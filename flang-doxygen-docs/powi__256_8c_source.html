15/**
16 * \brief Compute R4(:)**I4(:)
17 * \param[in] _vx (__m256) R4(:)
18 * \param[in] _vi (__m256i) I4(:)
19 * \return (__m256) R4(:)**I4(:)
20 */
21
22vrs8_t
23__fs_powi_8_avx2(vrs8_t _vx, vis8_t _vi)
24{
25 __m256 vx = (__m256)_vx;
26 __m256i vi = (__m256i)_vi;
27 __m256 vt;
28 __m256 vr;
29 __m256i vm;
30 __m256i vj;
31 __m256 vf1p0 = _mm256_set1_ps(1.0);
32 __m256i vi0 = _mm256_setzero_si256();
33 __m256i vlsb = _mm256_set1_epi32(1);
34
35 vj = _mm256_abs_epi32(vi);
36 vr = vf1p0;
37
38 vm = _mm256_cmpgt_epi32(vj, vi0);
39 if (0 == _mm256_movemask_epi8(vm)) return vr;
40
41 for (;;) {
42 vm = _mm256_and_si256(vj, vlsb);
43 vm = _mm256_sub_epi32(vi0, vm);
44 // Where vm == -1, vt = vx, else 1.0
45 vt = _mm256_blendv_ps(vf1p0, vx, (__m256)vm);
46 vr = _mm256_mul_ps(vr, vt);
47 vj = _mm256_srli_epi32(vj, 1);
48 vm = _mm256_cmpgt_epi32(vj, vi0);
49 if (0 == _mm256_movemask_epi8(vm)) break;
50 vx = _mm256_blendv_ps(vf1p0, vx, (__m256)vm);
51 vx = _mm256_mul_ps(vx, vx);
52 }
53
54 if (0 != _mm256_movemask_ps((__m256)vi)) {
55 vx = _mm256_div_ps(vf1p0, vr);
56 vr = _mm256_blendv_ps(vr, vx, (__m256)vi);
57 }
58
59 return vr;
60}

62/**
63 * \brief Compute R4(:)**I4(:) under mask
64 * \param[in] _vx (__m256) R4(:)
65 * \param[in] _vi (__m256i) I4(:)
66 * \param[in] _vm (__m256i) I4(:)
67 * \return (__m256) WHERE(_vm(:) != 0) R4(:)**I4(:)
68 */
69vrs8_t
70__fs_powi_8m_avx2(vrs8_t _vx, vis8_t _vi, vis8_t _vm)
71{
72 __m256 vx = (__m256)_vx;
73 /*
74 * Intentionally use PS for integer values, simplifies numer of arguments
75 * that need to be cast when using _mm256_blendv_ps() intrinsic.
76 */
77 __m256 vi = (__m256)_vi;
78 __m256 vm = (__m256)_vm;
79
80 vx = _mm256_blendv_ps(_mm256_set1_ps(0.0), vx, vm);
81 vi = _mm256_blendv_ps((__m256)_mm256_set1_epi32(0), vi, vm);
82
83 return __fs_powi_8_avx2((vrs8_t)vx, (vis8_t)vi);
84}

183/**
184 * \brief (internal) Kernel to compute R4(0:3)**I8_lower(:), Compute R4(4:7)**I8_upper(:)
185 * \param[in] _vx (__m256d) R4(:)
186 * \param[in] _vl (__m256i) I8(:)
187 * \param[in] _vu (__m256i) I8(:)
188 * \return (__m256d) (R4(4:7)**I8_upper(:))<<128 | R4(0:3)**I8_lower(:)
189 */
190vrs8_t
191__fs_powk_2x4_avx2 (vrs8_t _vx, vid4_t _vl, vid4_t _vu)
192{
193 __m256 vx;
194 __m256 vt;
195 __m256 vr;
196 __m256 vrl;
197 __m256 vf1p0 = _mm256_set1_ps(1.0);
198 __m256i vmi; // Inner loop mask
199 __m256i vmo; // Outer loop mask
200 __m256i vi;
201 __m256i vj;
202 __m256i vi0 = _mm256_setzero_si256();
203 __m256i vlsb = _mm256_set1_epi64x(1);
204 __m128 vxmm;
205
206 int i;
207
208 vxmm = _mm256_extractf128_ps(_vx, 0);
209 vx = _mm256_insertf128_ps(_vx, _mm_shuffle_ps(vxmm, vxmm ,0xe), 1);
210 vx = _mm256_shuffle_ps(vx, vx, 0x50);
211 vi = (__m256i)_vl;
212
213 for(i = 0 ; i < 2 ; i++) {
214 vr = vf1p0;
215 vj = _mm256_sub_epi64(vi0, vi);
216 vmo = _mm256_cmpgt_epi64(vi0, vi);
217 vj = (__m256i)_mm256_blendv_pd((__m256d)vi, (__m256d)vj, (__m256d)vmo);
218 for (;;) {
219 vmi = _mm256_and_si256(vj, vlsb);
220 vmi = _mm256_sub_epi64(vi0, vmi);
221 vt = (__m256) _mm256_blendv_pd((__m256d)vf1p0, (__m256d)vx, (__m256d)vmi);
222 vr = _mm256_mul_ps(vr, vt);
223 vj = _mm256_srli_epi64(vj, 1);
224 vmi = _mm256_cmpgt_epi64(vj, vi0);
225 if (0 == _mm256_movemask_epi8(vmi)) break;
226 vx = (__m256) _mm256_blendv_pd((__m256d)vf1p0, (__m256d)vx, (__m256d)vmi);
227 vx = _mm256_mul_ps(vx, vx);
228 }
229 if (0 != _mm256_movemask_pd((__m256d)vmo)) {
230 vx = _mm256_div_ps(vf1p0, vr);
231 vr = (__m256) _mm256_blendv_pd((__m256d)vr, (__m256d)vx, (__m256d)vmo);
232 }
233 if (i == 1) break;
234 vrl = vr;
235 vxmm = _mm256_extractf128_ps(_vx, 1);
236 vx = _mm256_insertf128_ps(_vx, _mm_shuffle_ps(vxmm, vxmm ,0x40), 0);
237 vx = _mm256_shuffle_ps(vx, vx, 0xfa);
238 vi = (__m256i)_vu;
239 }
240 vrl = _mm256_shuffle_ps(vrl, vrl, 0x8);
241 vr = _mm256_shuffle_ps(vr, vr, 0x80);
242 vt = (__m256) _mm256_blend_pd((__m256d)vr, (__m256d)vrl, 0x5);
243 return (vrs8_t) _mm256_permute4x64_pd((__m256d)vt, 0xd8);
244}

246/**
247 * \brief (external) Compute R4(0:3)**I8_lower(:), Compute R4(4:7)**I8_upper(:)
248 * \param[in] _vx (__m256d) R4(:)
249 * \param[in] _vl (__m256i) I8(:)
250 * \param[in] _vu (__m256i) I8(:)
251 * \return (__m256d) (R4(4:7)**I8_upper(:))<<128 | R4(0:3)**I8_lower(:)
252 */
253vrs8_t
254__fs_powk_8_avx2(vrs8_t _vx, vid4_t _vl, vid4_t _vu)
255{
256 __m256i vu = (__m256i) _vu;
257 __m256i vl = (__m256i) _vl;
258 __m256i vabsu;
259 __m256i vabsl;
260 __m256i vi0 = _mm256_setzero_si256();
261 __m256i vi2to31m1 = _mm256_set1_epi64x((1ll<<31)-1);
262 __m256i vmu;
263 __m256i vml;
264 __m256i vi;
265
266 vabsl = _mm256_sub_epi64(vi0, vl);
267 vabsu = _mm256_sub_epi64(vi0, vu);
268 vml = _mm256_cmpgt_epi64(vi0, vl);
269 vmu = _mm256_cmpgt_epi64(vi0, vu);
270 vabsl = (__m256i)_mm256_blendv_pd((__m256d)vl, (__m256d)vabsl, (__m256d)vml);
271 vabsu = (__m256i)_mm256_blendv_pd((__m256d)vu, (__m256d)vabsu, (__m256d)vmu);
272
273 vml = _mm256_cmpgt_epi64(vabsl, vi2to31m1);
274 vmu = _mm256_cmpgt_epi64(vabsu, vi2to31m1);
275
276 if (0 != _mm256_movemask_epi8(vmu|vml)) return __fs_powk_2x4_avx2(_vx, _vl, _vu);
277
278 vabsl = _mm256_shuffle_epi32(vl, 0x08);
279 vabsu = _mm256_shuffle_epi32(vu, 0x80);
280
281 vi = _mm256_permute4x64_epi64 (_mm256_blend_epi32(vabsu, vabsl, 0x33), 0xd8);
282 return __fs_powi_8_avx2(_vx, (vis8_t) vi);

310 vt = (__m256d)_mm256_cvtepi32_epi64((__m128i)_mm256_extractf128_pd(vm, 1));
311 vu = _mm256_blendv_pd(_mm256_set1_pd(0), vu, vt);
