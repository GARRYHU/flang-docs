28void
29ENTF90(GETARGA, getarga)(__INT_T *n, DCHAR(arg), __INT_T *int_kind DCLEN64(arg))
30{
31 char *p;
32 __INT_T i, len = CLEN(arg);
33 __INT_T argc;
34 char **argv;
35
36 i = I8(__fort_varying_int)(n, int_kind);
37 argc = __io_get_argc();
38 argv = __io_get_argv();
39
40 if (i < 0 || i >= argc) {
41 i = 0;
42 } else {
43 p = argv[i];
44 /* copy characters into arg: */
45 for (i = 0; p[i] != 0 && i < len; ++i)
46 CADR(arg)[i] = p[i];
47 }
48 /* blank fill if necessary: */
49 for (; i < len; ++i)
50 CADR(arg)[i] = ' ';
51}

76void
77ENTF90(GET_CMD_ARGA, get_cmd_arga)(__INT_T *number, DCHAR(value),
78 __INT_T *length, __INT_T *status,
79 __INT_T *int_kind DCLEN64(value))
80{
81 __CLEN_T len;
82 char *p, *q;
83 int n;
84 __CLEN_T i;
85 char **v;
86 __CLEN_T arg_len;
87 int stat;
88 int fail;
89
90 fail = 0;
91 i = 0;
92 arg_len = 0;
93 len = CLEN(value);
94 n = I8(__fort_varying_int)(number, int_kind);
95 q = (char *)CADR(value);
96 if (0 <= n && n < __io_get_argc()) {
97 v = __io_get_argv();
98 p = v[n];
99 /* compute length of argument: */
100 arg_len = strlen(p);
101 if (ISPRESENTC(value)) {
102 /* copy characters into value: */
103 for (i = 0; *p != 0 && i < len; q++, p++, i++) {
104 *q = *p;
105 }
106 }
107 } else
108 fail = 1;
109 if (ISPRESENTC(value)) {
110 /* blank fill if necessary: */
111 for (; i < len; q++, i++)
112 *q = ' ';
113 }
114 if (ISPRESENT(length))
115 store_int_kind(length, int_kind, arg_len);
116 if (ISPRESENT(status)) {
117 stat = 0;
118 if (ISPRESENTC(value) && len < arg_len)
119 stat = -1;
120 else if (fail)
121 stat = 1;
122 store_int_kind(status, int_kind, stat);
123 }
124}

136void
137ENTF90(GET_CMDA, get_cmda)(DCHAR(command), __INT_T *length, __INT_T *status,
138 __INT_T *int_kind DCLEN64(command))
139{
140 __CLEN_T len;
141 char *p, *q;
142 int argc;
143 int n;
144 __CLEN_T i;
145 char **v;
146 __CLEN_T arg_len;
147 int stat;
148 int fail;
149
150 fail = 0;
151 arg_len = 0;
152 len = CLEN(command);
153 argc = __io_get_argc();
154 q = (char *)CADR(command);
155 v = __io_get_argv();
156 i = 0;
157 for (n = 0; n < argc; n++) {
158 p = v[n];
159 if (ISPRESENTC(command)) {
160 /* copy characters into command: */
161 if (n && i < len) {
162 /* insert separator */
163 i++;
164 *q = ' ';
165 arg_len++;
166 q++;
167 }
168 for (; *p != 0 && i < len; q++, p++, i++) {
169 *q = *p;
170 arg_len++;
171 }
172 } else {
173 /* just compute length of argument: */
174 if (n) {
175 arg_len++; /* separator */
176 }
177 for (; *p != 0; p++, i++) {
178 arg_len++;
179 }
180 }
181 }
182 if (ISPRESENTC(command)) {
183 /* blank fill if necessary: */
184 for (; i < len; q++, i++)
185 *q = ' ';
186 }
187 if (ISPRESENT(length))
188 store_int_kind(length, int_kind, arg_len);
189 if (ISPRESENT(status)) {
190 stat = 0;
191 if (ISPRESENTC(command) && len < arg_len)
192 stat = -1;
193 else if (fail)
194 stat = 1;
195 store_int_kind(status, int_kind, stat);
196 }
197}

208void
209ENTF90(GET_ENV_VARA, get_env_vara)(DCHAR(name), DCHAR(value), __INT_T *length,
210 __INT_T *status, __LOG_T *trim_name,
211 __INT_T *int_kind DCLEN64(name) DCLEN64(value))
212{
213 __CLEN_T len;
214 char *envnm;
215 char *p, *q;
216 __CLEN_T i = 0;
217 __CLEN_T env_len;
218 int stat;
219 int sigblanks;
220
221 sigblanks = 0;
222 if (ISPRESENT(trim_name)) {
223 if (I8(__fort_varying_log)(trim_name, int_kind))
224 /* trailing blanks in the name are significant
225 * --- and currently not supported.
226 */
227 sigblanks = 1;
228 }
229 stat = 0;
230 env_len = 0;
231 len = CLEN(value);
232 envnm = __fstr2cstr(CADR(name), CLEN(name));
233 p = getenv(envnm);
234 __cstr_free(envnm);
235 q = (char *)CADR(value);
236 if (p) {
237 char *pv;
238 /* compute length of the environment variable's value */
239 pv = p;
240 for (i = 0; *pv != 0; pv++, i++) {
241 env_len++;
242 }
243 if (ISPRESENTC(value)) {
244 /* copy characters into value: */
245 for (i = 0; *p != 0 && i < len; q++, p++, i++) {
246 *q = *p;
247 }
248 }
249 } else
250 stat = 1;
251 if (ISPRESENTC(value)) {
252 /* blank fill if necessary: */
253 for (; i < len; q++, i++)
254 *q = ' ';
255 }
256 if (ISPRESENT(length))
257 store_int_kind(length, int_kind, env_len);
258 if (ISPRESENT(status)) {
259 if (ISPRESENTC(value) && len < env_len)
260 stat = -1;
261 store_int_kind(status, int_kind, stat);
262 }
263}
