39/*
40 PI_A to PI_D are constants that satisfy the following two conditions.
41
42 * For PI_A, PI_B and PI_C, the last 28 bits are zero.
43 * PI_A + PI_B + PI_C + PI_D is close to PI as much as possible.
44
45 The argument of a trig function is multiplied by 1/PI, and the
46 integral part is divided into two parts, each has at most 28
47 bits. So, the maximum argument that could be correctly reduced
48 should be 2^(28*2-1) PI = 1.1e+17. However, due to internal
49 double precision calculation, the actual maximum argument that can
50 be correctly reduced is around 2^50 = 1.1e+15.
51 */
52
53#define PI_A 3.1415926218032836914
54#define PI_B 3.1786509424591713469e-08
55#define PI_C 1.2246467864107188502e-16
56#define PI_D 1.2736634327021899816e-24
57#define TRIGRANGEMAX 1e+15
58
59/*
60 PI_A2 and PI_B2 are constants that satisfy the following two conditions.
61
62 * The last 3 bits of PI_A2 are zero.
63 * PI_A2 + PI_B2 is close to PI as much as possible.
64
65 The argument of a trig function is multiplied by 1/PI, and the
66 integral part is multiplied by PI_A2. So, the maximum argument that
67 could be correctly reduced should be 2^(3-1) PI = 12.6. By testing,
68 we confirmed that it correctly reduces the argument up to around 15.
69 */
70

72#define PI_B2 1.2246467991473532072e-16
73#define TRIGRANGEMAX2 15
