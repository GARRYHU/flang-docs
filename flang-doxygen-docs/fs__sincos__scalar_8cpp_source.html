46/* Payne-Hanek style argument reduction. */
47static void __attribute__((noinline))
48reduction_slowpath(float const a,
49S(float *rs, int32_t *hs) SINCOS_COMMA C(float *rc, int32_t *hc))
50{
51 uint2 m;
52 uint32_t ia = float_as_int(a);
53 S(uint32_t ss = ia & 0x80000000;)
54 uint32_t result[7];
55 uint32_t hi, lo;
56 uint32_t e;
57 int32_t idx;
58 int32_t q;
59 e = ((ia >> 23) & 0xff) - 127;
60 ia = (ia << 8) | 0x80000000;
61
62 /* compute x * 1/pi */
63 idx = 4 - ((e >> 5) & 3);
64
65 hi = 0;
66 for (q = 0; q < 6; q++) {
67 m = umad32wide(i1opi_f[q], ia, hi);
68 lo = m.x;
69 hi = m.y;
70 result[q] = lo;
71 }
72 result[q] = hi;
73
74 e = e & 31;
75 /* shift result such that hi:lo<63:63> is the least significant
76 integer bit, and hi:lo<62:0> are the fractional bits of the result
77 */
78
79 uint64_t p = ((uint64_t)result[idx + 2] << 32) | result[idx + 1];
80
81 if (e) {
82 q = 32 - e;
83 p = (p << e) | (result[idx] >> q);
84 }
85
86 p &= 0x7fffffffffffffffULL;
87
88 S(
89 uint64_t ps = p;
90 int32_t qs = q;
91
92 /* fraction */
93 qs = (result[idx + 2] << e) & 0x80000000;
94
95 if (ps & 0x4000000000000000ULL) {
96 ps |= 0x8000000000000000ULL;
97 qs ^= 0x80000000;
98 }
99 *hs = qs ^ ss;
100
101 double ds = (double)(int64_t)ps;
102 ds *= PI_2_M63;
103 *rs = (float)ds;
104 )
105
106
107 C(
108 uint64_t pc = p;
109 /* fraction */
110 *hc = (result[idx + 2] << e) & 0x80000000;
111 /* subtract 0.5 */
112 pc = (int64_t)pc - 0x4000000000000000LL;
113
114 double dc = (double)(int64_t)pc;
115 dc *= PI_2_M63;
116 *rc = (float)dc;
117 )
118}

120float
121__attribute__((noinline)) FCN_NAME(const float x)
122{
123
124 float p;
125 S(float ks, rs, ss, ts, xs;)
126 S(int hs = 0;)
127 C(float kc, rc, sc, tc, xc;)
128 C(int hc = 0;)
129
130 p = int_as_float(float_as_int(x) & 0x7fffffff);
131
132#if defined(COSINE)
133 // Cosine only! Don't use macro "C"!
134 if (float_as_int(p) <= 0x39800000) {
135 return 1.0f;
136 }
137#endif
138
139 if (float_as_int(p) > float_as_int(THRESHOLD_F)) {
140 if (float_as_int(p) >= 0x7f800000) {
141 S(xs = x * 0.0f;)
142 C(xc = x * 0.0f;)
143 } else {
144 reduction_slowpath(x, S(&xs, &hs) SINCOS_COMMA C(&xc, &hc));
145 }
146 } else {
147 S(ks = FMAF(x, _1_OVER_PI_F, 12582912.0f);)
148 S(hs = float_as_int(ks) << 31;)
149 S(ks -= 12582912.0f;)
150
151 C(kc = FMAF(p, _1_OVER_PI_F, -0.5f);)
152 C(kc += 12582912.0f;)
153 C(hc = float_as_int(kc) << 31;)
154 C(kc -= 12582912.0f;)
155 C(kc += 0.5;)
156
157 S(xs = x;)
158 C(xc = p;)
159 S(xs = FMAF(ks, -PI_HI_F, xs);)
160 C(xc = FMAF(kc, -PI_HI_F, xc);)
161 S(xs = FMAF(ks, -PI_MI_F, xs);)
162 C(xc = FMAF(kc, -PI_MI_F, xc);)
163 S(xs = FMAF(ks, -PI_LO_F, xs);)
164 C(xc = FMAF(kc, -PI_LO_F, xc);)
165 }
166
167#if defined(SINCOS) && defined(__x86_64__) && defined(__AVX2__)
168 {
169 const __m128 ama = _mm_set_ps(0.0, 0.0, -A_F, A_F);
170 const __m128 bmb = _mm_set_ps(0.0, 0.0, -B_F, B_F);
171 const __m128 cmc = _mm_set_ps(0.0, 0.0, -C_F, C_F);
172 const __m128 dmd = _mm_set_ps(0.0, 0.0, -D_F, D_F);
173 const __m128 eme = _mm_set_ps(0.0, 0.0, -E_F, E_F);
174 const __m128 omo = _mm_set_ps(0.0, 0.0, -1.0, 1.0);
175 __m128 va, vf, vs, vr, vt;
176
177 vr = ama;
178 va = _mm_unpacklo_ps(_mm_set1_ps(xs), _mm_set1_ps(xc));
179 vs = va * va;
180 vr = _mm_fmadd_ps(vr, vs, bmb);
181 vr = _mm_fmadd_ps(vr, vs, cmc);
182 vr = _mm_fmadd_ps(vr, vs, dmd);
183 vr = _mm_fmadd_ps(vr, vs, eme);
184 vf = _mm_castsi128_ps(_mm_set_epi32(0, 0, hc, hs));
185
186 vf = _mm_xor_ps(va, vf);
187 vt = _mm_fmadd_ps(vs, vf, _mm_set1_ps(0.0));
188 vf = _mm_mul_ps(vf, omo);
189 vr = _mm_fmadd_ps(vr, vt, vf);
190
191 rs = _mm_cvtss_f32(vr);
192 rc = _mm_cvtss_f32(_mm_permute_ps(vr, 1));
193 }
194#else
195 S(ss = xs * xs;)
196 C(sc = xc * xc;)
197
198 S(rs = A_F;)
199 C(rc = -A_F;)
200
201 S(rs = FMAF(rs, ss, B_F);)
202 C(rc = FMAF(rc, sc, -B_F);)
203
204 S(rs = FMAF(rs, ss, C_F);)
205 C(rc = FMAF(rc, sc, -C_F);)
206
207 S(rs = FMAF(rs, ss, D_F);)
208 C(rc = FMAF(rc, sc, -D_F);)
209
210 S(rs = FMAF(rs, ss, E_F);)
211 C(rc = FMAF(rc, sc, -E_F);)
212
213 S(xs = int_as_float(float_as_int(xs) ^ hs);)
214 C(xc = int_as_float(float_as_int(xc) ^ hc);)
215
216 S(ts = FMAF(ss, xs, 0.0);)
217 C(tc = FMAF(sc, xc, 0.0);)
218
219 S(rs = FMAF(rs, ts, xs);)
220 C(rc = FMAF(rc, tc, -xc);)
221#endif

223#if defined(SINCOS)
224// {
225 /*
226 * Probably not the best choice to use memory to get the proper value
227 * for "rc" when p <= 0x39800000, but the above block (#if 0) causes
228 * the sine/cosine FMAs and other operations to no longer be
229 * interleaved when compiling on x86_64 with clang 5.0.1.
230 *
231 * BTW, using a mask and merging good values and 1.0 also breaks the
232 * interleaved FMA operations.
233 */
234 bool pgt0x39800000 = float_as_int(p) > 0x39800000;
235 float rc_ret_val[2] = {1.0, rc}; // DO NOT MAKE STATIC!
236 rc = rc_ret_val[pgt0x39800000];
237
238 // Variant 2b
239 //float zero_one[] = {0.0,1.0,0.0}; // false, true, true+1
240 //rc = (rc * zero_one[pgt0x39800000]) + (1.0 * zero_one[1+pgt0x39800000]);
241
242 // Variant 2c
243 //uint32_t mask = 0 - pgt0x39800000;
244 //uint32_t t = (mask & *(uint32_t *)&rc) | (~mask & 0x3f800000);
245 //rc = *(float *)&t;
246 // }
247
248 asm("vmovss\t%0,%%xmm1" : : "m"(rc) : "%xmm1");
249 return rs;
250#else
251 S(return rs;)
252 C(return rc;)
253#endif
254}
