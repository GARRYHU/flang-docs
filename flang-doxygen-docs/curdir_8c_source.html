27/* fix pathname for "funny" NFS mount points */
28
29void __fort_fixmnt(char *new, char *old)
30{
31 const char *q;
32 char s[MAXPATHLEN]; /* substitute patterns */
33 char *smat; /* match string */
34 char *srep; /* replace string */
35 char *snxt; /* next pattern */
36 int n;
37
38 q = __fort_getopt("-mount"); /* pattern */
39 if (q == NULL) {
40 q = "/tmp_mnt";
41 }
42 strcpy(s, q);
43
44 snxt = s;
45 while (snxt != NULL) {
46 smat = snxt;
47 snxt = strchr(snxt, ',');
48 if (snxt != NULL) {
49 *snxt = '\0';
50 snxt++;
51 }
52 srep = strchr(smat, ':'); /* replace string */
53 if (srep != NULL) {
54 *srep = '\0';
55 srep++;
56 }
57 n = strlen(smat); /* match string length */
58 if (strncmp(old, smat, n) == 0) {
59 strcpy(new, srep ? srep : "");
60 strcat(new, old + n);
61 return;
62 }
63 }
64 strcpy(new, old);
65}
