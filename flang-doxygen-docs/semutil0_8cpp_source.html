167/**
168 \brief Convert a hollerith constant to a numeric value. The
169 destination can be any one of the 5 available sizes 1 byte, 2
170 bytes, 4 bytes, 8 bytes, or 16 bytes.
171
172 \param cp is a character pointer to hollerith character string.
173 \param num is the result of conversion of hollerith to numeric.
174 \param bc is the byte count of destination area i.e. *1, *2, *4, *8 or *16
175 */
176static void
177holtonum(char *cp, INT num[4], int bc)
178{
179 unsigned char *p, buf[18];
180 int sc, i;
181 int lc;
182
183 /*
184 * There are 4 32-bit parcels. Index 'i' starts at the parcel to begin
185 * filling and moves upward. For example, for a 8 byte quantity 'i' would
186 * start at 2 and end at 3 thus the last two words of 'num' array contain
187 * the 64-bit number.
188 */
189 num[0] = num[1] = num[2] = num[3] = 0;
190 sprintf((char *)buf, "%-17.17s", cp); /* Need 1 xtra char to detect trunc */
191 p = buf;
192 /* Select the initial parcel based on size of destination area */
193 i = 3;
194 if (bc > 4)
195 i = 2;
196 if (bc > 8)
197 i = 0;
198 if (flg.endian) {
199 /*
200 * The big endian byte order simply shifts each new character left 8
201 * bits FEWER than the previous shifted character producing the order
202 * ABCDEF...
203 */
204 while (i <= 3) {
205 sc = (bc < 4) ? bc : 4; /* Initial shift count */
206 while (sc--)
207 num[i] |= *p++ << (sc * 8);
208 i++;
209 }
210 } else {
211 /*
212 * The little endian byte order simply shifts each new character left 8
213 * bits MORE than the previous shifted character producing the order
214 * ...FEDCBA
215 */
216 while (i <= 3) {
217 sc = (bc < 4) ? bc : 4; /* Initial shift count */
218 lc = sc - 1;
219 while (sc--)
220 num[i] |= *p++ << ((lc - sc) * 8);
221 i++;
222 }
223 }
224
225 if (*p != '\0' && *p != ' ')
226 errwarn((error_code_t)24);
227}

229/**
230 \brief Convert doubleword hex/octal value to a character.
231
232 Function return value is the symbol table entry of the character
233 constant. The conversion is performed by copying an 8-bit value (2
234 hex digits) to a character position which is endian-dependent. The
235 endian-dependency is handled as if the hex value is "equivalenced"
236 with a character value of the same length. The length of the
237 character constant returned is determined by the magnitude of the
238 hex values (leading 0's are not converted). Note that this
239 conversion returns the same character value in context of an
240 assignment or data initialization. We may be incompatible with
241 other implementations with respect to data initialization:
242 1. if the value is smaller than the char item being initialized,
243 the conversion process results in appending blanks; other
244 systems may pad with 'nulls'
245 2. if the value is larger, truncation of the least significant
246 characters ("rightmost") occurs; other systems truncate the
247 most significant characters ("leftmost").
248
249 hexval[0] is msw, hexval[1] is lsw.
250 */
251static int
252hex2char(INT hexval[2])
253{
254 UINT val;
255 int i;
256 int len;
257 char *p;
258 char buf[8];
259
260 len = 0;
261 if (flg.endian) {
262 /* big endian: rightmost 2 hex digits are in last byte position */
263 p = buf + 7;
264 i = -1;
265 } else {
266 /* little endian: rightmost 2 hex digits are in first byte position */
267 p = buf;
268 i = 1;
269 }
270 val = hexval[1];
271 while (val) {
272 *p = val & 0xff;
273 p += i;
274 len++;
275 val >>= 8;
276 }
277 val = hexval[0];
278 while (val) {
279 *p = val & 0xff;
280 p += i;
281 len++;
282 val >>= 8;
283 }
284
285 if (len == 0) {
286 len = 1;
287 *p = '\0';
288 } else if (flg.endian)
289 p++;
290 else
291 p = buf;
292
293 return getstring(p, len);
294}

296/**
297 \brief Convert doubleword hex/octal value to an ncharacter.
298
299 Function return value is the symbol table entry of the character
300 constant. The conversion is performed by copying an 8-bit value (2
301 hex digits) to a character position which is endian-dependent. The
302 endian-dependency is handled as if the hex value is "equivalenced"
303 with a ncharacter value of the same length. The length of the
304 ncharacter constant returned is determined by the magnitude of the
305 hex values (leading 0's are not converted). Note that this
306 conversion returns the same ncharacter value in context of an
307 assignment or data initialization. We may be incompatible with
308 other implementations with respect to data initialization:
309 1. if the value is smaller than the nchar item being initialized,
310 the conversion process results in appending blanks; other
311 systems may pad with 'nulls'
312 2. if the value is larger, truncation of the least significant
313 characters ("rightmost") occurs; other systems truncate the most
314 significant characters ("leftmost").
315
316 hexval[0] is msw, hexval[1] is lsw.
317 */
318static int
319hex2nchar(INT hexval[2])
320{
321 UINT val;
322 int i;
323 int len;
324 unsigned short *p;
325 unsigned short buf[4];
326
327 len = 0;
328 if (flg.endian) {
329 /* big endian: rightmost 2 hex digits are in last byte position */
330 p = buf + 3;
331 i = -1;
332 } else {
333 /* little endian: rightmost 2 hex digits are in first byte position */
334 p = buf;
335 i = 1;
336 }
337 val = hexval[1];
338 while (val) {
339 *p = val & 0xffff;
340 p += i;
341 len += 2;
342 val >>= 16;
343 }
344 val = hexval[0];
345 while (val) {
346 *p = val & 0xffff;
347 p += i;
348 len += 2;
349 val >>= 16;
350 }
351 if (len == 0) {
352 len = 1;
353 *p = '\0';
354 } else if (flg.endian)
355 p++;
356 else
357 p = buf;
358
359 return getstring((char *)p, len);
360}

362/**
363 * \brief Convert constant from oldtyp to newtyp.
364 *
365 * Issue error messages only for impossible conversions. Return constant value
366 * for 32-bit constants, or symbol table pointer. Can only be used for scalar
367 * constants.
368 *
369 * Remember: Non-decimal constants are octal, hexadecimal, or hollerith
370 * constants which are represented by DT_WORD, DT_DWORD and DT_HOLL.
371 * Non-decimal constants 'assume' data types rather than go thru a conversion.
372 * Hollerith constants have a data type of DT_HOLL in the semantic stack,
373 * however, the symbol table entry they point to has a data type of DT_CHAR.
374 *
375 * Hollerith constants are always treated as scalars while octal or
376 * hexadecimal constants can be promoted to vectors.
377 */
378INT
379cngcon(INT oldval, DTYPE oldtyp, DTYPE newtyp)
380{
381 int to, from;
382 char *cp;
383 int newcvlen, oldcvlen, blnk;
384 INT num[4], result;
385 INT num1[4];
386#ifdef TARGET_SUPPORTS_QUADFP
387 INT num2[4];
388#endif
389 INT swap;
390 if (is_empty_typedef(newtyp) && oldtyp == DT_INT) {
391 /* FS#17600 - special case for emptyy derived type */
392 newtyp = DT_INT;
393 }
394
395 if (newtyp == oldtyp)
396 return oldval;
397 to = DTY(newtyp);
398 from = DTY(oldtyp);
399 if (!TY_ISSCALAR(to) || !TY_ISSCALAR(from))
400 goto type_conv_error;
401
402 switch (to) {
403 case TY_WORD:
404 case TY_DWORD:
405 return oldval;
406
407 case TY_BLOG:
408 case TY_BINT:
409 /* decimal integer constants are 32-bits, BUT, PARAMETER
410 may be TY_SLOG, TY_SINT, TY_BLOG, or TY_BINT.
411 */
412 switch (from) {
413 case TY_WORD:
414 if (oldval & 0xFFFFFF00)
415 errwarn((error_code_t)15);
416 return (ARSHIFT(LSHIFT(oldval, 24), 24));
417 case TY_DWORD:
418 result = CONVAL2G(oldval);
419 if (CONVAL1G(oldval) || (result & 0xFFFFFF00))
420 errwarn((error_code_t)15);
421 return (ARSHIFT(LSHIFT(result, 24), 24));
422 case TY_INT8:
423 case TY_LOG8:
424 result = CONVAL2G(oldval);
425 if ((((result & 0xFFFFFF80) != 0xFFFFFF80) && (result & 0xFFFFFF00)) ||
426 (CONVAL1G(oldval) && CONVAL1G(oldval) != (INT)0xFFFFFFFF))
427 SERROR(128, "%d", result & 0xFF);
428 return (ARSHIFT(LSHIFT(result, 24), 24));
429 case TY_BINT:
430 case TY_SINT:
431 case TY_INT:
432 case TY_BLOG:
433 case TY_SLOG:
434 case TY_LOG:
435 if (((oldval & 0xFFFFFF80) != 0xFFFFFF80) && (oldval & 0xFFFFFF00))
436 SERROR(128, "%d", oldval & 0xFF);
437 return (ARSHIFT(LSHIFT(oldval, 24), 24));
438 default:
439 goto other_int_cases;
440 break;
441 }
442 break;
443 case TY_SLOG:
444 case TY_SINT:
445 switch (from) {
446 case TY_WORD:
447 if (oldval & 0xFFFF0000)
448 errwarn((error_code_t)15);
449 return (ARSHIFT(LSHIFT(oldval, 16), 16));
450 case TY_DWORD:
451 result = CONVAL2G(oldval);
452 if (CONVAL1G(oldval) || (result & 0xFFFF0000))
453 errwarn((error_code_t)15);
454 return (ARSHIFT(LSHIFT(result, 16), 16));
455 case TY_INT8:
456 case TY_LOG8:
457 result = CONVAL2G(oldval);
458 if ((((result & 0xFFFF8000) != 0xFFFF8000) && (result & 0xFFFF0000)) ||
459 (CONVAL1G(oldval) && CONVAL1G(oldval) != (INT)0xFFFFFFFF))
460 SERROR(128, "%d", result & 0xFFFF);
461 return (ARSHIFT(LSHIFT(result, 16), 16));
462 case TY_BINT:
463 case TY_SINT:
464 case TY_INT:
465 case TY_BLOG:
466 case TY_SLOG:
467 case TY_LOG:
468 if (((oldval & 0xFFFF8000) != 0xFFFF8000) && (oldval & 0xFFFF0000))
469 SERROR(128, "%d", oldval & 0xFFFF);
470 return (ARSHIFT(LSHIFT(oldval, 16), 16));
471 default:
472 goto other_int_cases;
473 break;
474 }
475 break;
476 case TY_LOG:
477 case TY_INT:
478 if (from == TY_DWORD) {
479 result = CONVAL2G(oldval);
480 if (CONVAL1G(oldval))
481 errwarn((error_code_t)15);
482 return (result);
483 }
484 if (from == TY_INT8) {
485 result = CONVAL2G(oldval);
486 if (CONVAL1G(oldval) && CONVAL1G(oldval) != (INT)0xFFFFFFFF)
487 SERROR(128, "%d", CONVAL1G(oldval));
488 return (result);
489 }
490 if (from == TY_LOG8) {
491 result = CONVAL2G(oldval);
492 return (result);
493 } else if (from == TY_WORD || TY_ISINT(from))
494 return oldval;
495 else {
496 other_int_cases:
497 switch (from) {
498 case TY_CMPLX:
499 oldval = CONVAL1G(oldval);
500 FLANG_FALLTHROUGH;
501 case TY_REAL:
502 xfix(oldval, &result);
503 return result;
504 case TY_DCMPLX:
505 oldval = CONVAL1G(oldval);
506 FLANG_FALLTHROUGH;
507 case TY_DBLE:
508 num[0] = CONVAL1G(oldval);
509 num[1] = CONVAL2G(oldval);
510 xdfix(num, &result);
511 return result;
512#ifdef TARGET_SUPPORTS_QUADFP
513 case TY_QCMPLX:
514 oldval = CONVAL1G(oldval);
515 FLANG_FALLTHROUGH;
516#endif
517 case TY_QUAD:
518#ifdef TARGET_SUPPORTS_QUADFP
519 num[0] = CONVAL1G(oldval);
520 num[1] = CONVAL2G(oldval);
521 num[2] = CONVAL3G(oldval);
522 num[3] = CONVAL4G(oldval);
523 xqfix(num, &result);
524 return result;
525#else
526 uf("QUAD");
527 return 0;
528#endif
529 case TY_CHAR:
530 if (flg.standard)
531 ERR170("conversion of CHARACTER constant to numeric");
532 FLANG_FALLTHROUGH;
533 case TY_HOLL:
534 cp = stb.n_base + CONVAL1G(oldval);
535 oldcvlen = 4;
536 if (to == TY_BLOG || to == TY_BINT)
537 oldcvlen = 1;
538 if (to == TY_SLOG || to == TY_SINT)
539 oldcvlen = 2;
540 if (to == TY_LOG8 || to == TY_INT8)
541 oldcvlen = 8;
542 holtonum(cp, num, oldcvlen);
543 return num[3];
544 default: /* TY_NCHAR comes here */
545 break;
546 }
547 }
548 break;
549
550 case TY_LOG8:
551 case TY_INT8:
552 if (from == TY_DWORD || from == TY_INT8 || from == TY_LOG8) {
553 num[0] = CONVAL1G(oldval);
554 num[1] = CONVAL2G(oldval);
555 return getcon(num, newtyp);
556 } else if (from == TY_WORD) {
557 num[0] = 0;
558 num[1] = oldval;
559 return getcon(num, newtyp);
560 } else if (TY_ISINT(from)) {
561 if (oldval < 0) {
562 num[0] = -1;
563 num[1] = oldval;
564 } else {
565 num[0] = 0;
566 num[1] = oldval;
567 }
568 return getcon(num, newtyp);
569 } else {
570 switch (from) {
571 case TY_CMPLX:
572 oldval = CONVAL1G(oldval);
573 FLANG_FALLTHROUGH;
574 case TY_REAL:
575 xfix64(oldval, num);
576 return getcon(num, newtyp);
577 case TY_DCMPLX:
578 oldval = CONVAL1G(oldval);
579 FLANG_FALLTHROUGH;
580 case TY_DBLE:
581 num1[0] = CONVAL1G(oldval);
582 num1[1] = CONVAL2G(oldval);
583 xdfix64(num1, num);
584 return getcon(num, newtyp);
585#ifdef TARGET_SUPPORTS_QUADFP
586 case TY_QCMPLX:
587 oldval = CONVAL1G(oldval);
588 FLANG_FALLTHROUGH;
589#endif
590 case TY_QUAD:
591#ifdef TARGET_SUPPORTS_QUADFP
592 num1[0] = CONVAL1G(oldval);
593 num1[1] = CONVAL2G(oldval);
594 num1[2] = CONVAL3G(oldval);
595 num1[3] = CONVAL4G(oldval);
596 xqfix64(num1, num);
597 return getcon(num, newtyp);
598#else
599 uf("QUAD");
600 return 0;
601#endif
602 case TY_CHAR:
603 if (flg.standard)
604 ERR170("conversion of CHARACTER constant to numeric");
605 FLANG_FALLTHROUGH;
606 case TY_HOLL:
607 cp = stb.n_base + CONVAL1G(oldval);
608 holtonum(cp, num, 8);
609 if (flg.endian == 0) {
610 /* for little endian, need to swap words in each double word
611 * quantity. Order of bytes in a word is okay, but not the
612 * order of words.
613 */
614 swap = num[2];
615 num[2] = num[3];
616 num[3] = swap;
617 }
618 return getcon(&num[2], newtyp);
619 default: /* TY_NCHAR comes here */
620 break;
621 }
622 }
623 break;
624 case TY_REAL:
625 if (from == TY_WORD)
626 return oldval;
627 else if (from == TY_DWORD) {
628 result = CONVAL2G(oldval);
629 if (CONVAL1G(oldval))
630 errwarn((error_code_t)15);
631 return result;
632 } else if (from == TY_INT8 || from == TY_LOG8) {
633 num[0] = CONVAL1G(oldval);
634 num[1] = CONVAL2G(oldval);
635 xflt64(num, &result);
636 return result;
637 } else if (TY_ISINT(from)) {
638 xffloat(oldval, &result);
639 return result;
640 } else {
641 switch (from) {
642 case TY_CMPLX:
643 return CONVAL1G(oldval);
644 case TY_DCMPLX:
645 oldval = CONVAL1G(oldval);
646 FLANG_FALLTHROUGH;
647 case TY_DBLE:
648 num[0] = CONVAL1G(oldval);
649 num[1] = CONVAL2G(oldval);
650 xsngl(num, &result);
651 return result;
652#ifdef TARGET_SUPPORTS_QUADFP
653 case TY_QCMPLX:
654 oldval = CONVAL1G(oldval);
655 FLANG_FALLTHROUGH;
656#endif
657 case TY_QUAD:
658#ifdef TARGET_SUPPORTS_QUADFP
659 num[0] = CONVAL1G(oldval);
660 num[1] = CONVAL2G(oldval);
661 num[2] = CONVAL3G(oldval);
662 num[3] = CONVAL4G(oldval);
663 xqtof(num, &result);
664 return result;
665#else
666 uf("QUAD");
667 return oldval;
668#endif
669 case TY_CHAR:
670 if (flg.standard)
671 ERR170("conversion of CHARACTER constant to numeric");
672 FLANG_FALLTHROUGH;
673 case TY_HOLL:
674 cp = stb.n_base + CONVAL1G(oldval);
675 holtonum(cp, num, 4);
676 return num[3];
677 default:
678 break;
679 }
680 }
681 break;
682
683 case TY_DBLE:
684 if (from == TY_WORD) {
685 num[0] = 0;
686 num[1] = oldval;
687 } else if (from == TY_DWORD) {
688 num[0] = CONVAL1G(oldval);
689 num[1] = CONVAL2G(oldval);
690 } else if (from == TY_INT8 || from == TY_LOG8) {
691 num1[0] = CONVAL1G(oldval);
692 num1[1] = CONVAL2G(oldval);
693 xdflt64(num1, num);
694 } else if (TY_ISINT(from))
695 xdfloat(oldval, num);
696#ifdef TARGET_SUPPORTS_QUADFP
697 else if (from == TY_QCMPLX) {
698 oldval = CONVAL1G(oldval);
699 num1[0] = CONVAL1G(oldval);
700 num1[1] = CONVAL2G(oldval);
701 num1[2] = CONVAL3G(oldval);
702 num1[3] = CONVAL4G(oldval);
703 xqtod(num1, num);
704 }
705#endif
706 else if (from == TY_DCMPLX)
707 return CONVAL1G(oldval);
708 else if (from == TY_CMPLX) {
709 oldval = CONVAL1G(oldval);
710 xdble(oldval, num);
711 } else if (from == TY_REAL) {
712 xdble(oldval, num);
713#ifdef TARGET_SUPPORTS_QUADFP
714 } else if (from == TY_QUAD) {
715 num1[0] = CONVAL1G(oldval);
716 num1[1] = CONVAL2G(oldval);
717 num1[2] = CONVAL3G(oldval);
718 num1[3] = CONVAL4G(oldval);
719 xqtod(num1, num);
720#endif
721 } else if (from == TY_HOLL || from == TY_CHAR) {
722 if (flg.standard && from == TY_CHAR)
723 ERR170("conversion of CHARACTER constant to numeric");
724 cp = stb.n_base + CONVAL1G(oldval);
725 holtonum(cp, num, 8);
726 if (flg.endian == 0) {
727 /* for little endian, need to swap words in each double word
728 * quantity. Order of bytes in a word is okay, but not the
729 * order of words.
