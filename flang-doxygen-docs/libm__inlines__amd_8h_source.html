55/* Splits double x into exponent e and mantissa m, where 0.5 <= abs(m) < 1.0.
56 Assumes that x is not zero, denormal, infinity or NaN, but these conditions
57 are not checked */
58static inline void
59splitDouble(double x, int *e, double *m)
60{
61 __UINT8_T ux, uy;
62 GET_BITS_DP64(x, ux);
63 uy = ux;
64 ux &= EXPBITS_DP64;
65 ux >>= EXPSHIFTBITS_DP64;
66 *e = (int)ux - EXPBIAS_DP64 + 1;
67 uy = (uy & (SIGNBIT_DP64 | MANTBITS_DP64)) | HALFEXPBITS_DP64;
68 PUT_BITS_DP64(uy, x);
69 *m = x;
70}
71#endif /* USE_SPLITDOUBLE */

73#if defined(USE_SPLITDOUBLE_2)
74/* Splits double x into exponent e and mantissa m, where 1.0 <= abs(m) < 4.0.
75 Assumes that x is not zero, denormal, infinity or NaN, but these conditions
76 are not checked. Also assumes EXPBIAS_DP is odd. With this
77 assumption, e will be even on exit. */
78static inline void
79splitDouble_2(double x, int *e, double *m)
80{
81 __UINT8_T ux, vx;
82 GET_BITS_DP64(x, ux);
83 vx = ux;
84 ux &= EXPBITS_DP64;
85 ux >>= EXPSHIFTBITS_DP64;
86 if (ux & 1) {
87 /* The exponent is odd */
88 vx = (vx & (SIGNBIT_DP64 | MANTBITS_DP64)) | ONEEXPBITS_DP64;
89 PUT_BITS_DP64(vx, x);
90 *m = x;
91 *e = ux - EXPBIAS_DP64;
92 } else {
93 /* The exponent is even */
94 vx = (vx & (SIGNBIT_DP64 | MANTBITS_DP64)) | TWOEXPBITS_DP64;
95 PUT_BITS_DP64(vx, x);
96 *m = x;
97 *e = ux - EXPBIAS_DP64 - 1;
98 }
99}
100#endif /* USE_SPLITDOUBLE_2 */

123 Assumes EMIN <= n <= EMAX, though this condition is not checked. */
124static inline double
125scaleDouble_1(double x, int n)
126{
127 double t;
128 /* Construct the number t = 2.0**n */

134#if defined(USE_SCALEDOUBLE_2)
135/* Scales the double x by 2.0**n.
136 Assumes 2*EMIN <= n <= 2*EMAX, though this condition is not checked. */
137static inline double
138scaleDouble_2(double x, int n)
139{
140 double t1, t2;
141 int n1, n2;
142 n1 = n / 2;
143 n2 = n - n1;
144 /* Construct the numbers t1 = 2.0**n1 and t2 = 2.0**n2 */
145 PUT_BITS_DP64(((__INT8_T)n1 + EXPBIAS_DP64) << EXPSHIFTBITS_DP64, t1);
146 PUT_BITS_DP64(((__INT8_T)n2 + EXPBIAS_DP64) << EXPSHIFTBITS_DP64, t2);
147 return (x * t1) * t2;
148}
149#endif /* USE_SCALEDOUBLE_2 */

151#if defined(USE_SCALEDOUBLE_3)
152/* Scales the double x by 2.0**n.
153 Assumes 3*EMIN <= n <= 3*EMAX, though this condition is not checked. */
154static inline double
155scaleDouble_3(double x, int n)
156{
157 double t1, t2, t3;
158 int n1, n2, n3;
159 n1 = n / 3;
160 n2 = (n - n1) / 2;
161 n3 = n - n1 - n2;
162 /* Construct the numbers t1 = 2.0**n1, t2 = 2.0**n2 and t3 = 2.0**n3 */
163 PUT_BITS_DP64(((__INT8_T)n1 + EXPBIAS_DP64) << EXPSHIFTBITS_DP64, t1);
164 PUT_BITS_DP64(((__INT8_T)n2 + EXPBIAS_DP64) << EXPSHIFTBITS_DP64, t2);
165 PUT_BITS_DP64(((__INT8_T)n3 + EXPBIAS_DP64) << EXPSHIFTBITS_DP64, t3);
166 return ((x * t1) * t2) * t3;

1528 /* x is denormalised or zero */
1529 if (ux == 0)
1530 /* x is zero */
1531 return x;

1534 /* Normalize x by increasing the exponent by 26
1535 and subtracting a correction to account for the implicit
1536 bit. This replaces a slow denormalized
1537 multiplication by a fast normal subtraction. */
1538 static const float corr = 7.888609052210118054e-31F; /* 0x0d800000 */
1539 denorm = 1;

1541 PUT_BITS_SP32(ux | 0x0d800000, x);
1542 x -= corr;
1543 GET_BITS_SP32(x, ux);
1544 }

1548
1549 /*
1550 Find y and e such that x = 2^e * y, where y in [1,4).
1551 This is done using an in-lined variant of splitFloat,
1552 which also ensures that e is even.
1553 */

1555 ux &= EXPBITS_SP32;
1556 ux >>= EXPSHIFTBITS_SP32;
1557 if (ux & 1) {
1558 GET_BITS_SP32(y, u);
1559 u &= (SIGNBIT_SP32 | MANTBITS_SP32);
1560 u |= ONEEXPBITS_SP32;
1561 PUT_BITS_SP32(u, y);

1572

1576

1578 rtc_trail = rt_jby32_trail_table_float[index - 32];

1580 r = (y - c) / c;
1581
1582 /*
1583 Find q = sqrt(1+r) - 1.
1584 From one step of Newton on (q+1)^2 = 1+r
1585 */
1586
1587 p = r * 0.5F - r * r * (0.1250079870F - r * (0.6250522999e-01F));
1588 twop = p + p;
1589 q = p - (p * p + (twop - r)) / (twop + 2.0);
1590
1591 /* Reconstruction */
1592

1598 /* Scale by 2**(e-13) */
1599 PUT_BITS_SP32(((e - 13) + EXPBIAS_SP32) << EXPSHIFTBITS_SP32, r);
1600 z *= r;
1601 } else {
1602 /* Scale by 2**e */
1603 PUT_BITS_SP32((e + EXPBIAS_SP32) << EXPSHIFTBITS_SP32, r);
1604 z *= r;
1605 }
1606
1607 return z;
1608}

1769 /* Approximating polynomial coefficients for x near 1.0 */

1771 8.33333333333317923934e-02, /* 0x3fb55555555554e6 */

1777 static const double cb_1 =
1778 8.33333333333333593622e-02, /* 0x3fb5555555555557 */
1779 cb_2 = 1.24999999978138668903e-02, /* 0x3f89999999865ede */
1780 cb_3 = 2.23219810758559851206e-03; /* 0x3f6249423bd94741 */

1788 /* Arguments close to 1.0 are handled separately to maintain
1789 accuracy.
1790
1791 The approximation in this region exploits the identity
1792 log( 1 + r ) = log( 1 + u/2 ) / log( 1 - u/2 ), where
1793 u = 2r / (2+r).
1794 Note that the right hand side has an odd Taylor series expansion
1795 which converges much faster than the Taylor series expansion of
1796 log( 1 + r ) in r. Thus, we approximate log( 1 + r ) by
1797 u + A1 * u^3 + A2 * u^5 + ... + An * u^(2n+1).
1798
1799 One subtlety is that since u cannot be calculated from
1800 r exactly, the rounding error in the first u should be
1801 avoided if possible. To accomplish this, we observe that
1802 u = r - r*r/(2+r).
1803 Since x (=1+r) is the input argument, and thus presumed exact,
1804 the formula above approximates u accurately because
1805 u = r - correction,
1806 and the magnitude of "correction" (of the order of r*r)
1807 is small.
1808 With these observations, we will approximate log( 1 + r ) by
1809 r + ( (A1*u^3 + ... + An*u^(2n+1)) - correction ).
1810
1811 We approximate log(1+r) by an odd polynomial in u, where
1812 u = 2r/(2+r) = r - r*r/(2+r).

1822 *r2 = z2;

1829 F1 = 1 + j/64, j ranges from 0 to 64, and |F2| <= 1/128.
1830

1832 in U, where U = 2 F2 / (2 F2 + F1).
1833 Note that log( 1 + F2/F1 ) = log( 1 + U/2 ) - log( 1 - U/2 ).
1834 The core approximation calculates
1835 Poly = [log( 1 + U/2 ) - log( 1 - U/2 )]/U - 1.
1836 Note that log(1 + U/2) - log(1 - U/2) = 2 arctanh ( U/2 ),
1837 thus, Poly = 2 arctanh( U/2 ) / U - 1.
1838

1840 log(x) = M*log(2) + log(F1) + log( 1 + F2/F1 ).
1841 Hence, we return Z1 = log(F1), and Z2 = log( 1 + F2/F1).
1842 The values of log(F1) are calculated beforehand and stored
1843 in the program.

1850 and subtracting a correction to account for the implicit

1852 multiplication by a fast normal subtraction. */

1854 2.5653355008114851558350183e-290; /* 0x03d0000000000000 */

1856 ux |= 0x03d0000000000000;
1857 PUT_BITS_DP64(ux, f);
1858 f -= corr;
1859 GET_BITS_DP64(f, ux);
1860 expadjust = 60;
1861 } else
1862 expadjust = 0;
1863
1864 /* Store the exponent of x in xexp and put
1865 f into the range [0.5,1) */
1866 *xexp = (int)((ux & EXPBITS_DP64) >> EXPSHIFTBITS_DP64) - EXPBIAS_DP64 -
1867 expadjust;
1868 PUT_BITS_DP64((ux & MANTBITS_DP64) | HALFEXPBITS_DP64, f);
1869
1870 /* Now x = 2**xexp * f, 1/2 <= f < 1. */
1871
1872 /* Set index to be the nearest integer to 128*f */
1873 r = 128.0 * f;
1874 index = (int)(r + 0.5);
1875
1876 z1 = ln_lead_table[index - 64];
1877 q = ln_tail_table[index - 64];
1878 f1 = index * 0.0078125; /* 0.0078125 = 1/128 */
1879 f2 = f - f1;
1880 /* At this point, x = 2**xexp * ( f1 + f2 ) where
1881 f1 = j/128, j = 64, 65, ..., 128 and |f2| <= 1/256. */
1882
1883 /* Calculate u = 2 f2 / ( 2 f1 + f2 ) = f2 / ( f1 + 0.5*f2 ) */
1884 /* u = f2 / (f1 + 0.5 * f2); */
1885 u = f2 / (f1 + 0.5 * f2);
1886
1887 /* Here, |u| <= 2(exp(1/16)-1) / (exp(1/16)+1).
1888 The core approximation calculates
1889 poly = [log(1 + u/2) - log(1 - u/2)]/u - 1 */
1890 v = u * u;
1891 poly = (v * (cb_1 + v * (cb_2 + v * cb_3)));
1892 z2 = q + (u + u * poly);
1893 *r1 = z1;
1894 *r2 = z2;
1895 }
1896 return;
1897}
