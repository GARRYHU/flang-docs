173/* Induction Analysis Controller */
174/* The following steps occur: */
175/* 1. ind_loop_init - initialize for analyzing a loop */
176/* potential induction variables are found in the loop */
177/* 2. find_bivs - determines which of the potential ind. */
178/* variables are basic induction variables */
179/* 3. new_ivs - checks the uses of the induction var's */
180/* and creates new induction variables */

182void
183induction(int lp)
184{
185 (void)get_loop_count(lp);
186
187 if (OPTDBG(9, 2048)) {
188 fprintf(gbl.dbgfil, "\n* * * loop count %d * * *\n", loop_cnt);
189 if (loop_cnt)
190 dbg_print_ast(loop_cnt, gbl.dbgfil);
191 }
192
193 end_loop_count();
194

197static void
198ind_loop_init(void)
199{
200 /* initialize for induction analysis on the loop -- setup the
201 * storage needed and make a pass over all of the stores in the
202 * loop including any nested loops
203 */
204 register STL *stl;
205 register int nme, sym, i, ind;
206
207 induc.indb.stg_avail = 1;
208
209 /*
210 * scan through the store lists for any nested blocks. The names
211 * found cannot be induction variables
212 */
213 stl = LP_STL(lpx);
214 if (stl->childlst != 0)
215 removest(stl->childlst);
216
217 /*
218 * scan through the store lists for just the loop omitting certain
219 * names
220 */
221 stl = LP_STL(lpx);
222 for (i = stl->store; i; i = STORE_NEXT(i)) {
223 nme = STORE_NM(i);
224 if (NME_TYPE(nme) == NT_VAR && ISSCALAR(sym = NME_SYM(nme)) &&
225 DT_ISINT(DTYPEG(sym)) && NME_RFPTR(nme) == 0) {
226 /* the basic induction variable's family is itself. this
227 * will be propagated to all new induction variables based
228 * on the basic induction var.
229 */
230 ind = add_ind(nme, 0, 0, 0, NULL);
231 if (!is_sym_optsafe(nme, lpx)) {
232 IND_OMIT(ind) = 1;
233 if (OPTDBG(9, 2048))
234 fprintf(gbl.dbgfil, "--- omit nme %d - sym (%s) unsafe\n", nme,
235 getprint((int)sym));
236 } else {
237 if (OPTDBG(9, 2048))
238 fprintf(gbl.dbgfil, "--- potential ind nme %d (%s)\n", nme,
239 getprint(sym));
240
241 /*
242 * we can only delete c variables which are automatic or
243 * arguments or ftn variables which are local, and other
244 * criteria.
245 */
246 if (XBIT(19, 0x2)) /* #pragma nolstval */
247 IND_DELETE(ind) = 1;
248 else
249 IND_DELETE(ind) = !is_sym_imp_live(nme);
250#ifndef ALLOW_COMPLEX
251 if (DT_ISCMPLX(DTYPEG(sym)))
252 /* TEMPORARY - disallow complex; NOTE: global flow of
253 * complex vars is not done (defs are not created). This
254 * test needs to be removed when we allow complex.
255 */
256 IND_OMIT(ind) = 1;
257#endif
258 }
259 }
260 }
261
262 /*
263 * mark currently available locations of the use, def, and invariant
264 * tables -- the analysis will create new use and def items, and
265 * perhaps invariant ili which are needed only for the loop.
266 * The space will be re-used for the next loop.
267 */
268 mark_useb = opt.useb.stg_avail;
269 mark_defb = opt.defb.stg_avail;
270 mark_invb = opt.invb.stg_avail;
271
272 lastval_cnt = 0;
273 loop_cnt = 0;
274
275 cr_lastval_q = NULL;
276
277}

279/*
280 * scan through the store lists for any nested blocks. The names found
281 * cannot be induction variables removest is passed a list of children to
282 * (recursively) mark names as not valid induction variables.
283 */
284static void
285removest(STL *stlist)
286{
287 STL *tmp;
288 register int nme, i, ind;
289
290 for (tmp = stlist; tmp != NULL; tmp = tmp->nextsibl) {
291 for (i = tmp->store; i; i = STORE_NEXT(i)) {
292 nme = STORE_NM(i);
293 if (NME_TYPE(nme) == NT_VAR && ISSCALAR(NME_SYM(nme)) &&
294 NME_RFPTR(nme) == 0) {
295 ind = add_ind(nme, 0, 0, 0, NULL);
296 IND_OMIT(ind) = 1;
297 IND_RMST(ind) = 1;
298 if (OPTDBG(9, 2048))
299 fprintf(gbl.dbgfil, "--- omit nme %d (%s) - in nested loop\n", nme,
300 getprint((int)NME_SYM(nme)));
301 }
302 }
303 if (tmp->childlst != 0)
304 removest(tmp->childlst);
305 }
306}

308static int
309add_ind(int nme, int load, int tnode, int derived, Q_ITEM *astl)
310{
311 int i;
312
313 i = induc.indb.stg_avail++;
314 OPT_NEED(induc.indb, IND, 100);
315 IND_NM(i) = nme;
316 IND_LOAD(i) = load;
317 IND_FLAGS(i) = 0;
318 IND_BIVL(i) = NULL;
319 if (derived) {
320 IND_FAMILY(i) = IND_FAMILY(derived);
321 IND_DERIVED(i) = derived;
322 } else {
323 IND_FAMILY(i) = i;
324 IND_DERIVED(i) = 0;
325 }
326 IND_ASTL(i) = astl;
327 IND_OPC(i) = IND_SKIP(i) = 0;
328 IND_INITDEF(i) = 0;
329 NME_RFPTR(nme) = i;
330 return i;
331}

333/* step 1. determine which of the potential induction */
334/* variables are biv's */

336static void
337find_bivs(void)
338{
339 register int fgx, def, nme, ind, ilix;
340 int store_ili;
341 int top;
342
343 for (fgx = LP_FG(lpx); fgx; fgx = FG_NEXT(fgx)) {
344 for (def = FG_FDEF(fgx); def; def = DEF_LNEXT(def)) {
345 nme = DEF_NM(def);
346 ind = NME_RFPTR(nme);
347 if (NME_TYPE(nme) != NT_VAR || ind == 0)
348 continue;
349 if (IND_OMIT(ind))
350 continue;
351 if (DEF_CONFL(def)) {
352 if (OPTDBG(9, 2048))
353 fprintf(gbl.dbgfil, "--- find_bivs: omit nme %d (%s) - defconfl\n",
354 nme, getprint((int)NME_SYM(nme)));
355 IND_OMIT(ind) = 1;
356 continue;
357 }
358 store_ili = STD_AST(DEF_STD(def));
359 switch (A_TYPEG(store_ili)) {
360
361 case A_ENDDO:
362 ilix = is_biv(ind, def, (int)DEF_LHS(def), (int)DEF_RHS(def));
363 if (ilix == 0) {
364 /* The rhs is not in the correct form which seems odd
365 * for the def derived from a DO/ENDDO. An example of
366 * how this might occur is:
367 * m3 = 0
368 * do i = m1, m2, m3
369 * ...
370 * The def for the enddo will be entered as
371 * i = m1
372 * i.e., the increment is optimized away.
373 */
374 IND_OMIT(ind) = 1;
375 IND_LOAD(ind) = DEF_LHS(def);
376 break;
377 }
378 assert(ilix == DEF_LHS(def), "find_bivs():enddo load", ilix, 3);
379 IND_LOAD(ind) = ilix;
380 top = A_OPT2G(store_ili);
381 IND_INIT(ind) = A_M1G(top);
382 IND_PTR(ind) = 0;
383 break;
384
385 case A_ASN:
386 ilix = is_biv(ind, def, (int)DEF_LHS(def), (int)DEF_RHS(def));
387 if (ilix == 0) {
388 IND_OMIT(ind) = 1;
389 IND_LOAD(ind) = DEF_LHS(def);
390 } else if (IND_LOAD(ind) == 0) {
391 IND_INIT(ind) = IND_LOAD(ind) = ilix;
392 } else if (IND_LOAD(ind) != ilix)
393 IND_OMIT(ind) = 1;
394 IND_PTR(ind) = 0;
395 break;
396
397 default:
398 IND_OMIT(ind) = 1;
399 break;
400 }
401 if (OPTDBG(9, 2048)) {
402 if (IND_OMIT(ind))
403 fprintf(gbl.dbgfil, "--- find_bivs: omit nme %d (%s)\n", nme,
404 getprint((int)NME_SYM(nme)));
405 else
406 fprintf(gbl.dbgfil, "--- find_bivs: ind nme %d (%s) is a biv\n", nme,
407 getprint((int)NME_SYM(nme)));
408 }
409 }
410 }
411
412 {
413 /*
414 * for those induction variables which are deletable due to their
415 * storage class, determine if any of them are "live-out" simply due
416 * to control flow -- this is for the case where the value of the
417 * variable upon exit from the loop is used (later) upon entry to the
418 * loop. Typically, live-in/live-out analysis is done to solve this
419 * problem. Here, the solution is to check the OUT sets of the
420 * predecessors not in the loop of the loop head for the induction
421 * variables which are deletable.
422 *
423 * Also, for each basic induction variable, determine if there's
424 * a single definition of the induction variable which reaches the
425 * bottom of the predecessor of the loop head. If this definition's
426 * storing value is a constant and it occurs in a block which dominates
427 * the predecessor, then we can use the constant as the induction
428 * variable's initial value.
429 * Otherwise, we'll just use its default (i.e., the load of
430 * the variable). NOTE that we don't care if the induction variable
431 * cannot be deleted; this relies on the correctness of the reaching
432 * def information (i.e., a block's KILL set is computed correctly
433 * and call defs and ptr store defs are tracked correctly).
434 */
435 PSI_P pred;
436 Q_ITEM *p;
437 int rdef;
438
439 for (ind = 1; ind < induc.indb.stg_avail; ind++) {
440 if (IND_OMIT(ind))
441 continue;
442 p = IND_BIVL(ind);
443 if (p == NULL) {
444 IND_OMIT(ind) = 1;
445 if (OPTDBG(9, 2048))
446 fprintf(gbl.dbgfil, "--- find_bivs: omit nme %d (%s)\n",
447 (int)IND_NM(ind), getprint((int)NME_SYM(IND_NM(ind))));
448 continue;
449 }
450 if (!IND_DELETE(ind))
451 goto chk_reaching;
452 for (; p != NULL; p = p->next) {
453 def = p->info;
454 pred = FG_PRED(LP_HEAD(lpx));
455 for (; pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
456 fgx = PSI_NODE(pred);
457 if (FG_LOOP(fgx) == lpx)
458 continue;
459 if (def_in_out(fgx, def)) { /* is def in OUT of fgx */
460 if (OPTDBG(9, 2048))
461 fprintf(gbl.dbgfil, "--- def %d cannot be deleted (in OUT(%d))\n",
462 def, fgx);
463 IND_DELETE(ind) = 0;
464 goto chk_reaching;
465 }
466 }
467 }
468 chk_reaching:
469 /*
470 * First, find the loop head's predecessor; if we find more than
471 * one, give up.
472 */
473 fgx = pred_of_loop(lpx);
474 if (fgx == 0)
475 continue;
476 /*
477 * find a single reaching definition of the induction variable
478 * which reaches the end of the loop's predecessor.
479 */
480 nme = IND_NM(ind);
481 rdef = find_rdef(nme, fgx, FALSE); /* FALSE ==> end of block */
482
483 if (rdef && can_copy_def(rdef, fgx, FALSE)) {
484 /*
485 * it's safe to use the def's value as the init expression for
486 * the induction variable. If the def's value is a cse,
487 * the cse is removed (use its operand) -- may want to
488 * prune_cse.
489 */
490 int temp_ili;
491 temp_ili = DEF_RHS(rdef);
492 IND_INIT(ind) = temp_ili;
493 IND_INITDEF(ind) = rdef;
494 if (OPTDBG(9, 2048))
495 fprintf(gbl.dbgfil, "--- single init %d for %s\n", (int)IND_INIT(ind),
496 getprint((int)NME_SYM(nme)));
497 }
498 }
499 }
500
501 induc.last_biv = induc.indb.stg_avail - 1;
502
503 if (OPTDBG(9, 4096))
504 dump_ind();
505}

507/* must recursively search because some blocks are added after flow
508 * analysis and do not have OUT's defined.
509 */
510static int
511def_in_out(int fgx, int def)
512{
513 int ret;
514 int tempfgx;
515 PSI_P pred;
516
517 if (FG_OUT(fgx) != NULL) {
518 return bv_mem(FG_OUT(fgx), def);
519 }
520 /* else we are dealing with optimizer created block */
521 /* return true if def is found in any of this block's predecessors */
522 FG_VISITED(fgx) = 1; /* to prevent looping */
523 ret = 0;
524 pred = FG_PRED(fgx);
525 for (; pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
526 tempfgx = PSI_NODE(pred);
527 if (FG_LOOP(tempfgx) == lpx)
528 continue;
529 if (FG_VISITED(tempfgx) == 1)
530 continue;
531 if (def_in_out(tempfgx, def)) { /* recurse */
532 ret = 1;
533 break;
534 }
535 }
536 FG_VISITED(fgx) = 0;
537 return ret;
538}

540static int
541is_biv(int ind, int def, int lhs, int rhs)
542{
543 register int load, add;
544 register Q_ITEM *p;
545
546 add = rhs;
547 if (A_TYPEG(add) == A_BINOP && A_OPTYPEG(add) == OP_ADD) {
548 load = A_LOPG(add);
549 if (load == lhs) {
550 if (DEF_DOEND(def) || IS_INVARIANT((int)A_ROPG(add))) {
551 p = add_biv_def(ind, def);
552 p->flag = (2 << BIVL_OPN_SHF) | '+';
553 check_skip(ind, p);
554 return (load);
555 }
556 } else {
557 load = A_ROPG(add);
558 if (load == lhs) {
559 if (DEF_DOEND(def) || IS_INVARIANT((int)A_LOPG(add))) {
560 p = add_biv_def(ind, def);
561 p->flag = (1 << BIVL_OPN_SHF) | '+';
562 check_skip(ind, p);
563 return (load);
564 }
565 }
566 }
567 } else if (A_TYPEG(add) == A_BINOP && A_OPTYPEG(add) == OP_SUB) {
568 load = A_LOPG(add);
569 if (load == lhs) {
570 if (DEF_DOEND(def) || IS_INVARIANT((int)A_ROPG(add))) {
571 p = add_biv_def(ind, def);
572 p->flag = (2 << BIVL_OPN_SHF) | '-';
573 check_skip(ind, p);
574 return (load);
575 }
576 }
577 }
578 return (0);
579}

592/*
593 * for an induction definition just added, check its skip against what's
594 * already there (if any)
595 */
596static void
597check_skip(int ind, Q_ITEM *bivl)
598{
599 int def;
600 int value, opc, skip;
601
602 if (bivl->next)
603 IND_MIDF(ind) = 1;
604 def = bivl->info;
605 value = DEF_RHS(def);
606 if (((bivl->flag >> BIVL_OPN_SHF) & BIVL_OPN_MSK) == 1)
607 skip = A_LOPG(value);
608 else
609 skip = A_ROPG(value);
610 opc = A_OPTYPEG(value);
611 if (bivl->next) {
612 if (skip != IND_SKIP(ind) || opc != IND_OPC(ind))
613 IND_SKIP(ind) = 0;
614 } else {
615 IND_SKIP(ind) = skip;
616 IND_OPC(ind) = opc;
617 }

650static void
651scan_ind_uses(int ind)
652{
653 /*
654 * the first thing to do is to scan all of the uses of the
655 * induction definitions of the current induction variable
656 * marking those which are induction expressions
657 */
658 Q_ITEM *p;
659 ADU *adu;
660
661 srch.ind = ind;
662
663 srch.lastadu = srch.adu_queue = NULL;
664 /*
665 * process the uses of the definitions of the basic induction variable;
666 * this will discover aliases which will be added to the 'adu' queue.
667 */
668 for (p = IND_BIVL(ind); p != NULL; p = p->next)
669 scan_def((int)p->info, ind);
670 /*
671 * process any aliases found for this definition; 'aliases of aliases'
672 * are discovered during this process, in which case the queue is
673 * extended.
674 */
675 for (adu = srch.adu_queue; adu != NULL; adu = adu->next)
676 scan_def(adu->def, adu->ind);
677 /*
678 * for all of the aliases found for the induction variable:
679 * 1. ensure that the delete flag is correct (if set, the alias
680 * definition can be deleted);
681 * 2. compute the initial value of the derived induction variables.
682 */
683 for (adu = srch.adu_queue; adu != NULL; adu = adu->next) {
684 int drv;
685 int ii;
686
687 ii = adu->ind;
688 drv = IND_DERIVED(ii);
689 if (IND_NIU(drv))
690 IND_DELETE(ii) = 0;
691 ast_visit(1, 1);
692 ast_replace((int)IND_LOAD(drv), (int)IND_INIT(drv));
693 IND_INIT(ii) = ast_rewrite((int)DEF_RHS(adu->def));
694 ast_unvisit();
695 }
696
697 /*
698 * if a branch use was found in find_fam, it is determined if it can be
699 * turned into a check of an iteration count (i.e. a countable loop)
700 */
701 if (srch.branch.ili && srch.branch.ind == ind)
702 do_branch();
703
704}

707scan_def(int def, int ind)
708{
709 DU *du;
710 int use;
711 int lpuse;
712
713 srch.ind = ind;
714 srch.def = def;
715 srch.biv_ili = STD_AST(DEF_STD(def));
716 if (OPTDBG(9, 2048))
717 fprintf(gbl.dbgfil, " def %d\n", def);
718 for (du = DEF_DU(def); du != NULL; du = du->next) {
719 srch.use = use = du->use;
720 if (STD_DELETE(USE_STD(use))) {
721 if (OPTDBG(9, 2048))
722 fprintf(gbl.dbgfil, " use %d deleted\n", use);
723 goto next_du;
724 }
725 if ((lpuse = FG_LOOP(USE_FG(use))) != lpx) {
726 lpuse = LP_PARENT(lpuse);
727 for (; lpuse; lpuse = LP_PARENT(lpuse)) {
728 if (lpuse == lpx) {
729 IND_NIU(ind) = 1;
730 if (OPTDBG(9, 2048))
731 fprintf(gbl.dbgfil, " use %d in contained loop %d\n", use,
732 lpuse);
733 goto next_du;
734 }
735 }
736 IND_DELETE(ind) = 0;
737 if (OPTDBG(9, 2048))
738 fprintf(gbl.dbgfil, " use %d not in loop %d\n", use, lpuse);
739 } else if (USE_DOINIT(use)) {
740 /* The use was created when a doinit def was created. Its 'ast'
741 * actually appears in the next (FG_LNEXT) flowgraph node. This
742 * node is not a member of the loop region denoted by lpx;
743 * consequently, this use is skipped.
744 */
745 ;
746 } else {
747 srch.load = USE_AST(use);
748 if (OPTDBG(9, 2048))
749 fprintf(gbl.dbgfil, " use %d, load %d \"%s\"\n", use, srch.load,
750 astb.atypes[A_TYPEG(srch.load)]);
751 srch.new_skip_opc = IND_OPC(ind);
752 srch.new_skip = IND_SKIP(ind);
753 (void)find_fam((int)STD_AST(USE_STD(use)));
754 }
755 next_du:;
756 }
757#if DEBUG
758 if (OPTDBG(9, 2048))
759 fprintf(gbl.dbgfil, " def %d, non-biv uses: %d\n", def, IND_NIU(ind));
760#endif
761}

763static LOGICAL _fam(int ilix, int *fm_p);
764
765/*
766 * recurse thru an ili tree searching for the "load" of the current
767 * induction variable. Find all induction uses of the induction variable
768 * and create new induction variables (i.e., the current induction variable's
769 * family). Note that find_fam also is designed to work when we're
770 * processing a new induction variable (i.e., the "load" is actually some
771 * linear expression which was previously "replaced").
772 *
773 * This function returns 0 (not an induction use), or the induction table
774 * entry of the induction variable replacing the current ili (ilix).
775 */
776static int
777find_fam(int ilix)
778{
779 int fm;
780
781 fm = 0;
782 ast_traverse_all(ilix, _fam, NULL, &fm);
783 return fm;
784}
785
786static LOGICAL
787_fam(int ilix, int *fm_p)
788{
789 int opc, i;
790 int i1, i2;
791 int asd;
792 int subflg[7];
793
794 if (ilix == srch.load) {
795 srch.new_skip = IND_SKIP(srch.ind);
796 srch.new_skip_opc = IND_OPC(srch.ind);
797 *fm_p = srch.ind;
798 return TRUE;
799 }
800
801 i1 = i2 = 0;
802 switch (opc = A_TYPEG(ilix)) {
803
804 case A_DO:
805 /* don't search A_DOVAR */
806 ast_traverse_all((int)A_M1G(ilix), _fam, NULL, &i1);
807 if (i1)
808 goto not_iuse;
809 i1 = 0;
810 ast_traverse_all((int)A_M2G(ilix), _fam, NULL, &i1);
811 if (i1)
812 goto not_iuse;
813 if (A_M3G(ilix)) {
814 i1 = 0;
815 ast_traverse_all((int)A_M3G(ilix), _fam, NULL, &i1);
816 if (i1)
817 goto not_iuse;
818 }
819 if (A_M4G(ilix)) {
820 i1 = 0;
821 ast_traverse_all((int)A_M4G(ilix), _fam, NULL, &i1);
822 if (i1)
823 goto not_iuse;
824 }
825 break;
826 case A_ENDDO:
827 if (OPTDBG(9, 2048))
828 fprintf(gbl.dbgfil, " : enddo %d\n", ilix);
829 srch.branch.ind = srch.ind;
830 srch.branch.ili = ilix;
831 srch.branch.opn = 0;
832 break;
833
834 case A_BINOP:
835 switch (A_OPTYPEG(ilix)) {
836 case OP_ADD:
837 ast_traverse_all((int)A_LOPG(ilix), _fam, NULL, &i1);
838 if (i1) {
839 int s1, o1, s2, o2, d;
840 if (IS_INVARIANT(A_ROPG(ilix)))
841 goto is_iuse;
842 s1 = srch.new_skip;
843 o1 = srch.new_skip_opc;
844 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i2);
845 if (i2) {
846 s2 = srch.new_skip;
847 o2 = srch.new_skip_opc;
848 d = A_DTYPEG(ilix);
849 /* must add the two skips together */
850 if (o1 == OP_ADD && o2 == OP_ADD) {
851 srch.new_skip = mk_binop(OP_ADD, s1, s2, d);
852 srch.new_skip_opc = OP_ADD;
853 } else if (o1 == OP_ADD && o2 == OP_SUB) {
854 srch.new_skip = mk_binop(OP_SUB, s1, s2, d);
855 srch.new_skip_opc = OP_ADD;
856 } else if (o1 == OP_SUB && o2 == OP_ADD) {
857 srch.new_skip = mk_binop(OP_SUB, s2, s1, d);
858 srch.new_skip_opc = OP_ADD;
859 } else if (o1 == OP_SUB && o2 == OP_SUB) {
860 srch.new_skip = mk_binop(OP_ADD, s1, s2, d);
861 srch.new_skip_opc = OP_SUB;
862 } else {
863 goto not_iuse;
864 }
865 goto is_iuse;
866 }
867 goto not_iuse;
868 }
869 i1 = 0;
870 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i1);
871 if (i1) {
872 if (IS_INVARIANT(A_LOPG(ilix)))
873 goto is_iuse;
874 goto not_iuse;
875 }
876 break;
877 case OP_SUB:
878 ast_traverse_all((int)A_LOPG(ilix), _fam, NULL, &i1);
879 if (i1) {
880 int s1, o1, s2, o2, d;
881 if (IS_INVARIANT(A_ROPG(ilix)))
882 goto is_iuse;
883 s1 = srch.new_skip;
884 o1 = srch.new_skip_opc;
885 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i2);
886 if (i2) {
887 s2 = srch.new_skip;
888 o2 = srch.new_skip_opc;
889 d = A_DTYPEG(ilix);
890 /* must add the two skips together */
891 if (o1 == OP_ADD && o2 == OP_ADD) {
892 srch.new_skip = mk_binop(OP_SUB, s1, s2, d);
893 srch.new_skip_opc = OP_ADD;
894 } else if (o1 == OP_ADD && o2 == OP_SUB) {
895 srch.new_skip = mk_binop(OP_ADD, s1, s2, d);
896 srch.new_skip_opc = OP_ADD;
897 } else if (o1 == OP_SUB && o2 == OP_ADD) {
898 srch.new_skip = mk_binop(OP_ADD, s1, s2, d);
899 srch.new_skip_opc = OP_SUB;
900 } else if (o1 == OP_SUB && o2 == OP_SUB) {
901 srch.new_skip = mk_binop(OP_SUB, s2, s1, d);
902 srch.new_skip_opc = OP_ADD;
903 } else {
904 goto not_iuse;
905 }
906 goto is_iuse;
907 }
908 goto not_iuse;
909 }
910 i1 = 0;
911 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i1);
912 if (i1) {
913 if (IS_INVARIANT(A_LOPG(ilix))) {
914 if (srch.new_skip)
915 srch.new_skip_opc = srch.new_skip_opc == OP_ADD ? OP_SUB : OP_ADD;
916 goto is_iuse;
917 }
918 goto not_iuse;
919 }
920 break;
921 case OP_MUL:
922 ast_traverse_all((int)A_LOPG(ilix), _fam, NULL, &i1);
923 if (i1) {
924 if (IS_INVARIANT(A_ROPG(ilix))) {
925 if (srch.new_skip)
926 srch.new_skip = mk_binop(OP_MUL, srch.new_skip, (int)A_ROPG(ilix),
927 (int)A_DTYPEG(ilix));
928 goto is_iuse;
929 }
930 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i2);
931 if (i2) {
932 if (srch.new_skip)
933 srch.new_skip = mk_binop(OP_MUL, srch.new_skip, srch.new_skip,
934 (int)A_DTYPEG(srch.new_skip));
935 goto is_iuse;
936 }
937 goto not_iuse;
938 }
939 i1 = 0;
940 ast_traverse_all((int)A_ROPG(ilix), _fam, NULL, &i1);
941 if (i1) {
942 if (IS_INVARIANT(A_LOPG(ilix))) {
943 if (srch.new_skip)
944 srch.new_skip = mk_binop(OP_MUL, srch.new_skip, (int)A_LOPG(ilix),
945 (int)A_DTYPEG(ilix));
946 goto is_iuse;
947 }
948 goto not_iuse;
949 }
950 break;
951 default:
952 goto check_operands;
953 }
954 break;
955
956 case A_UNOP:
957 if (A_OPTYPEG(ilix) == OP_SUB) {
958 ast_traverse_all((int)A_LOPG(ilix), _fam, NULL, &i1);
959 if (i1) {
960 srch.new_skip_opc = srch.new_skip_opc == OP_ADD ? OP_SUB : OP_ADD;
961 goto is_iuse;
962 }
963 }
964 break;
965
966 case A_SUBSCR:
967 ast_traverse_all((int)A_LOPG(ilix), _fam, NULL, &i1);
968 asd = A_ASDG(ilix);
969 for (i = 0; i < (int)ASD_NDIM(asd); i++) {
970 i2 = 0;
971 ast_traverse_all((int)ASD_SUBS(asd, i), _fam, NULL, &i2);
972 if (i2)
973 subflg[i] = -1;
974 else
975 subflg[i] = IS_INVARIANT(ASD_SUBS(asd, i));
976 }
977 i1 = 0; /* other subscripts variant */
978 i2 = 0; /* any induction uses */
979 for (i = 0; i < (int)ASD_NDIM(asd); i++) {
980 if (subflg[i] < 0)
981 i2++;
982 else if (!subflg[i])
983 i1++;
984 }
985 if (i2 && i1)
986 goto not_iuse;
987 break;
988
989 case A_ASN:
990 if (is_biv_store(srch.ind, ilix)) {
991 if (OPTDBG(9, 2048))
992 fprintf(gbl.dbgfil, " -- biv store %d^\n", ilix);
993 break;
994 }
995
996 ast_traverse_all((int)A_DESTG(ilix), _fam, NULL, &i1);
997 ast_traverse_all((int)A_SRCG(ilix), _fam, NULL, &i1);
998 if (i1) {
999 /*
1000 * if the value of an induction variable (basic or derived)
1001 * is stored, check if this is an alias candidate if the stored
1002 * iv is the current iv. Otherwise, create a use item for
1003 * the expression and add it to the du chain of the (derived)
1004 * induction variable --- note that this defers the alias check
1005 * until the current iv is this induction variable.
1006 */
1007 if (check_alias(ilix))
1008 break;
1009 IND_NIU(i1) = 1;
1010 if (OPTDBG(9, 2048))
1011 fprintf(gbl.dbgfil, " : other stuse %d^(%s), ind %d\n", ilix,
1012 astb.atypes[A_TYPEG(ilix)], i1);
1013 }
1014 break;
1015
1016 default:
1017 check_operands:
1018 /* check the operands of the ast */
1019 i1 = 0;
1020 ast_trav_recurse(ilix, &i1);
1021 if (i1)
1022 goto not_iuse;
1023 break;
1024 }
1025 return TRUE; /* stop the traverse */
1026
1027is_iuse:
1028 *fm_p = srch.ind;
1029 return TRUE; /* stop the traverse */
1030
1031not_iuse:
1032 IND_NIU(srch.ind) = 1;
1033 if (OPTDBG(9, 2048))
1034 fprintf(gbl.dbgfil, " : other use %d^(%s), ind %d\n", ilix,
1035 astb.atypes[A_TYPEG(ilix)], i1);
1036 return TRUE; /* stop the traverse */
1037}

1108static void
1109do_branch(void)
1110{
1111 /*
1112 * if a branch use was found in find_fam, it is determined if it can be
1113 * turned into a check of an iteration count (i.e. a countable loop); the
1114 * conditions are:
1115 * 1. loop cannot contain calls (determined by find_fam).
1116 * 2. the branch is the last use in the tail (determined by find_fam).
1117 * 3. all induction variables are of the form
1118 * i = i + c OR i = i - c,
1119 * where c is a constant and is the same value in each definition.
1120 * 4. for each induction definition, the flow graph node containing the
1121 * definition dominates the tail of the loop.
1122 * 5. the comparison involved is one of >=, >, <=, or <.
1123 *
1124 * NOTE that an additional optimization is to use the iteration count
1125 * in a target's "loop count" instruction.
1126 */
1127 int astx;
1128 int ind;
1129
1130 ind = srch.branch.ind;
1131 /*
1132 * don't use loop instruction if this is a loop which has multiple
1133 * tails.
1134 */
1135 if (EDGE_NEXT(LP_EDGE(lpx)) >= 0)
1136 goto br_not_optz;
1137 /*
1138 * allow loop instruction if there are any non-induction uses unless
1139 * this restriction is requested.
1140 */
1141 if (XBIT(8, 0x80) && IND_NIU(ind))
1142 goto br_not_optz;
1143
1144 /* look at the type of the branch */
1145
1146 if (A_TYPEG(srch.branch.ili) == A_ENDDO) {
1147 int tt;
1148 astx = A_OPT2G(srch.branch.ili);
1149#if DEBUG
1150 assert(A_TYPEG(astx) == A_DO, "do_branch:mismatched do-enddo",
1151 srch.branch.ili, 3);
1152#endif
1153 opt.cntlp.cnt = conv_to_int(A_M2G(astx));
1154 tt = conv_to_int(A_M1G(astx));
1155 opt.cntlp.cnt = mk_binop(OP_SUB, opt.cntlp.cnt, tt, DT_INT);
1156 if ((tt = A_M3G(astx))) {
1157 tt = conv_to_int(tt);
1158 opt.cntlp.cnt = mk_binop(OP_ADD, opt.cntlp.cnt, tt, DT_INT);
1159 opt.cntlp.cnt = mk_binop(OP_DIV, opt.cntlp.cnt, tt, DT_INT);
1160 } else
1161 opt.cntlp.cnt = mk_binop(OP_ADD, opt.cntlp.cnt, astb.i1, DT_INT);
1162 } else
1163 goto br_not_optz;
1164
1165 opt.cntlp.branch = srch.branch.ili;
1166 induc.branch_ind = ind;
1167
1168 loop_cnt = opt.cntlp.cnt;
1169 /*
1170 * set the iteration count which is used for last value computation.
1171 * Also, this value indicates that the loop is countable.
1172 * A countable loop cannot have multiple exits.
1173 * NOTE: even though we can't compute last values, it's still
1174 * possible (and important) to use the loop instruction.
1175 */
1176 if (LP_MEXITS(lpx)) {
1177 if (OPTDBG(9, 2048))
1178 fprintf(gbl.dbgfil, "--- no lastval loop count, multiple exits\n");
1179 lastval_cnt = 0;
1180 } else {
1181 lastval_cnt = opt.cntlp.cnt;
1182 if (OPTDBG(9, 2048))
1183 fprintf(gbl.dbgfil, "--- lastval loop count is ili %d\n", lastval_cnt);
1184 }
1185
1186 goto br_complete;
1187
1188br_not_optz:
1189 if (OPTDBG(9, 2048))
1190 fprintf(gbl.dbgfil, "--- branch %d(%s) not optimized\n", srch.branch.ili,
1191 astb.atypes[A_TYPEG(srch.branch.ili)]);
1192 opt.cntlp.cnt = 0;
1193 IND_NIU(ind) = 1;
1194
1195br_complete:;
1196}

1217/* a store of an induction expression has been found. determine if the stored
1218 * variable can be treated as an alias for the induction expression.
1219 */
1220static LOGICAL
1221check_alias(int store)
1222{
1223 int nme, sym;
1224 int ind;
1225 int def;
1226 int i;
1227 int nuses;
1228 DU *du;
1229 ADU *adu;
1230 Q_ITEM *p;
1231 int use;
1232 int lpuse;
1233
1234 nme = A_NMEG(A_DESTG(store));
1235 if (NME_TYPE(nme) != NT_VAR || !ISSCALAR(sym = NME_SYM(nme)) ||
1236 !DT_ISINT(DTYPEG(sym)) || (ind = NME_RFPTR(nme)) == 0 || IND_RMST(ind) ||
1237 DT_ISCMPLX(DTYPEG(sym)) ||
1238 !is_sym_optsafe(nme, lpx))
1239 return FALSE;
1240
1241 if (IND_ALIAS(ind)) /* it's already an alias */
1242 return TRUE;
1243
1244 /* Only one def of sym allowed in loop */
1245
1246 def = 0;
1247 for (i = NME_DEF(nme); i; i = DEF_NEXT(i)) {
1248 if (FG_LOOP(DEF_FG(i)) == lpx) {
1249 if (def)
1250 return FALSE;
1251 def = i;
1252 }
1253 }
1254#if DEBUG
1255 assert(STD_AST(DEF_STD(def)) == store, "check_alias: def bad", def, 3);
1256#endif
1257 if (DEF_DELETE(def) && DEF_DU(def) == NULL) {
1258 /* can get here if delete_store (flow) deleted a def which does not
1259 * have any uses. If the def contains a use of the induction variable,
1260 * it still shows up in the induction variable's du. If we allow the
1261 * processing to continue, a "bug" could occur since we'll end up
1262 * deleting a def that has already been deleted (its ilt is removed
1263 * via unlnkilt more than once).
1264 */
1265 if (OPTDBG(9, 2048))
1266 fprintf(gbl.dbgfil, " alias use, store %d^, %s, already deleted\n",
1267 store, SYMNAME(sym));
1268 return TRUE;
1269 }
1270 /*
1271 * Scan all uses of the def. For those uses in the loop, check if this is
1272 * the only definition that reaches the uses along with other criteria
1273 * (such as if the def contains any external references, if it is defined
1274 * upon entry and the use can be reached from the entry, etc.).
1275 */
1276 nuses = 0; /* number of uses not found in the loop or loop nest */
1277 for (du = DEF_DU(def); du != NULL; du = du->next) {
1278 use = du->use;
1279 if ((lpuse = FG_LOOP(USE_FG(use))) != lpx) {
1280 lpuse = LP_PARENT(lpuse);
1281 for (; lpuse; lpuse = LP_PARENT(lpuse)) {
1282 if (lpuse == lpx) {
1283 IND_NIU(ind) = 1;
1284 goto next_du;
1285 }
1286 }
1287 nuses++;
1288 } else if (!single_ud(use)) {
1289 return FALSE;
1290 }
1291 next_du:;
1292 }
1293
1294 if (OPTDBG(9, 2048))
1295 fprintf(gbl.dbgfil, " alias use, store %d^, %s, # out uses %d\n",
1296 store, SYMNAME(sym), nuses);
1297 /*
1298 * create a new adu record and initialize its fields; it needs to be added
1299 * to the end of the queue (i.e., it's a queue).
1300 */
1301 adu = (ADU *)getitem(Q_AREA, sizeof(ADU));
1302 adu->def = def;
1303 adu->ind = ind;
1304 adu->next = NULL;
1305 if (srch.adu_queue == NULL)
1306 srch.adu_queue = adu;
1307 else
1308 srch.lastadu->next = adu;
1309 srch.lastadu = adu;
1310
1311 IND_ALIAS(ind) = 1;
1312 IND_INIT(ind) = DEF_RHS(def);
1313 IND_FAMILY(ind) = IND_FAMILY(srch.ind);
1314 IND_SKIP(ind) = srch.new_skip;
1315 IND_OPC(ind) = srch.new_skip_opc;
1316 IND_DERIVED(ind) = srch.ind;
1317 p = add_biv_def(ind, def); /* create a biv item so its def is recorded */
1318 p->flag = 'a';
1319
1320 /* currently, we don't try to compute an alias' last value. */
