19/*
20 * mode == ' ' means file is not open
21 * mode == 'r' means file is open for read
22 * mode == 'w' means file is open for write
23 * mode == 'e' means file was open for read but had reached end of file
24 */
25static char mode = ' ';
26static FILE *df = NULL;
27static void dump_buff(char);
28static DREC t;
29
30/*****************************************************************/
31
32void
33dinit_init(void)
34{
35 mode = ' '; /* neither read nor write */
36 if (df) {
37 mode = 'e';
38 fseek(df, 0L, 0);
39 }
40}

42/*****************************************************************/
43
44void
45dinit_put(DTYPE dtype, ISZ_T conval)
46{
47 int n;
48
49 if (mode == 'e') {
50 mode = 'w';
51 } else if (mode == ' ') {
52 if ((df = tmpfile()) == NULL)
53 errfatal(F_0005_Unable_to_open_temporary_file);
54 mode = 'w';
55 } else if (mode != 'w') {
56 error(F_0010_File_write_error_occurred_OP1, ERR_Fatal, 0,
57 "(data init file)", CNULL);
58 }
59
60 t.dtype = dtype;
61 t.conval = conval;
62 if (DBGBIT(6, 1))
63 dump_buff(mode);
64
65 n = fwrite((char *)&t, sizeof(t), 1, df);
66 if (n != 1)
67 error(F_0010_File_write_error_occurred_OP1, ERR_Fatal, 0,
68 "(data init file)", CNULL);
69}

71/*
72 * must appear after DINIT_STRING record, so the file must be open in mode 'w'
73 */
74void
75dinit_put_string(ISZ_T len, char *str)
76{
77 int n;
78 if (df == NULL || mode != 'w')
79 error(F_0010_File_write_error_occurred_OP1, ERR_Fatal, 0, "(data init file)", CNULL);
80 if (DBGBIT(6, 1))
81 fprintf(gbl.dbgfil, " string(%d)\n", (int)len);
82
83 n = fwrite(str, 1, len, df);
84 if (n != len)
85 error(F_0010_File_write_error_occurred_OP1, ERR_Fatal, 0, "(data init file)", CNULL);
86} /* dinit_put_string */

88/*******************************************************/
89
90DREC *
91dinit_read(void)
92{
93 register int n;
94
95 if (mode == ' ' || mode == 'e' || df == NULL)
96 return NULL;
97 if (mode == 'w') {
98 t.dtype = DINIT_ENDFILE;
99 t.conval = 0;
100 n = fwrite((char *)&t, sizeof(t), 1, df);
101 n = fseek(df, 0L, 0);
102 assert(n == 0, "dinit_read:bad rewind", n, ERR_Fatal);
103 mode = 'r';
104 }
105
106 n = fread((char *)&t, sizeof(t), 1, df);
107 if (n == 0) { /* end of file */
108 mode = 'e';
109 return NULL;
110 }
111 if (t.dtype == DINIT_ENDFILE) {
112 mode = 'e';
113 return NULL;
114 }
115
116 if (DBGBIT(6, 1))
117 dump_buff(mode);
118 return &t;
119}

133static void
134dump_buff(char mode)
135{
136 char buf[32];
137
138 fprintf(gbl.dbgfil, " %c ", mode);
139 if (t.dtype > 0 && t.dtype < stb.dt.stg_avail) {
140 getdtype(t.dtype, buf);
141 fprintf(gbl.dbgfil, "dtype: %s conval: %ld\n", buf, t.conval);
142 } else if (t.dtype == 0) {
143 }
144#ifdef DINIT_MODE
145 else if (t.dtype == DINIT_MODE)
146 fprintf(gbl.dbgfil, " mode %d\n", (int)t.conval);
147#endif
148#ifdef DINIT_FUNCCOUNT
149 else if (t.dtype == DINIT_FUNCCOUNT)
150 fprintf(gbl.dbgfil, " funccount %d\n", (int)t.conval);
151#endif
152 else {
153 if (t.dtype == DINIT_LOC)
154 fprintf(gbl.dbgfil, "DINIT_LOC %s ", getprint((int)t.conval));
155 else if (t.dtype == DINIT_SLOC)
156 fprintf(gbl.dbgfil, "DINIT_SLOC %s ", getprint((int)t.conval));
157 else if (t.dtype == DINIT_REPEAT)
158 fprintf(gbl.dbgfil, "DINIT_REPEAT ");
159 else if (t.dtype == DINIT_OFFSET)
160 fprintf(gbl.dbgfil, "DINIT_OFFSET ");
161 else if (t.dtype == DINIT_LABEL)
162 fprintf(gbl.dbgfil, "DINIT_LABEL %s ", getprint((int)t.conval));
163 else if (t.dtype == DINIT_ZEROES)
164 fprintf(gbl.dbgfil, "DINIT_ZEROES ");
165 else if (t.dtype == DINIT_SECT)
166 fprintf(gbl.dbgfil, "DINIT_SECT ");
167 else if (t.dtype == DINIT_DATASECT)
168 fprintf(gbl.dbgfil, "DINIT_DATASECT ");
169#ifdef DINIT_STRING
170 else if (t.dtype == DINIT_STRING)
171 fprintf(gbl.dbgfil, " string len=%d\n", (int)t.conval);
172#endif
173 else
174 fprintf(gbl.dbgfil, "dtype: %4d ", t.dtype);
175 fprintf(gbl.dbgfil, " conval: %10" ISZ_PF "d (0x%08" ISZ_PF "X)\n",
176 t.conval, t.conval);
177 }
178}

212void
213dinit_end(void)
214{
215 if (df) {
216 fclose(df);
217 df = NULL;
218 }
219 /* if this is block data, need to free the ilmb memory that
220 would ordinarily be freed in expand. purify MLK (memory
221 leak) error was being reported. */
222 if (gbl.rutype == RU_BDATA && ilmb.ilm_base) {
223 FREE(ilmb.ilm_base);
224 ilmb.ilm_base = NULL;
225 }
226 mode = ' '; /* no file */
227}
