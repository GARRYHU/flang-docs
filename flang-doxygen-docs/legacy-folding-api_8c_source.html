7/** \file
8 * \brief Implement legacy folding interfaces
9 *
10 * Implement the legacy compile-time evaluation library atop
11 * native host integer arithmetic, the floating-point
12 * folding package, and whatever 128-bit integer support
13 * has been supplied in int128.h.
14 *
15 * These interfaces date back to the days when IEEE-754 floating-point
16 * arithmetic was a novelty and cross-compilation was a common practice.
17 * They were formerly implemented with an IEEE-754 arithmetic emulation
18 * software package that had been translated into C from its original
19 * PDP-11 assembly language. These interfaces have been cleaned up
20 * to some degree, though more work remains to be done. This particular
21 * implementation is new, and comprises mostly conversions between
22 * the operand and result types of these legacy interfaces and those
23 * of the underlying integer and floating-point folding packages.
24 */

58static void
59unwrap_l(int64_t *x, const DBLINT64 y)
60{
61 *x = (int64_t) y[0] << 32 | (int64_t) (uint32_t) y[1];
62}

96static void
97unwrap_d(float64_t *x, const IEEE64 d)
98{
99 union {
100 float64_t x;
101 uint32_t i[2];
102 } u;
103 int le = is_host_little_endian();
104 u.i[le] = d[0]; /* big end */
105 u.i[le ^ 1] = d[1];
106 *x = u.x;
107}
108
109static void
110wrap_d(IEEE64 res, const float64_t *x)
111{
112 union {
113 float64_t d;
114 uint32_t i[2];
115 } u;
116 u.d = *x;
117 int le = is_host_little_endian();
118 res[0] = u.i[le]; /* big end */
119 res[1] = u.i[le ^ 1];
120}

378/*
379 * 128-bit integer routines
380 *
381 * Note: if GCC >= 4.8 were guaranteed to be available, its
382 * built-in __int128 type could be used here explicitly.
383 * For flexibility, the wrapper API in int128.h is used.
384 */
385
386static void
387unwrap_i128(int128_t *result, const INT128 i /* big-endian */)
388{
389 int j;
390 int128_from_int64(result, 0);
391 for (j = 0; j < 4; ++j) {
392 int128_t part, shifted;
393 int128_from_uint64(&part, (uint32_t) i[j ^ 3]);
394 int128_shift_left(&shifted, &part, 32 * j);
395 int128_or(result, result, &shifted);
396 }
397}
398
399static void
400wrap_i128(INT128 i, const int128_t *x)
401{
402 int j;
403 for (j = 0; j < 4; ++j) {
404 int128_t t;
405 int64_t least;
406 int128_shift_right_logical(&t, x, 32 * j);
407 int128_to_int64(&least, &t);
408 i[j ^ 3] = least;
409 }
410}

412void
413shf128(INT128 arg, int count, INT128 result)
414{
415 int128_t x, res;
416 unwrap_i128(&x, arg);
417 if (count < 0) {
418 /* negative count means right shift */
419 int128_shift_right_logical(&res, &x, -count);
420 } else {
421 int128_shift_left(&res, &x, count);
422 }
423 wrap_i128(result, &res);
424}

485check(enum fold_status status)
486{
487 switch (status) {
488 case FOLD_OK:
489 case FOLD_INEXACT:
490 break;

492 fperror(FPE_INVOP);
493 break;
494 case FOLD_OVERFLOW:
495 fperror(FPE_FPOVF);
496 break;
497 case FOLD_UNDERFLOW:
498 fperror(FPE_FPUNF);
499 break;
500 }
501}

785/*
786 * Copy a floating-point literal into a null-terminated buffer
787 * so that it may be passed to strtod() et al. Insert a leading "0x"
788 * after the sign, if requested; also transform a Fortran double-precision
789 * exponent character 'd'/'D'/'q'/'Q' into 'e'.
790 */
791static void
792get_literal(char *buffer, size_t length, const char *s, int n, bool is_hex)
793{
794 char *p = buffer;
795 assert(length > 0);
796 while (n > 0 && isspace(*s)) {
797 ++s;
798 --n;
799 }
800 if (n > 0 && (*s == '-' || *s == '+') && length > 1) {
801 *p++ = *s++;
802 --n;
803 --length;
804 }
805 if (is_hex && length > 3) {
806 *p++ = '0';
807 *p++ = 'x';
808 length -= 2;
809 }
810 while (n > 0 && *s && length > 1) {
811 char ch = *s++;
812 --n;
813 if (!is_hex && (ch == 'd' || ch == 'D' || ch == 'q' || ch == 'Q'))
814 ch = 'e';
815 *p++ = ch;
816 --length;
817 }
818 *p = '\0';
819}

823 int errno_capture, bool iszero) {
824 if (end != buffer + strlen(buffer))
825 return -1; /* format error */
826 /* The only error value documented for strtod() is ERANGE,
827 * but let's be defensive and produce at least a warning
828 * if some other error condition was raised.
829 */
830 if (errno_capture == ERANGE && iszero)
831 return iszero ? -3 /* underflow */ : -2 /* overflow */;
832 return errno_capture == 0 ? 0 : -2 /* overflow */;
833}
