376/** \brief getenv (uses env, not environ) */
377char *
378__fort_getenv(const char *nm)
379{
380 char **e;
381 int n;
382
383 n = strlen(nm);
384#if defined(TARGET_OSX)
385 e = env;
386#else
387 e = environ;
388#endif
389 while (*e != NULL) {
390 if ((strncmp(*e, nm, n) == 0) && ((*((*e) + n)) == '=')) {
391 return ((*e) + n + 1);
392 }
393 e++;
394 }
395 return (NULL);
396}

398/* init option processing */
399
400static void
401__fort_initopt()
402{
403 char *p, *q;
404 int i;
405
406 p = __fort_getenv("PGDIST_OPTS");
407 if (p == NULL) {
408 return;
409 }
410 if (optarea != NULL) {
411 __fort_free(optarea);
412 }
413 optarea = __fort_malloc(strlen(p) + 1);
414 q = optarea;
415 strcpy(q, p);
416 i = 0;
417 while (1) {
418 while (*q == ' ') {
419 q++;
420 }
421 if (*q == '\0') {
422 break;
423 }
424 if (i >= (MAXOPTS - 1)) {
425 __fort_abort("PGDIST_OPTS: too many options");
426 }
427 opts[i++] = q;
428 while ((*q != ' ') && (*q != '\0')) {
429 q++;
430 }
431 if (*q == ' ') {
432 *q++ = '\0';
433 }
434 }
435 opts[i] = NULL;
436}

438/* get option (command line -xx and environment */
439const char
440*__fort_getopt(const char *opt)
441{
442 char env[64];
443 char *p;
444 const char *q;
445 int n;
446
447 if (arg == NULL)
448 return NULL;
449 p = NULL;
450 for (n = 0; arg[n] != NULL; n++) {
451 if (strcmp(arg[n], opt) == 0) {
452 p = arg[n + 1];
453 if (p == NULL) {
454 return "";
455 }
456 break;
457 }
458 }
459 if (p == NULL) {
460 strcpy(env, "PGHPF_");
461 p = env + 6;
462 q = opt + 1;
463 while (*q != '\0') {
464 *p++ = toupper(*q++);
465 }
466 *p++ = '\0';
467 p = __fort_getenv(env);
468 }
469 if (p == NULL) {
470 for (n = 0; opts[n] != NULL; n++) {
471 if (strcmp(opts[n], opt) == 0) {
472 p = opts[n + 1];
473 if (p == NULL) {
474 return "";
475 }
476 break;
477 }
478 }
479 }
480 if ((strcmp(opt, "-g") == 0) && (p != NULL) && (*p == '-')) {
481 return "";
482 }
483 return p;
484}

543/* init stats (set options) */
544
545static void
546__fort_istat()
547{
548 const char *p;
549
550 p = __fort_getopt("-stat");
551 if (p == NULL) {
552 return;
553 }
554 if ((*p == '\0') || (*p == '-')) {
555 p = "all";
556 }
557 while (1) {
558 if (strncmp(p, "cpus", 4) == 0) {
559 __fort_quiet |= Q_CPUS;
560 } else if (strncmp(p, "mems", 4) == 0) {
561 __fort_quiet |= Q_MEMS;
562 } else if (strncmp(p, "msgs", 4) == 0) {
563 __fort_quiet |= Q_MSGS;
564 } else if (strncmp(p, "alls", 4) == 0) {
565 __fort_quiet |= (Q_CPUS | Q_MEMS | Q_MSGS);
566 } else if (strncmp(p, "cpu", 3) == 0) {
567 __fort_quiet |= Q_CPU;
568 } else if (strncmp(p, "mem", 3) == 0) {
569 __fort_quiet |= Q_MEM;
570 } else if (strncmp(p, "msg", 3) == 0) {
571 __fort_quiet |= Q_MSG;
572 } else if (strncmp(p, "all", 3) == 0) {
573 __fort_quiet |= (Q_CPU | Q_MEM | Q_MSG);
574 } else if (strncmp(p, "prof", 4) == 0) {
575 __fort_quiet |= Q_PROF;
576 } else if (strncmp(p, "trace", 5) == 0) {
577 __fort_quiet |= Q_TRAC;
578 } else if ((*p >= '0') && (*p <= '9')) {
579 __fort_quiet |= (int)strtol(p, (char **)0, 0);
580 } else {
581 getopt_abort("invalid format", "-stat");
582 }
583 p = strchr(p, ',');
584 if (p == NULL) {
585 break;
586 }
587 p++;
588 }
589}

591/* process (what used to be) generic command/environment options */
592
593static void
594__fort_initcom()
595{
596 const char *p;
597 char *q;
598 int n;
599
600 /* -test [<n>] */
601
602 p = __fort_getopt("-test");
603 if (p) {
604 __fort_test = (int)__fort_strtol(p, &q, 0);
605 if (q == p)
606 __fort_test = -1;
607 else if (*q != '\0')
608 getopt_abort("invalid numeric value", "-test");
609 }
610
611 /* -np <n> = number of processors */
612
613 p = __fort_getopt("-np");
614 if (p) {
615 n = (int)__fort_strtol(p, &q, 0);
616 if (q == p || *q != '\0' || n < 1)
617 getopt_abort("missing or invalid numeric value", "-np");
618 __fort_tcpus = n;
619 }
620
621 /* -g [<n>|all] = debug */
622
623 p = __fort_getopt("-g");
624 if (p) {
625 __fort_debug = 1;
626 __fort_debugn = (int)__fort_strtol(p, &q, 0);
627 if (q == p)
628 __fort_debugn = -1;
629 else if (*q != '\0' || __fort_debugn < 0 || __fort_debugn >= __fort_tcpus)
630 getopt_abort("invalid numeric value", "-g");
631 }
632
633 /* -stat ... */
634
635 __fort_istat();
636
637 /* -prof av[erage]|no[ne]|al[l] */
638
639 p = __fort_getopt("-prof");
640 if (p) {
641 int k = strlen(p);
642 if (k < 2)
643 k = 2;
644 if (strncmp(p, "average", k) == 0)
645 __fort_quiet |= Q_PROF_AVG;
646 else if (strncmp(p, "none", k) == 0)
647 __fort_quiet |= Q_PROF_NONE;
648 else if (strncmp(p, "all", k) != 0)
649 getopt_abort("invalid value", "-prof");
650 }
651}

667/* pass an arg to other processors, passing of a null is permitted */
668
669static char *
670__fort_passarg(int fr, int tol, int toh, char *val)
671{
672 int cpu;
673 int len;
674 char *p;
675
676 if (__fort_lcpu == fr) {
677 len = (val != NULL ? strlen(val) + 1 : 0);
678 for (cpu = tol; cpu < toh; cpu++) {
679 __fort_rsendl(cpu, &len, sizeof(len), 1, __UCHAR, 1);
680 if (len != 0) {
681 __fort_rsendl(cpu, val, len, 1, __UCHAR, 1);
682 }
683 }
684 p = val;
685 } else {
686 __fort_rrecvl(fr, &len, sizeof(len), 1, __UCHAR, 1);
687 if (len == 0) {
688 p = NULL;
689 } else {
690 p = __fort_malloc(len);
691 __fort_rrecvl(fr, p, len, 1, __UCHAR, 1);
692 }
693 }
694 return (p);
695}

697/* pass arglist */
698
699void
700__fort_passargs(int fr, int tol, int toh)
701{
702 char **toe;
703 char **fre;
704 int n;
705 int cpu;
706
707 if (__fort_lcpu == fr) {
708 n = 0;
709 while (env[n] != NULL) {
710 n++;
711 }
712 n++;
713 for (cpu = tol; cpu < toh; cpu++) {
714 __fort_rsendl(cpu, &n, sizeof(n), 1, __UCHAR, 1);
715 }
716 fre = env;
717 while (*fre != NULL) {
718 if ((strlen(*fre) > 6) && (strncmp("PGHPF_", *fre, 6) == 0)) {
719 __fort_passarg(fr, tol, toh, *fre);
720 }
721 fre++;
722 }
723 __fort_passarg(fr, tol, toh, NULL);
724 } else {
725 __fort_rrecvl(fr, &n, sizeof(n), 1, __UCHAR, 1);
726 env = (char **)__fort_malloc(n * sizeof(char *));
727 toe = env;
728 while (1) {
729 *toe = __fort_passarg(fr, tol, toh, NULL);
730 if (*toe == NULL) {
731 break;
732 }
733 toe++;
734 }
735 }
736
737 if (__fort_lcpu == fr) {
738 n = 0;
739 while (arg[n] != NULL) {
740 n++;
741 }
742 n++;
743 for (cpu = tol; cpu < toh; cpu++) {
744 __fort_rsendl(cpu, &n, sizeof(n), 1, __UCHAR, 1);
745 }
746 fre = arg;
747 while (*fre != NULL) {
748 __fort_passarg(fr, tol, toh, *fre);
749 fre++;
750 }
751 __fort_passarg(fr, tol, toh, NULL);
752 } else {
753 __fort_rrecvl(fr, &n, sizeof(n), 1, __UCHAR, 1);
754 arg = (char **)__fort_malloc(n * sizeof(char *));
755 toe = arg;
756 while (1) {
757 *toe = __fort_passarg(fr, tol, toh, NULL);
758 if (*toe == NULL) {
759 break;
760 }
761 toe++;
762 }
763 __fort_initopt(); /* init opt */
764 __fort_initcom(); /* init common arg/env */
765 }

780/* initialize everything */
781
782void ENTFTN(INIT, init)(__INT_T *n)
783{
784 __fort_setarg(); /* set __argv_save and __argc_save (maybe) */
785 if (!inited.consts) {
786 __fort_init_consts(); /* constants need initialization */
787 inited.consts = 1;
788 }
789 __fort_begpar(*n); /* TI-specific initialization */
790
791 /* smallest power of 2 >= number of processors */
792
793 for (__fort_np2 = 1; __fort_np2 < __fort_tcpus; __fort_np2 <<= 1)
794 ;
795
796 /* -V or -version */
797
798 if (__fort_lcpu == 0 && (__fort_getopt("-V") || __fort_getopt("-version")))
799 __fort_print_version();
800
801 __fort_zmem = __fort_getoptb("-zmem", 0);
802
803 __fort_entry_init(); /* start profiling/tracing/stats */
804
805 if (!inited.atexit) {
806 atexit(term); /* register term */
807 inited.atexit = 1;
808 }
809
810 ENTCOMN(NP, np)[0] = __fort_tcpus; /* for number_of_processors() */
811 ENTCOMN(ME, me)[0] = __fort_lcpu; /* for my_processor() */
812}
