41void
42sym_is_refd(int sptr)
43{
44 int a, dtype, stype;
45 ISZ_T size;
46 ISZ_T addr;
47 dtype = DTYPEG(sptr);
48 switch ((stype = STYPEG(sptr))) {
49 case ST_PLIST:
50 case ST_VAR:
51 case ST_ARRAY:
52 case ST_DESCRIPTOR:
53 case ST_STRUCT:
54 case ST_UNION:
55 if (REFG(sptr))
56 break;
57 if (gbl.internal > 1 && !INTERNALG(sptr))
58 break;
59 switch (SCG(sptr)) {
60 case SC_LOCAL:
61/*
62 * assign address to automatic variable: auto offsets are
63 * negative relative to the frame pointer. the current size of
64 * of the stack frame is saved as a positive value; the last
65 * offset assigned is the negative of the current frame size.
66 * The negative of the current frame size is aligned so that the
67 * variable ends on this boundary. The offset assigned is this
68 * value minus its size in bytes. The new size of the stack frame
69 * is the negative of the offset.
70 * ASSUMPTIONS:
71 * 1. the value frame pointer is an address whose alignment
72 * matches that of the scalar item having the most strict
73 * requirement.
74 * 2. there are not gaps between the address located by the
75 * frame pointer and the auto area (first offset is -1)
76 */
77#ifdef RECURBSS
78 if (!flg.recursive || DINITG(sptr) || SAVEG(sptr))
79#else
80 if (DINITG(sptr) || SAVEG(sptr))
81#endif
82 {
83 SCP(sptr, SC_STATIC);
84 goto static_shared;
85 }
86 if (stype == ST_PLIST)
87 size = PLLENG(sptr) * size_of(dtype);
88 else
89 size = size_of(dtype);
90 if (stype == ST_DESCRIPTOR && size > 4) {
91 a = alignment(DT_PTR);
92 size = ALIGN(size, a);
93 } else if ((flg.quad && size >= 16)) {
94 a = DT_QUADALIGN;
95 /* round-up size since sym's offset is 'aligned next' - size */
96 size = ALIGN(size, DT_QUADALIGN);
97 } else
98 a = alignment(dtype);
99#if DEBUG
100 assert(size == ALIGN(size, a), "sym_is_refd: sym unaligned", sptr, 2);
101#endif
102 addr = -gbl.locaddr;
103 addr = ALIGN_AUTO(addr, a) - size;
104 ADDRESSP(sptr, addr);
105 gbl.locaddr = -addr;
106 SYMLKP(sptr, gbl.locals);
107 gbl.locals = sptr;
108#if DEBUG
109 if (DBGBIT(5, 32)) {
110 fprintf(gbl.dbgfil,
111 "addr: %6" ISZ_PF "d size: %6" ISZ_PF "d %-32s (%s)\n", addr,
112 size, getprint(sptr), getprint((int)gbl.currsub));
113 }
114#endif
115 break;
116 case SC_STATIC:
117 static_shared:
118 if (DINITG(sptr))
119 addr = gbl.saddr;
120 else
121 addr = bss_addr;
122 if (stype == ST_PLIST)
123 size = PLLENG(sptr) * size_of(dtype);
124 else
125 size = size_of(dtype);
126 if (stype == ST_DESCRIPTOR) {
127 a = alignment(DT_PTR);
128 size = ALIGN(size, a);
129 } else if ((flg.quad && size >= 16)) {
130 a = DT_QUADALIGN;
131 } else
132 a = alignment(dtype);
133 addr = ALIGN(addr, a);
134 ADDRESSP(sptr, addr);
135 if (DINITG(sptr)) {
136 gbl.saddr = addr + size;
137 SYMLKP(sptr, gbl.statics);
138 gbl.statics = sptr;
139#if DEBUG
140 if (DBGBIT(5, 32)) {
141 fprintf(gbl.dbgfil,
142 "saddr: %6" ISZ_PF "d size: %6" ISZ_PF "d %-32s (%s)\n",
143 addr, size, getprint(sptr), getprint((int)gbl.currsub));
144 }
145#endif
146 } else {
147 bss_addr = addr + size;
148 if (!bss_addr && no_data_components(dtype)) {
149 /* FS#17746 - need to assign a default size to bss_addr
150 * for zero size derived types to make sure we declare
151 * the BSS symbol in the assembly file.
152 */
153 bss_addr = size_of(DT_INT4);
154 }
155#if DEBUG
156 if (DBGBIT(5, 32)) {
157 fprintf(gbl.dbgfil,
158 "baddr: %6" ISZ_PF "d size: %6" ISZ_PF "d %-32s (%s)\n",
159 addr, size, getprint(sptr), getprint((int)gbl.currsub));
160 }
161#endif
162 }
163 break;
164 case SC_CMBLK:
165 case SC_DUMMY:
166 break;
167 case SC_EXTERN:
168 break;
169 case SC_PRIVATE:
170 /* don't assign address to privates & don't set the REF bit */
171 return;
172 case SC_NONE:
173 default:
174 interr("sym_is_refd: bad sc\n", SCG(sptr), 3);
175 }
176 REFP(sptr, 1);
177 break;
178
179 case ST_PROC:
180 if (REFG(sptr) == 0 && SCG(sptr) == SC_EXTERN) {
181 SYMLKP(sptr, gbl.externs);
182 gbl.externs = sptr;
183 REFP(sptr, 1);
184 }
185 break;
186 case ST_CONST:
187 if (SYMLKG(sptr) == 0) {
188 SYMLKP(sptr, gbl.consts);
189 gbl.consts = sptr;
190 if (DTYPEG(sptr) == DT_ADDR && CONVAL1G(sptr))
191 sym_is_refd((int)CONVAL1G(sptr));
192 }
193 break;
194
195 case ST_ENTRY: /* (found on entry ili only) */
196 case ST_LABEL:
197 break;
198
199 default:
200 interr("sym_is_refd:bad sty", sptr, 2);
201 }
202
203}

205/** \brief The equivalence processor assigns positive offsets to the local
206 variables
207 which appear in equivalence statements. Target addresses must be
208 assigned using the offsets provided by the equivalence processor.
209 \param loc_list list of local symbols linked by SYMLK
210 \param loc_addr total size of the equivalenced locals
211 */
212void
213fix_equiv_locals(int loc_list, ISZ_T loc_addr)
214{
215 int sym;
216 int maxa;
217
218 if (loc_list != NOSYM) {
219 /*
220 * align beginning of this group just in case. Implementation note:
221 * could loop through the syms in the group to determine the maximum
222 * alignment required; this would involve looking at the size if -quad
223 * or just the alignment of the symbol.
224 */
225 if (flg.quad)
226 maxa = DT_QUADALIGN;
227 else
228 maxa = alignment(DT_DBLE);
229 gbl.locaddr = ALIGN(gbl.locaddr + loc_addr, maxa);
230 do {
231 /* NOTE: REF flag of sym set during equivalence processing */
232 sym = loc_list;
233 loc_list = SYMLKG(loc_list);
234 ADDRESSP(sym, -gbl.locaddr + ADDRESSG(sym));
235 SCP(sym, SC_LOCAL);
236 SYMLKP(sym, gbl.locals);
237 gbl.locals = sym;
238 } while (loc_list != NOSYM);
239 }
240
241}

243/** \brief Similiar to fix_equiv_locals except that these local variables were
244 saved and/or dinit'd. for these variables, switch the storage class to
245 SC_STATIC.
246 \param loc_list list of local symbols linked by SYMLK
247 \param loc_addr total size of the equivalenced locals
248 \param dinitflg variables were dinit'd
249
250 The equivalence processor assigns positive offsets to the local variables
251 which appear in equivalence statements. Target addresses must be
252 assigned using the offsets provided by the equivalence processor.
253 */
254void
255fix_equiv_statics(int loc_list, ISZ_T loc_addr, LOGICAL dinitflg)
256{
257 int sym;
258 int maxa;
259 ISZ_T addr;
260
261#if DEBUG
262 assert(loc_list != NOSYM, "fix_equiv_statics: bad loc_list", 0, 3);
263#endif
264 /*
265 * align beginning of this group just in case. Implementation note:
266 * could loop through the syms in the group to determine the maximum
267 * alignment required; this would involve looking at the size if -quad
268 * or just the alignment of the symbol.
269 */
270 if (flg.quad)
271 maxa = DT_QUADALIGN;
272 else
273 maxa = alignment(DT_DBLE);
274 if (dinitflg) {
275 addr = gbl.saddr;
276 addr = ALIGN(addr, maxa);
277 do {
278 /* NOTE: REF flag of sym set during equivalence processing */
279 sym = loc_list;
280 loc_list = SYMLKG(loc_list);
281 ADDRESSP(sym, addr + ADDRESSG(sym));
282 SCP(sym, SC_STATIC);
283 SYMLKP(sym, gbl.statics);
284 gbl.statics = sym;
285 DINITP(sym, 1); /* ensure getsname thinks it's in STATIC */
286 } while (loc_list != NOSYM);
287 gbl.saddr = addr += loc_addr;
288 } else {
289 addr = bss_addr;
290 addr = ALIGN(addr, maxa);
291 do {
292 /* NOTE: REF flag of sym set during equivalence processing */
293 sym = loc_list;
294 loc_list = SYMLKG(loc_list);
295 ADDRESSP(sym, addr + ADDRESSG(sym));
296 SCP(sym, SC_STATIC);
297 } while (loc_list != NOSYM);
298 bss_addr = addr += loc_addr;
299 }
300
301}

321ISZ_T
322pad_cmn_mem(int mem, ISZ_T msz, int *p_aln_n)
323{
324 int aln_n;
325
326 aln_n = *p_aln_n;
327#ifdef PDALN_IS_DEFAULT
328 if (!XBIT(57, 0x1000000) && !PDALN_IS_DEFAULT(mem) && !PDALN_IS_DEFAULT(CMBLKG(mem)) &&
329 msz > ALN_MINSZ) {
330 if (aln_n == 0)
331 aln_n = 1;
332 msz += ALN_UNIT * aln_n;
333 if (aln_n <= ALN_THRESH)
334 aln_n++;
335 else
336 aln_n = 1;
337 *p_aln_n = aln_n;
338 }
339#endif
340 return msz;
341}
