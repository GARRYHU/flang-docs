122/** \brief Check line number, findex after each various stages to make sure
123 * they don't contain 0 as a line number, so that ccff_info don't get
124 * linenumber as 0
125 */
126static void
127check_lineno(const char *phase)
128{
129 int block;
130 int ilt;
131 int first = 1;
132
133 if (DBGBIT(73, 1)) {
134 for (block = BIH_NEXT(0); block; block = BIH_NEXT(block)) {
135 /* For debugged code, the first block line number is 0 *
136 therefor we ignore it */
137 if (flg.debug || XBIT(120, 0x1000) || (XBIT(123, 0x400) && first)) {
138 first = 0;
139 if (BIH_LINENO(block) == 0)
140 continue;
141 }
142 if (BIH_LINENO(block) == 0) {
143 printf("ERROR: BIH_LINENO() is 0 after :%s\n", phase);
144 }
145 if (BIH_FINDEX(block) == 0) {
146 printf("ERROR: BIH_FINDEX() is 0 after :%s\n", phase);
147 }
148 for (ilt = BIH_ILTFIRST(block); ilt; ilt = ILT_NEXT(ilt)) {
149 if (ILT_LINENO(ilt) == 0) {
150 printf("ERROR: ILT_LINENO() is 0 after :%s\n", phase);
151 }
152 if (ILT_FINDEX(ilt) == 0) {
153 printf("ERROR: ILT_FINDEX() is 0 after :%s\n", phase);
154 }
155 }
156 }
157 }
158}

161/** \brief Process one input file
162 * \param argv0 path to initalize accelerator compilation
163 * \param need_cuda_constructor indicates that need to add a CUDA construction,
164 * this parameter is also written to
165 * \return false to indicated end of processing, true otherwise
166 */
167static bool
168process_input(char *argv0, bool *need_cuda_constructor)
169{
170 static int accsev = 0;
171 bool have_data_constructor = false;
172 bool is_constructor = false;
173 bool is_omp_recompile = false;
174 llvm_restart:
175 if (gbl.maxsev > accsev)
176 accsev = gbl.maxsev;
177
178#if DEBUG & sun
179 if (DBGBIT(7, 4))
180 if (malloc_verify() != 1)
181 interr("main: malloc_verify failsA", errno, ERR_Fatal);
182#endif
183 reinit();
184
185
186#if DEBUG & sun
187 if (DBGBIT(7, 4))
188 if (malloc_verify() != 1)
189 interr("main: malloc_verify failsB", errno, ERR_Fatal);
190#endif
191 xtimes[0] += get_rutime();
192 /* don't increment if it is outlined function because it
193 * uses STATICS/BSS from host routine.
194 */
195 if (flg.smp && IS_PARFILE) {
196 ll_set_outlined_currsub(is_omp_recompile);
197 is_omp_recompile = false;
198 }
199 gbl.func_count++;
200 gbl.multi_func_count = gbl.func_count;
201
202 TR("F90 ILM INPUT begins\n")
203 if (!IS_PARFILE)
204 {
205 upper(0);
206 if (gbl.eof_flag)
207 return false;
208 upper_assign_addresses();
209 }
210
211 is_constructor = gbl.cuda_constructor;
212 xtimes[1] += get_rutime();
213 DUMP("upper");
214
215 if (gbl.cuda_constructor) {
216 have_data_constructor = false;
217/* Generate the CUDA constructor */
218 gbl.cuda_constructor = 0;
219 return true;
220 }
221 if (DBGBIT(5, 1))
222 symdmp(gbl.dbgfil, DBGBIT(5, 8));
223
224 if (gbl.maxsev < 3 && (flg.object || flg.code) && !DBGBIT(2, 4)) {
225
226 if (flg.debug)
227 process_global_lifetime_debug();
228
229 gbl.nofperror = true;
230 if (gbl.rutype == RU_BDATA) {
231 } else {
232#ifdef OMP_OFFLOAD_LLVM
233 if (flg.omptarget) {
234 gbl.ompaccel_intarget = ompaccel_tinfo_has(gbl.currsub);
235 ompaccel_initsyms();
236 }
237#endif
238#ifdef OMP_OFFLOAD_LLVM
239 if (flg.omptarget) {
240 init_test();
241 ompaccel_initsyms();
242 ompaccel_create_reduction_wrappers();
243 }
244#endif
245 if (gbl.cuda_constructor) {
246 } else {
247 /*
248 * direct_rou_load() checks for assigned goto's and variable
249 * format specifiers.
250 *
251 * the optimizer doesn't handle assigned goto's correctly.
252 * (Doesn't know where to put loop exit code if you assign
253 * goto out of loop).
254 * the optimizer doesn't handle the control flow correctly
255 * for the variable-format functions.
256 */
257 direct_rou_load(gbl.currsub);
258 cuda_emu_start();
259 if (XBIT(14, 16) && !XBIT(89, 0x40)) {
260 }
261
262 if (XBIT(14, 0x20000) || !XBIT(14, 0x10000)) {
263 {
264 if (rdgilms(1) == 0) {
265 if (flg.smp) {
266 if (ll_reset_parfile()) {
267 gbl.eof_flag = false;
268 goto llvm_restart;
269 } else if (!IS_PARFILE) {
270 goto llvm_restart;
271 }
272 }
273 return false;
274 }
275
276 }
277 if (XBIT(70, 0x18080)) {
278 AssignAddresses(); /* exp_rte.c */
279 }
280 }
281 TR("F90 EXPANDER begins\n");
282
283 expand(); /* expand ILM's into ILI */
284 DUMP("expand");
285#if DEBUG
286 check_lineno("expand");
287#endif
288
289 }
290
291 {
292
293/* move call to intensity earlier to prevent too much loop restructuring */
294
295 if (XBIT_USE_SCOPE_LABELS)
296 remove_scope_labels();
297
298 if (flg.opt && !XBIT(6, 0x40000)) {
299 /* remove useless split blocks */
300 unsplit();
301#if DEBUG
302 check_lineno("unsplit");
303#endif
304 DUMP("unsplit");
305 }
306 rm_smove();
307 DUMP("rmsmove");
308
309#if defined(OMP_OFFLOAD_LLVM) || defined(OMP_OFFLOAD_PGI)
310 if(flg.omptarget && DBGBIT(61, 1))
311 dumpomptargets(); /* print all openmp target regions */
312#endif
313#if defined(OMP_OFFLOAD_LLVM)
314 if (flg.omptarget && ompaccel_tinfo_has(gbl.currsub))
315 gbl.ompaccel_isdevice = true;
316#endif
317
318 TR("F90 SCHEDULER begins\n");
319 DUMP("before-schedule");
320 schedule();
321 xtimes[5] += get_rutime();
322 DUMP("schedule");
323 } /* CUDAG(GBL_CURRFUNC) & CUDA_HOST */
324 }
325 TR("F90 ASSEMBLER begins\n");
326 assemble();
327 xtimes[6] += get_rutime();
328 upper_save_syminfo();
329 }
330 if (DBGBIT(5, 4))
331 symdmp(gbl.dbgfil, DBGBIT(5, 8));
332 if (DBGBIT(5, 16))
333 dmp_dtype();
334 if (gbl.rutype != RU_BDATA) {
335 cuda_emu_end();
336 /* TDB: make it look better!*/
337 if (!flg.smp)
338 direct_rou_end();
339 else if (!ll_has_more_outlined())
340 direct_rou_end();
341 else if (!ALLOW_NODEPCHK_SIMD)
342 direct_rou_end();
343 }
344 if (!flg.smp || !ll_has_more_outlined())
345 {
346 ccff_close_unit();
347 }
348
349 if (flg.inliner && !XBIT(117, 0x10000)
350 && !IS_PARFILE
351 ) {
352 }
353
354 if (flg.xref) {
355 xref(); /* write cross reference map */
356 xtimes[7] += get_rutime();
357 }
358 (void)summary(false, 0);
359 cg_llvm_fnend();
360 if (llProcessNextTmpfile()) {
361 if (ll_reset_parfile())
362 return true;
363 }
364#ifndef NO_OMP_OFFLOAD
365#endif
366 return true;
367}

369/** \brief Fortran backend main entry.
370 */
371int
372main(int argc, char *argv[])
373{
374 bool findex = false;
375 bool need_constructor = false;
376 int ilm_units_seen = 0;
377
378 get_rutime();
379 init(argc, argv);
380
381 saveoptflag = flg.opt;
382 savevectflag = flg.vect;
383 savex8flag = flg.x[8];
384 saverecursive = flg.recursive;
385 gbl.numversions = 1;
386
387#if DEBUG & sun
388 if (DBGBIT(7, 4))
389 malloc_debug(2);
390#endif
391 if (XBIT(14, 0x20000) || !XBIT(14, 0x10000)) {
392 init_global_ilm_mode();
393 }
394//TODO Need to add NO_OMP_OFFLOAD, otherwise compilation of main_ex fails.
395#ifndef NO_OMP_OFFLOAD
396#if defined(OMP_OFFLOAD_LLVM) || defined(OMP_OFFLOAD_PGI)
397 if (flg.omptarget) {
398 ompaccel_init();
399 }
400#endif
401#endif
402
403 if (STB_UPPER()) {
404 stb_upper_init();
405 gbl.findex = addfile(gbl.file_name, NULL, 0, 0, 0, 1, 0);
406 process_stb_file();
407 findex = true;
408 }
409
410 upper_init();
411 if (!findex)
412 gbl.findex = addfile(gbl.file_name, NULL, 0, 0, 0, 1, 0);
413#ifdef OMP_OFFLOAD_LLVM
414 if (flg.omptarget) {
415 init_test();
416 ompaccel_create_globalctor();
417 gbl.func_count--;
418 gbl.multi_func_count = gbl.func_count;
419 }
420#endif
421 do { /* loop once for each user program unit */
422
423 if (!process_input(argv[0], &need_constructor))
424 break;
425
426 // We cannot reuse gbl.func_count here since gbl.func_count must be
427 // incremented before calling process_input().
428 ilm_units_seen++;
429
430 } while (!gbl.eof_flag);
431
432 if (ilm_units_seen == 0)
433 errwarn((error_code_t)6);
434
435 // Finalize IR generation only if the output file has been initialized
436 // by cg_llvm_init or assemble.
437 if (FTN_HAS_INIT())
438 cg_llvm_end();
439
440 if (flg.smp) {
441 ll_unlink_parfiles();
442 }
443
444 finish(); /* finish does not return */
445 return 0; /* never reached */
446}

448static void
449reptime()
450{
451 char buf[80];
452 int i;
453 INT total;
454 int prct;
455 int tmp;
456
457 total = 0;
458 for (i = 0; i < _N_WHO; i++)
459 total += xtimes[i];
460
461 if (!DBGBIT(0, 8) || DBGBIT(14, 3))
462 goto xbitcheck;
463
464 if (flg.code || flg.list || flg.xref) {
465 list_line("");
466 list_line(" Timing stats:");
467 } else if (gbl.dbgfil)
468 fprintf(gbl.dbgfil, " Timing stats:\n");
469 for (i = 0; i < _N_WHO; i++) {
470 if (xtimes[i]) {
471 tmp = 100 * xtimes[i];
472 prct = tmp / total;
473 sprintf(buf, " %-10.10s %15d millisecs %5d%%", who[i], xtimes[i],
474 prct);
475 if (flg.code || flg.list || flg.xref)
476 list_line(buf);
477 else if (gbl.dbgfil)
478 fprintf(gbl.dbgfil, "%s\n", buf);
479 }
480 }
481
482 sprintf(buf, " Total time %15d millisecs", total);
483 if (flg.code || flg.list || flg.xref) {
484 list_line(buf);
485 } else if (gbl.dbgfil)
486 fprintf(gbl.dbgfil, "%s\n", buf);
487
488xbitcheck:
489 if (!XBIT(0, 1))
490 return;
491 fprintf(stderr, " Timing stats:\n");
492
493 for (i = 0; i < _N_WHO; i++) {
494 if (xtimes[i]) {
495 tmp = 100 * xtimes[i];
496 prct = tmp / total;
497 sprintf(buf, " %-10.10s %15d millisecs %5d%%", who[i], xtimes[i],
498 prct);
499 fprintf(stderr, "%s\n", buf);
500 }
501 }
502 sprintf(buf, " Total time %15d millisecs", total);
503 fprintf(stderr, "%s\n", buf);
504}

526/** \brief Initialize Fortran backend at the begining of compilation.
527 */
528static void
529init(int argc, char *argv[])
530{
531 char *sourcefile = NULL;
532 const char *stboutfile = NULL;
533 const char *listfile = NULL;
534 const char *tempfile = NULL;
535 const char *asmfile = NULL;
536 bool dbgflg;
537 bool errflg;
538 FILE *fd;
539 const char *file_suffix;
540 time_t now;
541
542 file_suffix = FTNFILE; /* default suffix for source files */
543 /*
544 * initialize error and symbol table modules in case error messages are
545 * issued:
546 */
547
548 /* Fill xflags array with zeros */
549 memset(flg.x, 0, sizeof(flg.x));
550
551 errini();
552 if (argc <= 1)
553 errfatal((error_code_t)1);
554
555 gbl.curr_file = NULL;
556 sym_init();
557
558 /* fill in date and time */
559 time(&now);
560 strftime(gbl.datetime, sizeof gbl.datetime, "%m/%d/%Y %H:%M:%S",
561 localtime(&now));
562
563 dbgflg = false;
564 errflg = false;
565
566 bool arg_reentrant; /* Argument to enable generating reentrant code */
567
568 /* Create a datastructure of various dump actions and their names */
569 action_map_t *dump_map; /* Deallocated after arguments are parsed */
570 create_action_map(&dump_map);
571 add_action(dump_map, "bih", dbih);
572 add_action(dump_map, "block", dbih);
573 add_action(dump_map, "blocks", dbih);
574 add_action(dump_map, "blockonly", dbihonly);
575 add_action(dump_map, "blocksonly", dbihonly);
576 add_action(dump_map, "diff", dumpdiff);
577 add_action(dump_map, "dtype", dumpdtypes);
578 add_action(dump_map, "dvl", dumpdvls);
579 add_action(dump_map, "ilts", dumpilts);
580 add_action(dump_map, "nme", dumpnmes);
581 add_action(dump_map, "pblock", printblocks);
582 add_action(dump_map, "pblocks", printblocks);
583 add_action(dump_map, "printblocks", printblocks);
584 add_action(dump_map, "stack", stackcheck);
585 add_action(dump_map, "stackvars", stackvars);
586 add_action(dump_map, "sym", dump_symbols);
587 add_action(dump_map, "syms", dump_symbols);
588 add_action(dump_map, "symtab", dump_symbols);
589 add_action(dump_map, "area", report_area);
590 add_action(dump_map, "gbl", dgbl);
591 add_action(dump_map, "flg", dflg);
592 /* Initialize the map that will be used by dump handler later */
593 create_action_map(&phase_dump_map);
594
595 /* Need to up optimization level if we modify unroller count */
596 int old_unroller_cnt = flg.x[9];
597 /* Target architecture string */
598 const char *tp;
599 /* OpenMP target triple architecture string */
600 const char *omptp = NULL;
601 const char *ompfile = NULL;
602 /* Vectorizer settings */
603 int vect_val;
604
605 /* Argument parser */
606 arg_parser_t *arg_parser;
607
608 /* Create argument parser, force errors on unknown flags (last argument is
609 * true) */
610 create_arg_parser(&arg_parser, true);
611
612 /* Input ILM file name */
613 register_filename_arg(arg_parser, &(gbl.src_file));
614 /* Fortran source file */
615 register_string_arg(arg_parser, "fn", &(gbl.file_name), NULL);
616 /* Other files to input or output */
617 register_string_arg(arg_parser, "stbfile", &stboutfile, NULL);
618 register_combined_bool_string_arg(arg_parser, "asm", (bool *)&(flg.asmcode),
619 &asmfile);
620
621 /* Register version arguments */
622 register_string_arg(arg_parser, "vh", &(version.host), "");
623
624 /* x flags */
625 register_xflag_arg(arg_parser, "x", flg.x);
626 register_yflag_arg(arg_parser, "y", flg.x);
627 /* Debug flags */
628 register_qflag_arg(arg_parser, "q", flg.dbg,
629 (sizeof(flg.dbg) / sizeof(flg.dbg[0])));
630 register_action_map_arg(arg_parser, "qq", phase_dump_map, dump_map);
631
632 /* Other flags */
633 register_integer_arg(arg_parser, "opt", &flg.opt, 1);
634 register_integer_arg(arg_parser, "ieee", &flg.ieee, 0);
635 register_inform_level_arg(arg_parser, "inform", (inform_level_t *)&flg.inform,
636 LV_Inform);
637 register_integer_arg(arg_parser, "endian", &flg.endian, 0);
638 register_boolean_arg(arg_parser, "mp", &flg.smp, false);
639 register_string_arg(arg_parser, "fopenmp-targets", &omptp, NULL);
640 register_string_arg(arg_parser, "fopenmp-targets-asm", &ompfile, NULL);
641 register_boolean_arg(arg_parser, "reentrant", &arg_reentrant, false);
642 register_integer_arg(arg_parser, "terse", &flg.terse, 1);
643 register_boolean_arg(arg_parser, "quad", &flg.quad, false);
644#ifdef TARGET_SUPPORTS_QUADFP
645 register_boolean_arg(arg_parser, "qp", &flg.qp, true);
646#endif
647 register_boolean_arg(arg_parser, "save", &flg.save, false);
648 register_string_arg(arg_parser, "tp", &tp, NULL);
649 register_integer_arg(arg_parser, "astype", &flg.astype, 0);
650 register_boolean_arg(arg_parser, "recursive", &flg.recursive, false);
651 register_integer_arg(arg_parser, "vect", &vect_val, 0);
652 register_string_arg(arg_parser, "cmdline", &flg.cmdline, NULL);
653 register_string_arg(arg_parser, "target_features", &flg.target_features, NULL);
654 register_integer_arg(arg_parser, "vscale_range_min", &flg.vscale_range_min, 0);
655 register_integer_arg(arg_parser, "vscale_range_max", &flg.vscale_range_max, 0);
656 register_boolean_arg(arg_parser, "debug", &flg.debug, false);
657
658 flg.linker_directives = (char **)getitem(8, argc * sizeof(char *));
659 register_string_list_arg(arg_parser, "linker", flg.linker_directives);
660 register_string_arg(arg_parser, "target", &(flg.llvm_target_triple), NULL);
661
662 /* Run argument parser */
663 parse_arguments(arg_parser, argc, argv);
664
665 /* Process debug output settings */
666 if (was_value_set(arg_parser, &(flg.dbg)) ||
667 was_value_set(arg_parser, phase_dump_map)) {
668#if DEBUG
669 dodebug = 1;
670#endif
671 if (flg.dbg[0] & 1) {
672 gbl.dbgfil = stderr;
673 } else {
674 const char *s, *ss, *t;
675 s = NULL;
676 t = gbl.file_name;
677 for (ss = gbl.file_name; *ss; ++ss) {
678 if (*ss == '.')
679 s = ss;
680 if ((*ss == '/'
681#ifdef _WIN64
682 || *ss == '\\'
683#endif
684 ) &&
685 *(ss + 1))
686 t = ss + 1;
687 }
688 if (s == NULL)
689 s = ".f90";
690 tempfile = mkfname(t, s, ".qdbg");
691 if ((gbl.dbgfil = fopen(tempfile, "w")) == NULL)
692 errfatal((error_code_t)5);
693 }
694 }
695
696 /* Enable reentrant code */
697 if (was_value_set(arg_parser, &arg_reentrant)) {
698 if (arg_reentrant) {
699 flg.x[7] |= 0x2; /* inhibit terminal func optz. */
700 flg.recursive = true; /* no static locals */
701 } else {
702 flg.x[7] &= ~(0x2);
703 flg.recursive = false;
704 }
705 }
706
707 /* Free memory */
708 destroy_arg_parser(&arg_parser);
709 destroy_action_map(&dump_map);
710
711 /* Go ahead and up optimization level if user modified maximum count for
712 * unroller */
713 if (flg.x[9] != old_unroller_cnt && flg.opt < 2)
714 flg.opt = 2;
715
716 /* Check optimization level */
717 if (flg.opt > 4) {

757 * flg.x[15] |= 1; '<x> / <const>' -> '<x> * (1 / <const>)'
758 * flg.x[15] |= 2; 'x/y/z' -> 'x/(y*z)'

783 }

786 finish();

788 /* open sourcefile */
789 if ((gbl.srcfil = fopen(gbl.src_file, "r")) == NULL) {

796 if (*s == '.')
797 file_suffix = s;
798 }
799 }

802
803 if (gbl.file_name == NULL)
804 gbl.file_name = gbl.src_file;
805 gbl.curr_file = gbl.file_name;
806 gbl.module = mkfname(sourcefile, file_suffix, "");
807 if (flg.smp) {
808 ll_open_parfiles();
809 }

811 /* process ilm and object file: */
812
813 gbl.ilmfil = gbl.objfil = NULL;
814 if (!flg.es && (flg.object || flg.code)) {
815 /* create temporary file for ilms */
816 if ((gbl.ilmfil = tmpfile()) == NULL)
817 errfatal((error_code_t)5);
818 }

820 if (flg.code || flg.list || flg.xref) {
821 if (listfile == NULL) {
822 /* make listing filename from sourcefile name */
823 listfile = mkfname(sourcefile, file_suffix, LISTFILE);
824 }
825 if ((fd = fopen(listfile, "w")) == NULL)
826 errfatal((error_code_t)3);
827 list_init(fd);
828 }
829 /* process assembly output file */
830 if (flg.asmcode) {
831 if (asmfile == NULL) {
832 /* make assembly filename from sourcefile name */
833 asmfile = mkfname(sourcefile, file_suffix, ASMFILE);
834 }
835 if ((gbl.asmfil = fopen(asmfile, "w")) == NULL)
836 errfatal((error_code_t)9);
837 } else /* do this for compilers which write asm code to stdout */
838 gbl.asmfil = stdout;
839
840 if (stboutfile) {
841 if ((gbl.stbfil = fopen(stboutfile, "r")) == NULL)
842 error((error_code_t)2, ERR_Fatal, 0, stboutfile, "");
843 } else {
844 gbl.stbfil = NULL;
845 }
846
847#if DEBUG
848 assert(flg.es == 0, "init:flg.esA", 0, ERR_unused);
849#endif
850 assemble_init(argc, argv, flg.cmdline);

852 gbl.func_count = 0;
853 gbl.multi_func_count = gbl.func_count;
854 direct_init();
855
856 if (XBIT(125, 0x8))
857 gbl.ftn_true = 1;
858 else
859 gbl.ftn_true = -1;
860
861 if (ccff_filename) {
862 ccff_open(ccff_filename, gbl.file_name ? gbl.file_name : gbl.src_file);
863 ccff_build(flg.cmdline, "F90");
864 }

866 free(sourcefile);
867}
