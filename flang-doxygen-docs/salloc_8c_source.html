22typedef char *PTR;
23#undef PTRSZ
24#undef ALIGN
25#define PTRSZ sizeof(PTR)
26#define ALIGN(o) (((o) + (PTRSZ - 1)) & (~(PTRSZ - 1)))

28static char *areap[ANUM] = {NULL, NULL, NULL, NULL, NULL, NULL, NULL,
29 NULL, NULL, NULL, NULL, NULL, NULL, NULL,
30 NULL, NULL, NULL, NULL, NULL, NULL, NULL};
31static int avail[ANUM];

33/**
34 \param area is an area
35 \param size is the size in bytes of item to be allocated.
36 */
37char *
38getitem(int area, int size)
39{
40 char *p;
41
42 assert(area >= 0 && area < ANUM, "getitem: bad area", area, ERR_Fatal);
43 size = ALIGN(size); /* round up to multiple of PTRSZ */
44
45 if (areap[area] == NULL) {
46 int sz = SIZE;
47 if (size + PTRSZ > SIZE)
48 sz = size + PTRSZ;
49 NEW(p, char, sz);
50 areap[area] = p;
51 if (p == NULL)
52 interr("getitem: no mem avail", area, ERR_Fatal);
53 *((PTR *)p) = NULL;
54 avail[area] = PTRSZ;
55 } else if (avail[area] + size > SIZE) {
56 int sz = SIZE;
57 if (size + PTRSZ > SIZE)
58 sz = size + PTRSZ;
59 NEW(p, char, sz);
60 if (p == NULL)
61 interr("getitem: no mem avail", area, ERR_Fatal);
62 *((PTR *)p) = areap[area];
63 areap[area] = p;
64 avail[area] = PTRSZ;
65 }
66 p = areap[area] + avail[area];
67 avail[area] += size;
68#if DEBUG
69 if (DBGBIT(0, 0x20000)) {
70 char *q, cc;
71 int s;
72 /* fill with junk */
73 cc = 0xa6;
74 for (s = size, q = p; s; --s, ++q) {
75 *q = cc;
76 cc = (cc << 1) | (cc >> 7);
77 }
78 }
79#endif
80 return p;
81}

83void
84freearea(int area)
85{
86 char *p, *q;
87
88 assert(area >= 0 && area < ANUM, "freearea: bad area", area, ERR_Fatal);
89 for (p = areap[area]; p != NULL; p = q) {
90 q = *((PTR *)p); /* get next before free!!! */
91 FREE(p);
92 }
93 areap[area] = NULL;
94}

119/*
120 * Functions to allow referring to pointers returned by getitem as ints
121 * for the purpose of storing pointers in the int fields of the symbol
122 * table and other structures.
123 */
124static struct {
125 void **base;
126 int size;
127 int avail;
128} tbl = {0, 0, 0};
129
130/**
131 \brief stores the pointer in the pointer table and returns its
132 index.
133 */
134int
135put_getitem_p(void *p)
136{
137 int r;
138
139 /* NULL <==> 0 */
140 if (p == NULL)
141 return 0;
142 if (tbl.size == 0) {
143 tbl.size = 100;
144 NEW(tbl.base, void *, tbl.size);
145 r = 1;
146 tbl.avail = 2;
147 } else {
148 r = tbl.avail++;
149 NEED(tbl.avail, tbl.base, void *, tbl.size, tbl.size + 100);
150 }
151 tbl.base[r] = p;
152 return r;
153}
154
155/**
156 \brief returns the pointer stored at index i.
157 */
158void *
159get_getitem_p(int i)
160{
161 /* 0 <==> NULL */
162 if (i == 0)
163 return NULL;
164#if DEBUG
165 assert(tbl.size, "get_getitem_p: null tbl", i, ERR_unused);
166 assert(i > 0 && i < tbl.avail, "get_getitem_p: i out of range", i, ERR_unused);
167#endif
168 return tbl.base[i];
169}

171/**
172 \brief frees the table of pointers. NOTE: it's safer to just call
173 this function at the end of compilation to allow a mix of pointers
174 which are only live during a phase and those which are live during
175 the entire compile such as the pointers locating items in area 8.
176 */
177void
178free_getitem_p(void)
179{
180 if (tbl.size) {
181 free(tbl.base);
182 tbl.size = 0;
183 }
