74{

154void
155int128_unsigned_multiply(int128_t *high, int128_t *low,
156 const int128_t *x, const int128_t *y)
157{
158 uint128_t xh = *x >> 64, xl = (uint64_t) *x;
159 uint128_t yh = *y >> 64, yl = (uint64_t) *y;
160 *low = *x * *y;
161 *high = ((((xl * yl) >> 64) + (xh * yl) + (yh * xl)) >> 64) + xh * yh;
162}

202int
203int128_unsigned_compare(const int128_t *x, const int128_t *y)
204{
205 int j;
206 for (j = 3; j >= 0; --j) {
207 if (x->part[j] < y->part[j])
208 return -1;
209 if (x->part[j] > y->part[j])
210 return 1;
211 }
212 return 0;

215int
216int128_signed_compare(const int128_t *x, const int128_t *y)
217{
218 if (x->part[3] & MSB32) {
219 if (!(y->part[3] & MSB32))
220 return -1;
221 return -int128_unsigned_compare(x, y);
222 } else {
223 if (y->part[3] & MSB32)
224 return 1;
225 return int128_unsigned_compare(x, y);
226 }

229int
230int128_count_leading_zeros(const int128_t *x)
231{
232 int j, k;
233 for (j = 0; j < 4; ++j) {
234 uint32_t w = x->part[3 - j];
235 if (w != 0) {
236 for (k = 0; k < 32; ++k) {
237 if (((w << k) & MSB32) != 0)
238 return 32 * j + k;
239 }
240 assert(!"int128_count_leading_zeros: can't happen");
241 }
242 }
243 return 128;
244}

289void
290int128_shift_left(int128_t *result, const int128_t *x, int count)
291{
292 int j, off, sh;
293 assert(count >= 0);
294 off = count >> 5;
295 sh = count & 31;
296 if (sh == 0) {
297 /* Be portable: don't assume that a shift by (32-0) below yields 0. */
298 for (j = 3; j - off >= 0; --j) {
299 result->part[j] = x->part[j - off];
300 }
301 } else {
302 for (j = 3; j - off - 1 >= 0; --j) {
303 result->part[j] = (x->part[j - off] << sh) |
304 (x->part[j - off - 1] >> (32 - sh));
305 }
306 if (j - off >= 0) {
307 result->part[j] = x->part[j - off] << sh;
308 --j;
309 }
310 }
311 for (; j >= 0; --j) {
312 result->part[j] = 0;
313 }
314}

316void
317int128_shift_right_logical(int128_t *result, const int128_t *x, int count)
318{
319 int j, off, sh;
320 assert(count >= 0);
321 off = count >> 5;
322 sh = count & 31;
323 if (sh == 0) {
324 /* Be portable: don't assume that a shift by (32-0) below yields 0. */
325 for (j = 0; j + off < 4; ++j) {
326 result->part[j] = x->part[j + off];
327 }
328 } else {
329 for (j = 0; j + off + 1 < 4; ++j) {
330 result->part[j] = (x->part[j + off] >> sh) |
331 (x->part[j + off + 1] << (32 - sh));
332 }
333 if (j + off < 4) {
334 result->part[j] = x->part[j + off] >> sh;
335 ++j;
336 }
337 }
338 for (; j < 4; ++j) {
339 result->part[j] = 0;
340 }

344int128_unsigned_add(int128_t *result, const int128_t *x, const int128_t *y)
345{
346 uint64_t carry = 0;
347 int j;
348 for (j = 0; j < 4; ++j) {
349 carry += x->part[j];
350 result->part[j] = carry += y->part[j];
351 carry >>= 32;
352 }
353 return carry != 0;

371void
372int128_unsigned_multiply(int128_t *high, int128_t *low,
373 const int128_t *x, const int128_t *y)
374{
375 int j, k, to;
376 uint32_t res[8];
377 for (j = 0; j < 8; ++j) {
378 res[j] = 0;
379 }
380 for (j = 0; j < 4; ++j) {
381 for (k = 0; k < 4; ++k) {
382 /* Next few expressions can't overflow, since they compute at most
383 * 0xffffffff * 0xffffffff + 0xffffffff
384 * = (0xffffffff + 1) * 0xffffffff
385 * = 0x100000000 * 0xffffffff
386 * = 0xffffffff00000000
387 * or, if you prefer,
388 * (2**32 - 1)**2 + 2**32 - 1
389 * = 2**64 - 2*(2**32) + 1 + 2**32 - 1
390 * = 2**64 - 2**32
391 */
392 uint64_t xy = (uint64_t) x->part[j] * y->part[k];
393 for (to = j + k; to < 8; ++to) {
394 if (xy == 0)
395 break;
396 res[to] = xy += res[to];
397 xy >>= 32;
398 }
399 }
400 }
401 for (j = 0; j < 4; ++j) {
402 low->part[j] = res[j];
403 high->part[j] = res[j + 4];
404 }
405}

412void
413int128_signed_multiply(int128_t *high, int128_t *low,
414 const int128_t *x, const int128_t *y)
415{
416 bool xneg = int128_signed_compare(x, &zero) < 0;
417 bool yneg = int128_signed_compare(y, &zero) < 0;
418 int128_t xtmp, ytmp;
419
420 if (xneg) {
421 int128_twos_complement(&xtmp, x);
422 x = &xtmp;
423 }
424 if (yneg) {
425 int128_twos_complement(&ytmp, y);
426 y = &ytmp;
427 }
428 int128_unsigned_multiply(high, low, x, y);
429 if (xneg != yneg) {
430 int128_ones_complement(low, low);
431 int128_ones_complement(high, high);
432 if (int128_unsigned_add(low, low, &one))
433 int128_unsigned_add(high, high, &one);
434 }
435}

437bool
438int128_unsigned_divide(int128_t *quotient, int128_t *remainder,
439 const int128_t *dividend, const int128_t *divisor)
440{
441 bool allzero = int128_unsigned_compare(divisor, &zero) == 0;
442 int128_t top;
443 int bits = int128_count_leading_zeros(dividend);
444
445 *quotient = zero;
446 *remainder = zero;
447 int128_shift_left(&top, dividend, bits);
448 for (; bits < 128; ++bits) {
449 int128_t tmp = *remainder;
450 int128_unsigned_add(remainder, &tmp, &tmp);
451 tmp = top;
452 if (int128_unsigned_add(&top, &tmp, &tmp)) {
453 tmp = *remainder;
454 int128_unsigned_add(remainder, &tmp, &one);
455 }
456 tmp = *quotient;
457 int128_unsigned_add(quotient, &tmp, &tmp);
458 if (int128_unsigned_compare(remainder, divisor) >= 0) {
459 tmp = *quotient;
460 int128_unsigned_add(quotient, &tmp, &one);
461 tmp = *remainder;
462 int128_signed_subtract(remainder, &tmp, divisor);
463 }
464 }
465 return allzero;
466}

468bool
469int128_signed_divide(int128_t *quotient, int128_t *remainder,
470 const int128_t *dividend, const int128_t *divisor)
471{
472 bool negate_quotient = false, negate_remainder = false;
473 bool overflow = false;
474 int128_t tmp[4];
475 const int128_t *num = dividend, *denom = divisor;
476 if (int128_signed_compare(dividend, &zero) < 0) {
477 negate_quotient = negate_remainder = true;
478 int128_twos_complement(&tmp[0], num);
479 num = &tmp[0];
480 }
481 if (int128_signed_compare(divisor, &zero) < 0) {
482 negate_quotient = !negate_quotient;
483 int128_twos_complement(&tmp[1], denom);
484 denom = &tmp[1];
485 }
486 overflow = int128_unsigned_divide(quotient, remainder, num, denom);
487 if (int128_signed_compare(quotient, &zero) < 0) {
488 /* Signed division can overflow in one case: MOST_NEGATIVE_INT / -1 */
489 overflow = true;
490 }
491 if (negate_quotient) {
492 tmp[2] = *quotient;
493 int128_twos_complement(quotient, &tmp[2]);
494 }
495 if (negate_remainder) {
496 tmp[3] = *remainder;
497 int128_twos_complement(remainder, &tmp[3]);
498 }
499 return overflow;
