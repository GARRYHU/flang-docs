15static void I8(eoshift_scalar)(char *rb, /* result base */
16 char *ab, /* array base */
17 __INT_T shift_amt, /* shift amount */
18 char *bb, /* boundary base */
19 __INT_T shift_dim, /* shift dimension */
20 F90_Desc *rs, /* result descriptor */
21 F90_Desc *as, /* array descriptor */
22 F90_Desc *rc, /* result subsection descriptor */
23 F90_Desc *ac, /* array subsection descriptor */
24 __INT_T sub_dim) /* subsection dimension */
25{
26 chdr *c;
27 char *ap, *rp;
28 DECL_DIM_PTRS(asd);
29 DECL_DIM_PTRS(rsd);
30 __INT_T aflags, albase, apbase, arepli, ascoff;
31 __INT_T rflags, rlbase, rpbase, rrepli, rscoff;
32 __INT_T aolb[MAXDIMS], aoub[MAXDIMS];
33 __INT_T rolb[MAXDIMS], roub[MAXDIMS];
34 __INT_T extent, i, sabs;
35
36#if defined(DEBUG)
37 if (__fort_test & DEBUG_EOSH) {
38 printf("%d eoshift_scalar shift=%d boundary=", GET_DIST_LCPU, shift_amt);
39 __fort_print_scalar(bb, F90_KIND_G(rs));
40 printf(" dim=%d\n", shift_dim);
41 }
42#endif
43
44 SET_DIM_PTRS(rsd, rs, shift_dim - 1);
45 SET_DIM_PTRS(asd, as, shift_dim - 1);
46
47 extent = F90_DPTR_EXTENT_G(asd);
48 if (extent < 0)
49 return;
50
51 /* save descriptor fields affected by set/finish_section */
52
53 aflags = F90_FLAGS_G(ac);
54 albase = F90_LBASE_G(ac);
55 apbase = DIST_PBASE_G(ac);
56 arepli = DIST_REPLICATED_G(ac);
57 ascoff = DIST_SCOFF_G(ac);
58 for (i = F90_RANK_G(ac); --i >= 0;) {
59 aolb[i] = DIST_DIM_OLB_G(ac, i);
60 aoub[i] = DIST_DIM_OUB_G(ac, i);
61 }
62 rflags = F90_FLAGS_G(rc);
63 rlbase = F90_LBASE_G(rc);
64 rpbase = DIST_PBASE_G(rc);
65 rrepli = DIST_REPLICATED_G(rc);
66 rscoff = DIST_SCOFF_G(rc);
67 for (i = F90_RANK_G(rc); --i >= 0;) {
68 rolb[i] = DIST_DIM_OLB_G(rc, i);
69 roub[i] = DIST_DIM_OUB_G(rc, i);
70 }
71
72 /* copy directly if shift amount is zero */
73
74 if (shift_amt == 0) {
75 I8(__fort_set_section)(rc, sub_dim, rs, shift_dim,
76 F90_DPTR_LBOUND_G(rsd), DPTR_UBOUND_G(rsd), 1);
77 I8(__fort_finish_section)(rc);
78 I8(__fort_set_section)(ac, sub_dim, as, shift_dim,
79 F90_DPTR_LBOUND_G(asd), DPTR_UBOUND_G(asd), 1);
80 I8(__fort_finish_section)(ac);
81
82 /* adjust base addresses for scalar subscripts and copy */
83
84 rp = rb + DIST_SCOFF_G(rc) * F90_LEN_G(rc);
85 ap = ab + DIST_SCOFF_G(ac) * F90_LEN_G(ac);
86 c = I8(__fort_copy)(rp, ap, rc, ac, NULL);
87 __fort_doit(c);
88 __fort_frechn(c);
89
90 /* restore descriptor fields */
91
92 F90_FLAGS_P(ac, aflags);
93 F90_LBASE_P(ac, albase);
94 DIST_PBASE_P(ac, apbase);
95 DIST_REPLICATED_P(ac, arepli);
96 DIST_SCOFF_P(ac, ascoff);
97 for (i = F90_RANK_G(ac); --i >= 0;) {
98 DIST_DIM_OLB_P(ac, i, aolb[i]);
99 DIST_DIM_OUB_P(ac, i, aoub[i]);
100 }
101 DIST_CACHED_P(ac, 0);
102
103 F90_FLAGS_P(rc, rflags);
104 F90_LBASE_P(rc, rlbase);
105 DIST_PBASE_P(rc, rpbase);
106 DIST_REPLICATED_P(rc, rrepli);
107 DIST_SCOFF_P(rc, rscoff);
108 for (i = F90_RANK_G(rc); --i >= 0;) {
109 DIST_DIM_OLB_P(rc, i, rolb[i]);
110 DIST_DIM_OUB_P(rc, i, roub[i]);
111 }
112 DIST_CACHED_P(rc, 0);
113 return;
114 }
115
116 /* if the absolute shift amount is greater than or equal to the
117 extent, just fill the result section with boundary values */
118
119 sabs = Abs(shift_amt);
120 if (sabs >= extent) {
121 I8(__fort_set_section)(rc, sub_dim, rs, shift_dim,
122 F90_DPTR_LBOUND_G(rsd), DPTR_UBOUND_G(rsd), 1);
123 I8(__fort_finish_section)(rc);
124
125 I8(__fort_fills)(rb, rc, bb);
126
127 /* restore descriptor fields */
128
129 F90_FLAGS_P(rc, rflags);
130 F90_LBASE_P(rc, rlbase);
131 DIST_PBASE_P(rc, rpbase);
132 DIST_REPLICATED_P(rc, rrepli);
133 DIST_SCOFF_P(rc, rscoff);
134 DIST_CACHED_P(rc, 0);
135 for (i = F90_RANK_G(rc); --i >= 0;) {
136 DIST_DIM_OLB_P(rc, i, rolb[i]);
137 DIST_DIM_OUB_P(rc, i, roub[i]);
138 }
139 return;
140 }
141
142 if (shift_amt < 0)
143 sabs = extent - sabs;
144
145 /* lower part of result */
146
147 I8(__fort_set_section)(rc, sub_dim, rs, shift_dim,
148 F90_DPTR_LBOUND_G(rsd), DPTR_UBOUND_G(rsd) - sabs, 1);
149
150 I8(__fort_finish_section)(rc);
151
152 if (shift_amt > 0) {
153
154 I8(__fort_set_section)(ac, sub_dim, as, shift_dim,
155 F90_DPTR_LBOUND_G(asd) + sabs,
156 DPTR_UBOUND_G(asd), 1);
157
158 I8(__fort_finish_section)(ac);
159 rp = rb + DIST_SCOFF_G(rc) * F90_LEN_G(rc);
160 ap = ab + DIST_SCOFF_G(ac) * F90_LEN_G(ac);
161 c = I8(__fort_copy)(rp, ap, rc, ac, NULL);
162 } else
163 I8(__fort_fills)(rb, rc, bb);
164
165 /* restore descriptor fields */
166
167 F90_FLAGS_P(ac, aflags);
168 F90_LBASE_P(ac, albase);
169 DIST_PBASE_P(ac, apbase);
170 DIST_REPLICATED_P(ac, arepli);
171 DIST_SCOFF_P(ac, ascoff);
172 DIST_CACHED_P(ac, 0);
173 for (i = F90_RANK_G(ac); --i >= 0;) {
174 DIST_DIM_OLB_P(ac, i, aolb[i]);
175 DIST_DIM_OUB_P(ac, i, aoub[i]);
176 }
177 F90_FLAGS_P(rc, rflags);
178 F90_LBASE_P(rc, rlbase);
179 DIST_PBASE_P(rc, rpbase);
180 DIST_REPLICATED_P(rc, rrepli);
181 DIST_SCOFF_P(rc, rscoff);
182 for (i = F90_RANK_G(rc); --i >= 0;) {
183 DIST_DIM_OLB_P(rc, i, rolb[i]);
184 DIST_DIM_OUB_P(rc, i, roub[i]);
185 }
186 DIST_CACHED_P(rc, 0);
187
188 /* upper part of result */
189
190 I8(__fort_set_section)(rc, sub_dim, rs, shift_dim,
191 F90_DPTR_LBOUND_G(rsd) + (extent - sabs),
192 DPTR_UBOUND_G(rsd), 1);
193
194 I8(__fort_finish_section)(rc);
195
196 if (shift_amt < 0) {
197
198 I8(__fort_set_section)(ac, sub_dim, as, shift_dim,
199 F90_DPTR_LBOUND_G(asd),
200 DPTR_UBOUND_G(asd) - (extent - sabs), 1);
201
202 I8(__fort_finish_section)(ac);
203 rp = rb + DIST_SCOFF_G(rc) * F90_LEN_G(rc);
204 ap = ab + DIST_SCOFF_G(ac) * F90_LEN_G(ac);
205 c = I8(__fort_copy)(rp, ap, rc, ac, NULL);
206 } else
207 I8(__fort_fills)(rb, rc, bb);
208
209 __fort_doit(c);
210 __fort_frechn(c);
211
212 /* restore descriptor fields */
213
214 F90_FLAGS_P(ac, aflags);
215 F90_LBASE_P(ac, albase);
216 DIST_PBASE_P(ac, apbase);
217 DIST_REPLICATED_P(ac, arepli);
218 DIST_SCOFF_P(ac, ascoff);
219 for (i = F90_RANK_G(ac); --i >= 0;) {
220 DIST_DIM_OLB_P(ac, i, aolb[i]);
221 DIST_DIM_OUB_P(ac, i, aoub[i]);
222 }
223 DIST_CACHED_P(ac, 0);
224
225 F90_FLAGS_P(rc, rflags);
226 F90_LBASE_P(rc, rlbase);
227 DIST_PBASE_P(rc, rpbase);
228 DIST_REPLICATED_P(rc, rrepli);
229 DIST_SCOFF_P(rc, rscoff);
230 for (i = F90_RANK_G(rc); --i >= 0;) {
231 DIST_DIM_OLB_P(rc, i, rolb[i]);
232 DIST_DIM_OUB_P(rc, i, roub[i]);
233 }
234 DIST_CACHED_P(rc, 0);
235}

237static void I8(eoshift_loop)(char *rb, /* result base */
238 char *ab, /* array base */
239 __INT_T *sb, /* shift base */
240 char *bb, /* boundary base */
241 __INT_T shift_dim, /* dimension to shift */
242 F90_Desc *rs, /* result descriptor */
243 F90_Desc *as, /* array descriptor */
244 F90_Desc *ss, /* shift descriptor */
245 F90_Desc *bs, /* boundary descriptor */
246 F90_Desc *rc, /* result subsection descriptor */
247 F90_Desc *ac, /* array subsection descriptor */
248 __INT_T soff, /* shift offset */
249 __INT_T boff, /* boundary offset */
250 __INT_T loop_dim) /* loop dim */
251{
252 DECL_DIM_PTRS(asd);
253 DECL_DIM_PTRS(bsd);
254 DECL_DIM_PTRS(rsd);
255 DECL_DIM_PTRS(ssd);
256 __INT_T aflags, albase, apbase, arepli, ascoff;
257 __INT_T rflags, rlbase, rpbase, rrepli, rscoff;
258 __INT_T ai, array_dim, bstr, ri, sstr;
259
260 /* shift rank = array rank - 1*/
261
262 array_dim = loop_dim;
263 if (array_dim >= shift_dim)
264 ++array_dim;
265
266 SET_DIM_PTRS(rsd, rs, array_dim - 1);
267 ri = F90_DPTR_LBOUND_G(rsd);
268
269 SET_DIM_PTRS(asd, as, array_dim - 1);
270 ai = F90_DPTR_LBOUND_G(asd);
271
272 if (F90_TAG_G(ss) == __DESC) {
273 SET_DIM_PTRS(ssd, ss, loop_dim - 1);
274 sstr = F90_DPTR_SSTRIDE_G(ssd) * F90_DPTR_LSTRIDE_G(ssd);
275 soff += (F90_DPTR_SSTRIDE_G(ssd) * F90_DPTR_LBOUND_G(ssd) +
276 F90_DPTR_SOFFSET_G(ssd)) *
277 F90_DPTR_LSTRIDE_G(ssd);
278 } else
279 sstr = soff = 0;
280
281 if (F90_TAG_G(bs) == __DESC) {
282 SET_DIM_PTRS(bsd, bs, loop_dim - 1);
283 bstr = F90_DPTR_SSTRIDE_G(bsd) * F90_DPTR_LSTRIDE_G(bsd);
284 boff += (F90_DPTR_SSTRIDE_G(bsd) * F90_DPTR_LBOUND_G(bsd) +
285 F90_DPTR_SOFFSET_G(bsd)) *
286 F90_DPTR_LSTRIDE_G(bsd);
287 } else
288 bstr = boff = 0;
289
290 /* save descriptor fields affected by set_single */
291
292 aflags = F90_FLAGS_G(ac);
293 albase = F90_LBASE_G(ac);
294 apbase = DIST_PBASE_G(ac);
295 arepli = DIST_REPLICATED_G(ac);
296 ascoff = DIST_SCOFF_G(ac);
297
298 rflags = F90_FLAGS_G(rc);
299 rlbase = F90_LBASE_G(rc);
300 rpbase = DIST_PBASE_G(rc);
301 rrepli = DIST_REPLICATED_G(rc);
302 rscoff = DIST_SCOFF_G(rc);
303
304 for (; ri <= DPTR_UBOUND_G(rsd); ++ri, ++ai, soff += sstr, boff += bstr) {
305 I8(__fort_set_single)(rc, rs, array_dim, ri, __SCALAR);
306 I8(__fort_set_single)(ac, as, array_dim, ai, __SCALAR);
307
308 if (loop_dim > 1)
309 I8(eoshift_loop)(rb, ab, sb, bb, shift_dim, rs, as, ss, bs, rc, ac,
310 soff, boff, loop_dim-1);
311 else
312
313 I8(eoshift_scalar)(rb, ab, sb[soff], bb + boff*F90_LEN_G(bs),
314 shift_dim, rs, as, rc, ac, 1);
315
316 /* restore descriptor fields */
317
318 F90_FLAGS_P(ac, aflags);
319 F90_LBASE_P(ac, albase);
320 DIST_PBASE_P(ac, apbase);
321 DIST_REPLICATED_P(ac, arepli);
322 DIST_SCOFF_P(ac, ascoff);
323 DIST_CACHED_P(ac, 0);
324
325 F90_FLAGS_P(rc, rflags);
326 F90_LBASE_P(rc, rlbase);
327 DIST_PBASE_P(rc, rpbase);
328 DIST_REPLICATED_P(rc, rrepli);
329 DIST_SCOFF_P(rc, rscoff);
330 DIST_CACHED_P(rc, 0);
331 }
332}

336void ENTFTN(EOSHIFTSZ, eoshiftsz)(char *rb, /* result base */
337 char *ab, /* array base */
338 __INT_T *sb, /* shift base */
339 __INT_T *db, /* dimension */
340 F90_Desc *rs, /* result descriptor */
341 F90_Desc *as, /* array descriptor */
342 F90_Desc *ss, /* shift descriptor */
343 F90_Desc *ds) /* dim descriptor */
344{
345 char *bb;
346 DECL_HDR_VARS(ac);
347 DECL_HDR_VARS(rc);
348 DECL_DIM_PTRS(asd);
349 DECL_DIM_PTRS(rsd);
350 __INT_T dim, i, shift;
351
352 shift = *sb;
353 dim = *db;
354 /* FIXME: bb is passed to many non-const parameters; just cast it for now. */
355 bb = (F90_KIND_G(rs) == __STR) ? (char *)" " : (char *)GET_DIST_ZED;
356
357#if defined(DEBUG)
358 if (__fort_test & DEBUG_EOSH) {
359 printf("%d r", GET_DIST_LCPU);
360 I8(__fort_show_section)(rs);
361 printf("@%x = EOSHIFT(a", rb);
362 I8(__fort_show_section)(as);
363 printf("@%x, shift=%d, dim=%d)\n", ab, shift, dim);
364 }
365#endif
366
367 /* initialize section descriptors */
368
369 __DIST_INIT_SECTION(ac, F90_RANK_G(as), as);
370 __DIST_INIT_SECTION(rc, F90_RANK_G(rs), rs);
371
372 for (i = 1; i <= F90_RANK_G(as); ++i) {
373 if (i == dim)
374 continue;
375 SET_DIM_PTRS(asd, as, i - 1);
376 I8(__fort_set_section)(ac, i, as, i, F90_DPTR_LBOUND_G(asd),
377 DPTR_UBOUND_G(asd), 1);
378 SET_DIM_PTRS(rsd, rs, i - 1);
379 I8(__fort_set_section)(rc, i, rs, i, F90_DPTR_LBOUND_G(rsd),
380 DPTR_UBOUND_G(rsd), 1);
381 }
382
383 I8(eoshift_scalar)(rb, ab, shift, bb, dim, rs, as, rc, ac, dim);
384}
