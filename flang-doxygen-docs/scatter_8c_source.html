12/** \file
13 * \brief Scatter reduction routines
14 *
15 * call ENTFTN(xxx_scatter)(rb, rs, mb, ms, bb, bs, ...)
16 * where xxx = all, any, count, parity
17 *
18 * call ENTFTN(yyy_scatter)(rb, rs, ab, as, bb, bs, mb, ms, ...)
19 * where yyy = copy, iall, iany, iparity, maxval, minval, product, sum
20 *
21 * rb = result base address
22 * rs = result descriptor
23 * ab = array base address
24 * as = array descriptor
25 * bb = base base address
26 * bs = base descriptor
27 * mb = mask base address
28 * ms = mask descriptor
29 *
30 * ... = { xb, xs, }* = base address and descriptors for the indexes,
31 * one pair for each result dimension.
32 *
33 * indexes must not be scalar, instead the result section may have a
34 * scalar subscript. Index values must be within the bounds of the
35 * corresponding result dimension.
36 *
37 * indexes must conform with array. Each dimension of every index
38 * must be either 1) replicated or 2) aligned with the corresponding
39 * dimension of array.
40 *
41 * mask must conform with array, may be scalar, and is assumed to be
42 * of type logical*4. Each dimension of mask must be either
43 * replicated or aligned with the corresponding dimension of array.
44 */

82/* ENTFTN(comm_start) function: gather, transfer, scatter */
83
84static void I8(gathscat_start)(void *skp, char *rb, char *sb, F90_Desc *rd,
85 F90_Desc *sd)
86{
87 gathscat_sked *sk = (gathscat_sked *)skp;
88 char *rp, *sp, *bufr, *bufs;
89 int cpu, j, k, lcpu, m, nr, ns, tcpus;
90
91 double t;
92 if (__fort_test & DEBUG_TIME)
93 t = __fort_second();
94
95 rp = rb + DIST_SCOFF_G(rd) * F90_LEN_G(rd);
96 sp = sb + DIST_SCOFF_G(sd) * F90_LEN_G(sd);
97
98 /* handle local elements first */
99
100 j = k = sk->lclcnt;
101 if (k > 0) {
102
103 /*
104 * make sure base type of object hasn't changed for local gather...
105 * This can occur when we share schedules across objects ...
106 */
107 /* Couldn't find where gathscatfn is set to local_gathscat_WRAPPER.
108 Possible dead code? Work around the compiler warning with an
109 explicit cast for now. */
110 if (sk->gathscatfn == (gathscatfn_t)local_gathscat_WRAPPER) {
111
112 local_gathscat_WRAPPER(k, rp, sk->soff, sp, sk->goff, F90_KIND_G(rd));
113
114 } else {
115 sk->gathscatfn(k, rp, sk->soff, sp, sk->goff);
116 }
117 }
118 /* now non-local elements */
119
120 lcpu = GET_DIST_LCPU;
121 if (sk->maxcnt > 0) {
122
123 /* allocate buffers for aggregated elements */
124
125 bufr = (char *)__fort_gmalloc(2 * sk->maxcnt * F90_LEN_G(sd));
126 bufs = bufr + sk->maxcnt * F90_LEN_G(sd);
127
128 /* exchange targets are chosen by xor'ing this processor number
129 with all masks in the range 1 .. 2**log2(__fort_tcpus). Target
130 values >= tcpus are skipped (which occurs only if tcpus is not
131 a power of two.) */
132
133 tcpus = GET_DIST_TCPUS;
134 for (m = 1; m < __fort_np2; ++m) {
135
136 /* identify target */
137
138 cpu = lcpu ^ m;
139 if (cpu >= tcpus)
140 continue;
141
142 nr = sk->countr[cpu];
143 ns = sk->counts[cpu];
144
145 /* exchange elements */
146
147 if (ns > 0)
148 local_gather_WRAPPER(ns, bufs, sp, sk->goff + j, F90_KIND_G(sd));
149
150 if (cpu < lcpu) {
151 if (nr > 0)
152 __fort_rrecvl(cpu, bufr, nr, 1, F90_KIND_G(sd), F90_LEN_G(sd));
153 if (ns > 0)
154 __fort_rsendl(cpu, bufs, ns, 1, F90_KIND_G(sd), F90_LEN_G(sd));
155 } else {
156 if (ns > 0)
157 __fort_rsendl(cpu, bufs, ns, 1, F90_KIND_G(sd), F90_LEN_G(sd));
158 if (nr > 0)
159 __fort_rrecvl(cpu, bufr, nr, 1, F90_KIND_G(sd), F90_LEN_G(sd));
160 }
161
162 if (nr > 0) {
163
164 /*
165 * Added call to wrapper routine for local scatter
166 * in order to handle cases where the schedule is
167 * shared between objects of different types...
168 */
169 /* Couldn't find where gathscatfn is set to local_gathscat_WRAPPER.
170 Possible dead code? Work around the compiler warning with an
171 explicit cast for now. */
172 if (sk->scatterfn == (scatterfn_t)local_scatter_WRAPPER) {
173 local_scatter_WRAPPER(nr, rp, sk->soff + k, bufr, F90_KIND_G(rd));
174 } else {
175 sk->scatterfn(nr, rp, sk->soff + k, bufr);
176 }
177 }
178
179 j += ns;
180 k += nr;
181 }
182
183 __fort_gfree(bufr);
184 }
185
186 /* replicate result if needed */
187
188 if (sk->repchn) {
189 __fort_adjbase(sk->repchn, rp, rp, F90_KIND_G(rd), F90_LEN_G(rd));
190 __fort_doit(sk->repchn);
191 }
192
193 if (__fort_test & DEBUG_TIME) {
194 t = __fort_second() - t;
195 printf("%d %s execute %.6f\n", lcpu, sk->what, t);
196 }
197}

216static void I8(gathscat_element)(gathscat_parm *z, __INT_T uoff, __INT_T xoff[])
217{
218 gathscat_dim *zd;
219 DECL_HDR_PTRS(ud);
220 DECL_HDR_PTRS(vd);
221 int cpu, k, ux, vx;
222 __INT_T roff;
223 __INT_T vi[MAXDIMS];
224
225 ud = z->ud;
226 vd = z->vd;
227
228 /* construct vectored array index tuple */
229
230 for (vx = F90_RANK_G(vd); --vx >= 0;) {
231 zd = &z->dim[vx];
232 if (z->indirect >> vx & 1)
233 vi[vx] = zd->xb[xoff[vx]];
234 else {
235 ux = *zd->xmap;
236 vi[vx] = z->ui[ux - 1];
237 }
238 }
239
240#if defined(DEBUG)
241 if (__fort_test & DEBUG_SCAT) {
242 for (vx = 0; vx < F90_RANK_G(vd); ++vx) {
243 zd = &z->dim[vx];
244 if (z->indirect >> vx & 1)
245 printf("%d vx %d xoff[vx] %d vi[vx] %d\n", GET_DIST_LCPU, vx, xoff[vx],
246 vi[vx]);
247 else
248 printf("%d vx %d ux %d vi[vx] %d\n", GET_DIST_LCPU, vx, *zd->xmap,
249 vi[vx]);
250 }
251 }
252#endif
253
254 k = ++z->outgoing; /* count outgoing elements */
255
256 if (z->communicate | z->replicate) {
257
258 /* identify target processor and element offset, count
259 elements for target processor */
260
261 I8(__fort_localize)(vd, vi, &cpu, &roff);
262 cpu += z->group_offset;
263 ++z->counts[cpu];
264
265 /* link to target cpu's list. using a linked list reverses
266 the order of elements processed */
267
268 z->next[k - 1] = z->head[cpu];
269 z->head[cpu] = k;
270 } else {
271 cpu = GET_DIST_LCPU;
272 roff = I8(__fort_local_offset)(vd, vi);
273 }
274
275 /* store offsets in local and remote arrays */
276
277 z->loff[k - 1] = uoff - DIST_SCOFF_G(ud);
278 z->roff[k - 1] = roff;
279
280#if defined(DEBUG)
281 if (__fort_test & DEBUG_SCAT) {
282 printf("%d %s remote cpu %d r", GET_DIST_LCPU, z->what, cpu);
283 if (z->dir == __SCATTER) {
284 I8(__fort_show_index)(F90_RANK_G(vd), vi);
285 printf("@%d = a", roff);
286 }
287 I8(__fort_show_index)(F90_RANK_G(ud), z->ui);
288 printf("@%d ", uoff);
289 if (z->dir == __GATHER) {
290 printf("= a");
291 I8(__fort_show_index)(F90_RANK_G(vd), vi);
292 printf("@%d", roff);
293 } else
294 __fort_print_scalar(z->ub + uoff * F90_LEN_G(ud), F90_KIND_G(ud));
295 printf("\n");
296 }
297#endif
298}

300/* loop over local elements of unvectored array, checking mask */
301
302static void
303 I8(gathscat_mask_loop)(gathscat_parm *z, /* parameters */
304 int uoff0, /* unvectored array element offset */
305 __INT_T xoff[], /* index array element offsets */
306 int dim) /* unvectored array/mask dimension */
307{
308 DECL_HDR_PTRS(ud);
309 DECL_DIM_PTRS(udd);
310 DECL_F90_DIM_PTR(xdd);
311 DECL_DIST_DIM_PTR(xdd);
312 xstuff *x;
313 __LOG_T *mb;
314 __LOG_T mask_log;
315 __INT_T n, ubl, ubu, ubn, ucl, ucn, uclof, uoff;
316
317 ud = z->ud;
318 SET_DIM_PTRS(udd, ud, dim - 1);
319 ucn = DIST_DPTR_CN_G(udd);
320 ucl = DIST_DPTR_CL_G(udd);
321 uclof = DIST_DPTR_CLOF_G(udd);
322 ubn = 0;
323
324 mb = (__LOG_T *)z->mb;
325
326 for (x = z->xhead[dim - 1]; x; x = x->next) {
327 if (z->aligned_x_u >> x->vx & 1) {
328 F90_DIM_NAME(xdd) = x->F90_DIM_NAME(xdd);
329 DIST_DIM_NAME(xdd) = x->DIST_DIM_NAME(xdd);
330 x->cn = DIST_DPTR_CN_G(xdd);
331 x->cl = DIST_DPTR_CL_G(xdd);
332 x->cs = DIST_DPTR_CS_G(xdd);
333 x->clof = DIST_DPTR_CLOF_G(xdd);
334 x->clos = DIST_DPTR_CLOS_G(xdd);
335 } else
336 x->cn = 0;
337 x->bn = 0;
338 x->off0 = xoff[x->vx]; /* save initial offset */
339 }
340
341 mask_log = GET_DIST_MASK_LOG;
342 while (ubn > 0 || ucn > 0) {
343 if (ubn == 0) {
344 ubn = I8(__fort_block_bounds)(ud, dim, ucl, &ubl, &ubu);
345 uoff = uoff0 +
346 (F90_DPTR_SSTRIDE_G(udd) * ubl - uclof) * F90_DPTR_LSTRIDE_G(udd);
347 ucl += DIST_DPTR_CS_G(udd);
348 uclof += DIST_DPTR_CLOS_G(udd);
349 --ucn;
350 z->ui[dim - 1] = ubl; /* unvectored array index */
351 }
352 n = ubn;
353 for (x = z->xhead[dim - 1]; x; x = x->next) {
354 if (x->bn == 0) { /* block exhausted */
355 __INT_T xbl, xbu;
356
357 F90_DIM_NAME(xdd) = x->F90_DIM_NAME(xdd);
358 DIST_DIM_NAME(xdd) = x->DIST_DIM_NAME(xdd);
359 if (z->aligned_x_u >> x->vx & 1) {
360
361 /* mask/index aligned with unvectored. local
362 blocks should synchronize with unvectored array */
363
364 if (x->cn <= 0)
365 gathscat_abort(z->what, "index misalignment");
366 x->bn = I8(__fort_block_bounds)(x->xd, x->xx + 1, x->cl, &xbl, &xbu);
367 } else {
368
369 /* mask/index not aligned with unvectored (but
370 local wherever unvectored is local). need to
371 set up loop bounds corresponding to each new
372 unvectored block */
373
374 __INT_T xl = F90_DPTR_LBOUND_G(xdd) + ubl - F90_DPTR_LBOUND_G(udd);
375 __INT_T xu = xl + ubn - 1;
376
377 if (x->cn <= 0) {
378 __INT_T xcu;
379 x->cn = I8(__fort_cyclic_loop)(x->xd, x->xx + 1, xl, xu, 1, &x->cl,
380 &xcu, &x->cs, &x->clof, &x->clos);
381 }
382 x->bn = I8(__fort_block_loop)(x->xd, x->xx + 1, xl, xu, 1, x->cl, &xbl,
383 &xbu);
384 }
385 xoff[x->vx] =
386 x->off0 +
387 (F90_DPTR_SSTRIDE_G(xdd) * xbl - x->clof) * F90_DPTR_LSTRIDE_G(xdd);
388 x->cl += x->cs;
389 x->clof += x->clos;
390 --x->cn;
391 }
392 if (x->bn < n)
393 n = x->bn;
394 }
395 ubl += n;
396 ubn -= n;
397 for (x = z->xhead[dim - 1]; x; x = x->next)
398 x->bn -= n;
399 while (--n >= 0) {
400 if (dim > 1)
401 I8(gathscat_mask_loop)(z, uoff, xoff, dim - 1);
402 else if (mb[xoff[MAXDIMS]] & mask_log)
403 I8(gathscat_element)(z, uoff, xoff);
404 uoff += F90_DPTR_SSTRIDE_G(udd) * F90_DPTR_LSTRIDE_G(udd);
405 for (x = z->xhead[dim - 1]; x; x = x->next)
406 xoff[x->vx] += x->str;
407 ++z->ui[dim - 1];
408 }
409 }
410 for (x = z->xhead[dim - 1]; x; x = x->next)
411 xoff[x->vx] = x->off0; /* restore offset */
412}

416static void I8(gathscat_loop)(gathscat_parm *z, /* parameters */
417 int uoff0, /* unvectored array element offset */
418 __INT_T xoff[], /* index array element offsets */
419 int dim) /* unvectored array dimension */
420{

422 DECL_DIM_PTRS(udd);
423 DECL_F90_DIM_PTR(xdd);

426 __INT_T n, ubl, ubu, ubn, ucl, ucn, uclof, uoff;
427
428 ud = z->ud;
429 SET_DIM_PTRS(udd, ud, dim - 1);
430 ucn = DIST_DPTR_CN_G(udd);
431 ucl = DIST_DPTR_CL_G(udd);
432 uclof = DIST_DPTR_CLOF_G(udd);
433 ubn = 0;
434
435 for (x = z->xhead[dim - 1]; x; x = x->next) {
436 if (z->aligned_x_u >> x->vx & 1) {
437 F90_DIM_NAME(xdd) = x->F90_DIM_NAME(xdd);
438 DIST_DIM_NAME(xdd) = x->DIST_DIM_NAME(xdd);

441 x->cs = DIST_DPTR_CS_G(xdd);
442 x->clof = DIST_DPTR_CLOF_G(xdd);
443 x->clos = DIST_DPTR_CLOS_G(xdd);
444 } else
445 x->cn = 0;
446 x->bn = 0;
447 x->off0 = xoff[x->vx]; /* save initial offset */
448 }
449
450 while (ubn > 0 || ucn > 0) {
451 if (ubn == 0) {
452 ubn = I8(__fort_block_bounds)(ud, dim, ucl, &ubl, &ubu);
453 uoff = uoff0 +
454 (F90_DPTR_SSTRIDE_G(udd) * ubl - uclof) * F90_DPTR_LSTRIDE_G(udd);
455 ucl += DIST_DPTR_CS_G(udd);
456 uclof += DIST_DPTR_CLOS_G(udd);
457 --ucn;
458 z->ui[dim - 1] = ubl; /* unvectored array index */
459 }
460 n = ubn;
461 for (x = z->xhead[dim - 1]; x; x = x->next) {
462 if (x->bn == 0) { /* block exhausted */
463 __INT_T xbl, xbu;
464
465 F90_DIM_NAME(xdd) = x->F90_DIM_NAME(xdd);
466 DIST_DIM_NAME(xdd) = x->DIST_DIM_NAME(xdd);
467 if (z->aligned_x_u >> x->vx & 1) {
468
469 /* index aligned with unvectored. local blocks
470 should synchronize with unvectored array */
471
472 if (x->cn <= 0)
473 gathscat_abort(z->what, "index misalignment");
474 x->bn = I8(__fort_block_bounds)(x->xd, x->xx + 1, x->cl, &xbl, &xbu);

476
477 /* index not aligned with unvectored (but local
478 wherever unvectored is local). need to set up
479 loop bounds corresponding to each new
480 unvectored block */
481
482 __INT_T xl = F90_DPTR_LBOUND_G(xdd) + ubl - F90_DPTR_LBOUND_G(udd);
483 __INT_T xu = xl + ubn - 1;
484
485 if (x->cn <= 0) {
486 __INT_T xcu;
487 x->cn = I8(__fort_cyclic_loop)(x->xd, x->xx + 1, xl, xu, 1, &x->cl,
488 &xcu, &x->cs, &x->clof, &x->clos);
489 }
490 x->bn = I8(__fort_block_loop)(x->xd, x->xx + 1, xl, xu, 1, x->cl, &xbl,
491 &xbu);
492 }
493 xoff[x->vx] =
494 x->off0 +
495 (F90_DPTR_SSTRIDE_G(xdd) * xbl - x->clof) * F90_DPTR_LSTRIDE_G(xdd);
496 x->cl += x->cs;
497 x->clof += x->clos;
498 --x->cn;
499 }
500 if (x->bn < n)
501 n = x->bn;
502 }
503 ubl += n;
504 ubn -= n;
505 for (x = z->xhead[dim - 1]; x; x = x->next)
506 x->bn -= n;
507 while (--n >= 0) {
508 if (dim > 1)
509 I8(gathscat_loop)(z, uoff, xoff, dim - 1);
510 else
511 I8(gathscat_element)(z, uoff, xoff);
512 uoff += F90_DPTR_SSTRIDE_G(udd) * F90_DPTR_LSTRIDE_G(udd);
513 for (x = z->xhead[dim - 1]; x; x = x->next)
514 xoff[x->vx] += x->str;
515 ++z->ui[dim - 1];
516 }
517 }
518 for (x = z->xhead[dim - 1]; x; x = x->next)
519 xoff[x->vx] = x->off0; /* restore offset */
520}

522/* u = unvectored, v = vectored, x = index */
523
524sked *I8(__fort_gathscat)(gathscat_parm *z)
525{
526 gathscat_sked *sk;
527 gathscat_dim *zd;
528 char *rp;
529 chdr *repchn;
530 DECL_HDR_PTRS(md);
531 DECL_HDR_PTRS(ud);
532 DECL_HDR_PTRS(vd);
533 DECL_HDR_PTRS(rd);
534 DECL_HDR_PTRS(xd);
535 DECL_F90_DIM_PTR(mdd);
536 DECL_DIST_DIM_PTR(mdd) = NULL;
537 DECL_DIM_PTRS(udd);
538 DECL_F90_DIM_PTR(xdd);
539 DECL_DIST_DIM_PTR(xdd) = NULL;
540 xstuff *x;
541
542 int *countbuf, *countr, *counts, *goff, *head, *loff, *next, *offr, *offs,
543 *offsetbuf, *roff, *soff;
544
545 int alike, cpu, different, incoming, i, j, k, m, n, nr, ns;
546 int lclcnt, lcpu, maxcnt, tempz, tcpus, u_covers_v, v_covers_u;
547#if defined(DEBUG)
548 int has_gb = 0;
549#endif
550 __INT_T mx, uoff, ux, vx, xx, xoff[MAXDIMS + 1];
551
552 repl_t u_repl; /* unvectored array replication descriptor */
553
554 double t;
555 if (__fort_test & DEBUG_TIME)
556 t = __fort_second();
557
558 md = z->md;
559 ud = z->ud;
560 vd = z->vd;
561 rd = z->rd;
562
563 lcpu = GET_DIST_LCPU;
564
565#if defined(DEBUG)
566 if (z->gathscatfn == NULL || z->scatterfn == NULL)
567 gathscat_abort(z->what, "unsupported data type");
568#endif
569
570 /* initial bit masks */
571
572 z->conform_x_u = 0; /* index conforms with unvectored */
573 z->aligned_x_u = 0; /* index aligned with unvectored */
574 z->aligned_v_u = 0; /* vectored aligned with unvectored */
575 z->aligned_u_v = 0; /* unvectored aligned with vectored */
576
577 if (~z->indirect & ~(-1 << F90_RANK_G(vd))) {
578 /* if any vectored dim is not indirectly indexed... */
579 u_covers_v = I8(__fort_covers_procs)(ud, vd);
580 v_covers_u = I8(__fort_covers_procs)(vd, ud);
581 }
582
583 /* initialize inverse mapping list of mask and index dimensions
584 subscripted by each unvectored axis */
585
586 uoff = F90_LBASE_G(ud) - 1;
587 for (ux = F90_RANK_G(ud); --ux >= 0;) {
588 SET_DIM_PTRS(udd, ud, ux);
589 uoff += F90_DPTR_SOFFSET_G(udd) * F90_DPTR_LSTRIDE_G(udd);
590 z->xhead[ux] = NULL; /* empty xlists */
591 }
592 z->xfree = z->xlist; /* first free xlist entry */
593
594 /* check mask -- finished already if mask is scalar .false. */
595
596 if (F90_TAG_G(md) == __DESC) {
597 if (F90_KIND_G(md) != __LOG)
598 gathscat_abort(z->what, "mask array must be logical");
599
600 /* treat mask as an extra index array for address calculations */
601
602 I8(__fort_cycle_bounds)(md);
603 xoff[MAXDIMS] = F90_LBASE_G(md) - 1;
604 for (mx = F90_RANK_G(md); --mx >= 0;) {
605 SET_DIM_PTRS(mdd, md, mx);
606 xoff[MAXDIMS] += F90_DPTR_SOFFSET_G(mdd) * F90_DPTR_LSTRIDE_G(mdd);
607
608 /* add mask dimension to xlist for the corresponding
609 unvectored dimension. mask axis permutation is not
610 supported, but the mask rank may be less than the
611 unvectored array rank */
612
613 x = z->xfree++;
614 x->next = z->xhead[mx];
615 z->xhead[mx] = x;
616 x->xd = md;
617 SET_DIM_PTRS(mdd, md, mx);
618 x->F90_DIM_NAME(xdd) = F90_DIM_NAME(mdd);
619 x->DIST_DIM_NAME(xdd) = DIST_DIM_NAME(mdd);
620 x->str = F90_DPTR_SSTRIDE_G(mdd) * F90_DPTR_LSTRIDE_G(mdd);
621 x->vx = MAXDIMS;
622 x->xx = mx;
623 }
624 z->conform_x_u |= I8(__fort_conform)(md, id_map, ud, id_map) << MAXDIMS;
625 z->aligned_x_u |= I8(__fort_aligned)(md, id_map, ud, id_map) << MAXDIMS;
626 if (!I8(__fort_aligned)(ud, id_map, md, id_map)) {
627#if defined(DEBUG)
628 if (__fort_test & DEBUG_SCAT) {
629 printf("%d %s unvectored array:\n", lcpu, z->what);
630 I8(__fort_describe)(z->ub, ud);
631 printf("%d %s misaligned mask array:\n", lcpu, z->what);
632 I8(__fort_describe)((char *)z->mb, md);
633 }
634#endif
635 gathscat_abort(z->what, "misaligned mask array");
636 }
637 } else if (!I8(__fort_fetch_log)(z->mb, md))
638 return NULL;
639
640 /* check index arrays */
641
642 for (vx = 0; vx < F90_RANK_G(vd); ++vx) {
643 zd = &z->dim[vx];
644
645 if (z->indirect >> vx & 1) {
646
647 /* indirection in this vectored array dimension */
648
649 xd = zd->xd;
650#if defined(DEBUG)
651 if (F90_TAG_G(xd) != __DESC)
652 gathscat_abort(z->what, "index must be array");
653 if (F90_KIND_G(xd) != __INT)
654 gathscat_abort(z->what, "index must be integer");
655#endif
656 I8(__fort_cycle_bounds)(xd);
657 xoff[vx] = F90_LBASE_G(xd) - 1;
658 for (xx = F90_RANK_G(xd); --xx >= 0;) {
659 SET_DIM_PTRS(xdd, xd, xx);
660 xoff[vx] += F90_DPTR_SOFFSET_G(xdd) * F90_DPTR_LSTRIDE_G(xdd);
661 }
662
663 if (z->permuted >> vx & 1) {
664
665 /* index axes are a permutation of the unvectored
666 array axes. check for identity permutation */
667
668 alike = 1;
669 for (xx = F90_RANK_G(xd); --xx >= 0;) {
670 ux = zd->xmap[xx] - 1;
671#if defined(DEBUG)
672 if (ux < 0 || ux >= F90_RANK_G(ud))
673 gathscat_abort(z->what, "invalid index axis");
674#endif
675 alike &= (ux == xx);
676
677 /* set up inverse mapping from unvectored axis to
678 list of vectored axis/index axis pairs */
679
680 x = z->xfree++;
681 x->next = z->xhead[ux];
682 z->xhead[ux] = x;
683 x->xd = xd;
684 SET_DIM_PTRS(xdd, xd, xx);
685 x->F90_DIM_NAME(xdd) = F90_DIM_NAME(xdd);
686 x->DIST_DIM_NAME(xdd) = DIST_DIM_NAME(xdd);
687 x->str = F90_DPTR_SSTRIDE_G(xdd) * F90_DPTR_LSTRIDE_G(xdd);
688 x->vx = vx;
689 x->xx = xx;
690 }
691 if (alike)
692 z->permuted &= ~(1 << vx);
693#if defined(DEBUG)
694 else if (__fort_test & DEBUG_SCAT) {
695 printf("%d %s dim %d index axis permutation", lcpu, z->what, vx + 1);
696 for (xx = 0; xx < F90_RANK_G(xd); ++xx)
697 printf(" %d", zd->xmap[xx]);
698 printf("\n");
699 }
700#endif
701 } else {
702
703 /* index axes not permuted */
704
705 zd->xmap = id_map;
706#if defined(DEBUG)
707 if (F90_RANK_G(xd) > F90_RANK_G(ud))
708 gathscat_abort(z->what, "invalid index rank");
709#endif
710
711 /* set up inverse mapping from unvectored axis to list
712 of vectored axis/index axis pairs */
713
714 for (xx = F90_RANK_G(xd); --xx >= 0;) {
715 ux = xx;
716 x = z->xfree++;
717 x->next = z->xhead[ux];
718 z->xhead[ux] = x;
719 x->xd = xd;
720 SET_DIM_PTRS(xdd, xd, xx);
721 x->F90_DIM_NAME(xdd) = F90_DIM_NAME(xdd);
722 x->DIST_DIM_NAME(xdd) = DIST_DIM_NAME(xdd);
723 x->str = F90_DPTR_SSTRIDE_G(xdd) * F90_DPTR_LSTRIDE_G(xdd);
724 x->vx = vx;
725 x->xx = xx;
726 }
727 }
728
729 z->conform_x_u |= I8(__fort_conform)(xd, zd->xmap, ud, id_map) << vx;
730 z->aligned_x_u |= I8(__fort_aligned)(xd, zd->xmap, ud, id_map) << vx;
731
732 if (!I8(__fort_aligned)(ud, id_map, xd, zd->xmap)) {
733#if defined(DEBUG)
734 if (__fort_test & DEBUG_SCAT) {
735 printf("%d %s unvectored array:\n", lcpu, z->what);
736 I8(__fort_describe)(z->ub, ud);
737 printf("%d %s misaligned index array:\n", lcpu, z->what);
738 I8(__fort_describe)((char *)zd->xb, xd);
739 }
740#endif
741 gathscat_abort(z->what, "misaligned index array");
742 }
743 } else {
744
745 /* no indirection in this vectored array dimension */
746
747 zd->xd = NULL;
748 zd->xb = NULL;
749
750 xoff[vx] = 0;
751
752 if (z->permuted >> vx & 1) {
753
754 /* this vectored axis is directly indexed by a
755 specified unvectored axis */
756
757 ux = *zd->xmap - 1;
758#if defined(DEBUG)
759 if (ux < 0 || ux >= F90_RANK_G(ud))
760 gathscat_abort(z->what, "invalid unvectored axis");
761#endif
762 if (ux == vx)
763 z->permuted &= ~(1 << vx);
764 } else {
765
766 /* this vectored axis is directly indexed by the
767 corresponding unvectored axis */
768
769 zd->xmap = &id_map[vx];
770 ux = vx;
771 }
772
773 if (u_covers_v)
774 z->aligned_v_u |= I8(__fort_aligned_axes)(vd, vx + 1, ud, ux + 1) << vx;
775 if (v_covers_u)
776 z->aligned_u_v |= I8(__fort_aligned_axes)(ud, ux + 1, vd, vx + 1) << vx;
777 }
778 }
779
780 /* check if any communication is required */
781
782 if (LOCAL_MODE || F90_FLAGS_G(ud) & F90_FLAGS_G(vd) & __LOCAL)
783 z->communicate = z->replicate = different = 0;
784 else {
785
786 /* check if vectored and unvectored arrays are identically
787 replicated over the same processor grid */
788
789 different = (DIST_DIST_TARGET_G(vd) != DIST_DIST_TARGET_G(ud) ||
790 DIST_REPLICATED_G(vd) != DIST_REPLICATED_G(ud));
791
792 /* need to broadcast the result if it is replicated and the
793 source array is not identically replicated over the same
794 processor grid. if the two arrays are identically
795 replicated, then the communication pattern is executed
796 redundantly by the replicants. otherwise the primaries
797 execute the communication and broadcast the result to the
798 replicants. */
799
800 z->replicate = (DIST_REPLICATED_G(rd) && different);
801 if (z->replicate)
802 I8(__fort_describe_replication)(rd, &z->r_repl);
803
804 /* communication occurs if there is a single alignment or
805 scalar subscript in a mapped dimension */
806
807 z->communicate =
808 ((DIST_SINGLE_G(vd) & DIST_MAPPED_G(DIST_ALIGN_TARGET_G(vd))) |
809 (DIST_SINGLE_G(ud) & DIST_MAPPED_G(DIST_ALIGN_TARGET_G(ud)))) != 0;
810
811 /* ... or if any mapped dimension of the vectored array is
812 indirectly indexed */
813
814 z->communicate |= z->indirect & DIST_MAPPED_G(vd);
815
816 /* ... or if any directly indexed dimension of the vectored
817 array is not mutually aligned with the corresponding
818 dimension of the unvectored array */
819
820 z->communicate |= ~(z->indirect | (z->aligned_v_u & z->aligned_u_v) |
821 -1 << F90_RANK_G(vd));
822 }
823
824 /* set up cyclic loops over unvectored section */
825
826 I8(__fort_cycle_bounds)(ud);
827
828 /* estimate the amount of temp space needed */
829
830 for (tempz = 1, i = F90_RANK_G(ud); --i >= 0;) {
831 SET_DIM_PTRS(udd, ud, i);
832 tempz *= (DIST_DPTR_UAB_G(udd) - DIST_DPTR_PO_G(udd)) -
833 (DIST_DPTR_LAB_G(udd) + DIST_DPTR_NO_G(udd)) + 1;
834 }
835
836#if defined(DEBUG)
837 if (__fort_test & DEBUG_SCAT) {
838 printf("%d %s indirect ", lcpu, z->what);
839 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
840 printf(" %d", z->indirect >> vx & 1);
841 printf("\n");
842
843 printf("%d %s permuted ", lcpu, z->what);
844 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
845 printf(" %d", z->permuted >> vx & 1);
846 printf("\n");
847
848 printf("%d %s conform_x_u", lcpu, z->what);
849 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
850 if (z->indirect >> vx & 1)
851 printf(" %d", z->conform_x_u >> vx & 1);
852 else
853 printf(" -");
854 if (F90_TAG_G(md) == __DESC)
855 printf(" %d", z->conform_x_u >> MAXDIMS & 1);
856 printf("\n");
857
858 printf("%d %s aligned_x_u", lcpu, z->what);
859 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
860 if (z->indirect >> vx & 1)
861 printf(" %d", z->aligned_x_u >> vx & 1);
862 else
863 printf(" -");
864 if (F90_TAG_G(md) == __DESC)
865 printf(" %d", z->aligned_x_u >> MAXDIMS & 1);
866 printf("\n");
867
868 printf("%d %s aligned_v_u", lcpu, z->what);
869 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
870 printf(" %d", z->aligned_v_u >> vx & 1);
871 printf("\n");
872
873 printf("%d %s aligned_u_v", lcpu, z->what);
874 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
875 printf(" %d", z->aligned_u_v >> vx & 1);
876 printf("\n");
877
878 printf("%d %s communicate", lcpu, z->what);
879 for (vx = 0; vx < F90_RANK_G(vd); ++vx)
880 printf(" %d", z->communicate >> vx & 1);
881 printf("\n");
882
883 printf("%d %s different %d replicate %d tempz %d\n", lcpu, z->what,
884 different, z->replicate, tempz);
885 }
886 if (z->communicate && LOCAL_MODE)
887 gathscat_abort(z->what, "mapped arguments in LOCAL mode");
888
889 /*
890 * check for a gen_block dimension.
891 * if we have one, then it's possible that we have a 0 lsize
892 * allocated to it. Therefore, we do not want the debug code
893 * aborting if lsize is 0.
894 */
895
896 for (i = 1; i <= F90_RANK_G(ud); ++i) {
897 if (DFMT(ud, i) == DFMT_GEN_BLOCK) {
898 has_gb = 1;
899 break;
900 }
901 }
902
903 if (has_gb && tempz >= 0)
904 goto zero_lsize_ok_for_gen_block;
905
906 if (tempz <= 0 || tempz > F90_LSIZE_G(ud)) {
907 printf("%d: has_gb=%d dfmt=%d tempz=%d lsize=%d\n", lcpu, has_gb,
908 DIST_DFMT_G(ud), tempz, F90_LSIZE_G(ud));
909 gathscat_abort(z->what, "temp estimate wrong #1");
910 }
911
912zero_lsize_ok_for_gen_block:
913
914#endif
915
916 /* group_offset is zero if this processor is the primary of the
917 unvectored array replication group */
918
919 if (DIST_REPLICATED_G(ud)) {
920 I8(__fort_describe_replication)(ud, &u_repl);
921 z->group_offset = lcpu - u_repl.plow;
922 } else
923 z->group_offset = 0;
924
925 z->outgoing = 0;
926
927 if (!(z->communicate | z->replicate)) {
928
929 /* local gather-scatter. allocate buffers for offsets */
930
931 offsetbuf = (int *)__fort_malloc(2 * tempz * sizeof(int) + sizeof(DIST_Desc));
932
933 soff = offsetbuf;
934 goff = offsetbuf + tempz;
935
936 z->roff = roff = soff;
937 z->loff = loff = goff;
938 z->head = z->next = NULL;
939 z->counts = counts = countr = countbuf = NULL;
940
941 if (~F90_FLAGS_G(ud) & __OFF_TEMPLATE &&
942 (z->group_offset == 0 || !different)) {
943
944 /* this processor is the primary of the replication group,
945 or the source and result are identically replicated --
946 loop over all local unvectored elements to determine
947 gather-scatter offsets */
948
949 if (F90_TAG_G(md) == __DESC)
950 I8(gathscat_mask_loop)(z, uoff, xoff, F90_RANK_G(ud));
951 else
952 I8(gathscat_loop)(z, uoff, xoff, F90_RANK_G(ud));
953
954#if defined(DEBUG)
955 if (z->outgoing > tempz)
956 gathscat_abort(z->what, "temp estimate wrong #2");
957#endif
958 }
959 lclcnt = z->outgoing;
960 maxcnt = 0;
961 } else {
962
963 /* non-local gather-scatter. allocate buffers for offsets and
964 linked list for aggregating by target processor */
965
966 tcpus = GET_DIST_TCPUS;
967 head = (int *)__fort_malloc((tcpus + 3 * tempz) * sizeof(int));
968 next = head + tcpus;
969 roff = next + tempz;
970 loff = roff + tempz;
971
972 for (i = tcpus; --i >= 0;)
973 head[i] = 0;
974
975 z->head = head;
976 z->next = next;
977 z->roff = roff;
978 z->loff = loff;
979
980 countbuf = (int *)__fort_gcalloc(tcpus + __fort_np2, sizeof(int));
981 counts = countbuf;
982 countr = countbuf + tcpus;
983
984 z->counts = counts;
985
986 if (~F90_FLAGS_G(ud) & __OFF_TEMPLATE &&
987 (z->group_offset == 0 || !different)) {
988
989 /* this processor is the primary of the replication group,
990 or the source and result are identically replicated --
991 loop over all local unvectored elements to determine
992 local offset, target cpu, and remote offset */
993
994 if (F90_TAG_G(md) == __DESC)
995 I8(gathscat_mask_loop)(z, uoff, xoff, F90_RANK_G(ud));
996 else
997 I8(gathscat_loop)(z, uoff, xoff, F90_RANK_G(ud));
998
999#if defined(DEBUG)
1000 if (z->outgoing > tempz)
1001 gathscat_abort(z->what, "temp estimate wrong #3");
1002#endif
1003
1004 for (i = tcpus; --i >= 0;)
1005 countr[i] = counts[i];
1006 }
1007
1008 /* exchange counts */
1009
1010 maxcnt = __fort_exchange_counts(countr);
1011
1012 incoming = 0;
1013 for (i = tcpus; --i >= 0;)
1014 incoming += countr[i];
1015
1016 /* head = head of linked list per target processor */
1017 /* next = pointers to next in linked list */
1018 /* roff = unaggregated offsets in remote vectored array */
1019 /* loff = unaggregated offsets in local unvectored array */
1020 /* counts = send element count per target processor */
1021 /* countr = receive element count per target processor */
1022 /* maxcnt = max send/receive count between any processor pair */
1023 /* outgoing = total number of outgoing elements */
1024 /* incoming = total number of incoming elements */
1025
1026 /* allocate buffers for aggregating incoming and outgoing
1027 offsets */
1028
1029 offr = (int *)__fort_gmalloc(2 * maxcnt * sizeof(int));
1030 offs = offr + maxcnt;
1031
1032 /* allocate buffers for aggregated local offsets */
1033
1034 offsetbuf = (int *)__fort_malloc((incoming + z->outgoing) * sizeof(int));
1035 soff = offsetbuf;
1036 goff = offsetbuf + incoming;
1037
1038 lclcnt = 0;
1039 for (i = head[lcpu]; i > 0; i = next[i - 1]) {
1040 goff[lclcnt] = z->loff[i - 1];
1041 soff[lclcnt] = z->roff[i - 1];
1042 ++lclcnt;
1043 }
1044
1045#if defined(DEBUG)
1046 if (lclcnt != counts[lcpu] || lclcnt != countr[lcpu])
1047 gathscat_abort(z->what, "local counts wrong");
1048#endif
1049
1050 /* exchange targets are chosen by xor'ing this processor number
1051 with all masks in the range 1 .. 2**log2(__fort_tcpus). Target
1052 values >= tcpus are skipped (which occurs only if tcpus is not
1053 a power of two.) */
1054
1055 j = k = lclcnt;
1056
1057 for (m = 1; m < __fort_np2; ++m) {
1058
1059 /* identify target */
1060
1061 cpu = lcpu ^ m;
1062 if (cpu >= tcpus)
1063 continue;
1064
1065 /* aggregate local and remote offsets for outgoing
1066 elements */
1067
1068 for (i = head[cpu], n = 0; i > 0; i = next[i - 1]) {
1069 goff[j++] = z->loff[i - 1];
1070 offs[n++] = z->roff[i - 1];
1071 }
1072
1073 nr = countr[cpu];
1074 ns = counts[cpu];
1075
1076#if defined(DEBUG)
1077 if (n != ns)
1078 gathscat_abort(z->what, "aggregation counts wrong #1");
1079#endif
1080
1081 /* exchange offsets */
1082
1083 if (cpu < lcpu) {
1084 if (nr > 0)
1085 __fort_rrecvl(cpu, offr, nr, 1, __CINT, sizeof(int));
1086 if (ns > 0)
1087 __fort_rsendl(cpu, offs, ns, 1, __CINT, sizeof(int));
1088 } else {
1089 if (ns > 0)
1090 __fort_rsendl(cpu, offs, ns, 1, __CINT, sizeof(int));
1091 if (nr > 0)
1092 __fort_rrecvl(cpu, offr, nr, 1, __CINT, sizeof(int));
1093 }
1094
1095 /* aggregate offsets for incoming elements. copying is
1096 required because we need to receive into a gmalloc'd
1097 buffer of equal size (maxcnt) on all processors */
1098
1099 for (i = 0; i < nr; ++i)
1100 soff[k++] = offr[i];
1101 }
1102
1103#if defined(DEBUG)
1104 if (j != z->outgoing || k != incoming)
1105 gathscat_abort(z->what, "aggregation counts wrong #2");
1106#endif
1107
1108 __fort_gfree(offr);
1109 __fort_free(head);
1110 }
1111
1112 if (z->replicate && ~F90_FLAGS_G(rd) & __OFF_TEMPLATE) {
1113
1114/* result elements are local and replicated, but the source
1115 array is not identically replicated -- broadcast the result */
1116
1117#if defined(DEBUG)
1118 if (__fort_test & DEBUG_SCAT) {
1119 printf("%d %s repl p0 %d pn ", lcpu, z->what, z->r_repl.plow);
1120 I8(__fort_show_index)(z->r_repl.ndim, z->r_repl.pcnt);
1121 printf(" ps ");
1122 I8(__fort_show_index)(z->r_repl.ndim, z->r_repl.pstr);
1123 printf("\n");
1124 }
1125#endif
1126 repchn =
1127 __fort_chn_1toN(NULL, z->r_repl.ndim, z->r_repl.plow, z->r_repl.pcnt,
1128 z->r_repl.pstr, 0, z->r_repl.plow, NULL, NULL);
1129 rp = z->rb + DIST_SCOFF_G(rd) * F90_LEN_G(rd);
1130 if (z->r_repl.plow == lcpu)
1131 __fort_sendl(repchn, 0, rp, F90_LSIZE_G(rd), 1, F90_KIND_G(rd),
1132 F90_LEN_G(rd));
1133 __fort_recvl(repchn, 0, rp, F90_LSIZE_G(rd), 1, F90_KIND_G(rd),
1134 F90_LEN_G(rd));
1135 __fort_chn_prune(repchn);
1136 __fort_setbase(repchn, rp, rp, F90_KIND_G(rd), F90_LEN_G(rd));
1137 } else
1138 repchn = NULL;
1139
1140 /* create a new gather-scatter schedule */
1141
1142 sk = (gathscat_sked *)__fort_malloc(sizeof(gathscat_sked));
1143 sk->sked.tag = __SKED;
1144 sk->sked.start = I8(gathscat_start);
1145 sk->sked.free = gathscat_free;
1146 sk->sked.arg = sk;
1147 sk->what = z->what;
1148 sk->gathscatfn = z->gathscatfn;
1149 sk->scatterfn = z->scatterfn;
1150 sk->repchn = repchn;
1151 sk->lclcnt = lclcnt;
1152 sk->maxcnt = maxcnt;
1153 sk->countbuf = countbuf;
1154 sk->offsetbuf = offsetbuf;
1155 if (z->dir == __GATHER) { /* swap */
1156 sk->counts = countr;
1157 sk->countr = counts;
1158 sk->goff = soff;
1159 sk->soff = goff;
1160 } else {
1161 sk->counts = counts;
1162 sk->countr = countr;
1163 sk->goff = goff;
1164 sk->soff = soff;
1165 }
1166
1167 if (__fort_test & DEBUG_TIME) {
1168 t = __fort_second() - t;
1169 printf("%d %s schedule %.6f\n", lcpu, sk->what, t);
1170 }
1171 return &sk->sked;
1172}

1174void *I8(__fort_adjust_index_array)(const char *what, char *idx_array,
1175 char *src, int dim, F90_Desc *is,
1176 F90_Desc *bs)
1177{
1178
1179 /* Adjust Index array for scatter routines. This needs to be called
1180 * when the BASE array argument's lower bound is not 1. We need to
1181 * then adjust the index array by adding LBASE(BASE, dim) - 1 to each
1182 * element in array.
1183 *
1184 *
1185 *
1186 * See HPF 2.0 Spec (6.4.4 Array Combining Scatter Functions)
1187 * page 93-94 for details.
1188 *
1189 *
1190 * Arguments:
1191 * what = type of scatter routine
1192 * idx_array = rslt of adjusted array here. If this is NULL, we
1193 * allocate an global array ...
1194 * src = if idx_array != NULL, this should be same as idx_array.
1195 * dim = dimension of index array (for use by is)
1196 * is = index array descriptor
1197 * bs = BASE array (argument in SCATTER routine) descriptor
1198 *
1199 */
1200
1201 __INT_T adj, i;
1202
1203#if defined(DEBUG)
1204
1205 if (ISSCALAR(is)) {
1206 printf("%d %s: index argument must be an array\n", GET_DIST_LCPU, what);
1207 __fort_abort((char *)0);
1208 }
1209
1210 if (idx_array && idx_array != src) {
1211 printf("%d %s: index_array address must be NULL or == src address\n",
1212 GET_DIST_LCPU, what);
1213 __fort_abort((char *)0);
1214 }
1215
1216#endif
1217
1218 if (idx_array == NULL) {
1219 idx_array = __fort_gmalloc(F90_GSIZE_G(is) * F90_LEN_G(is));
1220 }
1221
1222 adj = F90_DIM_LBOUND_G(bs, dim) - 1;
1223
1224 switch (F90_KIND_G(is)) {
1225
1226 case __INT1:
1227 for (i = 0; i < F90_LSIZE_G(is); ++i)
1228 *((__INT1_T *)idx_array + i) = *((__INT1_T *)src + i) + adj;
1229 break;
1230 case __INT2:
1231 for (i = 0; i < F90_LSIZE_G(is); ++i)
1232 *((__INT2_T *)idx_array + i) = *((__INT2_T *)src + i) + adj;
1233 break;
1234
