46/*
47 * depth-first numbering of loops.
48 * loops are numbered in Depth-first order, children numbered first
49 * +---- loop 1
50 * |+--- loop 2
51 * ||
52 * |+---
53 * |+--- loop 3
54 * ||+-- loop 4
55 * |||
56 * ||+--
57 * |+---
58 * +----
59 * loops are ordered: 2,4,3,1
60 * numbers are:
61 * loop dfn childdfn
62 * 1 4 1
63 * 2 1 1
64 * 3 3 2
65 * 4 2 2
66 * The DFN is the loop's position in the loop ordering
67 * ChildDFN is the DFN of the first descendant in the loop ordering
68 * loop l1 is a descendant of loop l2 if dfn(l1) >= dfn(l1) and childdfn(l1) <= dfn(l2)
69 */
70
71
72typedef struct {
73 int dfn, childdfn;
74 }lpinfo_t;
75STG_DECLARE(lpinfo, lpinfo_t);
76#define LP_DFN(l) lpinfo.stg_base[l].dfn
77#define LP_CHILDDFN(l) lpinfo.stg_base[l].childdfn

102 * return '1' if ast 'ast' is marked as modified in loop 'l', or
103 * in any loop contained inside of 'l'

117 } else {
118 /* just compare numbers */
119 if (m == l) {
120 return 1;
121 }
122 }
123 return 0;
124} /* modified */

155/*
156 * findssexprs is called several times with a 'type' parameter
157 * type == SS_ALL means put all subscript expressions and subexpressions
158 * on the SSlist.
159 * type == SS_ROOT means put all 'root' subscript expressions on the SSlist,
160 * that is, all expressions which are directly used, or for which its
161 * parent is not an optimizable subscript subexpression.
162 * type == SS_INV_ROOT means put all invariant 'root' subexpressions on the
163 * SSlist
164 * type == SS_ALL_MULTIPLE means the same as type == SS_ALL, but set A_MOREG if
165 * it
166 * it put on the list more than once
167 * type == SS_INV_ROOT_MULTIPLE means put all invariant 'root' subexpressions on
168 * the
169 * SSlist that have at least 2 uses, found by A_MOREG != 0.
170 */

172static void
173addSS(int ss, int type, int l)
174{
175 if (A_TYPEG(ss) == A_CONV )
176 ss = A_LOPG(ss);
177 switch (A_TYPEG(ss)) {
178 case A_ID:
179 if (!(type & SS_ALL))
180 return;
181 break;
182 case A_CNST:
183 return;
184 }
185 if (A_OPT1G(ss) != 0) {
186 /* already on the list */
187 if (type == SS_ALL_MULTIPLE) {
188 A_MOREP(ss, 1);
189 }
190 return;
191 }
192#if DEBUG
193 if (DBGBIT(39, 1)) {
194 fprintf(gbl.dbgfil, "Adding subscript expression %d to list type %d for "
195 "loop/node %d, opt2=%d",
196 ss, type, l, A_OPT2G(ss));
197 if (DBGBIT(39, 4)) {
198 fprintf(gbl.dbgfil, " ");
199 printast(ss);
200 }
201 fprintf(gbl.dbgfil, "\n");
202 }
203#endif
204 if (SSlisttail == 1) {
205 SSlisthead = ss;
206 } else {
207 A_OPT1P(SSlisttail, ss);
208 }
209 SSlisttail = ss;
210 A_OPT1P(ss, 1);
211 if (type == SS_ALL_MULTIPLE) {
212 A_MOREP(ss, 0); /* first time on list */
213 }
214} /* addSS */

216/*
217 * puts subexpressions on SSlist
218 * return '0' if there is a subexpression that should not be added to the SSlist
219 * return '1 'otherwise
220 */
221static int
222markss(int ast, int type, int l)
223{
224 int lop, rop, sptr, ssflags;
225
226 switch (A_TYPEG(ast)) {
227 default: /* only (), binop, unop, ID, const, MEM,
228 * section descriptor references are optimizable */
229 return 0;
230 case A_PAREN:
231 case A_UNOP:
232 case A_CONV:
233 lop = markss(A_LOPG(ast), type, l);
234 if (!lop)
235 return 0;
236 if (lop) {
237 if (type & SS_ALL)
238 addSS(A_LOPG(ast), type, l);
239 if (type != SS_INV_ROOT_MULTIPLE || A_MOREG(ast))
240 return 1;
241 }
242 break;
243 case A_BINOP:
244 lop = markss(A_LOPG(ast), type, l);
245 rop = markss(A_ROPG(ast), type, l);
246 if (lop && rop && (type != SS_INV_ROOT_MULTIPLE || A_MOREG(ast))) {
247 if (type & SS_ALL) {
248 addSS(A_LOPG(ast), type, l);
249 addSS(A_ROPG(ast), type, l);
250 }
251 return 1;
252 }
253 if (lop)
254 addSS(A_LOPG(ast), type, l);
255 if (rop)
256 addSS(A_ROPG(ast), type, l);
257 break;
258 case A_ID:
259 if (((type & SS_INV_ROOT) == SS_INV_ROOT) && modified(ast, l))
260 return 0;
261 if (type != SS_INV_ROOT_MULTIPLE || A_MOREG(ast))
262 return 1;
263 break;
264 case A_SUBSCR:
265 if (((type & SS_INV_ROOT) == SS_INV_ROOT) && modified(ast, l))
266 return 0;
267 if (type == SS_INV_ROOT_MULTIPLE && A_MOREG(ast) == 0)
268 return 0;
269 sptr = 0;
270 if (!XBIT(70, 0x8000000)) {
271 sptr = sptr_of_subscript(ast);
272 if (!POINTERG(sptr))
273 sptr = 0;
274 }
275 lop = markss(A_LOPG(ast), type, l);
276 if (lop) {
277 int asd, s, all, ss;
278 asd = A_ASDG(ast);
279 all = 1;
280 ssflags = 0;
281 for (s = 0; s < ASD_NDIM(asd); ++s) {
282 ss = ASD_SUBS(asd, s);
283 if (markss(ss, type, l)) {
284 ssflags |= (1 << s);
285 if (type == SS_ALL && !sptr) {
286 addSS(ss, type, l);
287 }
288 } else {
289 all = 0;
290 }
291 }
292 if (!sptr) {
293 if (all) {
294 if (type & SS_ALL) {
295 addSS(A_LOPG(ast), type, l);
296 }
297 return 1;
298 } else if (ssflags) {
299 for (s = 0; s < ASD_NDIM(asd); ++s) {
300 ss = ASD_SUBS(asd, s);
301 if (ssflags & (1 << s)) {
302 addSS(ss, type, l);
303 }
304 }
305 }
306 }
307 }
308 /* ### probably needs some work for section descriptor members */
309 break;
310 case A_CNST:
311 return 1; /* ok as subexpression */
312 case A_MEM:
313 /* allow these */
314 if (((type & SS_INV_ROOT) == SS_INV_ROOT) && modified(ast, l))
315 return 0;
316 if (type == SS_INV_ROOT_MULTIPLE && A_MOREG(ast) == 0)
317 return 0;
318 lop = markss(A_PARENTG(ast), type, l);
319 if (lop)
320 return 1;
321 break;
322 }
323 return 0;
324} /* markss */

326/*
327 * at a root subexpression, recurse to the subexpressions
328 * put eligible subexpressions on the SSlist
329 */
330static int
331markrootss(int root, int type, int l)
332{
333 if (markss(root, type, l)) {
334 int lop, rop;
335 /* for a root subexpression, don't put on the list if
336 * it is a simple unary op on a constant, or a binary
337 * op where one (or both) operands are constants */
338 switch (A_TYPEG(root)) {
339 case A_UNOP:
340 lop = A_LOPG(root);
341 if (A_TYPEG(lop) == A_CNST)
342 return 0;
343 break;
344 case A_BINOP:
345 lop = A_LOPG(root);
346 rop = A_ROPG(root);
347 if (A_TYPEG(lop) == A_CNST) {
348 if (A_TYPEG(rop) != A_CNST && type == SS_INV_ROOT_MULTIPLE) {
349 addSS(rop, type, l);
350 }
351 return 0;
352 } else if (A_TYPEG(rop) == A_CNST) {
353 if (type == SS_INV_ROOT_MULTIPLE) {
354 addSS(lop, type, l);
355 }
356 return 0;
357 }
358 break;
359 }
360 addSS(root, type, l);
361 return 1;
362 }
363 return 0;
364} /* markrootss */

384/*
385 * at an AST, look for a subscript expression.
386 * if we are in a subscript, add subexpressions to SSlist
387 */
388static void
389findss(int ast, int *dummy)
390{
391 if (A_TYPEG(ast) == A_SUBSCR) {
392 if (subscrnest == 1) {
393 int asd, s;
394 asd = A_ASDG(ast);
395 for (s = 0; s < ASD_NDIM(asd); ++s) {
396 int ss;
397 ss = ASD_SUBS(asd, s);
398 /* don't mark simple subscripts */
399 switch (A_TYPEG(ss)) {
400 case A_PAREN:
401 case A_UNOP:
402 case A_CONV:
403 case A_BINOP:
404 case A_SUBSCR:
405 case A_MEM:
406 markrootss(ss, findsstype, findssl);
407 }
408 }
409 }
410 --subscrnest;
411 }
412} /* findss */

414/*
415 * number any expressions in subscripts in a single fg node
416 */
417static void
418findssexprsfg(int l, int fg, int newstmts, int type)
419{
420 int std;
421 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
422 int ast;
423 ast = STD_AST(std);
424 Trace(("statement %d in node %d in loop %d type %d", std, fg, l, type));
425 if (std >= newstmts && A_TYPEG(ast) == A_ASN) {
426 if (!markrootss(A_SRCG(ast), type, l)) {
427 ast_traverse(ast, count_nest, findss, NULL);
428 }
429 } else {
430 ast_traverse(ast, count_nest, findss, NULL);
431 }
432 if (std == FG_STDLAST(fg))
433 break; /* leave loop */
434 }
435} /* findssexprsfg */

451/*
452 * find and number any expressions in subscripts in a single loop
453 */
454static void
455findssexprs(int l, int newstmts, int type, int onlyfg, int parlevel)
456{
457 /* build a list starting at SSlisthead, linked by A_OPT1 fields,
458 * terminated by '1', of optimizable expressions that appear in
459 * subscripts in loops. We terminate by '1' so we can easily tell
460 * whether an ast is already on the list (A_OPT1(ast) != 0 means on
461 * the list somewhere, A_OPT1(ast) == 0 means not on the list) */
462 clearSSlist();
463 Trace(("findssexprs, loop %d, type %d, parlevel %d", l, type, parlevel));
464 ast_visit(1, 1);
465 findsstype = type;
466 findssl = l;
467 if (onlyfg) {
468 findssexprsfg(l, onlyfg, newstmts, type);
469 } else {
470 int fg;
471 for (fg = LP_FG(l); fg; fg = FG_NEXT(fg)) {
472 if (FG_PAR(fg) == parlevel)
473 findssexprsfg(l, fg, newstmts, type);
474 }
475 }
476 ast_unvisit();
477
478#if DEBUG
479 if (DBGBIT(39, 1))
480 dumpSSlist();
481#endif
482} /* findssexprs */

487/*
488 * mark ast 'ast' as modified in loop 'l'
489 */
490static void
491mark(int ast, int l)
492{
493 int a, aa, sptr, dtype;
494 a = ast;
495 while (a > 0) {
496 aa = a;
497 switch (A_TYPEG(a)) {
498 case A_ID:
499 A_OPT2P(a, l);
500 sptr = A_SPTRG(a);
501 if (TARGETG(sptr) || F90POINTERG(sptr)) {
502 dtype = DTYPEG(sptr);
503 dtype = DDTG(dtype);
504 DtypeAssigned(dtype) = l;
505 }
506 a = 0;
507 break;
508 case A_SUBSCR:
509 A_OPT2P(a, l);
510 a = A_LOPG(a);
511 break;
512 case A_SUBSTR:
513 A_OPT2P(a, l);
514 a = A_LOPG(a);
515 break;
516 case A_MEM:
517 A_OPT2P(a, l);
518 A_OPT2P(A_MEMG(a), l);
519 sptr = A_SPTRG(A_MEMG(a));
520 if (TARGETG(sptr) || F90POINTERG(sptr)) {
521 dtype = DTYPEG(sptr);
522 dtype = DDTG(dtype);
523 DtypeAssigned(dtype) = l;
524 }
525 a = A_PARENTG(a);
526 break;
527 case A_CNST:
528 aa = 0;
529 a = 0;
530 break;
531 case A_BINOP:
532 mark(A_LOPG(a), l);
533 mark(A_ROPG(a), l);
534 a = 0;
535 aa = 0;
536 break;
537 default:
538 interr("redundss:mark unexpected AST type", A_TYPEG(a), 3);
539 a = 0;
540 aa = 0;
541 break;
542 }
543#if DEBUG
544 if (DBGBIT(39, 2) && aa) {
545 fprintf(gbl.dbgfil, "Modified ast %d = ", aa);
546 printast(aa);
547 fprintf(gbl.dbgfil, "\n");
548 }
549#endif
550 }
551} /* mark */

553static void
554markmodified(int ast, int *pl)
555{
556 int argt, argcnt, arg, a, astli, l, sptr, dtype;
557 int subr, subrast, paramct, dpdsc, templatecall;
558 l = *pl;
559 switch (A_TYPEG(ast)) {
560 /* for statement ASTs and function calls, mark modified variables */
561 case A_DO: /* modifies DO variable */
562 case A_MP_PDO: /* modifies DO variable */
563 mark(A_DOVARG(ast), l);
564 break;
565 case A_ASN: /* modifies LHS variable */
566 mark(A_DESTG(ast), l);
567 break;
568 case A_FUNC: /* potentially modifies arguments */
569 case A_CALL: /* potentially modifies arguments */
570 argt = A_ARGSG(ast);
571 argcnt = A_ARGCNTG(ast);
572 subrast = A_LOPG(ast);
573 if (A_TYPEG(subrast) == A_ID) {
574 subr = A_SPTRG(subrast);
575 paramct = PARAMCTG(subr);
576 dpdsc = DPDSCG(subr);
577 } else {
578 subr = -1;
579 paramct = -1;
580 dpdsc = -1;
581 }
582 templatecall = 0;
583 if (A_TYPEG(ast) == A_CALL && subr > 0 &&
584 getF90TmplSectRtn(SYMNAME(subr))) {
585 templatecall = 1;
586 }
587 for (a = 0; a < argcnt; ++a) {
588 int asym;
589 arg = ARGT_ARG(argt, a);
590 /* don't count descriptor arguments as modified
591 * if they need to be so marked, they will be at the array
592 * argument to which they correspond */
593 asym = -1;
594 if (a < paramct) {
595 asym = aux.dpdsc_base[dpdsc + a];
596 }
597 switch (A_TYPEG(arg)) {
598 case A_ID:
599 case A_MEM:
600 sptr = memsym_of_ast(arg);
601 if (!DESCARRAYG(sptr) || (templatecall && a == 0) ||
602 (!templatecall && dpdsc == 0)) {
603 /* regular symbol, or 1st argument to template call,
604 * or call to a function without an interface */
605 mark(arg, l);
606 }
607 if (asym > NOSYM && POINTERG(asym) && SDSCG(sptr)) {
608 /* there is an explicit interface
609 * the corresponding dummy is a pointer dummy
610 * there is a section descriptor */
611 mark(mk_id(SDSCG(sptr)), l);
612 }
613 break;
614 case A_SUBSTR:
615 mark(arg, l);
616 break;
617 case A_SUBSCR:
618 sptr = memsym_of_ast(arg);
619 if (!DESCARRAYG(sptr)) {
620 mark(arg, l);
621 }
622 break;
623 }
624 }
625 a = A_LOPG(ast);
626 if (A_TYPEG(a) != A_ID || !PUREG(A_SPTRG(a))) {
627 CallAppears = l;
628 }
629 break;
630 case A_ICALL: /* potentially modifies first argument */
631 argcnt = A_ARGCNTG(ast);
632 argt = A_ARGSG(ast);
633 arg = ARGT_ARG(argt, 0);
634 switch (A_TYPEG(arg)) {
635 case A_ID:
636 case A_MEM:
637 mark(arg, l);
638 }
639 switch (A_OPTYPEG(ast)) {
640 case I_PTR2_ASSIGN:
641 /* get datatype of first argument */
642 dtype = A_DTYPEG(arg);
643 DtypeAssigned(dtype) = l;
644 /* modifies the section descriptor also */
645 mark(ARGT_ARG(argt, 1), l);
646 break;
647
648 case I_NULLIFY:
649 case I_PTR_COPYOUT:
650 if (argcnt > 1) {
651 /* modifies the section descriptor also */
652 mark(ARGT_ARG(argt, 1), l);
653 }
654 break;
655
656 case I_COPYOUT:
657 /* modifies the section descriptor also */
658 mark(ARGT_ARG(argt, 2), l);
659 break;
660
661 case I_COPYIN:
662 /* modifies the section descriptor also */
663 mark(ARGT_ARG(argt, 3), l);
664 break;
665
666 case I_PTR_COPYIN:
667 /* modifies the section descriptor also */
668 mark(ARGT_ARG(argt, 4), l);
669 break;
670 }
671 break;
672 case A_ALLOC: /* modifies allocated object */
673 case A_REDIM: /* modifies redimensioned object */
674 mark(A_SRCG(ast), l);
675 sptr = sym_of_ast(A_SRCG(ast));
676 if (MIDNUMG(sptr)) {
677 mark(mk_id(MIDNUMG(sptr)), l);
678 }
679 if (PTROFFG(sptr)) {
680 mark(mk_id(PTROFFG(sptr)), l);
681 }
682 if (SDSCG(sptr)) {
683 mark(mk_id(SDSCG(sptr)), l);
684 }
685 break;
686 case A_FORALL: /* modified FORALL indices */
687 for (astli = A_LISTG(ast); astli; astli = ASTLI_NEXT(astli)) {
688 mark(mk_id(ASTLI_SPTR(astli)), l);
689 }
690 break;
691 case A_REALIGN: /* modifies realigned object */
692 case A_REDISTRIBUTE: /* modifies redistributed object */
693 mark(A_LOPG(ast), l);
694 sptr = sym_of_ast(A_SRCG(ast));
695 if (MIDNUMG(sptr)) {
696 mark(mk_id(MIDNUMG(sptr)), l);
697 }
698 if (PTROFFG(sptr)) {
699 mark(mk_id(PTROFFG(sptr)), l);
700 }
701 if (SDSCG(sptr)) {
702 mark(mk_id(SDSCG(sptr)), l);
703 }
704 break;
705 }
706} /* markmodified */

738/*
739 * at loop l, find which subexpressions are modified at this loop
740 * level, and mark them modified by setting A_OPT2 to the loop number
741 */
742static void
743markmodifiedexprs(int l, int onlyfg)
744{
745 /* go through the nodes in the loop and statements in the node */
746 /* use ast_traverse to get all function references */
747 ast_visit(1, 1);
748 if (onlyfg) {
749 markmodifiedexprsfg(l, onlyfg);
750 } else {
751 int fg;
752 for (fg = LP_FG(l); fg; fg = FG_NEXT(fg)) {
753 markmodifiedexprsfg(l, fg);
754 }
755 }
756 ast_unvisit_norepl();
757} /* markmodifiedexprs */

759/*
760 * now propagate 'modified' from subexpressions
761 * because of the way the list of subexpressions is ordered,
762 * all subsubexpressions are visited first;
763 * we have to go through ALL subsubexpressions, even if they are not
764 * used in loop l, because they might have been used in a contained loop
765 * of loop l, and we have to mark A_OPT2 with 'l' even in that case.
766 */
767static void
768propagatemodified(int l)
769{
770 int ast, dtype;
771 for (ast = SSlisthead; ast > 1; ast = A_OPT1G(ast)) {
772 if (!modified(ast, l)) {
773 int sptr, asd, s, mem;
774 switch (A_TYPEG(ast)) {
775 case A_PAREN:
776 case A_UNOP:
777 case A_CONV:
778 if (modified(A_LOPG(ast), l)) {
779 A_OPT2P(ast, l);
780 }
781 break;
782 case A_BINOP:
783 if (modified(A_LOPG(ast), l) || modified(A_ROPG(ast), l)) {
784 A_OPT2P(ast, l);
785 }
786 break;
787 case A_ID:
788 /* may be set by markmodified */
789 sptr = A_SPTRG(ast);
790 switch (SCG(sptr)) {
791 case SC_CMBLK:
792 case SC_STATIC:
793 /* call in this loop? */
794 if (inloop(CallAppears, l)) {
795 A_OPT2P(ast, l);
796 }
797 break;
798 default:;
799 }
800 /* if 'target' or 'pointer', and any assignment to pointer of that
801 * datatype in the loop, or any call in the loop,
802 * we must assume it might be modified secretly */
803 if (TARGETG(sptr) || POINTERG(sptr)) {
804 /* call in this loop? */
805 if (inloop(CallAppears, l)) {
806 A_OPT2P(ast, l);
807 }
808 /* assignment to pointer of that datatype? */
809 dtype = DTYPEG(sptr);
810 dtype = DDTG(dtype);
811 if (inloop(DtypeAssigned(dtype), l)) {
812 A_OPT2P(ast, l);
813 }
814 }
815 if (DESCARRAYG(sptr)) {
816 int ast2;
817 ast2 = mk_id(sptr);
818 if (ast2 != ast && modified(ast2, l)) {
819 A_OPT2P(ast, l);
820 }
821 }
822 break;
823 case A_CNST:
824 break;
825 case A_SUBSCR:
826 if (modified(A_LOPG(ast), l)) {
827 A_OPT2P(ast, l);
828 }
829 asd = A_ASDG(ast);
830 for (s = 0; s < ASD_NDIM(asd); ++s) {
831 if (modified(ASD_SUBS(asd, s), l)) {
832 A_OPT2P(ast, l);
833 }
834 }
835 break;
836 case A_MEM:
837 if (modified(A_PARENTG(ast), l)) {
838 A_OPT2P(ast, l);
839 }
840 mem = A_MEMG(ast);
841 if (modified(mem, l)) {
842 A_OPT2P(ast, l);
843 }
844 sptr = A_SPTRG(mem);
845 if (DESCARRAYG(sptr)) {
846 int ast2;
847 ast2 = mk_id(sptr);
848 if (ast2 != ast && modified(ast2, l)) {
849 A_OPT2P(ast, l);
850 }
851 }
852 /* if 'target' or 'pointer', and any assignment to pointer of that
853 * datatype in the loop, or any call in the loop,
854 * we must assume it might be modified secretly */
855 if (TARGETG(sptr) || POINTERG(sptr)) {
856 /* call in this loop? */
857 if (inloop(CallAppears, l)) {
858 A_OPT2P(ast, l);
859 }
860 /* assignment to pointer of that datatype? */
861 if (inloop(DtypeAssigned(DTYPEG(sptr)), l)) {
862 A_OPT2P(ast, l);
863 }
864 }
865 break;
866 default:
867 A_OPT2P(ast, l);
868 break;
869 }
870 }
871 }
872#if DEBUG
873 if (DBGBIT(39, 1))
874 dumpSSlist();
875#endif
876} /* propagatemodified */

878/* lop (+/-) rop, where lop or rop (or both) might be zero */
879static int
880addcomponents(int lop, int optype, int rop, int l)
881{
882 int ast;
883 if (rop == 0) {
884 ast = lop;
885 } else if (lop == 0) {
886 if (optype == OP_ADD) {
887 ast = rop;
888 } else if (optype == OP_SUB) {
889 if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
890 /* -(-rop) == rop */
891 ast = A_LOPG(rop);
892 } else if (A_TYPEG(rop) == A_BINOP && A_OPTYPEG(rop) == OP_SUB) {
893 /* -(rlop-rrop) = rrop-rlop */
894 ast = mk_binop(optype, A_ROPG(rop), A_LOPG(rop), A_DTYPEG(rop));
895 A_OPT2P(ast, A_OPT2G(rop));
896 } else {
897 ast = mk_unop(optype, rop, A_DTYPEG(rop));
898 A_OPT2P(ast, A_OPT2G(rop));
899 }
900 } else {
901 ast = mk_unop(optype, rop, A_DTYPEG(rop));
902 A_OPT2P(ast, A_OPT2G(rop));
903 }
904 } else if (optype == OP_ADD) {
905 /* look for (-a)+(-b), (-a)+b, a+(-b), (-a)+a, a+(-a) */
906 if (A_TYPEG(lop) == A_UNOP && A_OPTYPEG(lop) == OP_SUB) {
907 if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
908 int tmp;
909 /* (-a)+(-b) turns into -(a+b) */
910 lop = A_LOPG(lop);
911 rop = A_LOPG(rop);
912 tmp = mk_binop(OP_ADD, lop, rop, A_DTYPEG(rop));
913 if (!modified(tmp, l)) {
914 if (modified(lop, l) || modified(rop, l)) {
915 A_OPT2P(tmp, l);
916 }
917 }
918 ast = mk_unop(OP_SUB, tmp, A_DTYPEG(tmp));
919 A_OPT2P(ast, A_OPT2G(tmp));
920 } else {
921 /* (-a) + b turns into b-a */
922 lop = A_LOPG(lop);
923 if (lop == rop) {
924 /* -a + a */
925 ast = 0;
926 } else {
927 ast = mk_binop(OP_SUB, rop, lop, A_DTYPEG(rop));
928 if (!modified(ast, l)) {
929 if (modified(lop, l) || modified(rop, l)) {
930 A_OPT2P(ast, l);
931 }
932 }
933 }
934 }
935 } else if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
936 /* a + (-b) turns into a-b */
937 rop = A_LOPG(rop);
938 if (lop == rop) {
939 /* a - a */
940 ast = 0;
941 } else {
942 ast = mk_binop(OP_SUB, lop, rop, A_DTYPEG(rop));
943 if (!modified(ast, l)) {
944 if (modified(lop, l) || modified(rop, l)) {
945 A_OPT2P(ast, l);
946 }
947 }
948 }
949 } else {
950 ast = mk_binop(OP_ADD, lop, rop, A_DTYPEG(rop));
951 if (!modified(ast, l)) {
952 if (modified(lop, l) || modified(rop, l)) {
953 A_OPT2P(ast, l);
954 }
955 }
956 }
957 } else {
958 /* look for (-a)-(-b), (-a)-b, a-(-b), a-a */
959 if (A_TYPEG(lop) == A_UNOP && A_OPTYPEG(lop) == OP_SUB) {
960 if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
961 /* (-a)-(-b) turns into b-a */
962 lop = A_LOPG(lop);
963 rop = A_LOPG(rop);
964 if (lop == rop) {
965 /* (-a) - (-a) */
966 ast = 0;
967 } else {
968 ast = mk_binop(OP_SUB, rop, lop, A_DTYPEG(rop));
969 if (!modified(ast, l)) {
970 if (modified(lop, l) || modified(rop, l)) {
971 A_OPT2P(ast, l);
972 }
973 }
974 }
975 } else {
976 /* (-a) - b turns into -(a+b) */
977 int tmp;
978 lop = A_LOPG(lop);
979 tmp = mk_binop(OP_ADD, rop, lop, A_DTYPEG(rop));
980 if (!modified(tmp, l)) {
981 if (modified(lop, l) || modified(rop, l)) {
982 A_OPT2P(tmp, l);
983 }
984 }
985 ast = mk_unop(OP_SUB, tmp, A_DTYPEG(tmp));
986 A_OPT2P(ast, A_OPT2G(tmp));
987 }
988 } else if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
989 /* a - (-b) turns into a+b */
990 rop = A_LOPG(rop);
991 ast = mk_binop(OP_ADD, lop, rop, A_DTYPEG(rop));
992 if (!modified(ast, l)) {
993 if (modified(lop, l) || modified(rop, l)) {
994 A_OPT2P(ast, l);
995 }
996 }
997 } else {
998 if (lop == rop) {
999 /* a - a */
1000 ast = 0;
1001 } else {
1002 ast = mk_binop(OP_SUB, lop, rop, A_DTYPEG(rop));
1003 if (!modified(ast, l)) {
1004 if (modified(lop, l) || modified(rop, l)) {
1005 A_OPT2P(ast, l);
1006 }
1007 }
1008 }
1009 }
1010 }
1011 return ast;

1035/*
1036 * lop*rop, where lop or rop (or both) might be zero
1037 * loop for (-lop)*(-rop), (-lop)*rop, eliminate negatives
1038 */
1039static int
1040mulcomponents(int lop, int rop, int l)
1041{
1042 int ast, negate;
1043 if (lop == 0 || rop == 0) {
1044 ast = 0;
1045 } else {
1046 negate = 1;
1047 if (A_TYPEG(lop) == A_UNOP && A_OPTYPEG(lop) == OP_SUB) {
1048 negate = -negate;
1049 lop = A_LOPG(lop);
1050 }
1051 if (A_TYPEG(rop) == A_UNOP && A_OPTYPEG(rop) == OP_SUB) {
1052 negate = -negate;
1053 rop = A_LOPG(rop);
1054 }
1055 if (lop == astb.i1) {
1056 ast = rop;
1057 } else if (rop == astb.i1) {
1058 ast = lop;
1059 } else {
1060 ast = mk_binop(OP_MUL, lop, rop, A_DTYPEG(rop));
1061 if (!modified(ast, l)) {
1062 if (modified(lop, l) || modified(rop, l)) {
1063 A_OPT2P(ast, l);
1064 }
1065 }
1066 }
1067 if (negate == -1) {
1068 int nast;
1069 nast = mk_unop(OP_SUB, ast, A_DTYPEG(ast));
1070 if (!modified(nast, l)) {
1071 if (modified(ast, l)) {
1072 A_OPT2P(nast, l);
1073 }
1074 }
1075 ast = nast;
1076 }
1077 }
1078 return ast;
1079} /* mulcomponents */

1081/*
1082 * given an ast, see if it is a product, and if so, build two trees,
1083 * one the product of components that are invariant in loop l, and one
1084 * the product of components that are variant in loop l
1085 */
1086static void
1087findmulcomponents(int ast, int l, int *pinv, int *pvar)
1088{
1089 int linv, lvar, rinv, rvar;
1090 if (ast == 0) {
1091 *pvar = 0;
1092 *pinv = 0;
1093 return;
1094 }
1095 /* default result, unless some other case applies */
1096 if (modified(ast, l) || A_TYPEG(ast) == A_CNST) {
1097 *pvar = ast;
1098 *pinv = astb.i1;
1099 } else {
1100 *pvar = astb.i1;
1101 *pinv = ast;
1102 }
1103 switch (A_TYPEG(ast)) {
1104 case A_UNOP:
1105 switch (A_OPTYPEG(ast)) {
1106 case OP_ADD:
1107 findmulcomponents(A_LOPG(ast), l, pinv, pvar);
1108 break;
1109 case OP_SUB:
1110 findmulcomponents(A_LOPG(ast), l, &linv, &lvar);
1111 /* negate one of them */
1112 *pinv = negatecomponent(linv, l);
1113 *pvar = lvar;
1114 break;
1115 }
1116 break;
1117 case A_BINOP:
1118 switch (A_OPTYPEG(ast)) {
1119 case OP_MUL:
1120 findmulcomponents(A_LOPG(ast), l, &linv, &lvar);
1121 findmulcomponents(A_ROPG(ast), l, &rinv, &rvar);
1122 *pinv = mulcomponents(linv, rinv, l);
1123 *pvar = mulcomponents(lvar, rvar, l);
1124 break;
1125 }
1126 break;
1127 case A_PAREN:
1128 findmulcomponents(A_LOPG(ast), l, pinv, pvar);
1129 break;
1130 }
1131} /* findmulcomponents */

1133/*
1134 * given an ast, see if it is a sum, and if so, build two trees,
1135 * one the sum of components that are invariant in loop l, and one
1136 * the sum of components that are variant in loop l
1137 */
1138static void
1139findcomponents(int ast, int l, int *pinv, int *pvar)
1140{
1141 int linv, lvar, rinv, rvar, optype;
1142 /* default result, unless some other case applies */
1143 if (modified(ast, l) || A_TYPEG(ast) == A_CNST) {
1144 *pvar = ast;
1145 *pinv = 0;
1146 } else {
1147 *pvar = 0;
1148 *pinv = ast;
1149 }
1150 switch (A_TYPEG(ast)) {
1151 case A_UNOP:
1152 switch (A_OPTYPEG(ast)) {
1153 case OP_ADD:
1154 findcomponents(A_LOPG(ast), l, pinv, pvar);
1155 break;
1156 case OP_SUB:
1157 findcomponents(A_LOPG(ast), l, &linv, &lvar);
1158 /* negate both of them */
1159 *pinv = negatecomponent(linv, l);
1160 *pvar = negatecomponent(lvar, l);
1161 break;
1162 }
1163 break;
1164 case A_BINOP:
1165 optype = A_OPTYPEG(ast);
1166 switch (optype) {
1167 case OP_ADD:
1168 case OP_SUB:
1169 findcomponents(A_LOPG(ast), l, &linv, &lvar);
1170 findcomponents(A_ROPG(ast), l, &rinv, &rvar);
1171 *pinv = addcomponents(linv, optype, rinv, l);
1172 *pvar = addcomponents(lvar, optype, rvar, l);
1173 break;
1174 case OP_MUL:
1175 findcomponents(A_LOPG(ast), l, &linv, &lvar);
1176 findcomponents(A_ROPG(ast), l, &rinv, &rvar);
1177 if (lvar == 0) {
1178 int rvarinv, rvarvar;
1179 /* linv*(rinv+(rvarinv*rvarvar)) becomes
1180 * linv*rinv + (linv*rvarinv)*rvarvar */
1181 *pinv = mulcomponents(linv, rinv, l);
1182 findmulcomponents(rvar, l, &rvarinv, &rvarvar);
1183 linv = mulcomponents(linv, rvarinv, l);
1184 *pvar = mulcomponents(linv, rvarvar, l);
1185 } else if (rvar == 0) {
1186 int lvarinv, lvarvar;
1187 /* (linv+(lvarinv*lvarvar))*rinv becomes
1188 * linv*rinv + (lvarinv*rinv)*lvarvar */
1189 *pinv = mulcomponents(linv, rinv, l);
1190 findmulcomponents(lvar, l, &lvarinv, &lvarvar);
1191 rinv = mulcomponents(lvarinv, rinv, l);
1192 *pvar = mulcomponents(lvarvar, rinv, l);
1193 } else {
1194 /* (lvar+linv)*(rinv+rvar) just reassociate */
1195 *pinv = 0;
1196 lvar = addcomponents(lvar, OP_ADD, linv, l);
1197 rvar = addcomponents(rvar, OP_ADD, rinv, l);
1198 *pvar = mulcomponents(lvar, rvar, l);
1199 }
1200 break;
1201 }
1202 break;
1203 case A_PAREN:
1204 findcomponents(A_LOPG(ast), l, pinv, pvar);
1205 break;
1206 }
1207} /* findcomponents */

1209/*
1210 * return TRUE if ast is a sum of variant and invariant parts */
1211static LOGICAL
1212mixedsum(int ast, int l)
1213{
1214 LOGICAL mixed;
1215 int lop, rop, lmod, rmod;
1216 mixed = FALSE;
1217 switch (A_TYPEG(ast)) {
1218 case A_UNOP:
1219 switch (A_OPTYPEG(ast)) {
1220 case OP_ADD:
1221 case OP_SUB:
1222 mixed = mixedsum(A_LOPG(ast), l);
1223 }
1224 break;
1225 case A_BINOP:
1226 switch (A_OPTYPEG(ast)) {

1248/*
1249 * given 'factor' that is invariant in loop 'l',
1250 * and 'sum' that is partially invariant in loop 'l',
1251 * distribute 'factor' over 'sum':
1252 * (factor)*((sumvarinv*sumvarvar)+suminv) becomes
1253 * ((factor)*(sumvarinv))*(sumvarvar) + (factor)*(suminv)
1254 */
1255static int
1256distribute(int factor, int sum, int l)
1257{
1258 int suminv, sumvar, sumvarinv, sumvarvar, ast;
1259 findcomponents(sum, l, &suminv, &sumvar);
1260 suminv = mulcomponents(factor, suminv, l);
1261 findmulcomponents(sumvar, l, &sumvarinv, &sumvarvar);
1262 factor = mulcomponents(factor, sumvarinv, l);
1263 sumvar = mulcomponents(factor, sumvarvar, l);
1264 ast = addcomponents(suminv, OP_ADD, sumvar, l);
1265 return ast;
1266} /* distribute */

1268/*
1269 * return a replacement ast that is reassociated and distributed
1270 * to give better code floating.
1271 * (i*k+1)*2+3 will become (i*(k*2))+((1*2)+3)
1272 * We compute a sum of a loop-invariant expression and a loop-variant one.
1273 * Leave the constant part in the 'loop-variant' part.
1274 */
1275static int
1276MakeReplacement(int ast, int l)
1277{
1278 int optype, rast = 0;
1279 int lop, linv, lvar, rop, rinv, rvar, subscr[MAXSUBS], numdim, asd, s, any;
1280 switch (A_TYPEG(ast)) {
1281 case A_PAREN:
1282 return MakeReplacement(A_LOPG(ast), l);
1283 case A_CONV:
1284 rast = MakeReplacement(A_LOPG(ast), l);
1285 if (rast == A_LOPG(ast)) {
1286 rast = ast;
1287 } else {
1288 rast = mk_convert(rast, A_DTYPEG(ast));
1289 }
1290 break;
1291 case A_UNOP:
1292 switch (A_OPTYPEG(ast)) {
1293 case OP_SUB:
1294 rast = MakeReplacement(A_LOPG(ast), l);
1295 rast = negatecomponent(rast, l);
1296 break;
1297 case OP_ADD:
1298 rast = MakeReplacement(A_LOPG(ast), l);
1299 break;
1300 default:
1301 rast = ast;
1302 break;
1303 }
1304 break;
1305 case A_BINOP:
1306 optype = A_OPTYPEG(ast);
1307 switch (optype) {
1308 case OP_ADD:
1309 case OP_SUB:
1310 /* find 'components' of lop and rop, linv, lvar, rinv, rvar.
1311 * add (linv+rinv)+(lvar+rvar) */
1312 findcomponents(A_LOPG(ast), l, &linv, &lvar);
1313 findcomponents(A_ROPG(ast), l, &rinv, &rvar);
1314 linv = addcomponents(linv, optype, rinv, l);
1315 lvar = addcomponents(lvar, optype, rvar, l);
1316 rast = addcomponents(linv, OP_ADD, lvar, l);
1317 break;
1318 case OP_MUL:
1319 /* multiply is the most difficult case
1320 * ((a+b)*(c+d)) * ((e+f)*(g+h))
1321 * If one side is invariant and the other is variant ADD or SUB,
1322 * but not completely variant, distribute the one side over the
1323 * other.
1324 * If either or both sides are multiplies, find multiplicative
1325 * components and reassociate as for sums */
1326 lop = MakeReplacement(A_LOPG(ast), l);
1327 rop = MakeReplacement(A_ROPG(ast), l);
1328 rast = ast;
1329 if (!modified(lop, l) && mixedsum(rop, l)) {
1330 rast = distribute(lop, rop, l);
1331 } else if (!modified(rop, l) && mixedsum(lop, l)) {
1332 rast = distribute(rop, lop, l);
1333 } else {
1334 findmulcomponents(ast, l, &linv, &lvar);
1335 rast = mulcomponents(linv, lvar, l);
1336 }
1337 break;
1338 default:
1339 rast = ast;
1340 break;
1341 }
1342 break;
1343 case A_SUBSCR:
1344 /* see if any subscripts can be modified */
1345 if (!XBIT(70, 0x8000000)) {
1346 int sptr;
1347 sptr = sptr_of_subscript(ast);
1348 if (!POINTERG(sptr))
1349 break;
1350 }
1351 rast = ast;
1352 asd = A_ASDG(ast);
1353 numdim = ASD_NDIM(asd);
1354 any = 0;
1355 for (s = 0; s < numdim; ++s) {
1356 subscr[s] = MakeReplacement(ASD_SUBS(asd, s), l);
1357 if (subscr[s] != ASD_SUBS(asd, s))
1358 ++any;
1359 }
1360 if (any) {
1361 rast = mk_subscr(A_LOPG(ast), subscr, numdim, A_DTYPEG(ast));
1362 }
1363 break;
1364 default:
1365 rast = ast;
1366 }
1367 return rast;
1368} /* MakeReplacement */

1452/*
1453 * go through loops, innermost first,
1454 * call findssexprs to build SSlist of all subscript subexpressions
1455 * used in the loop.
1456 * call markmodifiedexprs for each loop to mark expressions that
1457 * are directly modified in this loop
1458 * call propagatemodified to mark which of these are modified in the loop.
1459 * call findssexprs to build SSlist of the root subscript subexpressions
1460 * used in the loop.
1461 * for each of these, call MakeReplacement to reassociate and distribute,
1462 * and to find the invariant part of the expression. Insert an assignment
1463 * for the invariant part outside of the loop, or move the assignment if the
1464 * whole expression is invariant.
1465 */
1466static void
1467reassociate(void)
1468{

1470 oldstdavl = astb.std.stg_avail;
1471 for (loop = 1; loop <= opt.nloops; ++loop) {
1472 int l, ast, any;
1473 int fgpre;
1474 l = LP_LOOP(loop);
1475 Trace(("REASSOCIATE loop %d: Find all subscript expressions", l));
1476 findssexprs(l, oldstdavl, SS_ALL, 0, LP_PARLOOP(l));
1477 Trace(("REASSOCIATE loop %d: Mark modified expressions", l));
1478 markmodifiedexprs(l, 0);
1479 Trace(("REASSOCIATE loop %d: propagate modified expressions", l));
1480 propagatemodified(l);
1481 Trace(("REASSOCIATE loop %d: find root subscript expressions", l));
1482 findssexprs(l, oldstdavl, SS_ROOT, 0, LP_PARLOOP(l));
1483 /* go through each root expression used in this loop.
1484 * reassociate and distribute for maximum code floating */
1485 any = 0;
1486 ast_visit(1, 1);
1487 for (ast = SSlisthead; ast > 1; ast = A_OPT1G(ast)) {
1488 /* ast is used in this loop */

1490 rast = MakeReplacement(ast, l);
1491 if (rast != ast) {

1493 ++any;

1500 printast(rast);
1501 fprintf(gbl.dbgfil ? gbl.dbgfil : stderr, "\n");
1502 }
1503#endif
1504 ast_replace(ast, rast);
1505 }
1506 }
1507 if (!any) {
1508 ast_unvisit();
1509 } else {
1510 int fg;
1511 for (fg = LP_FG(l); fg; fg = FG_NEXT(fg)) {
1512 int std;
1513 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
1514 int ast, rast;

1516 rast = ast_rewrite(ast);

1519 std, ast, rast));

1523 fprintf(gbl.dbgfil ? gbl.dbgfil : stderr, " ==> ");
1524 printast(rast);
1525 fprintf(gbl.dbgfil ? gbl.dbgfil : stderr, "\n");
1526 }
1527#endif
1528 STD_AST(std) = rast;
1529 }
1530 if (std == FG_STDLAST(fg))
1531 break; /* leave loop */
1532 }
1533 }
1534 ast_unvisit();
1535 /* now find subscript expressions after changes */
1536 Trace((
1537 "REASSOCIATE loop %d: find subscript expressions after reassociation",
1538 l));
1539 findssexprs(l, oldstdavl, SS_ALL, 0, LP_PARLOOP(l));
1540 Trace(("REASSOCIATE loop %d: mark modified subexpressions after "
1541 "reassociation",
1542 l));
1543 markmodifiedexprs(l, 0);
1544 Trace(("REASSOCIATE loop %d: propagate modified subexpressions after "
1545 "reassociation",
1546 l));
1547 propagatemodified(l);
1548 }
1549 Trace(
1550 ("REASSOCIATE loop %d: find invariant root subscript expressions", l));
1551 findssexprs(l, oldstdavl, SS_INV_ROOT, 0, LP_PARLOOP(l));
1552 /* make preheader flow graph node for loop l, where to insert code */

1555 FG_PAR(opt.pre_fg) = FG_PAR(fgpre);

1557 ast_visit(1, 1);
1558 any = addssassignments(0, opt.pre_fg, 0, STD_PAR(FG_STDFIRST(LP_HEAD(l))),
1559 STD_TASK(FG_STDFIRST(LP_HEAD(l))));
1560 wrilts(opt.pre_fg);
1561 if (any) {
1562 int fg;
