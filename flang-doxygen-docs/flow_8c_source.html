180/* visit is in postorder */
181static void
182_cr_nme(int ast, int *dummy)
183{
184 int sym;
185 int nme;
186 int asd;
187 int s;
188 int astli;
189 int sd, sda;
190
191 switch (A_TYPEG(ast)) {
192 case A_ID:
193 sym = A_SPTRG(ast);
194 for (s = sym; SCG(s) == SC_BASED;) {
195 int n, a;
196 s = MIDNUMG(s);
197 if (s > NOSYM && ST_ISVAR(STYPEG(s))) {
198 n = add_arrnme(NT_VAR, s, 0, (INT)0, 0, FALSE);
199 a = mk_id(s);
200 A_NMEP(a, n);
201 } else
202 break;
203 }
204 if (ST_ISVAR(STYPEG(sym))) {
205 sd = SDSCG(sym);
206 if (sd) {
207 /* FS#19312: Needed for source= in sourced allocaton of bld_ud() below.
208 * \
209 * Shouldn't we always do this for descriptors? \
210 */
211 _cr_nme(mk_id(sd), dummy);
212 }
213 nme = add_arrnme(NT_VAR, sym, 0, (INT)0, 0, FALSE);
214 } else {
215 nme = NME_UNK; /* NME_UNK == 0 */
216 }
217 A_NMEP(ast, nme);
218 break;
219 case A_MEM:
220 if (A_TYPEG(A_PARENTG(ast)) == A_FUNC) {
221 nme = A_NMEG(A_LOPG(A_PARENTG(ast)));
222 } else {
223 nme = A_NMEG(A_PARENTG(ast));
224 }
225 sym = A_SPTRG(A_MEMG(ast));
226 sd = SDSCG(sym);
227 if (sd) {
228 sda = check_member(ast, mk_id(sd));
229 _cr_nme(sda, dummy);
230 }
231 nme = add_arrnme(NT_MEM, (int)PSMEMG(sym), nme, (INT)0, 0, FALSE);
232 A_NMEP(ast, nme);
233 break;
234 case A_SUBSCR:
235 nme = A_NMEG(A_LOPG(ast));
236 asd = A_ASDG(ast);
237 nme = add_arrnme(NT_ARR, -1, nme, (INT)0, asd, 0);
238 A_NMEP(ast, nme);
239 break;
240 case A_FORALL:
241 for (astli = A_LISTG(ast); astli; astli = ASTLI_NEXT(astli)) {
242 int aa;
243 sym = ASTLI_SPTR(astli);
244 aa = mk_id(sym);
245 _cr_nme(aa, dummy);
246 }
247 break;
248 default:
249 break;
250 }
251}

255void
256flow(void)
257{
258
259 flow_init();
260
261 create_nmes();
262
263 dilt_l = NULL;
264 local_again = FALSE; /* first call to localflow */
265 do_lv = TRUE;
266 do_unv = TRUE;
267
268 live_var_init();
269
270 localflow();
271
272 live_var();
273
274 live_var_end();
275
276 if (flg.opt >= 2 && !XBIT(2, 0x400000)) {
277 /* live_var_init and localflow must be done first */
278 do_lv = FALSE;
279 uninit_var_init();
280 uninit_var();
281 uninit_var_end();
282 do_lv = TRUE;
283 }
284
285 reaching_defs();
286 du_ud();
287
288again:
289 jumps_deleted = FALSE;
290 if (flg.opt >= 2 && const_prop()) {
291 int i;
292
293 for (i = 1; i <= opt.num_nodes; i++) {
294 FG_ALL(i) = 0;
295 FG_FDEF(i) = 0;
296 FG_OUT(i) = FG_IN(i) = 0;
297 }
298 for (i = 0; i <= opt.nloops; i++) {
299 LP_EXT_STORE(i) = 0;
300 LP_PTR_LOAD(i) = 0;
301 LP_PTR_STORE(i) = 0;
302 LP_QJSR(i) = 0;
303 LP_STL(i) = 0;
304 LP_STL_PAR(i) = 0;
305 LP_CS(i) = 0;
306 LP_CSECT(i) = 0;
307 LP_MASTER(i) = 0;
308 LP_PARREGN(i) = 0;
309 LP_PARSECT(i) = 0;
310 /*
311 * These are initialized by findloop:
312 * LP_NOBLA(i)
313 * LP_CALLFG(i)
314 * LP_CALLINTERNAL(i)
315 * LP_ZEROTRIP(i)
316 * LP_JMP_TBL(i)
317 * LP_PARLOOP(i)
318 */
319 }
320 local_again = TRUE;
321 if (!jumps_deleted) {
322 do_lv = FALSE; /* no need to recompute live variables */
323 do_unv = FALSE;
324 }
325 flow_end();
326 freearea(STL_AREA);
327 freearea(DU_AREA);
328 if (jumps_deleted) {
329 freearea(PSI_AREA);
330
331 flowgraph(); /* build the flowgraph for the function */
332
333 findloop(HLOPT_ALL); /* find the loops */
334 }
335#if DEBUG
336 if (OPTDBG(29, 32)) {
337 fprintf(gbl.dbgfil, "***** After const prop *****\n");
338 dump_flowgraph();
339 dump_loops();
340 }
341 if (DBGBIT(29, 64)) {
342 int jj;
343 jj = gbl.entbih;
344 fprintf(gbl.dbgfil,
345 "\n***** BIHs for Function \"%s\" After const prop *****\n",
346 getprint((int)FG_LABEL(jj)));
347 for (;;) {
348 dmpilt(jj);
349 if (FG_XT(jj))
350 break;
351 jj = FG_LNEXT(jj);
352 }
353 }
354#endif
355
356 flow_init();
357
358 if (do_lv)
359 live_var_init();
360
361 localflow();
362
363 if (do_lv) {
364 live_var();
365 live_var_end();
366 }
367 if (flg.opt >= 2 && !XBIT(2, 0x400000)) {
368 /* relies on data structures in live_var_init */
369 LOGICAL tdo_lv = do_lv;
370 do_lv = FALSE;
371 if (do_unv && tdo_lv) {
372 uninit_var_init();
373 uninit_var();
374 uninit_var_end();
375 }
376 do_lv = tdo_lv;
377 }
378 reaching_defs();
379 du_ud();
380 if (jumps_deleted) {
381 /* since the flowgraph had to be rebuilt it's possible that
382 * additional const prop opportunities are available.
383 * This is because that certain defs could have occurred
384 * in what was determined to be dead blocks.
385 */
386 do_lv = TRUE;
387 do_unv = TRUE;
388 goto again;
389 }
390 }
391 do_lv = TRUE; /* so flow_end() functions properly */
392 do_unv = TRUE;
393 local_again = FALSE;
394
395 /* delete any ilts saved for deletion during localflow() */
396
397 if (dilt_l != NULL) {
398 do {
399 unlnkilt((int)dilt_l->ilt, (int)dilt_l->bih, TRUE);
400 dilt_l = dilt_l->next;
401 } while (dilt_l != NULL);
402 freearea(DLT_AREA);
403 }
404
405#if DEBUG
406 if (flg.dbg[9]) {
407 int ii;
408 for (ii = 0; ii < nmeb.stg_avail; ii++)
409 if (NME_STL(ii)) {
410 interr("flow: bad stl", ii, 3);
411 fprintf(gbl.dbgfil, "flow: bad stl, nme:%d, rf:%d\n", ii, NME_STL(ii));
412 }
413 }
414#endif
415
416#if DEBUG
417 if (DBGBIT(9, 128))
418 dump_global(TRUE);
419#endif
420
421}

423/** \brief Cleanup up structures created by flow analysis and after a function
424 has been optimized.
425 */
426void
427flow_end(void)
428{
429 int dfx, nme;
430
431 for (dfx = FIRST_DEF; dfx <= opt.ndefs; dfx++) {
432 nme = DEF_NM(dfx);
433 NME_DEF(nme) = 0;
434 }
435#if DEBUG
436 for (nme = 0; nme < nmeb.stg_avail; nme++)
437 if (NME_DEF(nme))
438 interr("flow_end: bad rfptr", nme, 1);
439#endif
440
441 FREE(opt.def_setb.stg_base); /* allocated in globalflow */
442 if (do_lv) {
443 FREE(lv.lp_stg_base);
444 FREE(opt.def_setb.nme_base);
445 }
446 use_hash_free();
447}

476static void
477localflow(void)
478{
479 int i, parent;
480 STL *p;
481 int nxtfg;
482
483 /* first allocate an stl for each loop + region 0 hence i = 0 not 1 */
484 for (i = 0; i <= opt.nloops; i++) {
485 int tmp;
486 if (i == 0)
487 tmp = 0;
488 else
489 tmp = LP_LOOP(i);
490 cur_stl = (STL *)getitem(STL_AREA, sizeof(STL));
491 LP_STL(tmp) = cur_stl;
492 cur_stl->store = 0;
493 cur_stl->nextsibl = 0;
494 cur_stl->childlst = 0;
495 }
496
497#if DEBUG
498 {
499 int ii;
500 for (ii = 1; ii < astb.stg_avail; ii++)
501 if (A_VISITG(ii)) {
502 interr("local_flow: ast visited", ii, 3);
503 A_VISITP(ii, 0);
504 }
505 }
506#endif
507 for (i = 1; i <= opt.nloops; i++) {
508 cur_lp = LP_LOOP(i);
509 cur_stl = LP_STL(cur_lp);
510 /* link me into my parent's childlst */
511 parent = LP_PARENT(cur_lp);
512 p = LP_STL(parent);
513 cur_stl->nextsibl = p->childlst;
514 p->childlst = cur_stl;
515 }
516
517 for (cur_fg = gbl.entbih; cur_fg; cur_fg = nxtfg) {
518 nxtfg = FG_LNEXT(cur_fg); /* just in case cur_fg is changed */
519 lflow_of_block();
520 }
521
522 for (i = 1; i <= opt.nloops; i++) {
523 cur_lp = LP_LOOP(i);
524 parent = LP_PARENT(cur_lp);
525 LP_EXT_STORE(parent) |= LP_EXT_STORE(cur_lp);
526 LP_PTR_STORE(parent) |= LP_PTR_STORE(cur_lp);
527 LP_PTR_LOAD(parent) |= LP_PTR_LOAD(cur_lp);
528 LP_CALLFG(parent) |= LP_CALLFG(cur_lp);
529 LP_CALLINTERNAL(parent) |= LP_CALLINTERNAL(cur_lp);
530 LP_QJSR(parent) |= LP_QJSR(cur_lp);
531 LP_NOBLA(parent) |= LP_NOBLA(cur_lp);
532 LP_JMP_TBL(parent) |= LP_JMP_TBL(cur_lp);
533 LP_CSECT(parent) |= LP_CSECT(cur_lp);
534 LP_PARREGN(parent) |= LP_PARREGN(cur_lp);
535 LP_PARSECT(parent) |= LP_PARSECT(cur_lp);
536 }
537
538#if DEBUG
539 if (OPTDBG(9, 64))
540 prntstl(LP_STL(0));
541#endif
542
543}

545static void
546lflow_of_block(void)
547{
548 if (FG_DFN(cur_fg) == -1)
549 return;
550 cur_lp = FG_LOOP(cur_fg);
551 cur_stl = LP_STL(cur_lp);
552 if (cur_lp)
553 /* apply loop-scoped pragmas available for this loop */
554 open_pragma((int)FG_LINENO(LP_HEAD(cur_lp)));
555
556#if DEBUG
557 if (OPTDBG(9, 64))
558 fprintf(gbl.dbgfil, "\n---------- local flow trace of node %d, loop %d\n",
559 cur_fg, cur_lp);
560 if (OPTDBG(9, 64))
561 fprintf(gbl.dbgfil, " flow graph node %d\n", cur_fg);
562#endif
563 /*
564 * Use 0'th def entry as the head of the list of defs, linked
565 * together via the lnext field, created for the current flowgraph
566 * node. last_def is set to the 0'th entry; when build_ud is
567 * done for this node, DEF_LNEXT(0) (possibly 0 => "no defs") is
568 * the list of defs for the node. NOTE that the defs in the
569 * list are in lexical order.
570 */
571 last_def = 0;
572 DEF_LNEXT(0) = 0;
573 first_use = opt.useb.stg_avail;
574 num_uses = 0;
575 use_hash_reset();
576 if (FG_CS(cur_fg)) {
577 LP_CS(cur_lp) = 1;
578 LP_CSECT(cur_lp) = 1;
579 }
580 if (FG_TASK(cur_fg))
581 LP_TASK(cur_lp) = 1;
582 if (FG_MASTER(cur_fg))
583 LP_MASTER(cur_lp) = 1;
584 if (FG_PAR(cur_fg))
585 LP_PARREGN(cur_lp) = 1;
586 if (FG_PARSECT(cur_fg))
587 LP_PARSECT(cur_lp) = 1;
588 rdilts(cur_fg);
589 for (cur_std = FG_STDFIRST(cur_fg); cur_std; cur_std = STD_NEXT(cur_std)) {
590#if DEBUG
591 if (OPTDBG(9, 64))
592 fprintf(gbl.dbgfil, " std %d\n", cur_std);
593#endif
594 build_ud((int)STD_AST(cur_std));
595 STD_EX(cur_std) = cur_callfg;
596 A_CALLFGP(STD_AST(cur_std), cur_callfg);
597 FG_EX(cur_fg) |= STD_EX(cur_std);
598 LP_CALLFG(cur_lp) |= FG_EX(cur_fg);
599 LP_CALLINTERNAL(cur_lp) |= cur_callinternal;
600 }
601 wrilts(cur_fg);
602
603 if (cur_lp)
604 close_pragma();
605
606 if (!FG_XT(cur_fg)) {
607 int nxtfg;
608
609 nxtfg = FG_LNEXT(cur_fg);
610 cur_std = FG_STDFIRST(nxtfg);
611 if (cur_std) {
612 switch (A_TYPEG(STD_AST(cur_std))) {
613 case A_DO:
614#ifdef A_MP_PDO
615 case A_MP_PDO:
616#endif
617 case A_FORALL:
618 build_do_init((int)STD_AST(cur_std));
619 break;
620 default:
621 break;
622 }
623 }
624 }
625
626 FG_FDEF(cur_fg) = DEF_LNEXT(0);
627 clean_names(cur_stl);
628
629}

640static void
641build_ud(int tree)
642{
643 /* setup to record list of nodes visited; use nonzero value
644 * to denote that an ili has been visited AND the next ili
645 * in the list. astb.i0 represents the end of the list; must start
646 * out with this ili as being visited.
647 */
648 ast_visit(1, 1);
649 cur_callfg = 0;
650 cur_callinternal = 0;
651 ast_traverse(tree, bld_ud, NULL, NULL);
652 ast_unvisit();
653

656static LOGICAL
657bld_ud(int ast, int *dummy)
658{
659 int atype;
660 int sym;
661 int i, j;
662 int astli;
663 int argt;
664 int cnt;
665 int top; /* ast of matching do/forall of enddo/endforall */
666 int opnd, baseopnd;
667 int dtype;
668 int u;
669 DEF *df;
670 int idpdsc, iface;
671 int intent;
672 int parmcnt;
673 int templatecall;
674
675 cur_callfg |= A_CALLFGG(ast);
676 switch (atype = A_TYPEG(ast)) {
677 case A_ID:
678 case A_MEM:
679 case A_SUBSCR:
680 case A_SUBSTR:
681 (void)bld_use(ast, FALSE);
682 return TRUE;
683 case A_ICALL:
684 ast_traverse((int)A_LOPG(ast), bld_ud, NULL, NULL);
685 argt = A_ARGSG(ast);
686 switch (A_OPTYPEG(ast)) {
687 case I_ALL:
688 case I_ANY:
689 case I_COUNT:
690 case I_PRODUCT:
691 case I_SUM:
692 case I_FINDLOC:
693 case I_MAXVAL:
694 case I_MINVAL:
695 case I_CSHIFT:
696 case I_DOT_PRODUCT:
697 case I_EOSHIFT:
698 case I_MAXLOC:
699 case I_MINLOC:
700 case I_PACK:
701 case I_RESHAPE:
702 case I_SPREAD:
703 case I_TRANSPOSE:
704 case I_UNPACK:
705 case I_TRANSFER:
706 case I_REDUCE_SUM:
707 case I_REDUCE_PRODUCT:
708 case I_REDUCE_MAXVAL:
709 case I_REDUCE_MINVAL:
710 case I_REDUCE_ALL:
711 case I_REDUCE_IALL:
712 case I_REDUCE_ANY:
713 case I_REDUCE_IANY:
714 case I_REDUCE_PARITY:
715 case I_REDUCE_IPARITY:
716 /* for these intrinsic calls, the first argument is intent inout.
717 * the remaining arguments are uses
718 */
719 cnt = A_ARGCNTG(ast);
720 if (cnt == 0)
721 break;
722 opnd = ARGT_ARG(argt, 0);
723 if (opnd) {
724 u = bld_use(opnd, TRUE);
725 (void)bld_lhs(opnd, ast, TRUE, 1);
726 }
727
728 /* the remaining arguments are uses */
729 for (i = 1; i < cnt; i++) {
730 opnd = ARGT_ARG(argt, i);
731 if (opnd) {
732 switch (A_TYPEG(opnd)) {
733 case A_ID:
734 case A_MEM:
735 case A_SUBSCR:
736 case A_SUBSTR:
737 /* for those ASTs which are lvalues, create use items
738 * now and mark as 'arg' uses.
739 */
740 u = bld_use(opnd, TRUE);
741 break;
742 default:
743 ast_traverse(opnd, bld_ud, NULL, NULL);
744 break;
745 }
746 }
747 }
748 break;
749
750 case I_MVBITS:
751 default:
752 idpdsc = 0;
753 parmcnt = 0;
754 templatecall = 0;
755 goto like_call;
756 }
757 return TRUE;
758
759 case A_CALL:
760 case A_FUNC:
761 sym = procsym_of_ast(A_LOPG(ast));
762 proc_arginfo(sym, NULL, &idpdsc, &iface);
763 parmcnt = 0;
764 templatecall = 0;
765 if (iface) {
766 parmcnt = PARAMCTG(iface);
767 if (INTERNALG(iface)) {
768 cur_callinternal = 1;
769 }
770 if (HCCSYMG(iface) && NODESCG(iface) &&
771 getF90TmplSectRtn(SYMNAME(iface))) {
772 templatecall = 1;
773 }
774 }
775 like_call:
776 cur_callfg = 1;
777 ast_traverse((int)A_LOPG(ast), bld_ud, NULL, NULL);
778 argt = A_ARGSG(ast);
779 j = 0;
780 for (i = 0, cnt = A_ARGCNTG(ast); cnt; cnt--) {
781 int arg = 0;
782 intent = INTENT_DFLT;
783 if (templatecall) {
784 if (i == 0) {
785 intent = INTENT_OUT;
786 } else {
787 intent = INTENT_IN;
788 }
789 } else if (idpdsc) {
790 if (i < parmcnt) {
791 intent = INTENTG(aux.dpdsc_base[idpdsc + i]);
792 } else {
793 /* this is a descriptor argument for some fortran argument */
794 while (j < parmcnt && !needs_descriptor(aux.dpdsc_base[idpdsc + j]))
795 ++j;
796 if (j < parmcnt) {
797 arg = aux.dpdsc_base[idpdsc + j];
798 /* if this is a F90 pointer or HPF dynamic argument,
799 * the descriptor may be changed by the call */
800 if (!POINTERG(arg)) {
801 intent = INTENT_IN;
802 }
803 }
804 }
805 }
806 opnd = ARGT_ARG(argt, i);
807 if (opnd && intent != INTENT_OUT)
808 switch (A_TYPEG(opnd)) {
809 case A_ID:
810 case A_MEM:
811 case A_SUBSCR:
812 case A_SUBSTR:
813 u = bld_use(opnd, TRUE);
814 break;
815 default:
816 ast_traverse(opnd, bld_ud, NULL, NULL);
817 break;
818 }
819 if (opnd && A_ISLVAL(A_TYPEG(opnd)) && intent != INTENT_IN)
820 (void)bld_lhs(opnd, ast, TRUE, 1);
821 ++i;
822 }
823 return TRUE;
824
825 case A_ASN:
826 (void)bld_lhs((int)A_DESTG(ast), (int)A_SRCG(ast), FALSE, 1);
827 return TRUE;
828 case A_MP_ATOMICWRITE:
829 case A_MP_ATOMICUPDATE:
830 case A_MP_ATOMICCAPTURE:
831 (void)bld_lhs((int)A_LOPG(ast), (int)A_ROPG(ast), FALSE, 1);
832 return TRUE;
833 case A_MP_ATOMICREAD:
834 (void)bld_use(A_SRCG(ast), FALSE);
835 return TRUE;
836 case A_ASNGOTO:
837 df = bld_lhs((int)A_DESTG(ast), (int)A_SRCG(ast), FALSE, 1);
838 df->flags.bits.other = 1;
839 return TRUE;
840 case A_ENDDO:
841#ifdef A_MP_ENDPDO
842 case A_MP_ENDPDO:
843#endif
844 top = A_OPT2G(ast);
845 if (A_TYPEG(top) == A_DOWHILE)
846 break;
847#if DEBUG
848#ifdef A_MP_PDO
849 assert(A_TYPEG(top) == A_DO || A_TYPEG(top) == A_MP_PDO,

851#else

853#endif
854#endif
855 ast = A_DOVARG(top);
856 dtype = A_DTYPEG(ast);
857 if (A_M3G(top))
858 cnt = A_M3G(top);
859 else if (dtype == DT_REAL4)
860 cnt = mk_cnst(stb.flt1);
861 else if (dtype == DT_DBLE)
862 cnt = mk_cnst(stb.dbl1);
863 else
864 cnt = astb.i1;
865 cnt = mk_binop(OP_ADD, ast, cnt, dtype);
866 df = bld_lhs(ast, cnt, FALSE, 1);
867 df->flags.bits.doend = 1;
868 return TRUE;
869 case A_ALLOC:
870 opnd = A_SRCG(ast);
871 /* If the symbol being allocated has a section descriptor,
872 * add a def of the section descriptor here also */
873 baseopnd = opnd;
874 while (A_TYPEG(baseopnd) == A_SUBSCR) {
875 baseopnd = A_LOPG(baseopnd);
876 }
877 if (A_TYPEG(baseopnd) == A_ID) {
878 int sd;
879 sym = A_SPTRG(baseopnd);
880 sd = SDSCG(sym);
881 if (sd) {
882 int n, sda;
883 sda = mk_id(sd);
884 n = A_NMEG(sda);
885 df = bld_lhs(sda, ast, FALSE, 0);
886 if (df)
887 df->flags.bits.other = 1;
888 }
889 } else if (A_TYPEG(baseopnd) == A_MEM) {
890 int sd;
891 sym = A_SPTRG(A_MEMG(baseopnd));
892 sd = SDSCG(sym);
893 if (sd) {
894 int n, sda;
895 sda = check_member(baseopnd, mk_id(sd));
896 n = A_NMEG(sda);
897 df = bld_lhs(sda, ast, FALSE, 0);
898 if (df)
899 df->flags.bits.other = 1;
900 }
901 }
902 df = bld_lhs(opnd, ast, FALSE, 1);
903 if (df)
904 df->flags.bits.other = 1;
905 opnd = A_LOPG(ast);
906 if (opnd) {
907 df = bld_lhs(opnd, ast, FALSE, 1);
908 if (df)
909 df->flags.bits.other = 1;
910 }
911 opnd = A_DESTG(ast);
912 if (opnd) {
913 df = bld_lhs(opnd, ast, FALSE, 1);
914 if (df)
915 df->flags.bits.other = 1;
916 }
917 opnd = A_M3G(ast);
918 if (opnd) {
919 df = bld_lhs(opnd, ast, FALSE, 1);
920 if (df)
921 df->flags.bits.other = 1;
922 }
923 opnd = A_STARTG(ast);
924 if (opnd) {
925 /* FS#19312: Check for section descriptor in the source=
926 * input argument. If one is present, add a def of the
927 * section descriptor.
928 */
929 baseopnd = opnd;
930 while (A_TYPEG(baseopnd) == A_SUBSCR) {
931 baseopnd = A_LOPG(baseopnd);
932 }
933 if (A_TYPEG(baseopnd) == A_ID) {
934 int sd;
935 sym = A_SPTRG(baseopnd);
936 sd = SDSCG(sym);
937 if (sd) {
938 int sda;
939 sda = mk_id(sd);
940 df = bld_lhs(sda, ast, FALSE, 0);
941 if (df)
942 df->flags.bits.other = 1;
943 }
944 } else if (A_TYPEG(baseopnd) == A_MEM) {
945 int sd;
946 sym = A_SPTRG(A_MEMG(baseopnd));
947 sd = SDSCG(sym);
948 if (sd) {
949 int sda;
950 sda = check_member(baseopnd, mk_id(sd));
951 df = bld_lhs(sda, ast, FALSE, 0);
952 if (df)
953 df->flags.bits.other = 1;
954 }
955 }
956 }
957 return TRUE;
958 case A_FORALL:
959 LP_FORALL(cur_lp) = 1;
960 for (astli = A_LISTG(ast); astli; astli = ASTLI_NEXT(astli))
961 ast_traverse((int)ASTLI_TRIPLE(astli), bld_ud, NULL, NULL);
962 if (A_IFEXPRG(ast))
963 ast_traverse((int)A_IFEXPRG(ast), bld_ud, NULL, NULL);
964 if (A_IFSTMTG(ast)) {
965 ast_traverse((int)A_IFSTMTG(ast), bld_ud, NULL, NULL);
966 goto endforall_shared;
967 }
968 return TRUE;
969 case A_ENDFORALL:
970 top = A_OPT2G(ast);
971#if DEBUG
972 assert(A_TYPEG(top) == A_FORALL, "bld_ud:mismatched endforall", ast, 3);
973#endif
974 ast = top;
975 endforall_shared:
976 for (astli = A_LISTG(ast); astli; astli = ASTLI_NEXT(astli)) {
977 ast = mk_id((int)ASTLI_SPTR(astli));
978 cnt = ASTLI_TRIPLE(astli);
979 if (A_STRIDEG(cnt))
980 cnt = A_STRIDEG(cnt);
981 else
982 cnt = astb.i1;
983 cnt = mk_binop(OP_ADD, ast, cnt, DT_INT);
984 df = bld_lhs(ast, cnt, FALSE, 1);
985 df->flags.bits.doend = 1;
986 }
987 return TRUE;
988 case A_REDIM:
989 opnd = A_SRCG(ast);
990 df = bld_lhs(opnd, ast, FALSE, 1);
991 df->flags.bits.other = 1;
992 return TRUE;
993 case A_PRAGMA:
994 /* treat appearances in ACC pragmas as uses */
995 switch (A_PRAGMATYPEG(ast)) {

1097static int
1098bld_use(int lval, LOGICAL isarg)
1099{
1100 int ast;
1101 int sym, nme;
1102 int i, asd;
1103 int precise;
1104 int s;
1105 int u;
1106
1107 ast = lval;
1108 precise = 1;
1109 u = 0;
1110again:
1111 switch (A_TYPEG(ast)) {
1112 case A_CONV:
1113 ast = A_LOPG(ast);
1114 goto again;
1115 case A_ID:
1116 sym = A_SPTRG(ast);
1117 for (s = sym; SCG(s) == SC_BASED;) {
1118 int n, a;
1119 s = MIDNUMG(s);
1120 if (s > NOSYM && ST_ISVAR(STYPEG(s))) {
1121 a = mk_id(s);
1122 n = A_NMEG(a);
1123 u = add_use(n, a, a, TRUE);
1124 USE_ARG(u) |= isarg;
1125#if DEBUG
1126 if (OPTDBG(9, 64))
1127 fprintf(
1128 gbl.dbgfil,
1129 " ptrast %d, addr %d: use of nme %d, ulist %d \"%s\"\n",
1130 a, a, n, u, getprint((int)NME_SYM(n)));
1131#endif
1132 chk_ptr_load(n);
1133 } else
1134 break;
1135 }
1136 if (ST_ISVAR(STYPEG(sym))) {
1137 nme = A_NMEG(ast);
1138 if (POINTERG(sym))
1139 chk_ptr_load(nme);
1140 if (isarg && ast != lval && SEQG(sym))
1141 /* an aggregate reference or subscripted array as an argument
1142 * where the object is marked sequential.
1143 */
1144 u = add_use(nme, ast, lval, 0);
1145 else
1146 u = add_use(nme, ast, lval, precise);
1147 USE_ARG(u) |= isarg;
1148#if DEBUG
1149 if (OPTDBG(9, 64))
1150 fprintf(gbl.dbgfil,
1151 " ast %d, addr %d: use of nme %d, ulist %d \"%s\"\n",
1152 ast, lval, nme, u, getprint((int)NME_SYM(nme)));
1153#endif
1154 }
1155 break;
1156 case A_MEM:
1157 nme = A_NMEG(ast);
1158 ast = A_PARENTG(ast);
1159 if (NME_TYPE(nme) != NT_MEM) {
1160 lval = ast;
1161 precise = 0;
1162 } else if (POINTERG(NME_SYM(nme))) {
1163 chk_ptr_load(nme);
1164/* ADDRTKN hack - be conservative with the parent symbol;
1165 * the parent's nme is the one which is actually recorded
1166 * in the USE and DEF structures.
1167 */
1168#ifdef PTRSTOREP
1169 ptrstore_of(nme);
1170#else
1171 loc_of(nme);
1172#endif
1173 }
1174 goto again;
1175 case A_SUBSCR:
1176 asd = A_ASDG(ast);
1177 for (i = 0; i < (int)ASD_NDIM(asd); i++) {
1178 ast_traverse((int)ASD_SUBS(asd, i), bld_ud, NULL, NULL);
1179 if (A_ALIASG(ASD_SUBS(asd, i)) == 0)
1180 precise = 0;
1181 }
1182 /* build a use of the section descriptor */
1183 sym = memsym_of_ast(ast);
1184 if ((POINTERG(sym) || ALLOCG(sym)) && SDSCG(sym)) {
1185 int a;
1186 a = check_member(A_LOPG(ast), mk_id(SDSCG(sym)));
1187 u = add_use(A_NMEG(a), a, a, TRUE);
1188 USE_ARG(u) |= isarg;
1189#if DEBUG
1190 if (OPTDBG(9, 64))
1191 fprintf(gbl.dbgfil,
1192 " sdsc %d, addr %d: use of nme %d, ulist %d \"%s\"\n",
1193 a, a, A_NMEG(a), u, getprint((int)NME_SYM(A_NMEG(a))));
1194#endif
1195 }
1196 ast = A_LOPG(ast);
1197 goto again;
1198 case A_SUBSTR:
1199 if (A_LEFTG(ast)) {
1200 ast_traverse((int)A_LEFTG(ast), bld_ud, NULL, NULL);
1201 }
1202 if (A_RIGHTG(ast)) {
1203 ast_traverse((int)A_RIGHTG(ast), bld_ud, NULL, NULL);
1204 }
1205 precise = 0;
1206 ast = A_LOPG(ast);
1207 goto again;
1208 case A_FUNC:
1209 break;
1210 case A_CNST:
1211 if (!XBIT(7, 0x100000))
1212 break;
1213 FLANG_FALLTHROUGH;
1214 default:
1215 interr("bld_use, ast nyd", ast, 3);
1216 }
1217 return u;
1218}

1220static DEF *
1221bld_lhs(int lhs, int rhs, LOGICAL isarg, int precise)
1222{
1223 int ast;
1224 int sym, nme;
1225 int i, asd;
1226 DEF *df = NULL;
1227 int mark_use; /* to mark beginning of uses found in a rhs */
1228 int s;
1229
1230 mark_use = opt.useb.stg_avail;
1231 ast_traverse(rhs, bld_ud, NULL, NULL);
1232 ast = lhs;
1233
1234again:
1235 switch (A_TYPEG(ast)) {
1236 case A_ID:
1237 sym = A_SPTRG(ast);
1238 if (!ST_ISVAR(STYPEG(sym)))
1239 return NULL;
1240 for (s = sym; SCG(s) == SC_BASED;) {
1241 int n, a;
1242 s = MIDNUMG(s);
1243 if (s > NOSYM && ST_ISVAR(STYPEG(s))) {
1244 a = mk_id(s);
1245 n = A_NMEG(a);
1246 i = add_use(n, a, a, TRUE);
1247#if DEBUG
1248 if (OPTDBG(9, 64))
1249 fprintf(
1250 gbl.dbgfil,
1251 " ptrast %d, addr %d: use of nme %d, ulist %d \"%s\"\n",
1252 a, a, n, i, getprint((int)NME_SYM(n)));
1253#endif
1254 chk_ptr_load(n);
1255 } else
1256 break;
1257 }
1258 nme = A_NMEG(ast);
1259 if (POINTERG(sym))
1260 chk_ptr_load(nme);
1261 add_store(nme);
1262 df = add_def(nme, ast, lhs, rhs, precise);
1263
1264 /* if the def is due to an appearance as an actual argument, don't
1265 * turn off the GEN flag of any defs which precede this point.
1266 */
1267 if (!isarg) {
1268 i = df->next;
1269 if (precise && ast == lhs && /* def defines the whole variable */
1270 DEF_FG(i) == cur_fg) /* i may be zero, but flags of */
1271 DEF_GEN(i) = 0; /* DEF 0 are zero */
1272 else {
1273 for (; i && DEF_FG(i) == cur_fg; i = DEF_NEXT(i))
1274 if (precise && DEF_LHS(i) == ast && DEF_ADDR(i) == lhs)
1275 DEF_GEN(i) = 0;
1276 }
1277 } else
1278 df->flags.bits.arg = 1;
1279
1280 df->flags.bits.gen = 1;
1281 if (A_ALIASG(rhs))
1282 df->flags.bits.cnst = 1;
1283 if (g_mark_use.first)
1284 for (i = g_mark_use.first; i <= g_mark_use.last; i++)
1285 if (USE_NM(i) == nme) {
1286 df->flags.bits.self = 1;
1287 break;
1288 }
1289 while (mark_use < opt.useb.stg_avail) {
1290 if (USE_NM(mark_use) == nme) {
1291 df->flags.bits.self = 1;
1292 break;
1293 }
1294 mark_use++;
1295 }
1296#if DEBUG
1297 if (OPTDBG(9, 64))
1298 fprintf(gbl.dbgfil, " lhs %d: store of nme %d \"%s\"\n", lhs, nme,
1299 getprint((int)NME_SYM(nme)));
1300#endif
1301 if ((FG_CS(cur_fg) || FG_PARSECT(cur_fg) || FG_PAR(cur_fg)) &&
1302 !IS_PRIVATE(NME_SYM(nme))) {
1303 Q_ITEM *q;
1304 int ll;
1305
1306#if DEBUG
1307 if (OPTDBG(9, 64))
1308 fprintf(gbl.dbgfil, " store of nonprivate %s\n",
1309 getprint((int)NME_SYM(nme)));
1310#endif
1311 /*
1312 * For this loop and all enclosing loops, add this variable
1313 * (its nme) to the list of non-private variables which were
1314 * stored while in a critical section, a parallel section, or
1315 * parallel region.
1316 */
1317 for (ll = cur_lp;; ll = LP_PARENT(ll)) {
1318 for (q = LP_STL_PAR(ll); q != NULL; q = q->next)
1319 if (q->info == nme)
1320 goto nme_in_par;
1321
1322 q = (Q_ITEM *)getitem(STL_AREA, sizeof(Q_ITEM));
1323 q->info = nme;
1324 q->next = LP_STL_PAR(ll);
1325 LP_STL_PAR(ll) = q;
1326 nme_in_par:
1327 if (ll == 0)
1328 break;
1329 }
1330 }
1331 break;
1332 case A_MEM:
1333 nme = A_NMEG(ast);
1334 add_store((int)A_NMEG(ast));
1335 ast = A_PARENTG(ast);
1336 if (NME_TYPE(nme) != NT_MEM) {
1337 lhs = ast;
1338 precise = 0;
1339 } else if (POINTERG(NME_SYM(nme))) {
1340 chk_ptr_load(nme);
1341 if (!is_ptr_safe(nme)) {
1342 LP_PTR_STORE(cur_lp) = 1;
1343 FG_PTR_STORE(cur_fg) = 1;
1344 }
1345/* ADDRTKN hack - be conservative with the parent symbol;
1346 * the parent's nme is the one which is actually recorded
1347 * in the USE and DEF structures.
1348 */
1349#ifdef PTRSTOREP
1350 ptrstore_of(nme);
1351#else
1352 loc_of(nme);
1353#endif
1354 }
1355 goto again;
1356 case A_SUBSCR:
1357 add_store((int)A_NMEG(ast));
1358 asd = A_ASDG(ast);
1359 for (i = 0; i < (int)ASD_NDIM(asd); i++) {
1360 ast_traverse((int)ASD_SUBS(asd, i), bld_ud, NULL, NULL);
1361 if (A_ALIASG(ASD_SUBS(asd, i)) == 0)
1362 precise = 0;
1363 }
1364 /* build a use of the section descriptor */
1365 sym = memsym_of_ast(ast);
1366 if ((POINTERG(sym) || ALLOCG(sym)) && SDSCG(sym)) {
1367 int a, u;
1368 a = check_member(A_LOPG(ast), mk_id(SDSCG(sym)));
1369 u = add_use(A_NMEG(a), a, a, TRUE);
1370 USE_ARG(u) |= isarg;
1371#if DEBUG
1372 if (OPTDBG(9, 64))
1373 fprintf(gbl.dbgfil,
1374 " sdsc %d, addr %d: use of nme %d, ulist %d \"%s\"\n",
1375 a, a, A_NMEG(a), u, getprint((int)NME_SYM(A_NMEG(a))));
1376#endif
1377 }
1378 ast = A_LOPG(ast);
1379 goto again;
1380 case A_SUBSTR:
1381 if (A_LEFTG(ast)) {
1382 ast_traverse((int)A_LEFTG(ast), bld_ud, NULL, NULL);
1383 }

1487add_use(int nme, int ilix, int addr, int precise)
1488{
1489 int usex, def;
1490
1491 usex = use_hash_lookup(false, true, addr, nme, cur_std);
1492#if (DEBUG && DEBUG_USE_HASH)
1493 {
1494 int nuses = num_uses;
1495 int lusex = 0;
1496 bool found = false;
1497 /* Perform the linear use search, and compare to the hash lookup */
1498 for (lusex = first_use; nuses--; lusex++) {
1499 if (USE_NM(lusex) == nme && USE_STD(lusex) == cur_std && USE_ADDR(lusex) == addr) {
1500 found = true;
1501 break;
1502 }
1503 }
1504 use_hash_check(usex, lusex, found);
1505 }
1506#endif /* DEBUG && DEBUG_USE_HASH */
1507 if (usex) return usex;
1508
1509 usex = opt.useb.stg_avail++;
1510 OPT_NEED(opt.useb, USE, 100);
1511
1512 num_uses++;
1513 use_hash_insert(usex, false, true, addr, nme, cur_std);
1514 BZERO(opt.useb.stg_base + usex, USE, 1);
1515 USE_NM(usex) = nme;
1516 USE_FG(usex) = cur_fg;
1517 USE_STD(usex) = cur_std;
1518 USE_AST(usex) = ilix;
1519 USE_ADDR(usex) = addr;
1520 USE_UD(usex) = NULL;
1521 USE_PRECISE(usex) = precise;
1522 USE_DOINIT(usex) = in_doinit;
1523 if ((def = NME_DEF(nme)) && DEF_FG(def) == cur_fg) {
1524 /* in the same flowgraph node, there exists a def of the same
1525 * variable prior to the use
1526 */
1527 if (ilix == addr) {
1528 /* use is a whole use or just a scalar variable */
1529 if (!precise || !DEF_PRECISE(def) || DEF_ADDR(def) != addr)
1530 USE_EXPOSED(usex) = 1;
1531 /*
1532 * add use to du of the definition for nme which is in the block
1533 * and immediately precedes the use
1534 */
1535 add_du_ud(def, usex);
1536#if DEBUG
1537 if (OPTDBG(9, 64) && !USE_EXPOSED(usex))
1538 fprintf(gbl.dbgfil, "-- use %d reached by def %d (use not exposed1)\n",
1539 usex, def);
1540#endif
1541 /* since this an aggregate use, add this as a use of any
1542 * remaining defs in the same block which precede the use.
1543 * Stop if the def's addr is the aggregate (scalar falls into this
1544 * condition) or when we run out of defs for the variable in this
1545 * block.
1546 */
1547 while (TRUE) {
1548 if (precise && DEF_PRECISE(def) && DEF_ADDR(def) == addr)
1549 break;
1550 def = DEF_NEXT(def);
1551 if (def == 0 || DEF_FG(def) != cur_fg)
1552 break;
1553 if (!DEF_GEN(def))
1554 continue;
1555 add_du_ud(def, usex);
1556#if DEBUG
1557 if (OPTDBG(9, 64))
1558 fprintf(gbl.dbgfil,
1559 "-- use %d reached by def %d (use not exposed2)\n", usex, def);
1560#endif
1561 }
1562 return (usex);
1563 }
1564 /* use is a partial use of an array or structure */
1565 while (TRUE) {
1566 if (!precise || !DEF_PRECISE(def) || DEF_LHS(def) == ilix ||
1567 DEF_ADDR(def) == addr || DEF_ADDR(def) == ilix) {
1568 add_du_ud(def, usex);
1569#if DEBUG
1570 if (OPTDBG(9, 64))
1571 fprintf(gbl.dbgfil,
1572 "-- use %d reached by def %d (use not exposed3)\n", usex, def);
1573#endif
1574 if (precise && DEF_PRECISE(def) && DEF_ADDR(def) == addr)
1575 return usex;
1576 }
1577 def = DEF_NEXT(def);
1578 if (def == 0 || DEF_FG(def) != cur_fg)
1579 break;
1580 }
1581 }
1582 /*
1583 * this use is exposed to the beginning of the block -- defs which
1584 * reach this use (those which are in IN(cur_fg)) must be determined
1585 * later
1586 */
1587 USE_EXPOSED(usex) = 1;
1588 if (do_lv)
1589 bv_set(LUSE(cur_fg), nme);
1590 else if (do_unv && flg.opt >= 2 && !XBIT(2, 0x400000)) {
1591 bv_set(LUSE(cur_fg), nme);
1592 }
1593
1594 return (usex);
1595} /* add_use */

1650add_store(int nme)
1651{
1652 int sym, store, mark;
1653
1654 if (cur_lp == 0) {
1655 /*
1656 * for region 0, only care about finding stores via ptrs - this is
1657 * for global register assignment only
1658 */
1659 for (; 1; nme = NME_NM(nme)) {
1660 switch (NME_TYPE(nme)) {
1661 default:
1662 break;
1663
1664 case NT_ARR:
1665 case NT_MEM:
1666 continue;
1667
1668 case NT_VAR:
1669 sym = NME_SYM(nme);
1670 if (SCG(sym) == SC_BASED && !is_ptr_safe(nme)) {
1671 LP_PTR_STORE(cur_lp) = 1;
1672 FG_PTR_STORE(cur_fg) = 1;
1673 } else if (POINTERG(sym) && !is_ptr_safe(nme)) {
1674 LP_PTR_STORE(cur_lp) = 1;
1675 FG_PTR_STORE(cur_fg) = 1;
1676 }
1677 break;
1678
1679 case NT_UNK:
1680 LP_PTR_STORE(cur_lp) = 1;
1681 FG_PTR_STORE(cur_fg) = 1;
1682 break;
1683
1684 case NT_IND:
1685 if (!is_ptr_safe(nme)) {
1686 LP_PTR_STORE(cur_lp) = 1;
1687 FG_PTR_STORE(cur_fg) = 1;
1688 }
1689 break;
1690 }
1691 break;
1692 }
1693 return;
1694 }
1695 /*
1696 * for the nme which is being stored, create a store item for it and
1697 * any "parent" nmes (if an array reference, member reference, etc.).
1698 * Also, each store item is marked indicating if it represents an
1699 * aggregate store or it's an array reference with variable subscripts;
1700 * the mark is propagated to any ancestors. Initially mark is 0 unless
1701 * the nme represents an aggregate store (e.g., created by expander to
1702 * optimize structure assignments).
1703 */
1704 if (NME_TYPE(nme) == NT_VAR)
1705 mark = !ISSCALAR(NME_SYM(nme));
1706 else if (NME_TYPE(nme) == NT_MEM) {
1707 if (NME_SYM(nme) <= 1) /* CMPLX real or imag nme */
1708 mark = 0;
1709 else
1710 mark = !ISSCALAR(NME_SYM(nme));
1711 } else
1712 mark = 0;
1713 for (; 1; nme = NME_NM(nme)) {
1714 if ((store = NME_STL(nme)) == 0) {
1715 store = new_storeitem(nme);
1716 } else if (STORE_TYPE(store))
1717 break;
1718
1719 STORE_TYPE(store) |= mark;
1720#if DEBUG
1721 if (OPTDBG(9, 64))
1722 fprintf(gbl.dbgfil, " nme %d marked %d, store %d\n", nme,
1723 STORE_TYPE(store), store);
1724#endif
1725
1726 switch (NME_TYPE(nme)) {
1727 default:
1728 break;
1729
1730 case NT_ARR:
1731 if (NME_SYM(nme))
1732 mark = 1;
1733 continue;
1734
1735 case NT_MEM:
1736 if (!XBIT(7, 0x100))
1737 mark = 0; /* assume ref is within member's boundaries */
1738 continue;
1739
1740 case NT_VAR:
1741 sym = NME_SYM(nme);
1742 if (SCG(sym) == SC_BASED && !is_ptr_safe(nme)) {
1743 LP_PTR_STORE(cur_lp) = 1;
1744 FG_PTR_STORE(cur_fg) = 1;
1745 } else if (POINTERG(sym) && !is_ptr_safe(nme)) {
1746 LP_PTR_STORE(cur_lp) = 1;
1747 FG_PTR_STORE(cur_fg) = 1;
1748 }
1749 if (IS_EXTERN(sym) || ADDRTKNG(sym))
1750 LP_EXT_STORE(cur_lp) = 1;
1751 break;
1752
1753 case NT_UNK:
1754 STORE_TYPE(store) = 1;
1755 LP_PTR_STORE(cur_lp) = 1;
1756 FG_PTR_STORE(cur_fg) = 1;
1757 break;
1758
1759 case NT_IND:
1760 if (!is_ptr_safe(nme)) {
1761 LP_PTR_STORE(cur_lp) = 1;
1762 FG_PTR_STORE(cur_fg) = 1;
1763 }
1764 break;
1765 }
1766 break;
1767 }
1768

1826static void
1827live_var_init(void)
1828{

1830
1831 /*
1832 * Allocate the space which will contain pointers to the flowgraph
1833 * nodes' live variable sets. This space, organized as a table,
1834 * parallels the flowgraph table.
1835 */

1842 * Then, the space needed for all of the sets in the flow graph is
1843 * allocated. This will be 4*num_nodes+1 BV units (don't use dfn even
1844 * though it's possible that a node will be deleted; algorithm depends

1846 * The additional BV unit (the first one in the space) is used
1847 * as a scratch bit vector.
1848 */
1849 lv.n = nmeb.stg_avail - 1;

1859 fprintf(gbl.dbgfil, "lv: size of bv space %d\n", lv.bv_sz);
1860 }
1861#endif
1862
1863 i = lv.bv_len;
1864 for (v = 1; v <= opt.num_nodes; v++) {
1865 LIN(v) = lv.stg_base + i;
1866 i += lv.bv_len;
1867 LOUT(v) = lv.stg_base + i;
1868 i += lv.bv_len;

1870 i += lv.bv_len;
1871 LDEF(v) = lv.stg_base + i;
1872 i += lv.bv_len;

1878 * are live upon entry. For the exit node, initialize its LUSE to those

1880 */

1892static void
1893live_var_end(void)
1894{
1895 /*
1896 * After the live variable IN/OUT sets have been computed for the flowgraph
1897 * nodes, we compute the live variable sets for each loop (i.e., those
1898 * variables which are live upon entry to the loop and those variables
1899 * which are live out of a loop. Once this computation is completed,
1900 * the space used for the flowgraph sets is freed -- the live-in/-out
1901 * sets of the flowgraph are not preserved.
1902 */
1903 int i, j;
1904 int v;
1905 PSI_P p;
1906
1907 /*
1908 * calculate the size (d) in units of BV required for each loop's
1909 * LIN, and LOUT sets.
1910 * Then, the space needed for all of the sets in the loop is
1911 * allocated. This will be 2*(opt.nloops+1) BV units.
1912 * the additional BV units (the first pair in the space) are used
1913 * as region 0's LIN and LOUT sets.
1914 */
1915 lv.lp_bv_sz = (2 * (opt.nloops + 1)) * lv.bv_len;
1916 NEW(lv.lp_stg_base, BV, lv.lp_bv_sz);
1917 if (lv.lp_stg_base == NULL)
1918 error(7, 4, 0, CNULL, CNULL);
1919 bv_zero(lv.lp_stg_base, lv.lp_bv_sz);
1920#if DEBUG
1921 if (OPTDBG(29, 4)) {
1922 fprintf(gbl.dbgfil, "lvlp: size of each BV %d,", lv.bv_len);
1923 fprintf(gbl.dbgfil, "lvlp: size of bv space %d\n", lv.lp_bv_sz);
1924 }
1925#endif
1926
1927 i = 0;
1928 for (j = 0; j <= opt.nloops; j++) {
1929 LP_LIN(j) = lv.lp_stg_base + i;
1930 i += lv.bv_len;
1931 LP_LOUT(j) = lv.lp_stg_base + i;
1932 i += lv.bv_len;
1933 }
1934 /*
1935 * compute the set of variables which are live into a loop and live
1936 * out of a loop:
1937 * live-in = U LOUT(v), for each predecessor of head not in loop
1938 * live-out = U LIN(v), for each natural exit from the loop
1939 */
1940 for (j = 1; j <= opt.nloops; j++) {
1941 for (p = FG_PRED(LP_HEAD(j)); p; p = PSI_NEXT(p)) {
1942 v = PSI_NODE(p);
1943 if (FG_LOOP(v) == j)
1944 continue;
1945 bv_union(LP_LIN(j), LOUT(v), lv.bv_len);
1946 }
1947 for (p = LP_EXITS(j); p != PSI_P_NULL; p = PSI_NEXT(p)) {
1948 v = PSI_NODE(p);
1949 bv_union(LP_LOUT(j), LIN(v), lv.bv_len);
1950 }
1951 }
1952 /*
1953 * compute the set of variables which are live into a function and live
1954 * out of a function:
1955 * live-in = LIN(entry node)
1956 * live-out = LIN(exit node)
1957 * Note that this is only done for consistency; could just use
1958 * is_entry_live() and is_exit_live() at the time when it's necessary
1959 * to check liveness.
1960 */
1961 bv_copy(LP_LIN(0), LIN(1), lv.bv_len);
1962 bv_copy(LP_LOUT(0), LIN(opt.exitfg), lv.bv_len);
1963#if DEBUG
1964 if (OPTDBG(29, 4))
1965 for (i = 0; i <= opt.nloops; i++) {
1966 fprintf(gbl.dbgfil, "--- lv for loop %d\n", i);
1967 fprintf(gbl.dbgfil, "LIN :");
1968 lv_print(LP_LIN(i));
1969 fprintf(gbl.dbgfil, "LOUT:");
1970 lv_print(LP_LOUT(i));
1971 }
1972#endif
1973 if (flg.opt >= 2 && !XBIT(2, 0x400000))
1974 return;
1975
1976#if DEBUG
1977 BZERO(lv.sets, LV_SETS, opt.num_nodes + 1);
1978#endif
1979 FREE(lv.sets);
1980 FREE(lv.stg_base);
1981
1982}

1984/*
1985 Compute the live variable sets of the nodes in the flowgraph.
1986 The purpose of this analysis is to compute live variable sets for
1987 the loops in the function.
1988
1989 Algorithm:
1990 for each node v in the flowgraph in reverse depth first order {
1991 LIN(v) = LUSE(v);
1992 LOUT(v) = 0;
1993 add v to queue;
1994 }
1995 do {
1996 remove node v from queue;
1997 new_out = 0;
1998 new_out U= LIN(s), for each successor s of v;
1999 if (new_out != LOUT(v)) {
2000 LOUT(v) = new_out;
2001 LIN(v) = (LOUT(v) - LDEF(v)) U LUSE(v);
2002 for each predecessor s of v {
2003 if (s is not in queue)
2004 add s to queue;
2005 }
2006 }
2007 } while (first of queue != last of queue);
2008*/
2009static void
2010live_var(void)
2011{
2012 Q_ITEM *f_q, *l_q, *q;
2013 BV *new_out, *bv;
2014 PSI_P p;
2015 int i, v;
2016
2017 l_q = GET_Q_ITEM(f_q); /* a queue is empty when the first and */
2018 l_q->next = Q_NULL; /* and last entries locate the same */
2019 /* item */
2020
2021 /*
2022 * scan the flow graph nodes in reverse depth first order and define
2023 * lin(fg) to luse(fg) and lout(fg) to the empty set (done by zeroing
2024 * out the entire area at allocation.
2025 * Also, the queue of all the nodes is created.
2026 */
2027 for (i = opt.dfn; i >= 1; i--) {
2028 v = VTX_NODE(i);
2029

2405 * DFO order for a flow graph visits each node,
2406 * numbering each node as it is visited before its successors.
2407 * RDFO order visits each node, numbering each node
2408 * AFTER its successors. We use the reverse of that order,
2409 * or, equivalently, build the list backwards.
2410 */

2415{

2419 int w;

2422 _rdfo(w);
2423 }

2434{

2436 for (v = 0; v <= opt.num_nodes; ++v) {
2437 FG_RDFO(v) = 0;
2438 RDFOVTX_NODE(v) = 0;
2439 }
2440 rdfon = opt.dfn;
2441 _rdfo(1);
2442 if (rdfon != 0)
2443 interr("more DFN nodes than RDFO nodes", 0, 4);

2446/* definition analysis */
2447
2448/*
2449 Algorithm:
2450 for each node v in the flowgraph in depth first order {
2451 IN(v) = 0;
2452 OUT(v) = GEN(v);
2453 }
2454 build reverse-depth-first-order
2455 top = 1
2456 do{
2457 changed = 0
2458 oldtop = top
2459 top = opt.dfn+1
2460 do dfo = oldtop to opt.dfn{
2461 v = RDFOVTX_NODE(dfo)
2462 IN(v) = 0;
2463 IN(v) U= OUT(p), for each predecessor p of v;
2464 new_out = (IN(v) - KILL(v)) U GEN(v);
2465 if( new_out != OUT(v) ){
2466 OUT(v) = new_out
2467 for each successor s of v {
2468 if( FG_DFN(s) < top && FG_DFN(s) < dfo ){
2469 top = FG_DFN(s)
2470 changed = 1
2471 }
2472 }
2473 }
2474 }

2476*/
2477static void
2478reaching_defs(void)

2481 PSI_P p, s;
2482 int i, v, def, bih, top, oldtop, changed, dfo, iter;
2483
2484 /*
2485 * calculate the size (d) in units of BV required for each flow graph's
2486 * IN or OUT set. Then, the space needed for the entire flow graph is
2487 * allocated. This will be 2*dfn+1 BV units. (opt.dfn is the number
2488 * of nodes which are in the flowgraph and not dead -- dead nodes have a
2489 * dfn of -1); the addtional BV unit (the first one in the space) is used
2490 * as a scratch bit vector.
2491 */
2492 build_rdfo();
2493 def_bv_len = (opt.ndefs + BV_BITS - 1) / BV_BITS;
2494 i = (2 * opt.dfn + 1) * def_bv_len;
2495 NEW(opt.def_setb.stg_base, BV, i);
2496 bv_zero(opt.def_setb.stg_base, i);
2497 opt.def_setb.stg_avail = 0;
2498 new_out = opt.def_setb.stg_base + opt.def_setb.stg_avail;
2499 opt.def_setb.stg_avail += def_bv_len;
2500#if DEBUG

2502 fprintf(gbl.dbgfil, " size of each def BV %d,", def_bv_len);
2503 fprintf(gbl.dbgfil, " size of bv space %d\n", i);
2504 }
2505#endif

2513 * in(fg) to empty (done by zeroing out the entire area at allocation.
2514 * out(fg) is set to the definitions in fg which reach the end of the
2515 * node (gen flag). Also, the queue of all the nodes is created.
2516 */

2518 v = RDFOVTX_NODE(i);

2536 * "call def"
2537 */

2540 bv_set(bv, CALL_DEF);
2541 /*
2542 * if a qjsr occurs in this block, then the GEN set contains
2543 * "qjsr def"
2544 */
2545 if (FG_QJSR(bih))

2548 * if a store via a pointer occurs in this block, then the GEN set
2549 * contains "ptr store def"
2550 */

2552 bv_set(bv, PTR_STORE_DEF);
2553#if DEBUG

2556 bv_print(FG_OUT(v), opt.ndefs);
2557 }
2558#endif
2559 }
2560 /*
2561 * do the flow equations
2562 */

2564 iter = 0;
2565 do {
2566 ++iter;
2567 changed = 0;
2568 oldtop = top;
2569 top = opt.dfn + 1;
2570 for (dfo = oldtop; dfo <= opt.dfn; ++dfo) {
2571 /* do the union of the OUT sets for the predecessors of v */
2572 v = RDFOVTX_NODE(dfo);
2573 bv = FG_IN(v);
2574 bv_zero(bv, def_bv_len);
2575 for (p = FG_PRED(v); p != PSI_P_NULL; p = PSI_NEXT(p))
2576 bv_union(bv, FG_OUT(PSI_NODE(p)), def_bv_len);
2577

2580
2581/* new_out = IN(v) - KILL(v) U GEN(v) */
2582/* go through the definitions in GEN(v) */

2586#endif

2643#endif

2646#if DEBUG

2650 }

2726static LOGICAL
2727const_prop(void)
2728{
2729 LOGICAL keep_def;
2730 LOGICAL changes;
2731 int df;
2732 int df_ilt;
2733 int nme;
2734 DU *du;
2735 Q_ITEM *q;
2736 int dvl;
2737
2738 if (XBIT(6, 0x1))
2739 return FALSE;
2740#if DEBUG
2741 if (OPTDBG(9, 32768)) {
2742 fprintf(gbl.dbgfil, "const_prop: file %s, function %s\n", gbl.src_file,
2743 getprint(FG_LABEL(gbl.entbih)));
2744 dump_global(FALSE);
2745 }
2746#endif
2747 /*
2748 * Use a queue to record the definitions which are candidates for
2749 * constant propagation; if a definition's value is propagated, it
2750 * is possible to create more opportunities for constant propagation.
2751 * The new candidates are added to the end of the queue.
2752 * NOTES:
2753 * 1. a queue is empty when the first and last entries locate the same
2754 * item
2755 * 2. the queue is initially filled with the constant definitions
2756 * found during localflow.
2757 * 3. it's sufficient to use the DEF_CONST flag to indicate that
2758 * the definition is in the queue.
2759 * 4. for FORTRAN, before the def table is scanned, opportunities
2760 * for const prop due to data inits are processed. The queue
2761 * to which new def opportunities are added during copy_const is
2762 * released. Subsequent processing of the def table will still
2763 * process the new defs since these defs will have their
2764 * DEF_CONST flags set.
2765 */
2766 cp.l_q = GET_Q_ITEM(cp.f_q);
2767 cp.l_q->next = Q_NULL;
2768 changes = FALSE;
2769 if (!XBIT(7, 0x100000)) {
2770 for (dvl = 0; dvl < aux.dvl_avl; dvl++) {
2771 int use;
2772 int sym;
2773 int nocp;
2774 LOGICAL changes_sym;
2775
2776 sym = DVL_SPTR(dvl);
2777 /* allow only if
2778 * symbol is a local (SC_LOCAL, or SC_STATIC with SAVE flag)
2779 * symbol is never assigned, address it not taken, no storage
2780 * overlap, not a host subprogram
2781 */
2782 if ((sym &&
2783 (SCG(sym) != SC_LOCAL && (SCG(sym) != SC_STATIC || !SAVEG(sym)))) ||
2784 (gbl.internal == 1)
2785#ifdef THREADG
2786 || THREADG(sym)
2787#endif
2788 || ASSNG(sym) || ADDRTKNG(sym) || SOCPTRG(sym) || ARGG(sym))
2789 continue;
2790 cp.val = DVL_CONVAL(dvl);
2791 nme = addnme(NT_VAR, sym, 0, (INT)0);
2792 nocp = 0;
2793 /* keep track of changes for this sym. */
2794 changes_sym = FALSE;
2795 for (use = 1; use < opt.useb.stg_avail; use++) {
2796 if (nme == USE_NM(use)) {
2797 if (!copy_const(use)) {
2798 changes = TRUE;
2799 changes_sym = TRUE;
2800 } else {
2801 /* found a use where constant has not been propagated */
2802 nocp++;
2803 }
2804 }
2805 }
2806 if (!nocp && changes_sym)
2807 DATACONSTP(sym, 1);
2808 }
2809 aux.dvl_avl = 0; /* ensure doesn't happen again */
2810 if (changes) {
2811 freearea(Q_AREA);
2812 cp.l_q = GET_Q_ITEM(cp.f_q);
2813 cp.l_q->next = Q_NULL;
2814 }
2815 }
2816
2817 for (df = FIRST_DEF; df <= opt.ndefs; df++) {
2818 if (!can_prop_fg(DEF_FG(df)))
2819 continue;
2820 if (DEF_CONST(df) && can_prop_def(df) && DEF_PRECISE(df)) {
2821 cp.l_q->next = GET_Q_ITEM(q);
2822 cp.l_q = q;
2823 cp.l_q->info = df;
2824 }
2825 }
2826
2827 while (cp.f_q != cp.l_q) {
2828 cp.f_q = cp.f_q->next;
2829 df = cp.f_q->info;
2830 nme = DEF_NM(df);
2831 df_ilt = DEF_STD(df);
2832 cp.val = DEF_RHS(df);
2833#if DEBUG
2834 if (OPTDBG(9, 32768))
2835 fprintf(gbl.dbgfil, "--- cand df:%d, val ili %d, %s\n", df, cp.val,
2836 getprint(NME_SYM(nme)));
2837#endif
2838 keep_def = is_sym_imp_live(nme) || XBIT(6, 0x2);
2839 for (du = DEF_DU(df); du; du = du->next) {
2840 int use;
2841
2842 use = du->use;
2843 if (!single_ud(use) || self_use(use) || !USE_PRECISE(use)) {
2844 keep_def = TRUE;
2845 continue;
2846 }
2847
2848 if (copy_const(use)) {
2849 keep_def = TRUE;
2850 }
2851 changes = TRUE;
2852 }
2853 }
2854
2855 freearea(Q_AREA);
2856 return changes;
2857}
