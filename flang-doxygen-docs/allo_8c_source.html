46/*
47 * The implementation of a normal allocte pre-dates memalign; to accomodate:
48 * + NULL is 0,
49 * + we want to align the area to improve cache utiltization
50 * the allocate mallocs a large enough space to whose address is massaged
51 * to give the desired alignment and the pointer to the malloc area is
52 * stored 1 pointer-element back from the returned address.
53 *
54 * USE_MEMALIGN says we're going to use memalign to support the use of our
55 * ALIGN= extension in ALLOCATE. This method coexists with the 'nornmal'
56 * method; therefore, a list is needed to keep track of ALIGN'd allocaale
57 * The allocation via memalign gives us the pointer returned to the user;
58 * we DO NOT use the 1 pointer-element idea. For alloc/deallocate, we
59 * must always be sure that the list is always checked/set before continuing
60 * to the normal method.
61 *
62 * This can all be simplified to just use memalign and never randomize the
63 * alignment of allocations (could impact apps the have automatic arrsys,
64 * e.g, bwaves.
65 */

67/* header structure on every allocated block */
68
69typedef struct ALLO_HDR ALLO_HDR;
70struct ALLO_HDR {
71 ALLO_HDR *next; /* list of allocated blocks */
72 char *area; /* address of area returned to caller */
73 /* NOTE:
74 * 01/04/2010 -- area may be overwritten by a pointer which
75 * locates the header. The address of the header is stored
76 * in the pointer-size location immediately preceeding the
77 * address returned to the caller. The basic idea is to
78 * deallocate without having to first search a list. */
79};

105/** \brief
106 * Allocate ALLO_HDR list */
107static void
108allhdr()
109{
110 char *p, *q;
111 long n;
112
113 MP_P(sem);
114 if (allo_list != (ALLO_HDR *)0) { /* check again */
115 MP_P(sem);
116 return;
117 }
118 p = getenv("F90_ALLOCATE_HDRS"); /* check environment */
119 if (p != NULL) {
120 num_hdrs = strtol(p, &q, 0);
121 if ((*q == 'k') || (*q == 'K')) {
122 num_hdrs *= 1024;
123 } else if ((*q == 'm') || (*q == 'M')) {
124 num_hdrs *= 1024 * 1024;
125 } else if ((*q == 'g') || (*q == 'G')) {
126 num_hdrs *= 1024 * 1024 * 1024;
127 }
128 }
129 n = 8; /* must be power of 2 */
130 while (n < num_hdrs) {
131 n = n << 1;
132 }
133 num_hdrs = n; /* allocate headers */
134 allo_list = (ALLO_HDR *)calloc(num_hdrs, sizeof(ALLO_HDR));
135 if (allo_list == (ALLO_HDR *)0) {
136 __abort(1, "No memory for allocate headers");
137 }
138
139 MP_V(sem);
140}

143/** \brief
144 * Return nonzero if addresses p1 and p2 are aligned with respect to a
145 * multiple of the length of the data type.
146 */
147int
148I8(__fort_ptr_aligned)(char *p1, dtype kind, int len, char *p2)
149{
150 __POINT_T diff, off;
151
152 off = diff = p1 - p2;
153 if (kind != __STR && kind != __DERIVED)
154 off >>= GET_DIST_SHIFTS(kind);
155 else
156 off /= len;
157 return (off * len == diff);
158}

160/** \brief
161 * Compute pointer-sized index offset to replace Cray pointer, return
162 * aligned address. offset assumes 1-based arrays.
163 */
164char *
165I8(__fort_ptr_offset)(char **pointer, __POINT_T *offset, char *base,
166 dtype kind, __CLEN_T len, char *area)
167{
168 __POINT_T off;
169 char *aligned;
170
171 if (ISPRESENT(offset)) {
172 if (ISPRESENT(pointer) && *pointer == base) {
173
174 /* offset is present but cray pointer is being used */
175
176 *offset = 0;
177 *pointer = area;
178 return area;
179 }
180
181 /* offset is present and being used */
182
183 off = (area >= base) ? area - base + len - 1 : base - area;
184 if (kind != __STR && kind != __DERIVED)
185 off >>= GET_DIST_SHIFTS(kind);
186 else
187 off /= len;
188
189 if (area < base)
190 off = -off;
191
192 *offset = off + 1;
193 aligned = base + off * len;
194#if defined(DEBUG)
195 if (__fort_test & DEBUG_ALLO)
196 printf("%d ptr_offset: area %p base %p + (%d - 1)*%lu = %p\n",
197 GET_DIST_LCPU, area, base, (int)*offset, len, aligned);
198#endif
199 } else
200 aligned = area;
201 if (ISPRESENT(pointer))
202 *pointer = aligned;
203 return aligned;
204}

229char *
230I8(__fort_alloc)(__INT_T nelem, dtype kind, size_t len, __STAT_T *stat,
231 char **pointer, __POINT_T *offset, char *base, int check,
232 void *(*mallocfn)(size_t))
233{
234 ALLO_HDR *p;
235 char *area;
236 size_t need, size, slop, sizeof_hdr;
237 __POINT_T off;
238 char msg[80];
239 char *p_env;
240
241#if (defined(_WIN64))
242#define ALN_LARGE
243#else
244#undef ALN_LARGE
245#endif
246/* always use the larger padding - used be for just win, but
247 * now it makes a bug difference on linux with newer (since Jul 2007)
248 * processor.
249 */
250#define ALN_LARGE
251
252 size_t ALN_MINSZ = 128000;
253 size_t ALN_UNIT = 64;
254 size_t ALN_MAXADJ = 4096;
255
256#define ALN_THRESH (ALN_MAXADJ / ALN_UNIT)
257 static size_t aln_n = 0;
258 static int env_checked = 0;
259 size_t myaln;
260
261 sizeof_hdr = AUTOASZ;
262
263 if (!env_checked) {
264 env_checked = 1;
265
266 p_env = getenv("F90_ALN_MINSZ");
267 if (p_env != NULL)
268 ALN_MINSZ = atol(p_env);
269
270 p_env = getenv("F90_ALN_UNIT");
271 if (p_env != NULL)
272 ALN_UNIT = atol(p_env);
273
274 p_env = getenv("F90_ALN_MAXADJ");
275 if (p_env != NULL)
276 ALN_MAXADJ = atol(p_env);
277 }
278
279 ALLHDR();
280
281 if (!ISPRESENT(stat))
282 stat = NULL;
283 if (!ISPRESENT(pointer))
284 pointer = NULL;
285 if (!ISPRESENT(offset))
286 offset = NULL;
287 need = (nelem <= 0) ? 0 : (size_t)nelem * len;
288 slop = 0;
289 /* SLOPPY COMMENT:
290 * here, we add some slop so we can align the eventual
291 * allocated address after adding the size of the ALLO_HDR.
292 * We are going to align to ASZ. Since we know that the malloc
293 * function returns something at least 8-byte aligned, we only
294 * should have to add (ASZ-8) slop.
295 * Actually, we shouldn't even have to do this if we know
296 * that the malloc function returns aligned results and
297 * skipping ALLO_HDR will keep it aligned */
298 if (nelem > 1 || need > 2 * sizeof_hdr)
299 slop = (offset && len > (ASZ - 8)) ? len : (ASZ - 8);
300 size = (sizeof_hdr + slop + need + ASZ - 1) & ~(ASZ - 1);
301 MP_P(sem);
302 if (size > ALN_MINSZ) {
303 myaln = aln_n;
304 size += ALN_UNIT * myaln;
305 if (aln_n < ALN_THRESH)
306 aln_n++;
307 else
308 aln_n = 0;
309 }
310 p = (size < need) ? NULL : (ALLO_HDR *)mallocfn(size);
311 MP_V(sem);
312 if (p == NULL) {
313 if (pointer)
314 *pointer = NULL;
315 if (offset)
316 *offset = 1;
317 if (stat) {
318 *stat = 1;
319 return NULL;
320 }
321 MP_P_STDIO;
322 sprintf(msg, "ALLOCATE: %lu bytes requested; not enough memory", need);
323 MP_V_STDIO;
324 __fort_abort(msg);
325 }
326 if (stat)
327 *stat = 0;
328 area = (char *)p + sizeof_hdr;
329 if (offset) {
330 off = area - base + len - 1;
331 if (kind != __STR && kind != __DERIVED)
332 off >>= GET_DIST_SHIFTS(kind);
333 else
334 off /= len;
335
336 *offset = off + 1;
337 area = base + off * len;
338#if defined(DEBUG)
339 if (__fort_test & DEBUG_ALLO)
340 printf("%d alloc: need %lu size %lu p %p area %p end %p"
341 " base %p offset %ld len %lu\n",
342 GET_DIST_LCPU, need, size, p, area, (char *)p + size - 1, base,
343 *offset, len);
344#endif
345 } else {
346 /* see SLOPPY COMMENT above */
347 if (nelem > 1 || need > 2 * sizeof_hdr)
348 area = ALIGNP(area, ASZ);
349#if defined(DEBUG)
350 if (__fort_test & DEBUG_ALLO)
351 printf("%d alloc: need %lu size %lu p %p area %p end %p\n",
352 GET_DIST_LCPU, need, size, p, area, (char *)p + size - 1);
353#endif
354 }
355 if (size > ALN_MINSZ)
356 area += ALN_UNIT * myaln;
357 XYZZYP(area, p);
358 if (pointer)
359 *pointer = area;
360 return area;
361}

755static void
756save_alloc(__POINT_T nelem, __INT_T len, char **pointer)
757{
758 if (savedalloc.valid >= 0) {
759 /* we aren't saving some old space here */
760 __INT8_T l;
761 l = nelem;
762 MP_P_ALLO;
763 if (savedalloc.valid >= 0 && l > 0) { /* check still valid */
764 /* now, save the most recently allocated space for later reuse */
765 l *= (len);
766 savedalloc.valid = 1;
767 savedalloc.pointer = *pointer;
768 savedalloc.len = l;
769 }
770 MP_V_ALLO;
771 }

774static void *
775use_alloc(__POINT_T nelem, __INT_T len)
776{
777 if (savedalloc.valid == -1) {
778 void *salp;
779 /* for allocate/free in a loop, see if we have a recently
780 * allocated and freed space that we can immediately reuse;
781 * don't do this with stat argument
782 */
783 __INT8_T l;
784 l = nelem;
785 if (l > 0)
786 l *= (len);
787 /* l holds the length, see fi the space is long enough, but not too long */
788 MP_P_ALLO;
789 if (savedalloc.valid != -1) { /* still valid (in case of threading) */
790 MP_V_ALLO;
791 } else {
792 if (l <= savedalloc.len && l > (savedalloc.len >> 1)) {
793 salp = savedalloc.pointer;
794 savedalloc.valid = 1; /* in use */
795 /* success, it's just long enough, use it */
796 MP_V_ALLO;
797 return salp;
798 } else {
799 char *pp;
800 pp = savedalloc.pointer;
801 savedalloc.valid = 0; /* not allocated */
802 savedalloc.pointer = NULL;
803 savedalloc.len = 0;
804 MP_V_ALLO; /* get out of the critical section */
805 /* failure; just free the space we had saved */
806 (void)I8(__fort_dealloc)(pp, (__STAT_T *)(ENTCOMN(0, 0)), __fort_free);
807 }
808 }
809 }
810 return NULL;
811}

813static void *
814reuse_alloc(__STAT_T *stat, char *area)
815{
816 if (savedalloc.pointer == area && savedalloc.pointer != NULL) {
817 MP_P_ALLO;
818 /* now test again inside the critical region */
819 if (savedalloc.pointer == area && savedalloc.pointer != NULL) {
820 if (!ISPRESENT(stat)) {
821 /* if this was the 'recently allocated' space, mark it as ready for
822 * reuse */
823 savedalloc.valid = -1; /* ready for reuse */
824 MP_V_ALLO;
825 return area;
826 }
827 savedalloc.valid = 0; /* will be free-ed */
828 savedalloc.pointer = NULL;
829 savedalloc.len = 0;
830 }
831 MP_V_ALLO;
832 }
833 return NULL;
834}
