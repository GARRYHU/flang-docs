315/** \brief Open ccff_file, write initial tags
316 *
317 * called from main()
318 */
319void
320ccff_open(const char *ccff_filename, const char *srcfile)
321{
322 char *cwd;
323 int cwdlen;
324 int i, slash;
325 ccff_file = fopen(ccff_filename, "wb");
326 if (ccff_file == NULL) {
327 return; /* give error message? */
328 }
329 xmlopenattrs2("ccff", "ccff", "version", CCFFVERSION, "xmlns",
330 "http://www.pgroup.com/ccff");
331 /* get the file name path */
332 slash = -1;
333 for (i = 0; srcfile[i] != '\0'; ++i) {
334 if (srcfile[i] == '/')
335 slash = i;
336#ifdef HOST_WIN
337 else if (srcfile[i] == '\\')
338 slash = i;
339#endif
340 }
341 xmlopen("source", "s");
342 if (slash >= 0) {
343 char *srcpath = (char *)malloc(strlen(srcfile) + 1);
344 strcpy(srcpath, srcfile);
345 srcpath[slash] = '\0';
346 xmlentity("sourcename", "sn", srcfile + slash + 1);
347 xmlentity("sourcepath", "sp", srcpath);
348 free(srcpath);
349 } else {
350 xmlentity("sourcename", "sn", srcfile);
351 xmlentity("sourcepath", "sp", ".");
352 }
353 cwdlen = 100;
354 cwd = (char *)malloc(cwdlen);
355 while ((void *)getcwd(cwd, cwdlen - 1) == NULL) {
356 cwdlen *= 2;
357 cwd = (char *)realloc(cwd, cwdlen);
358 }
359 xmlentity("sourcedir", "sd", cwd);
360 xmlclose("source", "s");
361 free(cwd);
362 unitstatus = 0; /* file open */
363} /* ccff_open */

573/*
574 * heap sort by line number
575 */
576static void
577_childsort(int l, int h)
578{
579 int m1, m2;
580 int c, c1, c2;
581 c = childlist[l];
582 while (1) {
583 m1 = l * 2 + 1;
584 if (m1 > h)
585 break; /* done */
586 c1 = childlist[m1];
587 m2 = l * 2 + 2;
588 if (m2 <= h) {
589 c2 = childlist[m2];
590 if (FIH_LINENO(c2) > FIH_LINENO(c1) ||
591 (FIH_LINENO(c2) == FIH_LINENO(c1) && c2 > c1)) {
592 m1 = m2;
593 c1 = c2;
594 }
595 }
596 if (FIH_LINENO(c1) > FIH_LINENO(c) ||
597 (FIH_LINENO(c1) == FIH_LINENO(c) && c1 > c)) {
598 childlist[l] = c1;
599 childlist[m1] = c;
600 l = m1;
601 } else {
602 break;
603 }
604 }
605} /* _childsort */

643/*
644 * all children of fihx appear in FIH_CHILD/FIH_NEXT linked list.
645 * sort them by line number
646 */
647static void
648fih_sort_children(int fihx)
649{
650 int child, nchildren, n;
651 if (FIH_CHILD(fihx) == 0)
652 return;
653 nchildren = 0;
654 for (child = FIH_CHILD(fihx); child; child = FIH_NEXT(child)) {
655 ++nchildren;
656 }
657 if (nchildren > childlistsize) {
658 childlistsize = nchildren + 100;
659 childlist = GETITEMS(CCFFAREA, int, childlistsize);
660 }
661 nchildren = 0;
662 for (child = FIH_CHILD(fihx); child; child = FIH_NEXT(child)) {
663 childlist[nchildren] = child;
664 ++nchildren;
665 }
666 /* heap sort */
667 for (n = nchildren / 2; n > 0; --n) {
668 _childsort(n - 1, nchildren - 1);
669 }
670 for (n = nchildren - 1; n > 0; --n) {
671 int c;
672 c = childlist[n];
673 childlist[n] = childlist[0];
674 childlist[0] = c;
675 _childsort(0, n - 1);
676 }
677 FIH_CHILD(fihx) = childlist[0];
678 for (n = 0; n < nchildren - 1; ++n) {
679 FIH_NEXT(childlist[n]) = childlist[n + 1];
680 }
681 FIH_NEXT(childlist[nchildren - 1]) = 0;
682} /* fih_sort_children */

744/*
745 * compare two MESSAGEs.
746 * keys:
747 * line number
748 * if sortorder==0 or sortorder==1
749 * if the same message type:
750 * message id
751 * varname, if any
752 * funcname, if any
753 * arguments
754 * if sortorder==1 || sortorder == 2
755 * insertion order
756 * if sortorder == 2
757 * suborder
758 */
759static int
760_messagecmp(MESSAGE *m1, MESSAGE *m2, int sortorder)
761{
762 int r;
763 ARGUMENT *a1, *a2;
764 if (m1->lineno > m2->lineno)
765 return +1;
766 if (m1->lineno < m2->lineno)
767 return -1;
768 if (m1->msgtype == m2->msgtype && sortorder != 2) {
769 r = strcmp(m1->msgid, m2->msgid);
770 if (r)
771 return r;
772 if (m1->varname && m2->varname) {
773 r = strcmp(m1->varname, m2->varname);
774 if (r)
775 return r;
776 } else if (m1->varname) {
777 return +1;
778 } else if (m2->varname) {
779 return -1;
780 }
781 if (m1->funcname && m2->funcname) {
782 r = strcmp(m1->funcname, m2->funcname);
783 if (r)
784 return r;
785 } else if (m1->funcname) {
786 return +1;
787 } else if (m2->funcname) {
788 return -1;
789 }
790 for (a1 = m1->args, a2 = m2->args; a1 && a2; a1 = a1->next, a2 = a2->next) {
791 if (a1->argstring && a2->argstring) {
792 r = strcmp(a1->argstring, a2->argstring);
793 if (r)
794 return r;
795 } else if (a1->argstring) {
796 return +1;
797 } else if (a2->argstring) {
798 return -1;
799 }
800 if (a1->argvalue && a2->argvalue) {
801 int v1, v2;
802 if (_numeric(a1->argvalue, &v1) && _numeric(a2->argvalue, &v2)) {
803 if (v1 > v2)
804 return +1;
805 if (v1 < v2)
806 return -1;
807 } else {
808 r = strcmp(a1->argvalue, a2->argvalue);
809 if (r)
810 return r;
811 }
812 } else if (a1->argvalue) {
813 return +1;
814 } else if (a2->argvalue) {
815 return -1;
816 }
817 }
818 } else if (!sortorder) {
819 /* when sortorder != 0, we are only comparing for equality,
820 * so just return a nonzero value to mean not equal */
821 return 1;
822 }
823 if (sortorder > 0) {
824 if (m1->order > m2->order)
825 return +1;
826 if (m1->order < m2->order)
827 return -1;
828 }
829 if (sortorder == 2) {
830 if (m1->suborder > m2->suborder)
831 return +1;
832 if (m1->suborder < m2->suborder)
833 return -1;
834 }
835 return 0;
836} /* _messagecmp */

838/*
839 * heap sort by line number
840 */
841static void
842_messagesort(int l, int h, int sortorder)
843{
844 int m1, m2, r;
845 MESSAGE *c, *c1, *c2;
846 c = messagelist[l];
847 while (1) {
848 m1 = l * 2 + 1;
849 if (m1 > h)
850 break; /* done */
851 c1 = messagelist[m1];
852 m2 = l * 2 + 2;
853 if (m2 <= h) {
854 c2 = messagelist[m2];
855 r = _messagecmp(c1, c2, sortorder);
856 if (r < 0) {
857 /* compare to c2 */
858 m1 = m2;
859 c1 = c2;
860 } else {
861 /* compare to c1 */
862 }
863 }
864 r = _messagecmp(c, c1, sortorder);
865 if (r < 0) {
866 messagelist[l] = c1;
867 messagelist[m1] = c;
868 l = m1;
869 } else {
870 break;
871 }
872 }
873} /* _messagesort */

875/*
876 * sort messages by line number
877 */
878static MESSAGE *
879sort_message_list(MESSAGE *msglist)
880{
881 int nmessages, n, prevn;
882 MESSAGE *mptr;
883 MESSAGE *newmsglist;
884 nmessages = 0;
885 for (mptr = msglist; mptr; mptr = mptr->next)
886 ++nmessages;
887 if (nmessages == 0)
888 return msglist;
889 if (nmessages > messagelistsize) {
890 messagelistsize = nmessages + 100;
891 messagelist = GETITEMS(CCFFAREA, MESSAGE *, messagelistsize);
892 }
893 nmessages = 0;
894 for (mptr = msglist; mptr; mptr = mptr->next)
895 messagelist[nmessages++] = mptr;
896 /* heap sort */
897 for (n = nmessages / 2; n > 0; --n)
898 _messagesort(n - 1, nmessages - 1, 1);
899 for (n = nmessages - 1; n > 0; --n) {
900 mptr = messagelist[n];
901 messagelist[n] = messagelist[0];
902 messagelist[0] = mptr;
903 _messagesort(0, n - 1, 1);
904 }
905 newmsglist = messagelist[0];
906 prevn = 0;
907 for (n = 1; n < nmessages; ++n) {
908 /* look for duplicate messages, with the same arguments, on the same line */
909 MESSAGE *mmptr;
910 mptr = messagelist[n];
911 mmptr = messagelist[n - 1];
912 if (_messagecmp(mptr, mmptr, 0) != 0) {
913 /* not a duplicate, include this message */
914 messagelist[prevn]->next = messagelist[n];
915 prevn = n;
916 /* if this message has the same message id, reset order to match */
917 if (strcmp(mptr->msgid, mmptr->msgid) == 0) {
918 mmptr->order = mptr->order;
919 mmptr->suborder = mptr->suborder + 1;
920 } else {
921 mmptr->suborder = 1;
922 }
923 }
924 }
925 if (prevn >= 0)
926 messagelist[prevn]->next = NULL;
927
928 /* resort by line number, sort order, suborder,
929 * after the duplicate messages are all removed */
930 nmessages = 0;
931 for (mptr = newmsglist; mptr; mptr = mptr->next) {
932 messagelist[nmessages] = mptr;
933 ++nmessages;
934 }
935 /* heap sort */
936 for (n = nmessages / 2; n > 0; --n) {
937 _messagesort(n - 1, nmessages - 1, 2);
938 }
939 for (n = nmessages - 1; n > 0; --n) {
940 mptr = messagelist[n];
941 messagelist[n] = messagelist[0];
942 messagelist[0] = mptr;
943 _messagesort(0, n - 1, 2);
944 }
945 mptr = messagelist[0];
946 for (n = 1; n < nmessages; ++n) {
947 mptr->next = messagelist[n];
948 mptr = messagelist[n];
949 }
950 mptr->next = NULL;
951 newmsglist = messagelist[0];
952 return newmsglist;
953} /* sort_message_list */

991{

1008 s2 = arg2->argstring;

1010 return false;

1015 listarg = 1; /* list arguments may differ */

1019 }

1023 return false;
1024 }
1025 }

1027 return false;
