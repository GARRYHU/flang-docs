11/** \file
12 * \brief General-Purpose Hash Tables.
13 *
14 * These two hash table implementations store void pointers as keys, using
15 * user-provided functions for hashing and equality testing. The void pointers
16 * are not interpreted by the hash table implementation, except:
17 *
18 * - The void pointers are passed to the provided hash() and equals() functions.
19 *
20 * - The sentinel values NULL and (void*)~0UL have special meanings and cannot
21 * be used as keys.
22 *
23 * The equality function should return non-zero for equal hash keys. At a
24 * minimum, the provided functions must satisfy:
25 *
26 * equals(a, b) ==> hash(a) = hash(b)
27 *
28 * The hash() function should avoid clustering in the low bits of the hash
29 * value.
30 *
31 * A NULL equals() function is equivalent to a function returning a == b, but
32 * faster.
33 */
34
35typedef unsigned hash_value_t;
36typedef const void *hash_key_t;
37
38typedef hash_value_t (*hash_function_t)(hash_key_t);
39typedef int (*hash_equality_t)(hash_key_t, hash_key_t);

41typedef struct hash_functions_ {
42 hash_function_t hash;
43 hash_equality_t equals;
44} hash_functions_t;

208/*
209 * Helpers for computing hash values for composite data structures, using a
210 * hash accumulator. These macros implement parts of the Jenkins hash function.
211 *
212 * See also the functions string_hash() and direct_hash() in hash.c.
213 *
214 * hash_value_T hash_function(const struct foo *data)
215 * {
216 * hash_accu_t hacc = HASH_ACCU_INIT;
217 *
218 * HASH_ACCU_ADD(hacc, data->int_member);
219 * HASH_ACCU_ADD(hacc, data->pointer_member);
220 * HASH_ACCU_FINISH(accu);
221 * return HASH_ACCU_VALUE(accu);
222 * }
223 *
224 */
225
226typedef struct hash_accu_ {
227 hash_value_t a;
228} hash_accu_t;
229
230#define HASH_ACCU_INIT \
231 { \
232 0 \
233 }
234
235#define HASH_ACCU_ADD(accu, data) \
236 do { \
237 (accu).a += (hash_value_t)(data); \
238 (accu).a += (accu).a << 10; \
239 (accu).a ^= (accu).a >> 6; \
240 } while (0)
241
242#define HASH_ACCU_FINISH(accu) \
243 do { \
244 (accu).a += (accu).a << 3; \
245 (accu).a ^= (accu).a >> 11; \
246 (accu).a += (accu).a << 15; \
247 } while (0)
248
249#define HASH_ACCU_VALUE(accu) ((accu).a + 0)
