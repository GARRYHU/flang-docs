97/** \brief Hoist invariant communication expressions out of loops. */
98void
99comm_invar(void)
100{
101 int lp;
102
103 hlopt_init(HLOPT_INDUC);
104
105#if DEBUG
106 if (DBGBIT(43, 64)) {
107 fprintf(gbl.dbgfil, "----------- STDs before invariant hoisting\n");
108 dump_std();
109 }
110#endif
111
112 get_loopcounts();
113
114 clean_flags();
115
116 for (lp = LP_CHILD(0); lp; lp = LP_SIBLING(lp))
117 hoist_loop(lp);
118
119 mk_ztrips(0);
120#if DEBUG
121 if (DBGBIT(43, 64)) {
122 fprintf(gbl.dbgfil, "----------- STDs after invariant hoisting\n");
123 dump_std();
124 }
125#endif
126
127 if (pastCounts) {
128 FREE(pastCounts);
129 pastCounts = NULL;
130 }
131 hlopt_end(HLOPT_INDUC, 0);
132}

134/* Clear flags & fields in STDs & ASTs that will be used for hoisting. */
135static void
136clean_flags(void)
137{
138 int std;
139
140 for (std = STD_FIRST; std; std = STD_NEXT(std)) {
141 STD_MOVED(std) = FALSE;
142 A_CONDP(STD_AST(std), 0);
143 }
144}

146/* For each loop lp, set pastCounts[lp] to its count. */
147static void
148get_loopcounts(void)
149{
150 int lp;
151
152 NEW(pastCounts, int, opt.nloops + 1);
153
154 for (lp = 1; lp <= opt.nloops; lp++) {
155 /* Mark invariant ASTs within lp. */
156 invariant_nomotion(lp);
157 pastCounts[lp] = get_loop_count(lp);
158 end_loop_count();
159 invariant_unmarkv();
160 }
161}

163/* Hoist invariant communication expressions out of loop #lp. Return TRUE
164 * if lp is a candidate for hoisting. */
165static LOGICAL
166hoist_loop(int lp)
167{
168 int lpi;
169 LOGICAL bHoistInner = TRUE;
170 LOGICAL bx49;
171
172 for (lpi = LP_CHILD(lp); lpi; lpi = LP_SIBLING(lpi))
173 bHoistInner &= hoist_loop(lpi);
174
175 if (!bHoistInner)
176 /* Don't analyze if an inner loop can't be analyzed. */
177 return FALSE;
178 if (LP_MEXITS(lp))
179 /* Don't analyze if lp contains multiple exits. */
180 return TRUE;
181 if (!pastCounts[lp])
182 /* Loop not countable. */
183 return TRUE;
184 if (contains_alloc(lp))
185 /* Don't analyze if user-defined arrays are allocated. */
186 return FALSE;
187
188 /* Check for a disabling directive. */
189 open_pragma(FG_LINENO(LP_HEAD(lp)));
190 bx49 = (XBIT(49, 0x20000) != 0);
191 close_pragma();
192 if (bx49)
193 return TRUE;
194
195 /* Mark invariant ASTs within lp. */
196 invariant_nomotion(lp);
197
198 fgPre = fgPost = 0;
199
200 /* Hoist communication ASTs. */
201 hoist_comms(lp);
202
203 invariant_unmarkv();
204 return TRUE;
205}

207/* Return TRUE if loop #lp contains an ALLOCATE/DEALLOCATE statement of
208 * a user-defined array. */
209static LOGICAL
210contains_alloc(int lp)
211{
212 int fg;
213 int std, stdend;
214 int ast, sptr;
215
216 for (fg = LP_FG(lp); fg; fg = FG_NEXT(fg)) {
217 std = FG_STDFIRST(fg);
218 if (!std)
219 continue;
220 stdend = STD_NEXT(FG_STDLAST(fg));
221 for (; std != stdend; std = STD_NEXT(std)) {
222 ast = STD_AST(std);
223 if (A_TYPEG(ast) != A_ALLOC)
224 continue;
225 ast = A_SRCG(ast);
226 if (A_TYPEG(ast) == A_SUBSCR)
227 ast = A_LOPG(ast);
228 switch (A_TYPEG(ast)) {
229 case A_ID:
230 sptr = A_SPTRG(ast);
231 break;
232 case A_MEM:
233 sptr = A_SPTRG(A_MEMG(ast));
234 break;
235 default:
236 interr("contains_alloc: ID operand not found", lp, 4);
237 }
238 if (!CCSYMG(sptr))
239 return TRUE;
240 }
241 }
242 return FALSE;
243}

245/* Hoist invariant communication ASTs within loop #lp into a
246 * pre-header (and trailer) located at fgPre (and fgPost). */
247static void
248hoist_comms(int lp)
249{
250 int fg, fgtail, fgend;
251 int std, stdend, stdnext;
252 int par;
253
254 bMovedComm = FALSE;
255 bMovedAssn = FALSE;
256 fgtail = LP_TAIL(lp);
257 fgend = FG_LNEXT(fgtail);
258 fg = LP_HEAD(lp);
259 par = STD_PAR(FG_STDFIRST(fg));
260 for (; fg != fgend; fg = FG_LNEXT(fg)) {
261 if (FG_LOOP(fg) != lp)
262 continue;
263 if (!is_dominator(fg, fgtail))
264 continue;
265 std = FG_STDFIRST(fg);
266 if (!std)
267 continue;
268 stdend = STD_NEXT(FG_STDLAST(fg));
269 stdnext = STD_NEXT(std);
270 for (; std != stdend; std = stdnext, stdnext = STD_NEXT(stdnext)) {
271 if (par != STD_PAR(std))
272 continue;
273 if (!STD_DELETE(std))
274 hoist_std(fg, std, lp);
275 }
276 }
277
278 if (!bMovedComm && !bMovedAssn)
279 return;
280
281 /* Clear the flags. */
282 std = FG_STDFIRST(fgPre);
283 if (std) {
284 stdend = STD_NEXT(FG_STDLAST(fgPre));
285 for (; std != stdend; std = STD_NEXT(std))
286 STD_MOVED(std) = FALSE;
287 }
288
289 if (bMovedComm && flg.hpf)
290 ccff_info(MSGFTN, "FTN012", 1, FG_LINENO(LP_HEAD(lp)),
291 "Invariant communication calls hoisted out of loop", NULL);
292 if (bMovedAssn)
293 ccff_info(MSGFTN, "FTN013", 1, FG_LINENO(LP_HEAD(lp)),
294 "Invariant assignments hoisted out of loop", NULL);
295}

297/* Return TRUE if the A_HALLOBNDS at allo_std is unique in lp. That is,
298 no other A_HALLOBNDS in the loop allocates/deallocates the same symbol. */
299static LOGICAL
300unique_allo(int allo_std, int allo_ft, int lp)
301{
302 int fg;
303 int std, stdend;
304 int ast, ft;
305
306 for (fg = LP_FG(lp); fg; fg = FG_NEXT(fg)) {
307 std = FG_STDFIRST(fg);
308 if (!std)
309 continue;
310 stdend = STD_NEXT(FG_STDLAST(fg));
311 for (; std != stdend; std = STD_NEXT(std)) {
312 if (std == allo_std)
313 continue;
314 ast = STD_AST(std);
315 if (A_TYPEG(ast) != A_HALLOBNDS)
316 continue;
317 ft = A_OPT1G(ast);
318 if (!ft)
319 continue;
320 if (FT_ALLOC_SPTR(ft) == FT_ALLOC_SPTR(allo_ft))
321 return FALSE;
322 }
323 }
324 return TRUE;
325}

327/* Hoist an invariant communication AST at STD std to fgPre. STD std occurs
328 * within node #fg. Set bMovedComm or bMovedAssn to TRUE when hoisting
329 * occurs. */
330static void
331hoist_std(int fg, int std, int lp)
332{
333 int ast, ast1, astcomm;
334 int ft;
335 int stdfree;
336 int nme;
337 int ndim, dim;
338 int asd;
339
340 if (!std)
341 return;
342
343 ast = STD_AST(std);
344
345 if (A_CONDG(ast) && !is_invariant(A_CONDG(ast)))
346 return;
347
348 if (A_TYPEG(ast) == A_ASN && A_TYPEG(A_DESTG(ast)) == A_ID
349 && STYPEG(A_SPTRG(A_DESTG(ast))) == ST_VAR) {
350 /* Hoist unique scalar assignments with invariant right-hand sides,
351 * which may be allocatable array bounds. */
352 nme = A_NMEG(A_DESTG(ast));
353 if (is_sym_invariant_safe(nme, lp) &&
354 check_outer_defs(std, FG_LOOP(fg), nme) &&
355 check_inner_uses(std, FG_LOOP(fg), nme) && is_invariant(A_SRCG(ast)) &&
356 is_invariant_base(nme)) {
357 invariant_mark(A_DESTG(ast), INV);
358 NME_STL(nme) = 0;
359 move_invar_std(std, fg);
360 bMovedAssn = TRUE;
361 return;
362 }
363 }
364
365 if (A_TYPEG(ast) == A_ASN)
366 astcomm = A_SRCG(ast);
367 else
368 astcomm = ast;
369 ft = A_OPT1G(astcomm);
370
371 switch (A_TYPEG(astcomm)) {
372 case A_HALLOBNDS:
373 if (!ft || !unique_allo(std, ft, lp) || !has_invar_subs(A_LOPG(astcomm)))
374 return;
375 move_invar_std(std, fg);
376 stdfree = mk_dealloc_stub(A_CONDG(ast));
377 FT_ALLOC_FREE(ft) = stdfree;
378 bMovedComm = TRUE;
379 return;
380 case A_HSECT:
381 if (!ft || !has_invar_subs(A_LOPG(astcomm)) ||
382 !check_moved(FT_SECT_ALLOC(ft)))
383 return;
384 move_invar_std(std, fg);
385 stdfree = mk_dealloc_stub(A_CONDG(ast));
386 FT_SECT_FREE(ft) = stdfree;
387 bMovedComm = TRUE;
388 return;
389 case A_HLOCALIZEBNDS:
390 case A_HCYCLICLP:
391 if (!ft || !is_invariant(A_ITRIPLEG(astcomm)))
392 return;
393 move_invar_std(std, fg);
394 bMovedComm = TRUE;
395 return;
396 case A_HCOPYSECT:
397 if (!ft || !has_invar_subs(A_SRCG(astcomm)) ||
398 !check_moved(FT_CCOPY_ALLOC(ft)) || !check_moved(FT_CCOPY_SECTL(ft)) ||
399 !check_moved(FT_CCOPY_SECTR(ft)))
400 return;
401 move_invar_std(std, fg);
402 stdfree = mk_dealloc_stub(A_CONDG(ast));
403 FT_CCOPY_FREE(ft) = stdfree;
404 bMovedComm = TRUE;
405 return;
406 case A_HCSTART:
407 assert(A_TYPEG(A_SRCG(astcomm)) == A_SUBSCR,
408 "hoist_std: source operand not subscript", ast, 4);
409 assert(A_TYPEG(A_DESTG(astcomm)) == A_SUBSCR,
410 "hoist_std: dest operand not subscript", ast, 4);
411 if (!ft || !is_invariant(A_SRCG(astcomm)) ||
412 !is_invariant(A_DESTG(astcomm)) || !check_moved(FT_CSTART_COMM(ft)) ||
413 !check_moved(FT_CSTART_SECTL(ft)) || !check_moved(FT_CSTART_SECTR(ft)))
414 return;
415 move_invar_std(std, fg);
416 stdfree = mk_dealloc_stub(A_CONDG(ast));
417 FT_CSTART_FREE(ft) = stdfree;
418 bMovedComm = TRUE;
419 FT_CSTART_INVMVD(ft) = 1;
420 return;
421 case A_HOVLPSHIFT:
422 if (!ft || !has_invar_subs(A_SRCG(astcomm)) ||
423 (FT_SHIFT_BOUNDARY(ft) && !is_invariant(FT_SHIFT_BOUNDARY(ft))))
424 return;
425 move_invar_std(std, fg);
426 stdfree = mk_dealloc_stub(A_CONDG(ast));
427 FT_SHIFT_FREE(ft) = stdfree;
428 bMovedComm = TRUE;
429 return;
430 case A_HSCATTER:
431 case A_HGATHER:
432 if (!ft)
433 return;
434 ndim = ASD_NDIM(A_ASDG(FT_CGATHER_VSUB(ft)));
435 if (!has_invar_subs(FT_CGATHER_VSUB(ft)) ||
436 !has_invar_subs(FT_CGATHER_NVSUB(ft)) ||
437 (FT_CGATHER_MASK(ft) && !is_invariant(FT_CGATHER_MASK(ft))) ||
438 !is_invariant_list(ndim, &FT_CGATHER_V(ft, 0)) ||
439 !check_moved(FT_CGATHER_SECTVSUB(ft)) ||
440 !check_moved(FT_CGATHER_SECTNVSUB(ft)) ||
441 !check_moved(FT_CGATHER_SECTM(ft)) ||
442 !check_moved_list(ndim, &FT_CGATHER_SECTV(ft, 0)))
443 return;
444 move_invar_std(std, fg);
445 stdfree = mk_dealloc_stub(A_CONDG(ast));
446 FT_CGATHER_FREE(ft) = stdfree;
447 bMovedComm = TRUE;
448 return;
449 case A_HGETSCLR:
450 if (!is_invariant(A_SRCG(astcomm)))
451 return;
452 move_invar_std(std, fg);
453 bMovedComm = TRUE;
454 return;
455 case A_HOWNERPROC:
456 ast1 = A_LOPG(astcomm);
457 assert(A_TYPEG(ast1) == A_SUBSCR, "hoist_std: missing array ref", astcomm,
458 4);
459 asd = A_ASDG(ast1);
460 dim = CONVAL2G(A_SPTRG(A_DIMG(astcomm)));
461 if (!is_invariant(ASD_SUBS(asd, dim)))
462 return;
463 move_invar_std(std, fg);
464 bMovedComm = TRUE;
465 return;
466 default:
467 return;
468 }
469}

471/* Return TRUE if every definition of nme that is not at std occurs in
472 * a parent of loop lp. */
473static LOGICAL
474check_outer_defs(int std, int lp, int nme)
475{
476 int def;
477
478 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
479 if (DEF_STD(def) == std)
480 continue;
481 if (is_parent_loop(lp, FG_LOOP(DEF_FG(def))))
482 return FALSE;
483 }
484 return TRUE;
485}

487/* Return TRUE if every use of nme within loop lp is not reached by
488 * a def of nme other than the def at STD std. */
489static LOGICAL
490check_inner_uses(int std, int lp, int nme)
491{
492 int def;
493 DU *du;
494 int use;
495
496 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
497 if (DEF_STD(def) == std)
498 continue;
499 for (du = DEF_DU(def); du; du = du->next) {
500 use = du->use;
501 if (is_parent_loop(lp, FG_LOOP(USE_FG(use))))
502 return FALSE;
503 }
504 }
505 return TRUE;
506}

508/*
509 * return FALSE if this NME is for a BASED symbol, and the
510 * base pointer (from MIDNUM) is not already marked invariant
511 */
512static LOGICAL
513is_invariant_base(int nme)
514{
515 int sym;
516 sym = NME_SYM(nme);
517 if (SCG(sym) == SC_BASED && MIDNUMG(sym) > NOSYM &&
518 !is_invariant(mk_id(MIDNUMG(sym))))
519 return FALSE;
520 return TRUE;
521} /* is_invariant_base */

523/* Return TRUE if lpParent is a parent loop of lpChild, or if they are
524 * identical. */
525static LOGICAL
526is_parent_loop(int lpParent, int lpChild)
527{
528 int lpo;
529
530 for (lpo = lpChild; lpo; lpo = LP_PARENT(lpo))
531 if (lpo == lpParent)
532 return TRUE;
533 return FALSE;
534}

536/* Return TRUE if the subscript expressions of AST ast are all invariant. */
537static LOGICAL
538has_invar_subs(int ast)
539{
540 int asd;
541 int ndims, sub;
542
543 if (!ast)
544 return TRUE;
545
546 if (A_TYPEG(ast) == A_ID) {
547 assert(A_SHAPEG(ast), "has_invar_subs: ast has no shape", ast, 4);
548 return TRUE;
549 }
550 while (A_TYPEG(ast) != A_ID) {
551 switch (A_TYPEG(ast)) {
552 case A_SUBSCR:
553 asd = A_ASDG(ast);
554 ndims = ASD_NDIM(asd);
555 for (sub = 0; sub < ndims; sub++) {
556 if (!is_invariant(ASD_SUBS(asd, sub)))
557 return FALSE;
558 }
559 ast = A_LOPG(ast);
560 break;
561 case A_MEM:
562 ast = A_PARENTG(ast);
563 break;
564 default:
565 interr("reference not ID,SUBSCR,MEM", ast, 4);
566 }
567 }
568 return TRUE;
569}

595/* Return TRUE if std has been moved. */
596static LOGICAL
597check_moved(int std)
598{
599 if (!std)
600 return TRUE;
601 if (STD_DELETE(std))
602 return FALSE;
603 return STD_MOVED(std) != 0;

606/* Move the invariant AST at STD #std within node #fg to the last statement
607 * in the preheader of fg's loop. Set the std's MOVED flag. */
608static void
609move_invar_std(int std, int fg)
610{
611 int stdprev;
612 int astCont, astStmt, astCond;
613
614 mk_loop_hdtl(FG_LOOP(fg));
615
616 rdilts(fg);
617 stdprev = STD_PREV(std);
618 if (STD_LABEL(std)) {
619 astCont = mk_stmt(A_CONTINUE, 0);
620 stdprev = add_stmt_after(astCont, stdprev);
621 STD_LABEL(stdprev) = STD_LABEL(std);
622 STD_LABEL(std) = 0;
623 STD_LINENO(stdprev) = STD_LINENO(std);
624 }
625 STD_NEXT(stdprev) = STD_NEXT(std);
626 STD_PREV(STD_NEXT(std)) = stdprev;
627 wrilts(fg);
628
629 rdilts(fgPre);
630 stdprev = STD_LAST;
631 STD_NEXT(stdprev) = std;
632 STD_PREV(std) = stdprev;
633 STD_NEXT(std) = 0;
634 STD_LAST = std;
635 wrilts(fgPre);
636
637 STD_MOVED(std) = TRUE;
638 astStmt = STD_AST(std);
639 astCond = mk_count_test(A_CONDG(astStmt), pastCounts[FG_LOOP(fg)]);
640 A_CONDP(astStmt, astCond);
641 /*
642 * This is where fgPost needs be associated with astCond & saved
643 * in a data structure for mk_ztrips(int lp). Probably need to
644 * repurpose A_COND.
645fprintf(STDERR,
646"XXX lp:%d, fgPre:%d, fgPost:%d, fgPost_stdfirst:%d\n",
647FG_LOOP(fg), fgPre, fgPost, FG_STDFIRST(fgPost));
648 */
649}

651/* If necessary create new preheader and trailer nodes for loop #lp,
652 * and set fgPre & fgPost to these nodes. */
653static void
654mk_loop_hdtl(int lp)
655{
656 int ast;
657 int std;
658
659 if (fgPre)
660 return;
661
662 fgPre = add_fg(FG_LPREV(LP_HEAD(lp)));
663 opt.pre_fg = fgPre;
664 FG_FT(fgPre) = TRUE;
665 std = FG_STDFIRST(fgPre);
666 if (!std) {
667 /* Node empty; add a CONTINUE statement. */
668 rdilts(fgPre);
669 ast = mk_stmt(A_CONTINUE, 0);
670 std = add_stmt_after(ast, 0);
671 wrilts(fgPre);
672 }
673 FG_LINENO(fgPre) = STD_LINENO(std) = FG_LINENO(LP_HEAD(lp));
674 add_loop_preheader(lp);
675
676 add_single_loop_exit(lp);
677 fgPost = FG_LNEXT(LP_TAIL(lp));
678}

680/** \brief If necessary create new preheader and trailer nodes for loop \#lp,
681 and set fgPre & fgPost to these nodes. */
682void
683add_loop_hd(int lp)
684{
685 int ast;
686 int std;
687
688 fgPre = add_fg(FG_LPREV(LP_HEAD(lp)));
689 opt.pre_fg = fgPre;
690 FG_FT(fgPre) = TRUE;
691 std = FG_STDFIRST(fgPre);

737/*
738 * Create zero-trip tests around hoisted communication statements within
739 * loop lp.
740 */
741static void
742mk_ztrips(int lp)
743{
744 int lpi;
745 int fg;
746 int std, stdend;
747 int ast, astCond, astNewCond;
748 int *pastConds = NULL;
749 int maxconds = opt.nloops + 1;
750 int nconds;
751
752 for (lpi = LP_CHILD(lp); lpi; lpi = LP_SIBLING(lpi))
753 mk_ztrips(lpi);
754
755 NEW(pastConds, int, maxconds);
756 for (fg = LP_FG(lp); fg; fg = FG_NEXT(fg)) {
757 nconds = 0;
758 std = FG_STDFIRST(fg);
759 if (!std)
760 continue;
761 stdend = STD_NEXT(FG_STDLAST(fg));
762 for (; std != stdend; std = STD_NEXT(std)) {
763 astCond = A_CONDG(STD_AST(std));
764 astNewCond = astCond;
765 for (; nconds; nconds--) {
766 if (astCond)
767 astNewCond = remove_cond(astCond, pastConds[nconds - 1]);
768 else
769 astNewCond = astCond;
770 if (astNewCond != astCond)
771 /* ...new condition contained in old. */
772 break;
773 /* ...new condition not in old, so terminate old condition. */
774 ast = mk_stmt(A_ENDIF, 0);
775 add_stmt_before(ast, std);
776 }
777 if (!astCond || !astNewCond)
778 continue;
779 ast = mk_stmt(A_IFTHEN, 0);
780 A_IFEXPRP(ast, astNewCond);
781 add_stmt_before(ast, std);
782 pastConds[nconds++] = astCond;
783 assert(nconds <= opt.nloops, "mk_ztrips: too many conditions", lp, 4);
784 }
785 /*
786 * f20730 - hoisting of invariant assignments is suboptimal for
787 * reaching defs. Hoisting an invariant assignment from a loop
788 * occurs as follows:
789 *
790 * the loop
791 * ---------
792 * do __the_loop__
793 * XX = __invariant_expression__
794 * __work__ << uses XX>
795 * enddo
796 *
797 * the hoist
798 * ---------
799 * if (zero-trip-test) then
800 * XX = __invariant_expression__
801 * endif
802 * do __the_loop__
803 * __work__ <<< uses XX>>>
804 * enddo
805 *
806 * this needs to be
807 * ----------------
808 * if (zero-trip-test) then
809 * XX = __invariant_expression__
810 * do __the_loop__ !!! and somehow elide the backend's ZT
811 * __work__ << uses XX>
812 * enddo
813 * endif
814 *
815 * THE PROBLEM
816 * -----------
817 * IF there exists a def of XX prior to the original loop, that def
818 * now reaches the XX within the transformed loop (the hoisted
819 * assignment is now guarded). Without the hoisting, XX was killed
820 * by the def of XX in the loop. This can lead to false uses of
821 * XX -- for WRF, this was especially bad given that the XX was
822 * considered a live-out of a loop, which inhibited vectorizing
823 * that loop.
824 *
825 * To correctly place the ENDIF, move_invar_std() needs to associate
826 * A_COND with fgPost (FG_STDFIRST(fgPost) would be the correct
827 * insert point for the ENDIF.
828 *
829 * AT THIS TIME, it is not be worth the effort -- the backend will
830 * still hoist the RHS within the correct zero-trip context.
831 */
832 for (; nconds; nconds--) {
833 ast = mk_stmt(A_ENDIF, 0);
834 add_stmt_before(ast, std);
835 }
836 }
837 FREE(pastConds);
838}

840/* Remove conjunction astSrc from conjunction astTarg. If astSrc not in
841 * astTarg, return astTarg. If astSrc = astTarg, return 0. */
842static int
843remove_cond(int astTarg, int astSrc)
844{
845 int astNewl, astNewr, astNew;
846
847 if (astSrc == astTarg)
848 return 0;
849 if (A_TYPEG(astTarg) != A_BINOP || A_OPTYPEG(astTarg) != OP_LAND)
850 return astTarg;
851 astNewl = remove_cond(A_LOPG(astTarg), astSrc);
852 if (!astNewl) {
853 astNewr = remove_cond(A_ROPG(astTarg), astSrc);
854 return astNewr;
855 }
856 astNewr = remove_cond(A_ROPG(astTarg), astSrc);
857 if (!astNewr) {
858 astNewl = remove_cond(A_LOPG(astTarg), astSrc);
859 return astNewl;
860 }
861 if (astNewl == A_LOPG(astTarg) && astNewr == A_ROPG(astTarg))
862 astNew = astTarg;
863 else
864 astNew = mk_binop(OP_LAND, astNewl, astNewr, DT_LOG);
865 return astNew;
866}
