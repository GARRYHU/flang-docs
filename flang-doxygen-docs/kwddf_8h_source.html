8/* define structures and data for keyword processing: */

10typedef struct {
11 const char *keytext; /* keyword text in lower case */
12 int toktyp; /* token id (as used in parse tables) */
13 LOGICAL nonstandard; /* TRUE if nonstandard (extension to f90) */
14} KWORD;

16typedef struct {
17 int kcount; /* number of keywords in this table */
18 KWORD *kwds; /* pointer to first in array of KWORD */
19 /* the following members are filled in by init_ktable() to record
20 * the indices of the first and last keywords beginning with a
21 * certain letter. If first[] is zero, there does not exist a
22 * keyword which begins with the respective letter. A nonzero
23 * value is the index into the keyword table.
24 */
25 short first[26]; /* indexed by ('a' ... 'z') - 'a' */
26 short last[26]; /* indexed by ('a' ... 'z') - 'a' */
27} KTABLE;

29/* NOTE: When entering keywords in the tables, two or more consecutive tokens
30 * that can be seen in the grammar should be combined into a single token.
31 * This is because the keyword search routine is not designed to extract two
32 * or more tokens from a single identifier. For example, consider the
33 * statement DO WHILE ( <expr> ). One way of looking at the statement is
34 * that it begins with the two keywords DO and WHILE. The routine alpha
35 * sees the name DOWHILE; relevant tokens are "do", "doublecomplex", and
36 * "doubleprecision". When the keyword search routine looks at "dowhile",
37 * it determines that it should look AFTER "doubleprecision" --- the character
38 * 'w' is greater than 'u'. Consequently, no keyword, in particular DO, is
39 * found. For the statement, DO 10 WHILE ( <expr> ), the keyword DO is
40 * found in "do10while"; it is determined that it must look BEFORE
41 * "doublecomplex" ('1' < 'u'). DO statements of the form,
42 * DO <var> =
43 * are not a problem because of the '='; the keyword DO is found as a special
44 * case in routine alpha since there is an "exposed" equals sign.
45 *
46 * If a keyword is a prefix of another keyword, it is possible that the
47 * keyword search routine will not find the 'shorter' prefix of an identifier.
48 * E.g., real and realign are keywords
49 * The identifier realpi will not be found since 'realpi' > 'realign';
50 * the search routine searches past realign instead of before.
51 * For cases like this, the keyword table must only contain the 'shorter'
52 * prefix; it's up to alpha() to determine look for the additional keywords;
53 * e.g., 'realign' is 'real' followed by 'ign'.
54 *
55 */
56
57/*
58 * When the input is freeform, certain keywords may contain blanks.
59 * For those whose 'first' ident is also a keyword, the processing in
60 * alpha() will search ahead to determine if what follows is an ident
61 * which can be combined with 'first' to form a keyword; e.g.,
62 * end if, end program, etc.
63 * There are a few whose 'first' ident is not a keyword. For these,
64 * define special macros (TKF_ ...) and initialize in the keyword
65 * table and let alpha() do the rest. (NOTE that 0 is returned by
66 * keyword() if a name is not found in a keyword table).
67 */
