21int
22mk_ceil_div(int num, int den)
23{
24 /* compute CD(num, den) = (num + den - 1) / den */
25 int one = mk_cval(1, DT_INT);
26 int a;
27
28 if (den == 0 || den == one)
29 return num;
30 a = opt_binop(OP_ADD, num, den, DT_INT);
31 a = opt_binop(OP_SUB, a, one, DT_INT);
32 a = opt_binop(OP_DIV, a, den, DT_INT);
33 return opt_ast(a);
34}

62int
63mk_size(int low, int high)
64{
65 int t;
66
67 if (low == 0)
68 low = mk_cval(1, DT_INT);
69 if (high == 0)
70 high = low;
71 if (A_TYPEG(low) == A_CNST && A_TYPEG(high) == A_CNST) {
72 t = get_int_cval(A_SPTRG(high)) - get_int_cval(A_SPTRG(low)) + 1;
73 return mk_cval(t, DT_INT);
74 }
75 t = opt_binop(OP_ADD, opt_binop(OP_SUB, high, low, DT_INT),
76 mk_cval(1, DT_INT), DT_INT);
77 return opt_ast(t);
78}

80int
81search_basearr(int ast)
82{
83 /* search the ast for the symbol describing the base array. For
84 * now, ast can be either subscript or ID. Eventually, we will
85 * need to handle records */
86
87 if (A_TYPEG(ast) == A_ID)
88 return A_SPTRG(ast);
89 if (A_TYPEG(ast) == A_SUBSCR)
90 return search_basearr(A_LOPG(ast));
91 interr("search_basearr: can't find basearr", ast, 3);
92 return 0;
93}
