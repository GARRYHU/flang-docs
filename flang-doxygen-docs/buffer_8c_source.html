46/* open */
47
48void
49__fort_zopen(char *path)
50{
51 int ioproc;
52 int n;
53 int s;
54
55 if (buf == (char *)0) {
56 buf = __fort_gmalloc(MAXBUF);
57 }
58 ioproc = GET_DIST_IOPROC;
59 if (GET_DIST_LCPU != ioproc) {
60 bufp = buf + sizeof(int);
61 *((int *)buf) = 0;
62 bufe = buf + MAXBUF;
63 __fort_rrecv(ioproc, &off, sizeof(off), 1, __UCHAR);
64 } else {
65 off = 0;
66 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC, 0666);
67 if (fd == -1) {
68 __fort_abortp(path);
69 }
70 for (n = 0; n < ioproc; n++) {
71 __fort_rsend(n, &off, sizeof(off), 1, __UCHAR);
72 __fort_rrecv(n, buf, MAXBUF, 1, __UCHAR);
73 while (*((int *)buf) != 0) {
74 s = write(fd, buf + sizeof(int), *((int *)buf));
75 if (s == -1) {
76 __fort_abortp("__fort_zopen/write");
77 }
78 off += *((int *)buf);
79 __fort_rrecv(n, buf, MAXBUF, 1, __UCHAR);
80 }
81 }
82 }
83}

85/* write */
86
87void __fort_zwrite(char *adr, int len)
88{
89 int ioproc;
90 int s;
91
92 ioproc = GET_DIST_IOPROC;
93 if (GET_DIST_LCPU != ioproc) {
94 if ((len + sizeof(int)) > (bufe - bufp)) {
95 if (len > MAXBUF) {
96 __fort_abort("__fort_zwrite: message too big");
97 }
98 __fort_rsend(ioproc, buf, MAXBUF, 1, __UCHAR);
99 off += *((int *)buf);
100 bufp = buf;
101 *((int *)buf) = 0;
102 bufp += sizeof(int);
103 }
104 *((int *)buf) += len;
105 __fort_bcopy(bufp, adr, len);
106 bufp += len;
107 } else {
108 s = write(fd, adr, len);
109 if (s == -1) {
110 __fort_abortp("__fort_zwrite");
111 }
112 off += len;
113 }
114}

142 } else {
143 tcpus = GET_DIST_TCPUS;
144 for (n = ioproc + 1; n < tcpus; n++) {
145 __fort_rsend(n, &off, sizeof(off), 1, __UCHAR);
146 __fort_rrecv(n, buf, MAXBUF, 1, __UCHAR);
147 while (*((int *)buf) != 0) {
148 s = write(fd, buf + sizeof(int), *((int *)buf));
149 if (s == -1) {
150 __fort_abortp("__fort_zclose/write");
151 }
152 off += *((int *)buf);
153 __fort_rrecv(n, buf, MAXBUF, 1, __UCHAR);
154 }
155 }
