33/// Check if \p dtype is in legal range
34inline bool DTyValidRange(DTYPE dtype) {
35 return (dtype != DT_NONE) &&
36 (static_cast<unsigned>(dtype) < stb.dt.stg_avail);
37}

56/// A valid sequence type is one of vector of T, pointer to T, or array of T
57inline bool DTySeqTyValid(DTYPE dtype) {
58#ifdef TY_VECT
59 if (DTY(dtype) == TY_VECT)
60 return true;
61#endif
62 return (DTY(dtype) == TY_PTR) || (DTY(dtype) == TY_ARRAY);
63}
