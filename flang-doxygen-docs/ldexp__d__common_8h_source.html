12static double INLINE __ldexp_d_scalar_kernel(double a, long long int scale)
13{
14#if (defined __AVX512F__)
15 double res = _mm_cvtsd_f64(_mm_scalef_sd(_mm_set_sd(a), _mm_set_sd((double)scale)));
16 return res;
17#else
18 PRINT(a); PRINT(scale);
19 // Input is allowed to be such that signed |scale| < 2048,
20 // |a| may be in {+-0} or +-[2^-1074, 2^0] as it comes from sin/cos,
21 // but we took precaution outside this routine and normalized a,
22 // so that it is within +-[2^-1074 + 512, 2^512] or zero.
23
24 // Zeros and Inf/NaNs are handled separately.
25 // Input denormals end up here too and yield incorrect result.
26 // FIXME: assert(this function assumes no denormals on input !!!);
27 unsigned long long int exp_bits = D2L(a) & DB_EXP_MASK;
28 unsigned long long int zeroinfnan_mask =
29 ((exp_bits == DB_EXP_MASK) || (exp_bits == 0ull))
30 ? -1ll : 0ll; PRINT(zeroinfnan_mask);
31 // Preserve sign of input, quiet NaN
32 double zeroinfnan_res = a + a; PRINT(zeroinfnan_res);
33
34 // biased exponent bits, shifted to least significant position
35 unsigned long long int getexp_a = exp_bits >> (DB_PREC_BITS-1); PRINT(getexp_a);
36
37 // For a * 2^scale to fit in double we need getexp(a) + scale
38 // to fit in exponents range of double: bias + (DB_EXP_MIN-1, DB_EXP_MAX).
39 // DB_EXP_MIN-1 is less than the smallest denormal, but it may round up.
40 long long int sumexp = getexp_a + scale; PRINT(sumexp);
41
42 // Return Inf of correct sign if overflow
43 unsigned long long int ovf_mask =
44 (sumexp > (signed long long int)(DB_EXP_MAX + DB_EXP_BIAS))
45 ? -1ll : 0ll;
46 unsigned long long int sign_a = D2L(a) & DB_SIGN_BIT; PRINT(sign_a);
47 unsigned long long int ovf_res = (sign_a | DB_EXP_MASK); PRINT(ovf_res);
48
49 // If underflow, return zero of correct sign
50 unsigned long long int udf_mask =
51 (sumexp < (signed long long int)(DB_EXP_MIN-1 + DB_EXP_BIAS) )
52 ? -1ll : 0ll;
53 unsigned long long int udf_res = sign_a; PRINT(udf_res);
54
55 // Check if result is within denormalized numbers range
56 // and doesn't completely underflow
57 unsigned long long int den_mask = ~udf_mask &
58 (((signed long long int )(sumexp) <= 0ll) ? -1ll : 0ll);
59
60 // If scaling leads to denormals: we shall do it via FP multiplication
61 // 2^scale * a. But 2^scale alone may not be representable in FP, while
62 // the product is OK. Thus we would like the sum of exponents sumexp in
63 // range for FP. Since sumexp already contains the value of biased exponent
64 // of a, we will first compensate a by reducing its exponent to biased zero:
65 // a = a * 2^(-(getexp_a - bias)), or set exponent bits of a to DB_EXP_BIAS.
66 // Now we would like sumexp become positive, for that we may add as little
67 // as -(DB_EXP_MIN-2 + DB_EXP_BIAS). We'd have to compensate exponent of a
68 // by this same quantity, so in the end we'll be setting exponent of a to
69 // DB_EXP_BIAS + (DB_EXP_MIN-2 + DB_EXP_BIAS) = 2*DB_EXP_BIAS + DB_EXP_MIN-2
70 long long int new_scale =
71 ((unsigned long long int)(sumexp -(DB_EXP_MIN-2 + DB_EXP_BIAS)))
72 << (DB_PREC_BITS-1); PRINT(new_scale);
73 double new_a = L2D((D2L(a) & (~DB_EXP_MASK)) |
74 ((2*DB_EXP_BIAS + DB_EXP_MIN-2ll) << (DB_PREC_BITS-1))); PRINT(new_a);
75 double den_res = new_a * L2D(new_scale); PRINT(den_res);
76
77 // normal case, just add scale to exponent bits
78 unsigned long long int gen_res = D2L(a) +
79 (((unsigned long long int)scale) << (DB_PREC_BITS-1)); PRINT(gen_res);
80 unsigned long long int gen_mask = ~(ovf_mask | udf_mask | den_mask);
81
82 double result = L2D((D2L(zeroinfnan_res) & zeroinfnan_mask) |
83 ((~zeroinfnan_mask) & ((ovf_res & ovf_mask) |
84 (udf_res & udf_mask) |
85 (D2L(den_res) & den_mask) |
86 (gen_res & gen_mask)))); PRINT(result);
87
88 return result;
89#endif //#if (defined __AVX512F__)
90}

92static vdouble INLINE
93//static vdouble __attribute__((noinline))
94__vldexp_manual(vdouble va, vdouble vscale)
95{
96 PRINT(va); PRINT(vscale);
97 // Input is allowed to be such that signed |scale| < 2048,
98 // |a| may be in {+-0} or +-[2^-1074, 2^0] as it comes from sin/cos,
99 // but we took precaution outside this routine and normalized a,
100 // so that it is within +-[2^-1074 + 512, 2^512] or zero.
101
102 // Zeros and Inf/NaNs are handled separately.
103 // Input denormals end up here too and yield incorrect result.
104 // FIXME: assert(this function assumes no denormals on input !!!);
105 vint2 exp_bits = vand_vi2_vi2_vi2(vD2L(va), vSETll(DB_EXP_MASK));
106 vopmask zero_mask = veq64_vo_vm_vm(exp_bits, vSETll(0));
107 vopmask infnan_mask = veq64_vo_vm_vm(exp_bits, vSETll(DB_EXP_MASK));
108 vopmask zeroinfnan_mask = vor_vo_vo_vo(zero_mask, infnan_mask); PRINT(zeroinfnan_mask);
109
110 // Preserve sign of input, quiet NaN
111 vdouble zeroinfnan_res = vadd_vd_vd_vd(va, va); PRINT(zeroinfnan_res);
112
113 // biased exponent bits, shifted to least significant position
114 vint2 getexp_a = vsrl64_vi2_vi2_i(exp_bits, DB_PREC_BITS-1); PRINT(getexp_a);
115
116 // For a * 2^scale to fit in double we need getexp(a) + scale
117 // to fit in exponents range of double: bias + (DB_EXP_MIN-1, DB_EXP_MAX).
118 // DB_EXP_MIN-1 is less than the smallest denormal, but it may round up.
119 vint2 sumexp = vadd64_vi2_vi2_vi2(getexp_a, vD2L(vscale)); PRINT(sumexp);
120
121 // Return Inf of correct sign if overflow
122 vopmask ovf_mask = vgt64_vo_vm_vm(sumexp, vSETll(DB_EXP_MAX + DB_EXP_BIAS));
123 vint2 sign_a = vand_vi2_vi2_vi2(vD2L(va), vSETll(DB_SIGN_BIT)); PRINT(sign_a);
124 vint2 ovf_res = vor_vi2_vi2_vi2(sign_a, vSETll(DB_EXP_MASK)); PRINT(ovf_res);
125
126 // If underflow, return zero of correct sign
127 vopmask udf_mask = vgt64_vo_vm_vm(vSETll(DB_EXP_MIN-1 + DB_EXP_BIAS), sumexp);
128 vint2 udf_res = sign_a; PRINT(udf_res);
129
130 // Check if result is within denormalized numbers range
131 // and doesn't completely underflow
132 vopmask den_mask = vandnot_vo_vo_vo(udf_mask, vgt64_vo_vm_vm(vSETll(1), sumexp));
133
134 // If scaling leads to denormals: we shall do it via FP multiplication
135 // 2^scale * a. But 2^scale alone may not be representable in FP, while
136 // the product is OK. Thus we would like the sum of exponents sumexp in
137 // range for FP. Since sumexp already contains the value of biased exponent
138 // of a, we will first compensate a by reducing its exponent to biased zero:
139 // a = a * 2^(-(getexp_a - bias)), or set exponent bits of a to DB_EXP_BIAS.
140 // Now we would like sumexp become positive, for that we may add as little
141 // as -(DB_EXP_MIN-2 + DB_EXP_BIAS). We'd have to compensate exponent of a
142 // by this same quantity, so in the end we'll be setting exponent of a to
143 // DB_EXP_BIAS + (DB_EXP_MIN-2 + DB_EXP_BIAS) = 2*DB_EXP_BIAS + DB_EXP_MIN-2
144 vint2 new_scale =
145 vsll64_vi2_vi2_i(
146 vadd_vi2_vi2_vi2(sumexp, vSETll(-(DB_EXP_MIN-2 + DB_EXP_BIAS))),
147 DB_PREC_BITS-1); PRINT(new_scale);
148 vdouble new_a = vL2D(vor_vi2_vi2_vi2(
149 vand_vi2_vi2_vi2(vD2L(va), vSETll(~DB_EXP_MASK)),
150 vSETll((2ULL*DB_EXP_BIAS + DB_EXP_MIN-2) << (DB_PREC_BITS-1)))); PRINT(new_a);
151 vdouble den_res = vmul_vd_vd_vd(new_a, vL2D(new_scale)); PRINT(den_res);
152
153 // normal case, just add scale to exponent bits
154 vint2 gen_res = vadd_vi2_vi2_vi2(vD2L(va),
155 vsll64_vi2_vi2_i( vD2L(vscale), DB_PREC_BITS-1)); PRINT(gen_res);
156 vopmask ngen_mask =
157 vor_vo_vo_vo(vor_vo_vo_vo(ovf_mask, udf_mask), den_mask);
158
159 vdouble result = vL2D(
160 vor_vi2_vi2_vi2(
161 vand_vm_vo64_vm(zeroinfnan_mask, vD2L(zeroinfnan_res)),
162 vandnot_vm_vo64_vm(zeroinfnan_mask,
163 vor_vi2_vi2_vi2(
164 vand_vm_vo64_vm(ovf_mask, ovf_res),
165 vor_vi2_vi2_vi2(
166 vand_vm_vo64_vm(udf_mask, udf_res),
167 vor_vi2_vi2_vi2(
168 vand_vm_vo64_vm(den_mask, vD2L(den_res)),
169 vandnot_vm_vo64_vm(ngen_mask, gen_res))))))); PRINT(result);
170
171 return result;
172}

174static vdouble INLINE
175//static vdouble __attribute__((noinline))
176__vldexp_kernel(vdouble va, vdouble vscale)
177{
178 PRINT(va); PRINT(vscale);
179#if (defined __AVX512F__) && (defined __AVX512VL__) && (defined __AVX512DQ__)
180 vdouble vfres = JOIN(__SIMD_TYPE,_scalef_pd)(va, JOIN(__SIMD_TYPE,_cvtepi64_pd)(vD2L(vscale))); PRINT(vfres);
181 return vfres;
182#elif (defined __AVX512F__)
183 __mmask8 mask = (__mmask8)((1 << (2*_VL)) - 1); PRINT(mask);
184 #define _mm512_castpd512_pd512(x) (x) // no cast operation needed if in full width
185 __m512d fullwidth_va = JOIN3(_mm512_castpd,__SIMD_BITS,_pd512)(va); PRINT(fullwidth_va);
186 __m512d fullwidth_vscale = JOIN3(_mm512_castpd,__SIMD_BITS,_pd512)(vscale); PRINT(fullwidth_vscale);
187 // need to emulate conversion from signed 64-bit integer to double
188 // we know that |scale| < 2^31, so the trick works
189 __m512d fullwidth_vscale_dp =
190 _mm512_castsi512_pd(
191 _mm512_add_epi32(
192 _mm512_castpd_si512(fullwidth_vscale),
193 _mm512_castpd_si512(_mm512_set1_pd(0x1.8p52))
194 )
195 ); PRINT(fullwidth_vscale_dp);
196 fullwidth_vscale_dp =
197 _mm512_maskz_sub_pd(mask, fullwidth_vscale_dp, _mm512_set1_pd(0x1.8p52)); PRINT(fullwidth_vscale_dp);
198 __m512d fullwidth_vfres =
199 _mm512_maskz_scalef_pd(mask, fullwidth_va, fullwidth_vscale_dp); PRINT(fullwidth_vfres);
200 vdouble vfres = JOIN(_mm512_castpd512_pd,__SIMD_BITS)(fullwidth_vfres); PRINT(vfres);
201 #undef _mm512_castpd512_pd512
202 return vfres;
203#else
204 return __vldexp_manual(va, vscale);
205#endif
206}
