134static void
135save_gbl()
136{
137 if (gbl_avl) {
138 gbl->fcb = fcb;
139 gbl->accessed = accessed;
140 gbl->byte_cnt = byte_cnt;
141 gbl->n_irecs = n_irecs;
142 gbl->internal_file = internal_file;
143 gbl->rec_len = rec_len;
144 gbl->gbl_dtype = gbl_dtype;
145 if (rbuf_size > gbl->rbuf_size) {
146 gbl->rbufp = malloc(rbuf_size);
147 gbl->rbuf_size = rbuf_size;
148 } else {
149 gbl->rbufp = gbl->rbuf;
150 gbl->rbuf_size = RBUF_SIZE;
151 }
152 memcpy(gbl->rbufp, rbufp, rbuf_size);
153 if (currc) {
154 assert(currc == (rbufp + (currc - rbufp)));
155 gbl->currc = gbl->rbufp + (currc - rbufp);
156 } else {
157 gbl->currc = NULL;
158 }
159 gbl->in_recp = in_recp;
160
161 /* may need to revisit this for derived type io */
162 gbl->tknval = tknval;
163 gbl->tkntyp = tkntyp;
164 gbl->scan_err = scan_err;
165 gbl->repeat_cnt = repeat_cnt;
166 gbl->prev_tkntyp = prev_tkntyp;
167 gbl->comma_seen = comma_seen;
168 }

252static void
253allocate_new_gbl()
254{
255 G *tmp_gbl;
256 int gsize = sizeof(G);
257 if (gbl_avl >= gbl_size) {
258 if (gbl_size == GBL_SIZE) {
259 gbl_size = gbl_size + GBL_SIZE;
260 tmp_gbl = (G *)malloc(gsize * gbl_size);
261 memcpy(tmp_gbl, gbl_head, gsize * gbl_avl);
262 memset(tmp_gbl + gbl_avl, 0, gsize * GBL_SIZE);
263 gbl_head = tmp_gbl;
264 } else {
265 gbl_size = gbl_size + GBL_SIZE;
266 gbl_head = (G *)realloc(gbl_head, gsize * gbl_size);
267 memset(gbl_head + gbl_avl, 0, gsize * GBL_SIZE);
268 }
269 }
270 gbl = &gbl_head[gbl_avl];
271 if (gbl->rbufp != gbl->rbuf) {
272 free(gbl->rbufp);
273 }
274 memset(gbl, 0, gsize);
275 gbl->rbufp = gbl->rbuf;
276 gbl->rbuf_size = RBUF_SIZE;
277 ++gbl_avl;
278}

281free_gbl()
282{
283 --gbl_avl;
284 if (gbl_avl <= 0)
285 gbl_avl = 0;
286 if (gbl_avl == 0)
287 gbl = &gbl_head[gbl_avl];
288 else
289 gbl = &gbl_head[gbl_avl - 1];
290}

296static int
297_f90io_ldr_init(__INT_T *unit, /* unit number */
298 __INT_T *rec, /* record number for direct access I/O */
299 __INT_T *bitv, /* same as for ENTF90IO(open_) */
300 __INT_T *iostat) /* same as for ENTF90IO(open_) */
301{
302
303 int i;
304 G *tmp_gbl;
305 save_gbl();
306 __fortio_errinit03(*unit, *bitv, iostat, "list-directed read");
307 allocate_new_gbl();
308 fcb = __fortio_rwinit(*unit, FIO_FORMATTED, rec, 0 /*read*/);
309
310 if (fcb == NULL) {
311 if (fioFcbTbls.eof)
312 return EOF_FLAG;
313 /* TBD - does there need to be fioFcbTbls.eor */
314 return ERR_FLAG;
315 }
316
317 rec_len = fcb->reclen;
318 internal_file = FALSE;
319
320 gbl->decimal = fcb->decimal;
321
322 /* check if recursive io on same external file */
323 tmp_gbl = NULL;
324 if (gbl_avl > 1) {
325 for (i = gbl_avl - 2; i >= 0; --i) {
326 if (gbl_head[i].fcb == fcb) {
327 tmp_gbl = &gbl_head[i];
328 break;
329 }
330 }
331 }
332 if (tmp_gbl) {
333 gbl->same_fcb = tmp_gbl;
334 gbl->same_fcb_idx = i;
335 gbl->blank_zero = tmp_gbl->blank_zero;
336 gbl->pad = tmp_gbl->pad;
337 gbl->decimal = tmp_gbl->decimal;
338 gbl->round = tmp_gbl->round;
339
340 accessed = tmp_gbl->accessed;
341 byte_cnt = tmp_gbl->byte_cnt;
342 prev_tkntyp = tmp_gbl->prev_tkntyp;
343 repeat_cnt = tmp_gbl->repeat_cnt;
344 n_irecs = tmp_gbl->n_irecs;
345 rec_len = tmp_gbl->rec_len;
346 gbl_dtype = tmp_gbl->gbl_dtype;
347 in_recp = tmp_gbl->in_recp;
348 internal_file = tmp_gbl->internal_file;
349 if (tmp_gbl->rbuf_size > rbuf_size) {
350 if (rbufp != rbuf)
351 rbufp = realloc(rbufp, tmp_gbl->rbuf_size);
352 else
353 rbufp = malloc(tmp_gbl->rbuf_size);
354 rbuf_size = tmp_gbl->rbuf_size;
355 } else {
356 rbufp = rbuf;
357 }
358 memcpy(rbufp, tmp_gbl->rbufp, tmp_gbl->rbuf_size);
359 if (tmp_gbl->currc) {
360 currc = rbufp + (tmp_gbl->currc - tmp_gbl->rbufp);
361 } else {
362 currc = NULL;
363 }
364 comma_seen = FALSE;
365 return 0;
366 } else {
367 gbl->same_fcb = tmp_gbl;
368 gbl->same_fcb_idx = 0;
369 fcb->skip = 0;
370 }
371
372 shared_init();
373 return 0;
374}

396__INT_T
397ENTF90IO(LDR_INIT03A, ldr_init03a)
398(__INT_T *istat, DCHAR(blank), DCHAR(decimal), DCHAR(pad),
399 DCHAR(round) DCLEN64(blank) DCLEN64(decimal) DCLEN64(pad) DCLEN64(round))
400{
401 int s = *istat;
402
403 if (s)
404 return DIST_STATUS_BCST(s);
405
406 if (LOCAL_MODE || GET_DIST_LCPU == GET_DIST_IOPROC) {
407 if (ISPRESENTC(blank)) {
408 if (__fortio_eq_str(CADR(blank), CLEN(blank), "ZERO")) {
409 gbl->blank_zero = FIO_ZERO;
410 } else if (__fortio_eq_str(CADR(blank), CLEN(blank), "NULL")) {
411 gbl->blank_zero = FIO_NULL;
412 } else
413 s = __fortio_error(FIO_ESPEC);
414 }
415 if (ISPRESENTC(decimal) && s == 0) {
416 if (__fortio_eq_str(CADR(decimal), CLEN(decimal), "COMMA")) {
417 gbl->decimal = FIO_COMMA;
418 } else if (__fortio_eq_str(CADR(decimal), CLEN(decimal), "POINT")) {
419 gbl->decimal = FIO_POINT;
420 } else
421 s = __fortio_error(FIO_ESPEC);
422 }
423 if (ISPRESENTC(pad) && s == 0) {
424 if (__fortio_eq_str(CADR(pad), CLEN(pad), "YES"))
425 gbl->pad = FIO_YES;
426 else if (__fortio_eq_str(CADR(pad), CLEN(pad), "NO"))
427 gbl->pad = FIO_NO;
428 else
429 s = __fortio_error(FIO_ESPEC);
430 }
431 if (ISPRESENTC(round) && s == 0) {
432 if (__fortio_eq_str(CADR(round), CLEN(round), "UP")) {
433 gbl->round = FIO_UP;
434 } else if (__fortio_eq_str(CADR(round), CLEN(round), "DOWN")) {
435 gbl->round = FIO_DOWN;
436 } else if (__fortio_eq_str(CADR(round), CLEN(round), "ZERO")) {
437 gbl->round = FIO_ZERO;
438 } else if (__fortio_eq_str(CADR(round), CLEN(round), "NEAREST")) {
439 gbl->round = FIO_NEAREST;
440 } else if (__fortio_eq_str(CADR(round), CLEN(round), "COMPATIBLE")) {
441 gbl->round = FIO_COMPATIBLE;
442 } else if (__fortio_eq_str(CADR(round), CLEN(round),
443 "PROCESSOR_DEFINED")) {
444 gbl->round = FIO_PROCESSOR_DEFINED;
445 } else
446 s = __fortio_error(FIO_ESPEC);
447 }
448 }
449 if (s != 0) {
450 free_gbl();
451 restore_gbl();
452 __fortio_errend03();
453 }
454 return DIST_STATUS_BCST(s);
455}

617int
618__f90io_ldr(int type, /* data type (as defined in pghpft.h) */
619 long length, /* # items of type to read */
620 int stride, /* distance in bytes between items */
621 char *item, /* where to transfer data to */
622 __CLEN_T itemlen)
623{
624 int ret_err = 0; /* return error (for iostat) */
625 long item_num;
626 char *tmpitem;
627
628 if (fioFcbTbls.error) {
629 ret_err = ERR_FLAG;
630 goto ldr_err;
631 }
632 if (fioFcbTbls.eof) {
633 ret_err = EOF_FLAG;
634 goto ldr_err;
635 }
636
637 if (DBGBIT(0x1)) {
638 __io_printf("\n__fortio_ldr: ");
639 __io_printf("item=%p, ", item);
640 __io_printf("type=%d, ", type);
641 __io_printf("length=%ld, ", length);
642 __io_printf("stride=%d\n", stride);
643 }
644
645 if (length <= 0)
646 /* no items to read */
647 return 0;
648
649 if (prev_tkntyp == TK_SLASH)
650 return 0;
651 if (byte_cnt == 0 && (ret_err = read_record()) != 0) {
652 ret_err = __fortio_error(ret_err);
653 goto ldr_err;
654 }
655
656 /* main loop is driven by number (length) of items to be read */
657
658 tmpitem = item;
659 gbl_dtype = type;
660 for (item_num = 0; item_num < length; item_num++, tmpitem += stride) {
661 get_token();
662 if (tkntyp == TK_SLASH)
663 return 0;
664 if (tkntyp == TK_ERROR) {
665 ret_err = __fortio_error(scan_err);
666 goto ldr_err;
667 }
668 if (tkntyp == TK_NULL)
669 continue;
670
671 if (tkntyp == TK_VALS) {
672 tkntyp = TK_VAL;
673 if (gbl_dtype != __STR && gbl_dtype != __NCHAR) {
674 ret_err = __fortio_error(FIO_EERR_DATA_CONVERSION);
675 goto ldr_err;
676 }
677 }
678 scan_err = __fortio_assign(tmpitem, type, itemlen, &tknval);
679 if (scan_err) {
680 ret_err = __fortio_error(scan_err);
681 goto ldr_err;
682 }
683 }
684 return 0;
685
686ldr_err:
687 free_gbl();
688 restore_gbl();
689 __fortio_errend03();
690 return (ret_err);
691}

868#undef BEGINS_NUM
869#undef ISDELIMITER
870
871#define BEGINS_NUM(c) (ISDIGIT(c) || (c) == '.')
872#define ISDELIMITER(c) \
873 ((c) == ',' || (c) == ' ' || (c) == '\t' || (c) == '/' || (c == '\n'))
874
875static int is_repeat_count(char *);
876
877static void
878get_token()
879{
880 char ch;
881
882 if (DBGBIT(0x8))
883 __io_printf("get_token enter: repeat_cnt:%d\n", repeat_cnt);
884 if (repeat_cnt) {
885 repeat_cnt--;
886 return;
887 }
888 scan_err = 0;
889 tkntyp = 0;
890 if (gbl_dtype == __STR || gbl_dtype == __NCHAR)
891 /* current item is type character */
892 do {
893 switch (*currc++) {
894 case '\n':
895 /* read another record */
896 if ((scan_err = read_record()) != 0)
897 tkntyp = TK_ERROR;
898 break;
899
900 /* whitespace => delimiters */
901 case ' ':
902 case '\t':
903 break;
904
905 case ';':
906 if (gbl->decimal != FIO_COMMA)
907 break;
908 if (comma_seen) {
909 /* consecutive commas implies null value */
910 tkntyp = TK_NULL;
911 goto multiple_commas;
912 }
913 /* otherwise, it's just a delimiter */
914 comma_seen = TRUE;
915 break;
916 case ',':
917 if (gbl->decimal == FIO_COMMA)
918 break;
919 if (comma_seen) {
920 /* consecutive commas implies null value */
921 tkntyp = TK_NULL;
922 goto multiple_commas;
923 }
924 /* otherwise, it's just a delimiter */
925 comma_seen = TRUE;
926 break;
927
928 case '/':
929 tkntyp = TK_SLASH;
930 break;
931
932 case '0':
933 case '1':
934 case '2':
935 case '3':
936 case '4':
937 case '5':
938 case '6':
939 case '7':
940 case '8':
941 case '9':
942 currc--;
943 if (is_repeat_count(currc)) {
944 int rc;
945 get_number();
946 rc = tknval.val.i;
947 ch = *++currc;
948 if (ISDELIMITER(ch))
949 tkntyp = TK_NULL;
950 else
951 get_token();
952 repeat_cnt = rc - 1;
953 } else
954 get_junk();
955 break;
956
957 case '\'':
958 get_qstr('\'');
959 break;
960
961 case '\"':
962 get_qstr('\"');
963 break;
964
965 default:
966 currc--;
967 get_junk();
968 break;
969 }
970
971 } while (tkntyp == 0); /* end current item is type character */
972 else
973 /* current item is not type character */
974 do {
975 switch (*currc++) {
976 case '\n':
977 /* read another record */
978 if ((scan_err = read_record()) != 0)
979 tkntyp = TK_ERROR;
980 break;
981
982 /* whitespace => delimiters */
983 case ' ':
984 case '\t':
985 break;
986 case ';':
987 if (gbl->decimal != FIO_COMMA)
988 break;
989 if (comma_seen) {
990 /* consecutive commas implies null value */
991 tkntyp = TK_NULL;
992 goto multiple_commas;
993 }
994 /* otherwise, it's just a delimiter */
995 comma_seen = TRUE;
996 break;
997
998 case ',':
999 if (gbl->decimal == FIO_COMMA) {
1000 ch = *currc;
1001 if (ISDIGIT(ch)) {
1002 currc--;
1003 get_number();
1004 break;
1005 }
1006 /* check for logical */
1007 if (ch == 'T' || ch == 't') {
1008 currc++;
1009 goto have_true;
1010 }
1011 if (ch == 'F' || ch == 'f') {
1012 currc++;
1013 goto have_false;
1014 }
1015 currc--;
1016 get_number(); /* VMS extension (only if numeric or log */
1017 break;
1018 } else {
1019 if (comma_seen) {
1020 /* consecutive commas implies null value */
1021 tkntyp = TK_NULL;
1022 goto multiple_commas;
1023 }
1024 /* otherwise, it's just a delimiter */
1025 comma_seen = TRUE;
1026 break;
1027 }
1028
1029 case '/':
1030 tkntyp = TK_SLASH;
1031 break;
1032
1033 case '+':
1034 case '-':
1035 ch = *currc;
1036 if (BEGINS_NUM(ch)) {
1037 currc--;
1038 get_number();
1039 break;
1040 }
1041 /* Could also support NaN here too??? */
1042 if (ch == 'i' || ch == 'I') {
1043 currc--;
1044 get_infinity();
1045 break;
1046 }
1047 currc--;
1048 get_number(); /* VMS extension (only if numeric or log */
1049 break;
1050
1051 case '0':
1052 case '1':
1053 case '2':
1054 case '3':
1055 case '4':
1056 case '5':
1057 case '6':
1058 case '7':
1059 case '8':
1060 case '9':
1061 currc--;
1062 get_number();
1063 /* check if repeat count */
1064 if (tkntyp == TK_VAL && tknval.dtype == __BIGINT && tknval.val.i != 0 &&
1065 *currc == '*') {
1066 int rc;
1067 rc = tknval.val.i;
1068 ch = *++currc;
1069 if (ISDELIMITER(ch))
1070 tkntyp = TK_NULL;
1071 else
1072 get_token();
1073 repeat_cnt = rc - 1;
1074 }
1075 break;
1076
1077 case '.':
1078 ch = *currc;
1079 if (ISDIGIT(ch)) {
1080 currc--;
1081 get_number();
1082 break;
1083 }
1084 /* check for logical */
1085 if (ch == 'T' || ch == 't') {
1086 currc++;
1087 goto have_true;
1088 }
1089 if (ch == 'F' || ch == 'f') {
1090 currc++;
1091 goto have_false;
1092 }
1093 currc--;
1094 get_number(); /* VMS extension (only if numeric or log */
1095 break;
1096
1097 case '(':
1098 get_cmplx();
1099 break;
1100
1101 case 't':
1102 case 'T':
1103 have_true:
1104 tknval.val.i = FTN_TRUE;
1105 goto have_logical;
1106
1107 case 'f':
1108 case 'F':
1109 have_false:
1110 tknval.val.i = FTN_FALSE;
1111
1112 have_logical:
1113 while (TRUE) {
1114 ch = *currc;
1115 if (ISDELIMITER(ch))
1116 break;
1117 ++currc;
1118 }
1119 tkntyp = TK_VAL;
1120 tknval.dtype = __BIGLOG;
1121 break;
1122
1123 case '\'':
1124 get_qstr('\'');
1125 break;
1126
1127 case '\"':
1128 get_qstr('\"');
1129 break;
1130
1131 case 'e':
1132 case 'E':
1133 case 'd':
1134 case 'D':
1135 currc--;
1136 get_number(); /* VMS extension (only if numeric or log */
1137 break;
1138
1139 case 'i':
1140 case 'I':
1141 currc--;
1142 get_infinity();
1143 break;
1144 case 'n':
1145 case 'N':
1146 currc--;
1147 get_nan();
1148 break;
1149
1150 default:
1151 tkntyp = TK_ERROR;
1152 scan_err = FIO_ELEX; /* unknown token */
1153 break;
1154 }
1155
1156 } while (tkntyp == 0); /* end current item is not type character */
1157 /*
1158 * if a value is found, we did not see a comma.
1159 */
1160 comma_seen = FALSE;
1161
1162multiple_commas:
1163
1164 prev_tkntyp = tkntyp;
1165 if (DBGBIT(0x8))
1166 __io_printf("get_token: new token %d\n", tkntyp);
1167}

1169/** \brief Given that a string begins with a digit, is it a repeat count?
1170 *
1171 * If a string of digits is immediately followed by '*', then the
1172 * digit string is a repeat count.
1173 */
1174static int
1175is_repeat_count(char *p)
1176{
1177 char *q;
1178 char ch;
1179
1180 q = p + 1;
1181 while (TRUE) {
1182 ch = *q;
1183 if (!ISDIGIT(ch))
1184 break;
1185 q++;
1186 }
1187 if ((q - p) == 1 && *p == '0')
1188 return 0;
1189 if (ch == '*')
1190 return 1;
1191 return 0;
1192}

1195 * Extract integer, real, or double precision constant token:
1196 */
1197static void
1198get_number(void)
1199{
1200 int ret_err;
1201 int type;
1202 union {
1203 __BIGINT_T i;
1204 __BIGREAL_T d;
1205 __INT8_T i8v;
1206 } val;
1207 int len;
1208
1209 if (gbl->decimal == FIO_COMMA)
1210 ret_err = __fortio_getnum(currc, &type, &val, &len, TRUE);
1211 else
1212 ret_err = __fortio_getnum(currc, &type, &val, &len, FALSE);
1213 currc += len;
1214 if (ret_err) {
1215 scan_err = ret_err;
1216 tkntyp = TK_ERROR;
1217 return;
1218 }
1219 if (type == 1) {
1220 tknval.dtype = __BIGREAL;
1221 tknval.val.d = val.d;
1222 }
1223 else if (type == 2) {
1224 tknval.dtype = __INT8;
1225 tknval.val.i8v = val.i8v;
1226 }
1227 else if (type == 3) {
1228 /* Degenerate VMS REAL, allow integer value 0 to be returned only if
1229 * a REAL type was expected. Otherwise its an error.
1230 */
1231 if (!REAL_ALLOWED(gbl_dtype)) {
1232 scan_err = FIO_EERR_DATA_CONVERSION;
1233 tkntyp = TK_ERROR;
1234 return;
1235 }
1236 tknval.dtype = __BIGINT;
1237 tknval.val.i = val.i;
1238 } else {
1239 tknval.dtype = __BIGINT;
1240 tknval.val.i = val.i;
1241 }
1242 tkntyp = TK_VAL;
1243

1345/** \brief
1346 * An 'N' has been found. Is it a NaN? Valid are nan and nan(ddd...)
1347 */
1348static void
1349get_nan(void)
1350{
1351 union ieee ieee_v;
1352 char c;
1353 unsigned int nval, ntval, nshplaces;
1354 c = *currc++;
1355 ieee_v.i[0] = 0x0;
1356 ieee_v.i[1] = 0x7ff00000;
1357 if (c == 'n' || c == 'N') {
1358 c = *currc++;
1359 if (c == 'a' || c == 'A') {
1360 c = *currc++;
1361 if (c == 'n' || c == 'N') {
1362 c = *currc++;
1363 if (c == '(') {
1364 ieee_v.i[0] = 0x0;
1365 ieee_v.v.hm = 0x0;
1366 ieee_v.v.e = 2047;
1367 ntval = 0;
1368 c = *currc++;
1369 nshplaces = 48;
1370 while (TRUE) {
1371 if ((c >= '0') && (c <= '9'))
1372 nval = c - '0';
1373 else if ((c >= 'a') && (c <= 'f'))
1374 nval = c - 'a' + 10;
1375 else if ((c >= 'A') && (c <= 'F'))
1376 nval = c - 'A' + 10;
1377 else if (c == ')')
1378 break;
1379 else
1380 goto conv_nan_error;
1381 if (nshplaces > 28)
1382 ieee_v.v.hm |= nval << (nshplaces - 32);
1383 else
1384 ieee_v.v.lm |= nval << nshplaces;
1385 ntval += nval;
1386 c = *currc++;
1387 if (nshplaces)
1388 nshplaces -= 4;
1389 }
1390 if (c == ')') {
1391 /* Set this quiet */
1392 if (ntval == 0)
1393 ieee_v.v.hm |= 0x80000;
1394 c = *currc;
1395 }
1396 } else {
1397 ieee_v.v.hm |= 0x80000; /* quiet */
1398 }
1399 if (ISDELIMITER(c)) {
1400 tknval.dtype = __BIGREAL;
1401 tknval.val.d = ieee_v.d;
1402 tkntyp = TK_VAL;
1403 return;
1404 }
1405 }
1406 }
1407 }
1408conv_nan_error:
1409 scan_err = FIO_ELEX; /* unknown token */
1410 tkntyp = TK_ERROR;
1411}
