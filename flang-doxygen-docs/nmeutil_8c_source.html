51/** \brief Query whether the given nme is a PRE temp. */
52bool
53is_presym(int nme)
54{
55 if (nme && (NME_TYPE(nme) == NT_VAR)) {
56 SPTR sptr = NME_SYM(nme);
57
58 if ((sptr > 0) && CCSYMG(sptr) && SYMNAME(sptr) &&
59 (strncmp(SYMNAME(sptr), ".pre", 4) == 0))
60 return true;
61 }
62
63 return false;
64}

112/*
113 * for F90:
114 * type dt
115 * integer :: i, j
116 * end type
117 * type, extends(dt)::et
118 * integer :: k
119 * end type
120 * type(et) :: m
121 * a reference to m%i may appear in the ILMs as
122 * IM_MEMBER(IM_BASE(m),IM_MEMBER(i))
123 * but in the datatype table as
124 * dt-> TY_STRUCT ( i(integer), j(integer) )
125 * et-> TY_STRUCT ( dt(dt), k(integer) )
126 * so the datatype table has an extra member, named dt, for the datatype dt
127 * we want to insert the reference to %dt%i here
128 */
129static int
130find_parent_member(int nmex, DTYPE dt, int sym)
131{
132 SPTR mem;
133 int newnmex;
134 SPTR sptr;
135 mem = DTyAlgTyMember(dt);
136 if (mem > NOSYM && PARENTG(mem)) {
137 DTYPE dtmem = DTYPEG(mem);
138 for (sptr = DTyAlgTyMember(dtmem); sptr > NOSYM; sptr = SYMLKG(sptr)) {
139 if (sym == sptr)
140 break;
141 }
142 newnmex = add_arrnme(NT_MEM, mem, nmex, 0, 0, 0);
143 if (mem > NOSYM)
144 return newnmex;
145 return find_parent_member(newnmex, dtmem, sym);
146 }
147 return 0;
148} /* find_parent_member */

150int
151add_arrnme(NT_KIND type, SPTR insym, int nm, ISZ_T cnst, int sub, bool inlarr)
152{
153 int val, i;
154 DTYPE nmdt;
155 SPTR sym;
156
157 if (EXPDBG(10, 256))
158 fprintf(gbl.dbgfil,
159 "type = %d sym = %d nm = %d cnst = %" ISZ_PF "d sub=%d inlarr=%d\n",
160 type, insym, nm, cnst, sub, inlarr);
161#if DEBUG
162 //assert(true, "add_arrnme: bad inlarr", inlarr, ERR_Severe);
163#endif
164
165 /* evaluate nme and fold any values */
166
167 if (insym < SPTR_NULL) {
168 DEBUG_ASSERT(insym == NME_NULL, "add_arrnme: bad negative insym");
169 } else {
170 DEBUG_ASSERT(insym <= SPTR_MAX, "add_arrnme: out of bounds insym");
171 }
172 sym = insym;
173
174 switch (type) {
175
176 case NT_SAFE:
177 case NT_VAR:
178 case NT_IND:
179 break;
180
181 case NT_ARR:
182#ifdef NT_INDARR
183 case NT_INDARR:
184#endif
185#ifndef FE90
186 if (sub)
187 sub = ili_subscript(sub);
188#endif
189 if (nm == NME_VOL)
190 return NME_VOL;
191 { /* tpr 564:
192 * for a subscripted reference, ensure the 'base' is not
193 * a union or structure.
194 */
195 DTYPE dt = dt_nme(nm);
196 if (DTY(dt) == TY_UNION)
197 return nm;
198 }
199 break;
200
201 case NT_MEM:
202 /*
203 * don't create union NMEs for fortran; fortran's conflict() does not
204 * have the left-to-right walk of the nmes on which checking of unions
205 * depends.
206 */
207 nmdt = dt_nme(nm);
208 if (DTY(nmdt) == TY_STRUCT
209#ifdef TY_DERIVED
210 || DTY(nmdt) == TY_DERIVED
211#endif
212 ) {
213 if (sym > 0) {
214#if defined(INLNG)
215 if (!INLNG(sym))
216#else
217 if (1)
218#endif
219 {
220 for (i = DTyAlgTyMember(nmdt); i > NOSYM; i = SYMLKG(i)) {
221 if (sym == i)
222 goto is_member;
223 }
224 {
225 int nnm;
226 /* look if this is an extended member */
227 nnm = find_parent_member(nm, nmdt, sym);
228 if (nnm) {
229 nm = nnm;
230 goto is_member;
231 }
232 }
233 } else {
234 SPTR i;
235 /* the member names are inlined separately;
236 * the datatype gets duplicated.
237 * allow for this */
238 for (i = DTyAlgTyMember(nmdt); i > NOSYM; i = SYMLKG(i)) {
239 if (sym == i || (strcmp(SYMNAME(i), SYMNAME(sym)) == 0 &&
240 ADDRESSG(i) == ADDRESSG(sym))) {
241 sym = i;
242 goto is_member;
243 }
244 }
245 }
246 }
247 }
248 if (nm == NME_VOL || sym > 1) {
249 return nm;
250 }
251 break; /* => real/imag (sym = 0/1) parts of complex */
252 is_member:
253 break;
254
255 case NT_ADD:
256 switch (NME_TYPE(nm)) {
257 case NT_INDARR:
258 case NT_ADD:
259 case NT_SAFE:
260 break;
261
262 case NT_VAR:
263 case NT_MEM:
264 case NT_ARR:
265 /*
266 * attempt to create an array names entry for these cases. This
267 * is only done when nm is for an item which is an array.
268 */
269 if (DTY(dt_nme(nm)) != TY_ARRAY) {
270
271 /*
272 * since nm is not for an array, it is determine if the
273 * amount being added is zero.
274 */
275 if (sym == 0 && cnst == 0)
276 return nm; /* go ahead and use nm */
277
278 /*
279 * this occurs when adding a nonzero or variable value to
280 * an & expression; i.e., &x + 4, &x + i -- it is not
281 * known what is actually being referenced. The addrtkn
282 * flag is set for the variable in nm and the unknown
283 * names entry is returned.
284 */
285 loc_of(nm);
286 return NME_UNK;
287 }
288 type = NT_ARR; /* create an array names entry */
289 break;
290
291 case NT_IND:
292 if (sym == 0) {
293 if (NME_SYM(nm) == 0) {
294 cnst += NME_CNST(nm);
295 nm = NME_NM(nm);
296 } else
297 return nm;
298 } else {
299 nm = NME_NM(nm);
300 cnst = 0;
301 }
302 type = NT_IND;
303 sub = 0;
304 inlarr = false;
305 break;
306
307 case NT_UNK:
308 return nm;
309 }
310 break;
311 case NT_UNK:
312 return NME_UNK;
313 }
314
315 /* compute the hash index for this NME */
316 val = (int)((type ^ sym ^ nm ^ sub) & 0x7fff) % NMEHSZ;
317
318 /* search the hash links for this NME */
319 for (i = nmehsh[val]; i != 0; i = NME_HSHLNK(i))
320 if (NME_TYPE(i) == type && NME_INLARR(i) == inlarr && NME_SYM(i) == sym &&
321 NME_NM(i) == nm && NME_CNST(i) == cnst && NME_SUB(i) == sub &&
322#ifdef NME_PTE
323 NME_PTE(i) == 0 &&
324#endif
325 NME_RPCT_LOOP(i) == 0)
326 return i; /* F O U N D */
327
328 /*
329 * N O T F O U N D -- if no more storage is available, try to get more
330 * storage
331 */
332 i = STG_NEXT(nmeb);
333 if (i > MAXNME)
334 error((error_code_t)7, ERR_Fatal, 0, CNULL, CNULL);
335 /*
336 * NEW ENTRY - add the nme to the nme area and to its hash chain
337 */
338 if (EXPDBG(10, 256))
339 fprintf(gbl.dbgfil,
340 "adding nme %d:type = %d sym = %d nm = %d cnst = %" ISZ_PF
341 "d sub = %d, inlarr=%d\n",
342 i, type, sym, nm, cnst, sub, inlarr);
343 NME_TYPE(i) = type;
344 NME_INLARR(i) = inlarr;
345 NME_SYM(i) = sym;
346 NME_NM(i) = nm;
347 NME_CNST(i) = cnst;
348 NME_HSHLNK(i) = nmehsh[val];
349 NME_SUB(i) = sub;
350 nmehsh[val] = i;
351 return i;
352}

357 int i, val, sub = 0, sym = insym;

361

370 return i; /* F O U N D */
371 }

375/** \brief Add a new nme based on this nme with a new PTE list; otherwise all
376 fields the same.
377 */
378int
379add_nme_with_pte(int nm, int ptex)
380{
381#ifndef NME_PTE
382 return nm;
383#else
384 int val, i;
385
386 val =
387 (int)((NME_TYPE(nm) ^ NME_SYM(nm) ^ NME_NM(nm) ^ NME_SYM(nm)) & 0x7fff) %
388 NMEHSZ;
389 for (i = nmehsh[val]; i > 0; i = NME_HSHLNK(i)) {
390 if (NME_TYPE(i) == NME_TYPE(nm) && NME_INLARR(i) == NME_INLARR(nm) &&
391 NME_SYM(i) == NME_SYM(nm) && NME_NM(i) == NME_NM(nm) &&
392 NME_CNST(i) == NME_CNST(nm) && NME_SUB(i) == NME_SUB(nm) &&
393 NME_RPCT_LOOP(i) == NME_RPCT_LOOP(nm) && NME_PTE(i) == ptex)
394 return i; /* F O U N D */
395 }
396 i = STG_NEXT(nmeb);
397 if (i > MAXNME)
398 error((error_code_t)7, ERR_Fatal, 0, CNULL, CNULL);
399 if (EXPDBG(10, 256))
400 fprintf(gbl.dbgfil, "adding based nme %d, based on %d with pte %d\n", i, nm,
401 ptex);
402 BCOPY(nmeb.stg_base + i, nmeb.stg_base + nm, NME, 1);
403 NME_BASE(i) = nm;
404 NME_PTE(i) = ptex;
405 NME_HSHLNK(i) = nmehsh[val];
406 nmehsh[val] = i;
407 return i;
408#endif
409} /* add_nme_with_pte */

411/**
412 * This creates and returns a new NME, 'rpct_nme', whose fields are
413 * the same as those of 'orig_nme' except that it has
414 * NME_RPCT_LOOP( rpct_nme ) == rpct_loop (> 0), whereas
415 * NME_RPCT_LOOP( orig_nme ) == 0. 'rpct_nme' will replace all
416 * occurrences of 'orig_nme' within the loop denoted by 'rpct_loop',
417 * whereas 'orig_nme' will continue to be used elsewhere. This loop
418 * is guarded by runtime pointer conflict tests ('RPCT's) which prove
419 * at runtime that the 'rpct_nme' references don't conflict with
420 * various other references in the loop that they could potentially
421 * conflict with, as indicated in the 'RPCT table', and 'conflict()'
422 * will use this information to return NOCONFLICT for pairs of such
423 * references.
424 */
425int
426add_rpct_nme(int orig_nme, int rpct_loop)
427{
428 int hashval, rpct_nme;
429
430 asrt(NME_RPCT_LOOP(orig_nme) == 0 && rpct_loop > 0);
431
432 /* Compute the hash index for 'orig_nme' and 'rpct_nme' (they both
433 * have the same hash index).
434 */
435 hashval = (int)((NME_TYPE(orig_nme) ^ NME_SYM(orig_nme) ^ NME_NM(orig_nme) ^
436 NME_SUB(orig_nme)) &
437 0x7fff) %
438 NMEHSZ;
439
440#if DEBUG
441 /* Search the nme hash links for this NME. If it already exists
442 * we return it and generate an 'asrt()' error message, since this
443 * shouldn't happen.
444 */
445 for (rpct_nme = nmehsh[hashval]; rpct_nme; rpct_nme = NME_HSHLNK(rpct_nme)) {
446 if (NME_TYPE(rpct_nme) == NME_TYPE(orig_nme) &&
447 NME_INLARR(rpct_nme) == NME_INLARR(orig_nme) &&
448 NME_SYM(rpct_nme) == NME_SYM(orig_nme) &&
449 NME_NM(rpct_nme) == NME_NM(orig_nme) &&
450 NME_CNST(rpct_nme) == NME_CNST(orig_nme) &&
451 NME_SUB(rpct_nme) == NME_SUB(orig_nme) &&
452#ifdef NME_PTE
453 NME_PTE(rpct_nme) == NME_PTE(orig_nme) &&
454#endif
455 NME_RPCT_LOOP(rpct_nme) == rpct_loop) {
456 /* Found.
457 */
458 asrt(false); /* we don't expect it to be found! */
459 return rpct_nme;
460 }
461 }
462#endif
463
464 /* Not found, so create and initialise a new nme, and link it info
465 * its hash chain. If necessary get more storage.
466 */
467 rpct_nme = STG_NEXT(nmeb);
468
469 if (rpct_nme > MAXNME)
470 error((error_code_t)7, ERR_Fatal, 0, CNULL, CNULL);
471
472 if (EXPDBG(10, 256))
473 fprintf(gbl.dbgfil,
474 "adding rpct nme %d, based on nme %d, in rpct loop %d\n", rpct_nme,
475 orig_nme, rpct_loop);
476
477 BCOPY(nmeb.stg_base + rpct_nme, nmeb.stg_base + orig_nme, NME, 1);
478
479 NME_RPCT_LOOP(rpct_nme) = rpct_loop;
480 NME_HSHLNK(rpct_nme) = nmehsh[hashval];
481 nmehsh[hashval] = rpct_nme;
482
483 return rpct_nme;
484
485} /* end add_rpct_nme( int orig_nme, int rpct_loop ) */

493/** \brief Build an nme entry using a sym and an offset relative
494 to base address of this symbol */
495int
496build_sym_nme(SPTR sym, int offset, bool ptr_mem_op)
497{
498 int nme;
499 int sub;
500 bool inlarr;
501 DTYPE dt;
502 int i;
503
504 if (!sym)
505 return 0;
506
507 if (ptr_mem_op)
508 return 0;
509
510 nme = addnme(NT_VAR, sym, 0, 0);
511 dt = DTYPEG(sym);
512 if (DTY(dt) == TY_PTR) {
513 sub = 0;
514 inlarr = false;
515 for (i = nme; i < nmeb.stg_avail; i++) {
516 /* Check if subscript doesn't already exist */
517 if (NME_TYPE(i) == NT_IND && NME_NM(i) == nme && NME_SYM(i) == 0 &&
518 NME_CNST(i) == 0) {
519 sub = NME_SUB(i);
520#ifndef FE90
521 if (!ili_isdeleted(sub)) {
522#endif
523 inlarr = NME_INLARR(i);
524 break;
525#ifndef FE90
526 } else
527 sub = 0;
528#endif
529 }
530 }
531 nme = add_arrnme(NT_IND, SPTR_NULL, nme, 0, sub, inlarr);
532 dt = DTySeqTyElement(dt);
533 }
534 return 0;
535}

537/** \brief Get data type of a names entry
538 */
539DTYPE
540dt_nme(int nm)
541{
542 int i;
543 switch (NME_TYPE(nm)) {
544 case NT_INDARR:
545 case NT_ADD:
546 case NT_UNK:
547 break;
548 case NT_VAR:
549 return DTYPEG(NME_SYM(nm));
550 case NT_MEM:
551 if (NME_SYM(nm) == 0 || NME_SYM(nm) == 1) {
552 if (dt_nme((int)NME_NM(nm)) == DT_DCMPLX) {
553 return DT_DBLE;
554 } else {
555 return DT_REAL;
556 }
557 }
558 if ((i = NME_CNST(nm)) == 0)
559 return DTYPEG(NME_SYM(nm));
560 else
561 return DTYPEG(i);
562 case NT_ARR: {
563 DTYPE i = dt_nme((int)NME_NM(nm));
564 if (DTY(i) == TY_ARRAY)
565 return DTySeqTyElement(i);
566 /*
567 * for fortran, the TY_ARRAY dtype only occurs once; the element
568 * type is returned after the array dtype is returned when we hit
569 * the NT_VAR or NT_MEM case. All we need to do is to just pass
570 * up the dtype. NOTE that this will work if try to fake arrays
571 * using PLISTs.
572 */
573 return i;
574 }
575 case NT_IND: {
576 DTYPE i = dt_nme(NME_NM(nm));
577 if (DTY(i) == TY_PTR)
578 return DTySeqTyElement(i);
579 return i;
580 }
581 case NT_SAFE:
582 return dt_nme(NME_NM(nm));
583 }
584 return DT_NONE;
585}

587/** \brief Location of a names entry
588 *
589 * LOC (explicit or implicit) has been performed. The names entry for the
590 * lvalue is scanned to determine if a symbol's ADDRTKN flag must be set
591 */
592void
593loc_of(int nme)
594{
595 int type;
596
597 while (true) {
598 if ((type = NME_TYPE(nme)) == NT_VAR) {
599 if (!is_presym(nme))
600 ADDRTKNP(NME_SYM(nme), 1);
601 break;
602 }
603 if (type == NT_ARR || type == NT_MEM)
604 nme = NME_NM(nme);
605 else
606 break;
607 }
608}

777/**
778 * \brief Add a "runtime pointer conflict test" record
779 *
780 * This creates and returns a new RPCT (runtime pointer conflict test)
781 * record containing the pair of NMEs (rpct_nme1, rpct_nme2). These
782 * NMEs are 'RPCT NMEs' that were created by 'add_rpct_nme()'. The
783 * existence of this RPCT record indicates that a pointer conflict
784 * test has been generated which proves at runtime that this pair of
785 * references do not conflict, so 'conflict( rpct_nme1, rpct_nme2 )'
786 * will return NOCONFLICT.
787 */
788void
789add_rpct(int rpct_nme1, int rpct_nme2)
790{
791 int hashval, rpct;
792
793 asrt(rpct_nme1 != rpct_nme2 && NME_RPCT_LOOP(rpct_nme1) &&
794 NME_RPCT_LOOP(rpct_nme2) == NME_RPCT_LOOP(rpct_nme1));
795
796 /* If necessary swap 'rpct_nme1' & 'rpct_nme2' so that
797 * (rpct_nme1 < rpct_nme2), since that makes it quicker to search
798 * the RPCT table for a match.
799 */
800 if (rpct_nme1 > rpct_nme2) {
801 int tmp;
802
803 tmp = rpct_nme1;
804 rpct_nme1 = rpct_nme2;
805 rpct_nme2 = tmp;
806 }
807
808 /* Compute the hash index for this RPCT record.
809 */
810 hashval = (int)((rpct_nme1 ^ rpct_nme2) & 0x7fff) % NMEHSZ;
811
812#if DEBUG
813 /* Search the RPCT hash links for this RPCT. If it already exists
814 * we generate an 'asrt()' error message, since this shouldn't happen.
815 */
816 for (rpct = rpcthsh[hashval]; rpct; rpct = RPCT_HSHLNK(rpct)) {
817 if (RPCT_NME1(rpct) == rpct_nme1 && RPCT_NME2(rpct) == rpct_nme2) {
818 /* Found.
819 */
820 asrt(false); /* we don't expect it to be found! */
821 return;
822 }
823 }
824#endif
825
826 /* Create and initialise a new RPCT record and link it into its hash chain.
827 */
828 rpct = STG_NEXT(nmeb.rpct);
829 RPCT_NME1(rpct) = rpct_nme1;
830 RPCT_NME2(rpct) = rpct_nme2;
831 RPCT_HSHLNK(rpct) = rpcthsh[hashval];
832 rpcthsh[hashval] = rpct;
833
834} /* end add_rpct( int rpct_nme1, int rpct_nme2 ) */

947/**\brief Determine if the two names table entries, nm1 and nm2, conflict.
948 *
949 * Return SAME, CONFLICT, NOCONFLICT or UNKCONFLICT
950 */
951int
952conflict(int nm1, int nm2)
953{
954 int t1, t2, n1, n2, sptr2, sptr1, c;
955
956 /* if necessary, switch nm1 and nm2 so that t2 >= t1
957 (reduces number of cases later): */
958
959 if (nm1 == NME_VOL || nm2 == NME_VOL)
960 return CONFLICT;
961
962 if ((t2 = NME_TYPE(nm2)) < (t1 = NME_TYPE(nm1))) {
963 c = nm1;
964 nm1 = nm2;
965 nm2 = c;
966 t1 = t2;
967 t2 = NME_TYPE(nm2);
968 }
969
970 if (gbl.internal > 1 && nm1 != nm2) {
971 if (NME_TYPE(nm1) == NT_VAR && NME_SYM(nm1) == aux.curr_entry->display)
972 return NOCONFLICT;
973 if (NME_TYPE(nm2) == NT_VAR && NME_SYM(nm2) == aux.curr_entry->display)
974 return NOCONFLICT;
975 }
976
977 if (t2 == NT_MEM || t2 == NT_ARR) {
978 /* scan back past member and array records: */
979
980 for (n1 = nm1; NME_TYPE(n1) >= NT_MEM; n1 = NME_NM(n1))
981 ;
982 for (n2 = nm2; NME_TYPE(n2) >= NT_MEM; n2 = NME_NM(n2))
983 ;
984
985 c = conflict(n1, n2);
986 /* check if one if .real and one is .imag */
987 if (t1 == NT_MEM && t2 == NT_MEM &&
988 ((int)(NME_SYM(nm1) | NME_SYM(nm2)) <= 1) &&
989 NME_SYM(nm1) != NME_SYM(nm2)) {
990 return NOCONFLICT;
991 }
992 if (!XBIT(104, 0x10) && (c == CONFLICT || c == UNKCONFLICT)) {
993 int m1 = 0, m2 = 0;
994 int n1m, n2m;
995 /* if either nm1 or nm2 has a member or array of a member
996 * that is marked as noconflict, there can be no conflict
997 * this is particularly to handle section descriptor members */
998 n1m = nm1;
999 while (NME_TYPE(n1m) == NT_ARR)
1000 n1m = NME_NM(n1m);
1001 /* problem with block moves */
1002 if (NME_TYPE(n1m) == NT_IND && NME_SYM(n1m) == NME_NULL) {
1003 return UNKCONFLICT;
1004 }
1005 if (NME_TYPE(n2) == NT_IND && NME_SYM(n2) == NME_NULL) {
1006 return UNKCONFLICT;
1007 }
1008 if (NME_TYPE(n1m) == NT_MEM) {
1009 m1 = NME_SYM(n1m);
1010 }
1011 n2m = nm2;
1012 while (NME_TYPE(n2m) == NT_ARR)
1013 n2m = NME_NM(n2m);
1014 if (NME_TYPE(n2m) == NT_MEM) {
1015 m2 = NME_SYM(n2m);
1016 }
1017 if (m1 > NOSYM && m1 == m2 && NOCONFLICTG(m1)) {
1018 /* same noconflict member */
1019 if (NME_TYPE(nm1) == NT_ARR && NME_TYPE(nm2) == NT_ARR) {
1020 /* constant dimension? */
1021 if (NME_SYM(nm1) == 0 && NME_SYM(nm2) == 0) {
1022 if (NME_CNST(nm1) != NME_CNST(nm2))
1023 return NOCONFLICT;
1024 return SAME;
1025 }
1026 }
1027 /* same member */
1028 return UNKCONFLICT;
1029 } else {
1030/* if either is a descriptor symbol, no conflict */
1031#ifdef DESCARRAYG
1032 if (m1 > NOSYM && DESCARRAYG(m1))
1033 return NOCONFLICT;
1034 if (m2 > NOSYM && DESCARRAYG(m2))
1035 return NOCONFLICT;
1036#endif
1037 /* if either comes from inlining, there might be a conflict
1038 * even if NOCONFLICT is set on the member */
1039 {
1040 int nn1, nn2, sptr1, sptr2;
1041 for (nn1 = n1; NME_NM(nn1); nn1 = NME_NM(nn1))
1042 ;
1043 for (nn2 = n2; NME_NM(nn2); nn2 = NME_NM(nn2))
1044 ;
1045 sptr1 = NME_SYM(nn1);
1046 sptr2 = NME_SYM(nn2);
1047 if (sptr1 > NOSYM && INLNG(sptr1) && UNSAFEG(sptr1))
1048 return c;
1049 if (sptr2 > NOSYM && INLNG(sptr2) && UNSAFEG(sptr2))
1050 return c;
1051 }
1052
1053 /* if either member is a NOCONFLICT symbol, no conflict */
1054 if (m1 > NOSYM && NOCONFLICTG(m1))
1055 return NOCONFLICT;
1056 if (m2 > NOSYM && NOCONFLICTG(m2))
1057 return NOCONFLICT;
1058 }
1059 }
1060 if (c != SAME)
1061 return c;
1062
1063 /* check if these are complex elements: */
1064
1065 if (t1 == NT_MEM && t2 == NT_MEM) {
1066
1067 /* If complex, real NME_SYM==0, imag NME_SYM==1 */
1068
1069 /* ****************** */
1070 /* Test for 2 Complex */
1071 /* ****************** */
1072
1073 if ((int)(NME_SYM(nm1) | NME_SYM(nm2)) <= 1) {
1074
1075 /* If both complex but one is real part *
1076 * and one is complex part, then OK */
1077
1078 if (NME_SYM(nm1) != NME_SYM(nm2))
1079 return NOCONFLICT;
1080
1081 /* Set up for traversing back into the array processing */
1082 nm1 = NME_NM(nm1);
1083 nm2 = NME_NM(nm2);
1084
1085 }
1086 /* ****************************** */
1087 /* Test for 1 Complex w/ 1 Member */
1088 /* ****************************** */
1089 else if ((NME_SYM(nm1) <= 1 || NME_SYM(nm2) <= 1)) {
1090 if (!XBIT(70, 0x40000000))
1091 return NOCONFLICT;
1092 else if (NME_SYM(nm1) <= 1) {
1093 if (NME_SYM(nm2) == NME_NM(nm1))
1094 return CONFLICT;
1095 else
1096 return NOCONFLICT;
1097 } else {
1098 if (NME_SYM(nm1) == NME_NM(nm2))
1099 return CONFLICT;
1100 else
1101 return NOCONFLICT;
1102 }
1103 }
1104 /* ****************** */
1105 /* Test for 2 Members */
1106 /* ****************** */
1107 else {
1108 int sptr1, sptr2;
1109
1110 if (nm1 == nm2)
1111 return conflict(NME_NM(nm1), NME_NM(nm2));
1112
1113 sptr1 = NME_SYM(nm1);
1114 sptr2 = NME_SYM(nm2);
1115 for (;;) {
1116 sptr1 = SYMLKG(sptr1);
1117 if (sptr1 <= NOSYM)
1118 break;
1119 if (sptr1 == sptr2)
1120 return NOCONFLICT;
1121 }
1122 sptr1 = NME_SYM(nm1);
1123 for (;;) {
1124 sptr2 = SYMLKG(sptr2);
1125 if (sptr2 <= NOSYM)
1126 break;
1127 if (sptr2 == sptr1)
1128 return NOCONFLICT;
1129 }
1130 return CONFLICT;
1131 }
1132 } else if (t1 == NT_MEM) {
1133 /* Then t2 == NT_ARR and they are SAME */
1134 int sptr1, sptr2;
1135
1136 if (nm1 == nm2)
1137 return conflict(NME_NM(nm1), nm2);
1138
1139 sptr1 = NME_SYM(nm1);
1140
1141 for (n2 = nm2; NME_TYPE(n2) == NT_ARR; n2 = NME_NM(n2))
1142 ;
1143
1144 sptr2 = NME_SYM(n2);
1145 if (sptr2 <= NOSYM) {
1146 assert(0, "conflict: Unexpected member/array conflict. nm2: ", nm2,
1147 ERR_Informational);
1148 return CONFLICT;
1149 }
1150
1151 for (;;) {
1152 sptr1 = SYMLKG(sptr1);
1153 if (sptr1 <= NOSYM)
1154 break;
1155 if (sptr1 == sptr2)
1156 return NOCONFLICT;
1157 }
1158 sptr1 = NME_SYM(nm1);
1159 for (;;) {
1160 sptr2 = SYMLKG(sptr2);
1161 if (sptr2 <= NOSYM)
1162 break;
1163 if (sptr2 == sptr1)
1164 return NOCONFLICT;
1165 }
1166 return CONFLICT;
1167 } else if (t2 == NT_MEM) {
1168/* same base nme, but one is a MEM and one a VAR */
1169#if DEBUG
1170 assert(t1 == NT_VAR, "conflict: t1 not NT_VAR", t1, ERR_Severe);
1171#endif
1172 return CONFLICT;
1173 }
1174
1175 /* They are possibly both arrays. Check now that they
1176 * might be arrays that are members
1177 */
1178 for (n1 = nm1, n2 = nm2; NME_TYPE(n1) == NT_ARR && NME_TYPE(n2) == NT_ARR;
1179 n1 = NME_NM(n1), n2 = NME_NM(n2))
1180 ;
1181 if (NME_TYPE(n1) == NT_MEM || NME_TYPE(n2) == NT_MEM) {
1182 if (NME_TYPE(n1) == NT_MEM && NME_TYPE(n2) == NT_MEM && n1 != n2) {
1183 /* If members are not identical then noconflict (right?) */
1184 if (!XBIT(104, 1))
1185 return NOCONFLICT;
1186 }
1187 return CONFLICT;
1188 }
1189
1190 /* scan back thru array dimensions to find non-equal const dim: */
1191
1192 c = SAME;
1193 for (n1 = nm1, n2 = nm2; NME_TYPE(n1) == NT_ARR;
1194 n1 = NME_NM(n1), n2 = NME_NM(n2)) {
1195 if (NME_TYPE(n2) != NT_ARR) {
1196 if (flg.depchk)
1197 return CONFLICT;
1198 return UNKCONFLICT;
1199 }
1200 if (NME_SYM(n1) == 0 && NME_SYM(n2) == 0) {
1201 /* both dimensions are constant. */
1202 if (NME_CNST(n1) != NME_CNST(n2))
1203 return NOCONFLICT;
1204 } else
1205 c = UNKCONFLICT;
1206 }
1207
1208 return c;
1209 }
1210
1211 switch (t1) {
1212 case NT_UNK:
1213 if (t2 == NT_UNK)
1214 return CONFLICT;
1215 else if (t2 != NT_VAR)
1216 break;
1217 sptr2 = NME_SYM(nm2);
1218 if (STYPEG(sptr2) == ST_CONST) {
1219 return NOCONFLICT;
1220 }
1221 if (SOCPTRG(sptr2))
1222 break;
1223 return NOCONFLICT;
1224
1225 case NT_VAR:
1226 if (t2 == NT_VAR) {
1227 if (nm1 == nm2)
1228 return SAME;
1229 sptr1 = NME_SYM(nm1);
1230 sptr2 = NME_SYM(nm2);
1231 if (STYPEG(sptr1) == ST_CONST) {
1232 return NOCONFLICT;
1233 }
1234 if (STYPEG(sptr2) == ST_CONST) {
1235 return NOCONFLICT;
1236 }
1237 /* check for overlapping symbols: */
1238 if (SOCPTRG(sptr1) && flg.depchk) {
1239 if (SOCPTRG(sptr2))
1240 for (t1 = SOCPTRG(sptr1); t1; t1 = SOC_NEXT(t1))
1241 if (sptr2 == SOC_SPTR(t1))
1242 return CONFLICT;
1243 }
1244
1245 /* If sptr1 and sptr2 are overlapping common block variables, do not
1246 * optimize their loads/stores, like how variables in an equivalence
1247 * statement are handled.
1248 */
1249 if (is_overlap_cmblk_var(sptr1, sptr2))
1250 return CONFLICT;
1251 if (!hlcf || XBIT(104, 0x8)) {
1252 if (SCG(sptr1) == SC_BASED && INLNG(sptr1) && UNSAFEG(sptr1) &&
1253 SCG(sptr2) == SC_BASED && INLNG(sptr2) && UNSAFEG(sptr2))
1254 /* HACK-in dependence for multiple instances of inlined
1255 * functions and inline-created Cray pointees.
1256 */
1257 return UNKCONFLICT;
1258
1259 if (SCG(sptr1) == SC_BASED && INLNG(sptr1) && UNSAFEG(sptr1)) {
1260 int dty;
1261 dty = DTY(DTYPEG(sptr2));
1262 if ((dty == TY_ARRAY || dty == TY_STRUCT || dty == TY_UNION) &&
1263 ADDRTKNG(sptr2)
1264#ifdef PTRSAFEG
1265 && !PTRSAFEG(sptr2)
1266#endif
1267 ) {
1268 return UNKCONFLICT;
1269 }
1270 } else if (SCG(sptr2) == SC_BASED && INLNG(sptr2) && UNSAFEG(sptr2)) {
1271 int dty;
1272 dty = DTY(DTYPEG(sptr1));
1273 if ((dty == TY_ARRAY || dty == TY_STRUCT || dty == TY_UNION) &&
1274 ADDRTKNG(sptr1)
1275#ifdef PTRSAFEG
1276 && !PTRSAFEG(sptr1)
1277#endif
1278 ) {
1279 return UNKCONFLICT;
1280 }
1281 }
1282 }
1283 return NOCONFLICT;
1284 }
1285 break;
1286 case NT_IND:
1287#ifndef FE90
1288 /* F90 back end */
1289 if (!F90_nme_conflict(nm1, nm2)) {
1290 return NOCONFLICT;
1291 }
1292#endif
1293 if (t2 == NT_VAR) {
1294 sptr2 = NME_SYM(nm2);
1295 if (STYPEG(sptr2) == ST_CONST) {
1296 return NOCONFLICT;
1297 }
1298 if (SOCPTRG(sptr2) == 0 &&
1299 (NOCONFLICTG(sptr2) || (CCSYMG(sptr2) && XBIT(104, 0x20)))) {
1300 /* nme1 is a pointer, nme2 is a symbol
1301 * not declared as TARGET, so no conflict */
1302 return NOCONFLICT;
1303 }
1304 if (PTRSAFEG(sptr2)) {
1305 return NOCONFLICT;
1306 }
1307 } else if (t2 == NT_ARR) {
1308 int nm22, t22;
1309 for (nm22 = nm2; NME_TYPE(nm22) == NT_ARR; nm22 = NME_NM(nm22))
1310 ;
1311 t22 = NME_TYPE(nm22);
1312 if (t22 == NT_VAR) {
1313 sptr2 = NME_SYM(nm2);
1314 if (STYPEG(sptr2) == ST_CONST) {
1315 return NOCONFLICT;
1316 }
1317 if (SOCPTRG(sptr2) == 0 && NOCONFLICTG(sptr2)) {
1318 /* nme1 is a pointer, nme2 is an array symbol
1319 * not declared as TARGET, so no conflict */
1320 return NOCONFLICT;
1321 }
1322 }
1323 }
1324 break;
1325 }
1326
1327 /* for fortran, may reach here because of inlining or use of ptr types... */
1328 return CONFLICT;
1329} /* endroutine conflict */
