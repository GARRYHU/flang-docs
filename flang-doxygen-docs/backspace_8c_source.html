17/** \brief this must match RCWSZ defined in unf.c - is the number of bytes used
18 to represent the length that is stored with an unformatted record */
19#define RCWSZ sizeof(int)

21static int
22_f90io_backspace(__INT_T *unit, __INT_T *bitv, __INT_T *iostat, int swap_bytes)
23{
24 FIO_FCB *f;
25 FILE *fp;
26
27 __fortio_errinit03(*unit, *bitv, iostat, "BACKSPACE");
28 if (ILLEGAL_UNIT(*unit))
29 return __fortio_error(FIO_EUNIT);
30
31 f = __fortio_find_unit(*unit);
32
33 if (f == NULL) /* is unit connected? */
34 return 0; /* for VMS compat, treat as non-error */
35
36 /* check for outstanding async i/o */
37
38 if (f->asy_rw) { /* stop any async i/o */
39 f->asy_rw = 0;
40 if (Fio_asy_disable(f->asyptr) == -1) {
41 return (__fortio_error(__io_errno()));
42 }
43 }
44
45 /* no backspace for direct access files: */
46 if (f->acc == FIO_DIRECT)
47 return __fortio_error(FIO_EDIRECT);
48
49 /* if eof record has just been written, toggle eof_flag and return: */
50 if (f->eof_flag) {
51 f->eof_flag = FALSE;
52 return 0;
53 }
54
55 if (f->binary)
56 return 0; /* MODE=binary is a nop */
57
58 if (f->byte_swap)
59 swap_bytes = 1;
60 else if (f->native)
61 swap_bytes = 0;
62
63 /* append carriage return (maybe) */
64
65 if (f->nonadvance) {
66 f->nonadvance = FALSE;
67#if defined(_WIN64)
68 if (__io_binary_mode(f->fp))
69 __io_fputc('\r', f->fp);
70#endif
71 __io_fputc('\n', f->fp);
72 if (__io_ferror(f->fp))
73 return __io_errno();
74 }
75
76 fp = f->fp;
77 /* if already at the beginning just return without error */
78 /* if (f->nextrec < 2) */
79 if (__io_ftell(fp) == 0) /* use ftell in case file opened 'append' */
80 return 0;
81
82 if (f->form == FIO_UNFORMATTED) { /* CASE 1: unformatted file */
83 int reclen;
84 /* variable length record is stored as length:record:length
85 so back up over trailing length field, read the size, then
86 back up over the record and both length fields */
87
88 rec_continued:
89 if (__io_fseek(fp, -((seekoffx_t)RCWSZ), SEEK_CUR) != 0)
90 return __fortio_error(__io_errno());
91
92 if (__io_fread(&reclen, RCWSZ, 1, fp) != 1)
93 return __fortio_error(__io_errno());
94
95 /* NOTE: reclen in FCB and in file is always in units of bytes */
96 if (swap_bytes)
97 __fortio_swap_bytes((char *)&reclen, __INT, 1);
98 if (__io_fseek(fp, -((reclen & 0x7fffffff) + (seekoffx_t)(2 * RCWSZ)),
99 SEEK_CUR) != 0)
100 return __fortio_error(__io_errno());
101 if (reclen & 0x80000000)
102 goto rec_continued;
103 f->coherent = 0; /* avoid unnecessary seek later on */
104 } else { /* CASE 2: formatted file */
105 seekoffx_t pos;
106 assert(f->form == FIO_FORMATTED);
107 pos = __io_ftell(fp) - 1;
108 assert(pos >= 0);
109 while (TRUE) {
110 if (pos > 0)
111 --pos;
112 if (__io_fseek(fp, pos, SEEK_SET) != 0)
113 return __fortio_error(__io_errno());
114
115 if (pos == 0 || __io_fgetc(fp) == '\n') {
116 /* must set coherent flag to 'read' in case the next operation
117 on this file is a write: */
118 f->coherent = 2 /*read*/;
119 break;
120 }
121 }
122 }
123
124 --f->nextrec;
125 f->truncflag = TRUE; /* if next operation is write, truncate */
126 return 0; /* no error occurred */
127}
