42<pre>
43 the type NT_INDARR is used to distinguish the cases:
44 typedef float* fp;
45 typedef float f10[10];
46 fp* ppf; // pointer to pointer to float
47 fp a10pf[10]; // array of 10 pointers to float
48 f10* pa10f; // pointer to array of 10 floats
49 ppf[i][j] // appears as IM_ELEMENT -> IM_PLD -> IM_ELEMENT ->
50 IM_PLD -> ppf
51 a10pf[i][j] // appears as IM_ELEMENT -> IM_PLD -> IM_ELEMENT ->
52 a10pf
53 pa10f[i][j] // appears as IM_ELEMENT -> IM_ELEMENT -> IM_PLD ->
54 pa10f
55 NT_INDARR is used when an IM_ELEMENT -> IM_PLD appears in the ILM file.
56</pre>
57 */
