11vfloat __attribute__((noinline)) atan2_vec(vfloat const y, vfloat const x) {
12
13 // Helpful constants:
14 const vfloat pi_3_over_4 = vcast_vf_f(PI_3_OVER_4);
15 const vfloat pi_over_4 = vcast_vf_f(PI_OVER_4);
16 const vfloat pi_2 = vcast_vf_f(PI_OVER_2);
17 const vfloat pi = vcast_vf_f(PI);
18
19 unsigned long long int AbsMask = 0x7FFFFFFF;
20 float AbsMask_as_float = *(float *)&AbsMask;
21
22 vfloat yAbs = vreinterpret_vf_vm(
23 vand_vm_vm_vm(vreinterpret_vm_vf(y),
24 vreinterpret_vm_vf(vcast_vf_f(AbsMask_as_float))));
25
26 vfloat xAbs = vreinterpret_vf_vm(
27 vand_vm_vm_vm(vreinterpret_vm_vf(x),
28 vreinterpret_vm_vf(vcast_vf_f(AbsMask_as_float))));
29
30 vopmask yBigger = vgt_vo_vf_vf(yAbs, xAbs);
31
32#ifdef PERF_USE_TWO_DIVIDES
33 // Different vector sizes seems to prefer different code here, most are
34 // faster with using the 2 divides here, with the exception of avx512 which
35 // is (noticably) faster with just the 1 divide.
36 vfloat xReduced = vsel_vf_vo_vf_vf(yBigger, x / y, y / x);
37#else
38 // This seems to have better performance on avx512, while the other is
39 // better for everything else
40 vfloat xReduced =
41 vsel_vf_vo_vf_vf(yBigger, x, y) / vsel_vf_vo_vf_vf(yBigger, y, x);
42#endif
43
44 // The same Estrin scheme as is used in atan(x):
45 vfloat x2 = xReduced * xReduced;
46 vfloat x4 = x2 * x2;
47 vfloat x8 = x4 * x4;
48 vfloat x16 = x8 * x8;
49
50 // Convert our polynomial constants into vectors:
51 const vfloat D1 = vcast_vf_f(C1);
52 const vfloat D2 = vcast_vf_f(C2);
53 const vfloat D3 = vcast_vf_f(C3);
54 const vfloat D4 = vcast_vf_f(C4);
55 const vfloat D5 = vcast_vf_f(C5);
56 const vfloat D6 = vcast_vf_f(C6);
57 const vfloat D7 = vcast_vf_f(C7);
58 const vfloat D8 = vcast_vf_f(C8);
59
60 // First layer of Estrin:
61 vfloat L1 = vfma_vf_vf_vf_vf(x2, D2, D1);
62 vfloat L2 = vfma_vf_vf_vf_vf(x2, D4, D3);
63 vfloat L3 = vfma_vf_vf_vf_vf(x2, D6, D5);
64 vfloat L4 = vfma_vf_vf_vf_vf(x2, D8, D7);
65
66 // Second layer of estrin
67 vfloat M1 = vfma_vf_vf_vf_vf(x4, L2, L1);
68 vfloat M2 = vfma_vf_vf_vf_vf(x4, L4, L3);
69
70 vfloat poly = vfma_vf_vf_vf_vf(x8, M2, M1);
71
72 vfloat result_f = poly;
73
74 // A vfloat that contains 0x80000000:
75 unsigned long long int SignBit = 0x80000000;
76 float SignBit_as_float = *(float *)&SignBit;
77 vfloat SignBit_as_floatV = vcast_vf_f(SignBit_as_float);
78
79 // pi/2 with the sign of xReduced:
80 vfloat signedPi_2 = vreinterpret_vf_vm(
81 vreinterpret_vm_vf(pi_2) |
82 (vreinterpret_vm_vf(xReduced) & vreinterpret_vm_vf(SignBit_as_floatV)));
83
84 vfloat pi_factor = vsel_vf_vo_vf_vf(yBigger, signedPi_2, vcast_vf_f(0.0f));
85
86 xReduced = vsel_vf_vo_vf_vf(yBigger, -xReduced, xReduced);
87
88 result_f = vfma_vf_vf_vf_vf(x2 * xReduced, poly, xReduced);
89
90 // copysign(pi, y):
91 const vfloat signedPi = vreinterpret_vf_vm(
92 vreinterpret_vm_vf(pi) |
93 (vreinterpret_vm_vf(y) & vreinterpret_vm_vf(SignBit_as_floatV)));
94
95 // Needs to include -0.0 in the negative comparison
96 // We do this by cast to an int and comparing to -1:
97 vint2 pi_factor_as_vmask =
98 (~vgt_vi2_vi2_vi2(vreinterpret_vi2_vf(x), vcast_vi2_i(-1))) &
99 vreinterpret_vi2_vf(signedPi);
100
101 pi_factor += vreinterpret_vf_vi2(pi_factor_as_vmask);
102
103 result_f += pi_factor;
104
105 // We need to mask off some special values, mainly infinities and 0's
106 // Fortunately for all cases we always have (xAbs == yAbs)
107 // Get special return value:
108 vfloat special_return_value = vsel_vf_vo_vf_vf(
109 vlt_vo_vf_vf(x, vcast_vf_f(0.0f)), pi_3_over_4, pi_over_4);
110
111 // Special return values for (y, x) = (+-0, +-0)
112 vint2 special_yx_zero_zero_as_int =
113 ~vgt_vi2_vi2_vi2(vreinterpret_vi2_vf(x), vcast_vi2_i(-1)) &
114 vreinterpret_vi2_vf(pi);
115
116 vfloat special_yx_zero_zero =
117 vreinterpret_vf_vi2(special_yx_zero_zero_as_int);
118
119 // Check for (y, x) = (+-0.0, +-0.0)
120 special_return_value =
121 vsel_vf_vo_vf_vf(veq_vo_vf_vf(x, vcast_vf_f(0.0f)),
122 special_yx_zero_zero, special_return_value);
123
124 result_f = vsel_vf_vo_vf_vf(veq_vo_vf_vf(yAbs, xAbs), special_return_value,
125 result_f);
126
127 // copysign(result_f, y):
128 result_f = vreinterpret_vf_vm(
129 vreinterpret_vm_vf(result_f) |
130 (vreinterpret_vm_vf(y) & vreinterpret_vm_vf(SignBit_as_floatV)));
131
132 return result_f;
133}
