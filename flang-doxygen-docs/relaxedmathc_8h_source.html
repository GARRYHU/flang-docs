132 /*
133 * This algorithm was orignally developed by Peter Tang. It is based on
134 * the work described in:
135 * P. T. P. Tang. Table driven implementation of the exponential function
136 * in IEEE floating-point arithmetic. ACM Transactions on Mathematical
137 * Software, 15(2):144-157, June 1989
138 *
139 * I have short circuited some of the things Tang did for higher accuracy in
140 * the interest
141 * of performance.
142 * Testing indicates that for degree-2 polynomial, accuracy ~ 2.0x10-7 and for
143 * degree-3 polynomial, accuracy ~4.0x10-10.
144 */
145
146 long long itemp;
147 int m, j;
148 double r, q, s, result;
149 double xprime, temp, tempx;
150 double a2r, a1r, rsq, rcube;
151 union {
152 unsigned long long i;
153 double d;
154 } convert;
155 double half = 0.5;
156 int n, n1, n2;
157 /*
158 * for now we will introduce ntemp;
159 */
160 int ntemp;
161 int addmod = 0;
162 int n1pj;
163 double n1pjln2inv32;
164 tempx = x * invl2x32; // x * 32.0/ln(2) -- scaled x
165 if (fabs(x) > 200.0) // guard against over/underflow
166 return (CONCAT(__fsd_exp_,TARGET_VEX_OR_FMA)(x));
167 if (x < 0.0) { // We need to know if x < 0.0
168 half = -0.5; // For rounding to nearest
169 addmod = 32; // Add to computer % for correct mod
170 }
171 tempx += half; // Assures rounding to nearest
172 n = (int)(tempx); // integer part of tempx rounded to nearest
173 /*
174 * Create our own modulo function
175 *
176 */
177 ntemp = n >> 5; // Divide by 32
178 ntemp = ntemp << 5; // And then multiply by 32
179 j = n - ntemp; // this is n%32 modulo a sign issue
180 n1 = n - j; // this is almost m ...
181 m = n1 >> 5; // ... once we divide by 32
182 n1pj = n1 + j; // add j to n1
183 n1pjln2inv32 = ln2xinv32 * (double)n1pj; // then multiply it by ln(2)/32.0
184 r = x - n1pjln2inv32; // get the remainder
185 convert.i = ten23 + m; // 1023 << 52 = 2.0
186 convert.i = convert.i << 52; // convert.d now has 2^m
187 s = table[j]; // get 2^(j/32)
188 s *= convert.d; // * 2^m
189 /*
190 * We're going to use just a few terms of the expansion.
191 *
192 * p(t) = t + t^2/2.0 + t^3/6.0. We can add 3rd term if needed.
193 *
194 */
195 /*
196 * for degree 3 polynomial,
197 *
198 * q = r * ( 1 + r * ( a1 + a2 * r ) )
199 *
200 */
201
202 q = r * a2;
203 q += a1;
204 q *= r;
205 q += 1.0;
206 q *= r;
207
208 /*
209 * For degree 2 polynomial
210 *
211 * q = r * ( 1 + a1 * r )
212 */
213 /*
214 q = a1 * r;
215 q += 1.0;
216 q *= r;
217 */
218 result = 1.0 + q; // ~exp(r)
219 result *= s; // ~exp(x)
220 return (result);
221}

254/*
255 * Because the range of input values for sines and cosines this code has to
256 * solve
257 * is limited to angles between +/- 100, and because the accuracy constraints
258 * are
259 * fairly loose, this program does the reduction phase using the most obvious
260 * method, namely dividing the input angle by pi/2.0. Higher accuracy is not
261 * required.
262 * Once the reduced angle is determined, we need six more steps.
263 * 1. Approximate the sincos for the reduced angle. We do this with a lookup
264 * table.
265 * 2. Subtract the angle for which we have entries in the table from the
266 * reduced angle. This will leave a small residual angle for which the sine
267 * and cosine need to be computed.
268 * 3. Use a minmax polynomial to compute the remainder angle. (See Muller, pages
269 * 58,
270 * 59)
271 * 4. Determine in which quadrant the angle resides to properly assine signs and
272 * whether, for instance, and then decide how properly to use the information
273 * in step 5. A modulo operation determines this.
274 * 5.. Use the trig identities for computing sines and cosines for sums of
275 * angles:
276 * sin( a + b ) = sin( a ) * cos( b ) + cos( a ) * sin( b )
277 * cos( a + b ) = cos( a ) * cos( b ) - sin( a ) * sin( b )
278 * 6. Correct the sign for the sine depending on whether x >= 0 or x < 0.
279 */

282 * The next two coeffiecients used for low order sincos approximation

285#define S33 -0.1666596 /* Multiplies x^3 for sin */
286#define S22 -0.49996629 /* Multiples x^2 for cos */

288void
289CONCAT(__rsd_sincos_c_,TARGET_VEX_OR_FMA)(double x, double *s, double *c)
290{
291 double kpihalves; // pihalves * k
292 int k; // x/pihalves
293 int sindex, cindex; // indices into table
294 double tsin, tcos; // sine and cosine from table
295 double tsinss, tsinsc, tcosss, tcossc; // products of step 5 above
296 double mysin, mycos; // final sine and cosine
297 double x2, x3; // remains**2 and remains * x2
298 double smallsin, smallcos; // sine and cosine of remainder
299 double x2s22, x3s33; // x2 * s22 and x3 * s33
300 double remainder, remains; // used to compute x - k * pi/2
301 double sign; // sign( x )
302 double fabsx; // absolute value of x
303 double r; // temporary storage for remainder
304 int mod4; // This is used to compute k%4
305
306 fabsx = fabs(x); // Need |x|
307 if (fabsx < 100.0) { // We'll work only on "small" angles
308 sign = 1.0 - 2.0 * (x < 0.0); // Faster than test and branch
309 k = fabsx * invpihalves; // How many 1/2 rotations in |x|
310 kpihalves = (double)k * pihalves; // now we should be at k * pi/2
311 remainder = fabsx - kpihalves; // x - k * pi/2
312 sindex = remainder * invangle_inc; // generate index into sine table
313 cindex = ts - sindex; // generate index inot cosine table
314 tsin = scArray[sindex]; // fetch sine from table
315 tcos = scArray[cindex]; // fetch cosine from table
316 r = (double)sindex * angle_inc; // Need x - k * pi/2 - small-angle
317 remains = remainder - r; // sin( remains ) has to be computed
318 x2 = remains * remains; // x'^2 starts polynomial expansion
319 x3 = x2 * remains; // x'^3
320 x3s33 = x3 * S33; // x'^3 * S33
321 x2s22 = x2 * S22; // x'^2 * S22
322 smallsin = x3s33 + remains; // ~sin( x' )
323 smallcos = x2s22 + 1.0; // ~cos( x' )
324 tsinss = tsin * smallsin; // part of sin( a + b )
325 tsinsc = tsin * smallcos; // " "
326 tcosss = tcos * smallsin; // " "
327 tcossc = tcos * smallcos; // " "
328 mod4 = k >> 2; // Computing mod with shifts
329 mod4 = mod4 << 2; // for k%4 is faster than k%4
330 mod4 = k - mod4; //
331
332 switch (mod4) { // mod4 selects the quadrant to ..
333 case 0: // .. properly compute sincos()
334 mysin = tsinsc + tcosss;
335 mysin = sign * mysin;
336 mycos = tcossc - tsinss;
337 *s = mysin;
338 *c = mycos;
339 break;
340 case 1:
341 mysin = tcossc - tsinss;
342 mysin = sign * mysin;
343 mycos = -(tsinsc + tcosss);
344 *s = mysin;
345 *c = mycos;
346 break;
347 case 2:
348 mysin = -(tsinsc + tcosss);
349 mysin = sign * mysin;
350 mycos = tsinss - tcossc;
351 *s = mysin;
352 *c = mycos;
353 break;
354 case 3:
355 mysin = tsinss - tcossc;
356 mysin = sign * mysin;
357 mycos = tsinsc + tcosss;
358 *s = mysin;
359 *c = mycos;
360 }
361 } else {
362 *s = sin(x); // We go here if |x| > 100.0
363 *c = cos(x);
364 }
365}
