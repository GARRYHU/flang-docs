108/* Allocate memory for reference symbols with size of stb.stg_avail */
109void
110allocate_refsymbol(int symavl)
111{
112 if (dbgref_symbol.symnum == NULL) {
113 dbgref_symbol.symnum = (int *)(malloc((symavl + 10) * sizeof(int)));
114 dbgref_symbol.altname =
115 (mod_altptr *)(malloc((symavl + 10) * sizeof(mod_altptr)));
116 dbgref_symbol.size = symavl + 10;
117 BZERO((void *)dbgref_symbol.symnum, int, (dbgref_symbol.size));
118 BZERO((void *)dbgref_symbol.altname, mod_altptr, (dbgref_symbol.size));
119 } else if (dbgref_symbol.size <= symavl) {
120 dbgref_symbol.symnum =
121 (int *)(realloc(dbgref_symbol.symnum, (symavl + 10) * sizeof(int)));
122 dbgref_symbol.altname = (mod_altptr *)(realloc(
123 dbgref_symbol.altname, (symavl + 10) * sizeof(mod_altptr)));
124
125 BZERO((void *)(dbgref_symbol.symnum + dbgref_symbol.size), int,
126 symavl - dbgref_symbol.size + 10);
127 BZERO((void *)(dbgref_symbol.altname + dbgref_symbol.size), mod_altptr,
128 symavl - dbgref_symbol.size + 10);
129 dbgref_symbol.size = symavl + 10;
130 }
131}

133/* reinitialize reference symbols from symavl on,
134 * we want to keep anything under symavl because that could come from module.
135 */
136static void
137reinit_refsymbol(int symavl)
138{
139 int i;
140 mod_altptr symptr;
141
142 if (symavl > dbgref_symbol.size)
143 return;
144
145 /* zero out all symbols that are referenced in previous routine if any */
146 BZERO((void *)dbgref_symbol.symnum, int, dbgref_symbol.size);
147
148 /* Keep USEd names around for module */
149 for (i = symavl; i < dbgref_symbol.size; ++i) {
150 for (; dbgref_symbol.altname[i]; dbgref_symbol.altname[i] = symptr) {
151 symptr = dbgref_symbol.altname[i]->next;
152 FREE(dbgref_symbol.altname[i]);
153 }
154 dbgref_symbol.altname[i] = NULL;
155 }
156}

158/* Create link list of renames */
159void
160set_modusename(int local, int global)
161{
162 if (dbgref_symbol.size <= stb.stg_avail) {
163 allocate_refsymbol(stb.stg_avail);
164 }
165
166 /* To avoid duplicate names, because of _parser
167 * symnum should be set -2
168 */
169 if (dbgref_symbol.symnum[local] == -2) {
170 dbgref_symbol.symnum[local] = 0;
171 return;
172 }
173
174 if (dbgref_symbol.altname[local]) {
175 if (dbgref_symbol.altname[global] == NULL) {
176 dbgref_symbol.altname[global] = dbgref_symbol.altname[local];
177 } else {
178 mod_altptr symptr = dbgref_symbol.altname[global];
179 while (symptr->next) {
180 symptr = symptr->next;
181 }
182 symptr->next = dbgref_symbol.altname[local];
183 }
184 dbgref_symbol.symnum[local] = -2;
185 dbgref_symbol.altname[local] = NULL;
186 } else {
187 const char *localname = SYMNAME(local);
188 mod_altptr symptr = dbgref_symbol.altname[global];
189 if (!symptr) {
190 /* Don't do anything if name is not changed */
191 if (strcmp(SYMNAME(global), localname) == 0) {
192 dbgref_symbol.symnum[local] = -2;
193 return;
194 }
195 }
196 /* Check if localname is already in altname list */
197 while (symptr) {
198 if (strcmp(SYMNAME(symptr->sptr), localname) == 0)
199 break;
200 symptr = symptr->next;
201 }
202 if (!symptr) {
203 symptr = (mod_altptr)malloc(sizeof(module_altname));
204 symptr->sptr = local;
205 symptr->next = dbgref_symbol.altname[global];
206 dbgref_symbol.altname[global] = symptr;
207 }
208 dbgref_symbol.symnum[local] = -2;
209 }
210}

255/** \brief Process a USE ONLY statement, optionally renaming 'global'
256 * as 'local'. The module is specified in 'module_id'.
257 * \return The updated \a global symbol.
258 *
259 * The USE statement can be any of these forms:
260 * USE module, ONLY: global
261 * USE module, ONLY: local => global
262 * USE module, ONLY: OPERATOR(.xx.)
263 * USE module, ONLY: ASSIGNMENT(=)
264 * is_operator is set for the last two.
265 */
266SPTR
267add_use_rename(SPTR local, SPTR global, LOGICAL is_operator)
268{
269 RENAME *pr;
270
271 assert(module_id != NO_MODULE, "module_id must be set", 0, ERR_Fatal);
272 assert(global > NOSYM, "global must be set", global, ERR_Fatal);
273 pr = (RENAME *)getitem(USE_AREA, sizeof(RENAME));
274 pr->complete = 0;
275 pr->is_operator = is_operator;
276 pr->next = usedb.base[module_id].rename;
277 usedb.base[module_id].rename = pr;
278
279 if (local && local == global)
280 /* treat "use m, only: global => global" as "use m, only: global"*/
281 local = SPTR_NULL;
282 pr->local = local;
283 pr->global = global;
284 pr->lineno = gbl.lineno;
285
286 /* Add rename 'use module, abc=>b' */
287 if (flg.debug && local && strcmp(SYMNAME(local), SYMNAME(global)) != 0)
288 set_modusename(local, global);
289
290 return global;
291}

293/* Look for other generic or operator symbols that should be added to
294 * the 'only' list.
295 */
296static int
297add_only(int listitem, int save_sem_scope_level)
298{
299 SCOPESTACK *scope;
300 int sptr = SYMI_SPTR(listitem);
301 int stype = STYPEG(sptr);
302 int newglobal, nextnew;
303 for (newglobal = HASHLKG(sptr); newglobal; newglobal = nextnew) {
304 nextnew = HASHLKG(newglobal);
305 if (HIDDENG(newglobal))
306 continue;
307 if (NMPTRG(newglobal) != NMPTRG(sptr))
308 continue;
309 switch (STYPEG(newglobal)) {
310 case ST_ISOC:
311 case ST_CRAY:
312 /* predefined symbol, but not active in this routine */
313 continue;
314 case ST_MEMBER:
315 /* can't rename a member name */
316 continue;
317 default:;
318 }
319 scope = next_scope_sptr(curr_scope(), SCOPEG(newglobal));
320 /* found this in anything just USEd? */
321 if (get_scope_level(scope) >= save_sem_scope_level) {
322 /* check on 'except' list and private module variable */
323 if (!is_except_in_scope(scope, newglobal) && !PRIVATEG(newglobal)) {
324 /* look for generic with same name */
325 int ng = newglobal;
326 while ((STYPEG(ng) == ST_ALIAS || STYPEG(ng) == ST_MODPROC) &&
327 SYMLKG(ng) && NMPTRG(SYMLKG(ng)) == NMPTRG(newglobal)) {
328 ng = SYMLKG(ng);
329 }
330 if (STYPEG(ng) == ST_PROC && GSAMEG(ng) &&
331 SCOPEG(GSAMEG(ng)) == SCOPEG(newglobal)) {
332 /* generic with same name as specific, use the generic */
333 newglobal = GSAMEG(ng);
334 }
335 if (STYPEG(newglobal) == ST_MODPROC && SYMLKG(newglobal)) {
336 newglobal = SYMLKG(newglobal);
337 }
338 if (STYPEG(newglobal) == stype) {
339 listitem = add_symitem(newglobal, listitem);
340 }
341 }
342 }
343 }
344 return listitem;
345}

347/* We're at the beginning of the statement after a sequence of USE statements.
348 * Apply the use statements seen.
349 * Clean up after processing the sequence of USE statements.
350 */
351void
352apply_use_stmts(void)
353{
354 int save_lineno;
355 MODULE_ID m_id;
356 SPTR ancestor_mod;
357
358 ancestor_mod = NOSYM;
359 module_id = NO_MODULE;
360 if (ANCESTORG(gbl.currmod))
361 ancestor_mod = ANCESTORG(gbl.currmod);
362
363 /*
364 * A user error could have occurred which created a situation where
365 * sem.pgphase is still PHASE_USE (USE statements have appeared) and the
366 * use table is empty.
367 */
368 if (usedb.base == NULL) {
369 usedb.ipasave_avl = 0;
370 return;
371 }
372 save_lineno = gbl.lineno;
373
374 if (!gbl.currmod && gbl.internal <= 1) {
375 for (m_id = FIRST_USER_MODULE; m_id < usedb.avl; m_id++) {
376 if (usedb.base[m_id].scope > sem.scope_level) {
377 remove_from_use_tree(SYMNAME(usedb.base[m_id].module));
378 }
379 }
380 }
381 if (usedb.base[ISO_C_MOD].module) {
382 if (usedb.base[ISO_C_MOD].module == ancestor_mod)
383 error(1211, ERR_Severe, gbl.lineno, SYMNAME(ancestor_mod), CNULL);
384 /* use iso_c_binding */
385 add_predefined_isoc_module();
386 if (sem.interface == 0 && IN_MODULE)
387 exportb.iso_c_library = TRUE;
388 apply_use(ISO_C_MOD);
389 }
390 if (usedb.base[IEEE_ARITH_MOD].module) {
391 if (usedb.base[IEEE_ARITH_MOD].module == ancestor_mod)
392 error(1211, ERR_Severe, gbl.lineno, SYMNAME(ancestor_mod), CNULL);
393 /* use ieee_arithmetic */
394 add_predefined_ieeearith_module();
395 if (sem.interface == 0 && IN_MODULE)
396 exportb.ieee_arith_library = TRUE;
397 apply_use(IEEE_ARITH_MOD);
398 }
399 if (usedb.base[IEEE_FEATURES_MOD].module) {
400 if (usedb.base[IEEE_FEATURES_MOD].module == ancestor_mod)
401 error(1211, ERR_Severe, gbl.lineno, SYMNAME(ancestor_mod), CNULL);
402 /* use ieee_features */
403 sem.ieee_features = TRUE;
404 apply_use(IEEE_FEATURES_MOD);
405 }
406 if (usedb.base[ISO_FORTRAN_ENV].module) {
407 if (usedb.base[ISO_FORTRAN_ENV].module == ancestor_mod)
408 error(1211, ERR_Severe, gbl.lineno, SYMNAME(ancestor_mod), CNULL);
409 /* use iso_fortran_env */
410 add_predefined_iso_fortran_env_module();
411 if (sem.interface == 0 && IN_MODULE)
412 exportb.iso_fortran_env_library = TRUE;
413 apply_use(ISO_FORTRAN_ENV);
414 }
415
416 for (m_id = FIRST_USER_MODULE; m_id < usedb.avl; m_id++) {
417 apply_use(m_id);
418 }
419
420 gbl.lineno = save_lineno;
421 if (usedb.base) {
422 if (XBIT(89, 2) && usedb.avl > FIRST_USER_MODULE) {
423 usedb.ipasave_avl = 0;
424 if (usedb.ipasave_modname == NULL) {
425 usedb.ipasave_sz = usedb.sz;
426 NEW(usedb.ipasave_modname, int, usedb.ipasave_sz);
427 } else {
428 NEED(usedb.ipasave_avl + usedb.avl, usedb.ipasave_modname, int,
429 usedb.ipasave_sz, usedb.ipasave_sz + usedb.avl + 10);
430 }
431 for (m_id = FIRST_USER_MODULE; m_id < usedb.avl; ++m_id) {
432 if (usedb.base[m_id].module) {
433 usedb.ipasave_modname[usedb.ipasave_avl++] = usedb.base[m_id].module;
434 }
435 }
436 }
437 FREE(pd_mod_entries.iso_c);
438 FREE(pd_mod_entries.iso_fortran);
439 FREE(usedb.base);
440 usedb.base = NULL;
441 usedb.sz = 0;
442 usedb.avl = NO_MODULE;
443 }
444
445 freearea(USE_AREA);
446}

448static int
449find_def_in_most_recent_scope(int sptr, LOGICAL is_operator,
450 int save_sem_scope_level)
451{
452 int sptr1;
453 SCOPESTACK *scope;
454
455 for (sptr1 = first_hash(sptr); sptr1; sptr1 = HASHLKG(sptr1)) {
456 int sptr1_prv_recov = PRIVATEG(sptr1); /* Store the value for recovery */
457 if (NMPTRG(sptr1) != NMPTRG(sptr))
458 continue;
459 if (STYPEG(sptr1) == ST_ALIAS && aliased_sym_visible(sptr1)) {
460 PRIVATEP(sptr1, 0);
461 HIDDENP(SYMLKG(sptr1), 0);
462 }
463 if (STYPEG(sptr1) == ST_ALIAS) {
464 if (PRIVATEG(sptr1))
465 continue;
466 } else if (HIDDENG(sptr1)) {
467 continue;
468 }
469
470 switch (STYPEG(sptr1)) {
471 case ST_ISOC:
472 case ST_IEEEARITH:
473 case ST_CRAY:
474 /* predefined symbol, but not active in this routine */
475 continue;
476 case ST_MEMBER:
477 /* can't rename a member name */
478 continue;
479 default:;
480 }
481
482 scope = curr_scope();
483 while ((scope = next_scope_sptr(scope, SCOPEG(sptr1))) != 0) {
484 int ng;
485 int scopelevel = get_scope_level(scope);
486 if (scopelevel < save_sem_scope_level) {
487 break;
488 }
489 /* FS#14884 If sptr1 is ST_ALIAS then the PRIVATE
490 * flag is not valid. Look at the PRIVATE flag of the
491 * symbol the alias points to.
492 */
493 ng = sptr1;
494 while (STYPEG(ng) == ST_ALIAS && SYMLKG(ng) &&
495 NMPTRG(SYMLKG(ng)) == NMPTRG(sptr)) {
496 ng = SYMLKG(ng);
497 }
498 if ((is_operator && STYPEG(ng) != ST_OPERATOR) ||
499 (!is_operator && STYPEG(ng) == ST_OPERATOR))
500 continue;
501 /* is the symbol visible in this scope: i.e. not on except list or
502 in private USE or a private module variable */
503 if (!is_except_in_scope(scope, sptr1) &&
504 !is_private_in_scope(scope, sptr1) &&
505 ((STYPEG(ng) == ST_PROC) ||
506 (STYPEG(ng) == ST_USERGENERIC || !PRIVATEG(ng)))) {
507 return sptr1;
508 }
509 }
510 PRIVATEP(sptr1, sptr1_prv_recov); /* Put back the original value */
511 }
512 return NOSYM;
513}

515static void
516apply_use(MODULE_ID m_id)
517{
518 int save_sem_scope_level, exceptlist, onlylist;
519 RENAME *pr;
520 FILE *use_fd;
521 USED *used = &usedb.base[m_id];
522 char *use_file_name = used->fullname;
523 SPTR sptr;
524
525 if (DBGBIT(0, 0x10000))
526 fprintf(gbl.dbgfil, "Open module file: %s\n", use_file_name);
527 use_fd = fopen(use_file_name, "r");
528 /* -M option: Print list of include files to stdout */
529 /* -MD option: Print list of include files to file <program>.d */
530 if (sem.which_pass == 0 && ((XBIT(123, 2) || XBIT(123, 8)))) {
531 if (gbl.dependfil == NULL) {
532 if ((gbl.dependfil = tmpfile()) == NULL)
533 errfatal(5);
534 } else
535 fprintf(gbl.dependfil, "\\\n ");
536 if (!XBIT(123, 0x40000))
537 fprintf(gbl.dependfil, "%s ", use_file_name);
538 else
539 fprintf(gbl.dependfil, "\"%s\" ", use_file_name);
540 }
541 if (use_fd == NULL) {
542 set_exitcode(19);
543 if (XBIT(0, 0x20000000))
544 erremit(0);
545 error(4, 0, gbl.lineno, "Unable to open MODULE file",
546 SYMNAME(used->module));
547 return;
548 }
549 /* save this so we can tell what new symbols were added below */
550 save_sem_scope_level = sem.scope_level;
551 SCOPEP(used->module, 0);
552 /* Use INCLUDE_PRIVATES, parent privates are visible to inherited submodules.*/
553 used->module = import_module(use_fd, use_file_name, used->module,
554 INCLUDE_PRIVATES, save_sem_scope_level);
555 DINITP(used->module, TRUE);
556 RESTRICTEDP(used->module, used->unrestricted ? 0 : 1);
557 dbg_dump("apply_use", 0x2000);
558
559 if ((seen_contains && sem.mod_cnt) || gbl.internal > 1 || sem.interface) {
560 /*
561 adjust symbol visibility if module has renames and processing a (module
562 or subroutine)
563 contained subroutine or a subroutine interface
564 */
565 adjust_symbol_accessibility(used->module);
566 }
567
568 /* mark syms that are not accessible based on the USE ONLY list */
569 /* step1: set up NOT_IN_USEONLYP flags to 1 for all syms from the used module */
570 if (used->rename) {
571 for (sptr = stb.firstusym; sptr < stb.stg_avail; ++sptr) {
572 if (SCOPEG(sptr) == used->module)
573 NOT_IN_USEONLYP(sptr, 1);
574 }
575 }
576
577 exceptlist = 0;
578 onlylist = 0;
579 for (pr = used->rename; pr != NULL; pr = pr->next) {
580 SPTR newglobal;
581 SPTR ng = 0;
582 SPTR oldglobal = pr->global;
583 SPTR oldlocal = pr->local;
584 char *name = SYMNAME(pr->global);
585
586 if (pr->complete) {
587 /* already found as an iso_c intrinsic */
588 continue;
589 }
590
591 newglobal = find_def_in_most_recent_scope(pr->global, pr->is_operator,
592 save_sem_scope_level);
593
594 /* mark syms that are not accessible based on the USE ONLY list */
595 /* step2: reverse NOT_IN_USEONLYP flag to 0 for syms on the USE ONLY list*/
596 if (newglobal > NOSYM && SCOPEG(newglobal) == used->module)
597 NOT_IN_USEONLYP(newglobal, 0);
598
599 if (newglobal > NOSYM) {
600 /* look for generic with same name */
601 ng = newglobal;
602 while ((STYPEG(ng) == ST_ALIAS || STYPEG(ng) == ST_MODPROC) &&
603 SYMLKG(ng) && NMPTRG(SYMLKG(ng)) == NMPTRG(newglobal)) {
604 ng = SYMLKG(ng);
605 }
606 if (STYPEG(ng) == ST_PROC && GSAMEG(ng) &&
607 SCOPEG(GSAMEG(ng)) == SCOPEG(newglobal)) {
608 /* generic with same name as specific, use the generic */
609 newglobal = GSAMEG(ng);
610 }
611 }
612
613 if (newglobal <= NOSYM || newglobal < stb.firstosym ||
614 STYPEG(newglobal) == ST_UNKNOWN) {
615 if (!sem.which_pass)
616 continue;
617 error(84, 3, pr->lineno, name, "- not public entity of module");
618 IGNOREP(newglobal, 1);
619 continue;
620 }
621
622 if (newglobal != oldglobal && STYPEG(oldglobal) == ST_UNKNOWN) {
623 /* ignore the fake symbol added by the 'use' clause */
624 if (pr->local) {
625 IGNOREP(oldglobal, 1);
626 HIDDENP(oldglobal, 1);
627 } else {
628 pr->local = oldglobal;
629 }
630 }
631 if (STYPEG(newglobal) == ST_MODPROC && SYMLKG(newglobal))
632 newglobal = SYMLKG(newglobal);
633 pr->global = newglobal;
634 gbl.lineno = pr->lineno;
635 if (!pr->local) {
636 pr->local = insert_sym(pr->global);
637 } else if (STYPEG(pr->local) != ST_UNKNOWN) {
638 pr->local = insert_sym(pr->local);
639 }
640 SCOPEP(pr->local, stb.curr_scope);
641 IGNOREP(pr->local, 0);
642 if (!oldlocal)
643 DCLDP(pr->local, 1); /* declared, not renamed */
644 if (STYPEG(ng /*pr->global*/) == ST_OPERATOR) {
645 STYPEP(pr->local, ST_OPERATOR);
646 INKINDP(pr->local, INKINDG(pr->global));
647 PDNUMP(pr->local, PDNUMG(pr->global));
648 copy_specifics(ng, pr->local);
649 } else if (STYPEG(ng /*pr->global*/) == ST_USERGENERIC && !GTYPEG(ng)) {
650 if (NMPTRG(pr->local) == NMPTRG(pr->global)) {
651 STYPEP(pr->local, ST_ALIAS);
652 SYMLKP(pr->local, pr->global);
653 } else {
654 STYPEP(pr->local, ST_USERGENERIC);
655 copy_specifics(ng, pr->local);
656 IGNOREP(SYMLKG(pr->global), 1);
657 }
658 } else {
659 STYPEP(pr->local, ST_ALIAS);
660 if (STYPEG(pr->global) == ST_ALIAS) {
661 SYMLKP(pr->local, SYMLKG(pr->global));
662 IGNOREP(pr->global, 1);
663 } else {
664 SYMLKP(pr->local, pr->global);
665 }
666 }
667 if (used->unrestricted) {
668 /* add the original module symbol to its except list */
669 exceptlist = add_symitem(pr->global, exceptlist);
670 } else {
671 onlylist = add_symitem(pr->global, onlylist);
672 }
673 }
674 if (used->unrestricted) {
675 /* add this stuff to the exception list */
676 int nexte, e;
677 for (e = exceptlist; e; e = nexte) {
678 SPTR sptr = SYMI_SPTR(e);
679 SCOPESTACK *scope = next_scope_sptr(curr_scope(), SCOPEG(sptr));
680 nexte = SYMI_NEXT(e);
681 if (get_scope_level(scope) >= save_sem_scope_level) {
682 SYMI_NEXT(e) = scope->except;
683 scope->except = e;
684 if (STYPEG(sptr) == ST_ALIAS && STYPEG(SYMLKG(sptr)) == ST_PROC) {
685 /* hide original alias for a renamed subprogram */
686 int s;
687 HIDDENP(sptr, 1); /* hide original alias for a renamed subprogram */
688 HIDDENP(SYMLKG(sptr), 1); /* hide subprogram itself,
689 doesn't seem to be necessary */
690 for (s = first_hash(sptr); s; s = HASHLKG(s)) {
691 if (STYPEG(s) == ST_MODPROC && SYMLKG(s) == sptr) {
692 HIDDENP(s, 1); /* hide any associated ST_MODPROC */
693 break;
694 }
695 }
696 }
697 }
698 }
699 update_use_tree_exceptions();
700 } else {
701 /* the SCOPE_USE will be pushed at the scope
702 * level of the old SCOPE_NORMAL */
703 SCOPESTACK *scope = curr_scope();
704 while ((scope = next_scope(scope)) != 0 &&
705 get_scope_level(scope) >= save_sem_scope_level) {
706 int o, nexto;
707 scope->Private = TRUE;
708 for (o = onlylist; o; o = nexto) {
709 nexto = SYMI_NEXT(o);
710 if (SCOPEG(SYMI_SPTR(o)) == scope->sptr) {
711 SYMI_NEXT(o) = scope->only;
712 scope->only = add_only(o, save_sem_scope_level);
713 }
714 }
715 }
716 }
717 fclose(use_fd);
718}

864{

866 char *fullname;

868
869 if (STYPEG(use) != ST_MODULE && STYPEG(use) != ST_UNKNOWN &&
870 SCG(use) != SC_NONE) {
871 /* a variable of this name had been declared, perhaps in an enclosing
872 * subprogram */
873 SPTR sptr;
874 NEWSYM(sptr);
875 NMPTRP(sptr, NMPTRG(use));
876 SYMLKP(sptr, NOSYM);
877 use = sptr;
878 }
879 name = SYMNAME(use);
880
881 for (module_id = FIRST_MODULE; module_id < usedb.avl; module_id++)
882 if (strcmp(SYMNAME(usedb.base[module_id].module), name) == 0)
883 return;

885#define MAX_FNAME_LEN 2050
886
887 fullname = getitem(8, MAX_FNAME_LEN + 1);
888 modu_file_name = getitem(8, strlen(name) + strlen(MOD_SUFFIX) + 1);
889 strcpy(modu_file_name, name);
890 convert_2dollar_signs_to_hyphen(modu_file_name);
891 strcat(modu_file_name, MOD_SUFFIX);

900 /* if module has the same name as some predefined thing */
901 use = insert_sym(use);
902 }
903 if (strcmp(name, "iso_c_binding") == 0) {
904 module_id = ISO_C_MOD;
905 } else if (strcmp(name, "ieee_arithmetic") == 0) {
906 module_id = IEEE_ARITH_MOD;
907 } else if (strcmp(name, "ieee_arithmetic_la") == 0) {
908 module_id = IEEE_ARITH_MOD;
909 } else if (strcmp(name, "ieee_features") == 0) {
910 module_id = IEEE_FEATURES_MOD;
911 } else if (strcmp(name, "iso_fortran_env") == 0) {
912 module_id = ISO_FORTRAN_ENV;

953}

1469static void
1470check_sc(int sptr)
1471{
1472 ITEMX *px;
1473 int dty;
1474 int ty, tysize;
1475 int acc; /* access type: 0 = PUBLIC, 1 = PRIVATE */
1476 int chr; /* 0 => non-character; 1 => character */
1477 int islong; /* 0 => not long; 1 => long */
1478 int initd; /* 0 => not initd; 1 => initd */
1479 int idx, dev, con, link, cpyin;
1480
1481 if (IGNOREG(sptr))
1482 return;
1483 switch (SCG(sptr)) {
1484 case SC_BASED:
1485 case SC_DUMMY:
1486 dty = DTYG(DTYPEG(sptr));
1487 if (XBIT(58, 0x10000) ||
1488 (dty != TY_DERIVED && dty != TY_CHAR && dty != TY_NCHAR)) {
1489 if (POINTERG(sptr) && !F90POINTERG(sptr) && MIDNUMG(sptr) &&
1490 SCG(MIDNUMG(sptr)) != SC_CMBLK) {
1491 /* process pointer variables later; a pointer variable's
1492 * associated variables need to placed in its own common
1493 * block. Can't process here since they would be added
1494 * to the module's common block.
1495 */
1496 px = (ITEMX *)getitem(0, sizeof(ITEMX));
1497 px->val = sptr;
1498 px->next = pointer_list;
1499 pointer_list = px;
1500 /*
1501 * Give the pointer attribute precedence over module
1502 * allocatable.
1503 */
1504 MDALLOCP(sptr, 0);
1505 }
1506 }
1507 if (ALLOCATTRG(sptr)) {
1508 /* process module allocatable arrays later; a variable's
1509 * associated variables need to placed in its own common
1510 * block. Can't process here since they would be added
1511 * to the module's common block.
1512 */
1513 px = (ITEMX *)getitem(0, sizeof(ITEMX));
1514 px->val = sptr;
1515 px->next = mdalloc_list;
1516 mdalloc_list = px;
1517 break;
1518 }
1519 FLANG_FALLTHROUGH;
1520 case SC_CMBLK:
1521 MDALLOCP(sptr, 0);
1522 break;
1523 case SC_NONE:
1524 /* see if we should handle these pointer vars or pass them through */
1525 dty = DTYG(DTYPEG(sptr));
1526 if (XBIT(58, 0x10000) ||
1527 (dty != TY_DERIVED && dty != TY_CHAR && dty != TY_NCHAR)) {
1528 if (POINTERG(sptr) && !F90POINTERG(sptr)) {
1529 /* process pointer variables later; a pointer variable's
1530 * associated variables need to placed in its own common
1531 * block. Can't process here since they would be added
1532 * to the module's common block.
1533 */
1534 px = (ITEMX *)getitem(0, sizeof(ITEMX));
1535 px->val = sptr;
1536 px->next = pointer_list;
1537 pointer_list = px;
1538 /*
1539 * Give the pointer attribute precedence over module
1540 * allocatable.
1541 */
1542 MDALLOCP(sptr, 0);
1543 break;
1544 }
1545 if (ALLOCG(sptr) && !F90POINTERG(sptr)) {
1546 /* process module allocatable arrays later; a variable's
1547 * associated variables need to placed in its own common
1548 * block. Can't process here since they would be added
1549 * to the module's common block.
1550 */
1551 px = (ITEMX *)getitem(0, sizeof(ITEMX));
1552 px->val = sptr;
1553 px->next = mdalloc_list;
1554 mdalloc_list = px;
1555 break;
1556 }
1557 }
1558 FLANG_FALLTHROUGH;
1559 default:
1560#ifdef DEVICEG
1561 propagate_device_flags(sptr);
1562#endif
1563 if (EQVG(sptr)) {
1564 /* don't add to module common, its equivalenced var will be */
1565 break;
1566 }
1567 dev = 0;
1568 cpyin = 0;
1569 link = 0;
1570#ifdef DEVICEG
1571 if (DEVICEG(sptr) || MANAGEDG(sptr) || ACCCREATEG(sptr) ||
1572 ACCCOPYING(sptr) || ACCRESIDENTG(sptr))
1573 dev = 1;
1574 if (ACCCOPYING(sptr))
1575 cpyin = 1;
1576 if (ACCLINKG(sptr)) {
1577 dev = 1;
1578 link = 1;
1579 }
1580 con = CONSTANTG(sptr);
1581#endif
1582 if (XBIT(57, 0x800000) && !dev && !con) {
1583 /* don't set this for device or constant commons? */
1584 if (DTY(DTYPEG(sptr)) == TY_ARRAY && !DESCARRAYG(sptr)) {
1585#ifdef QALNP
1586 QALNP(sptr, 1); /* quad-word align */
1587#endif
1588#ifdef PDALNP
1589 PDALNP(sptr, 4); /* quad-word align */
1590#endif
1591 }
1592 }
1593 ty = basedtype(sptr);
1594 if (ty == 0)
1595 return; /* don't add to module common */
1596 if (CFUNCG(sptr)) {
1597 SCP(sptr, SC_EXTERN);
1598 return; /* C visable module variable not
1599 in common block */
1600 }
1601 tysize = size_of(ty);
1602 acc = PRIVATEG(sptr);
1603 chr = (DTY(ty) == TY_CHAR || DTY(ty) == TY_NCHAR);
1604 islong = chr ? 0 : size_of(ty) == 8;
1605 initd = DINITG(sptr);
1606 idx = MOD_CMN_IDX(acc, chr, islong, initd, THREADG(sptr), dev, con, cpyin,
1607 link);
1608 if (mod_cmn[idx] == 0)
1609 mod_cmn[idx] =
1610 make_module_common(idx, acc, THREADG(sptr), dev, con, cpyin, link);
1611
1612 if (SOCPTRG(sptr)) {
1613 /* may have to add 'padding' to the front of this symbol
1614 * if its offset is nonzero; may have to add 'padding' to
1615 * the end of this symbol if its overlap list has any
1616 * variables that extend over the end.
1617 * NOTE that the ADDRESS fields of the equivalenced variables
1618 * are still offsets relative to this symbol and the sptr's
1619 * relative offset from the beginning of its module common
1620 * has not been assigned.
1621 */
1622 ISZ_T offset = ADDRESSG(sptr);
1623 if (offset > 0) {
1624 ISZ_T arraysize = (offset + tysize - 1) / tysize;
1625 int p, pad;
1626 pad = add_padding(sptr, ty, arraysize, idx);
1627 for (p = SOCPTRG(sptr); p; p = SOC_NEXT(p)) {
1628 int overlap = SOC_SPTR(p);
1629 ISZ_T overlap_offset = ADDRESSG(overlap);
1630 if (overlap_offset < offset) {
1631 NEED(soc.avail + 2, soc.base, SOC_ITEM, soc.size, soc.size + 1000);
1632 SOC_SPTR(soc.avail) = pad;
1633 SOC_NEXT(soc.avail) = SOCPTRG(overlap);
1634 SOCPTRP(overlap, soc.avail);
1635 ++soc.avail;
1636 SOC_SPTR(soc.avail) = overlap;
1637 SOC_NEXT(soc.avail) = SOCPTRG(pad);
1638 SOCPTRP(pad, soc.avail);
1639 ++soc.avail;
1640 }
1641 }
1642 }
1643 }
1644 add_to_common(idx, sptr, 0);
1645 if (SOCPTRG(sptr)) {
1646 /* may have to add padding after the variable to account
1647 * for the extra space taken up by the other variables
1648 * equivalenced to this one.
1649 * NOTE that the ADDRESS fields of the equivalenced variables
1650 * are still offsets relative to this symbol and the sptr's
1651 * relative offset from the beginning of its module common
1652 * has been assigned.
1653 */
1654 ISZ_T offset = ADDRESSG(sptr);
1655 ISZ_T sptrsize = size_of(DTYPEG(sptr));
1656 ISZ_T padsize = 0;
1657 int p;
1658 for (p = SOCPTRG(sptr); p; p = SOC_NEXT(p)) {
1659 int overlap = SOC_SPTR(p);
1660 ISZ_T overlap_offset = ADDRESSG(overlap) + offset;
1661 ISZ_T overlap_size = size_of(DTYPEG(overlap));
1662 if (overlap_offset + overlap_size > offset + sptrsize + padsize) {
1663 padsize = overlap_offset + overlap_size - offset - sptrsize;
1664 }
1665 /* add to common block also */
1666 ADDRESSP(overlap, overlap_offset);
1667 add_to_common(idx, overlap, 0);
1668 }
1669 if (padsize > 0) {
1670 int p, pad;
1671 padsize = (padsize + tysize - 1) / tysize;
1672 pad = add_padding(sptr, ty, padsize, idx);
1673 for (p = SOCPTRG(sptr); p; p = SOC_NEXT(p)) {
1674 int overlap = SOC_SPTR(p);
1675 ISZ_T overlap_offset = ADDRESSG(overlap);
1676 ISZ_T overlap_size = size_of(DTYPEG(overlap));
1677 if (overlap_offset + overlap_size > offset + sptrsize) {
1678 int sp;
1679 /* it may already have been added in add_padding */
1680 for (sp = SOCPTRG(overlap); sp; sp = SOC_NEXT(sp)) {
1681 if (SOC_SPTR(sp) == pad)
1682 break;
1683 }
1684 if (sp == 0) {
1685 NEED(soc.avail + 2, soc.base, SOC_ITEM, soc.size,
1686 soc.size + 1000);
1687 SOC_SPTR(soc.avail) = pad;
1688 SOC_NEXT(soc.avail) = SOCPTRG(overlap);
1689 SOCPTRP(overlap, soc.avail);
1690 ++soc.avail;
1691 SOC_SPTR(soc.avail) = overlap;
1692 SOC_NEXT(soc.avail) = SOCPTRG(pad);
1693 SOCPTRP(pad, soc.avail);
1694 ++soc.avail;
1695 }
1696 }
1697 }
1698 }
1699 }
1700 break;
1701 }
1702} /* check_sc */

1717static void
1718fix_module_common(void)
1719{
1720 int sptr, symavl;
1721 int i;
1722 ITEMX *px;
1723 LOGICAL err;
1724 int evp, firstevp;
1725
1726 if (gbl.maxsev >= 3) {
1727 gbl.currsub = modu_sym; /* trick semfin & summary */
1728 semfin(); /* to cleanup, free space, etc. */
1729 return;
1730 }
1731
1732 BZERO(mod_cmn, char, sizeof(mod_cmn));
1733 BZERO(mod_cmn_naln, char, sizeof(mod_cmn_naln));
1734
1735 for (sptr = stb.firstusym; sptr < stb.stg_avail; sptr++) {
1736 if (IGNOREG(sptr))
1737 continue;
1738 switch (STYPEG(sptr)) {
1739 case ST_PARAM:
1740 if (DTY(DTYPEG(sptr)) == TY_ARRAY) {
1741 /* emit the data inits for the named array constant */
1742 init_named_array_constant(sptr, modu_sym);
1743 }
1744 break;
1745 default:
1746 break;
1747 }
1748 }
1749
1750 gbl.rutype = RU_SUBR; /* trick semfin */
1751 gbl.currsub = modu_sym; /* trick semfin */
1752
1753 semfin();
1754
1755 mdalloc_list = pointer_list = NULL;
1756 symavl = stb.stg_avail;
1757 for (sptr = stb.firstusym; sptr < symavl; sptr++) {
1758 if (IGNOREG(sptr))
1759 continue;
1760 if (SCOPEG(sptr) != stb.curr_scope)
1761 continue;
1762 if (ENCLFUNCG(sptr) == 0)
1763 ENCLFUNCP(sptr, modu_sym);
1764 if (ENCLFUNCG(sptr) != modu_sym)
1765 continue;
1766 if (NOMDCOMG(sptr))
1767 continue;
1768 switch (STYPEG(sptr)) {
1769 case ST_ARRAY:
1770 case ST_VAR:
1771 case ST_STRUCT:
1772 case ST_UNION:
1773 err = 0;
1774 if (SCG(sptr) != SC_DUMMY) {
1775 int dtype, dty;
1776 dtype = DTYPEG(sptr);
1777 if (DTY(dtype) == TY_ARRAY && ADJARRG(sptr)) {
1778 error(310, 3, gbl.lineno,
1779 "Automatic arrays are not allowed in a MODULE -",
1780 SYMNAME(sptr));
1781 err = 1;
1782 }
1783 dty = DTYG(dtype);
1784 if ((dty == TY_CHAR || dty == TY_NCHAR) && ADJLENG(sptr)) {
1785 error(310, 3, gbl.lineno,
1786 "Adjustable-length character variables are "
1787 "not allowed in a MODULE -",
1788 SYMNAME(sptr));
1789 err = 1;
1790 }
1791 }
1792 if (!err)
1793 check_sc(sptr);
1794 break;
1795 case ST_IDENT:
1796 STYPEP(sptr, ST_VAR);
1797 err = 0;
1798 if (SCG(sptr) != SC_DUMMY) {
1799 int dtype, dty;
1800 dtype = DTYPEG(sptr);
1801 dty = DTYG(dtype);
1802 if ((dty == TY_CHAR || dty == TY_NCHAR) && ADJLENG(sptr)) {
1803 error(310, 3, gbl.lineno,
1804 "Adjustable-length character variables are "
1805 "not allowed in a MODULE -",
1806 SYMNAME(sptr));
1807 err = 1;
1808 }
1809 }
1810 if (!err)
1811 check_sc(sptr);
1812 break;
1813 case ST_UNKNOWN: /* ignore */
1814 break;
1815 case ST_NML:
1816 if (mod_cmn[NML_MOD] == 0)
1817 mod_cmn[NML_MOD] = make_module_common(NML_MOD, 0, 0, 0, 0, 0, 0);
1818 add_to_common(NML_MOD, ADDRESSG(sptr), 0);
1819 /* mark as referenced, so it gets declared everywhere */
1820 REFP(sptr, 1);
1821 break;
1822 default:
1823 break;
1824 }
1825 }
1826 /* make sure all overlapped variables are listed in the module common */
1827 for (i = 0; i < N_MOD_CMN; ++i) {
1828 if (mod_cmn[i] <= 0)
1829 continue;
1830 for (sptr = CMEMFG(mod_cmn[i]); sptr != NOSYM; sptr = SYMLKG(sptr)) {
1831 int p;
1832 for (p = SOCPTRG(sptr); p; p = SOC_NEXT(p)) {
1833 int s = SOC_SPTR(p);
1834 if (SCG(s) != SC_CMBLK)
1835 add_to_common(i, s, 0);
1836 }
1837 }
1838 }
1839 /* Get correct addresses in the module common blocks */
1840 /* Store in the SCOPE field a symbol pointer to the symbol to which
1841 * this symbol is equivalenced. If SCOPEG(sptr)!=module then
1842 * SCOPEG(sptr) is the symbol to which sptr is equivalenced.
1843 * Also, ADDRESSG(sptr) is the byte offset of sptr relative to
1844 * the address of SCOPEG(sptr). */
1845 firstevp = 0;
1846 for (evp = sem.eqvlist; evp; evp = EQV(evp).next) {
1847 if (!HCCSYMG(CMBLKG(EQV(evp).sptr))) {
1848 /* skip user common blocks */
1849 continue;
1850 }
1851 if (EQV(evp).is_first < 0) {
1852 firstevp = 0;
1853 } else if (EQV(evp).is_first > 0) {
1854 firstevp = evp;
1855 } else if (firstevp != 0) {
1856 /* if EQVG(evp->sptr), set address of evp->sptr relative to
1857 * that of firstevp; otherwise, the other way around */
1858 if (EQVG(EQV(evp).sptr)) {
1859 /* see if we've already done this */
1860 if (SCOPEG(EQV(evp).sptr) == stb.curr_scope) {
1861 SCOPEP(EQV(evp).sptr, EQV(firstevp).sptr);
1862 ADDRESSP(EQV(evp).sptr,
1863 EQV(firstevp).byte_offset - EQV(evp).byte_offset);
1864 }
1865 } else {
1866 if (SCOPEG(EQV(firstevp).sptr) == stb.curr_scope) {
1867 /* EQV(evp).sptr already has an address; set address of
1868 * firstevp relative to that of evp->sptr */
1869 ADDRESSP(EQV(firstevp).sptr, ADDRESSG(EQV(evp).sptr) +
1870 EQV(evp).byte_offset -
1871 EQV(firstevp).byte_offset);
1872 }
1873 }
1874 }
1875 }
1876 firstevp = 0;
1877 for (evp = sem.eqvlist; evp; evp = EQV(evp).next) {
1878 if (!HCCSYMG(CMBLKG(EQV(evp).sptr))) {
1879 /* skip user common blocks */
1880 continue;
1881 }
1882 if (EQV(evp).is_first < 0) {
1883 firstevp = 0;
1884 } else if (EQV(evp).is_first > 0) {
1885 firstevp = evp;
1886 } else if (firstevp != 0) {
1887 if (EQVG(EQV(evp).sptr) && SCOPEG(EQV(evp).sptr) != stb.curr_scope) {
1888 ISZ_T addr = get_address(SCOPEG(EQV(evp).sptr));
1889 addr += ADDRESSG(EQV(evp).sptr);
1890 ADDRESSP(EQV(evp).sptr, addr);
1891 SCOPEP(EQV(evp).sptr, stb.curr_scope);
1892 }
1893 }
1894 }
1895 for (px = mdalloc_list; px != NULL; px = px->next)
1896 /* for each allocatable variable, create its run-time descriptor
1897 * "module-name$array-name$al"
1898 */
1899 make_rte_descriptor(px->val, "al");
1900
1901 for (px = pointer_list; px != NULL; px = px->next)
1902 /* for each pointer variable, create its run-time descriptor
