71void ENTFTN(CSENDCHARA, csendchara)(__INT_T *cpu, DCHAR(buf), __INT_T *cnt,
72 __INT_T *str DCLEN64(buf))
73{
74 char *adr;
75 __CLEN_T n, len, skip;

79 }
