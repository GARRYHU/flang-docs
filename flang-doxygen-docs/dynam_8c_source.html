19/* reallocate and copy. ad = new descriptor, dd = old descriptor, pd
20 = descriptor associated with pointer and offset variables, i.e. the
21 original descriptor location. */
22static void
23I8(recopy)(F90_Desc *ad, F90_Desc *dd, F90_Desc *pd)
24{
25 char *ab, *af, *db, *df, *base, **ptr;
26 __POINT_T *off;
27 chdr *ch;
28
29 if (F90_FLAGS_G(ad) & __TEMPLATE)
30 return;
31
32 ptr = ((char **)pd) - 2; /* array pointer variable */
33 off = (__POINT_T *)(ptr + 1); /* array offset variable */
34
35 db = *ptr; /* array address */
36
37 if (!ISPRESENT(db))
38 return;
39
40 /* allocate the new array */
41
42 base = db - (*off - 1) * F90_LEN_G(ad);
43 ab = I8(__fort_allocate)(F90_LSIZE_G(ad), F90_KIND_G(ad), F90_LEN_G(ad), base,
44 ptr, off);
45
46 /* copy the old into the new */
47
48 af = ab + DIST_SCOFF_G(ad) * F90_LEN_G(ad);
49 df = db + DIST_SCOFF_G(dd) * F90_LEN_G(dd);
50 ch = I8(__fort_copy)(af, df, ad, dd, NULL);
51 __fort_doit(ch);
52 __fort_frechn(ch);
53
54 /* free the old array */
55
56 if (~F90_FLAGS_G(dd) & __NOT_COPIED)
57 I8(__fort_deallocate)(db);
58
59 F90_FLAGS_P(ad, F90_FLAGS_G(ad) & ~__NOT_COPIED);
60#if defined(DEBUG)
61 if (__fort_test & DEBUG_RDST) {
62 printf("%d recopy ab=%x base=%x offset=%x\n", GET_DIST_LCPU, ab, base,
63 *off);
64 }
65#endif
66}

68/* \brief realign the alignee with the align-target template 'td'
69 *
70 *<pre>
71 * varargs are:
72 * [ __INT_T *collapse,
73 * { __INT_T *taxis, __INT_T *tstride, __INT_T *toffset, }*
74 * __INT_T *single, { __INT_T *coordinate, }* ]
75 *</pre>
76 */
77void
78ENTFTN(REALIGN, realign)(F90_Desc *ad, __INT_T *p_rank, __INT_T *p_flags,
79 F90_Desc *td, __INT_T *p_conform, ...)
80{
81 va_list va;
82 DECL_HDR_VARS(dd);
83 DECL_HDR_PTRS(ud);
84 DECL_HDR_PTRS(prev);
85 DECL_HDR_PTRS(next);
86 DECL_DIM_PTRS(add);
87 DECL_DIM_PTRS(ddd);
88 DECL_DIM_PTRS(tdd);
89 proc *ap, *tp;
90 __INT_T flags, collapse, m, single = 0;
91 __INT_T taxis[MAXDIMS], tstride[MAXDIMS], toffset[MAXDIMS];
92 __INT_T coordinate[MAXDIMS];
93 __INT_T ak, i, realign, rank, tk, tm, px, tx;
94
95 rank = *p_rank;
96 flags = *p_flags;
97
98#if defined(DEBUG)
99 if (__fort_test & DEBUG_RDST) {
100 printf("%d REALIGN alignee=%x new-align-target=%x\n", GET_DIST_LCPU, ad,
101 td);
102 __fort_show_flags(flags);
103 printf("\n");
104 }
105 if (ad == NULL || F90_TAG_G(ad) != __DESC)
106 __fort_abort("REALIGN: invalid alignee descriptor");
107 if (td == NULL || F90_TAG_G(td) != __DESC)
108 __fort_abort("REALIGN: invalid new-align-target descriptor");
109 if (~F90_FLAGS_G(ad) & __DYNAMIC)
110 __fort_abort("REALIGN: alignee is not DYNAMIC");
111 if (F90_RANK_G(ad) != rank)
112 __fort_abort("REALIGN: alignee rank differs");
113 if (flags &
114 (__DIST_TARGET_MASK << __DIST_TARGET_SHIFT |
115 __DIST_FORMAT_MASK << __DIST_FORMAT_SHIFT | __INHERIT | __SEQUENCE))
116 __fort_abort("REALIGN: distribution, inherit, or sequence disallowed");
117#endif
118
119 va_start(va, p_conform);
120
121 if (flags & __IDENTITY_MAP) {
122 collapse = 0;
123 for (i = 1; i <= rank; ++i) {
124 taxis[i - 1] = i;
125 tstride[i - 1] = 1;
126 toffset[i - 1] = 0;
127 }
128 } else {
129 collapse = *va_arg(va, __INT_T *);
130
131 for (i = 0; i < rank; ++i) {
132 if (collapse >> i & 1) {
133 taxis[i] = 0;
134 tstride[i] = 1;
135 toffset[i] = 0;
136 } else {
137 taxis[i] = *va_arg(va, __INT_T *);
138 tstride[i] = *va_arg(va, __INT_T *);
139 toffset[i] = *va_arg(va, __INT_T *);
140 }
141 }
142 single = *va_arg(va, __INT_T *);
143 if (single >> F90_RANK_G(td))
144 __fort_abort("REALIGN: invalid single alignment axis");
145 for (i = 0; i < F90_RANK_G(td); ++i) {
146 if (single >> i & 1)
147 coordinate[i] = *va_arg(va, __INT_T *);
148 else
149 coordinate[i] = 0;
150 }
151 }
152 va_end(va);
153
154 ap = DIST_DIST_TARGET_G(ad);
155 tp = DIST_DIST_TARGET_G(td);
156
157 realign = (ap->base != tp->base || ap->size != tp->size);
158
159 for (i = 0; !realign && i < rank; ++i) {
160 SET_DIM_PTRS(add, ad, i);
161
162 /* realignment required if different processor axes are
163 targeted or if the processor shapes or strides differ */
164
165 tx = taxis[i];
166 if (tx > 0) {
167 SET_DIM_PTRS(tdd, td, tx - 1);
168 px = DIST_DPTR_PAXIS_G(tdd);
169 } else
170 px = 0;
171
172 realign = (px != DIST_DPTR_PAXIS_G(add));
173 if (realign)
174 break;
175
176 if (px == 0)
177 continue; /* collapsed dimension */
178
179 realign = (DIST_DPTR_PSHAPE_G(add) != DIST_DPTR_PSHAPE_G(tdd) ||
180 DIST_DPTR_PSTRIDE_G(add) != DIST_DPTR_PSTRIDE_G(tdd));
181 if (realign)
182 break;
183
184 /* realignment required if the template mappings aren't
185 equivalent... */
186
187 /* offset in ultimately-aligned template of actual array */
188
189 ak = DIST_DPTR_TSTRIDE_G(add) * F90_DPTR_LBOUND_G(add) +
190 DIST_DPTR_TOFFSET_G(add) - DIST_DPTR_TLB_G(add);
191
192 /* mapping onto ultimate align-target */
193
194 tm = DIST_DPTR_TSTRIDE_G(tdd) * tstride[i];
195 tk = DIST_DPTR_TSTRIDE_G(tdd) * toffset[i] + DIST_DPTR_TOFFSET_G(tdd);
196
197 /* offset in ultimately-aligned template of align-target */
198
199 tk = tm * F90_DPTR_LBOUND_G(tdd) + tk - DIST_DPTR_TLB_G(tdd);
200
201#if defined(DEBUG)
202 if (__fort_test & DEBUG_RDST) {
203 printf("%d target tm=%d tk=%d tb=%d tmab=%d tkab=%d\n", GET_DIST_LCPU,
204 tm, tk, DIST_DPTR_BLOCK_G(tdd), tm * DIST_DPTR_BLOCK_G(add),
205 tk * DIST_DPTR_BLOCK_G(add));
206 printf("%d actual am=%d ak=%d ab=%d amtb=%d aktb=%d\n", GET_DIST_LCPU,
207 DIST_DPTR_TSTRIDE_G(add), ak, DIST_DPTR_BLOCK_G(add),
208 DIST_DPTR_TSTRIDE_G(add) * DIST_DPTR_BLOCK_G(tdd),
209 ak * DIST_DPTR_BLOCK_G(tdd));
210 }
211#endif
212 realign = (DIST_DPTR_BLOCK_G(tdd) * DIST_DPTR_TSTRIDE_G(add) !=
213 DIST_DPTR_BLOCK_G(add) * tm ||
214 DIST_DPTR_BLOCK_G(tdd) * ak != DIST_DPTR_BLOCK_G(add) * tk);
215 }
216
217 ud = DIST_ALIGN_TARGET_G(ad);
218#if defined(DEBUG)
219 if (ud == NULL || F90_TAG_G(ud) != __DESC)
220 __fort_abort("REALIGN: invalid old align-target descriptor");
221 if (DIST_ALIGN_TARGET_G(ud) != ud)
222 __fort_abort("REALIGN: old align-target is not ultimate align-target");
223#endif
224
225 if (F90_FLAGS_G(ud) & __DYNAMIC) {
226 if (ud == ad) {
227
228 /* array is distributee */
229
230 if (DIST_NEXT_ALIGNEE_G(ad) != NULL)
231 __fort_abort("REALIGN: array is dynamic align-target");
232 } else {
233
234 /* unlink from old ultimate align-target's alignees list */
235
236 prev = ud;
237 next = DIST_NEXT_ALIGNEE_G(ud);
238 while (next != NULL && next != ad) {
239 prev = next;
240 next = DIST_NEXT_ALIGNEE_G(prev);
241 }
242 if (next != ad)
243 __fort_abort("REALIGN: alignee not in old align-target's list");
244
245 DIST_NEXT_ALIGNEE_P(prev, DIST_NEXT_ALIGNEE_G(ad));
246#if defined(DEBUG)
247 if (__fort_test & DEBUG_RDST) {
248 printf("%d unlinked ud=%x prev=%x next=%lx\n", GET_DIST_LCPU, ud, prev,
249 DIST_NEXT_ALIGNEE_G(prev));
250 }
251#endif
252 }
253 }
254
255 if (realign) {
256
257 /* make a copy of the old descriptor */
258
259 I8(__fort_copy_descriptor)(dd, ad);
260
261 /* update the descriptor in place. init_descriptor links the
262 descriptor to the new ultimate align-target. */
263
264 ud = DIST_ALIGN_TARGET_G(td);
265
266 __DIST_INIT_DESCRIPTOR(ad, F90_RANK_G(ad), F90_KIND_G(ad), F90_LEN_G(ad),
267 flags, ud);
268 for (i = 1; i <= rank; ++i) {
269 tx = taxis[i - 1];
270 if (tx > 0) {
271 SET_DIM_PTRS(tdd, td, tx - 1);
272 tx = DIST_DPTR_TAXIS_G(tdd);
273 }
274 if (tx > 0) {
275 tm = DIST_DPTR_TSTRIDE_G(tdd) * tstride[i - 1];
276 tk = DIST_DPTR_TSTRIDE_G(tdd) * toffset[i - 1] + DIST_DPTR_TOFFSET_G(tdd);
277 } else {
278 tm = 1;
279 tk = 0;
280 }
281 SET_DIM_PTRS(ddd, dd, i - 1);
282
283 /*
284 * added last arg which passes the gen_block field in...
285 */
286
287 I8(__fort_set_alignment)(ad, i, F90_DPTR_LBOUND_G(ddd),
288 DPTR_UBOUND_G(ddd), tx, tm, tk,
289 (tx>0)?(&DIST_DPTR_GEN_BLOCK_G(tdd)):
290 (&DIST_DPTR_GEN_BLOCK_G(ddd)));
291 }
292 /* NEC 127 / tpr 2597 */
293
294 m = single;
295 for (i = 1; m > 0; ++i, m >>= 1) {
296 if (m & 1)
297 I8(__fort_set_single)(ad, td, i, coordinate[i - 1], __SINGLE);
298 }
299 m = DIST_SINGLE_G(td);
300 for (i = 1; m > 0; ++i, m >>= 1) {
301 if (m & 1)
302 I8(__fort_set_single)(ad, DIST_ALIGN_TARGET_G(td), i,
303 DIST_INFO_G(td, i-1), __SINGLE);
304 }
305
306 for (i = 1; i <= rank; ++i) {
307 SET_DIM_PTRS(ddd, dd, i - 1);
308 if (~F90_FLAGS_G(dd) & __TEMPLATE)
309 __DIST_SET_ALLOCATION(ad, i, DIST_DPTR_NO_G(ddd), DIST_DPTR_PO_G(ddd));
310 }
311
312 I8(__fort_finish_descriptor)(ad);
313
314#if defined(DEBUG)
315 if (__fort_test & DEBUG_RDST) {
316 printf("%d linked ud=%x next=%lx\n", GET_DIST_LCPU, ud,
317 DIST_NEXT_ALIGNEE_G(ud));
318 }
319#endif
320
321 /* reallocate and copy the old into the new */
322
323 I8(recopy)(ad, dd, ad);
324 } else {
325
326 /* link to new align-target. descriptor does not need to
327 change and array does not need to be copied. */
328
329 DIST_ALIGN_TARGET_P(ad, DIST_ALIGN_TARGET_G(td));
330 DIST_DIST_TARGET_P(ad, DIST_DIST_TARGET_G(td));
331 DIST_NEXT_ALIGNEE_P(ad, DIST_NEXT_ALIGNEE_G(td));
332 DIST_NEXT_ALIGNEE_P(td, ad);
333 }
334}

336/** \brief redistribute the distributee and all objects that are currently
337 * ultimately-aligned with it (within the scope of the calling
338 * subprogram). redistribution does not change alignment
339 * relationships.
340 *
341 *<pre>
342 * varargs are:
343 * [ proc *dist_target, ]
344 * __INT_T *isstar,
345 * { [__INT_T paxis,](__INT_T *dstfmt, |
346 * (__INT_T * gen_block_array, __INT_T extent) ) }*
347 *</pre>
348 */
349void
350ENTFTN(REDISTRIBUTE, redistribute)(F90_Desc *dd, __INT_T *p_rank,
351 __INT_T *p_flags, ...)
352{
353 va_list va;
354 DECL_HDR_PTRS(ad);
355 DECL_HDR_PTRS(ud);
356 DECL_HDR_PTRS(next);
357 DECL_DIM_PTRS(odd);
358 DECL_DIM_PTRS(udd);
359 proc *tp, *up;
360 DECL_HDR_VARS(od);
361 DECL_HDR_VARS(td);
362 __INT_T nmapped, block[MAXDIMS];
363 __INT_T flags, dist_format_spec, dist_target_spec;
364 __INT_T isstar, paxis[MAXDIMS];
365 __INT_T dfmt, ddfmt, tdfmt, i, rank, redistribute = 0, ux;
366
367 __INT_T *gbCopy[MAXDIMS]; /*hold gen_block dims*/
368 __INT_T gbIdx = 0, j;
369
370 for (i = 0; i < MAXDIMS; ++i)
371 gbCopy[i] = 0;
372
373 rank = *p_rank;
374 flags = *p_flags;
375
376 dist_target_spec =
377 (_io_spec)(flags >> __DIST_TARGET_SHIFT & __DIST_TARGET_MASK);
378 dist_format_spec =
379 (_io_spec)(flags >> __DIST_FORMAT_SHIFT & __DIST_FORMAT_MASK);
380
381#if defined(DEBUG)
382 if (dd == NULL || F90_TAG_G(dd) != __DESC)
383 __fort_abort("REDISTRIBUTE: invalid distributee descriptor");
384 if (F90_RANK_G(dd) != rank)
385 __fort_abort("REDISTRIBUTE: distributee has incorrect rank");
386 if (flags & (__ALIGN_TARGET_MASK << __ALIGN_TARGET_SHIFT | __SEQUENCE))
387 __fort_abort("REDISTRIBUTE: invalid flags");
388#endif
389
390 ud = DIST_ALIGN_TARGET_G(dd);
391
392#if defined(DEBUG)
393 if (ud == NULL || F90_TAG_G(ud) != __DESC)
394 __fort_abort("REDISTRIBUTE: invalid ultimate template descriptor");
395 if (DIST_ALIGN_TARGET_G(ud) != ud)
396 __fort_abort("REDISTRIBUTE: template is not ultimate align-target");
397 if (~F90_FLAGS_G(ud) & __DYNAMIC)
398 __fort_abort("REDISTRIBUTE: ultimate template is not DYNAMIC");
399
400 if (__fort_test & DEBUG_RDST) {
401 printf("%d REDISTRIBUTE distributee=%x ultimate template=%x\n",
402 GET_DIST_LCPU, dd, ud);
403 __fort_show_flags(flags);
404 printf("\n");
405 }
406#endif
407 if (F90_RANK_G(ud) != rank)
408 __fort_abort("REDISTRIBUTE: ultimate template has incorrect rank");
409
410 /* get distribution target spec */
411
412 va_start(va, p_flags);
413
414 switch (dist_target_spec) {
415
416 case __PRESCRIPTIVE:
417 tp = va_arg(va, proc *);
418 break;
419
420 case __OMITTED:
421 tp = NULL;
422 break;
423
424 case __DESCRIPTIVE:
425 case __TRANSCRIPTIVE:
426 default:
427 __fort_abort("REDISTRIBUTE: bad dist-target flags");
428 }
429
430 /* get distribution format spec */
431
432 nmapped = 0; /* no. of distributed dimensions */
433 ddfmt = 0;
434
435 switch (dist_format_spec) {
436
437 case __PRESCRIPTIVE:
438 isstar = *va_arg(va, __INT_T *);
439 for (i = 0; i < rank; ++i) {
440 if (isstar >> i & 1) {
441 paxis[i] = 0;
442 block[i] = 0;
443 } else if (((isstar & EXTENSION_BLOCK_MASK) >> (7 + 3 * i)) & 0x01) {
444
445 /*
446 * got a gen_block dimension. The arguments for
447 * ENTFTN(redistribute) are slightly different for
448 * gen_block, so we need to handle this as a special
449 * case.
450 */
451
452 if (flags & __DIST_TARGET_AXIS) {
453 paxis[i] = *va_arg(va, __INT_T *);
454 if (paxis[i] != 0)
455 ++nmapped;
456 } else
457 paxis[i] = ++nmapped;
458
459 gbCopy[gbIdx++] = va_arg(va, __INT_T *);
460 block[i] = *va_arg(va, __INT_T *);
461 ddfmt |= DFMT_GEN_BLOCK << DFMT__WIDTH * i;
462 if (DFMT(ud, i + 1) == DFMT_GEN_BLOCK) {
463
464 /* NEC problem 211 / tpr 2488
465 * redistribute if gen_block array changed.
466 */
467
468 int elem;
469 __INT_T *newgb, *oldgb;
470
471 newgb = DIST_DIM_GEN_BLOCK_G(ud, i);
472 oldgb = gbCopy[gbIdx - 1];
473 for (elem = 0; elem < block[i]; ++elem)
474
475 if (*(oldgb + elem) != *(newgb + elem)) {
476 redistribute = 1;
477 break;
478 }
479 }
480 }
481
482 else {
483 int dstfmt;
484
485 if (flags & __DIST_TARGET_AXIS) {
486 paxis[i] = *va_arg(va, __INT_T *);
487 if (paxis[i] != 0)
488 ++nmapped;
489 } else
490 paxis[i] = ++nmapped;
491
492 dstfmt = *va_arg(va, __INT_T *);
493 if (dstfmt >= 0) {
494
495 block[i] = dstfmt;
496 if (dstfmt == 0)
497 ddfmt |= DFMT_BLOCK << DFMT__WIDTH * i;
498 else
499 ddfmt |= DFMT_BLOCK_K << DFMT__WIDTH * i;
500 } else {
501 block[i] = -dstfmt;
502 if (dstfmt == -1)
503 ddfmt |= DFMT_CYCLIC << DFMT__WIDTH * i;
504 else
505 ddfmt |= DFMT_CYCLIC_K << DFMT__WIDTH * i;
506 }
507 }
508 }
509 break;
510
511 case __OMITTED:
512 for (i = 0; i < rank; ++i) {
513 paxis[i] = 0;
514 block[i] = 0;
515 }
516 break;
517
518 case __DESCRIPTIVE:
519 case __TRANSCRIPTIVE:
520 default:
521 __fort_abort("REDISTRIBUTE: bad dist-format flags");
522 }
523 va_end(va);
524
525 if (tp == NULL)
526 tp = __fort_defaultproc(nmapped);
527 else if (tp->tag != __PROC || tp->rank < nmapped)
528 __fort_abort("REDISTRIBUTE: invalid dist-target");
529
530 /* shuffle dist-formats to match align-target axis permutation */
531
532 tdfmt = 0;
533 dfmt = ddfmt;
534 for (i = 0; i < rank; ++i, dfmt >>= DFMT__WIDTH) {
535 if (dfmt & DFMT__MASK) {
536 ux = DIST_DIM_TAXIS_G(dd, i);
537 if (ux > 0)
538 tdfmt |= (dfmt & DFMT__MASK) << DFMT__WIDTH * (ux - 1);
539 else
540 __fort_abort("REDISTRIBUTE: no align-target axis for mapped dim");
541 }
542 }
543
544 /* check conformance of actual distribution vs. dist-target */
545
546 redistribute |= (tdfmt != DIST_DFMT_G(ud));
547 up = DIST_DIST_TARGET_G(ud);
548 if (!redistribute && up != tp) {
549 redistribute =
550 (up->rank != tp->rank || up->base != tp->base || up->size != tp->size);
551 for (i = 0; !redistribute && i < tp->rank; ++i)
552 redistribute = (up->dim[i].shape != tp->dim[i].shape);
553 }
554 for (i = 0; !redistribute && i < rank; ++i) {
555 ux = DIST_DIM_TAXIS_G(dd, i);
556 SET_DIM_PTRS(udd, ud, ux - 1);
557 redistribute = (DIST_DPTR_PAXIS_G(udd) != paxis[i]);
558 if (redistribute)
559 break;
560 switch (DFMT(ud, ux)) {
561 case DFMT_COLLAPSED:
562 case DFMT_BLOCK:
563 case DFMT_CYCLIC:
564 case DFMT_GEN_BLOCK:
565 break;
566 case DFMT_BLOCK_K:
567 case DFMT_CYCLIC_K:
568 redistribute = (DIST_DPTR_BLOCK_G(udd) != block[i]);
569 break;
570 default:
571 __fort_abort("REDISTRIBUTE: unsupported dist-format");
572 }
573 }
574#if defined(DEBUG)
575 if (__fort_test & DEBUG_RDST) {
576 for (i = 0; i < rank; ++i) {
577 printf("%d dim=%d ddfmt=%d paxis=%d block=%d\n", GET_DIST_LCPU, i + 1,
578 ddfmt >> DFMT__WIDTH * i & DFMT__MASK, paxis[i], block[i]);
579 }
580 printf("%d nmapped=%d redistribute=%d\n", GET_DIST_LCPU, nmapped,
581 redistribute);
582 }
583#endif
584
585 if (!redistribute)
586 return;
587
588 /* create a duplicate of the new align-target first. After all
589 the alignees have been redistributed, then the original
590 align-target is updated in place and the align-target pointer
591 in each alignee is reset back to the original align-target. */
592
593 __DIST_INIT_DESCRIPTOR(td, rank, F90_KIND_G(ud), F90_LEN_G(ud), flags, tp);
594 j = 0; /*for gen_block*/
595 for (i = 1; i <= rank; ++i) {
596 ux = DIST_DIM_TAXIS_G(dd, i - 1);
597#if defined(DEBUG)
598 if (ux <= 0)
599 __fort_abort("REDISTRIBUTE: invalid distributee align axis");
600#endif
601 SET_DIM_PTRS(udd, ud, ux - 1);
602 DIST_DFMT_P(td, tdfmt);
603
604 if ((tdfmt >> DFMT__WIDTH * (i - 1) & DFMT__MASK) == DFMT_GEN_BLOCK) {
605 DIST_DIM_GEN_BLOCK_P(td, i - 1, gbCopy[j++]);
606 } else {
607 DIST_DIM_GEN_BLOCK_P(td, i - 1, 0);
608 }
609
610 __DIST_SET_DISTRIBUTION(td, ux, F90_DPTR_LBOUND_G(udd), DPTR_UBOUND_G(udd),
611 paxis[i - 1], &block[i - 1]);
612
613 if (~F90_FLAGS_G(ud) & __TEMPLATE)
614 __DIST_SET_ALLOCATION(td, ux, DIST_DPTR_NO_G(udd), DIST_DPTR_PO_G(udd));
615 }
616 if (~F90_FLAGS_G(ud) & __TEMPLATE)
617 I8(__fort_finish_descriptor)((td));
618
619 /* reallocate and copy the old into the new */
620
621 I8(recopy)(td, ud, ud);
622
623 /* redistribute each alignee */
624
625 ad = DIST_NEXT_ALIGNEE_G(ud);
626 while (ad != NULL) {
627
628#if defined(DEBUG)
629 if (ad == ud)
630 __fort_abort("REDISTRIBUTE: distributee in own alignee's list");
631 if (DIST_ALIGN_TARGET_G(ad) != ud)
632 __fort_abort("REDISTRIBUTE: alignee has different align-target");
633#endif
634
635 /* make a copy of the old alignee descriptor */
636
637 I8(__fort_copy_descriptor)(od, ad);
638
639 /* update alignee descriptor in place */
640
641 __DIST_INIT_DESCRIPTOR(ad, F90_RANK_G(od), F90_KIND_G(od), F90_LEN_G(od),
642 F90_FLAGS_G(od), td);
643 for (i = 1; i <= F90_RANK_G(od); ++i) {
644 int tx;
645
646 SET_DIM_PTRS(odd, od, i - 1);
647
648 /*
649 * pasing gen_block field in thru last arg
650 */
651
