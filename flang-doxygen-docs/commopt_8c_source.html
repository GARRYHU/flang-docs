90static void
91selection_sort(void)
92{
93 int i, j;
94 int hd, hd1;
95 int dom;
96 int t;
97 int ast, ast1;
98 int std, std1;
99
100 lpsort = (INT *)getitem(FORALL_AREA, (opt.nloops + 1) * sizeof(INT));
101 for (i = 1; i <= opt.nloops; ++i)
102 lpsort[i] = i;
103
104 for (i = 1; i <= opt.nloops - 1; ++i) {
105 dom = i;
106 for (j = i + 1; j <= opt.nloops; ++j) {
107 hd = LP_HEAD(lpsort[dom]);
108 hd1 = LP_HEAD(lpsort[j]);
109 std = FG_STDFIRST(hd);
110 ast = STD_AST(std);
111 std1 = FG_STDFIRST(hd1);
112 ast1 = STD_AST(std1);
113 if (is_dominator(hd1, hd))
114 dom = j;
115 }
116 t = lpsort[dom];
117 lpsort[dom] = lpsort[i];
118 lpsort[i] = t;
119 }
120}

181void
182forall_init(void)
183{
184 int std;
185 int ast;
186 SPTR current_block = 0;
187 int parallel_depth;
188 int task_depth;
189
190 init_ftb();
191 init_dtb();
192 init_optsum();
193 init_bnd();
194
195 for (std = STD_NEXT(0); std;) {
196 ast = STD_AST(std);
197 if (A_TYPEG(ast) == A_FORALL) {
198 forall_opt1(ast);
199 }
200 std = STD_NEXT(std);
201 }
202 /* put calls into calls table */
203 parallel_depth = 0;
204 task_depth = 0;
205 for (std = STD_NEXT(0); std;) {
206 ast = STD_AST(std);
207 switch (A_TYPEG(ast)) {
208 case A_CONTINUE: {
209 // Set current_block to the current ST_BLOCK sym.
210 SPTR lab = STD_LABEL(std);
211 SPTR block_sptr;
212 if (!lab)
213 break;
214 block_sptr = ENCLFUNCG(lab);
215 if (!block_sptr || STYPEG(block_sptr) != ST_BLOCK)
216 break;
217 if (lab == STARTLABG(block_sptr)) {
218 current_block = block_sptr;
219 } else if (lab == ENDLABG(block_sptr)) {
220 block_sptr = ENCLFUNCG(block_sptr);
221 current_block =
222 block_sptr && STYPEG(block_sptr) == ST_BLOCK ? block_sptr : 0;
223 }
224 break;
225 }
226 case A_FORALL:
227 put_forall_pcalls(std);
228 forall_make_same_idx(std);
229 if (!STD_BLKSYM(std))
230 // Mark the loop with the current block, for use in fuse_forall.
231 STD_BLKSYM(std) = current_block;
232 break;
233 case A_MP_PARALLEL:
234 ++parallel_depth;
235 /*symutl.sc = SC_PRIVATE;*/
236 set_descriptor_sc(SC_PRIVATE);
237 break;
238 case A_MP_ENDPARALLEL:
239 --parallel_depth;
240 if (parallel_depth == 0 && task_depth == 0) {
241 /*symutl.sc = SC_LOCAL;*/
242 set_descriptor_sc(SC_LOCAL);
243 }
244 break;
245 case A_MP_TASK:
246 case A_MP_TASKLOOP:
247 ++task_depth;
248 set_descriptor_sc(SC_PRIVATE);
249 break;
250 case A_MP_ENDTASK:
251 case A_MP_ETASKLOOP:
252 --task_depth;
253 if (parallel_depth == 0 && task_depth == 0) {
254 set_descriptor_sc(SC_LOCAL);
255 }
256 break;
257 default:
258 break;
259 }
260 std = STD_NEXT(std);
261 }
262}

329static void
330fuse_forall(int nested)
331{
332 int i, j;
333 int std, std1;
334 int forall, forall1;
335 int nd, nd1;
336 int hd, hd1;
337 int ii, jj;
338 int fg, fg1;
339 int number_of_try;
340
341 for (ii = 1; ii <= opt.nloops; ++ii) {
342 i = lpsort[ii];
343 if (!LP_FORALL(i))
344 continue;
345 hd = LP_HEAD(i); /*the flow graph node which is the loop head*/
346 fg = LP_FG(i);
347 std = FG_STDFIRST(hd);
348 forall = STD_AST(std);
349 nd = A_OPT1G(forall);
350 if (FT_NFUSE(nd, nested) >= MAXFUSE)
351 continue;
352 if (FT_FUSED(nd))
353 continue;
354 number_of_try = 0;
355 for (jj = ii + 1; jj <= opt.nloops; ++jj) {
356 j = lpsort[jj];
357 if (LP_PARENT(i) != LP_PARENT(j))
358 continue;
359 if (FT_NFUSE(nd, nested) >= MAXFUSE)
360 break;
361 if (number_of_try > 25)
362 continue;
363 if (LP_FORALL(j)) {
364 /*the flow graph node which is the loop head */
365 hd1 = LP_HEAD(j);
366 fg1 = LP_FG(j);
367 if (!is_dominator(hd, hd1))
368 continue;
369 std1 = FG_STDFIRST(hd1);
370 if (STD_BLKSYM(std) != STD_BLKSYM(std1))
371 continue;
372 if (STD_ACCEL(std) != STD_ACCEL(std1))
373 continue;
374 forall1 = STD_AST(std1);
375 nd1 = A_OPT1G(forall1);
376 if (FT_FUSED(nd1))
377 continue;
378 if (!same_forall_bnds(i, j, nested))
379 continue;
380 if (!is_same_idx(std, std1))
381 continue;
382 number_of_try++;
383 if (!is_fusable(i, j, nested))
384 continue;
385
386 FT_FUSELP(nd, nested, FT_NFUSE(nd, nested)) = j;
387 FT_FUSEDSTD(nd, nested, FT_NFUSE(nd, nested)) = std1;
388 FT_NFUSE(nd, nested)++;
389 FT_FUSED(nd1) = 1;
390 FT_HEADER(nd1) = FT_HEADER(nd);
391 FT_FUSED(nd) = 1;
392 optsum.fuse++;
393 }
394 }
395 }

398static LOGICAL
399is_same_idx(int std, int std1)
400{
401 int idx[7], idx1[7];
402 int list, list1, listp;
403 int forall, forall1;
404 int nidx, nidx1;
405 int isptr, isptr1;
406 int i;
407
408 forall = STD_AST(std);
409 list = A_LISTG(forall);
410 nidx = 0;
411 for (listp = list; listp != 0; listp = ASTLI_NEXT(listp)) {
412 idx[nidx] = listp;
413 nidx++;
414 }
415
416 forall1 = STD_AST(std1);
417 list1 = A_LISTG(forall1);
418 nidx1 = 0;
419 for (listp = list1; listp != 0; listp = ASTLI_NEXT(listp)) {
420 idx1[nidx1] = listp;
421 nidx1++;
422 }
423
424 if (nidx != nidx1)
425 return FALSE;
426
427 for (i = 0; i < nidx; i++) {
428 isptr = ASTLI_SPTR(idx[i]);
429 isptr1 = ASTLI_SPTR(idx1[i]);
430 if (isptr != isptr1)
431 return FALSE;
432 }
433 return TRUE;
434}

517static LOGICAL
518same_forall_bnds(int lp1, int lp2, int nested)
519{
520 int idx1[7], idx2[7];
521 int itriple1, itriple2;
522 int lb1, ub1, st1;
523 int lb2, ub2, st2;
524 int std1, std2;
525 int fg1, fg2;
526 int hd1, hd2;
527 int list1, list2, listp;
528 int forall1, forall2;
529 int nidx1, nidx2;
530 int i, k;
531 int asd1, asd2;
532 int ndim1, ndim2;
533 int order2[MAXDIMS];
534 int no;
535 int lhs1, lhs2, newlhs2, l, l2;
536 int sptr1, sptr2;
537 int isptr1, isptr2;
538 int dim1, dim2;
539 int newast, oldast;
540 LOGICAL same;
541
542 hd1 = LP_HEAD(lp1); /*the flow graph node which is the loop head*/
543 fg1 = LP_FG(lp1);
544 std1 = FG_STDFIRST(hd1);
545 forall1 = STD_AST(std1);
546 lhs1 = A_DESTG(A_IFSTMTG(forall1));
547 list1 = A_LISTG(forall1);
548 nidx1 = 0;
549 for (listp = list1; listp != 0; listp = ASTLI_NEXT(listp)) {
550 idx1[nidx1] = listp;
551 ++nidx1;
552 }
553
554 hd2 = LP_HEAD(lp2);
555 fg2 = LP_FG(lp2);
556 if (!is_dominator(hd1, hd2))
557 return FALSE;
558 std2 = FG_STDFIRST(hd2);
559 forall2 = STD_AST(std2);
560 lhs2 = A_DESTG(A_IFSTMTG(forall2));
561 list2 = A_LISTG(forall2);
562 nidx2 = 0;
563 for (listp = list2; listp != 0; listp = ASTLI_NEXT(listp)) {
564 idx2[nidx2] = listp;
565 nidx2++;
566 }
567
568 if (nidx1 != nidx2)
569 return FALSE;
570
571 for (i = 0; i < nidx1 - nested; i++) {
572 itriple1 = ASTLI_TRIPLE(idx1[i]);
573 lb1 = A_LBDG(itriple1);
574 ub1 = A_UPBDG(itriple1);
575 st1 = A_STRIDEG(itriple1);
576 if (st1 == 0)
577 st1 = astb.i1;
578 itriple2 = ASTLI_TRIPLE(idx2[i]);
579 lb2 = A_LBDG(itriple2);
580 ub2 = A_UPBDG(itriple2);
581 st2 = A_STRIDEG(itriple2);
582 if (st2 == 0)
583 st2 = astb.i1;
584 same = FALSE;
585 if (lb1 == lb2 && ub1 == ub2 && st1 == st2) {
586 if ((is_avail_expr(lb1, std1, fg1, std2, fg2) &&
587 is_avail_expr(ub1, std1, fg1, std2, fg2) &&
588 is_avail_expr(st1, std1, fg1, std2, fg2)))
589 same = TRUE;
590 }
591 if (!same)
592 return FALSE;
593 }
594
595 if (is_duplicate(lhs1, list1))
596 return FALSE;
597 if (is_duplicate(lhs2, list2))
598 return FALSE;
599
600 ast_visit(1, 1);
601 for (i = 0; i < nidx1; i++) {
602 isptr1 = ASTLI_SPTR(idx1[i]);
603 newast = mk_id(isptr1);
604 isptr2 = ASTLI_SPTR(idx2[i]);
605 oldast = mk_id(isptr2);
606 ast_replace(oldast, newast);
607 }
608 newlhs2 = ast_rewrite(lhs2);
609 ast_unvisit();
610 if (!is_ordered(lhs1, newlhs2, list1, order2, &no))
611 return FALSE;
612
613 /* has to have same distribution */
614 l = left_subscript_ast(lhs1);
615 sptr1 = left_array_symbol(lhs1);
616 l2 = left_subscript_ast(newlhs2);
617 sptr2 = left_array_symbol(newlhs2);
618
619 assert(A_TYPEG(l) == A_SUBSCR, "same_forall_bnds: not a subscript", l, 4);
620 asd1 = A_ASDG(l);
621 ndim1 = ASD_NDIM(asd1);
622 assert(A_TYPEG(l2) == A_SUBSCR, "same_forall_bnds: not a subscript", l, 4);
623 asd2 = A_ASDG(l2);
624 ndim2 = ASD_NDIM(asd2);
625
626 for (i = 0; i < nidx1; i++) {
627 isptr1 = ASTLI_SPTR(idx1[i]);
628 dim1 = -1;
629 dim2 = -1;
630 for (k = 0; k < ndim1; k++)
631 if (is_name_in_expr(ASD_SUBS(asd1, k), isptr1))
632 dim1 = k;
633 for (k = 0; k < ndim2; k++)
634 if (is_name_in_expr(ASD_SUBS(asd2, k), isptr1))
635 dim2 = k;
636 if (dim1 == -1 || dim2 == -1) {
637 /* index does not appear in any distributed dimensions;
638 * until I figure out just what this is doing, be safe */
639 return FALSE;
640 }
641 if (!is_same_align_shape(sptr1, dim1, sptr2, dim2))
642 return FALSE;
643 if (ASD_SUBS(asd1, dim1) != ASD_SUBS(asd2, dim2))
644 return FALSE;
645 }
646
647 return TRUE;
648}

675/* This routine will return TRUE iff,
676 * lhs array appears at sink and their subscripts are different.
677 */
678static LOGICAL
679Conflict_(int sink)
680{
681 LOGICAL l, r;
682 int argt;
683 int nargs;
684 int i;
685 int result;
686 int src, sptr, sptr1;
687
688 if (sink == 0)
689 return FALSE;
690 switch (A_TYPEG(sink)) {
691 case A_CMPLXC:
692 case A_CNST:
693 case A_SUBSTR:
694 return FALSE;
695 case A_MEM:
696 if (Conflict_(A_PARENTG(sink))) {
697 /* see if this 'member' appears in the 'conf.src' tree */
698 int a, p, member;
699 a = conf.src;
700 member = A_SPTRG(A_MEMG(sink));
701 while (1) {
702 switch (A_TYPEG(a)) {
703 case A_SUBSCR:
704 a = A_LOPG(a);
705 break;
706 case A_ID:
707 return TRUE;
708 break;
709 case A_MEM:
710 p = A_PARENTG(a);
711 if (DDTG(A_DTYPEG(p)) == ENCLDTYPEG(member)) {
712 /* same member? different? */
713 if (A_MEMG(a) == A_MEMG(sink))
714 return TRUE;
715 return FALSE;
716 }
717 a = p;
718 break;
719 default:
720 interr("Conflict_: unexpected AST in member tree", a, 3);
721 return FALSE;
722 }
723 }
724 }
725 return FALSE;
726 case A_ID:
727 if (A_SPTRG(sink) == sym_of_ast(conf.src))
728 return TRUE;
729 else
730 return FALSE;
731 case A_BINOP:
732 l = Conflict_(A_LOPG(sink));
733 if (l)
734 return TRUE;
735 r = Conflict_(A_ROPG(sink));
736 if (r)
737 return TRUE;
738 return FALSE;
739 case A_UNOP:
740 return Conflict_(A_LOPG(sink));
741 case A_PAREN:
742 case A_CONV:
743 return Conflict_(A_LOPG(sink));
744 case A_SUBSCR:
745 if (Conflict_(A_LOPG(sink))) {
746 if (conf.order)
747 result = dd_array_conflict(conf.list, sink, conf.src, conf.after);
748 else
749 result = dd_array_conflict(conf.list, conf.src, sink, conf.after);
750 return result;
751 }
752 return FALSE;
753 case A_TRIPLE:
754 l = Conflict_(A_LBDG(sink));
755 if (l)
756 return TRUE;
757 r = Conflict_(A_UPBDG(sink));
758 if (r)
759 return TRUE;
760 return Conflict_(A_STRIDEG(sink));
761 case A_INTR:
762 case A_FUNC:
763 nargs = A_ARGCNTG(sink);
764 argt = A_ARGSG(sink);
765 /* dd_array_conflict does not work if the constructed section is
766 * out of bound. a(10);forall(i=1:10) b(i) = cshift(a(i+2))
767 * here a(3:12) is not in bounds.
768 */
769 if (A_OPTYPEG(sink) == I_CSHIFT || A_OPTYPEG(sink) == I_EOSHIFT) {
770 src = ARGT_ARG(argt, 0);
771 sptr = sym_of_ast(src);
772 sptr1 = sym_of_ast(conf.src);
773 if (sptr == sptr1)
774 return TRUE;
775 }
776 for (i = 0; i < nargs; ++i) {
777 l = Conflict_(ARGT_ARG(argt, i));
778 if (l)
779 return TRUE;
780 }
781 return FALSE;
782 case A_LABEL:
783 default:
784 interr("Conflict_: unexpected ast", sink, 2);
785 return FALSE;
786 }
787}

789static LOGICAL
790_olap_conflict(int expr, int expr1)
791{
792 LOGICAL l, r;
793 int argt;
794 int nargs;
795 int i;
796
797 if (expr == 0)
798 return FALSE;
799 switch (A_TYPEG(expr)) {
800 case A_CMPLXC:
801 case A_CNST:
802 case A_ID:
803 case A_SUBSTR:
804 case A_MEM:
805 return FALSE;
806 case A_BINOP:
807 l = _olap_conflict(A_LOPG(expr), expr1);
808 if (l)
809 return TRUE;
810 r = _olap_conflict(A_ROPG(expr), expr1);
811 if (r)
812 return TRUE;
813 return FALSE;
814 case A_UNOP:
815 return _olap_conflict(A_LOPG(expr), expr1);
816 case A_PAREN:
817 case A_CONV:
818 return _olap_conflict(A_LOPG(expr), expr1);
819 case A_SUBSCR:
820 return FALSE;
821 case A_TRIPLE:
822 l = _olap_conflict(A_LBDG(expr), expr1);
823 if (l)
824 return TRUE;
825 r = _olap_conflict(A_UPBDG(expr), expr1);
826 if (r)
827 return TRUE;
828 return _olap_conflict(A_STRIDEG(expr), expr1);
829 case A_INTR:
830 case A_FUNC:
831 nargs = A_ARGCNTG(expr);
832 argt = A_ARGSG(expr);
833 for (i = 0; i < nargs; ++i) {
834 l = _olap_conflict(ARGT_ARG(argt, i), expr1);
835 if (l)
836 return TRUE;
837 }
838 if (A_OPTYPEG(expr) == I_CSHIFT || A_OPTYPEG(expr) == I_EOSHIFT)
839 return is_shift_conflict(expr, argt, expr1);
840 return FALSE;
841 case A_LABEL:
842 default:
843 interr("_olap_conflict: unexpected ast", expr, 2);
844 return FALSE;
845 }

879/**
880 \brief ...
881
882 <pre>
883 S1: A(i) = ...
884 S2: ... = ... A(f(i)) ...
885 </pre>
886
887 The loops cannot be fused if either of the following calls to
888 dd_array_conflict() return TRUE:
889 <pre>
890 dd_array_conflict(tripletList, A(f(i)), A(i), FALSE)
891 dd_array_conflict(tripletList, A(i), A(f(i)), TRUE)
892 </pre>
893
894 Alternatively, the statements within the original loops may be of the
895 following form:
896
897 <pre>
898 S1: ... = ... A(f(i))
899 S2: A(i) = ...
900 </pre>
901
902 The loops cannot be fused if the following call to dd_array_conflict()
903 returns TRUE.
904 <pre>
905 dd_array_conflict(tripletList, A(i), A(f(i)), FALSE)
906 </pre>
907
908 The back end recognizes loops with 0 RHS and transforms these into calls
909 to _mzero, so don't fuse assignments with just 0 on the RHS.
910 */
911static LOGICAL
912is_fusable(int lp, int lp1, int nested)
913{
914
915 int k;
916 int std, std1;
917 int forall, forall1;
918 int nd, nd1;
919 int hd, hd1;
920 int expr, expr1;
921 int asn, asn1;
922 int rhs, rhs1;
923 int lhs, lhs_array, lhs1, lhs1_array;
924 int lhs_sptr, lhs_sptr1;
925 int list, list1, listp;
926 int isptr;
927 int fg, fg1;
928 LOGICAL fuseable;
929 int fuselp;
930 int triple;
931 int idx[7];
932 int cnt;
933 LOGICAL fuse_cnst_rhs;
934
935 if (XBIT(47, 0x80000000))
936 return FALSE;
937
938 hd = LP_HEAD(lp); /*the flow graph node which is the loop head*/
939 fg = LP_FG(lp);
940 std = FG_STDFIRST(hd);
941 forall = STD_AST(std);
942 list = A_LISTG(forall);
943 /*
944 * NOTES:
945 * XBIT(47,0x4000000) - fuse even if constant RHS
946 * XBIT(8,0x8000000) - inhibit mem idioms
947 */
948 fuse_cnst_rhs = FALSE;
949 if (XBIT(47, 0x4000000))
950 fuse_cnst_rhs = TRUE;
951 else if (XBIT(8, 0x8000000))
952 fuse_cnst_rhs = TRUE;
953 else {
954 }
955
956 expr = A_IFEXPRG(forall);
957 asn = A_IFSTMTG(forall);
958 rhs = A_SRCG(asn);
959 lhs = A_DESTG(asn);
960 for (lhs_array = lhs; A_TYPEG(lhs_array) == A_MEM;
961 lhs_array = A_PARENTG(lhs_array))
962 ;
963 lhs_sptr = sym_of_ast(lhs);
964 nd = A_OPT1G(forall);
965
966 if (LP_PARENT(lp) != LP_PARENT(lp1))
967 return FALSE;
968
969 if (A_TYPEG(rhs) == A_CONV)
970 rhs = A_LOPG(rhs);
971 if (flg.opt >= 2 && !fuse_cnst_rhs && A_TYPEG(rhs) == A_CNST) {
972 /*
973 * prefer calling a tuned mzero/mem rather fusing.
974 */
975 return FALSE;
976 }
977 /*the flow graph node which is the loop head */
978 hd1 = LP_HEAD(lp1);
979 fg1 = LP_FG(lp1);
980 if (!is_dominator(hd, hd1))
981 return FALSE;
982 if (smp_conflict(hd, hd1))
983 return FALSE;
984 std1 = FG_STDFIRST(hd1);
985 forall1 = STD_AST(std1);
986 nd1 = A_OPT1G(forall1);
987 list1 = A_LISTG(forall1);
988 expr1 = A_IFEXPRG(forall1);
989 asn1 = A_IFSTMTG(forall1);
990 rhs1 = A_SRCG(asn1);
991 lhs1 = A_DESTG(asn1);
992 for (lhs1_array = lhs1; A_TYPEG(lhs1_array) == A_MEM;
993 lhs1_array = A_PARENTG(lhs1_array))
994 ;
995 lhs_sptr1 = sym_of_ast(lhs1);
996 if (A_TYPEG(rhs1) == A_CONV)
997 rhs1 = A_LOPG(rhs1);
998 if (flg.opt >= 2 && !fuse_cnst_rhs && A_TYPEG(rhs1) == A_CNST) {
999 /*
1000 * prefer calling a tuned mzero/mem rather fusing.
1001 */
1002 return FALSE;
1003 }
1004
1005 /* forall1 values should not changed between std and std1 */
1006 /* Except that if they are in block, nothing change them */
1007 if (!is_in_block(std, std1)) {
1008 cnt = 0;
1009 for (listp = list1; listp != 0; listp = ASTLI_NEXT(listp)) {
1010 isptr = ASTLI_SPTR(listp);
1011 idx[cnt] = mk_id(isptr);
1012 cnt++;
1013 }
1014 if (!is_avail_expr_with_list(expr1, std, fg, std1, fg1, idx, cnt) ||
1015 !is_avail_expr_with_list(rhs1, std, fg, std1, fg1, idx, cnt) ||
1016 !is_avail_expr_with_list(lhs1, std, fg, std1, fg1, idx, cnt))
1017 return FALSE;
1018 }
1019
1020 /* dependency */
1021 if (subscr_dependent(rhs1, lhs, std1, std))
1022 return FALSE;
1023 if (rhs1 && A_LOPG(rhs1) && A_TYPEG(rhs1) == A_SUBSCR &&
1024 A_TYPEG(A_LOPG(rhs1)) == A_MEM)
1025 if (subscr_dependent(A_LOPG(rhs1), lhs, std1, std))
1026 return FALSE;
1027 if (subscr_dependent(expr1, lhs, std1, std))
1028 return FALSE;
1029 if (expr1 && A_LOPG(expr1) && A_TYPEG(expr1) == A_SUBSCR &&
1030 A_TYPEG(A_LOPG(expr1)) == A_MEM)
1031 if (subscr_dependent(A_LOPG(expr1), lhs, std1, std))
1032 return FALSE;
1033 if (subscr_dependent(lhs1, lhs, std1, std))
1034 return FALSE;
1035 if (lhs1 && A_LOPG(lhs1) && A_TYPEG(lhs1) == A_SUBSCR &&
1036 A_TYPEG(A_LOPG(lhs1)) == A_MEM)
1037 if (subscr_dependent(A_LOPG(lhs1), lhs, std1, std))
1038 return FALSE;
1039 if (rhs && lhs1 && A_TYPEG(rhs) == A_SUBSCR && A_LOPG(lhs1) &&
1040 A_TYPEG(lhs1) == A_SUBSCR)
1041 if (subscr_dependent(rhs, A_LOPG(lhs1), std1, std))
1042 return FALSE;
1043
1044 /* don't let LHSs appears at FORALL list */
1045 /* forall(i=x(1):x(2)) x(i) = */
1046 for (listp = list1; listp != 0; listp = ASTLI_NEXT(listp)) {
1047 triple = ASTLI_TRIPLE(listp);
1048 if (A_TYPEG(lhs_array) == A_SUBSCR &&
1049 contains_ast(triple, A_LOPG(lhs_array)))
1050 return FALSE;
1051 if (A_TYPEG(lhs1_array) == A_SUBSCR &&
1052 contains_ast(triple, A_LOPG(lhs1_array)))
1053 return FALSE;
1054 }
1055
1056 /* because of using mask pghpf_vsub_gather */
1057 /* this can be optimized, iff expr1 and rhs1
1058 * does not have indirections */
1059 if (A_TYPEG(lhs_array) == A_SUBSCR && contains_ast(expr1, A_LOPG(lhs_array)))
1060 return FALSE;
1061
1062 if (expr1)
1063 if (Conflict(A_LISTG(forall), lhs, expr1, FALSE, 1, 0))
1064 return FALSE;
1065 if (Conflict(A_LISTG(forall), lhs, rhs1, FALSE, 1, 0))
1066 return FALSE;
1067
1068 /* for communication */
1069 conf.otherlhs = lhs1;
1070 if (expr1)
1071 if (Conflict(A_LISTG(forall), lhs, expr1, TRUE, 0, 1))
1072 return FALSE;
1073 if (Conflict(A_LISTG(forall), lhs, rhs1, TRUE, 0, 1))
1074 return FALSE;
1075
1076 if (expr)
1077 if (Conflict(A_LISTG(forall), lhs1, expr, FALSE, 0, 0))
1078 return FALSE;
1079 if (Conflict(A_LISTG(forall), lhs1, rhs, FALSE, 0, 0))
1080 return FALSE;
1081
1082 if (is_branch_between(lp, lp1))
1083 return FALSE;
1084 if (!is_in_block(std, std1))
1085 if (A_TYPEG(lhs1_array) == A_SUBSCR &&
1086 is_contains_ast_between(lp, lp1, A_LOPG(lhs1_array)))
1087 return FALSE;
1088
1089 if (is_olap_conflict(forall, forall1))
1090 return FALSE;
1091 fuseable = TRUE;
1092 for (k = 0; k < FT_NFUSE(nd, nested); k++) {
1093 fuselp = FT_FUSELP(nd, nested, k);
1094 if (!is_fusable(fuselp, lp1, nested))
1095 fuseable = FALSE;
1096 }
1097 if (!fuseable)
1098 return FALSE;
1099
1100 return TRUE;
1101}

1103/*
1104 * Determine if two flowgraph nodes conflict with respect to smp
1105 * execution given that fg1 dominates fg2.
1106 */
1107static LOGICAL
1108smp_conflict(int fg1, int fg2)
1109{
1110 int fg;
1111 int std;
1112 int ast;
1113
1114 if (!flg.smp)
1115 return FALSE;
1116 if (fg1 == fg2)
1117 /* no conflict if the same node */
1118 return FALSE;
1119 if ((FG_PAR(fg1) && !FG_PAR(fg2)) || (!FG_PAR(fg1) && FG_PAR(fg2)))
1120 /* fg1 is serial & fg2 is parallel, or vice versa */
1121 return TRUE;
1122 if (FG_PAR(fg2)) {
1123 /* both are within a parallel region; determine if it's the
1124 * same region.
1125 */
1126 for (fg = fg1; fg != fg2; fg = FG_LNEXT(fg)) {
1127 rdilts(fg);
1128 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
1129 ast = STD_AST(std);
1130 if (A_TYPEG(ast) == A_MP_ENDPARALLEL) {
1131 /* endparallel was seen in a node before fg2 - it must
1132 * be a different parallel region.
1133 */
1134 wrilts(fg);
1135 return TRUE;
1136 }
1137 }
1138 wrilts(fg);
1139 }
1140 }
1141 if (FG_CS(fg1) ^ FG_CS(fg2))
1142 /* fg1 is in a critical section & fg2 is not, or vice versa */
1143 return TRUE;
1144 if (FG_CS(fg2)) {
1145 /* both are within a critical section; determine if it's the
1146 * same critical section.
1147 */
1148 for (fg = fg1; fg != fg2; fg = FG_LNEXT(fg)) {
1149 rdilts(fg);
1150 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
1151 ast = STD_AST(std);
1152 if (A_TYPEG(ast) == A_MP_ENDCRITICAL) {
1153 wrilts(fg);
1154 return TRUE;
1155 }
1156 }
1157 wrilts(fg);
1158 }
1159 }
1160 if (FG_PARSECT(fg1) ^ FG_PARSECT(fg2))
1161 /* fg1 is in a parallel section (master, single, sections) & fg2 is
1162 * not, or vice versa
1163 */
1164 return TRUE;
1165 if (FG_PARSECT(fg2)) {
1166 /* both are within a parallel section; determine if it's the
1167 * same parallel section.
1168 */
1169 for (fg = fg1; fg != fg2; fg = FG_LNEXT(fg)) {
1170 rdilts(fg);
1171 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
1172 ast = STD_AST(std);
1173 switch (A_TYPEG(ast)) {
1174 case A_MP_ENDMASTER:
1175 case A_MP_ENDSINGLE:
1176 case A_MP_ENDSECTIONS:
1177 wrilts(fg);
1178 return TRUE;
1179 }
1180 }
1181 wrilts(fg);
1182 }
1183 }
1184
1185 return FALSE;
1186}

1283/* This routine is same as is_avail_expr except that
1284 * it will look at the cnt number of variables given by idx.
1285 * idx has variable A_ID ast. it will decompose_expression
1286 * and eliminate idx variable from list.
1287 * it helps forall triplet variables ignored.
1288 */
1289static LOGICAL
1290is_avail_expr_with_list(int expr, int std, int fg, int std1, int fg1, int idx[],
1291 int cnt)
1292{

1295 int ele;
1296 int found;
1297 int i, j;
1298 int nvar1;
1299
1300 if (!expr)
1301 return TRUE;
1302 size = 100;
1303 nvar = 0;
1304
1305 decompose_expression(expr, lst, size, &nvar, NULL);
1306 if (nvar > size)
1307 return FALSE;
1308
1309 /* eliminate duplicate variables from the lst */
1310 nvar1 = 0;
1311 for (i = 0; i < nvar; i++) {
1312 found = 0;
1313 for (j = 0; j < nvar1; j++)
1314 if (lst[i] == lst1[j])
1315 found = 1;

1323 ele = lst1[i];
1324 found = 0;
1325 for (j = 0; j < cnt; j++)
1326 if (idx[j] == ele)
1327 found = 1;

1329 continue;

1331 return FALSE;
1332 }
1333 return TRUE;

1427 type1 = A_TYPEG(rt1);
1428 if (type1 == A_ASN)
1429 type1 = A_TYPEG(A_SRCG(rt1));
1430 if (STD_DELETE(rt1_std))
1431 continue;
1432 if (rt == rt1)
1433 continue;
1434 if (type != type1)
1435 continue;

1437 if (k > l)
1438 continue;

1614static LOGICAL
1615is_same_array_bounds(int sub, int sub1, int std, int std1, int fg, int fg1)
1616{
1617 int ndim, ndim1;

1619 int count;
1620 int lb, ub, st;
1621 int lb1, ub1, st1;
1622 int i;
1623 int itriple, itriple1;

1625 while (1) {
1626 if (A_TYPEG(sub) != A_TYPEG(sub1))
1627 return FALSE;
1628 switch (A_TYPEG(sub)) {
1629 case A_ID:
1630 return TRUE;
1631 case A_MEM:
1632 sub = A_PARENTG(sub);
1633 sub1 = A_PARENTG(sub1);

1635 case A_SUBSTR:
1636 sub = A_LOPG(sub);
1637 sub1 = A_LOPG(sub1);

1652 st = A_STRIDEG(itriple);

1654 st = astb.i1;

1657 ub = astb.i1;
1658 st = astb.i1;
1659 }

1670 ub1 = astb.i1;
1671 st1 = astb.i1;

1675 if (is_avail_expr(lb, std, fg, std1, fg1) &&

1688 return FALSE;
1689 }
1690 }
1691}

3089 dest = A_DESTG(ast);

3092 defstd = DEF_STD(def);
3093 deffg = DEF_FG(def);

3099 if (A_DESTG(defast) != dest)
