15static int _1 = 1;
16static chdr *exch[16] = {NULL};
17static int *rbuf, *sbuf, *vcounts, *vrbuf, *vsbuf;
18static int rmax, smax, vrmax; /* * * * NOT THREAD SAFE * * * */
19
20/* counts is an integer array of length 2**ceil(log2 np). For each
21 pair of processors {i,j}, counts[j] on processor i is exchanged
22 with counts[i] on processor j, and the maximum count value is
23 returned. Ceil(log2 np) communication steps are required. If the
24 number of processors is not a power of two, some processors do
25 double duty as "virtual processors". */
26
27int
28__fort_exchange_counts(int *counts)
29{
30 chdr *c;
31 int cpu, lcpu, i, j, k, l, m, n, tcpus, vcpu, vme;
32 int *tcpus_addr;
33
34 lcpu = GET_DIST_LCPU;
35 tcpus = GET_DIST_TCPUS;
36 tcpus_addr = GET_DIST_TCPUS_ADDR;
37 smax = counts[0]; /* initial maximum */
38 if (tcpus == 1)
39 return smax;
40
41 for (i = tcpus; --i > 0;) {
42 if (counts[i] > smax)
43 smax = counts[i];
44 }
45
46#if defined(DEBUG)
47 if (__fort_test & DEBUG_EXCH) {
48 for (i = tcpus; i < __fort_np2; ++i)
49 counts[i] = -lcpu;
50 printf("%d exch counts", lcpu);
51 for (i = 0; i < tcpus; ++i)
52 printf(" %d", counts[i]);
53 printf("\n");
54 }
55#endif
56
57 m = __fort_np2 >> 1; /* message length, also most
58 significant bit of processor number */
59 vme = lcpu ^ m; /* virtual processor number */
60
61 /* one-time setup... */
62
63 if (exch[0] == NULL) {
64
65 /* allocate buffers */
66
67 n = __fort_np2;
68 if (__fort_np2 != tcpus)
69 n *= 3; /* more buffer needed for virtual processor */
70 rbuf = (int *)__fort_gmalloc(n * sizeof(int));
71 sbuf = rbuf + m;
72 if (__fort_np2 != tcpus) {
73 vrbuf = sbuf + m;
74 vsbuf = vrbuf + m;
75 vcounts = vsbuf + m;
76 } else
77 vsbuf = vrbuf = vcounts = NULL;
78
79 /* initialize channels */
80
81 for (n = 0, l = 1; l < tcpus; ++n, l <<= 1) {
82 c = __fort_chn_1to1(NULL, 1, 0, tcpus_addr, &_1, 1, 0, tcpus_addr, &_1);
83 cpu = lcpu ^ l;
84 if (cpu >= tcpus)
85 cpu ^= m;
86 vcpu = vme ^ l;
87 if (vcpu >= tcpus)
88 vcpu ^= m;
89#if defined(DEBUG)
90 if (__fort_test & DEBUG_EXCH) {
91 printf("%d exch l=%d cpu=%d(%d)\n", lcpu, l, lcpu ^ l, cpu);
92 if (vme >= tcpus) {
93 printf("%d exch l=%d cpu=%d(%d)\n", vme, l, vme ^ l, vcpu);
94 }
95 }
96#endif
97 __fort_sendl(c, cpu, &smax, 1, 1, __CINT, sizeof(int));
98 __fort_sendl(c, cpu, sbuf, m, 1, __CINT, sizeof(int));
99 if (vme >= tcpus) {
100 __fort_sendl(c, vcpu, &smax, 1, 1, __CINT, sizeof(int));
101 __fort_sendl(c, vcpu, vsbuf, m, 1, __CINT, sizeof(int));
102 if (cpu == lcpu) {
103 __fort_recvl(c, vcpu, &vrmax, 1, 1, __CINT, sizeof(int));
104 __fort_recvl(c, vcpu, vrbuf, m, 1, __CINT, sizeof(int));
105 __fort_recvl(c, cpu, &rmax, 1, 1, __CINT, sizeof(int));
106 __fort_recvl(c, cpu, rbuf, m, 1, __CINT, sizeof(int));
107 } else {
108 __fort_recvl(c, cpu, &rmax, 1, 1, __CINT, sizeof(int));
109 __fort_recvl(c, cpu, rbuf, m, 1, __CINT, sizeof(int));
110 __fort_recvl(c, vcpu, &vrmax, 1, 1, __CINT, sizeof(int));
111 __fort_recvl(c, vcpu, vrbuf, m, 1, __CINT, sizeof(int));
112 }
113 } else {
114 __fort_recvl(c, cpu, &rmax, 1, 1, __CINT, sizeof(int));
115 __fort_recvl(c, cpu, rbuf, m, 1, __CINT, sizeof(int));
116 }
117 __fort_chn_prune(c);
118 exch[n] = c;
119 }
120 }
121
122#if defined(DEBUG)
123 if (vme >= tcpus) {
124 for (i = 0; i < __fort_np2; ++i)
125 vcounts[i] = -vme;
126 }
127#endif
128
129 /* do the exchanges */
130
131 for (n = 0, l = 1; l < tcpus; ++n, l <<= 1) {
132 cpu = lcpu ^ l;
133 vcpu = vme ^ l;
134 for (i = cpu & l, j = 0; j < m; i += l) {
135 for (k = j + l; j < k; ++i, ++j) {
136 sbuf[j] = counts[i];
137 }
138 }
139 if (vme >= tcpus) {
140 for (i = vcpu & l, j = 0; j < m; i += l) {
141 for (k = j + l; j < k; ++i, ++j) {
142 vsbuf[j] = vcounts[i];
143 }
144 }
145 }
146 __fort_doit(exch[n]);
147 if (rmax > smax)
148 smax = rmax;
149 for (i = cpu & l, j = 0; j < m; i += l) {
150 for (k = j + l; j < k; ++i, ++j) {
151 counts[i] = rbuf[j];
152 }
153 }
154 if (vme >= tcpus) {
155 if (vrmax > smax)
156 smax = vrmax;
157 for (i = vcpu & l, j = 0; j < m; i += l) {
158 for (k = j + l; j < k; ++i, ++j) {
159 vcounts[i] = vrbuf[j];
160 }
161 }
162 }
163#if defined(DEBUG)
164 if (__fort_test & DEBUG_EXCH) {
165 printf("%d exch l=%d cpu=%d rmax=%d", lcpu, l, cpu, rmax);
166 for (i = 0; i < __fort_np2; ++i)
167 printf(" %d", counts[i]);
168 printf("\t>");
169 for (i = 0; i < m; ++i)
170 printf(" %d", sbuf[i]);
171 printf("\t<");
172 for (i = 0; i < m; ++i)
173 printf(" %d", rbuf[i]);
174 printf("\n");
175 if (vme >= tcpus) {
176 printf("%d exch l=%d cpu=%d rmax=%d", vme, l, vcpu, vrmax);
177 for (i = 0; i < __fort_np2; ++i)
178 printf(" %d", vcounts[i]);
179 printf("\t>");
180 for (i = 0; i < m; ++i)
181 printf(" %d", vsbuf[i]);
182 printf("\t<");
183 for (i = 0; i < m; ++i)
184 printf(" %d", vrbuf[i]);
185 printf("\n");
186 }
187 }
188#endif
189 }
190 return smax;
191}
