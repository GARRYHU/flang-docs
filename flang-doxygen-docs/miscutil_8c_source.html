21 \brief Allocate space for and make new filename using mkperm.
22 */
23char *
24mkfname(const char *oldname, const char *oldsuf, const char *newsuf)
25{
26 char *p;
27
28 /* get enough space for oldname, newsuf, '.', '\0', and 1 extra: */
29 p = getitem(8, strlen(oldname) + strlen(newsuf) + 3);
30 strcpy(p, oldname);
31 return (mkperm(p, oldsuf, newsuf));
32}

34/** \brief Create literal string for STATIC STRING
35 *
36 * If isStringW is TRUE(1), oldstr is passed in as a wide string
37 * literal and a letter "L" needs to be prepended to the target string.
38*/
39char *
40literal_string(char *oldstr, int userlen, bool isStringW)
41{
42 static char newstr[MAX_FILENAME_LEN];
43 char *from, *end, *curr, c;
44 int len, skip, start;
45
46 from = oldstr;
47 end = newstr + sizeof(newstr);
48 len = userlen - 1;
49
50 if (!isStringW) {
51 newstr[0] = '\"';
52 skip = 1;
53 start = 1;
54 } else {
55 newstr[0] = 'L';
56 newstr[1] = '\"';
57 skip = 4; /* wide char size */
58 start = 2;
59 }
60
61 for (curr = newstr + start; len-- && curr < end;) {
62 c = *from & 0xff;
63 from += skip;
64 if (c == '\"' || c == '\'' || c == '\\') {
65 *curr++ = '\\';
66 *curr++ = c;
67 } else if (c >= ' ' && c <= '~') {
68 *curr++ = c;
69 } else if (c == '\n') {
70 *curr++ = '\\';
71 *curr++ = 'n';
72 } else {
73 *curr++ = '\\';
74 snprintf(curr, newstr + MAX_FILENAME_LEN - curr, "%03o", c);
75 curr += 3;
76 }
77 }
78
79 *curr++ = '\"';
80 *curr = '\0';
81 return newstr;
82}

155/*
156 * clear 'n' elements of the data structure starting at 'r'
157 * reset stg_cleared if we're initializing or extending the cleared region
158 */
159void
160stg_clear_force(STG *stg, BIGUINT64 r, BIGUINT64 n, bool force)
161{
162 if (r >= 0 && n > 0) {
163 STG *thisstg;
164 if (r == stg->stg_cleared) {
165 stg->stg_cleared += n;
166 } else if (r == 0 && n > stg->stg_cleared) {
167 stg->stg_cleared = n;
168 }
169 for (thisstg = stg; thisstg; thisstg = (STG *)thisstg->stg_sidecar) {
170 thisstg->stg_cleared = stg->stg_cleared;
171 if (force || !STG_CHECKFLAG((*thisstg), STG_FLAG_NOCLEAR))
172 memset((char *)(thisstg->stg_base) + (r * thisstg->stg_dtsize), 0,
173 n * thisstg->stg_dtsize);
174 }
175 }
176} /* stg_clear_force */

179stg_clear(STG *stg, int r, int n)

234/*
235 * reallocate STG structure if we need the extra size (if stg_avail > stg_size)
236 * reallocate any sidecars as well
237 * the new size will be 2*(stg_avail-1), which must be >= 2*stg_size
238 */
239void
240stg_need(STG *stg)
241{
242 STG *thisstg;
243 /* if the compiler has recycled some previously allocated space,
244 * we need to reset the stg_cleared region */
245 if (stg->stg_cleared > stg->stg_avail)
246 stg->stg_cleared = stg->stg_avail;
247 if (stg->stg_avail > stg->stg_size) {
248 BIGUINT64 newsize, oldsize;
249 oldsize = stg->stg_size;
250 newsize = (stg->stg_avail - 1) * 2;
251 /* reallocate stg and all its sidecars */
252 for (thisstg = stg; thisstg; thisstg = (STG *)thisstg->stg_sidecar) {
253 if (DBGBIT(7,0x10))
254 fprintf(gbl.dbgfil, "stg_need(stg=%p, dtsize=%d, size=%d, newsize=%d, name=%s)\n",
255 thisstg, thisstg->stg_dtsize, thisstg->stg_size, newsize, thisstg->stg_name);
256 thisstg->stg_size = newsize;
257 thisstg->stg_base = (void *)sccrelal(
258 (char *)thisstg->stg_base, newsize * thisstg->stg_dtsize);
259 }
260 /* we have to clear all newly allocated elements, in case there
261 * are sidecars with the NOCLEAR flag set, so they get initially cleared */
262 stg_clear_force(stg, oldsize, newsize - oldsize, true);
263 }
264 if (stg->stg_avail > stg->stg_cleared) {
265 /* clear any new elements */
266 stg_clear_force(stg, stg->stg_cleared, stg->stg_avail - stg->stg_cleared, true);
267 }
268} /* stg_need */

270/*
271 * Allocate a sidecar, attach to list of sidecars
272 */
273void
274stg_alloc_sidecar(STG *basestg, STG *stg, int dtsize, const char *name)
275{
276 if (DBGBIT(7,0x10))
277 fprintf(gbl.dbgfil, "stg_alloc_sidecar(basestg=%p, name=%s, stg=%p, dtsize=%d, name=%s)\n",
278 basestg, basestg->stg_name, stg, dtsize, name);
279 if (stg->stg_sidecar) {
280 interrf(ERR_Fatal, "%s: %s has a sidecar, may not add as sidecar to %s",
281 "stg_alloc_sidecar", stg->stg_name, basestg->stg_name);
282 }
283 stg_alloc_base(stg, dtsize, basestg->stg_size, name);
284 stg->stg_avail = basestg->stg_avail;
285 /* clear sidecar for any already-allocated elements */
286 stg_clear_force(stg, 0, stg->stg_size, true);
287 /* link this sidecar to the list of sidecars for the basestg */
288 stg->stg_sidecar = basestg->stg_sidecar;
289 basestg->stg_sidecar = (void *)stg;
290} /* stg_alloc_sidecar */

303/*
304 * Deallocate a sidecar, detach from list of sidecars
305 */
306void
307stg_delete_sidecar(STG *basestg, STG *stg)
308{
309 if (DBGBIT(7,0x10))
310 fprintf(gbl.dbgfil, "stg_delete_sidecar(basestg=%p, name=%s, stg=%p, "
311 "dtsize=%d, name=%s)\n", basestg, basestg->stg_name, stg,
312 stg->stg_dtsize, stg->stg_name);
313 if ((STG *)basestg->stg_sidecar == stg) {
314 basestg->stg_sidecar = stg->stg_sidecar;
315 } else {
316 STG *sidecar;
317 for (sidecar = (STG *)basestg->stg_sidecar; sidecar;
318 sidecar = (STG *)sidecar->stg_sidecar) {
319 if ((STG *)sidecar->stg_sidecar == stg) {
320 sidecar->stg_sidecar = stg->stg_sidecar;
321 break;
322 }
323 }
324 if (!sidecar) {
325 sidecar_not_found("stg_delete_sidecar", basestg, stg);
326 }
327 }
328 stg_delete(stg);
329} /* stg_delete_sidecar */

331/*
332 * reserve next n elements at stg_avail; increment stg_avail;
333 * grow, if necessary;
334 * clear newly allocated elements; return the first such element.
335 */
336int
337stg_next(STG *stg, int n)
338{
339 STG *thisstg;
340 unsigned int r = stg->stg_avail;
341 if (n == 0)
342 return 0;
343 if (n < 0) {
344 interrf(ERR_Fatal, "stg_next(%s,%d) called with n < 0", stg->stg_name, n);
345 return 0;
346 }
347 /* if the compiler has recycled some previously allocated space,
348 * we need to reset the stg_cleared region */
349 if (stg->stg_cleared > r)
350 stg->stg_cleared = r;
351 stg->stg_avail += n;
352 for (thisstg = (STG *)stg->stg_sidecar; thisstg;
353 thisstg = (STG *)thisstg->stg_sidecar) {
354 thisstg->stg_avail = stg->stg_avail;
355 thisstg->stg_cleared = stg->stg_cleared;
356 }
357 if (stg->stg_avail > stg->stg_size) {
358 stg_need(stg);
359 } else {
360 stg_clear(stg, stg->stg_cleared, stg->stg_avail - stg->stg_cleared);
361 }
362 return r;
363} /* stg_next */

375static char*
376freefield(STG* stg, BIGUINT64 r)
377{
378 char *base;
379 /* get stg_base */
380 base = (char *)stg->stg_base;
381 /* add the offset of the r'th element (r*dtsize) */
382 base += r * stg->stg_dtsize;
383 /* add freelink offset */
384 base += stg->stg_freelink_offset;
385 return base;
386} /* freefield */
387
388/*
389 * get next element from the free list, if it's not null.
390 * reset the free list from the free list link.
391 * otherwise, just get the next available element from stg_avail
392 * the link to the next free element is stored at 'word 0' of the structure
393 */
394int
395stg_next_freelist(STG *stg)
396{
397 int r = stg->stg_free;
398 if (!r) {
399 r = stg_next(stg, 1);
400 stg_clear(stg, r, 1);
401 } else {
402 char *base;
403 if (stg->stg_dtsize < sizeof(int))
404 too_small_for_freelist("stg_next_freelist", stg);
405 /* get freelink for entry 'r' */
406 base = freefield(stg, r);
407 /* move stg_free to the next free element */
408 stg->stg_free = *(int *)base;
409 /* clear the new element */
410 stg_clear(stg, r, 1);
411 }
412 return r;
413} /* stg_next_freelist */

415/*
416 * return latest entry (from stg_next)
417 */
418void
419stg_return(STG *stg)
420{
421 STG *thisstg;
422 unsigned int r = stg->stg_avail - 1;
423 stg->stg_avail = r;
424 if (stg->stg_cleared > r)
425 stg->stg_cleared = r;
426 for (thisstg = (STG *)stg->stg_sidecar; thisstg;
427 thisstg = (STG *)thisstg->stg_sidecar) {
428 thisstg->stg_avail = stg->stg_avail;
429 thisstg->stg_cleared = stg->stg_cleared;
430 }
431} /* stg_return */

433/*
434 * add element to the free list
435 * store the link to the next free element at 'word 0'
436 */
437void
438stg_add_freelist(STG *stg, int r)
439{
440 char *base;
