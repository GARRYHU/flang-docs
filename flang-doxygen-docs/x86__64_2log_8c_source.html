283 /*
284 First, we decompose the argument x to the form
285 x = 2**M * (F1 + F2),
286 where 1 <= F1+F2 < 2, M has the value of an integer,
287 F1 = 1 + j/64, j ranges from 0 to 64, and |F2| <= 1/128.
288
289 Second, we approximate log( 1 + F2/F1 ) by an odd polynomial
290 in U, where U = 2 F2 / (2 F2 + F1).
291 Note that log( 1 + F2/F1 ) = log( 1 + U/2 ) - log( 1 - U/2 ).
292 The core approximation calculates
293 Poly = [log( 1 + U/2 ) - log( 1 - U/2 )]/U - 1.
294 Note that log(1 + U/2) - log(1 - U/2) = 2 arctanh ( U/2 ),
295 thus, Poly = 2 arctanh( U/2 ) / U - 1.
296
297 It is not hard to see that
298 log(x) = M*log(2) + log(F1) + log( 1 + F2/F1 ).
299 Hence, we return Z1 = log(F1), and Z2 = log( 1 + F2/F1).
300 The values of log(F1) are calculated beforehand and stored
301 in the program.
302 */
303
304 f = x;
305
306 /* Store the exponent of x in xexp and put
307 f into the range [0.5,1) */
308 xexp = (int)((ux & EXPBITS_DP64) >> EXPSHIFTBITS_DP64) - EXPBIAS_DP64;
309 PUT_BITS_DP64((ux & MANTBITS_DP64) | HALFEXPBITS_DP64, f);
310
311 /* Now x = 2**xexp * f, 1/2 <= f < 1. */
312
313 /* Set index to be the nearest integer to 128*f */
314 /*
315 r = 128.0 * f;
316 index = (int)(r + 0.5);
317 */
318 /* This code instead of the above can save several cycles.
319 It only works because 64 <= r < 128, so
320 the nearest integer is always contained in exactly
321 7 bits, and the right shift is always the same. */
322 index = (int)((((ux & 0x000fc00000000000) | 0x0010000000000000) >> 46) +
323 ((ux & 0x0000200000000000) >> 45));
324 z1 = ln_lead_table[index - 64];
325 q = ln_tail_table[index - 64];
326 f1 = index * 0.0078125; /* 0.0078125 = 1/128 */
327 f2 = f - f1;
328 /* At this point, x = 2**xexp * ( f1 + f2 ) where
329 f1 = j/128, j = 64, 65, ..., 128 and |f2| <= 1/256. */
330
331 /* Calculate u = 2 f2 / ( 2 f1 + f2 ) = f2 / ( f1 + 0.5*f2 ) */
332 /* u = f2 / (f1 + 0.5 * f2); */
333 u = f2 / (f1 + 0.5 * f2);
334
335 /* Here, |u| <= 2(exp(1/16)-1) / (exp(1/16)+1).
336 The core approximation calculates
337 poly = [log(1 + u/2) - log(1 - u/2)]/u - 1 */
338 v = u * u;
339 poly = (v * (cb_1 + v * cb_2));
340 z2 = q + (u + u * poly);
341
342 /* Now z1,z2 is an extra-precise approximation of log(f).
343 Add xexp * log(2) to z1, z2 to get the result log(x). */
344
345 r = xexp * log2 + z1 + z2;
346#if defined(COMPILING_LOG10)
347 return (float)(log10e * r);
348#elif defined(COMPILING_LOG2)
349 return (float)(log2e * r);
350#else
351 return (float)r;
352#endif
