42/** \brief Function called from within fiolib to open a file.
43 */
44int
45__fortio_open(int unit, int action_flag, int status_flag, int dispose_flag,
46 int acc_flag, int blank_flag, int form_flag, int delim_flag,
47 int pos_flag, int pad_flag, __INT8_T reclen, char *name,
48 __CLEN_T namelen)
49{
50 char *q;
51 const char *perms;
52 char bfilename[MAX_NAMELEN + 1];
53 char *filename;
54 int long_name;
55 FILE *lcl_fp; /* local file fp */
56 FIO_FCB *f; /* local file control block ptr */
57 __CLEN_T i;
58 int fd;
59
60 if (ILLEGAL_UNIT(unit))
61 return __fortio_error(FIO_EUNIT);
62
63 lcl_fp = NULL;
64 f = NULL;
65
66 /* ------- if file name specified, delete trailing blanks, copy
67 into filename buffer and null-terminate it: */
68
69 long_name = 0;
70 filename = bfilename;
71
72#undef EXIT_OPEN
73#define EXIT_OPEN(ec) \
74 { \
75 if (long_name) \
76 free(filename); \
77 return ec; \
78 }
79
80 if (name != NULL) {
81 /* remove trailing blanks: */
82 while (namelen > 0 && name[namelen - 1] == ' ')
83 --namelen;
84 /* remove preceding blanks: */
85 while (namelen > 0 && name[0] == ' ')
86 name++, namelen--;
87 if (namelen <= 0)
88 return __fortio_error(FIO_ENAME);
89
90 if (namelen > MAX_NAMELEN) {
91 filename = malloc(namelen + 1);
92 long_name = 1;
93 }
94 for (i = 0; i < namelen; i++)
95 filename[i] = name[i];
96 filename[namelen] = '\0';
97#if defined(_WIN64)
98 if (filename[0] == '/' && filename[1] == '/' && filename[3] == '/') {
99 /* convert posix format to win32 format */
100 filename[0] = filename[2]; /* drive letter */
101 filename[1] = ':';
102 filename[2] = '\\';
103 strcpy(filename + 3, filename + 4);
104 namelen--;
105 }
106#endif
107 /* check that file is not already connected to different unit: */
108 for (f = fioFcbTbls.fcbs; f; f = f->next)
109 if (f->named && strcmp(filename, f->name) == 0)
110 if (unit != f->unit)
111 EXIT_OPEN(__fortio_error(FIO_EOPENED))
112 }
113
114 /* ------- handle situation in which unit is already connected: */
115
116 f = __fortio_find_unit(unit);
117
118 if (f != NULL) {
119 if (name == NULL || strcmp(filename, f->name) == 0) {
120 /* case 1: file to be connected is the same: */
121 /* make sure no specifier other than BLANK is different than
122 the one currently in effect */
123
124 if (status_flag == FIO_SCRATCH && f->status != FIO_SCRATCH)
125 EXIT_OPEN(__fortio_error(FIO_ECOMPAT))
126 if (acc_flag != f->acc || form_flag != f->form)
127 EXIT_OPEN(__fortio_error(FIO_ECOMPAT))
128 if (acc_flag == FIO_DIRECT && reclen != (f->reclen / f->wordlen))
129 EXIT_OPEN(__fortio_error(FIO_ECOMPAT))
130
131 f->blank = blank_flag;
132 if (pos_flag == FIO_REWIND) {
133 __io_fseek(f->fp, (seekoffx_t)0L, SEEK_SET);
134 } else if (pos_flag == FIO_APPEND) {
135 __io_fseek(f->fp, (seekoffx_t)0L, SEEK_END);
136 }
137 f->reclen = reclen * f->wordlen;
138
139 Fcb = f; /* save pointer to the fcb for any augmented opens */
140 EXIT_OPEN(0) /* no error occurred */
141 } else {
142 /* case 2: file to be connected is NOT the same: */
143 if (__fortio_close(f, 0 /*dispose flag*/) != 0)
144 EXIT_OPEN(ERR_FLAG)
145 }
146 }
147
148 /* ------- create default name if none specified: */
149
150 if (name == NULL) {
151 /* for unnamed unit, first check environment variable */
152 sprintf(filename, "FOR%03d", unit);
153 if ((q = __fort_getenv(filename)) != NULL)
154 strcpy(filename, q);
155 else if (status_flag != FIO_SCRATCH)
156 sprintf(filename, GET_FIO_CNFG_DEFAULT_NAME, unit);
157 else { /* unnamed SCRATCH file: */
158 while (1) {
159 __fortio_scratch_name(filename, unit);
160 fd = open(filename, O_RDWR | O_CREAT | O_TRUNC | O_EXCL, 0666);
161 if (fd == -1) {
162 continue;
163 }
164 close(fd);
165 break;
166 }
167 }
168 namelen = strlen(filename);
169 }
170
171 /* ******************************************************************
172 procede with opening of new file based on value of STATUS:
173 *******************************************************************/
174
175 if (status_flag == FIO_OLD) {
176 /* if OLD and doesn't exist, then error */
177 if (__fort_access(filename, 0) != 0)
178 EXIT_OPEN(__fortio_error(FIO_ENOEXIST))
179
180/* open file for readonly or read/write: */
181
182 perms = "r+";
183 if ((action_flag == FIO_READ) ||
184 (lcl_fp = __io_fopen(filename, perms)) == NULL) {
185 perms = "r";
186 if ((lcl_fp = (__io_fopen(filename, perms))) == NULL)
187 EXIT_OPEN(__fortio_error(__io_errno()))
188 }
189 } else if (status_flag == FIO_NEW) {
190 /* if NEW and exists then error */
191 if (__fort_access(filename, 0) == 0)
192 EXIT_OPEN(__fortio_error(FIO_EEXIST))
193
194 perms = "w+";
195 if ((lcl_fp = __io_fopen(filename, perms)) == NULL)
196 EXIT_OPEN(__fortio_error(__io_errno()))
197 } else if (status_flag == FIO_REPLACE) {
198/* if file does not exist, create a file;
199 * if file exists, delete the file and create a new file.
200 */
201 perms = "w+";
202 if ((lcl_fp = __io_fopen(filename, perms)) == NULL)
203 EXIT_OPEN(__fortio_error(__io_errno()))
204 } else if (status_flag == FIO_UNKNOWN) {
205 i = 0;
206 if (__fort_access(filename, 0) == 0) { /* file exists */
207 perms = "r+";
208 i = 1;
209 } else /* file does not exist */
210 perms = "w+";
211
212 if ((lcl_fp = __io_fopen(filename, perms)) == NULL) {
213 if (i == 0) /* file does not exist */
214 EXIT_OPEN(__fortio_error(__io_errno()))
215/* try again with different mode: */
216 perms = "r";
217 if ((lcl_fp = __io_fopen(filename, perms)) == NULL)
218 EXIT_OPEN(__fortio_error(__io_errno()))
219 }
220 } else {
221 assert(status_flag == FIO_SCRATCH);
222 perms = "w+";
223 if ((lcl_fp = __io_fopen(filename, perms)) == NULL) {
224 EXIT_OPEN(__fortio_error(__io_errno()))
225 }
226 __fort_unlink(filename);
227 }
228
229 /* ****************************************************************
230 allocate entry for file just opened and assign the
231 characteristics to the file:
232 ***************************************************************/
233
234 f = __fortio_alloc_fcb();
235
236 f->fp = lcl_fp;
237 assert(lcl_fp != NULL);
238 f->unit = unit;
239 f->action = action_flag;
240 f->status = FIO_OLD;
241 if (status_flag == FIO_SCRATCH)
242 f->status = FIO_SCRATCH;
243 f->delim = delim_flag;
244 f->dispose = dispose_flag;
245 f->blank = blank_flag;
246 f->form = form_flag;
247 f->pad = pad_flag;
248 f->pos = pos_flag;
249 f->wordlen = 1; /* default */
250 if (f->form == FIO_UNFORMATTED) {
251 if (envar_fortranopt != NULL && strstr(envar_fortranopt, "vaxio") != NULL)
252 f->wordlen = 4; /* WHAT */
253 }
254 f->reclen = reclen * f->wordlen;
255 f->nextrec = 1;
256 f->truncflag = FALSE;
257 f->skip = 0;
258 f->ispipe = FALSE;
259 f->nonadvance = FALSE;
260 f->pread = 0;
261 f->pback = 0;
262
263 if (acc_flag == FIO_DIRECT) {
264 /* compute number of records in direct access file: */
265 f->acc = FIO_DIRECT;
266 f->maxrec = 0;
267 if (status_flag == FIO_OLD || status_flag == FIO_UNKNOWN) {
268 seekoffx_t len;
269 if (__io_fseekx(lcl_fp, (seekoffx_t)0L, SEEK_END) != 0)
270 goto free_fcb_err;
271 len = (seekoffx_t)__io_ftellx(lcl_fp);
272 f->maxrec = len / f->reclen;
273 __io_fseek(lcl_fp, (seekoffx_t)0L,
274 SEEK_SET); /* re-position to beginning */
275 }
276 } else {
277 if (acc_flag == FIO_STREAM)
278 f->acc = FIO_STREAM;
279 else
280 f->acc = FIO_SEQUENTIAL;
281 if ((status_flag == FIO_OLD || status_flag == FIO_UNKNOWN) &&
282 pos_flag != FIO_APPEND)
283 f->truncflag = TRUE;
284 if (status_flag != FIO_SCRATCH && __fortio_ispipe(f->fp)) {
285 f->truncflag = FALSE;
286 f->ispipe = TRUE;
287 } else if (pos_flag == FIO_APPEND) /* position file at end of file */
288 if (__io_fseek(lcl_fp, (seekoffx_t)0L, SEEK_END) != 0)
289 goto free_fcb_err;
290 }
291
292 if (status_flag != FIO_SCRATCH)
293 f->named = TRUE;
294 else
295 f->named = FALSE;
296 f->name = STASH(filename);
297 f->coherent = 0;
298 f->eof_flag = FALSE;
299 f->eor_flag = FALSE;
300 f->stdunit = FALSE;
301 f->byte_swap = FALSE;
302 f->native = FALSE;
303 f->binary = FALSE;
304 f->asy_rw = 0; /* init async flags */
305 f->asyptr = (void *)0;
306 f->decimal = FIO_POINT;
307 f->encoding = FIO_DEFAULT;
308 f->round = FIO_COMPATIBLE;
309 f->sign = FIO_PROCESSOR_DEFINED;
310 Fcb = f; /* save pointer to the fcb for any augmented opens */
311
312 EXIT_OPEN(0) /* no error occurred */
313
314free_fcb_err:
315 __fortio_free_fcb(f); /* free up FCB for later use */
316 EXIT_OPEN(__fortio_error(__io_errno()))
317}

322static int
323f90_open(__INT_T *unit, __INT_T *bitv, char *acc_ptr, char *action_ptr,
324 char *blank_ptr, char *delim_ptr, char *name_ptr, char *form_ptr,
325 __INT_T *iostat, char *pad_ptr, char *pos_ptr, __INT8_T *reclen,
326 char *status_ptr, char *dispose_ptr, __CLEN_T acc_siz, __CLEN_T action_siz,
327 __CLEN_T blank_siz, __CLEN_T delim_siz, __CLEN_T name_siz, __CLEN_T form_siz, __CLEN_T pad_siz,
328 __CLEN_T pos_siz, __CLEN_T status_siz, __CLEN_T dispose_siz)
329{
330 int acc_flag, action_flag, delim_flag, form_flag, blank_flag;
331 int pad_flag, pos_flag, status_flag, dispose_flag;
332 __INT8_T tmpreclen;
333 int retv;
334 bool binary;
335
336 __fortio_errinit03(*unit, *bitv, iostat, "OPEN");
337
338 if (name_ptr != NULL) {
339 fioFcbTbls.fname = name_ptr;
340 fioFcbTbls.fnamelen = name_siz;
341 } else {
342 fioFcbTbls.fname = NULL;
343 fioFcbTbls.fnamelen = 0;
344 }
345 binary = FALSE;
346
347 /* -------- check specifiers and set flags appropriately: */
348
349 /* ACCESS: */
350
351 pos_flag = FIO_ASIS; /* default to handle "APPEND" */
352 acc_flag = FIO_SEQUENTIAL; /* default value */
353 if (acc_ptr != NULL) {
354 if (__fortio_eq_str(acc_ptr, acc_siz, "DIRECT")) {
355 acc_flag = FIO_DIRECT;
356 } else if (__fortio_eq_str(acc_ptr, acc_siz, "STREAM")) {
357 acc_flag = FIO_STREAM;
358 } else if (__fortio_eq_str(acc_ptr, acc_siz, "SEQUENTIAL")) {
359 acc_flag = FIO_SEQUENTIAL;
360 } else if (__fortio_eq_str(acc_ptr, acc_siz, "APPEND")) {
361 pos_flag = FIO_APPEND;
362 } else {
363 return __fortio_error(FIO_ESPEC);
364 }
365 }
366
367 /* ACTION: */
368
369 action_flag = FIO_READWRITE; /* default value */
370 if (action_ptr != NULL) {
371 if (__fortio_eq_str(action_ptr, action_siz, "READ"))
372 action_flag = FIO_READ;
373 else if (__fortio_eq_str(action_ptr, action_siz, "WRITE"))
374 action_flag = FIO_WRITE;
375 else if (__fortio_eq_str(action_ptr, action_siz, "READWRITE"))
376 action_flag = FIO_READWRITE;
377 else
378 return __fortio_error(FIO_ESPEC);
379 }
380
381 /* FORM: */
382
383 if (form_ptr != NULL) {
384 if (__fortio_eq_str(form_ptr, form_siz, "FORMATTED"))
385 form_flag = FIO_FORMATTED;
386 else if (__fortio_eq_str(form_ptr, form_siz, "UNFORMATTED"))
387 form_flag = FIO_UNFORMATTED;
388 else if (__fortio_eq_str(form_ptr, form_siz, "BINARY")) {
389 form_flag = FIO_UNFORMATTED;
390 binary = TRUE;
391 } else
392 return __fortio_error(FIO_ESPEC);
393 } else if (acc_flag == FIO_DIRECT || acc_flag == FIO_STREAM)
394 form_flag = FIO_UNFORMATTED;
395 else
396 form_flag = FIO_FORMATTED;
397
398 /* DELIM: */
399
400 delim_flag = FIO_NONE; /* default value */
401 if (delim_ptr != NULL) {
402 if (form_flag != FIO_FORMATTED)
403 return __fortio_error(FIO_ECOMPAT);
404 if (__fortio_eq_str(delim_ptr, delim_siz, "APOSTROPHE"))
405 delim_flag = FIO_APOSTROPHE;
406 else if (__fortio_eq_str(delim_ptr, delim_siz, "QUOTE"))
407 delim_flag = FIO_QUOTE;
408 else if (__fortio_eq_str(delim_ptr, delim_siz, "NONE"))
409 delim_flag = FIO_NONE;
410 else
411 return __fortio_error(FIO_ESPEC);
412 }
413
414 /* BLANK: */
415
416 blank_flag = FIO_NULL; /* default value */
417 if (blank_ptr != NULL) {
418 /* file must be connected for formatted I/O: */
419 if (form_flag != FIO_FORMATTED)
420 return __fortio_error(FIO_ECOMPAT);
421 if (__fortio_eq_str(blank_ptr, blank_siz, "ZERO"))
422 blank_flag = FIO_ZERO;
423 else if (!__fortio_eq_str(blank_ptr, blank_siz, "NULL"))
424 return __fortio_error(FIO_ESPEC);
425 }
426
427 /* PAD: */
428
429 pad_flag = FIO_YES; /* default */
430 if (pad_ptr != NULL) {
431 if (form_flag != FIO_FORMATTED)
432 return __fortio_error(FIO_ECOMPAT);
433 if (__fortio_eq_str(pad_ptr, pad_siz, "YES"))
434 pad_flag = FIO_YES;
435 else if (__fortio_eq_str(pad_ptr, pad_siz, "NO"))
436 pad_flag = FIO_NO;
437 else
438 return __fortio_error(FIO_ESPEC);
439 }
440
441 /* POSITION: */
442
443 /* moved to ACCESS to handle ACCESS=APPEND
444 pos_flag = FIO_ASIS;
445 */
446 if (pos_ptr != NULL) {
447 if (acc_flag != FIO_SEQUENTIAL && acc_flag != FIO_STREAM)
448 return __fortio_error(FIO_ECOMPAT);
449 if (__fortio_eq_str(pos_ptr, pos_siz, "ASIS"))
450 pos_flag = FIO_ASIS;
451 else if (__fortio_eq_str(pos_ptr, pos_siz, "REWIND"))
452 pos_flag = FIO_REWIND;
453 else if (__fortio_eq_str(pos_ptr, pos_siz, "APPEND"))
454 pos_flag = FIO_APPEND;
455 else
456 return __fortio_error(FIO_ESPEC);
457 }
458
459 /* STATUS: */
460
461 status_flag = FIO_UNKNOWN; /* default */
462 if (status_ptr != NULL) {
463 if (__fortio_eq_str(status_ptr, status_siz, "OLD"))
464 status_flag = FIO_OLD;
465 else if (__fortio_eq_str(status_ptr, status_siz, "NEW"))
466 status_flag = FIO_NEW;
467 else if (__fortio_eq_str(status_ptr, status_siz, "REPLACE"))
468 status_flag = FIO_REPLACE;
469 else if (__fortio_eq_str(status_ptr, status_siz, "UNKNOWN"))
470 status_flag = FIO_UNKNOWN;
471 else if (__fortio_eq_str(status_ptr, status_siz, "SCRATCH"))
472 status_flag = FIO_SCRATCH;
473 else
474 return __fortio_error(FIO_ESPEC);
475 }
476
477 /* DISPOSE: */
478
479 if (dispose_ptr != NULL) {
480 if (__fortio_eq_str(dispose_ptr, dispose_siz, "KEEP"))
481 dispose_flag = FIO_KEEP;
482 else if (__fortio_eq_str(dispose_ptr, dispose_siz, "SAVE"))
483 dispose_flag = FIO_KEEP;
484 else if (__fortio_eq_str(dispose_ptr, dispose_siz, "DELETE"))
485 dispose_flag = FIO_DELETE;
486 else
487 return __fortio_error(FIO_ESPEC);
488 } else if (status_flag == FIO_SCRATCH)
489 dispose_flag = FIO_DELETE;
490 else
491 dispose_flag = FIO_KEEP;
492
493 /* ------------- Check for compatibility between specifiers: */
494
495 if (get_gbl_newunit() && name_ptr == NULL && status_flag != FIO_SCRATCH) {
496 return __fortio_error(FIO_ENEWUNIT);
497 }
498
499 if (acc_flag == FIO_DIRECT) {
500 tmpreclen = 0;
501 if (reclen)
502 tmpreclen = *reclen;
503 if (tmpreclen < 1)
504 return __fortio_error(FIO_ERECLEN);
505 } else if (acc_flag == FIO_SEQUENTIAL && reclen) {
506 tmpreclen = *reclen;
507 if (tmpreclen < 1)
508 return __fortio_error(FIO_ERECLEN);
509 } else {
510 /* VMS allows RECL with non-DIRECT access. Just ignore reclen. */
511 tmpreclen = 0;
512 if (reclen) {
513 if (acc_flag == FIO_STREAM)
514 return __fortio_error(FIO_ERECL);
515 }
516 }
517
518 if (status_flag == FIO_SCRATCH) {
519 if (dispose_flag == FIO_KEEP)
520 return __fortio_error(FIO_EDISPOSE);
521 if (fioFcbTbls.fname != NULL)
522 return __fortio_error(FIO_ESCRATCH);
523 }
524
525 if (action_flag == FIO_READ) {
526 if (status_flag == FIO_SCRATCH || status_flag == FIO_REPLACE ||
527 dispose_flag == FIO_DELETE)
528 return __fortio_error(FIO_EREADONLY);
529 }
530
531 if ((acc_flag == FIO_STREAM) && (form_flag == FIO_UNFORMATTED))
532 binary = TRUE;
533
534 /* --------- call __fortio_open to complete process of opening file: */
535
536 retv = __fortio_open(*unit, action_flag, status_flag, dispose_flag, acc_flag,
537 blank_flag, form_flag, delim_flag, pos_flag, pad_flag,
538 tmpreclen, fioFcbTbls.fname, fioFcbTbls.fnamelen);
539 if (!retv && binary) {
540 if (acc_flag == FIO_DIRECT)
541 retv = __fortio_error(FIO_ESPEC);
542 else
543 Fcb->binary = TRUE;
544 }
545
546 return retv;
547}

552 */
553__INT_T
554ENTF90IO(OPENA, opena) (
555 __INT_T *unit, /* unit number */
556 __INT_T *bitv, /* determines action if error occurs */
557 DCHAR(acc), /* DIRECT, SEQUENTIAL, or NULL */
558 DCHAR(action), /* READ, WRITE, READWRITE, or NULL */
559 DCHAR(blank), /* ZERO or NULL */
560 DCHAR(delim), /* APOSTROPHE, QUOTE, NONE, or NULL */
561 DCHAR(name), /* file name */
562 DCHAR(form), /* FORMATTED, UNFORMATTED, or NULL */
563 __INT_T *iostat, /* IOSTAT variable */
564 DCHAR(pad), /* YES, NO, or NULL */
565 DCHAR(pos), /* ASIS, REWIND, APPEND, or NULL */
566 __INT_T *reclen, /* record length in bytes or words */
567 DCHAR(status), /* OLD, NEW, SCRATCH, REPLACE, or NULL */
568 DCHAR(dispose) /* KEEP, DELETE, SAVE, or NULL */
569 DCLEN64(acc) /* length of acc */
570 DCLEN64(action) /* length of action */
571 DCLEN64(blank) /* length of blank */
572 DCLEN64(delim) /* length of delim */
573 DCLEN64(name) /* length of name */
574 DCLEN64(form) /* length of form */
575 DCLEN64(pad) /* length of pad */
576 DCLEN64(pos) /* length of pos */
577 DCLEN64(status) /* length of status */
578 DCLEN64(dispose)) /* length of dispose */
579{
580 char *acc_ptr;
581 char *action_ptr;
582 char *blank_ptr;
583 char *delim_ptr;
584 char *name_ptr;
585 char *form_ptr;
586 char *pad_ptr;
587 char *pos_ptr;
588 char *status_ptr;
589 char *dispose_ptr;
590 __CLEN_T acc_siz;
591 __CLEN_T action_siz;
592 __CLEN_T blank_siz;
593 __CLEN_T delim_siz;
594 __CLEN_T name_siz;
595 __CLEN_T form_siz;
596 __CLEN_T pad_siz;
597 __CLEN_T pos_siz;
598 __CLEN_T status_siz;
599 __CLEN_T dispose_siz;
600 __INT8_T newreclen;
601
602 int s = 0;
603
604 acc_ptr = (ISPRESENTC(acc) ? CADR(acc) : NULL);
605 action_ptr = (ISPRESENTC(action) ? CADR(action) : NULL);
606 blank_ptr = (ISPRESENTC(blank) ? CADR(blank) : NULL);
607 delim_ptr = (ISPRESENTC(delim) ? CADR(delim) : NULL);
608 name_ptr = (ISPRESENTC(name) ? CADR(name) : NULL);
609 form_ptr = (ISPRESENTC(form) ? CADR(form) : NULL);
610 pad_ptr = (ISPRESENTC(pad) ? CADR(pad) : NULL);
611 pos_ptr = (ISPRESENTC(pos) ? CADR(pos) : NULL);
612 status_ptr = (ISPRESENTC(status) ? CADR(status) : NULL);
613 dispose_ptr = (ISPRESENTC(dispose) ? CADR(dispose) : NULL);
614 acc_siz = CLEN(acc);
615 action_siz = CLEN(action);
616 blank_siz = CLEN(blank);
617 delim_siz = CLEN(delim);
618 name_siz = CLEN(name);
619 form_siz = CLEN(form);
620 pad_siz = CLEN(pad);
621 pos_siz = CLEN(pos);
622 status_siz = CLEN(status);
623 dispose_siz = CLEN(dispose);
624
625 __fort_status_init(bitv, iostat);
626 newreclen = (__INT8_T)*reclen;
627 if (LOCAL_MODE || GET_DIST_LCPU == GET_DIST_IOPROC)
628 s = f90_open(unit, /* do real open (finally) */
629 bitv, acc_ptr, action_ptr, blank_ptr, delim_ptr, name_ptr,
630 form_ptr, iostat, pad_ptr, pos_ptr, &newreclen, status_ptr,
631 dispose_ptr, acc_siz, action_siz, blank_siz, delim_siz,
