76float FN_PROTOTYPE(mth_i_exp)(float x)
77{
78 static const float max_exp_arg = 8.8722839355E+01F, /* 0x42B17218 */
79 min_exp_arg = -1.0327893066E+02F, /* 0xC2CE8ED0 */
80 thirtytwo_by_log2 = 4.6166240692E+01F, /* 0x4238AA3B */
81 log2_by_32_lead = 2.1659851074E-02F, /* 0x3CB17000 */
82 log2_by_32_tail = 9.9831822808E-07F; /* 0x3585FDF4 */
83
84 float z1, z2, z;
85 int m;
86 unsigned int ux, ax;
87
88 /*
89 Computation of exp(x).
90
91 We compute the values m, z1, and z2 such that
92 exp(x) = 2**m * (z1 + z2), where
93 exp(x) is the natural exponential of x.
94
95 Computations needed in order to obtain m, z1, and z2
96 involve three steps.
97
98 First, we reduce the argument x to the form
99 x = n * log2/32 + remainder,
100 where n has the value of an integer and |remainder| <= log2/64.
101 The value of n = x * 32/log2 rounded to the nearest integer and
102 the remainder = x - n*log2/32.
103
104 Second, we approximate exp(r1 + r2) - 1 where r1 is the leading
105 part of the remainder and r2 is the trailing part of the remainder.
106
107 Third, we reconstruct the exponential of x so that
108 exp(x) = 2**m * (z1 + z2).
109 */
110
111 GET_BITS_SP32(x, ux);
112 if ((ux << 1) == 0) {
113 /* Quick exit if arg == +/- 0.0 */
114 return 1.0;
115 }
116 ax = ux & (~SIGNBIT_SP32);
117
118 if (ax >= 0x42B17218) /* abs(x) >= 88.7... */
119 {
120 if (ax >= 0x7f800000) {
121 /* x is either NaN or infinity */
122 if (ux & MANTBITS_SP32)
123 /* x is NaN */
124 return x + x; /* Raise invalid if it is a signalling NaN */
125 else if (ux & SIGNBIT_SP32)
126 /* x is negative infinity; return 0.0 with no flags */
127 return 0.0;
128 else
129 /* x is positive infinity */
130 return x;
131 }
132 if (x > max_exp_arg)
133 /* Return +infinity with overflow flag */
134 return retval_errno_erange_overflow(x);
135 else if (x < min_exp_arg)
136 /* x is negative. Return +zero with underflow and inexact flags */
137 return retval_errno_erange_underflow(x);
138 }
139
140 /* Handle small arguments separately */
141 if (ax < 0x3c800000) /* abs(x) < 1/64 */
142 {
143 if (ax < 0x32800000) /* abs(x) < 2^(-26) */
144 return 1.0F + x; /* Raises inexact if x is non-zero */
145 else
146 z = (((((((1.0F / 5040) * x + 1.0F / 720) * x + 1.0F / 120) * x +
147 1.0F / 24) *
148 x +
149 1.0F / 6) *
150 x +
151 1.0F / 2) *
152 x +
153 1.0F) *
154 x +
155 1.0F;
156 } else {
157 /* Find m and z such that exp(x) = 2**m * (z1 + z2) */
158
159 splitexpf(x, 1.0F, thirtytwo_by_log2, log2_by_32_lead, log2_by_32_tail, &m,
160 &z1, &z2);
161
162 /* Scale (z1 + z2) by 2.0**m */
163
164 if (m >= EMIN_SP32 && m <= EMAX_SP32)
165 z = scaleFloat_1((float)(z1 + z2), m);
166 else
167 z = scaleFloat_2((float)(z1 + z2), m);
168 }
169 return z;
