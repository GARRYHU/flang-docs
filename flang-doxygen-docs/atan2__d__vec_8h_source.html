11vdouble __attribute__((noinline))
12atan2_d_vec(vdouble const y, vdouble const x) {
13 // Helpful constants:
14 const vdouble pi_3_over_4 = vcast_vd_d(PI_3_OVER_4);
15 const vdouble pi_over_4 = vcast_vd_d(PI_OVER_4);
16 const vdouble pi_2 = vcast_vd_d(PI_OVER_2);
17 const vdouble pi = vcast_vd_d(PI);
18
19 unsigned long long int AbsMask = 0x7FFFFFFFFFFFFFFF;
20 double AbsMask_as_double = *(double *)&AbsMask;
21
22 vdouble yAbs = vreinterpret_vd_vm(
23 vand_vm_vm_vm(vreinterpret_vm_vd(y),
24 vreinterpret_vm_vd(vcast_vd_d(AbsMask_as_double))));
25
26 vdouble xAbs = vreinterpret_vd_vm(
27 vand_vm_vm_vm(vreinterpret_vm_vd(x),
28 vreinterpret_vm_vd(vcast_vd_d(AbsMask_as_double))));
29
30 // We need to check if x is negative, but include -0.0 in this.
31 // We do this by casting to int and 2's complement:
32 vmask xNegative = (vreinterpret_vm_vd(x) < 0);
33
34 vopmask yBigger = vgt_vo_vd_vd(yAbs, xAbs);
35
36#ifdef PERF_USE_TWO_DIVIDES
37 // Different vector sizes seems to prefer different code here, most are
38 // faster with using the 2 divides here, with the exception of avx512 which
39 // is (noticably) faster with just the 1 divide.
40 vdouble xReduced = vsel_vd_vo_vd_vd(yBigger, x / y, y / x);
41#else
42 // This seems to have better performance on avx512, while the other is
43 // better for everything else
44 vdouble xReduced =
45 vsel_vd_vo_vd_vd(yBigger, x, y) / vsel_vd_vo_vd_vd(yBigger, y, x);
46#endif
47
48 // The same Estrin scheme as is used in atan(x):
49 vdouble x2 = xReduced * xReduced;
50 vdouble x4 = x2 * x2;
51 vdouble x8 = x4 * x4;
52 vdouble x16 = x8 * x8;
53
54 // Convert our polynomial constants into vectors:
55 const vdouble D2 = vcast_vd_d(C2);
56 const vdouble D3 = vcast_vd_d(C3);
57 const vdouble D4 = vcast_vd_d(C4);
58 const vdouble D5 = vcast_vd_d(C5);
59 const vdouble D6 = vcast_vd_d(C6);
60 const vdouble D7 = vcast_vd_d(C7);
61 const vdouble D8 = vcast_vd_d(C8);
62 const vdouble D9 = vcast_vd_d(C9);
63 const vdouble D10 = vcast_vd_d(C10);
64 const vdouble D11 = vcast_vd_d(C11);
65 const vdouble D12 = vcast_vd_d(C12);
66 const vdouble D13 = vcast_vd_d(C13);
67 const vdouble D14 = vcast_vd_d(C14);
68 const vdouble D15 = vcast_vd_d(C15);
69 const vdouble D16 = vcast_vd_d(C16);
70 const vdouble D17 = vcast_vd_d(C17);
71 const vdouble D18 = vcast_vd_d(C18);
72 const vdouble D19 = vcast_vd_d(C19);
73 const vdouble D20 = vcast_vd_d(C20);
74
75 // The same Estrin scheme as is used in atan(x):
76 vdouble L1 = vfma_vd_vd_vd_vd(x2, D6, D5);
77 vdouble L2 = vfma_vd_vd_vd_vd(x2, D8, D7);
78 vdouble L3 = vfma_vd_vd_vd_vd(x2, D10, D9);
79 vdouble L4 = vfma_vd_vd_vd_vd(x2, D12, D11);
80 vdouble L5 = vfma_vd_vd_vd_vd(x2, D14, D13);
81 vdouble L6 = vfma_vd_vd_vd_vd(x2, D16, D15);
82 vdouble L7 = vfma_vd_vd_vd_vd(x2, D18, D17);
83 vdouble L8 = vfma_vd_vd_vd_vd(x2, D20, D19);
84
85 // L1 + x4*L2 + x8*L3 + x12*L4 + x16*L5 + x20*L6 + x24*L7 + x28*L8
86 vdouble M1 = vfma_vd_vd_vd_vd(x4, L2, L1);
87 vdouble M2 = vfma_vd_vd_vd_vd(x4, L4, L3);
88 vdouble M3 = vfma_vd_vd_vd_vd(x4, L6, L5);
89 vdouble M4 = vfma_vd_vd_vd_vd(x4, L8, L7);
90
91 // M1 + x8*M2 + x16*M3 + x24*M4
92 // (M1 + x8*M2) + x16*(M3 + x8*M4)
93 vdouble N1 = vfma_vd_vd_vd_vd(x8, M2, M1);
94 vdouble N2 = vfma_vd_vd_vd_vd(x8, M4, M3);
95
96 // D2 + x2*D3 + x4*D4 + x6*(N1 + x16*N2):
97 vdouble poly = vfma_vd_vd_vd_vd(x16, N2, N1);
98
99 poly = vfma_vd_vd_vd_vd(x4, vfma_vd_vd_vd_vd(x2, poly, D4),
100 vfma_vd_vd_vd_vd(x2, D3, D2));
101
102 vdouble result_d = poly;
103
104 // A vdouble that contains 0x8000000000000000:
105 unsigned long long int SignBit = 0x8000000000000000;
106 double SignBit_as_double = *(double *)&SignBit;
107 vdouble SignBit_as_doubleV = vcast_vd_d(SignBit_as_double);
108
109 // pi/2 with the sign of xReduced:
110 vdouble signedPi_2 = vreinterpret_vd_vm(
111 vreinterpret_vm_vd(pi_2) | (vreinterpret_vm_vd(xReduced) &
112 vreinterpret_vm_vd(SignBit_as_doubleV)));
113
114 vdouble pi_factor = vsel_vd_vo_vd_vd(yBigger, signedPi_2, vcast_vd_d(0.0));
115
116 xReduced = vsel_vd_vo_vd_vd(yBigger, -xReduced, xReduced);
117
118 result_d = vfma_vd_vd_vd_vd(x2 * xReduced, poly, xReduced);
119
120 // pi with the sign of y:
121 const vdouble signedPi = vreinterpret_vd_vm(
122 vreinterpret_vm_vd(pi) |
123 (vreinterpret_vm_vd(y) & vreinterpret_vm_vd(SignBit_as_doubleV)));
124
125 pi_factor += vreinterpret_vd_vm(xNegative & vreinterpret_vm_vd(signedPi));
126
127 result_d += pi_factor;
128
129 // We need to mask off some special values, mainly infinities and 0's
130 // Fortunately for all cases we always have (xAbs == yAbs)
131 // Get special return value:
132 vdouble special_return_value = vsel_vd_vo_vd_vd(
133 vlt_vo_vd_vd(x, vcast_vd_d(0.0f)), pi_3_over_4, pi_over_4);
134
135 vdouble special_yx_zero_zero =
136 vreinterpret_vd_vm(xNegative & vreinterpret_vm_vd(pi));
137
138 // Check for (y, x) = (+-0.0, +-0.0)
139 special_return_value =
140 vsel_vd_vo_vd_vd(veq_vo_vd_vd(x, vcast_vd_d(0.0)), special_yx_zero_zero,
141 special_return_value);
142
143 result_d = vsel_vd_vo_vd_vd(veq_vo_vd_vd(yAbs, xAbs), special_return_value,
144 result_d);
145
146 // copysign(special_return_value, y):
147 result_d = vreinterpret_vd_vm(
148 vreinterpret_vm_vd(result_d) |
149 (vreinterpret_vm_vd(y) & vreinterpret_vm_vd(SignBit_as_doubleV)));
150
151 return result_d;
152}
