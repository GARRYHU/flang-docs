17/*
18 * Bit offsets for various X8664 hardware features within X86IDFN(hw_features).
19 * If X86IDFN(hw_features) == 0, the variable is undefined and is
20 * initialized by calling X86IDFN(init_hw_features)().
21 *
22 * X86IDFN(hw_features) is intended to be use by runtime routines that have
23 * different execution paths depending on hardware characteristics. In
24 * particular different SSE and AVX implementations to avoid some processors'
25 * expensive AVX/SSE transition penalties.
26 *
27 * A prototype assembly pseudo code implementation would be:
28 *
29 * #if defined(TARGET_WIN_X8664)
30 * movl ENT(X86IDFN(hw_features))(%rip), %eax
31 * #else
32 * movq ENT(X86IDFN(is_avx_cached))@GOTPCREL(%rip), %rax
33 * movl (%rax), %eax
34 * #endif
35 *
36 * 1:
37 * testl $HW_AVX, %eax
38 * jnz do_avx
39 * testl $HW_SSE, %eax
40 * jnz do_sse // Can't assume do_sse on first pass
41 * subq $8, %rsp // Adjusted for number of local regs to save
42 * movq I1, (%rsp) // Or %xmm0, or I1, %xmm0, or %ymm0, ...
43 * movl %eax, I1W // Input to X86IDFN(init_hw_feature)()
44 * CALL (ENT(X86IDFN(init_hw_features))) // (%eax) = hw_features
45 * movq (%rsp), I1 // And possibly more regs
46 * addq $8, %rsp
47 * jmp 1b // Restart feature tests
48 *
49 * Note: X86IDFN(init_hw_feature)(X86IDFN(hw_features)) will abort if
50 * I1W on entry is the same as the return value.
51 *
52 */
53
