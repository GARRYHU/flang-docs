36{

58static int
59getEnclFunc(SPTR sptr)
60{
61 int currencl;
62 int enclsptr;
63 currencl = enclsptr = ENCLFUNCG(sptr);
64 while (enclsptr && STYPEG(enclsptr) != ST_ENTRY) {
65 currencl = enclsptr;
66 enclsptr = ENCLFUNCG(enclsptr);
67 }
68
69 if (currencl)
70 return SCOPEG(currencl);
71 return 0;
72}

86/** \brief Look for a symbol with same name as first and in an active scope.
87 \param first the symbol to match by name
88 \param overloadclass 0 or the value of stb.ovclass to match
89 \param paliassym return the symbol the result is an alias of
90 \param plevel return the scope nest at which symbol was found
91 \param multiple_use_error if true, report error if name is USE-associated
92 from two different modules
93 \return symbol (or alias) if found, else 0
94 */
95int
96sym_in_scope(int first, OVCLASS overloadclass, int *paliassym, int *plevel,
97 int multiple_use_error)
98{
99 SPTR sptrloop, bestsptr, bestsptrloop, cc_alias;
100 int bestsl, bestuse, bestuse2, bestusecount, bestuse2count;
101
102 // Construct names have their own OC_CONSTRUCT overloading class (name
103 // space), but are actually in the OC_OTHER class along with variables names
104 // and a number of other names. The compiler should flag multiple uses of
105 // a name in this class, even though it looks benign to allow overloading
106 // between construct names and other names in this class in most cases.
107 // Another consideration is that construct scoping information should
108 // (probably) also be taken into account, even though that information is
109 // not always available for symbols here. As a compromise to exact error
110 // reporting, look for name clashes of this sort primarily (but not
111 // exclusively) for symbols at the top construct scope level, to reduce
112 // scoping complications. As with prior code without an alt_overloadclass
113 // check, this continues to permit both false positives and false negatives.
114 // These should be rare and easy to work around by not overloading names.
115 OVCLASS alt_overloadclass =
116 overloadclass == OC_CONSTRUCT && sem.doif_depth == 0 ? OC_OTHER : 0;
117
118 if (paliassym)
119 *paliassym = 0;
120 if (plevel)
121 *plevel = 0;
122 bestsptr = bestsptrloop = 0;
123 bestuse = bestuse2 = bestusecount = bestuse2count = 0;
124 bestsl = -1;
125 for (sptrloop = first_hash(first); sptrloop; sptrloop = HASHLKG(sptrloop)) {
126 int want_scope, usecount, sptrlink;
127 SCOPESTACK *scope;
128 int sptr = sptrloop;
129 if (NMPTRG(sptr) != NMPTRG(first))
130 continue;
131 switch (STYPEG(sptr)) {
132 case ST_ISOC:
133 case ST_CRAY:
134 /* predefined symbol, but not active in this routine */
135 continue;
136 case ST_MODPROC:
137 case ST_PROC:
138 case ST_IDENT:
139 case ST_VAR:
140 case ST_ARRAY:
141 case ST_STRUCT:
142 case ST_UNION:
143 case ST_DESCRIPTOR:
144 case ST_TYPEDEF:
145 if (HIDDENG(sptr))
146 continue;
147 /* make sure it is in current function scope */
148 if (gbl.internal > 1 && SCG(sptr) == SC_PRIVATE && ENCLFUNCG(sptr)) {
149 if (!isLocalPrivate(sptr))
150 continue;
151 }
152 break;
153 default:;
154 }
155 if (sem.scope_stack == NULL) {
156 /* must be after the parser, such as in static-init */
157 if (overloadclass == 0 || STYPEG(sptr) == ST_UNKNOWN ||
158 stb.ovclass[STYPEG(sptr)] == overloadclass) {
159 if (STYPEG(sptr) == ST_ALIAS)
160 sptr = SYMLKG(sptr);
161 if (paliassym != NULL)
162 *paliassym = sptrloop;
163 if (plevel != NULL)
164 *plevel = -1;
165 return sptr;
166 }
167 continue;
168 }
169
170 /* in a current scope? */
171 want_scope = SCOPEG(sptr);
172 if (want_scope == 0 && STYPEG(sptr) == ST_MODULE) {
173 /* see if there is a USE clause for this module, use that level */
174 SCOPESTACK *scope = next_scope_kind_sptr(0, SCOPE_USE, sptr);
175 if (scope != 0) {
176 want_scope = scope->sptr; /* treat module as scoped at itself */
177 }
178 }
179 if (want_scope == 0) {
180 if (STYPEG(sptr) == ST_ALIAS)
181 sptr = SYMLKG(sptr);
182 if (bestsl == -1) {
183 bestsl = 0;
184 bestsptr = sptr;
185 bestsptrloop = sptrloop;
186 }
187 continue;
188 }
189 cc_alias = 0;
190 if (STYPEG(sptr) == ST_ALIAS && DCLDG(sptr) &&
191 NMPTRG(sptr) == NMPTRG(SYMLKG(sptr))) {
192 /* from a 'use module, only: i' statement;
193 * compiler inserts an alias 'i' in this scope,
194 * but the alias in this scope has no meaning; look at the
195 * original symbol 'i'.
196 * This is very different from 'use module, only: j=>i',
197 * where the alias 'j' in this scope DOES have meaning
198 *
199 * But (fs16195), do keep track of alias as an additional
200 * check of the except list.
201 */
202 cc_alias = sptr;
203 sptr = SYMLKG(sptr);
204 }
205 sptrlink = sptr;
206 while ((STYPEG(sptrlink) == ST_ALIAS || STYPEG(sptrlink) == ST_MODPROC) &&
207 SYMLKG(sptrlink)) {
208 sptrlink = SYMLKG(sptrlink);
209 }
210 usecount = 0;
211 /* look in the scope stack for an active scope containing this symbol */
212 scope = 0;
213 while ((scope = next_scope(scope)) != 0) {
214 /* past a SCOPE_NORMAL, association is HOST-association,
215 * not USE-association */
216 if (scope->kind == SCOPE_NORMAL)
217 ++usecount;
218 if (scope->sptr == want_scope ||
219 /* module procedures are 'scoped' at module level.
220 * treat as if they are scoped here */
221 scope->sptr == sptrloop ||
222 (scope->sptr && want_scope < stb.stg_avail &&
223 scope->sptr == find_explicit_interface(want_scope))) {
224 LOGICAL found = is_except_in_scope(scope, sptr) ||
225 is_except_in_scope(scope, cc_alias);
226 if (scope->Private &&
227 ((STYPEG(sptr) != ST_PROC && STYPEG(sptr) != ST_OPERATOR &&
228 STYPEG(sptr) != ST_USERGENERIC) ||
229 (!VTOFFG(sptr) && !TBPLNKG(sptr)) ||
230 (IS_TBP(sptr) && PRIVATEG(sptr)))) {
231 found = TRUE; /* in a private USE */
232 } else if (scope->kind == SCOPE_USE &&
233 (PRIVATEG(sptr) ||
234 PRIVATEG(sptrloop))) {
235 /* FE creates an alias when processing the case like:
236 'use mod_name, only : i'.
237 So, if found sptrloop is a type of ST_ALIAS, we need to check whether
238 current module is a submod of ENCLFUNCG(sptrloop). If yes, then this
239 variable is accessible.
240 */
241 if (STYPEG(sptrloop) == ST_ALIAS && ANCESTORG(gbl.currmod))
242 found = ENCLFUNCG(sptrloop) != ANCESTORG(gbl.currmod);
243 else
244 found = TRUE; /* private module variable */
245 /* private module variables are visible to inherited submodules*/
246 if (is_used_by_submod(gbl.currsub, sptr))
247 return sptr;
248 }
249 if (!found) { /* not found in 'except' list */
250 if (STYPEG(sptr) == ST_ALIAS)
251 sptr = SYMLKG(sptr);
252 if (overloadclass == 0 || STYPEG(sptrlink) == ST_UNKNOWN ||
253 stb.ovclass[STYPEG(sptrlink)] == overloadclass ||
254 (stb.ovclass[STYPEG(sptrlink)] == alt_overloadclass &&
255 !CONSTRUCTSYMG(sptrlink))) {
256 int sl = get_scope_level(scope);
257 if (sl > bestsl &&
258 (scope->kind != SCOPE_BLOCK || sptr >= scope->symavl)) {
259 if (scope->kind == SCOPE_USE &&
260 STYPEG(sptrlink) != ST_USERGENERIC &&
261 STYPEG(sptrlink) != ST_ENTRY && !VTOFFG(sptrlink) &&
262 !TBPLNKG(sptrlink)) {
263 if (bestuse && bestuse2 == 0 && bestsptr != sptrlink) {
264 bestuse2 = bestuse;
265 bestuse2count = bestusecount;
266 }
267 bestuse = scope->sptr;
268 bestusecount = usecount;
269 } else {
270 bestuse = 0;
271 }
272 bestsl = sl;
273 bestsptr = sptrlink;
274 bestsptrloop = sptrloop;
275 } else if (bestuse && scope->kind == SCOPE_USE &&
276 /* for submodule, use-association overwrites host-association*/
277 STYPEG(scope->sptr) == ST_MODULE &&
278 ANCESTORG(gbl.currmod) != scope->sptr &&
279 scope->sptr != bestuse &&
280 STYPEG(sptrlink) != ST_USERGENERIC &&
281 STYPEG(sptrlink) != ST_ENTRY && !VTOFFG(sptrlink) &&
282 !TBPLNKG(sptrlink) && bestsptr != sptrlink) {
283 bestuse2 = scope->sptr;
284 bestuse2count = usecount;
285 }
286 }
287 }
288 }
289 if (scope->closed && scope->kind != SCOPE_INTERFACE) {
290 if (!bestsptr && scope->kind == SCOPE_NORMAL && scope->import) {
291 if (sym_in_sym_list(sptr, scope->import)) {
292 if (STYPEG(sptr) == ST_ALIAS)
293 sptr = SYMLKG(sptr);
294 return sptr;
295 }
296 }
297 break; /* can't go farther out anyway */
298 }
299 }
300 }
301
302 if (bestuse && bestuse2 && multiple_use_error && bestuse != bestuse2 &&
303 !isSameNameGenericOrProcOrModproc(bestsptr, bestsptrloop) &&
304 bestusecount == bestuse2count && sem.which_pass == 1) {
305 /* oops; this name is USE-associated from two
306 * different modules */
307 char msg[200];
308 sprintf(msg,
309 "is use-associated from modules %s and %s,"
310 " and cannot be accessed",
311 SYMNAME(bestuse), SYMNAME(bestuse2));
312 error(155, 3, gbl.lineno, SYMNAME(first), msg);
313 }
314 if (paliassym != NULL)
315 *paliassym = bestsptrloop;
316 if (plevel != NULL)
317 *plevel = bestsl;
318 return bestsptr;
319} /* sym_in_scope */

321/** \brief IMPORT symbol from host scope -- not to be confused with interf
322 import stuff.
323 */
324void
325sem_import_sym(int s)
326{
327 int sptr;
328 int smi;
329 SCOPESTACK *scope;
330
331 sptr = find_in_host(s);
332 while (sptr > NOSYM && STYPEG(sptr) == ST_ALIAS && SYMLKG(sptr) > NOSYM &&
333 strcmp(SYMNAME(sptr), SYMNAME(SYMLKG(sptr))) == 0)
334 sptr = SYMLKG(sptr); /* FS#17251 - need to resolve alias */
335 if (sptr <= NOSYM) {
336 error(155, 3, gbl.lineno, "Cannot IMPORT", SYMNAME(s));
337 return;
338 }
339 /*
340 * <zero or more> SCOPE_USE
341 */
342 scope = next_scope_kind(0, SCOPE_NORMAL);
343 smi = add_symitem(sptr, scope->import);
344 scope->import = smi;
345}

347/*
348 * The current context is:
349 * interface
350 * ...
351 * subroutine/function ...
352 * INPORT FROMHOST <<< current context, find FROMHOST>>>>
353 * endsubroutine/endfunction
354 * ...
355 * endinterface
356 *
357 * There should be three scope entries corresponding to this context:
358 *
359 * scope_level-2 : SCOPE_INTERFACE
360 * scope_level-1 : SCOPE_NORMAL
361 * <zero or more> SCOPE_USE
362 * scope_level : SCOPE_SUBPROGRAM
363 *
364 */
365static int
366find_in_host(int s)
367{
368 int cap;
369 int sptr;
370 SCOPESTACK *scope, *iface_scope;
371
372 /*
373 * First check for the minimal scope entries.
374 */
375 cap = sem.scope_level - 3 * (sem.interface - 1);
376 if (cap < 4)
377 return -1;
378
379 scope = get_scope(cap);
380 if (scope->kind != SCOPE_SUBPROGRAM) {
381 return -1;
382 }
383 scope = next_scope_kind(scope, SCOPE_NORMAL);
384 if (scope == 0 || get_scope_level(scope) < 4) {
385 return -1;
386 }
387
388 iface_scope = next_scope(scope);
389 if (iface_scope->kind != SCOPE_INTERFACE) {
390 return -1;
391 }
392
393 /*
394 * Find symbol suitable for IMPORT from the hash list.
395 */
396 for (sptr = first_hash(s); sptr; sptr = HASHLKG(sptr)) {
397 if (NMPTRG(sptr) != NMPTRG(s))
398 continue;
399 if (stb.ovclass[STYPEG(sptr)] != OC_OTHER)
400 continue;
401 /*
402 * Now, search the scope entries.
403 */
404 /*
405 * Now, search the scope entries starting below the scope for the interface
406 */
407 scope = iface_scope;
408 while ((scope = next_scope(scope)) != 0) {
409 if (scope->sptr == SCOPEG(sptr)) {
410 LOGICAL ex;
411 if (scope->except) {
412 ex = is_except_in_scope(scope, sptr);
413 } else if (scope->Private) {
414 for (ex = scope->only; ex; ex = SYMI_NEXT(ex)) {
415 int sptr2 = SYMI_SPTR(ex);
416 if (sptr2 == sptr)
417 return sptr;
418 /* FS#14811 Check for symbol in GNDSC list. */
419 if (STYPEG(sptr2) == ST_ALIAS)
420 sptr2 = SYMLKG(sptr2);
421 if (sym_in_sym_list(sptr2, GNDSCG(sptr))) {
422 return sptr;
423 }
424 }
425 ex = TRUE; /* in a private USE */
426 } else {
427 ex = FALSE;
428 }
429 if (!ex) /* not on a 'except' list */
430 return sptr;
431 }
432 if (scope->closed)
433 break; /* can't go farther out anyway */
434 }
435 }
436 return -1;
437}

439int
440test_scope(int sptr)
441{
442 int sl;
443 for (sl = sem.scope_level; sl >= 0; --sl) {
444 SCOPESTACK *scope = get_scope(sl);
445 if (scope->sptr == SCOPEG(sptr)) {
446 int ex = is_except_in_scope(scope, sptr);
447 if (scope->Private) {
448 for (ex = scope->only; ex; ex = SYMI_NEXT(ex)) {
449 int sptr2 = SYMI_SPTR(ex);
450 if (sptr2 == sptr)
451 return sl;
452 /* FS#14811 Check for symbol in GNDSC list. */
453 if (STYPEG(sptr2) == ST_ALIAS)
454 sptr2 = SYMLKG(sptr2);
455 if (sym_in_sym_list(sptr2, GNDSCG(sptr))) {
456 return sl;
457 }
458 }
459 ex = 1; /* in a private USE */
460 } else if (scope->kind == SCOPE_USE && scope->sptr != gbl.currmod &&
461 PRIVATEG(sptr)) {
462 ex = 1; /* private module variable */
463 }
464 if (ex == 0) /* not on a 'except' list */
465 return sl;
466 }
467 if (scope->closed)
468 break; /* can't go farther out anyway */
469 }
470 return -1;
471} /* test_scope */

475/** \brief Look up symbol having a specific symbol type.
476
477 If a symbol is found in the same overloading class and has
478 the same symbol type, it is returned to the caller.
479 If a symbol is found in the same overloading class, the action
480 of declref depends on the stype of the existing symbol and
481 value of the argument def:
482 1. if symbol is an unfrozen intrinsic and def is 'd' (define),
483 its intrinsic property is removed and a new symbol is declared,
484 2. if def is 'd', a multiple declaration error occurs, or
485 3. if def is not 'd', an 'illegal use' error occurs
486
487 If an error occurs or a matching symbol is not found, one is
488 created and its symbol type is initialized.
489 */
490int
491declref(int first, SYMTYPE stype, int def)
492{
493 int sptr1, sptr;
494
495 sptr = sym_in_scope(first, stb.ovclass[stype], NULL, NULL, 0);
496 if (sptr) {
497 SYMTYPE st = STYPEG(sptr);
498 if (st == ST_UNKNOWN && sptr == first)
499 goto return1; /* stype not set yet, set it */
500 if ((int)SCOPEG(sptr) != stb.curr_scope && def == 'd')
501 goto return0;
502 if (stype != st) {
503 if (def == 'd') {
504 /* Redeclare of intrinsic symbol is okay unless frozen */
505 if (IS_INTRINSIC(st)) {
506 if ((sptr1 = newsym(sptr)) != 0)
507 sptr = sptr1;
508 goto return1;
509 }
510 /* multiple declaration */
511 error(44, 3, gbl.lineno, SYMNAME(first), CNULL);
512 } else
513 /* illegal use of symbol */
514 error(84, 3, gbl.lineno, SYMNAME(first), CNULL);
515 goto return0;
516 }
517 goto return2; /* found, return it */
518 }
519return0:
520 sptr = insert_sym(first); /* create new one if def or illegal use */
521return1:
522 STYPEP(sptr, stype);
523 SCOPEP(sptr, stb.curr_scope);
524 if (!sem.interface)
525 IGNOREP(sptr, 0);
526return2:
527 if (flg.xref)
528 xrefput(sptr, def);
529 return sptr;
530}

532/* If we see an ENTRY in a module with the same name as a variable
533 * in the module, we must change the variable into an ENTRY.
534 * We must remove the variable from the module common
535 * (actually, to simplify things, we replace it with another variable)
536 * and change the sptr to be an ENTRY. We can't add another ENTRY to
537 * the end, because postprocessing of the symbols added by this subprogram
538 * assumes that all new symbols are undeclared in the module specification
539 * part, and changes things like the PRIVATE/PUBLIC bit. */
540static int
541replace_variable(int sptr, SYMTYPE stype)
542{
543 int newsptr;
544 ACCL *accessp;
545 newsptr = insert_sym(sptr);
546 STYPEP(newsptr, stype);
547 DTYPEP(newsptr, DTYPEG(sptr));
548 /* add 'private' or 'public' for this symbol */
549 accessp = (ACCL *)getitem(3, sizeof(ACCL));
550 accessp->next = sem.accl.next;
551 sem.accl.next = accessp;
552 accessp->sptr = newsptr;
553 accessp->oper = ' ';
554 if (PRIVATEG(sptr)) {
555 accessp->type = 'v';
556 } else {
557 accessp->type = 'u';
558 }
559 HIDDENP(sptr, 1);
560 module_must_hide_this_symbol(sptr);
561 return newsptr;
562} /* replace_variable */

564static void
565set_internref_flag(int sptr)
566{
567 INTERNREFP(sptr, 1);
568 if (DTY(DTYPEG(sptr)) == TY_ARRAY || POINTERG(sptr) || ALLOCATTRG(sptr) ||
569 IS_PROC_DUMMYG(sptr) || ADJLENG(sptr)) {
570 int descr, sdsc, midnum, devcopy;
571 int cvlen = 0;
572 descr = DESCRG(sptr);
573 sdsc = SDSCG(sptr);
574 midnum = MIDNUMG(sptr);
575 devcopy = DEVCOPYG(sptr);
576 // adjustable char arrays can exist as single vars or array of arrays
577 if (STYPEG(sptr) == ST_VAR || STYPEG(sptr) == ST_ARRAY ||
578 STYPEG(sptr) == ST_IDENT)
579 cvlen = CVLENG(sptr);
580 if (descr)
581 INTERNREFP(descr, 1);
582 if (sdsc)
583 INTERNREFP(sdsc, 1);
584 if (midnum)
585 INTERNREFP(midnum, 1);
586 if (cvlen)
587 INTERNREFP(cvlen, 1);
588 if (devcopy)
589 INTERNREFP(devcopy, 1);
590 }
591 if (DTY(DTYPEG(sptr)) == TY_ARRAY) {
592 ADSC *ad;
593 ad = AD_DPTR(DTYPEG(sptr));
594 if (AD_ADJARR(ad) || ALLOCATTRG(sptr) || ASSUMSHPG(sptr)) {
595 int i, ndim;
596 ndim = AD_NUMDIM(ad);
597 for (i = 0; i < ndim; i++) {
598 internref_bnd(AD_LWAST(ad, i));
599 internref_bnd(AD_UPAST(ad, i));
600 internref_bnd(AD_MLPYR(ad, i));
601 internref_bnd(AD_EXTNTAST(ad, i));
602 }
603 internref_bnd(AD_NUMELM(ad));
604 internref_bnd(AD_ZBASE(ad));
605 }
606 }
607 if (SCG(sptr) == SC_DUMMY && CLASSG(sptr)) {
608 int parent = PARENTG(sptr);
609 if (parent && CLASSG(parent))
610 set_internref_flag(parent);
611 }
612}

635/** \brief Declare a new symbol.
636
637 An error can occur if the symbol is already in the symbol table.<br>
638 If the symbol types match: treat as in error if \a errflg is true; otherwise
639 return the symbol.<br>
640 If they don't match: if symbol is an intrinsic attempt to remove symbol's
641 intrinsic property; otherwise it is an error.
642 */
643int
644declsym(int first, SYMTYPE stype, LOGICAL errflg)
645{
646 SYMTYPE st;
647 int sptr1, sptr, sptralias, oldsptr, level;
648 int symi;
649
650 sptr = sym_in_scope(first, stb.ovclass[stype], &sptralias, &level, 0);
651 if (sptr) {
652 if (STYPEG(sptr) == ST_ENTRY && FVALG(sptr))
653 sptr = FVALG(sptr);
654 st = STYPEG(sptr);
655 if (st == ST_UNKNOWN && sptr == first && gbl.internal &&
656 sptr < stb.firstusym)
657 goto return0; /* New symbol at this scope. */
658 if ((st == ST_UNKNOWN ||
659 (st == ST_MODPROC && !SEPARATEMPG(sptr) && sem.interface)) &&
660 sptr == first && sptr >= stb.firstusym)
661 goto return1; /* Brand new symbol, return it. */
662 if ((int)SCOPEG(sptr) == stb.curr_scope && st == ST_IDENT &&
663 stb.ovclass[st] == stb.ovclass[stype]) {
664 /* Found an ST_IDENT in the same overloading class */
665 goto return1; /* OK (?) */
666 }
667 if (stype == ST_USERGENERIC) {
668 if ((STYPEG(sptr) == ST_PROC || STYPEG(sptr) == ST_MODPROC) &&
669 GSAMEG(sptr)) {

705 oldsptr = sptr;

707 if (st != ST_MODULE && DCLDG(oldsptr)) {
708 DCLDP(sptr, 1);

710 DCLDP(oldsptr, 0);
711 }

713 }

717 sptr = sptr1;
718 goto return1;
719 }
720 }

722 * if so, declare a new symbol */

725 /* declare a new symbol; the level at which
726 * the existing symbol was found is outside the
727 * current scope */

736 * of the ST_ENTRY.

739 STYPEP(sptr, stype);

742 }

744 if (SCOPEG(first) == 0 && stb.curr_scope) {
745 /* host (outer) routine with same-named
746 * identifier in inner scope

749 STYPEP(sptr, stype);
750 SCOPEP(sptr, stb.curr_scope);
751 return sptr;
752 }
753 }
754

756 sptr = insert_sym(first);
757 STYPEP(sptr, stype);
758 SCOPEP(sptr, stb.curr_scope);
759 return sptr;

762 goto return0;
763 }
764 goto return2;
765 }

768 (int)SCOPEG(sptr) == stb.curr_scope) {

770 * argument to the current subprogram; make it an
771 * entry and return it;

786 goto return1;
787 }

789 sptr = sptr1;

791 }
792 if (st == ST_USERGENERIC) {

794 sptr1 = insert_sym(first);
795 if (ENCLFUNCG(sptr) && STYPEG(ENCLFUNCG(sptr)) == ST_MODULE &&
796 ENCLFUNCG(sptr) != gbl.currmod) {
797 /* user generic was from a USE MODULE statement */

810 break;

815 }
816 }
817 if (stype == ST_ENTRY && st == ST_MODPROC && IN_MODULE &&
818 sem.interface == 0 && SYMLKG(sptr) == 0) {
819 sptr1 = insert_sym(first);
820 SYMLKP(sptr, sptr1);
821 export_append_sym(sptr);

823 GSAMEP(sptr1, GSAMEG(sptr));
824 GSAMEP(GSAMEG(sptr), sptr1);
825 }
826 if (PRIVATEG(sptr)) {
827 PRIVATEP(sptr1, 1);
828 }
829 sptr = sptr1;
830 goto return1;
831 }
832 if (stype == ST_ENTRY && STYPEG(sptralias) == ST_ALIAS && sem.mod_sym &&
833 st == ST_PROC && ENCLFUNCG(sptr) == sem.mod_sym) {
834 /* the existing symbol is the interface (ST_PROC) for
835 * a module contained subprogram.
836 */

838 IGNOREP(sptr, 1);

840 sptr = sptralias;
841 goto return1;
842 }
843 if (stype == ST_ENTRY && sptralias == sptr && sem.mod_sym &&
844 st == ST_PROC && ENCLFUNCG(sptr) == sem.mod_sym) {
845 /* the existing symbol is the interface (ST_PROC) for
846 * a module contained subprogram; no ST_ALIAS added
847 * for native-mode.
848 */

850 oldsptr = sptr;
851 /* create new one if def or illegal use */
852 sptr = insert_sym(first);
853 /* make sure this is the first symbol on the hash list */
854 pop_sym(sptr);
855 push_sym(sptr);
856 INMODULEP(sptr, INMODULEG(oldsptr));
857 goto return1;
858 }

861 ENCLFUNCG(sptr) == sem.mod_sym && !INTERNALG(sptr)) {

866 goto return1;
867 }

869 * if so, declare a new symbol */

872 /* declare a new symbol; the level at which
873 * the existing symbol was found is outside the
874 * current scope */
875 goto return0;
876 }
877 }
878 if (stype == ST_ENTRY && st == ST_PROC) {
879 goto return0;
880 }

886 case SCOPE_OUTER:
887 goto return0;
888 case SCOPE_MODULE:
889 if (STYPEG(sptr) == ST_PROC) {
890 oldsptr = sptr;
891 sptr = insert_sym(first); /* create new one */
892 SYMLKP(sptr, oldsptr); /* resolve ST_MODPROC */

894 }
895 break;

897 if (STYPEG(sptr) == ST_PROC) {
898 oldsptr = sptr;
899 sptr = insert_sym(first); /* create new one */
900 SYMLKP(sptr, oldsptr); /* resolve ST_MODPROC */

902 }
903 goto return0;
904 default:;
905 }
906 }
907 /* if we are in a module, creating a module subprogram,
908 * and the old symbol is a 'variable', override the variable. */
909 if (sem.mod_sym && sem.which_pass == 0 && gbl.internal == 0 &&
910 stype == ST_ENTRY && st == ST_VAR) {
911 sptr = replace_variable(sptr, stype);
912 goto return1;
913 }
914 error(43, 3, gbl.lineno, "symbol", SYMNAME(first));
915 }
916
917return0:
918 sptr = insert_sym(first); /* create new one if def or illegal use */
919return1:
920 STYPEP(sptr, stype);
921 SCOPEP(sptr, stb.curr_scope);
922 if (!sem.interface)
923 IGNOREP(sptr, 0);
924return2:

926 xrefput(sptr, 'd');
927#ifdef GSCOPEP
928 if (sem.which_pass && gbl.internal <= 1 &&
929 internal_proc_has_ident(sptr, gbl.currsub)) {
930 GSCOPEP(sptr, 1);
931 }
932#endif
933 if (gbl.internal > 1 && first == sptr) {
934 set_internref_flag(sptr);
935 }
936
937 return sptr;
938}

940/** \brief Look up a symbol having the given overloading class.
941
942 If the symbol with the overloading class is found its sptr is returned. If
943 no symbol with the given overloading class is found, a new sptr is returned.
944 */
945int
946refsym(int first, OVCLASS oclass)
947{
948 int sptr, sl;
949
950 sptr = sym_in_scope(first, oclass, NULL, NULL, 1);
951 if (sptr) {
952 SYMTYPE st = STYPEG(sptr);
953 if (st == ST_UNKNOWN && sptr == first)
954 goto return1;
955 if (stb.ovclass[st] == oclass) {
956 /* was this a reference to the return value? */
957 if (st == ST_ENTRY && !RESULTG(sptr) &&
958 (gbl.rutype == RU_FUNC || (sptr == gbl.outersub && FVALG(sptr)))) {
959 /* always a reference to the result variable */
960 sl = sptr;
961 sptr = ref_entry(sptr);
962 if (FVALG(sl) == sptr) {
963 if (gbl.internal > 1) {
964 set_internref_flag(sptr);
965 }
966 if ((sem.parallel || sem.task || sem.target || sem.teams)) {
967 set_parref_flag(sptr, sptr, BLK_UPLEVEL_SPTR(sem.scope_level));
968 }
969 }
970
971 } else if (SCG(sptr) == SC_DUMMY) {
972 if (gbl.internal > 1) {
973 if (SCOPEG(sptr) && SCOPEG(sptr) == SCOPEG(gbl.currsub))
974 set_internref_flag(sptr);
975 }
976 if ((sem.parallel || sem.task || sem.target || sem.teams)) {
977 set_parref_flag(sptr, sptr, BLK_UPLEVEL_SPTR(sem.scope_level));
978 }
979 }
980 goto returnit;
981 }
982 }
983
984 /* Symbol in given overloading class not found, create new one */
985 sptr = insert_sym(first);
986return1:
987 if (flg.xref)
988 xrefput(sptr, 'd');
989 if (!sem.interface)
990 IGNOREP(sptr, 0);
991returnit:
992 if (gbl.internal > 1 && first == sptr) {
993 if (STYPEG(sptr) == ST_PROC && SCG(sptr) == SC_DUMMY)
994 set_internref_flag(sptr);
995 else if (STYPEG(sptr) != ST_PROC && STYPEG(sptr) != ST_STFUNC)
996 set_internref_flag(sptr);
997 }
998 return sptr;
999}

1001/** \brief Similar to refsym() except that the current scope is taken into
1002 consideration.
1003
1004 If the symbol with the overloading class is found its sptr is returned.
1005 If no symbol with the given overloading class is found, a new sptr is
1006 returned.
1007 */
1008int
1009refsym_inscope(int first, OVCLASS oclass)
1010{
1011 int sptr, level;
1012
1013 sptr = sym_in_scope(first, oclass, NULL, &level, 0);
1014 if (sptr) {
1015 SYMTYPE st = STYPEG(sptr);
1016 /* if this is the symbol just created for this subprogram, use it */
1017 if (st == ST_UNKNOWN && sptr == first && sptr >= stb.firstusym)
1018 goto return1;
1019 if (stb.ovclass[st] == oclass) {
1020 if (gbl.currsub == sptr)
1021 /* && (int)SCOPEG(sptr) == (stb.curr_scope - 1)) */
1022 goto returnit;
1023 /* is this a symbol that was host-associated?
1024 * if so, declare a new symbol */
1025 if (level > 0) {
1026 int sl;
1027 for (sl = sem.scope_level; sl > level; --sl) {
1028 if (sem.scope_stack[sl].kind == SCOPE_NORMAL) {
1029 /* declare a new symbol; the level at which
1030 * the existing symbol was found is outside the
1031 * current scope */
1032 goto return0;
1033 }
1034 }
1035 } else if (level == 0 && st == ST_MODULE &&
1036 sptr == sem.mod_sym /* is the current module */
1037 && sptr != stb.curr_scope /* not in outer host scope */
1038 ) {
1039 /* context is a module which is being defined but not in its
1040 * module specification part -- the symbol is being declared
1041 * in a scope contained within the module.
1042 */
1043 goto return0;
1044 }
1045 if (ENCLFUNCG(sptr) && STYPEG(ENCLFUNCG(sptr)) == ST_MODULE &&
1046 ENCLFUNCG(sptr) != gbl.currmod) {
1047 /* see if the scope level makes this host associated */
1048 if (level < 0)
1049 goto return0;
1050 /* use associated symbol */
1051 if (IGNOREG(sptr) || PRIVATEG(sptr) ||
1052 (st == ST_PROC && PRIVATEG(SCOPEG(sptr))) ||
1053 ((st == ST_USERGENERIC || st == ST_OPERATOR) &&
1054 TBPLNKG(sptr)) /* FS#20696: needed for overloading */
1055 )
1056 goto return0; /* create new symbol */
1057 if (oclass == OC_CMBLK ||
1058 /* Check whether the gbl.currmod and ENCLFUNCG(sptr) share
1059 with the same ancestor, if yes then use host-association
1060 */
1061 (oclass == OC_OTHER &&
1062 (ANCESTORG(gbl.currmod) ?
1063 ANCESTORG(gbl.currmod) : gbl.currmod) ==
1064 (ANCESTORG(ENCLFUNCG(sptr)) ?
1065 ANCESTORG(ENCLFUNCG(sptr)) : ENCLFUNCG(sptr))))
1066 goto return0;
1067 error(155, 3, gbl.lineno, SYMNAME(sptr),
1068 "is use associated and cannot be redeclared");
1069 goto return0;
1070 }
1071 if (gbl.internal > 1 && !INTERNALG(sptr)) {
1072 /* This is a non-internal symbol in an internal subprogram. */
1073 if (IS_INTRINSIC(STYPEG(sptr)))
1074 goto returnit; // tentative intrinsic; may be overridden later
1075 goto return0; // declare a new symbol
1076 }
1077 /*if ((int)SCOPEG(sptr) == stb.curr_scope)*/
1078 goto returnit;
1079 /* break; don't create new symbol */
1080 }
1081 }
1082
1083return0:
1084 /* Symbol in given overloading class not found, create new one */
1085 sptr = insert_sym(first);
1086return1:
1087 SCOPEP(sptr, stb.curr_scope);
1088 if (!sem.interface)
1089 IGNOREP(sptr, 0);
1090 if (flg.xref)
1091 xrefput(sptr, 'd');
1092returnit:
1093 if (gbl.internal > 1 && first == sptr) {
1094 set_internref_flag(sptr);
1095 }
1096 return sptr;
1097}

1176/** \brief Reset fields of intrinsic or generic symbol, sptr, to zero in
1177 preparation for changing its symbol type by the Semantic Analyzer.
1178
1179 If the symbol type of the symbol has been 'frozen', issue an error message
1180 and notify the caller by returning a zero symbol pointer.
1181 */
1182int
1183newsym(int sptr)
1184{
1185 int sp2, sp1;
1186
1187 if (EXPSTG(sptr)) {
1188 /* Symbol previously frozen as an intrinsic */
1189 error(43, 3, gbl.lineno, "intrinsic", SYMNAME(sptr));
1190 return 0;
1191 }
1192 /*
1193 * try to find another sym in the same overloading class; we need to
1194 * try this first since there could be multiple occurrences of an
1195 * intrinsic and therefore the sptr appears more than once in the
1196 * semantic stack. E.g.,
1197 * call sub (sin, sin)
1198 * NOTE that in order for this to work we need to perform another getsym
1199 * to start at the beginning of the hash links for symbols whose names
1200 * are the same.
1201 */
1202 sp1 = getsym(LOCAL_SYMNAME(sptr), strlen(SYMNAME(sptr)));
1203 sp2 = getocsym(sp1, OC_OTHER, FALSE);
1204 if (sp2 != sptr)
1205 return sp2;
1206 /*
1207 * Create a new symbol with the same name:
1208 */
1209 error(35, 1, gbl.lineno, SYMNAME(sptr), CNULL);
1210 sp2 = insert_sym(sp1);
1211
1212 /* transfer dtype if it was explicitly declared for sptr: */
1213
1214 if (DCLDG(sptr)) {
1215 DTYPEP(sp2, DTYPEG(sptr));
1216 DCLDP(sp2, 1);
1217 DCLDP(sptr, 0);
1218 ADJLENP(sp2, ADJLENG(sptr));
1219 ADJLENP(sptr, 0);
1220 }
1221
1222 return sp2;
1223}

1397int
1398ref_based_object_sc(int sptr, SC_KIND sc)
1399{
1400 int sptr1;
1401#if DEBUG
1402 assert(SCG(sptr) == SC_BASED || POINTERG(sptr) || ALLOCATTRG(sptr) ||
1403 (SCG(sptr) == SC_CMBLK && ALLOCG(sptr)),
1404 "ref_based_object: sptr not based", sptr, 3);
1405#endif
1406 if (flg.xref)
1407 xrefput(sptr, 'r');

1409 if (DTY(DTYPEG(sptr)) != TY_ARRAY) {
1410 /* test for scalar pointer */

1412 if (SCG(sptr) == SC_NONE)
1413 SCP(sptr, SC_BASED);
1414 get_static_descriptor(sptr);
1415 get_all_descriptors(sptr);
1416 }

1420 * for an allocatable array, it's possible that the array is not
1421 * associated with a pointer (did not appear in a POINTER statement).
1422 * Create a compiler temporary to represent the pointer variable.
1423 */
1424 if (MIDNUMG(sptr) <= NOSYM && !F90POINTERG(sptr)) {
1425 if (F77OUTPUT) {
1426 sptr1 = sym_get_ptr(sptr);
1427 } else {
1428 sptr1 = getccsym('Z', sptr, ST_VAR);
1429 DTYPEP(sptr1, DT_PTR);
1430 }
1431 MIDNUMP(sptr, sptr1);
1432 /*
1433 * if an allocatable array is saved, need to ensure that all of its
1434 * associated temporary variables are marked save -- e.g., the internal
1435 * pointer variable, its bounds' variables, its zero-base temporary,
1436 * etc.
1437 */
1438 if (SAVEG(sptr) ||
1439 (in_save_scope(sptr) && !CCSYMG(sptr) && !HCCSYMG(sptr))) {
1440 ADSC *ad;
1441 int i, numdim;
1442
1443 SCP(sptr1, SC_STATIC);
1444 SAVEP(sptr1, 1);
1445
1446 if (DTY(DTYPEG(sptr)) == TY_ARRAY) {
1447 ad = AD_PTR(sptr);
1448 numdim = AD_NUMDIM(ad);
1449
1450 mark_static(AD_NUMELM(ad));
1451 mark_static(AD_ZBASE(ad));
1452 for (i = 0; i < numdim; i++) {
1453 mark_static(AD_LWAST(ad, i));
1454 mark_static(AD_UPAST(ad, i));
1455 mark_static(AD_MLPYR(ad, i));
1456 mark_static(AD_EXTNTAST(ad, i));
1457 }
1458 }
1459 }
1460 else if (GSCOPEG(sptr)) {
1461 fixup_reqgs_ident(sptr);
1462 }
1463 else
1464 SCP(sptr1, sc);
1465 }
1466 sptr1 = sptr;
1467 while (TRUE) {
1468 if (STYPEG(sptr1) == ST_IDENT)
1469 STYPEP(sptr1, ST_VAR);
1470 sptr1 = MIDNUMG(sptr1);
1471 if (SAVEG(sptr))
1472 SAVEP(sptr1, 1);
1473 if (flg.xref)
1474 xrefput(sptr1, 'r');
1475 if (SCG(sptr1) != SC_BASED)
1476 break;
1477#if DEBUG
1478 assert(sptr1 > NOSYM, "ref_based_object: bad list", sptr, 0);
1479#endif
1480 }

1482 SCP(sptr1, sc);
1483 if (gbl.internal > 1 && SCOPEG(sptr) == SCOPEG(gbl.currsub)) {
1484 set_internref_flag(sptr);
1485 }
1486 if (flg.smp)
1487 check_parref(sptr, sptr, sptr);
1488 return sptr1;
1489}

1601/** \brief Declare a private symbol which may be based on the attributes of
1602 an existing symbol.
1603
1604 If the symbol doesn't exist (its stype is ST_UNKNOWN), it's assumed that the
1605 private variable will be a scalar variable.
1606 */
1607int
1608decl_private_sym(int sptr)
1609{
1610 int sptr1;
1611 SYMTYPE stype;
1612 char *name;
1613 int new = 0;
1614 int rgn_level;
1615 /*
1616 * First, retrieve the first symbol in the hash list whose name is the same.
1617 * Then, use refsym to retrieve the first symbol whose overloading class
1618 * is the same. This is all necessary because a private symbol could
1619 * have already been created ahead of the existing symbol (sptr).
1620 */
1621 name = SYMNAME(sptr);
1622 sptr1 = getsymbol(name);
1623 sptr = refsym(sptr1, stb.ovclass[STYPEG(sptr)]);
1624 if (SCOPEG(sptr) == sem.scope_stack[sem.scope_level].sptr)
1625 return sptr; /* a variable can appear in more than 1 clause */
1626 if (checking_scope && sem.scope_stack[sem.scope_level].kind == SCOPE_PAR) {
1627 rgn_level = sem.scope_stack[sem.scope_level].rgn_scope;
1628 if (SCOPEG(sptr) == sem.scope_stack[rgn_level].sptr) {
1629 return sptr; /* a variable can appear in more than 1 clause */
1630 }
1631 }
1632 if (ALLOCG(sptr) || POINTERG(sptr)) {
1633 new = insert_sym(sptr1);
1634 STYPEP(new, STYPEG(sptr));
1635 if (DTY(DTYPEG(sptr)) == TY_ARRAY)
1636 DTYPEP(new, dup_array_dtype(DTYPEG(sptr)));
1637 else
1638 DTYPEP(new, DTYPEG(sptr));
1639 ALLOCP(new, ALLOCG(sptr));
1640 POINTERP(new, POINTERG(sptr));
1641 ALLOCATTRP(new, ALLOCATTRG(sptr));
1642 SCP(new, SC_BASED);
1643 set_descriptor_sc(SC_PRIVATE);
1644 get_static_descriptor(new);
1645 get_all_descriptors(new);
1646 new = add_private_allocatable(sptr, new);
1647 set_descriptor_sc(SC_LOCAL);
1648 if (ADJLENG(sptr)) {
1649 int cvlen = CVLENG(sptr);
1650 if (cvlen == 0) {
1651 cvlen = sym_get_scalar(SYMNAME(sptr), "len", DT_INT);
1652 CVLENP(sptr, cvlen);
1653 if (SCG(sptr) == SC_DUMMY)
1654 CCSYMP(cvlen, 1);
1655 }
1656 CVLENP(new, cvlen);
1657 ADJLENP(new, 1);
1658 }
1659 goto return_it;
1660 }
1661 stype = STYPEG(sptr);
1662 switch (stype) {
1663 case ST_UNKNOWN:
1664 new = sptr;
1665 STYPEP(new, ST_VAR);
1666 break;
1667 case ST_IDENT:
1668 case ST_VAR:
1669 new = insert_sym(sptr1);
1670 STYPEP(new, ST_VAR);
1671 DTYPEP(new, DTYPEG(sptr));
1672 if (ADJLENG(sptr)) {
1673 new = add_private_allocatable(sptr, new);
1674 goto return_it;
1675 } else if (ASSUMLENG(sptr)) {
1676 new = add_private_allocatable(sptr, new);
1677 goto return_it;
1678 }
1679 if (allocatable_member(sptr)) {
1680 if (checking_scope && sem.scope_stack[sem.scope_level].end_prologue != 0)
1681 nullify_member_after(
1682 mk_id(new), sem.scope_stack[sem.scope_level].end_prologue, sptr1);
1683 else
1684 nullify_member_after(mk_id(new), STD_PREV(0), sptr1);
1685 }
1686 break;
1687 case ST_STRUCT:
1688 case ST_UNION:
1689 new = insert_sym(sptr1);
1690 STYPEP(new, stype);
1691 DTYPEP(new, DTYPEG(sptr));
1692 break;
1693 case ST_ARRAY:
1694 new = insert_sym(sptr1);
1695 STYPEP(new, ST_ARRAY);
1696 DTYPEP(new, DTYPEG(sptr));
1697 if (SCG(sptr) == SC_DUMMY) {
1698 if (ASUMSZG(sptr))
1699 error(155, 3, gbl.lineno,
1700 "Assumed-size arrays cannot be specified as private",
1701 SYMNAME(sptr));
1702 }
1703 if (SCG(sptr) == SC_BASED && MIDNUMG(sptr) && !CCSYMG(MIDNUMG(sptr)) &&
1704 !HCCSYMG(MIDNUMG(sptr))) {
1705 /* Cray pointee: just copy ADJARR flag (fixes tpr3374) */
1706 ADJARRP(new, ADJARRG(sptr));
1707 } else if (ADJARRG(sptr) || RUNTIMEG(sptr) || ADJLENG(sptr)) {
1708 /*
1709 * The private copy of an adjustable/automatic array is an
1710 * allocated array. The bounds information of the adjustable array
1711 * and its private copy is the same. The private array will
1712 * be allocated from the heap; need to save the sptr of the
1713 * private copy so that it can be deallocated at the end
1714 * of the parallel construct.
1715 */
1716 new = add_private_allocatable(sptr, new);
1717 goto return_it;
1718 } else if (ASSUMSHPG(sptr)) {
1719 /*
1720 * The private copy of an assumed-shape array is an allocated
1721 * array. The bounds information of the assume-shape array
1722 * will be assigned to its private copy. The private array will
1723 * be allocated from the heap; need to save the sptr of the
1724 * private copy so that it can be deallocated at the end
1725 * of the parallel construct.
1726 */
1727 ADSC *ad;
1728 int i, ndim;
1729 int dt;
1730
1731 ad = AD_DPTR(DTYPEG(sptr));
1732 ndim = AD_NUMDIM(ad);
1733 for (i = 0; i < ndim; i++) {
1734 int lb;
1735 lb = AD_LWBD(ad, i);
1736 if (A_ALIASG(lb)) {
1737 sem.bounds[i].lowtype = S_CONST;
1738 sem.bounds[i].lowb = get_isz_cval(A_SPTRG(lb));
1739 } else {
1740 sem.bounds[i].lowtype = S_EXPR;
1741 sem.bounds[i].lowb = lb;
1742 }
1743 sem.bounds[i].lwast = lb;
1744 sem.bounds[i].uptype = S_EXPR;
1745 sem.bounds[i].upb = AD_UPBD(ad, i);
1746 sem.bounds[i].upast = AD_UPBD(ad, i);
1747 }
1748 sem.arrdim.ndim = ndim;
1749 sem.arrdim.ndefer = 0;
1750 dt = mk_arrdsc();
1751 DTY(dt + 1) = DTY(DTYPEG(sptr) + 1);
1752 DTYPEP(new, dt);
1753 new = add_private_allocatable(sptr, new);
1754 goto return_it;
1755 }
1756 break;
1757 default:
1758 sptr = new = insert_sym(sptr1);
1759 STYPEP(new, ST_VAR);
1760 break;
1761 }
1762
1763 if (SCG(sptr) != SC_BASED)
1764 SCP(new, sem.sc);
1765 else {
1766 int stp;
1767 stp = decl_private_sym(MIDNUMG(sptr));
1768 MIDNUMP(new, stp);
1769 SCP(new, SC_BASED);
1770 }
1771 if (sem.task && SCG(new) == SC_PRIVATE) {
1772 int i;
1773 for (i = sem.doif_depth; i; i--) {
1774 switch (DI_ID(i)) {
1775 default:
1776 break;
1777 case DI_TASK:
1778 case DI_TASKLOOP:
1779 TASKP(new, 1);
1780 goto td_exit;
1781 case DI_PAR:
1782 case DI_PARDO:
1783 case DI_PARSECTS:
1784 goto td_exit;
1785 }
1786 }
1787 td_exit:;
1788 }
1789return_it:
1790 if (checking_scope && sem.scope_stack[sem.scope_level].kind == SCOPE_PAR) {
1791 rgn_level = sem.scope_stack[sem.scope_level].rgn_scope;
1792 SCOPEP(new, sem.scope_stack[rgn_level].sptr);
1793 } else
1794 SCOPEP(new, sem.scope_stack[sem.scope_level].sptr);
1795 ENCLFUNCP(new, BLK_SYM(sem.scope_level));
1796 CCSYMP(new, CCSYMG(sptr));
1797 DCLDP(new, 1); /* so DCLCHK is quiet */
1798 TARGETP(new, TARGETG(sptr));
1799 if (flg.smp) {
1800 if (!ENCLFUNCG(new)) {
1801 ENCLFUNCP(new, BLK_SCOPE_SPTR(sem.scope_level));
1802 }
1803 set_private_encl(sptr, new);
1804 if (sem.task && SCG(new) == SC_BASED) {
1805 set_private_taskflag(new);
1806 }
1807 }
1808 return new;
1809}

1827static int
1828add_private_allocatable(int old, int new)
1829{
1830 /*
1831 * The private copy of an adjustable/automatic array is an allocated
1832 * array. The bounds information of the original object and its private
1833 * copy are the same. The private array will be allocated from the heap;
1834 * need to save the sptr of the private copy so that it can be deallocated
1835 * at the end of the parallel construct.
1836 *
1837 * NOTE, need to distinguish:
1838 * 1. allocatables - conditionally allocate/deallocate
1839 * 2. pointer - no allocate/deallocate
1840 * 3. other (adj., automatic) - unconditionally allocate/deallocate
1841 */
1842 ITEM *itemp;
1843 int pvar;
1844 int allo_obj;
1845 int where;
1846
1847 SCP(new, SC_BASED);
1848 if (!POINTERG(old) && !ALLOCATTRG(old)) {
1849 pvar = getccsym('Z', new, ST_VAR);
1850 DTYPEP(pvar, DT_PTR);
1851 SCP(pvar, sem.sc);
1852 SCOPEP(pvar, sem.scope_stack[sem.scope_level].sptr);
1853 ENCLFUNCP(pvar, BLK_SYM(sem.scope_level));
1854 MIDNUMP(new, pvar);
1855 }
1856 if (ADJLENG(old)) {
1857 int cvlen = CVLENG(old);
1858 if (cvlen == 0) {
1859 cvlen = sym_get_scalar(SYMNAME(old), "len", DT_INT);
1860 CVLENP(old, cvlen);
1861 if (SCG(old) == SC_DUMMY)
1862 CCSYMP(cvlen, 1);
1863 }
1864 CVLENP(new, cvlen);
1865 ADJLENP(new, 1);
1866 if (flg.smp) {
1867 if (SCG(old) == SC_BASED)
1868 ref_based_object(old);
1869 set_parref_flag(cvlen, cvlen, BLK_UPLEVEL_SPTR(sem.scope_level));
1870 set_parref_flag(old, old, BLK_UPLEVEL_SPTR(sem.scope_level));
1871 }
1872 } else if (STYPEG(new) != ST_ARRAY && ASSUMLENG(old)) {
1873 /* 1) we don't know the size of assumlen char at compile time
1874 * 2) make private copy adjustable len char
1875 * 3) make CVLEN a private copy for convenience.
1876 */
1877 int ast;
1878 int oldlen = ast_intr(I_LEN, astb.bnd.dtype, 1, mk_id(old));
1879 int cvlen = sym_get_scalar(SYMNAME(old), "len", DT_INT);
1880 ast = mk_assn_stmt(mk_id(cvlen), oldlen, DT_INT);
1881 (void)add_stmt(ast);
1882 CVLENP(new, cvlen);
1883 ADJLENP(new, 1);
1884 SCP(cvlen, SCG(MIDNUMG(new)));
1885 ENCLFUNCP(cvlen, ENCLFUNCG(new));
1886 SCOPEP(cvlen, sem.scope_stack[sem.scope_level].sptr);
1887 if (SCG(new) == SC_DUMMY)
1888 CCSYMP(cvlen, 1);
1889 if (flg.smp) {
1890 if (SCG(old) == SC_BASED)
1891 ref_based_object(old);
1892 set_parref_flag(old, old, BLK_UPLEVEL_SPTR(sem.scope_level));
1893 }
1894 }
1895 allo_obj = mk_id(new); /* base symbol of allocation */
1896 if (STYPEG(new) == ST_ARRAY) {
1897 int dt;
1898 if (ADJARRG(old) || RUNTIMEG(old))
1899 ADJARRP(new, 1);
1900 dt = DTYPEG(new);
1901 if (ASSUMSHPG(old)) {
1902 ADJARRP(new, 1);
1903 ADD_NOBOUNDS(dt) = 1;
1904 }
1905 if (ADD_NOBOUNDS(dt)) {
1906 /*
1907 * an adjustable array with this flag set is an automatic
1908 * array. Need to use the bounds of the array in the allocation
1909 * so that lower() will correctly assign the .A temporaries.
1910 */
1911 int numdim;
1912 int subs[MAXRANK];
1913 int i;
1914 if (ALLOCATTRG(old)) {
1915 /*
1916 * an allocatable inherits its bounds from the original;
1917 * switch to the dtype of the original to get the correct
1918 * bounds.
1919 */
1920 dt = DTYPEG(old);
1921 }
1922 numdim = ADD_NUMDIM(dt);
1923 for (i = 0; i < numdim; i++) {
1924 int lb, ub;
1925
1926 lb = ADD_LWAST(dt, i);
1927 if (!lb)
1928 lb = astb.bnd.one;
1929 ub = ADD_UPAST(dt, i);
1930 if (!ub)
1931 ub = astb.bnd.one;
1932 subs[i] = mk_triple(lb, ub, 0);
1933 }
1934 allo_obj = mk_subscr(allo_obj, subs, numdim, dt);
1935 }
1936 }
1937 if (checking_scope) {
1938 /* If checking_scope (handling variables in a PARALLEL directive
1939 * with a DEFAULT(PRIVATE) or DEFAULT(FIRSTPRIVATE) clause),
1940 * we have to do the allocation in the prologue of the PARALLEL
1941 * directive. We saved "end_prologue" in do_default_clause()
1942 * in semsmp.c
1943 */

1999/** \brief Check the current parallel scope of a variable.
2000 \param sym Represents the variable which actually references storage.
2001 \param orig Identifier to which the sym refers; for entries, orig will be
2002 the ST_ENTRY and sym will be its FVAL.
2003
2004 If the current scope is default private, need to ensure that any variables
2005 which have not been declared in this scope are declared as private
2006 variables. If the current scope is 'none', then need to ensure that the
2007 variables were actually declared in this scope.
2008
2009 Note that at this time, ST_UNKNOWN and ST_IDENT symbols should be have
2010 been resolved.
2011 */
2012int
2013sem_check_scope(int sym, int orig)
2014{
2015 int new;
2016 int no_scope;
2017
2018 checking_scope = TRUE;
2019 new = sym;
2020 if (sem.parallel || sem.task || sem.target || sem.teams
2021 || sem.orph
2022 ) {
2023 /* Cray pointees are special cases:
2024 * 1. the pointee is unaffected by the DEFAULT clause.
2025 * 2. the pointer's scope is determined at the point of the
2026 * pointee's use.
2027 *
2028 * For a Cray pointee, need to recursively check its pointer.
2029 * Check the scope of each pointer and create a private copy if
2030 * one is needed.
2031 */
2032 switch (STYPEG(orig)) {
2033 case ST_VAR:
2034 case ST_ARRAY:
2035 case ST_STRUCT:
2036 case ST_UNION:
2037 if (SCG(new) == SC_BASED && MIDNUMG(new) && !CCSYMG(MIDNUMG(new)) &&
2038 !HCCSYMG(MIDNUMG(new))) {
2039 int ptr;
2040 ptr = MIDNUMG(new);
2041 ptr = refsym(ptr, OC_OTHER);
2042 ptr = sem_check_scope(ptr, ptr);
2043 if (ptr != MIDNUMG(new)) {
2044 /* A new pointer was created, create a new pointee. */
2045 checking_scope = TRUE;
2046 new = decl_private_sym(new);
2047 }
2048 goto returnit;
2049 }
2050 break;
2051 default:
2052 break;
2053 }
2054 if (sem.scope_stack[sem.scope_level].par_scope != PAR_SCOPE_SHARED) {
2055 int s;
2056 switch (STYPEG(orig)) {
2057 case ST_ENTRY:
2058 case ST_VAR:
2059 case ST_ARRAY:
2060 case ST_STRUCT:
2061 case ST_UNION:
2062 if (STYPEG(new) != ST_ENTRY) {
2063 if (SCG(new) == SC_CMBLK) {
2064 if (THREADG(CMBLKG(new)))
2065 goto returnit;
2066 } else if (THREADG(new)) {
2067 goto returnit;
2068 }
2069 if (CCSYMG(new) || HCCSYMG(new))
2070 goto returnit;
2071 if (STYPEG(new) == ST_ARRAY && SCG(new) == SC_DUMMY && ASUMSZG(new)) {
2072 goto returnit;
2073 }
2074 }
2075 for (s = sem.scope_stack[sem.scope_level].rgn_scope;
2076 s <= sem.scope_level; s++) {
2077 if (SCOPEG(new) == sem.scope_stack[s].sptr)
2078 goto sym_ok;
2079 }
2080 no_scope = 0;
2081 if (!sem.ignore_default_none &&
2082 sem.scope_stack[sem.scope_level].par_scope == PAR_SCOPE_NONE) {
2083 no_scope = 1;
2084 }
2085 if (STYPEG(new) == ST_ENTRY)
2086 goto returnit;
2087 if (sem.scope_stack[sem.scope_level].par_scope ==
2088 PAR_SCOPE_TASKNODEFAULT) {
2089 if (sem.parallel) {
2090 /*
2091 * for a task appearing within the lexical extent
2092 * of a parallel region, only the private objects are
2093 * firstprivate
2094 */
2095 if (SCG(new) == SC_BASED && MIDNUMG(new)) {
2096 int ss, ptr;
2097 ss = new;
2098 ptr = MIDNUMG(new);
2099 while (TRUE) {
2100 if (SCG(ptr) != SC_PRIVATE)
2101 goto returnit;
2102 if (SCG(ptr) != SC_BASED)
2103 break;
2104 ss = ptr;
2105 }
2106 } else if (SCG(new) != SC_PRIVATE)
2107 goto returnit;
2108 } else {
2109 /*
2110 * for an orphaned task, all non-static objects are
2111 * firstprivate
2112 */
2113 if (SCG(new) == SC_CMBLK || SCG(new) == SC_STATIC || SAVEG(new))
2114 goto returnit;
2115 if (SCG(new) == SC_BASED && MIDNUMG(new)) {
2116 int ss, ptr;
2117 ss = new;
2118 ptr = MIDNUMG(new);
2119 while (TRUE) {
2120 if (SCG(ptr) == SC_STATIC || SCG(ptr) == SC_CMBLK)
2121 goto returnit;
2122 if (SCG(ptr) != SC_BASED)
2123 break;
2124 ss = ptr;
2125 }
2126 }
2127 }
2128 }
2129 new = decl_private_sym(new);
2130 if (no_scope) {
2131 add_no_scope_sptr(sym, new, gbl.lineno);
2132 }
2133 if (sem.scope_stack[sem.scope_level].par_scope ==
2134 PAR_SCOPE_FIRSTPRIVATE)
2135 add_assign_firstprivate(new, sym);
2136 else if (sem.scope_stack[sem.scope_level].par_scope ==
2137 PAR_SCOPE_TASKNODEFAULT)
2138 add_assign_firstprivate(new, sym);
2139 break;
2140 default:
2141 break;
2142 }
2143 sym_ok:;
2144 }
2145 }
2146returnit:
2147 check_parref(sym, new, orig);
2148 checking_scope = FALSE;
2149 return new;
2150}
