32/* Fortran-callable time routine
33 * calling:
34 * real*8 times(3)
35 * call ENTFTN(TIMES,times)(times)
36 * returns:
37 * times(1) = wallclock time
38 * times(2) = user process run time
39 * times(3) = system time
40 */
41
42void ENTFTN(TIMES, times)(double *times)
43{
44 struct tb t;
45
46 __fort_gettb(&t);
47 times[0] = t.r;
48 times[1] = t.u;
49 times[2] = t.s;
50}

115static void cpu(struct tb *tbp)
116{
117 int i, quiet, tcpus;
118 struct tb n, a, x, e;
119 char buf[256];
120 double d;
121
122 tcpus = GET_DIST_TCPUS;
123 fprintf(__io_stderr(), "\n");
124 fprintf(__io_stderr(),
125 "cpu real user sys ratio node\n");
126 n.r = x.r = e.r = tbp[0].r;
127 n.u = x.u = e.u = tbp[0].u;
128 n.s = x.s = e.s = tbp[0].s;
129 for (i = 1; i < tcpus; i++) {
130 if (tbp[i].r < n.r)
131 n.r = tbp[i].r;
132 if (tbp[i].r > x.r)
133 x.r = tbp[i].r;
134 if (tbp[i].u < n.u)
135 n.u = tbp[i].u;
136 if (tbp[i].u > x.u)
137 x.u = tbp[i].u;
138 if (tbp[i].s < n.s)
139 n.s = tbp[i].s;
140 if (tbp[i].s > x.s)
141 x.s = tbp[i].s;
142 e.r += tbp[i].r;
143 e.u += tbp[i].u;
144 e.s += tbp[i].s;
145 }
146
147 quiet = GET_DIST_QUIET;
148 if (quiet & Q_CPUS) {
149 for (i = 0; i < tcpus; i++) {
150 d = (tbp[i].r == 0.0 ? 0.0 : (tbp[i].u + tbp[i].s) / tbp[i].r);
151 sprintf(buf, "%4d%c%10.2f%10.2f%10.2f%9.0f%% %-s\n", i,
152 (i == GET_DIST_IOPROC ? '*' : ' '), tbp[i].r, tbp[i].u, tbp[i].s,
153 d * 100, tbp[i].host);
154 write(2, buf, strlen(buf));
155 }
156 }
157 if ((quiet & Q_CPUS) && (tcpus > 1)) {
158 a.r = e.r / (float)tcpus;
159 a.u = e.u / (float)tcpus;
160 a.s = e.s / (float)tcpus;
161 sprintf(buf, " min %10.2f%10.2f%10.2f\n", n.r, n.u, n.s);
162 write(2, buf, strlen(buf));
163 sprintf(buf, " avg %10.2f%10.2f%10.2f\n", a.r, a.u, a.s);
164 write(2, buf, strlen(buf));
165 sprintf(buf, " max %10.2f%10.2f%10.2f\n", x.r, x.u, x.s);
166 write(2, buf, strlen(buf));
167 }
168 d = (x.r == 0.0 ? 0.0 : (e.u + e.s) / x.r);
169 sprintf(buf, "total%10.2f%10.2f%10.2f%9.2fx\n", x.r, e.u, e.s, d);
170 write(2, buf, strlen(buf));
171}

175static void mem(struct tb *tbp)
176{
177 double tmaxrss; /* total max set size */
178 double tminflt; /* total minor fault */
179 double tmajflt; /* total major fault */
180 double tnvcsw; /* total voluntary switches */
181 double tnivcsw; /* total involuntary switches */
182 double tsbrk; /* total heap used (local) */
183 double tgsbrk; /* total heap used (global) */
184 int i, quiet, tcpus;
185 const char *s_sbrk, *s_gsbrk;
186 double d_sbrk, d_gsbrk;
187 char buf[256];
188
189 fprintf(__io_stderr(), "\n");
190 fprintf(__io_stderr(), "memory local global res size pag flts pag "
191 "flts voluntary involunt\n");
192 fprintf(__io_stderr(), " heap heap (pages) minor "
193 "major switches switches\n");
194 tmaxrss = 0.0;
195 tminflt = 0.0;
196 tmajflt = 0.0;
197 tnvcsw = 0.0;
198 tnivcsw = 0.0;
199 tsbrk = 0.0;
200 tgsbrk = 0.0;
201 quiet = GET_DIST_QUIET;
202 tcpus = GET_DIST_TCPUS;
203 for (i = 0; i < tcpus; i++) {
204 tmaxrss += tbp[i].maxrss;
205 tminflt += tbp[i].minflt;
206 tmajflt += tbp[i].majflt;
207 tnvcsw += tbp[i].nvcsw;
208 tnivcsw += tbp[i].nivcsw;
209 tsbrk += tbp[i].sbrk;
210 tgsbrk += tbp[i].gsbrk;
211 if (quiet & Q_MEMS) {
212 s_sbrk = scale_kbytes(tbp[i].sbrk, &d_sbrk);
213 s_gsbrk = scale_kbytes(tbp[i].gsbrk, &d_gsbrk);
214 sprintf(buf,
215 "%4d%c%8.0lf%2s%8.0lf%2s%10.0lf%10.0lf%10.0lf%10.0lf%10.0lf\n", i,
216 (i == GET_DIST_IOPROC ? '*' : ' '), d_sbrk, s_sbrk, d_gsbrk,
217 s_gsbrk, tbp[i].maxrss, tbp[i].minflt, tbp[i].majflt,
218 tbp[i].nvcsw, tbp[i].nivcsw);
219 write(2, buf, strlen(buf));
220 }
221 }
222 s_sbrk = scale_kbytes(tsbrk, &d_sbrk);
223 s_gsbrk = scale_kbytes(tgsbrk, &d_gsbrk);
224 sprintf(buf, "total%8.0lf%2s%8.0lf%2s%10.0lf%10.0lf%10.0lf%10.0lf%10.0lf\n",
225 d_sbrk, s_sbrk, d_gsbrk, s_gsbrk, tmaxrss, tminflt, tmajflt, tnvcsw,
226 tnivcsw);
227 write(2, buf, strlen(buf));
228}

231
232static void msg(struct tb *tbp)
233{
234 int i, quiet, tcpus;
235 double ds, dr, dc, dst, drt, dct;
236 double mst, mrt, mct, ast, art, act;
237 const char *ss, *sr, *sc, *as, *ar, *ac;
238 double d;
239 char buf[256];
240
241 fprintf(__io_stderr(), "\n");
242 fprintf(__io_stderr(), "messages send send send recv recv "
243 "recv copy copy copy\n");
244 fprintf(__io_stderr(), " cnt total avg cnt total "
245 "avg cnt total avg\n");
246 dst = 0;
247 drt = 0;
248 dct = 0;
249 mst = 0;
250 mrt = 0;
251 mct = 0;
252 quiet = GET_DIST_QUIET;
253 tcpus = GET_DIST_TCPUS;
254 for (i = 0; i < tcpus; i++) {
255 dst += tbp[i].bytes;
256 drt += tbp[i].byter;
257 dct += tbp[i].bytec;
258 mst += tbp[i].datas;
259 mrt += tbp[i].datar;
260 mct += tbp[i].datac;
261 if (quiet & Q_MSGS) {
262 ss = scale_bytes(tbp[i].bytes, &ds);
263 sr = scale_bytes(tbp[i].byter, &dr);
264 sc = scale_bytes(tbp[i].bytec, &dc);
265 d = (tbp[i].datas == 0 ? 0.0 : tbp[i].bytes / tbp[i].datas);
266 as = scale_bytes(d, &ast);
267 d = (tbp[i].datar == 0 ? 0.0 : tbp[i].byter / tbp[i].datar);
268 ar = scale_bytes(d, &art);
269 d = (tbp[i].datac == 0 ? 0.0 : tbp[i].bytec / tbp[i].datac);
270 ac = scale_bytes(d, &act);
271 sprintf(buf, "%4d%c%9.0lf%5.0lf%2s%5.0lf%2s%9.0lf%5.0lf%2s%5.0lf%2s%9."
272 "0lf%5.0lf%2s%5.0lf%2s\n",
273 i, (i == GET_DIST_IOPROC ? '*' : ' '), tbp[i].datas, ds, ss, ast,
274 as, tbp[i].datar, dr, sr, art, ar, tbp[i].datac, dc, sc, act, ac);
275 write(2, buf, strlen(buf));
276 }
277 }
278 ss = scale_bytes(dst, &ds);
279 sr = scale_bytes(drt, &dr);
280 sc = scale_bytes(dct, &dc);
281 d = (dst == 0 ? 0.0 : dst / mst);
282 as = scale_bytes(d, &ast);
283 d = (drt == 0 ? 0.0 : drt / mrt);
284 ar = scale_bytes(d, &art);
285 d = (dct == 0 ? 0.0 : dct / mct);
286 ac = scale_bytes(d, &act);
287 sprintf(buf, "total%9.0lf%5.0lf%2s%5.0lf%2s%9.0lf%5.0lf%2s%5.0lf%2s%9.0lf%5."
288 "0lf%2s%5.0lf%2s\n",
289 mst, ds, ss, ast, as, mrt, dr, sr, art, ar, mct, dc, sc, act, ac);
290 write(2, buf, strlen(buf));
291}

293/** \brief Print stats (called by all cpus) */
294void
295__fort_stat_term(void)
296{
297 int i, ioproc, quiet, tcpus;
298 struct tb *tbp;
299
300 __fort_gettb(&tb1);
301 tb1.r -= tb0.r;
302 tb1.u -= tb0.u;
303 tb1.s -= tb0.s;
304 if (tb1.r < (tb1.u + tb1.s)) {
305 tb1.r = tb1.u + tb1.s;
306 }
307 tb1.sbrk -= tb0.sbrk;
308 tb1.gsbrk -= tb0.gsbrk;
309
310 tcpus = GET_DIST_TCPUS;
311 tbp = (struct tb *)__fort_gmalloc(sizeof(struct tb) * tcpus);
312
313 ioproc = GET_DIST_IOPROC;
314 if (__fort_is_ioproc() == 0) { /* not i/o process */
315 __fort_rsend(ioproc, (char *)&(tb1), sizeof(struct tb), 1, __UCHAR);
316 } else {
317 for (i = 0; i < tcpus; i++) { /* i/o process */
318 if (i == ioproc) {
319 continue;
320 }
321 __fort_rrecv(i, (char *)&(tbp[i]), sizeof(struct tb), 1, __UCHAR);
322 }
323 tbp[ioproc] = tb1;
324 quiet = GET_DIST_QUIET;
325 if (quiet & (Q_CPU | Q_CPUS)) {
326 cpu(tbp);
327 }
328 if (quiet & (Q_MEM | Q_MEMS)) {
329 mem(tbp);
330 }
331 if (quiet & (Q_MSG | Q_MSGS)) {
332 msg(tbp);
333 }
334 }
335
336 __fort_gfree(tbp);
337}
