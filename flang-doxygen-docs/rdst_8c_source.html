34/** \brief Compare alignments and local storage sequences. Return true if all
35 elements of s1 are ultimately aligned to elements of s2 that reside
36 on the same processors in the same storage order.
37
38 Descriptors always have legitimate align-target pointers.
39
40 Sections s1 and s2 are stored alike if:
41
42 storage strides are identical,
43
44 extents are identical,
45
46 both are equivalently mapped onto processor arrangements, and
47
48 overlap allowances are identical.
49
50 Corresponding elements of sections s1 and s2 have identical offsets.
51
52 The sections may have different types, lengths, base addresses, and
53 linearized index base offsets.
54
55*/
56
57int
58I8(__fort_stored_alike)(F90_Desc *dd, F90_Desc *sd)
59{
60 DECL_DIM_PTRS(ddd);
61 DECL_DIM_PTRS(sdd);
62 __INT_T dim, bit;
63
64 if (dd == sd)
65 return 1;
66
67 if (dd == NULL || sd == NULL || F90_TAG_G(dd) != F90_TAG_G(sd))
68 return 0;
69
70 if (F90_TAG_G(dd) != __DESC)
71 return 1;
72
73 if (F90_RANK_G(dd) != F90_RANK_G(sd) || F90_GSIZE_G(dd) != F90_GSIZE_G(sd) ||
74 (F90_FLAGS_G(dd) | F90_FLAGS_G(sd)) & __OFF_TEMPLATE)
75 return 0; /* different rank, size, or not local */
76
77 for (dim = F90_RANK_G(dd); --dim >= 0;) {
78
79 SET_DIM_PTRS(ddd, dd, dim);
80 SET_DIM_PTRS(sdd, sd, dim);
81
82 if (F90_DPTR_EXTENT_G(ddd) != F90_DPTR_EXTENT_G(sdd))
83 return 0; /* different global extents */
84
85 if (F90_DPTR_SSTRIDE_G(ddd) != F90_DPTR_SSTRIDE_G(sdd))
86 return 0; /* different section stride */
87
88 if (F90_DPTR_SOFFSET_G(ddd) != F90_DPTR_SOFFSET_G(sdd))
89 return 0; /* different section offsets */
90
91 if (F90_DPTR_LSTRIDE_G(ddd) != F90_DPTR_LSTRIDE_G(sdd))
92 return 0; /* different local strides */
93
94 if (DIST_DPTR_LOFFSET_G(ddd) != DIST_DPTR_LOFFSET_G(sdd))
95 return 0; /* different local offsets */
96
97 if (DIST_DPTR_NO_G(ddd) != DIST_DPTR_NO_G(sdd) ||
98 DIST_DPTR_PO_G(ddd) != DIST_DPTR_PO_G(sdd))
99 return 0; /* different overlaps */
100
101 bit = 1 << dim;
102
103 if (DIST_MAPPED_G(dd) & DIST_MAPPED_G(sd) & bit) {
104
105 /* both partitioned... */
106
107 if (DIST_ALIGN_TARGET_G(dd) == DIST_ALIGN_TARGET_G(sd)) {
108
109 /* same align-target... */
110
111 if (DIST_DPTR_TAXIS_G(ddd) != DIST_DPTR_TAXIS_G(sdd))
112 return 0; /* different template axes */
113
114 if (F90_DPTR_LBOUND_G(ddd) * DIST_DPTR_TSTRIDE_G(ddd) +
115 DIST_DPTR_TOFFSET_G(ddd) !=
116 F90_DPTR_LBOUND_G(sdd) * DIST_DPTR_TSTRIDE_G(sdd) +
117 DIST_DPTR_TOFFSET_G(sdd))
118 return 0; /* different template origins */
119
120 if (DPTR_UBOUND_G(ddd) > F90_DPTR_LBOUND_G(ddd) &&
121 DIST_DPTR_TSTRIDE_G(ddd) != DIST_DPTR_TSTRIDE_G(sdd))
122 return 0; /* different template strides */
123 } else {
124
125 /* different align-target descriptors... */
126
127 if (DIST_DPTR_PSHAPE_G(ddd) != DIST_DPTR_PSHAPE_G(sdd) ||
128 DIST_DPTR_PSTRIDE_G(ddd) != DIST_DPTR_PSTRIDE_G(sdd))
129 return 0; /* different processor arrangements */
130
131 if (DIST_DPTR_BLOCK_G(sdd) * DIST_DPTR_TSTRIDE_G(ddd) !=
132 DIST_DPTR_BLOCK_G(ddd) * DIST_DPTR_TSTRIDE_G(sdd))
133 return 0;
134
135 if (DIST_DPTR_BLOCK_G(sdd) *
136 (F90_DPTR_LBOUND_G(ddd) * DIST_DPTR_TSTRIDE_G(ddd) +
137 DIST_DPTR_TOFFSET_G(ddd) - DIST_DPTR_TLB_G(ddd)) !=
138 DIST_DPTR_BLOCK_G(ddd) *
139 (F90_DPTR_LBOUND_G(sdd) * DIST_DPTR_TSTRIDE_G(sdd) +
140 DIST_DPTR_TOFFSET_G(sdd) - DIST_DPTR_TLB_G(sdd)))
141 return 0;
142 }
143 } else if ((DIST_MAPPED_G(dd) | DIST_MAPPED_G(sd)) & bit)
144 return 0; /* one partitioned and one not */
145 }
146 return 1;
147}

152/** \brief copy local data between global actual array and local dummy array */
153static void
154I8(local_copy)(char *db, F90_Desc *dd, __INT_T doffset, char *ab,
155 F90_Desc *ad, __INT_T aoffset, __INT_T dim,
156 copy_dir dir)
157{
158 DECL_DIM_PTRS(add);
159 DECL_DIM_PTRS(ddd);
160 __INT_T aoff, astr, cl, clof, cn, dstr, doff, l, alen, dlen, n, u;
161 char *aptr, *dptr;
162
163 SET_DIM_PTRS(ddd, dd, dim - 1);
164 SET_DIM_PTRS(add, ad, dim - 1);
165
166 astr = F90_DPTR_SSTRIDE_G(add) * F90_DPTR_LSTRIDE_G(add);
167 dstr = F90_DPTR_LSTRIDE_G(ddd);
168 doff = doffset + F90_DPTR_LBOUND_G(ddd) * dstr;
169
170 alen = F90_LEN_G(ad);
171 dlen = F90_LEN_G(dd);
172 cl = DIST_DPTR_CL_G(add);
173 clof = DIST_DPTR_CLOF_G(add);
174 for (cn = DIST_DPTR_CN_G(add); --cn >= 0;) {
175 n = I8(__fort_block_bounds)(ad, dim, cl, &l, &u);
176 aoff = aoffset +
177 (l * F90_DPTR_SSTRIDE_G(add) + F90_DPTR_SOFFSET_G(add) - clof) *
178 F90_DPTR_LSTRIDE_G(add);
179 if (dim > 1) {
180 for (; n > 0; --n) {
181 I8(local_copy)(db, dd, doff, ab, ad, aoff, dim - 1, dir);
182 aoff += astr;
183 doff += dstr;
184 }
185 } else if (n > 0) {
186 aptr = ab + aoff * alen;
187 dptr = db + doff * dlen;
188 if (alen == dlen) {
189 if (dir == __COPY_IN) {
190 __fort_bcopysl(dptr, aptr, n, dstr, astr, alen);
191 } else {
192 __fort_bcopysl(aptr, dptr, n, astr, dstr, alen);
193 }
194 } else {
195 __INT_T nn, dskip, askip;
196 char *dptrx, *aptrx;
197 dptrx = dptr;
198 aptrx = aptr;
199 dskip = dstr * dlen;
200 askip = astr * alen;
201 if (dir == __COPY_IN) {
202 for (nn = 0; nn < n; ++nn) {
203 __fort_bcopysl(dptrx, aptrx, 1, dstr, astr, dlen);
204 dptrx += dskip;
205 aptrx += askip;
206 }
207 } else {
208 for (nn = 0; nn < n; ++nn) {
209 __fort_bcopysl(aptrx, dptrx, 1, astr, dstr, dlen);
210 dptrx += dskip;
211 aptrx += askip;
212 }
213 }
214 }
215 doff += n * dstr;
216 }
217 cl += DIST_DPTR_CS_G(add);
218 clof += DIST_DPTR_CLOS_G(add);
219 }
220}

222/** \brief check if a descriptor is associated with a non-contiguous
223 * section.
224 *
225 * \param a is the descriptor we are checking.
226 * \param dim is the rank of the array we are checking.
227 *
228 * \returns 0 if contiguous, else the dimension that is non-contiguous.
229 */
230__INT_T
231I8(is_nonsequential_section)(F90_Desc *a, __INT_T dim)
232{
233 __INT_T is_nonseq_section;
234 __INT_T tmp_lstride;
235 int i;
236 DECL_DIM_PTRS(ad);
237
238 is_nonseq_section = 0;
239 tmp_lstride = 1;
240 for (i = 0; i < dim; i++) {
241 SET_DIM_PTRS(ad, a, i);
242 if (F90_DPTR_LSTRIDE_G(ad) != tmp_lstride || F90_DPTR_SSTRIDE_G(ad) != 1) {
243 is_nonseq_section = i + 1;
244 break;
245 }
246 tmp_lstride *= F90_DPTR_EXTENT_G(ad);
247 }
248
249 return is_nonseq_section;
250}

285/* Varargs: [ [ proc *dist_target, ]
286 __INT_T *isstar,
287 { [__INT_T paxis, ] __INT_T *dstfmt, | (__INT_T * array,
288 __INT_T * extent) }* ]
289 [ F90_Desc *align_target, __INT_T *conform,
290 [ __INT_T *collapse,
291 { __INT_T *taxis, __INT_T *tstride, __INT_T *toffset, }*
292 __INT_T *single,
293 { __INT_T *coordinate, }* ] ]
294 { __INT_T *lbound, [ __INT_T *ubound, ] }*
295 { [ __INT_T *no, __INT_T *po, ] }*
296 */
297void
298ENTFTN(QOPY_IN, qopy_in)(char **dptr, __POINT_T *doff, char *dbase,
299 F90_Desc *dd, char *ab, F90_Desc *ad,
300 __INT_T *p_rank, __INT_T *p_kind, __INT_T *p_len,
301 __INT_T *p_flags, ...)
302{
303 va_list va;
304 DECL_HDR_PTRS(au); /* (a)ctual, (d)ummy, (t)arget, (u)ltimate */
305 DECL_DIM_PTRS(add);
306 DECL_DIM_PTRS(ddd);
307 char *db, *df;
308
309 dtype kind;
310 __INT_T conform, collapse, single;
311 __INT_T rank, len;
312 __INT_T flags, isstar = 0;
313 __INT_T lbound[MAXDIMS], ubound[MAXDIMS];
314 __INT_T idx[MAXDIMS];
315
316 _io_intent intent;
317 _io_spec dist_target_spec;
318 _io_spec dist_format_spec;
319 _io_spec align_target_spec;
320
321 __INT_T i, m, n;
322 int action;
323 int pointer_present, offset_present;
324 __INT_T actual_extent[MAXDIMS], actual_size, aolb;
325 __INT_T dummy_extent[MAXDIMS], dummy_size, dolb;
326 void *unused;
327 __INT_T wrk_rank;
328
329 rank = *p_rank;
330 kind = (dtype)*p_kind;
331 len = *p_len;
332 flags = *p_flags;
333
334 if (LOCAL_MODE)
335 flags |= __LOCAL;
336
337 if (flags & __ASSUMED_SIZE)
338 flags |= __SEQUENCE;
339
340 intent = (_io_intent)(flags >> __INTENT_SHIFT & __INTENT_MASK);
341
342 dist_target_spec =
343 (_io_spec)(flags >> __DIST_TARGET_SHIFT & __DIST_TARGET_MASK);
344 dist_format_spec =
345 (_io_spec)(flags >> __DIST_FORMAT_SHIFT & __DIST_FORMAT_MASK);
346 align_target_spec =
347 (_io_spec)(flags >> __ALIGN_TARGET_SHIFT & __ALIGN_TARGET_MASK);
348
349#if defined(DEBUG)
350 if (__fort_test & DEBUG_RDST) {
351 printf("%d COPY_IN actual ab=0x%x ad=0x%x\n", GET_DIST_LCPU, ab, ad);
352 I8(__fort_describe)(ab, ad);
353 printf("%d COPY_IN dummy dptr=0x%x doff=0x%x dbase=0x%x dd=0x%x\n"
354 " %s len=%d rank=%d ",
355 GET_DIST_LCPU, dptr, doff, dbase, dd, GET_DIST_TYPENAMES(kind),
356 len, rank);
357 __fort_show_flags(flags);
358 printf("\n");
359 }
360#endif
361
362 if (!ISPRESENT(ab)) {
363 (void)I8(__fort_ptr_offset)(dptr, doff, dbase, kind, len, ab);
364 I8(__fort_copy_descriptor)(dd, ad);
365 return;
366 }
367
368 if (ISSCALAR(ad) || ISSEQUENCE(ad)) {
369 if (!(flags & (__SEQUENCE | __NO_OVERLAPS)))
370 copy_in_abort("scalar passed to nonsequential dummy");
371 if (flags & __ASSUMED_SHAPE)
372 copy_in_abort("scalar passed to assumed-shape dummy");
373
374 au = NULL; /* no descriptors */
375 } else if (F90_TAG_G(ad) == __DESC) {
376
377 /* actual argument must match nonsequential or assumed-shape
378 dummy type, kind, and rank */
379
380 if (flags & __ASSUMED_SHAPE || !(flags & (__SEQUENCE | __NO_OVERLAPS))) {
381 if (F90_RANK_G(ad) != rank)
382 copy_in_abort("actual argument rank differs from dummy");
383 else if (F90_KIND_G(ad) != kind)
384 copy_in_abort("actual argument type differs from dummy");
385 else if (F90_LEN_G(ad) != len)
386 copy_in_abort("actual argument length differs from dummy");
387 }
388
389 au = ad;
390 } else if (F90_TAG_G(ad) == 0)
391 /* tpr2467: allocable never allocated; the descriptor tag has been
392 * initialized to 0.
393 */
394 return;
395 else
396 copy_in_abort("invalid actual argument descriptor"
397 " (missing or incorrect\ninterface block in caller,"
398 " or wrong number of arguments?)");
399
400 conform = 0;
401 collapse = 0;
402 single = 0;
403 isstar = 0;
404 action = 0;
405
406 va_start(va, p_flags);
407
408 if (align_target_spec) {
409 /* consume args not used by F90 */
410 unused = va_arg(va, F90_Desc *);
411 conform = *va_arg(va, __INT_T *);
412 if (!(flags & __IDENTITY_MAP)) {
413 collapse = *va_arg(va, __INT_T *);
414 for (i = 0; i < rank; ++i) {
415 if (!(collapse >> i & 1)) {
416 unused = va_arg(va, __INT_T *);
417 unused = va_arg(va, __INT_T *);
418 unused = va_arg(va, __INT_T *);
419 }
420 }
421
422 /* single alignments */
423
424 single = *va_arg(va, __INT_T *);
425 for (i = 0; single >> i; ++i) {
426 if (single >> i & 1)
427 unused = va_arg(va, __INT_T *);
428 }
429 }
430 } else {
431 /* distribution target spec, consume unused args */
432 if ((dist_target_spec == __PRESCRIPTIVE) || (dist_target_spec == __DESCRIPTIVE)) {
433 unused = va_arg(va, proc *);
434 }
435
436 if ((dist_format_spec == __PRESCRIPTIVE) || (dist_format_spec == __DESCRIPTIVE)) {
437 isstar = *va_arg(va, __INT_T *);
438
439 for (i = 0; i < rank; ++i) {
440 if (isstar >> i & 1) {
441 if (flags & __DUMMY_COLLAPSE_PAXIS) {
442 unused = va_arg(va, __INT_T *);
443 unused = va_arg(va, __INT_T *);
444 }
445 } else {
446 if (flags & __DIST_TARGET_AXIS) {
447 unused = va_arg(va, __INT_T *);
448 }
449
450 if (!(isstar & EXTENSION_BLOCK_MASK) ||
451 !(((isstar & EXTENSION_BLOCK_MASK) >> (7 + 3 * i)) & 0x01)) {
452 unused = va_arg(va, __INT_T *);
453 } else {
454 unused = va_arg(va, __INT_T *);
455 if (!(flags & __ASSUMED_GB_EXTENT))
456 unused = va_arg(va, __INT_T *);
457 }
458 }
459 }
460 }
461 }
462
463 /* compute actual extents and length */
464
465 if (F90_TAG_G(ad) != __DESC) {
466 /* actual arg is scalar or sequential */
467
468 actual_size = 1;
469 for (i = rank; --i >= 0;)
470 actual_extent[i] = 1;
471 } else {
472 /* normal case; actual extents are global extents */
473
474 actual_size = F90_GSIZE_G(ad);
475 for (i = F90_RANK_G(ad); --i >= 0;) {
476 actual_extent[i] = F90_DIM_EXTENT_G(ad, i);
477 }
478 }
479
480 /* get array bounds; handle assumed-shape or size */
481
482 if (flags & __ASSUMED_SHAPE) {
483 dummy_size = actual_size;
484 for (i = 0; i < rank; ++i) {
485 lbound[i] = *va_arg(va, __INT_T *);
486 ubound[i] = lbound[i] + actual_extent[i] - 1;
487 dummy_extent[i] = actual_extent[i];
488 }
489 } else if (flags & __ASSUMED_SIZE) {
490 dummy_size = 1;
491 for (i = 1; i < rank; ++i) {
492 lbound[i - 1] = *va_arg(va, __INT_T *);
493 ubound[i - 1] = *va_arg(va, __INT_T *);
494 n = ubound[i - 1] - lbound[i - 1] + 1;
495 if (n < 0)
496 n = 0;
497 dummy_extent[i - 1] = n;
498 dummy_size *= n;
499 }
500 if (dummy_size <= 0)
501 n = 0;
502 else if (F90_TAG_G(ad) != __DESC)
503 n = 1;
504 else {
505 n = (actual_size * F90_LEN_G(ad)) / (dummy_size * len);
506 if (n < 0)
507 n = 0;
508 }
509 lbound[rank - 1] = *va_arg(va, __INT_T *);
510 ubound[rank - 1] = lbound[rank - 1] + n - 1;
511 dummy_extent[rank - 1] = n;
512 dummy_size *= n;
513 } else { /* explicit-shape dummy */
514 dummy_size = 1;
515 for (i = 0; i < rank; ++i) {
516 lbound[i] = *va_arg(va, __INT_T *);
517 ubound[i] = *va_arg(va, __INT_T *);
518 n = ubound[i] - lbound[i] + 1;
519 if (n < 0)
520 n = 0;
521 dummy_extent[i] = n;
522 dummy_size *= n;
523 }
524
525 if (!(flags & (__SEQUENCE | __NO_OVERLAPS | __LOCAL)) &&
526 (dummy_size > actual_size))
527 copy_in_abort("argument shape conformance error");
528 }
529
530 va_end(va);
531
532#if defined(DEBUG)
533 if (__fort_test & DEBUG_RDST) {
534 printf(" dim star px dfmt cnf col tx str off"
535 " lbnd ubnd no po\n");
536 for (i = 0; i < rank; ++i) {
537 printf("%5d%5d%5d%5d%5d%5d%5d%5d%5d%5d%5d%5d%5d\n", i + 1, 1, 0, 0, 0, 0,
538 0, 0, 0, lbound[i], ubound[i], 0, 0);
539 }
540 }
541#endif
542
543 /* determine copy_in action. force copying if index offset
544 required and base variables can't be aligned */
545
546 pointer_present = ISPRESENT(dptr);
547 offset_present = ISPRESENT(doff);
548
549 if (offset_present && !I8(__fort_ptr_aligned)(dbase, kind, len, ab))
550 action |= RELOCATE;
551
552 if (F90_TAG_G(ad) != __DESC) {
553 if (action & RELOCATE)
554 copy_in_abort("unable to align dummy base with sequential arg");
555
556 /* actual argument is sequential. make new descriptor */
557
558 action |= REDESCRIBE;
559 } else {
560
561 /* make new descriptors if lower bounds are different */
562
563 for (i = rank; --i >= 0;) {
564 if (F90_DIM_LBOUND_G(ad, i) != lbound[i] ||
565 F90_DIM_SOFFSET_G(ad, i) != 0) {
566 action |= REDESCRIBE;
567 break;
568 }
569 }
570
571 /* make new descriptors for assumed-size dummy or if rank
572 changes (sequential dummy) */
573
574 if (flags & __ASSUMED_SIZE || rank != F90_RANK_G(ad))
575 action |= REDESCRIBE;
576
577 /* squeeze if the actual argument is discontiguous (ignoring
578 overlaps) */
579
580 if (I8(is_nonsequential_section)(ad, F90_RANK_G(ad)))
581 action |= SQUEEZE;
582
583 /* copy if the local shape is different */
584
585 if (dummy_size > 0 && !(action & ~REDESCRIBE)) {
586 m = 1;
587 for (i = 0; i < rank; ++i) {
588 SET_DIM_PTRS(add, ad, i);
589 if (dummy_extent[i] > 1 &&
590 (F90_DPTR_SSTRIDE_G(add) * F90_DPTR_LSTRIDE_G(add) != m ||
591 dummy_extent[i] != actual_extent[i])) {
592#if defined(DEBUG)
593 if (__fort_test & DEBUG_RDST)
594 printf("%d nonsequential->local squeeze\n", GET_DIST_LCPU);
595#endif
596 action |= SQUEEZE;
597 break;
598 }
599 m *= dummy_extent[i];
600 }
601 }
602 }
603
604 /* make new descriptors if actual argument isn't ultimately
605 aligned to self */
606
607 if (au != ad)
608 action |= REDESCRIBE;
609
610 if (action) {
611 __DIST_INIT_DESCRIPTOR(dd, rank, kind, len, flags, NULL);
612 for (i = 0; i < rank; ++i) {
613 SET_DIM_PTRS(ddd, dd, i);
614 F90_DPTR_LBOUND_P(ddd, lbound[i]);
615 DPTR_UBOUND_P(ddd, ubound[i]);
616 F90_DPTR_SSTRIDE_P(ddd, 1); /* section stride */
617 F90_DPTR_SOFFSET_P(ddd, 0); /* section offset */
618 F90_DPTR_LSTRIDE_P(ddd, 0);
619 }
620
621 F90_FLAGS_P(dd, F90_FLAGS_G(dd) & ~__TEMPLATE);
622 I8(__fort_finish_descriptor)(dd);
623 } else {
624 I8(__fort_copy_descriptor)(dd, ad);
625
626 aolb = F90_LBASE_G(dd) - 1;
627 for (i = F90_RANK_G(dd); --i >= 0;) {
628 SET_DIM_PTRS(ddd, dd, i);
629 aolb += F90_DPTR_LSTRIDE_G(ddd) * F90_DPTR_LBOUND_G(ddd);
630 }
631 *dptr = ab + aolb * F90_LEN_G(dd);
632 F90_LBASE_P(dd, F90_LBASE_G(dd) - aolb);
633
634 F90_FLAGS_P(dd, F90_FLAGS_G(dd) | __NOT_COPIED);
635 return;
636 }

638#if defined(DEBUG)
639 if (__fort_test & DEBUG_RDST && action)
640 printf("%d%s%s%s%s%s%s%s%s%s\n", GET_DIST_LCPU,
641 action & REDESCRIBE ? " redescribe" : "",
642 action & RETARGET ? " retarget" : "",
643 action & REFORMAT ? " reformat" : "",
644 action & REALIGN ? " realign" : "",
645 action & RELOCATE ? " relocate" : "",
646 action & SQUEEZE ? " squeeze" : "", action & SHRINK ? " shrink" : "",
647 action & EXPAND ? " expand" : "", action & RECOPY ? " recopy" : "");
648#endif
649
650 if (action & ~REDESCRIBE) {
651
652 /* copying needed for all actions except REDESCRIBE */
653
654 DECL_HDR_VARS(c);
655 DECL_HDR_PTRS(cd) = dd; /* descriptor to use for copy */
656
657 if (!(pointer_present | offset_present))
658 copy_in_abort("cannot copy actual arg to sequential dummy");
659
660 /* allocate space and copy the array */
661
662 F90_FLAGS_P(dd, F90_FLAGS_G(dd) & ~__NOT_COPIED);
663
664 if (flags & (__SEQUENCE | __NO_OVERLAPS)) {
665
666 /* sequential dummy may not have same shape as the actual
667 argument, in which case we need to create a descriptor
668 with the actual argument shape just to do copy */
669
670 if (F90_RANK_G(ad) != rank)
671 cd = c;
672 else {
673 for (i = rank; --i >= 0;) {
674 if (actual_extent[i] != dummy_extent[i]) {
675 cd = c;
676 break;
677 }
678 }
679 }
680 if (cd != dd) {
681 __DIST_INIT_DESCRIPTOR(cd, F90_RANK_G(ad), F90_KIND_G(ad), F90_LEN_G(ad),
682 flags, NULL);
683 wrk_rank = F90_RANK_G(ad);
684 for (i = 1; i <= wrk_rank; ++i) {
685 SET_DIM_PTRS(ddd, dd, i - 1);
686 F90_DPTR_LBOUND_P(ddd, 1);
687 DPTR_UBOUND_P(ddd, actual_extent[i - 1]);
688 F90_DPTR_SSTRIDE_P(ddd, 1); /* section stride */
689 F90_DPTR_SOFFSET_P(ddd, 0); /* section offset */
690 F90_DPTR_LSTRIDE_P(ddd, 0);
691 }
692 F90_FLAGS_P(cd, F90_FLAGS_G(cd) & ~__TEMPLATE);
693 I8(__fort_finish_descriptor)(cd);
694 }
695 }
696
697 if (__fort_test & DEBUG_COPYIN_LOCAL)
698 __fort_tracecall("local argument copied");
699
700 db = I8(__fort_local_allocate)(F90_LSIZE_G(cd), F90_KIND_G(cd),
701 F90_LEN_G(cd), dbase, dptr, doff);
702 if (intent != __OUT) {
703 I8(local_copy)(db, cd, F90_LBASE_G(dd)-1, ab, ad,
704 F90_LBASE_G(ad)-1, rank, __COPY_IN);
705 }
706 } else {
707
708/* the array does not need to be copied. if there is local
709 data, then both the dummy and the actual should have the
710 same first local element. the base address and offset of
711 the dummy argument needs to be adjusted for unaccounted
712 single alignments, scalar subscripts, and overlap allowance
713 differences. */
714
715#if defined(DEBUG)
716 if (__fort_test & DEBUG_RDST) {
717 printf("%d COPY_IN initial ab=0x%x dd=0x%x\n", GET_DIST_LCPU, ab, dd);
718 }
719#endif
720
721 F90_FLAGS_P(dd, F90_FLAGS_G(dd) | __NOT_COPIED);
722 if (F90_TAG_G(ad) == __DESC && ~F90_FLAGS_G(dd) & __OFF_TEMPLATE &&
723 F90_GSIZE_G(dd) > 0) {
724 for (i = F90_RANK_G(ad); --i >= 0;)
725 idx[i] = F90_DIM_LBOUND_G(ad, i);
726 aolb = I8(__fort_local_offset)(ad, idx);
727
728 for (i = rank; --i >= 0;) {
729 F90_DIM_SOFFSET_P(dd, i, 0);
730 idx[i] = F90_DIM_LBOUND_G(dd, i);
731 }
732 dolb = I8(__fort_local_offset)(dd, idx);
733
734 if (flags & (__SEQUENCE | __NO_OVERLAPS)) {
735
736 /* Because a sequential dummy may be passed to an f77
737 extrinsic, the dummy address (db) must point
738 precisely at the actual first local element and the
739 dummy descriptor must produce a zero offset when
740 its first element is indexed */
741
742 db = ab + aolb * F90_LEN_G(ad);
743 F90_LBASE_P(dd, F90_LBASE_G(dd) - dolb);
744 } else { /* nonsequential dummy */
745 db = ab;
746
747#if defined(DEBUG)
748 if (aolb != dolb) {
749 I8(__fort_describe)(db, dd);
750 printf("aolb=%d dolb=%d\n", aolb, dolb);
751 copy_in_abort("nonsequential arg offsets differ");
752 }
753#endif
754 }
755 } else
756 db = ab;
757
758 df = I8(__fort_ptr_offset)(dptr, doff, dbase, kind, len, db);
759 if (df != db || (df != ab && !(pointer_present | offset_present)))
760 copy_in_abort("cannot align dummy base with actual arg");
761 }
762
763 if (flags & __NO_OVERLAPS)
764 /* coerced to sequential */
765 F90_FLAGS_P(dd, F90_FLAGS_G(dd) | __SEQUENCE);
766
767#if defined(DEBUG)
768 if (__fort_test & DEBUG_RDST) {
769 printf("%d COPY_IN final db=0x%x dd=0x%x\n", GET_DIST_LCPU, db, dd);
770 I8(__fort_describe)(db, dd);
771 }
772#endif
773}

787void I8(__fort_copy_out)(void *ab, /**< actual base address */
788 void *db, /**< dummy base address */
789 F90_Desc *ad, /**< actual descriptor */
790 F90_Desc *dd, /**< dummy descriptor */
791 __INT_T flags)
792{
793 DECL_HDR_VARS(c);
794 DECL_HDR_PTRS(cd); /* descriptor to use for copy */
795 __INT_T actual_extent[MAXDIMS], dummy_extent, i, intent;
796 __INT_T wrk_rank;
797
798 if (!ISPRESENT(ab))
799 return; /* actual arg is absent */
800
801 intent = (flags >> __INTENT_SHIFT & __INTENT_MASK);
802
803 if (dd && F90_TAG_G(dd) == 0)
804 /* tpr2467: allocable never allocated; the descriptor tag has been
805 * initialized to 0.
806 */
807 return;
808
809#if defined(DEBUG)
810 if (!ISPRESENT(db))
811 copy_out_abort("dummy not present");
812 if (dd == NULL || F90_TAG_G(dd) != __DESC)
813 copy_out_abort("invalid dummy descriptor");
814 if (F90_TAG_G(DIST_ALIGN_TARGET_G(dd)) != __DESC)
815 copy_out_abort("invalid dummy align-target descriptor");
816 if (__fort_test & DEBUG_RDST)
817 printf("%d COPY_OUT ab=%x db=%x INTENT(%s)%s\n", GET_DIST_LCPU, ab, db,
818 intents[intent],
819 F90_FLAGS_G(dd) & __NOT_COPIED ? " NOT COPIED" : "");
820#endif
821
822 if (F90_FLAGS_G(dd) & __NOT_COPIED)
823 return;
824
825 cd = dd;
826 if (F90_FLAGS_G(dd) & (__SEQUENCE | __NO_OVERLAPS)) {
827
828 /* sequential dummy may not have same shape as the actual
829 argument, in which case we need to create a descriptor
830 with the actual argument shape just to do copy */
831
832 for (i = F90_RANK_G(ad); --i >= 0;) {
833 actual_extent[i] = F90_DIM_EXTENT_G(ad, i);
834 }
835
836 if (F90_RANK_G(ad) != F90_RANK_G(dd))
837 cd = c;
838 else {
839 for (i = F90_RANK_G(ad); --i >= 0;) {
840 dummy_extent = F90_DIM_EXTENT_G(dd, i);
841 if (actual_extent[i] != dummy_extent) {
842 cd = c;
843 break;
844 }
845 }
846 }
847 if (cd != dd) {
848 __DIST_INIT_DESCRIPTOR(cd, F90_RANK_G(ad), F90_KIND_G(ad), F90_LEN_G(ad),
849 F90_FLAGS_G(dd), DIST_DIST_TARGET_G(dd));
850 wrk_rank = F90_RANK_G(ad);
851 for (i = 1; i <= wrk_rank; ++i) {
852 __DIST_SET_DISTRIBUTION(cd, i, 1, actual_extent[i - 1], 0, NULL);
853 }
854 F90_FLAGS_P(cd, F90_FLAGS_G(cd) & ~__TEMPLATE);
855 I8(__fort_finish_descriptor)(cd);
856 }
857 }
858
859 if (intent != __IN) {
860 I8(__fort_cycle_bounds)(ad);
861 I8(local_copy)(db, cd, F90_LBASE_G(dd)-1, ab, ad, F90_LBASE_G(ad)-1,
862 F90_RANK_G(ad), __COPY_OUT);
863 }
864 I8(__fort_local_deallocate)(db);
865}

926/** \brief General routine to create a template (alignment or distribution)
927 * descriptor
928 *
929 *<pre>
930 * Varargs: [ [ proc *dist_target, ]
931 * __INT_T *isstar,
932 * { [__INT_T paxis, ] ( __INT_T *dstfmt, | __INT_T * genBlockArray,
933 * __INT_T * genBlockArrayExtent,) }* ]
934 * [ F90_Desc *align_target, __INT_T *conform,
935 * [ __INT_T *collapse,
936 * { __INT_T *taxis, __INT_T *tstride, __INT_T *toffset, }*
937 * __INT_T *single,
938 * { __INT_T *coordinate, }* ] ]
939 * { __INT_T *lbound, __INT_T *ubound, }*
940 *</pre>
941 */
942void
943ENTFTN(TEMPLATE, template)
944 (F90_Desc *dd, __INT_T *p_rank, __INT_T *p_flags, ...)
945{
946 va_list va;
947 DECL_HDR_PTRS(td);
948 DECL_HDR_PTRS(tu);
949 DECL_DIM_PTRS(tdd);
950 proc *tp;
951
952 __INT_T rank, flags, isstar = 0;
953 /* isstar format: bits 0..6 set for each */
954 /* collapsed dimension. */
955 /* bits 7..27 dist. format*/
956 /* for each non-collapsed */
957 /* dim. (3-bit fields:) */
958 /* */
959 /* 0 - block,block(k), */
960 /* cyclic, cyclic(k) */
961 /* 1 - gen_block */
962 /* 2 - indirect (if we */
963 /* ever support it) */
964 /* 3..7 - future expansion. */
965
966 __INT_T collapse, single;
967 __INT_T coordinate[MAXDIMS];
968 __INT_T lbound[MAXDIMS], ubound[MAXDIMS];
969
970 _io_spec dist_target_spec;
971 _io_spec dist_format_spec;
972 _io_spec align_target_spec;
973
974 __INT_T i, m;
975 void *unused;
976
977 rank = *p_rank;
978 flags = *p_flags;
979
980 if (LOCAL_MODE)
981 flags |= __LOCAL;
982
983 dist_target_spec =
984 (_io_spec)(flags >> __DIST_TARGET_SHIFT & __DIST_TARGET_MASK);
985 dist_format_spec =
986 (_io_spec)(flags >> __DIST_FORMAT_SHIFT & __DIST_FORMAT_MASK);
987 align_target_spec =
988 (_io_spec)(flags >> __ALIGN_TARGET_SHIFT & __ALIGN_TARGET_MASK);
989
990#if defined(DEBUG)
991 if (__fort_test & DEBUG_RDST) {
992 printf("%d TEMPLATE %x ", GET_DIST_LCPU, dd);
993 __fort_show_flags(flags);
994 printf("\n");
995 }
996 if (dd == NULL)
997 __fort_abort("TEMPLATE: invalid descriptor");
998 if ((align_target_spec && (dist_target_spec | dist_format_spec)) ||
999 flags & __INHERIT) {
1000 __fort_abort("TEMPLATE: invalid flags");
1001 }
1002#endif
1003
1004 va_start(va, p_flags);
1005
1006 if (align_target_spec) {
1007
1008#if defined(DEBUG)
1009 if (align_target_spec != __PRESCRIPTIVE &&
1010 align_target_spec != __DESCRIPTIVE)
1011 __fort_abort("TEMPLATE: bad align-target flags");
1012#endif
1013
1014 /* get align-target spec */
1015
1016 td = va_arg(va, F90_Desc *);
1017
1018#if defined(DEBUG)
1019 if (td == NULL || F90_TAG_G(td) != __DESC)
1020 __fort_abort("TEMPLATE: invalid align-target descriptor");
1021#endif
1022
1023 tu = DIST_ALIGN_TARGET_G(td);
1024
1025#if defined(DEBUG)
1026 if (tu == NULL || F90_TAG_G(tu) != __DESC)
1027 __fort_abort("TEMPLATE: invalid ultimate align-target");
1028#endif
1029
1030 unused = va_arg(va, __INT_T *);
1031
1032 if (flags & __ASSUMED_SHAPE) {
1033
1034 /* template assumes shape of ultimate align-target but
1035 bounds are shifted to lbound == 1 */
1036
1037 collapse = 0;
1038 td = tu;
1039 for (i = 1; i <= rank; ++i) {
1040 SET_DIM_PTRS(tdd, td, i - 1);
1041 lbound[i - 1] = 1;
1042 ubound[i - 1] = F90_DPTR_EXTENT_G(tdd);
1043 }
1044 single = 0;
1045 } else {
1046
1047 if (flags & __IDENTITY_MAP) {
1048 collapse = 0;
1049 single = 0;
1050#if defined(DEBUG)
1051 if (F90_RANK_G(td) != rank)
1052 __fort_abort("TEMPLATE: align-target rank mismatch");
1053#endif
1054 } else {
1055 collapse = *va_arg(va, __INT_T *);
1056 for (i = 0; i < rank; ++i) {
1057 if (!(collapse >> i & 1)) {
1058 unused = va_arg(va, __INT_T *);
1059 unused = va_arg(va, __INT_T *);
1060 unused = va_arg(va, __INT_T *);
1061 }
1062 }
1063 single = *va_arg(va, __INT_T *);
1064 if (single >> F90_RANK_G(td))
1065 __fort_abort("TEMPLATE: invalid single alignment axis");
1066 m = single;
1067 for (i = 0; m > 0; ++i, m >>= 1) {
1068 if (m & 1)
1069 coordinate[i] = *va_arg(va, __INT_T *);
1070 }
1071 }
1072
1073 /* get array bounds */
1074
1075 for (i = 1; i <= rank; ++i) {
1076 lbound[i - 1] = *va_arg(va, __INT_T *);
1077 ubound[i - 1] = (i == rank && flags & __ASSUMED_SIZE)
1078 ? lbound[i - 1]
1079 : *va_arg(va, __INT_T *);
1080 }
1081 }
1082
1083 /* create alignment descriptor */
1084 __DIST_INIT_DESCRIPTOR(dd, rank, __NONE, 0, flags, tu);
1085
1086 for (i = 1; i <= rank; ++i) {
1087 I8(__fort_set_alignment)(dd, i, lbound[i - 1], ubound[i - 1], 0, 0, 0);
1088 }
1089
1090 } else {
1091
1092#if defined(DEBUG)
1093 if (flags & __ASSUMED_SHAPE)
1094 __fort_abort("TEMPLATE: assumed shape needs align-target");
1095#endif
1096
1097 /* get distribution target spec --- for F90 consume var args */
1098
1099 switch (dist_target_spec) {
1100
1101 case __PRESCRIPTIVE:
1102 case __DESCRIPTIVE:
1103 tp = va_arg(va, proc *);
1104 break;
1105
1106 case __OMITTED:
1107 tp = NULL;
1108 break;
1109
1110 case __TRANSCRIPTIVE:
1111 __fort_abort("TEMPLATE: bad dist-target flags");
1112 }
1113
1114 /* get distribution format spec */
1115
1116 switch (dist_format_spec) {
1117
1118 case __PRESCRIPTIVE:
1119 case __DESCRIPTIVE:
1120
1121 isstar = *va_arg(va, __INT_T *);
1122
1123 for (i = 0; i < rank; ++i) {
1124 if (isstar >> i & 1) {
1125 if (flags & __DUMMY_COLLAPSE_PAXIS) {
1126 unused = va_arg(va, __INT_T *);
1127 unused = va_arg(va, __INT_T *);
1128 }
1129 } else {
1130
1131 if (flags & __DIST_TARGET_AXIS) {
1132 unused = va_arg(va, __INT_T *);
1133 }
1134
1135 if (!(isstar & EXTENSION_BLOCK_MASK) ||
1136 !(((isstar & EXTENSION_BLOCK_MASK) >> (7 + 3 * i)) & 0x01)) {
1137 unused = va_arg(va, __INT_T *);
1138 } else {
1139 unused = va_arg(va, __INT_T *);
1140
1141 if (!(flags & __ASSUMED_GB_EXTENT))
1142 unused = va_arg(va, __INT_T *);
1143 }
1144 }
1145 } /*end for i<rank*/
1146 break;
1147
1148 case __OMITTED:
1149 break;
1150
1151 case __TRANSCRIPTIVE:
1152 __fort_abort("TEMPLATE: bad dist-format flags");
1153 }
1154
1155 /* get array bounds */
1156
1157 for (i = 1; i <= rank; ++i) {
1158 lbound[i - 1] = *va_arg(va, __INT_T *);
1159 ubound[i - 1] = (i == rank && flags & __ASSUMED_SIZE)
1160 ? lbound[i - 1]
1161 : *va_arg(va, __INT_T *);
1162 }
1163
1164 if (tp == NULL) {
1165 tp = __fort_localproc();
1166 } else if (tp->tag != __PROC)
1167 __fort_abort("TEMPLATE: invalid dist-target");
1168
1169 /* create distribution descriptor */
1170
1171 __DIST_INIT_DESCRIPTOR(dd, rank, __NONE, 0, flags, tp);
1172
1173 /* Set gen_block copy for each dimension and set distribution for
1174 * each dimension...
1175 */
1176
1177 for (i = 1; i <= rank; ++i) {
1178 __DIST_SET_DISTRIBUTION(dd, i, lbound[i - 1], ubound[i - 1], 0, NULL);
1179 }
1180 }
1181 va_end(va);
1182
1183#if defined(DEBUG)
1184 if (__fort_test & DEBUG_RDST) {
1185 printf("TEMPLATE: %d dd=%x align_target=%lx dist_target=%lx\n",
1186 GET_DIST_LCPU, dd, DIST_ALIGN_TARGET_G(dd), DIST_DIST_TARGET_G(dd));
1187 }
1188#endif
1189}
1190/** \brief Optimized routine to create a template
1191 *
1192 * Varargs: { __INT_T *lbound, __INT_T *ubound, }*
1193 */
1194void
1195ENTF90(TEMPLATE, template)(F90_Desc *dd, __INT_T *p_rank, __INT_T *p_flags,
1196 __INT_T *p_kind, __INT_T *p_len, ...)
1197{
1198 va_list va;
1199
1200 __INT_T rank, flags, len;
1201 dtype kind;
1202 __INT_T i;
1203 __INT_T gsize, lbase;
1204
1205 rank = *p_rank;
1206 flags = *p_flags;
1207 kind = (dtype)*p_kind;
1208 len = *p_len;
1209
1210#if defined(DEBUG)
1211 if (__fort_test & DEBUG_RDST) {
1212 printf("%d TEMPLATE %x ", GET_DIST_LCPU, dd);
1213 __fort_show_flags(flags);
1214 printf("\n");
1215 }
1216 if (dd == NULL)
1217 __fort_abort("TEMPLATE: invalid descriptor");
1218#endif

1221
1222#if defined(DEBUG)
1223 if (flags & __ASSUMED_SHAPE)
1224 __fort_abort("TEMPLATE: assumed shape unsupported");
1225#endif
1226
1227 /* create distribution descriptor */
1228
1229 __DIST_INIT_DESCRIPTOR(dd, rank, __NONE, 0, flags, NULL);
1230
1231 gsize = lbase = 1;
1232 for (i = 1; i <= rank; ++i) {

1247 F90_DPTR_SOFFSET_P(_dd, 0);
1248 lbase -= gsize * l;
1249 F90_DPTR_LSTRIDE_P(_dd, gsize);
1250 gsize *= __extent;
1251 }
1252 F90_LBASE_P(dd, lbase);
1253 F90_LSIZE_P(dd, gsize);
1254 F90_GSIZE_P(dd, gsize);
1255 F90_KIND_P(dd, kind);
1256 F90_LEN_P(dd, len);
1257 va_end(va);

1259#if defined(DEBUG)
1260 if (__fort_test & DEBUG_RDST) {
1261 printf("TEMPLATE: %d dd=%x align_target=%lx dist_target=%lx\n",
1262 GET_DIST_LCPU, dd, DIST_ALIGN_TARGET_G(dd), DIST_DIST_TARGET_G(dd));
1263 }
1264#endif

1597
1598 dtype kind;

1600

1643void
1644ENTFTN(FREE, free)(F90_Desc *d)
1645{
1646 DECL_HDR_PTRS(t);
1647 DECL_HDR_PTRS(p);
1648 DECL_HDR_PTRS(n);
1649
1650 if (d == NULL || F90_TAG_G(d) != __DESC)
1651 __fort_abort("FREE: invalid descriptor (already freed?)");
1652
1653 t = DIST_ALIGN_TARGET_G(d);
1654 if (t == NULL || F90_TAG_G(t) != __DESC) {
1655 if (DIST_NEXT_ALIGNEE_G(d) != NULL)
1656 __fort_abort("FREE: alignee has invalid align-target");
1657 } else if (t != d) {
1658
1659 /* freeing an alignee. */
1660
1661 if (F90_FLAGS_G(t) & __DYNAMIC) {
1662
1663 /* find descriptor in align-target's alignees list */
1664
1665 p = t;
1666 n = DIST_NEXT_ALIGNEE_G(t);
1667 while (n != NULL && n != d) {
1668 p = n;
1669 n = DIST_NEXT_ALIGNEE_G(n);
1670 }
1671 if (n != d)
1672 __fort_abort("FREE: alignee not in alignees list");
1673
1674 /* unlink descriptor from alignees list */
1675
1676 DIST_NEXT_ALIGNEE_P(p, DIST_NEXT_ALIGNEE_G(d));
1677 DIST_NEXT_ALIGNEE_P(d, NULL);
1678 }
1679 } else {
1680
1681 /* freeing an ultimate align-target. the compiler
1682 sometimes frees align-targets first, so we break the
1683 links from all accessible alignees. */
1684
1685 n = d;
1686 while (n != NULL) {
1687 p = n;
1688 n = DIST_NEXT_ALIGNEE_G(n);
1689 DIST_ALIGN_TARGET_P(p, NULL);
1690 DIST_NEXT_ALIGNEE_P(p, NULL);
1691 }
1692 }
1693 F90_TAG_P(d, __NONE);
1694}
