25/*
26 * Constants that are larger than 32 bits are expected to have been
27 * broken into big-endian arrays of 32-bit int chunks, regardless of
28 * the endianness of the host and target.
29 */

35typedef int32_t IEEE128[4]; /* IEEE quad precision float number */
36typedef int32_t INT128[4];
