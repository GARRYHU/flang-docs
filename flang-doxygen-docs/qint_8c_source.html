23#define GET_EXP(u) (__int128_t)(((u) & (BITS1 << 64)) >> MANTBITS)

25float128_t __mth_i_qint(float128_t xx)
26{
27 __int128_t xexp;
28 __uint128_t ux, mask;
29 float128_t x;
30 FPI fpi;
31
32 x = xx;
33 fpi.f = x;
34 ux = fpi.i;
35 xexp = GET_EXP(ux) - EXPBIAS;
36 if (xexp < 0) {
37 /* |x| < 0 => zero with the original sign */
38 fpi.i = (ux & SIGN_MASK);
39 } else if (xexp < MANTBITS) {
40 /* 1 <= |x| < 2^113:
41 * just mask out the trailing bits of the mantiassa beyond the
42 * range of the exponent; mask out the exponent field as well.
43 */
44 mask = ((__uint128_t)1 << (MANTBITS - xexp)) - 1;
45 fpi.i = ux & ~mask;
46 }
47
48 /* else illegal input, nan, inf, overflow, ...; just return it */
49 return fpi.f;
50}
