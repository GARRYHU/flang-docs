18/* overlap shift communication schedule */
19
20typedef struct {
21 sked sked;
22 chdr *ch1[MAXDIMS], *ch2[MAXDIMS], *ch3[MAXDIMS];
23 double boundary[2];
24 enum { __SHIFT, __EOSHIFT, __CSHIFT } style;
25 int rank;
26 int nsh[MAXDIMS]; /* negative shift amounts */
27 int psh[MAXDIMS]; /* positive shift amounts */
28 int span; /* highest span-able dimension */
29 int dim; /* dimension being shifted */
30 int dir; /* shift direction */
31 int extent; /* dimension extent */
32 int pcoord; /* my processor coordinate */
33 int plow; /* low processor number */
34 int pshape; /* processor shape */
35 int pstride; /* processor stride */
36 int fullcycle[MAXDIMS]; /* optimizable cyclic case if != 0 */
37#if defined(DEBUG)
38 char *base, *limit;
39 int lb[MAXDIMS];
40 int ub[MAXDIMS];
41#endif
42} olap_sked;

44/* ENTFTN(comm_start) function: adjust base addresses and call doit */
45
46static void I8(olap_start)(void *op, char *rb, char *sb, F90_Desc *rs,
47 F90_Desc *ss)
48{
49 olap_sked *o = (olap_sked *)op;
50 __INT_T i;
51
52#if defined(DEBUG)
53 if (rb != sb || rs != ss)
54 __fort_abort("olap_start: source/dest not same");
55#endif
56 for (i = 0; i < o->rank; ++i) {
57 if (o->ch1[i] != NULL) {
58 __fort_adjbase(o->ch1[i], rb, rb, F90_KIND_G(rs), F90_LEN_G(rs));
59 __fort_doit(o->ch1[i]);
60 }
61 if (o->ch2[i] != NULL) {
62 __fort_adjbase(o->ch2[i], (void *)o->boundary, rb, F90_KIND_G(rs),
63 F90_LEN_G(rs));
64 __fort_doit(o->ch2[i]);
65 }
66 if (o->ch3[i] != NULL) {
67 __fort_adjbase(o->ch3[i], rb, rb, F90_KIND_G(rs), F90_LEN_G(rs));
68 __fort_doit(o->ch3[i]);
69 }
70 }
71}

91static void I8(olap_send)(char *src, F90_Desc *as, olap_sked *o, int len)
92{
93 DECL_DIM_PTRS(dd);
94 char *adr;
95 __INT_T ak, bl, bn, bu, cl, cn, cnt, i, l, clof, n, nl, nu, pc, pstep, str,
96 tl, tu, u, wrap;
97
98 SET_DIM_PTRS(dd, as, o->dim - 1);
99
100 ak = F90_DPTR_LSTRIDE_G(dd) * F90_LEN_G(as);
101
102 pstep = o->dir;
103 if (DIST_DPTR_TSTRIDE_G(dd) < 0)
104 pstep = -pstep;
105
106 cl = DIST_DPTR_CL_G(dd);
107 cn = DIST_DPTR_CN_G(dd);
108 clof = DIST_DPTR_CLOF_G(dd) + DIST_DPTR_LAB_G(dd);
109
110 for (; cn > 0; --cn, cl += DIST_DPTR_CS_G(dd), clof += DIST_DPTR_CLOS_G(dd)) {
111 bn = I8(__fort_block_bounds)(as, o->dim, cl, &bl, &bu);
112
113 pc = o->pcoord;
114
115 do {
116
117 /* find next neighbor's processor coordinate */
118
119 pc += pstep;
120 if (pc < 0)
121 pc += o->pshape;
122 else if (pc >= o->pshape)
123 pc -= o->pshape;
124
125 wrap = (pc - o->pcoord) * pstep <= 0;
126
127 if (wrap && o->style != __CSHIFT)
128 break;
129
130 /* neighbor's template block bounds */
131
132 if (DIST_DPTR_GEN_BLOCK_G(dd)) {
133
134 /*
135 * get neighbor's block bounds from gen_block
136 */
137
138 __INT_T *gb, *tgb;
139 __INT_T i, direction, tExtent, aExtent;
140
141 nl = DIST_DPTR_TLB_G(dd);
142 tExtent = (DIST_DPTR_TUB_G(dd) - nl) + 1;
143 aExtent = F90_DPTR_EXTENT_G(dd);
144
145 if (tExtent != aExtent) {
146
147 /* recompute gen_block */
148
149 gb = tgb = I8(__fort_new_gen_block)(as, o->dim - 1);
150
151 } else {
152 tgb = 0;
153 gb = DIST_DPTR_GEN_BLOCK_G(dd);
154 }
155
156 direction = (DIST_DPTR_TSTRIDE_G(dd) < 0);
157
158 if (!direction)
159 i = 0;
160 else {
161 i = (DIST_DPTR_PSHAPE_G(dd)) - 1;
162 gb += i;
163 }
164 nu = *gb + (nl - 1);
165 for (; i != pc;) {
166 if (!direction) {
167 ++gb;
168 ++i;
169 } else {
170 --i;
171 --gb;
172 }
173 nl = nu + 1;
174 nu += *gb;
175 }
176
177 if (tgb)
178 __fort_free(tgb);
179
180 goto check_nl_and_nu;
181 }
182
183 tl = DIST_DPTR_TLB_G(dd) + pc * DIST_DPTR_BLOCK_G(dd);
184 tu = tl + DIST_DPTR_BLOCK_G(dd) - 1;
185
186 /* neighbor's owned bounds */
187
188 nl = tl - DIST_DPTR_TOFFSET_G(dd);
189 nu = tu - DIST_DPTR_TOFFSET_G(dd);
190 if (DIST_DPTR_TSTRIDE_G(dd) != 1) {
191 if (DIST_DPTR_TSTRIDE_G(dd) < 0)
192 i = nl, nl = nu, nu = i;
193 if (DIST_DPTR_TSTRIDE_G(dd) == -1) {
194 nl = -nl;
195 nu = -nu;
196 } else {
197 nl /= DIST_DPTR_TSTRIDE_G(dd);
198 nu /= DIST_DPTR_TSTRIDE_G(dd);
199 }
200 }
201 if (nl < F90_DPTR_LBOUND_G(dd))
202 nl = F90_DPTR_LBOUND_G(dd);
203 if (nu > DPTR_UBOUND_G(dd))
204 nu = DPTR_UBOUND_G(dd);
205
206 check_nl_and_nu:
207
208 if (nl > nu)
209 continue; /* neighbor has no data */
210
211 /* find overlap bounds */
212
213 if (o->dir < 0) { /* negative shift */
214 l = nu + 1;
215 u = nu + o->nsh[o->dim - 1];
216 if (l > bu) {
217 l -= o->extent;
218 u -= o->extent;
219 }
220 } else { /* positive shift */
221 l = nl - o->psh[o->dim - 1];
222 u = nl - 1;
223 if (u < bl) {
224 l += o->extent;
225 u += o->extent;
226 }
227 }
228 if (l < bl)
229 l = bl;
230 if (u > bu)
231 u = bu;
232
233 n = (u - l + 1);
234 if (n <= 0)
235 break; /* no more overlaps */
236
237 /* move the data */
238
239 adr = src + (l - clof) * ak;
240 cnt = len;
241 str = ak;
242 if (o->dim <= o->span) {
243 cnt *= n;
244 str *= n;
245 n = 1;
246 } else
247 u = l;
248
249 for (; n > 0; --n, adr += str) {
250#if defined(DEBUG)
251 o->lb[o->dim - 1] = l++;
252 o->ub[o->dim - 1] = u++;
253 if (__fort_test & DEBUG_OLAP) {
254 printf("%d olap_send %d ", GET_DIST_LCPU, o->plow + pc * o->pstride);
255 for (i = 0; i < F90_RANK_G(as); ++i)
256 printf("%c%d:%d", i == 0 ? '(' : ',', o->lb[i], o->ub[i]);
257 printf(")[%d]@%x ", cnt, adr);
258 __fort_show_scalar(adr, F90_KIND_G(as));
259 printf("\n");
260 }
261 if (adr < o->base || adr + cnt * F90_LEN_G(as) > o->limit)
262 __fort_abort("olap_send: bad address");
263#endif
264 __fort_sendl(o->ch1[o->dim - 1], pc, adr, cnt, 1, F90_KIND_G(as),
265 F90_LEN_G(as));
266 }
267 } while (pc != o->pcoord);
268 }
269}

354 nu += *gb;
355 }
356
357 if (tgb)
358 __fort_free(tgb);

361 }
362
363 /* neighbor's template block bounds */
364
365 tl = DIST_DPTR_TLB_G(dd) + pc * DIST_DPTR_BLOCK_G(dd);
366 tu = tl + DIST_DPTR_BLOCK_G(dd) - 1;
367
368 /* neighbor's owned bounds */
369
370 nl = tl - DIST_DPTR_TOFFSET_G(dd);
371 nu = tu - DIST_DPTR_TOFFSET_G(dd);
372 if (DIST_DPTR_TSTRIDE_G(dd) != 1) {
373 if (DIST_DPTR_TSTRIDE_G(dd) < 0)
374 i = nl, nl = nu, nu = i;
375 if (DIST_DPTR_TSTRIDE_G(dd) == -1) {

382 }

384 nl = F90_DPTR_LBOUND_G(dd);
385 if (nu > DPTR_UBOUND_G(dd))
386 nu = DPTR_UBOUND_G(dd);
387
388 check_nl_and_nu:
389
390 if (nl > nu)
391 continue; /* neighbor has no data */
392
393 /* find overlap bounds */
394

451 __fort_sendl(o->ch2[o->dim - 1], 0, o->boundary, cnt, 0,
452 F90_KIND_G(as), F90_LEN_G(as));
453 __fort_recvl(o->ch2[o->dim - 1], 0, adr, cnt, 1, F90_KIND_G(as),
454 F90_LEN_G(as));
455 } else
456 __fort_recvl(o->ch1[o->dim - 1], pc, adr, cnt, 1, F90_KIND_G(as),
457 F90_LEN_G(as));
458 }
459 } while (pc != o->pcoord);
460 }

463/* replicate data locally when shift amount > array extent */
464
465static void I8(olap_copy)(char *adr0, F90_Desc *as, olap_sked *o, int len)
466{
467 DECL_DIM_PTRS(dd);
468 char *adr1, *adr2;
469 __INT_T ak, m, n, nl, nu, l, u;
470#if defined(DEBUG)
471 __INT_T i;
472#endif
473
474 SET_DIM_PTRS(dd, as, o->dim - 1);
475
476 ak = F90_DPTR_LSTRIDE_G(dd) * F90_LEN_G(as);
477
478 if (DIST_DPTR_OLB_G(dd) > DIST_DPTR_OUB_G(dd))
479 return; /* no local data */
480
481 if (o->dir < 0) {
482 l = nl = DIST_DPTR_UAB_G(dd) - DIST_DPTR_PO_G(dd) + 1;
483 m = o->nsh[o->dim - 1];
484 } else {
485 u = nu = DIST_DPTR_LAB_G(dd) + DIST_DPTR_NO_G(dd) - 1;
486 m = o->psh[o->dim - 1];
487 }
488
489 for (m -= o->extent; m > 0; m -= o->extent) {
490
491 n = Min(m, o->extent);
492
493 if (o->dir < 0) {
494 nu = nl + n - 1;
495 l += o->extent;
496 u = l + n - 1;
497 } else {
498 nl = nu - n + 1;
499 u -= o->extent;
500 l = u - n + 1;
501 }
502
503 n *= len;
504 adr1 = adr0 + (nl - DIST_DPTR_LAB_G(dd)) * ak;
505 adr2 = adr0 + (l - DIST_DPTR_LAB_G(dd)) * ak;
506#if defined(DEBUG)
507 o->lb[o->dim - 1] = l;
508 o->ub[o->dim - 1] = u;
509 if (__fort_test & DEBUG_OLAP) {
510 printf("%d olap_copy ", GET_DIST_LCPU);
511 for (i = 0; i < F90_RANK_G(as); ++i)
512 printf("%c%d:%d", i == 0 ? '(' : ',', o->lb[i], o->ub[i]);
513 printf(")[%d]@%x = ", n, adr2);
514 o->lb[o->dim - 1] = nl;
515 o->ub[o->dim - 1] = nu;
516 for (i = 0; i < F90_RANK_G(as); ++i)
517 printf("%c%d:%d", i == 0 ? '(' : ',', o->lb[i], o->ub[i]);
518 printf(")@%x ", adr1);
519 __fort_show_scalar(adr1, F90_KIND_G(as));
520 printf("\n");
521 }
522 if (adr1 < o->base || adr1 + n * F90_LEN_G(as) > o->limit ||
523 adr2 < o->base || adr2 + n * F90_LEN_G(as) > o->limit)
524 __fort_abort("olap_recv: bad address");
525#endif
526 __fort_sendl(o->ch3[o->dim - 1], 0, adr1, n, 1, F90_KIND_G(as),
527 F90_LEN_G(as));
528 __fort_recvl(o->ch3[o->dim - 1], 0, adr2, n, 1, F90_KIND_G(as),
529 F90_LEN_G(as));
530 }
531}

533static void I8(olap_loop)(char *adr0, F90_Desc *as, olap_sked *o, int len,
534 int dim)
535{
536 DECL_DIM_PTRS(dd);
537 char *adr1;
538 __INT_T ak, bl, bn, bu, cl, cn, clof;
539
540 if (dim == o->dim)
541 --dim;
542
543 if (dim < 1) {
544 I8(olap_send)(adr0, as, o, len);
545 I8(olap_recv)(adr0, as, o, len);
546 I8(olap_copy)(adr0, as, o, len);
547 return;
548 }
549
550 SET_DIM_PTRS(dd, as, dim - 1);
551
552 ak = F90_DPTR_LSTRIDE_G(dd) * F90_LEN_G(as);
553
554 cl = DIST_DPTR_CL_G(dd);
555 cn = DIST_DPTR_CN_G(dd);
556 clof = DIST_DPTR_CLOF_G(dd) + DIST_DPTR_LAB_G(dd);
557
558 for (; cn > 0; --cn, cl += DIST_DPTR_CS_G(dd), clof += DIST_DPTR_CLOS_G(dd)) {
559
560 if (cn > 1 && /* optimize cyclic cases */
561 (DIST_DPTR_CLOS_G(dd) == 0 || (dim < o->dim && o->fullcycle[dim - 1]))) {
562 bl = DIST_DPTR_LAB_G(dd) + DIST_DPTR_NO_G(dd);
563 bu = DIST_DPTR_UAB_G(dd) - DIST_DPTR_PO_G(dd);
564 bn = bu - bl + 1;
565 cn = 1; /* short-circuit cyclic loop */
566 clof = DIST_DPTR_LAB_G(dd); /* cyclic indices localized */
567 } else
568 bn = I8(__fort_block_bounds)(as, dim, cl, &bl, &bu);
569
570 if (bn <= 0)
571 continue; /* no local data */
572
573 if (dim < o->dim) { /* include data shifted earlier */
574 if (dim > o->span) {
575 bl -= o->psh[dim - 1];
576 bu += o->nsh[dim - 1];
577 } else { /* span this dimension */
578 bl = DIST_DPTR_LAB_G(dd);
579 bu = DIST_DPTR_UAB_G(dd);
580 cn = 1; /* short-circuit cyclic loop */
581 clof = DIST_DPTR_LAB_G(dd); /* cyclic indices localized */
582 }
583 bn = bu - bl + 1;
584 }
585 adr1 = adr0 + (bl - clof) * ak;
586 if (dim > o->span) {
587 for (; bn > 0; --bn, adr1 += ak) {
588#if defined(DEBUG)
589 o->lb[dim - 1] = o->ub[dim - 1] = bl++;
590#endif
591 I8(olap_loop)(adr1, as, o, len, dim - 1);
592 }
593 } else {
594#if defined(DEBUG)
595 o->lb[dim - 1] = bl;
596 o->ub[dim - 1] = bu;
597#endif
598 I8(olap_loop)(adr1, as, o, len * bn, dim - 1);
599 }
600 }
601}

603static sked *I8(olap_shift)(char *ab, F90_Desc *as, olap_sked *o)
604{
605 DECL_DIM_PTRS(dd);
606 __INT_T i, abstr, lcpu, maxb, maxc;
607
608 I8(__fort_cycle_bounds)(as);
609
610 o->sked.tag = __SKED;
611 o->sked.start = I8(olap_start);
612 o->sked.free = olap_free;
613 o->sked.arg = o;
614
615 o->rank = F90_RANK_G(as);
616 o->span = 0;
617
618#if defined(DEBUG)
619 o->base = ab;
620 o->limit = ab + F90_LSIZE_G(as) * F90_LEN_G(as);
621#endif
622
623 lcpu = GET_DIST_LCPU;
624
625 for (i = 0; i < o->rank; ++i) {
626 SET_DIM_PTRS(dd, as, i);
627
628 if (DIST_DPTR_GEN_BLOCK_G(dd)) {
629
630 if (strncmp(GET_DIST_TRANSNAM, "smp", 3) == 0) {
631
632 /*
633 * find max block based on gen_block.
634 * (max block is defined as the largest element in the
635 * genblock array when using -SMP or using T3E) ...
636 */
637 __INT_T *gb, *tgb;
638 __INT_T i, tExtent, aExtent, pshape;
639
640 tExtent = (DIST_DPTR_TUB_G(dd) - DIST_DPTR_TLB_G(dd)) + 1;
641 aExtent = F90_DPTR_EXTENT_G(dd);
642
643 if (tExtent != aExtent) {
644
645 /* recompute gen_block */
646
647 gb = tgb = I8(__fort_new_gen_block)(as, o->dim - 1);
648
649 } else {
650 tgb = 0;
651 gb = DIST_DPTR_GEN_BLOCK_G(dd);
652 }
653
654 gb = DIST_DPTR_GEN_BLOCK_G(dd);
655 maxb = *gb;
656 pshape = DIST_DPTR_PSHAPE_G(dd);
657 for (i = 0; i < pshape; i++) {
658 ++gb;
659 if (*gb > maxb) {
660 maxb = *gb;
661 }
662 }
663
664 if (tgb)
665 __fort_free(tgb);
666
667 abstr = Abs(DIST_DPTR_TSTRIDE_G(dd));
668 maxc = DIST_DPTR_CYCLE_G(dd) + abstr - 1;
669
670 } else {
671
672 /* asymmetrical memory allocation, no optimization
673 * can be performed.
674 */
675
676 o->fullcycle[i] = 0;
677 continue;
678 }
679
680 } else {
681
682 abstr = Abs(DIST_DPTR_TSTRIDE_G(dd));
683 maxb = DIST_DPTR_BLOCK_G(dd) + abstr - 1;
684 maxc = DIST_DPTR_CYCLE_G(dd) + abstr - 1;
685 if (abstr != 1) {
686 maxb /= abstr;
687 maxc /= abstr;
688 }
689 }
690 o->fullcycle[i] = (maxb + o->nsh[i] + o->psh[i] >= maxc);
691 }
692
693 for (i = 0; i < o->rank; ++i) {
694 o->dim = i + 1;
695 o->ch1[i] = o->ch2[i] = o->ch3[i] = NULL;
696
697 SET_DIM_PTRS(dd, as, i);
698
699#if defined(DEBUG)
700 if (__fort_test & DEBUG_OLAP)
701 printf("%d olap dim=%d nsh=%d psh=%d po=%d no=%d\n", GET_DIST_LCPU,
702 o->dim, o->nsh[i], o->psh[i], DIST_DPTR_PO_G(dd),
703 DIST_DPTR_NO_G(dd));
704#endif
705
706 /* A lower dimension (below the one being shifted) can be
707 spanned (transferred all at once) if consecutive elements
708 are contiguous and the shift amounts for that dimension and
709 all lower dimensions are equal to their respective overlap
710 allowances. */
711
712 if (o->span == i && DIST_DPTR_COFSTR_G(dd) == 0 &&
713 o->psh[i] == DIST_DPTR_NO_G(dd) && o->nsh[i] == DIST_DPTR_PO_G(dd))
714 o->span = o->dim;
715
716 if (o->nsh[i] == 0 && o->psh[i] == 0)
717 continue; /* nothing to do */
718
719 if (o->nsh[i] > DIST_DPTR_PO_G(dd) || o->psh[i] > DIST_DPTR_NO_G(dd))
720 __fort_abort("olap_shift: shift amount exceeds allowance");
721
722 if (DIST_DPTR_PAXIS_G(dd) > 0) {
723 o->pcoord = DIST_DPTR_PCOORD_G(dd);
724 o->pshape = DIST_DPTR_PSHAPE_G(dd);
725 o->pstride = DIST_DPTR_PSTRIDE_G(dd);
726 } else {
727 o->pcoord = 0;
728 o->pshape = 1;
729 o->pstride = 1;
730 }
731 o->plow = lcpu - o->pcoord * o->pstride;
732
733 o->ch1[i] = __fort_chn_1to1(NULL, 1, o->plow, &o->pshape, &o->pstride, 1,
734 o->plow, &o->pshape, &o->pstride);
735
736 /* channel for filling boundary values */
737
738 if (o->style == __EOSHIFT)
739 o->ch2[i] =
740 __fort_chn_1to1(NULL, 0, lcpu, NULL, NULL, 0, lcpu, NULL, NULL);
741
742 /* channel for replicating data when shift > array extent */
743
744 o->extent = F90_DPTR_EXTENT_G(dd);
745 if (o->extent < 0)
746 o->extent = 0;
747 if (o->nsh[i] > o->extent || o->psh[i] > o->extent)
748 o->ch3[i] =
749 __fort_chn_1to1(NULL, 0, lcpu, NULL, NULL, 0, lcpu, NULL, NULL);
750
751 /* negative (left) shift */
752
753 if (o->nsh[i] > 0 && ~F90_FLAGS_G(as) & __OFF_TEMPLATE) {
754 o->dir = -1;
755 I8(olap_loop)(ab, as, o, 1, o->rank);
756 }
757
758 /* positive (right) shift */
759
760 if (o->psh[i] > 0 && ~F90_FLAGS_G(as) & __OFF_TEMPLATE) {
761 o->dir = 1;
762 I8(olap_loop)(ab, as, o, 1, o->rank);
763 }
764
765 __fort_chn_prune(o->ch1[i]);
766 __fort_chn_prune(o->ch2[i]);
767 __fort_chn_prune(o->ch3[i]);
768
769 __fort_setbase(o->ch1[i], ab, ab, F90_KIND_G(as), F90_LEN_G(as));
770 __fort_setbase(o->ch2[i], (void *)o->boundary, ab, F90_KIND_G(as),
771 F90_LEN_G(as));
772 __fort_setbase(o->ch3[i], ab, ab, F90_KIND_G(as), F90_LEN_G(as));
773 }
774 return &o->sked;
775}
