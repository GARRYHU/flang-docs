85#if defined(TARGET_WIN)
86static int
87asy_wait(struct asy *asy)
88{
89 long len;
90 int s;
91 int tn;
92 BOOL aio_complete;
93
94 if (!(asy->flags & ASY_IOACT)) {
95 return (0);
96 }
97 asy->flags &= ~ASY_IOACT;
98
99 for (tn = 0; tn < asy->outstanding_transactions; tn++) {
100 do {
101 aio_complete =
102 GetOverlappedResult(asy->handle, &(asy->overlap[tn]), &len, FALSE);
103 } while ((aio_complete == FALSE));
104
105 if (slime)
106 printf("---Fio_asy_wait %d\n", asy->fd);
107
108 if (asy->atd[tn].len != len) {
109 __io_set_errno(FIO_EEOF);
110 printf("Wait FIO_EEOF error forthcoming.\n");
111 return (-1);
112 }
113 }
114 asy->atd[0].off = asy->atd[asy->outstanding_transactions].off;
115 asy->outstanding_transactions = 0;
116 return (0);
117}
118#else
119static int
120asy_wait(struct asy *asy)
121{
122 long len;
123 int s;
124 int tn;
125 struct aiocb *p[1];
126
127 if (!(asy->flags & ASY_IOACT)) { /* i/o active? */
128 return (0);
129 }
130 asy->flags &= ~ASY_IOACT;
131
132 for (tn = 0; tn < asy->outstanding_transactions; tn++) {
133 p[0] = &(asy->aiocb[tn]);
134 do {
135 s = aio_suspend((const struct aiocb * const *)p, 1,
136 (const struct timespec *)0);
137 } while ((s == -1) && (__io_errno() == EINTR));
138 if (s == -1) {
139 return (-1);
140 }
141 if (slime)
142 printf("---Fio_asy_wait %d\n", asy->fd);
143 len = aio_return(&(asy->aiocb[tn]));
144 if (len == -1) {
145 s = aio_error(&(asy->aiocb[tn]));
146 __io_set_errno(s);
147 return (-1);
148 }
149 if (asy->atd[tn].len != len) { /* incomplete transfer? */
150 __io_set_errno(FIO_EEOF); /* ..yes */
151 return (-1);
152 }
153 }
154 /* Reset the number of outstanding transactions back to 0,
155 * and set the offset to the end of the last transaction list
156 */
157 asy->atd[0].off = asy->atd[asy->outstanding_transactions].off;
158 asy->outstanding_transactions = 0;
159 return (0);
160}

163int
164Fio_asy_fseek(struct asy *asy, long offset, int whence)
165{
166 if (slime)
167 printf("--Fio_asy_seek %d %ld\n", asy->fd, offset);
168
169 if (whence == SEEK_CUR) {
170 asy->atd[asy->outstanding_transactions].off += offset;
171 } else {
172 asy->atd[asy->outstanding_transactions].off = offset;
173 }
174 return (0);
175}

177/* enable fd, disable fp */
178
179int
180Fio_asy_enable(struct asy *asy)
181{
182 int n;
183
184 if (slime)
185 printf("--Fio_asy_enable %d\n", asy->fd);
186 if (asy->flags & ASY_IOACT) { /* i/o active? */
187 if (asy_wait(asy) == -1) {
188 return (-1);
189 }
190 }
191 if (asy->flags & ASY_FDACT) { /* fd already active? */
192 return (0);
193 }
194
195 asy->atd[0].off = __io_ftellx(asy->fp);
196 asy->outstanding_transactions = 0;
197 if (asy->atd[0].off == -1) {
198 return (-1);
199 }
200 n = __io_fflush(asy->fp);
201 if (n != 0) {
202 return (-1);
203 }
204 asy->flags |= ASY_FDACT; /* fd is now active */
205 return (0);
206}

208/* disable fd, enable fp */
209
210int
211Fio_asy_disable(struct asy *asy)
212{
213 int n;
214 int offset;
215
216 if (slime)
217 printf("--Fio_asy_disable %d\n", asy->fd);
218 if (asy->flags & ASY_IOACT) { /* i/o active? */
219 if (asy_wait(asy) == -1) {
220 return (-1);
221 }
222 }
223 if (!(asy->flags & ASY_FDACT)) { /* fd not active? */
224 return (0);
225 }
226 /* Seek to the end of the the list. */
227 offset = asy->atd[asy->outstanding_transactions].off;
228 n = __io_fseekx(asy->fp, offset, 0);
229 if (n == -1) {
230 return (-1);
231 }
232 asy->flags &= ~ASY_FDACT; /* fd is now inactive */
233 return (0);
234}

236/* init file for asynch i/o, called from open */

238int
239Fio_asy_open(FILE *fp, struct asy **pasy)
240{
241 struct asy *asy;
242#if defined(TARGET_WIN)
243 HANDLE temp_handle;
244#endif
245 asy = (struct asy *)calloc(sizeof(struct asy), 1);
246 if (asy == (struct asy *)0) {
247 __io_set_errno(ENOMEM);
248 return (-1);
249 }
250 asy->fp = fp;
251 asy->fd = __io_getfd(fp);
252#if defined(TARGET_WIN)
253 temp_handle = _get_osfhandle(asy->fd);
254 asy->handle =
255 ReOpenFile(temp_handle, GENERIC_READ | GENERIC_WRITE,
256 FILE_SHARE_READ | FILE_SHARE_WRITE, FILE_FLAG_OVERLAPPED);
257 if (asy->handle == INVALID_HANDLE_VALUE) {
258 __io_set_errno(EBADF);
259 return (-1);
260 }
261#endif
262 if (slime)
263 printf("--Fio_asy_open %d\n", asy->fd);
264 *pasy = asy;
265 return (0);
266}

268/* start an asynch read */
269
270int
271Fio_asy_read(struct asy *asy, void *adr, long len)
272{
273 int n;
274 int tn;
275
276#if defined(TARGET_WIN)
277 union Converter converter;
278#endif
279 if (slime)
280 printf("--Fio_asy_read %d %p %ld\n", asy->fd, adr, len);
281
282#if defined(TARGET_WIN)
283 if (asy->flags & ASY_IOACT) { /* i/o active? */
284 if (asy_wait(asy) == -1) { /* ..yes, wait */
285 return (-1);
286 }
287 }
288 tn = asy->outstanding_transactions;
289 asy->overlap[tn].Internal = 0;
290 asy->overlap[tn].InternalHigh = 0;
291 asy->overlap[tn].Pointer = 0;
292 /* Load asy->off into OffsetHigh/Offset */
293 converter.offset = asy->atd[tn].off;
294 asy->overlap[tn].Offset = converter.wOffset;
295 asy->overlap[tn].OffsetHigh = converter.wOffsetHigh;
296 asy->overlap[tn].hEvent = 0;
297 if (ReadFile(asy->handle, adr, len, NULL, &(asy->overlap[tn])) == FALSE &&
298 GetLastError() != ERROR_IO_PENDING) {
299 n = -1;
300 }
301#else
302 tn = asy->outstanding_transactions;
303 asy->aiocb[tn].aio_fildes = asy->fd;
304 asy->aiocb[tn].aio_reqprio = 0;
305 asy->aiocb[tn].aio_buf = adr;
306 asy->aiocb[tn].aio_nbytes = len;
307 memset(&(asy->aiocb[tn].aio_sigevent), 0, sizeof(struct sigevent));
308 asy->aiocb[tn].aio_offset = asy->atd[tn].off;
309 n = aio_read(&(asy->aiocb[tn]));
310#endif
311
312 if (n == -1) {
313 return (-1);
314 }
315 asy->atd[tn].len = len;
316 asy->atd[tn + 1].off = asy->atd[tn].off + len;
317 asy->flags |= ASY_IOACT; /* i/o now active */
318 asy->outstanding_transactions += 1;
319 return (0);
320}

390
