29/** brief Fortran file close
30 *
31 * \param f 0 (use default stream)
32 * \param flag FIO_KEEP, FIO_DELETE
33 */
34int
35__fortio_close(FIO_FCB *f, int flag)
36{
37
38 /* append carriage return (maybe) */
39
40 if (f->nonadvance) {
41 f->nonadvance = FALSE;
42#if defined(_WIN64)
43 if (__io_binary_mode(f->fp))
44 __io_fputc('\r', f->fp);
45#endif
46 __io_fputc('\n', f->fp);
47 if (__io_ferror(f->fp))
48 return __io_errno();
49 }
50
51 if (!f->stdunit) {
52 if (__io_fclose(f->fp) != 0) {
53 return __fortio_error(__io_errno());
54 }
55 if (flag == 0 && f->dispose == FIO_DELETE)
56 flag = FIO_DELETE;
57
58 /* (note SCRATCH files are unlinked when they are opened... ) */
59 if (flag == FIO_DELETE && f->status != FIO_SCRATCH) {
60 if (__fort_access(f->name, 2)) /* check write permission */
61 __fortio_error(FIO_EREADONLY);
62 else
63 __fort_unlink(f->name);
64 }
65#if defined(_WIN64)
66 else if (f->status == FIO_SCRATCH)
67 unlink(f->name);
68#endif
69 } else { /* stdin, stdout, stderr - just flush */
70#if defined(TARGET_OSX)
71 if (f->unit != 5 && f->unit != -5)
72#endif
73 if (__io_fflush(f->fp) != 0)
74 return __fortio_error(__io_errno());
75 }
76
77 __fortio_free_fcb(f); /* free up FCB for later use */
78
79 return 0;
80}

82static int
83_f90io_close(__INT_T *unit, __INT_T *bitv, __INT_T *iostat, char *status,
84 __CLEN_T status_siz)
85{
86 int status_flag;
87 FIO_FCB *f;
88
89 __fortio_errinit03(*unit, *bitv, iostat, "CLOSE");
90 if (ILLEGAL_UNIT(*unit))
91 return __fortio_error(FIO_EUNIT); /* illegal unit number */
92
93 /* find the FIO_FCB struct, else return */
94
95 f = __fortio_find_unit(*unit);
96 if (f == NULL)
97 return 0; /* no error occurred */
98
99 /* check for legal status parameter and assign status flag: */
100
101 status_flag = 0;
102 if (status != NULL) {
103 if (__fortio_eq_str(status, status_siz, "DELETE")) {
104 if (f->acc == FIO_READ) /* cannot delete readonly file */
105 return __fortio_error(FIO_EREADONLY);
106 status_flag = FIO_DELETE;
107 } else if (__fortio_eq_str(status, status_siz, "KEEP") ||
108 __fortio_eq_str(status, status_siz, "SAVE")) {
109 if (f->status == FIO_SCRATCH) /* cannot keep scratch file */
110 return __fortio_error(FIO_ECOMPAT);
111 status_flag = FIO_KEEP;
112 } else
113 return __fortio_error(FIO_ESPEC);
114 }
115
116 return __fortio_close(f, status_flag);
117}

178/** \brief IO cleanup routine */
179void
180__fortio_cleanup(void)
181{
182 FIO_FCB *f, *f_next;
183
184 if (!LOCAL_MODE) {
185 __fort_barrier();
186 }
187 if ((LOCAL_MODE || (GET_DIST_LCPU == GET_DIST_IOPROC))) {
188 for (f = fioFcbTbls.fcbs; f != (FIO_FCB *)0; f = f_next) {
189 /*
190 * WARNING: __fortio_close() calls __fortio_free_fcb()
191 * which removes 'f' from the fioFcbTbls.fcbs list;
192 * consequently, need to extract the 'next' field now.
193 */
194 f_next = f->next;
195 if (f->fp == NULL) { /* open? */
196 continue;
197 }
198 __io_fflush(f->fp);
199 if (f->stdunit) { /* standard unit? */
200 continue;
201 }
202 __fortio_close(f, 0);
203 if ((f->dispose == FIO_DELETE) && (f->status != FIO_SCRATCH) &&
204 (__fort_access(f->name, 2) == 0)) {
205 __fort_unlink(f->name);
206 }
207 }
208 __fortio_cleanup_fcb();
209 }
210}
