8/** \file state.h
9 \brief macros to read/write the 'state' file
10*/
11
12/*
13 * to save/restore an array to the state file:
14 * RW_FD( address, datatype, number-elements )
15 */
16#define RW_FD(b, s, n) \
17 { \
18 nw = (*p_rw)((char *)(b), sizeof(s), n, fd); \
19 if (nw != (n)) \
20 error(10, 40, 0, "(state file)", CNULL); \
21 }
22
23/*
24 * to save/restore a scalar to the state file:
25 * RW_SCALAR( variable )
26 */
27#define RW_SCALAR(b) RW_FD(&b, b, 1)
28
29/*
30 * the rw_routine should be declared:
31 * void rw_routine( RW_ROUTINE, RW_FILE )
32 * this declares the proper names for the rw_routine and file
33 * used in the RW_ macros
34 */
35#define RW_ROUTINE int (*p_rw)(void *, size_t, size_t, FILE *)
36#define RW_FILE FILE *fd
37#define RW_ROUTINE_TYPE int (*)(void *, size_t, size_t, FILE *)
38
39/*
40 * sometimes special action is taken on read or write.
41 * use these macros to test whether this is a write (save) or read (restore)
42 */
43#define ISREAD() (p_rw == (RW_ROUTINE_TYPE)fread)
44#define ISWRITE() (p_rw == (RW_ROUTINE_TYPE)fwrite)
