59void
60parser(void)
61{
62 static int maxsev;
63 LOGICAL skip_first_pass;
64 ISZ_T save_bss_addr, save_saddr;
65 int save_lineno;
66
67 skip_first_pass = TRUE;
68 if (gbl.internal == 0 && sem.mod_cnt <= 1) {
69 /*
70 * Perform the first pass on a subprogram or all module-contained
71 * contained subprograms. Semantic analysis is not performed on
72 * the executable statements.
73 */
74 skip_first_pass = FALSE;
75 sem.which_pass = 0;
76 reset_internal_subprograms();
77 save_bss_addr = get_bss_addr();
78 save_saddr = gbl.saddr;
79 save_lineno = gbl.lineno;
80 if (!XBIT(120, 0x4000000)) {
81 set_allfiles(0); /* Store file indexes */
82 }
83 fe_init(); /* init the scanner for the first parse */
84#if DEBUG
85 if (DBGBIT(1, 1) || DBGBIT(2, 1))
86 fprintf(gbl.dbgfil, "----- First Parse -----\n");
87#endif
88 _parser();
89 maxsev = gbl.maxsev;
90 if (sem.mod_cnt) {
91 /*
92 * Either a module specification or the first module-contained
93 * subprogram was just parsed. If an error occurred, issue the
94 * error summary now.
95 */
96 if (maxsev >= 3)
97 summary(FALSE, FALSE);
98 if (gbl.rutype == RU_BDATA)
99 /*
100 * A module specification part was parsed. Either:
101 * 1. a CONTAINS within the module terminated the first
102 * parse (scan sets scn.end_program_unit to TRUE), or
103 * 2. a module without contained subprograms was parsed.
104 * Now, just produce the module-created blockdata - the second
105 * parse isn't performed.
106 */
107 return;
108 /*
109 * At this point, either:
110 * 1. a module specification part with errors was parsed, or
111 * 2. the first module-contained subprogram was parsed (errors
112 * could have been detected).
113 * Perform the first parse on the remaining contained subprograms,
114 * recording the maximum error level detected for the all of the
115 * contained subprograms.
116 */
117 while (sem.mod_cnt == 1) {
118 reinit();
119 errini();
120 _parser();
121 if (gbl.maxsev > maxsev)
122 maxsev = gbl.maxsev;
123 if (maxsev >= 3 && gbl.currsub)
124 summary(FALSE, FALSE);
125 }
126 }
127 resolve_fwd_refs();
128 set_bss_addr(save_bss_addr);
129 gbl.saddr = save_saddr;
130 gbl.lineno = save_lineno;
131 if (!XBIT(120, 0x4000000)) {
132 set_allfiles(1); /* Retrieve file indexes */
133 }
134 }
135/*
136 * At this point, we're ready to perform the second parse (includes
137 * executable statements). The second parse is performed on a host
138 * subprogram, an internal subprogram, or module-contained subprogram
139 * if errors were not detected during the first parse.
140 */
141 if (maxsev < 3) {
142 fe_restart();
143 if (!skip_first_pass)
144 reinit();
145 sem.which_pass = 1;
146 p_semant1[1] = semant1;
147 p_semant2[1] = semant2;
148#if DEBUG
149 if (DBGBIT(1, 1) || DBGBIT(2, 1))
150 fprintf(gbl.dbgfil, "----- Second Parse -----\n");
151#endif
152 _parser();
153 resolve_fwd_refs();
154 if (gbl.internal || sem.mod_cnt > 1)
155 /*
156 * If we're processing module-contained subprograms, make sure
157 * the scanner is initialized for the next module subprogram.
158 */
159 fe_init();
160 } else if (sem.mod_cnt == 2) {
161 /*
162 * One or more errors were detected in a module subprogram during the
163 * the first parse. Don't perform the second parse on any of the
164 * module subprograms. The ensuing call to parser() will start the
165 * first parse on the statement after the ENDMODULE statement.
166 */
167 end_module();
168 sem.mod_cnt = 0;
169 sem.mod_sym = 0;
170 sem.submod_sym = 0;
171 }
172}

174static void
175_parser(void)
176{
177 int tkntyp, newtop, rednum, endflg;
178 int t;
179 int start, end, nstate;
180 int jstart, jend, ptr, i;
181 char *ptoken;
182
183 endflg = 0;
184 sst_size = SST_SIZE;
185 NEW(pstack, PSTACK, sst_size);
186 if (pstack == NULL)
187 error(7, 4, 0, CNULL, CNULL);
188 NEW(sst, SST, sst_size);
189 if (sst == NULL)
190 error(7, 4, 0, CNULL, CNULL);
191
192 /* set funcline to a best guess value in case profiling info is
193 requested for an unnamed program */
194 gbl.funcline = gbl.lineno + 1;
195
196 /* loop once for each statement in subprogram unit: */
197
198 LOOP
199 {
200
201 parse_init();
202
203 /* loop once for each token in current Fortran stmt: */
204
205 tkntyp = get_token(&ctknval); /* first token of a statement */
206
207 if (sem.which_pass == 0) {
208 /*
209 * For the first pass, we want to semantically analyze expressions
210 * only if they occur within a declaration statement. Also, there
211 * are semantic actions shared by the semantic analysis routines
212 * for declaration and executable statements; for the first
213 * pass, we only want these actions to be performed only if they
214 * occur within the declarations statements.
215 */
216 if (is_declaration(tkntyp) ||
217 (tkntyp == TK_GENERIC && sem.type_mode == 2) /* generic tbp */
218 ) {
219 p_semant1[0] = semant1;
220 p_semant2[0] = semant2;
221 } else {
222 p_semant2[0] = psemant2;
223 }
224 gbl.nowarn = FALSE;
225 } else {
226 if (is_declaration(tkntyp)) {
227 /* warnings issued in first pass for declarations */
228 gbl.nowarn = TRUE;
229 } else {
230 gbl.nowarn = FALSE;
231 }
232 }
233
234 LOOP
235 {
236#if DEBUG
237 if (tkntyp < 1 || tkntyp >= (sizeof(tokname) / sizeof(char *))) {
238 interr("scan error in parser", tkntyp, 3);
239 tkntyp = TK_END;
240 }
241#endif
242 if (scn.end_program_unit)
243 endflg = 1;
244
245 if (DBGBIT(1, 1)) {
246 fprintf(gbl.dbgfil, "tkntyp: %s tknval: %d", tokname[tkntyp], ctknval);
247 if (tkntyp == TK_IDENT)
248 fprintf(gbl.dbgfil, " (%s)", scn.id.name + ctknval);
249 fprintf(gbl.dbgfil, " lineno: %d \n", gbl.lineno);
250 }
251
252 /*
253 * loop once for each reduction which can be made with tkntyp as
254 * look ahead token (note that a production index may be 0):
255 */
256 LOOP
257 {
258
259 /*
260 * perform binary search on parse tables to determine if a
261 * reduction can be made:
262 */
263 start = fred[cstate];
264 end = fred[cstate + 1] - 1;
265 if (start > end)
266 break; /* no reduction */
267 for (i = start; i <= end; i++) {
268 jstart = lset[nset[i]];
269 jend = lset[nset[i] + 1] - 1;
270 while (jstart <= jend) {
271 ptr = (jstart + jend) >> 1;
272 t = ls[ptr];
273 if (t == tkntyp)
274 goto perform_reduction;
275 if (t < tkntyp)
276 jstart = ptr + 1;
277 else
278 jend = ptr - 1;
279 }
280 }
281 break; /* no reduction found */
282
283 perform_reduction:
284 rednum = prod[i];
285 sem.tkntyp = tkntyp;
286 if (DBGBIT(2, 1))
287#if DEBUG
288 fprintf(gbl.dbgfil, "%4d %crod(%4d) %s\n",
289 gbl.lineno, sem.which_pass ? 'P' : 'p',
290 rednum, prodstr[rednum]);
291#else
292 fprintf(gbl.dbgfil, " %cednum: %d\n",
293 sem.which_pass ? 'R' : 'r', rednum);
294#endif
295
296 /* call appropriate semantic action routine: */
297
298 newtop = stktop - len[rednum] + 1;
299 if (rednum < SEM2)
300 p_semant1[sem.which_pass](rednum, &sst[newtop]);
301 else if (rednum < SEM3)
302 p_semant2[sem.which_pass](rednum, &sst[newtop]);
303 else if (rednum < SEM4)
304 p_semant3[sem.which_pass](rednum, &sst[newtop]);
305 else if (rednum < SEM5)
306 p_semantio[sem.which_pass](rednum, &sst[newtop]);
307 else
308 p_semsmp[sem.which_pass](rednum, &sst[newtop]);
309
310 if (sem.ignore_stmt) {
311 sem.ignore_stmt = FALSE;
312 goto ignore_stmt;
313 }
314
315 /* look for reduce transition: */
316
317 nstate = next_state(pstack[newtop - 1], (int)lhs[rednum]);
318 if (nstate == NOSTATE)
319 goto issue_error;
320 else {
321 cstate = nstate;
322 pstack[newtop] = nstate;
323 stktop = newtop;
324 }
325 } /* end of reduce loop. */
326
327 /* look for a read transition: */
328
329 nstate = next_state(cstate, tkntyp);
330 if (nstate == NOSTATE) {
331 /* tpr 535
332 * the grammar cannot be modified to support complex
333 * constants of the form '( const-expr , const-expr )' but
334 * can modified if a special token is returned for ',' (i.e.,
335 * a "complex comma").
336 * if a syntax error occurs when the current token is a comma,
337 * check if a "complex comma" is legal; if so, continue
338 * by parsing as if we have a complex constant, and semant
339 * will determine if the real & imag parts are constants.
340 */
341 if (tkntyp == TK_COMMA) {
342 nstate = next_state(cstate, TK_CMPLXCOMMA);
343 if (nstate != NOSTATE) {
344 if (DBGBIT(1, 1))
345 fprintf(gbl.dbgfil, ">>> comma changed to complex comma %d\n",
346 gbl.lineno);
347 goto read_trans;
348 }
349 }
350 issue_error:
351
352 ptoken = prettytoken(tkntyp, ctknval);
353 errWithSrc(34, 3, gbl.lineno, ptoken, CNULL, getCurrColumn(), 1, false,
354 getDeduceStr(ptoken));
355 sem.psfunc = FALSE; /* allow no stmt func defs */
356 break;
357 }
358 read_trans:
359 stktop++;
360 if (stktop >= sst_size) {
361 sst_size += SST_SIZE;
362 pstack = (PSTACK *)sccrelal((char *)pstack,
363 ((BIGUINT64)((sst_size) * sizeof(PSTACK))));
364 sst = (SST *)sccrelal((char *)sst, ((BIGUINT64)((sst_size) * sizeof(SST))));
365 assert(pstack != NULL, "parser:stack ovflw", stktop, 4);
366 assert(sst != NULL, "parser:stack ovflw", stktop, 4);
367 }
368 pstack[stktop] = nstate;
369 SST_SYMP(&sst[stktop], ctknval);
370 SST_LINENOP(&sst[stktop], gbl.lineno);
371 SST_COLUMNP(&sst[stktop], getCurrColumn());
372 cstate = nstate;
373
374 if (tkntyp == TK_EOL) {
375 if (endflg == 1)
376 goto parse_done;
377
378 if (!scn.multiple_stmts && gbl.eof_flag ) {
379 if (gbl.empty_contains) {
380 gbl.internal = 0;
381 goto parse_done;
382 }
383
384 errsev(22);
385 sem.mod_cnt = 0;
386 sem.mod_sym = 0;
387 sem.submod_sym = 0;
388 goto parse_done;
389 }
390 break;
391 }
392
393 tkntyp = get_token(&ctknval); /* next token in the statement */
394
395 } /* end foreach token LOOP */
396
397 ignore_stmt:;
398
399 } /* end foreach statement LOOP */
400
401parse_done:
402 FREE(pstack);
403 FREE(sst);
404 pstack = NULL;
405 sst = NULL;
406 sst_size = 0;
407}

409/* Initialize parser to begin parsing of next Fortran statement */
410void
411parse_init(void)
412{
413 pstack[0] = 0;
414 pstack[1] = 1;
415 cstate = 1;
416 stktop = 1;
417
418 scan_reset();
419}

421/* Return next parse state, given current state and look ahead
422 token. NOSTATE is returned if there is no next state (syntax
423 error).
424*/
425static int
426next_state(int state, int tkntyp)
427{
428 int start, end, ptr, t;
429
430 start = ftrn[state];
431 end = ftrn[state + 1] - 1;
432
433 while (start <= end) {
434 ptr = (start + end) >> 1;
435 t = ent[tran[ptr]];
436 if (t == tkntyp)
437 return (tran[ptr]);
438 if (t < tkntyp)
439 start = ptr + 1;
440 else
441 end = ptr - 1;
442 }
443 return (NOSTATE);
444}

446static char *
447prettytoken(int tkntyp, INT tknval)
448{
449 static char symbuf[132];
450 INT v[2];
451
452 switch (tkntyp) {
453 case TK_EOL:
454 sprintf(symbuf, "end of line");
455 break;
456 case TK_IDENT:
457 case TK_NAMED_CONSTRUCT:
458 sprintf(symbuf, "identifier %s", scn.id.name + tknval);
459 break;
460 case TK_LOGCONST:
461 sprintf(symbuf, "logical constant %s",
462 tknval == SCFTN_TRUE ? ".TRUE." : ".FALSE.");
463 break;
464 case TK_K_LOGCONST:
465 sprintf(symbuf, "logical literal %s", getprint((int)tknval));
466 break;
467 case TK_ICON:
468 sprintf(symbuf, "integer constant %d", tknval);
469 break;
470 case TK_K_ICON:
471 sprintf(symbuf, "integer literal %s", getprint((int)tknval));
472 break;
473 case TK_RCON:
474 strcpy(symbuf, "real constant ");
475 v[0] = tknval;
476 v[1] = 0;
477 cprintf(symbuf + 14, "%.7e", v);
478 break;
479 case TK_DCON:
480 sprintf(symbuf, "doubleprecision constant %s", getprint((int)tknval));
481 break;
482 case TK_CCON:
483 sprintf(symbuf, "complex constant %s", getprint((int)tknval));
484 break;
485 case TK_DCCON:
486 sprintf(symbuf, "doublecomplex constant %s", getprint((int)tknval));
487 break;
488 case TK_HOLLERITH:
489 sprintf(symbuf, "hollerith constant %10.10s",
490 stb.n_base + CONVAL1G(tknval));
491 break;
492 case TK_NONDDEC:
493 sprintf(symbuf, "%s", getprint((int)tknval));
494 break;
495 case TK_NONDEC:
496 sprintf(symbuf, "non-decimal constant %x", tknval);
497 break;
498 case TK_CMPLXCOMMA:
499 sprintf(symbuf, ",");
500 break;
501 case TK_IOLP:
502 case TK_IMPLP:
503 sprintf(symbuf, "(");
504 break;
505 case TK_EQ:
506 if (tknval)
507 sprintf(symbuf, "==");
508 else
509 sprintf(symbuf, "%s", tokname[tkntyp]);
510 break;
511 case TK_GE:
512 if (tknval)
513 sprintf(symbuf, ">=");
