19/** Uplevel data structure containing a list of shared variables for the region
20 * nest that this uplevel belongs to. The shared variables in this structure
21 * are represented as a list of unique sptrs.
22 */
23typedef struct {
24 int *vals; /* Array containing shared var sptrs */
25 int vals_size; /* Total allocated slots in vals */
26 int vals_count; /* Traditionally "available" or vals_avl */
27 DTYPE dtype; ///< The true dtype containing fields and their offsets
28 SPTR parent; /* sptr of its parent */
29 /* TODO: Consider using a hashset to speed-unique lookup */
30} LLUplevel;

32/** First private variable:
33 * This information is necessary to generate code that allows a particular task
34 * to know which variables are stored in its allocated memory (task memory).
35 * First privates for the task have to be stored in that memory.
36 */
37typedef struct _llprivate_t {
38 int shared_sptr; /**< Represents the caller's copy */
39 int private_sptr; /**< Represents the callee's copy (local) */
40} LLFirstPrivate;

42/// Task data structure containing a list of private variables for the task.
43typedef struct {
44 int scope_sptr; /**< Outlined task's scope sptr (BMPSCOPE ST_BLOCK) */
45 int task_sptr; /**< Outlined function representing the task */
46 LLFirstPrivate *privs; /**< Array of private sptrs for this task */
47 int privs_count;
48 int privs_size;
49 int actual_size; /**< Bytes in task: First priv size + base task size */
50} LLTask;

52/// Data attributes for each data reference used in an OpenMP target region.
53typedef enum {
54 // No flags
55 OMP_TGT_MAPTYPE_NONE = 0x000,
56 // copy data from host to device
57 OMP_TGT_MAPTYPE_TO = 0x001,
58 // copy data from device to host
59 OMP_TGT_MAPTYPE_FROM = 0x002,
60 // copy regardless of the reference count
61 OMP_TGT_MAPTYPE_ALWAYS = 0x004,
62 // force unmapping of data
63 OMP_TGT_MAPTYPE_DELETE = 0x008,
64 // map the pointer as well as the pointee
65 OMP_TGT_MAPTYPE_PTR_AND_OBJ = 0x010,
66 // pass device base address to kernel
67 OMP_TGT_MAPTYPE_TARGET_PARAM = 0x020,
68 // return base device address of mapped data
69 OMP_TGT_MAPTYPE_RETURN_PARAM = 0x040,
70 // private variable - not mapped
71 OMP_TGT_MAPTYPE_PRIVATE = 0x080,
72 // copy by value - not mapped
73 OMP_TGT_MAPTYPE_LITERAL = 0x100,
74 // mapping is implicit
75 OMP_TGT_MAPTYPE_IMPLICIT = 0x200,
76 // member of struct, member given by 4 MSBs - 1
77 OMP_TGT_MAPTYPE_MEMBER_OF = 0xffff000000000000
78} tgt_map_type;
