69void static INLINE
70__reduction_slowpath(vfloat const a,
71 S(vfloat *sr, vmask *sh) SINCOS_COMMA C(vfloat *cr, vmask *ch) )
72{
73 vint2 ia, e, idx, q, p;
74 S(vint2 s;)
75 vint2 ia_a, ia_b, p_a, p_b, hi_a, hi_b;
76 vint2 hi, lo, ll, prev, prev2;
77
78 vmask i1opi_vec[] = {
79 vcast_vm_i_i(0, i1opi_f[0]),
80 vcast_vm_i_i(0, i1opi_f[1]),
81 vcast_vm_i_i(0, i1opi_f[2]),
82 vcast_vm_i_i(0, i1opi_f[3]),
83 vcast_vm_i_i(0, i1opi_f[4]),
84 vcast_vm_i_i(0, i1opi_f[5]),
85 };
86
87 ia = (vint2)a;
88 S(s = vand_vi2_vi2_vi2(ia, vcast_vi2_i(0x80000000));)
89 /* e = ((ia >> 23) & 0xff) - 127; */
90 e = vsrl_vi2_vi2_i(ia, 23);
91 e = vand_vi2_vi2_vi2(e, vcast_vi2_i(0xff));
92 e = vsub_vi2_vi2_vi2(e, vcast_vi2_i(127));
93 /* ia = (ia << 8) | 0x80000000; */
94 ia = vsll_vi2_vi2_i(ia, 8);
95 ia = vor_vi2_vi2_vi2(ia, vcast_vi2_i(0x80000000));
96
97 /* compute x * 1/pi */
98 /* idx = 6 - ((e >> 5) & 3); */
99 idx = vsrl_vi2_vi2_i(e, 5);
100 idx = vand_vi2_vi2_vi2(idx, vcast_vi2_i(3));
101 idx = vsub_vi2_vi2_vi2(vcast_vi2_i(6), idx);
102
103 ia_a = vsrl64_vi2_vi2_i(ia, 32);
104 ia_b = ia;
105 hi_a = vcast_vi2_i(0);
106 hi_b = vcast_vi2_i(0);
107
108 q = vcast_vi2_i(0);
109 for (int i = 0; i < 6; i++) {
110 p_a = vmulu_vi2_vi2_vi2((vint2)i1opi_vec[i], ia_a);
111 p_b = vmulu_vi2_vi2_vi2((vint2)i1opi_vec[i], ia_b);
112 p_a = vadd64_vi2_vi2_vi2(p_a, hi_a);
113 p_b = vadd64_vi2_vi2_vi2(p_b, hi_b);
114
115 hi_a = vsrl64_vi2_vi2_i(p_a, 32);
116 hi_b = vsrl64_vi2_vi2_i(p_b, 32);
117
118 p_a = vsll64_vi2_vi2_i(p_a, 32);
119 p_b = vand_vi2_vi2_vi2(p_b, vcast_vm_i_i(0, 0xffffffff));
120
121 p = vor_vi2_vi2_vi2(p_a, p_b);
122
123 vopmask m = veq_vo_vi2_vi2(idx, q);
124 hi = vsel_vi2_vo_vi2_vi2(m, p, hi);
125 lo = vsel_vi2_vo_vi2_vi2(m, prev, lo);
126 ll = vsel_vi2_vo_vi2_vi2(m, prev2, ll);
127
128 prev2 = prev;
129 prev = p;
130
131 q = vadd_vi2_vi2_vi2(q, vcast_vi2_i(1));
132 }
133 p = vor_vi2_vi2_vi2(vsll64_vi2_vi2_i(hi_a, 32), hi_b);
134
135 vopmask m = veq_vo_vi2_vi2(idx, q);
136 hi = vsel_vi2_vo_vi2_vi2(m, p, hi);
137 lo = vsel_vi2_vo_vi2_vi2(m, prev, lo);
138 ll = vsel_vi2_vo_vi2_vi2(m, prev2, ll);
139
140 e = vand_vi2_vi2_vi2(e, vcast_vi2_i(31));
141
142 union {
143 vint2 v;
144 uint32_t t[sizeof(vint2) / sizeof(uint32_t)];
145 } ue, uhi, ulo, ull, us,
146 S(suh, sur, sus) SINCOS_COMMA C(cuh, cur)
147 ;
148
149 ue.v = e; uhi.v = hi; ulo.v = lo; ull.v = ll;
150 S(sus.v = s;)
151
152 for (unsigned i = 0; i < sizeof(vint2) / sizeof(uint32_t); i++) {
153 uint32_t e = ue.t[i];
154 S(uint32_t sq;)
155 S(uint64_t sp = ((uint64_t)uhi.t[i] << 32) | ulo.t[i];)
156 C(uint32_t cq;)
157 C(uint64_t cp = ((uint64_t)uhi.t[i] << 32) | ulo.t[i];)
158
159 if (e) {
160 S(sq = 32 - e;)
161 S(sp = (sp << e) | (ull.t[i] >> sq);)
162 C(cq = 32 - e;)
163 C(cp = (cp << e) | (ull.t[i] >> cq);)
164 }
165
166 S(
167 sq = (uhi.t[i] << e) & 0x80000000;
168 sp &= 0x7fffffffffffffffULL;
169
170 if (sp & 0x4000000000000000ULL) {
171 sp |= 0x8000000000000000ULL;
172 sq ^= 0x80000000;
173 }
174 suh.t[i] = sq ^ sus.t[i];
175 )
176
177 C(
178 cuh.t[i] = (uhi.t[i] << e) & 0x80000000;
179 cp &= 0x7fffffffffffffffULL;
180 cp = (int64_t)cp - 0x4000000000000000LL;
181 )
182
183 S(
184 double sd = (double)(int64_t)sp;
185 sd *= PI_2_M63;
186 float sr = (float)sd;
187 sur.t[i] = float_as_int(sr);
188 )
189
190 C(
191 double cd = (double)(int64_t)cp;
192 cd *= PI_2_M63;
193 float cr = (float)cd;
194 cur.t[i] = float_as_int(cr);
195 )
196
197 }
198
199 S(vstore_v_p_vf((float*)sh, (vfloat)suh.v);)
200 S(vstore_v_p_vf((float*)sr, (vfloat)sur.v);)
201 C(vstore_v_p_vf((float*)ch, (vfloat)cuh.v);)
202 C(vstore_v_p_vf((float*)cr, (vfloat)cur.v);)
203}

206vfloat __attribute__((noinline)) FCN_NAME(const vfloat x)
207{
208 vfloat xabs;
209 vfloat vt; // General temp register
210 S(vfloat as, ks, rs;)
211 S(vint2 hs;)
212 S(vfloat ss, fs, ts;)
213
214 C(vfloat ac, kc, rc;)
215 C(vint2 hc;)
216 C(vfloat sc, fc, tc;)
217
218 xabs = vabs_vf_vf(x);
219
220
221/*
222 * Reduce x for sine.
223 */
224 S(ks = vfma_vf_vf_vf_vf(x, vcast_vf_f(_1_OVER_PI_F), vcast_vf_f(12582912.0f));)
225 S(hs = vsll_vi2_vi2_i((vint2)ks, 31);)
226 S(ks = vsub_vf_vf_vf(ks, vcast_vf_f(12582912.0f));)
227
228
229/*
230 * Reduce x for cosine.
231 */
232 C(kc = vfma_vf_vf_vf_vf(xabs, vcast_vf_f(_1_OVER_PI_F), vcast_vf_f(-0.5f));)
233 C(kc = vadd_vf_vf_vf(kc, vcast_vf_f(12582912.0f));)
234 C(hc = vsll_vi2_vi2_i((vint2)kc, 31);)
235 C(kc = vsub_vf_vf_vf(kc, vcast_vf_f(12582912.0f));)
236 //C(kc = vfma_vf_vf_vf_vf(vcast_vf_f(2.0f), kc, vcast_vf_f(1.0f));)
237 C(kc = vadd_vf_vf_vf(kc, vcast_vf_f(0.5f));)
238
239
240 S(as = vfma_vf_vf_vf_vf(ks, vcast_vf_f(-PI_HI_F), x);)
241 C(ac = vfma_vf_vf_vf_vf(kc, vcast_vf_f(-PI_HI_F), xabs);)
242
243 S(as = vfma_vf_vf_vf_vf(ks, vcast_vf_f(-PI_MI_F), as);)
244 C(ac = vfma_vf_vf_vf_vf(kc, vcast_vf_f(-PI_MI_F), ac);)
245
246 S(as = vfma_vf_vf_vf_vf(ks, vcast_vf_f(-PI_LO_F), as);)
247 C(ac = vfma_vf_vf_vf_vf(kc, vcast_vf_f(-PI_LO_F), ac);)
248
249 vopmask m_lrg_args = vgt_vo_vi2_vi2((vint2)xabs, (vint2)vcast_vf_f(THRESHOLD_F));
250 if (__builtin_expect(!vtestz_i_vo(m_lrg_args), 0)) {
251 /*
252 * Handle large arguments.
253 */
254 S(vfloat spas;)
255 S(vint2 sphs;)
256 C(vfloat spac;)
257 C(vint2 sphc;)
258
259 __reduction_slowpath(x,
260 S(&spas, &sphs) SINCOS_COMMA C(&spac, &sphc));
261 S(as = vsel_vf_vo_vf_vf(m_lrg_args, spas, as);)
262 C(ac = vsel_vf_vo_vf_vf(m_lrg_args, spac, ac);)
263 S(hs = (vmask)vsel_vi2_vo_vi2_vi2(m_lrg_args, (vint2)sphs, (vint2)hs);)
264 C(hc = (vmask)vsel_vi2_vo_vi2_vi2(m_lrg_args, (vint2)sphc, (vint2)hc);)
265 }
266
267/*
268 *
269 rs = __sin_kernel(as, hs);
270 */
271
272 S(ss = vmul_vf_vf_vf(as, as);)
273 C(sc = vmul_vf_vf_vf(ac, ac);)
274
275 S(rs = vcast_vf_f(A_F);)
276 C(rc = vcast_vf_f(-A_F);)
277
278 S(rs = vfma_vf_vf_vf_vf(rs, ss, vcast_vf_f(B_F));)
279 C(rc = vfmapn_vf_vf_vf_vf(rc, sc, vcast_vf_f(B_F));)
280
281 S(rs = vfma_vf_vf_vf_vf(rs, ss, vcast_vf_f(C_F));)
282 C(rc = vfmapn_vf_vf_vf_vf(rc, sc, vcast_vf_f(C_F));)
283
284 S(rs = vfma_vf_vf_vf_vf(rs, ss, vcast_vf_f(D_F));)
285 C(rc = vfmapn_vf_vf_vf_vf(rc, sc, vcast_vf_f(D_F));)
286
287 S(rs = vfma_vf_vf_vf_vf(rs, ss, vcast_vf_f(E_F));)
288 C(rc = vfmapn_vf_vf_vf_vf(rc, sc, vcast_vf_f(E_F));)
289
290 S(fs = (vfloat)vxor_vi2_vi2_vi2((vint2)as, hs);)
291 C(fc = (vfloat)vxor_vi2_vi2_vi2((vint2)ac, hc);)
292
293 S(ts = vfma_vf_vf_vf_vf(ss, fs, vcast_vf_f(0.0));)
294 C(tc = vfma_vf_vf_vf_vf(sc, fc, vcast_vf_f(0.0));)
295
296 S(rs = vfma_vf_vf_vf_vf(rs, ts, fs);)
297 C(rc = vfmapn_vf_vf_vf_vf(rc, tc, fc);)
298
299 /*
300 * Cosine args less than equal to 0x39800000 return 1.0.
301 */
302 C(vopmask m0 = vgt_vo_vi2_vi2((vint2)xabs, vcast_vi2_i(0x39800000));)
303 C(rc = vsel_vf_vo_vf_vf(m0, rc, vcast_vf_f(1.0f));)
304
305 if (__builtin_expect(vtestz_i_vo(m_lrg_args), 0)) { // No large args
306 #if defined (SINCOS)
307 return __mth_return2vectors(rs, rc);
308 #else
309 S(return rs;)
310 C(return rc;)
311 #endif
312 }
313
314 vopmask ninf = vgt_vo_vi2_vi2(vcast_vi2_i(0x7f800000), (vint2)xabs);
315 S(rs = vsel_vf_vo_vf_vf(ninf, rs, vmul_vf_vf_vf(x, vcast_vf_f(0.0)));)
316 C(rc = vsel_vf_vo_vf_vf(ninf, rc, vmul_vf_vf_vf(x, vcast_vf_f(0.0)));)
317
318#if defined (SINCOS)
319 return __mth_return2vectors(rs, rc);
320#else
321 S(return rs;)
322 C(return rc;)
323#endif
324}
