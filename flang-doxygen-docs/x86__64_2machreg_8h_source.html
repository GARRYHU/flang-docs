417/* Machine Register Information -
418 *
419 * This information is in two pieces:
420 * 1. a structure exists for each machine's register class which defines
421 * the attributes of registers.
422 * These attributes define a register set with the following properties:
423 * 1). a set is just an increasing set of numbers,
424 * 2). scratch registers are allocated in increasing order (towards
425 * the globals),
426 * 3). globals are allocated in decreasing order (towards the
427 * scratch registers).
428 * 4). the scratch set that can be changed by a procedure
429 * [min .. first_global-1]
430 * 5). the scratch set that can be changed by an intrinisic
431 * [min .. intrinsic]
432 *
433 * 2. a structure exists for all of the generic register classes which will
434 * map a register type (macros in registers.h) to the appropriate
435 * machine register class.
436 */
437
438/***** Machine Register Table *****/
439
440typedef struct {
441 char min; /* minimum register # */
442 char max; /* maximum register # */
443 char intrinsic; /* last scratch that can be changed by an intrinsic */
444 const char first_global; /* first register # that can be global
445 * Note that the globals are allocated in increasing
446 * order (first_global down to last_global).
447 */
448 const char end_global; /* absolute last register # that can be global. */
449 /* the following two really define the working set
450 of registers that can be assigned. */
451 char next_global; /* next global register # */
452 char last_global; /* last register # that can be global. */
453 char nused; /* number of global registers assigned */
454 const char mapbase; /* offset in register bit vector where
455 this class of MACH_REGS begins. */
456 const char Class; /* class or type of register. code generator needs
457 to know what kind of registers these represent.
458 'i' (integer), 'f' (float stk), 'x' (float xmm) */
459} MACH_REG;

461/***** Register Mapping Table *****/
462

464 char max; /* maximum number of registers */
465 char nused; /* number of registers assigned */
466 char joined; /* non-zero value if registers are formed from
467 * multiple machine registers. 1==>next machine
468 * register is used; other values TBD.
469 */
470 int rcand; /* register candidate list */
471 MACH_REG *mach_reg; /* pointer to struct of the actual registers */
472 INT const_flag; /* flag controlling assignment of consts */
473} REG; /* [rtype] */
