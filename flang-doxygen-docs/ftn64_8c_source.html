13/*
14 * macros for accessing the endian-dependent halves of a 64-bit typeless
15 * quantity when stored in a union of a double and an int[2] array):
16 * _MSH -- most significant half ([1] - little endian, [0] - big endian)
17 * _LSH -- least significant half ([0] - little endian, [1] - big endian)

83ftn_i_shift64(double op1, int cnt)
84{
85 /*
86 logical shift:
87 cnt < 0 ==> shift op1 left by |cnt|
88 cnt > 0 ==> shift op1 right by cnt
89 |cnt| >= 64 ==> result is 0
90 */
91 union u u1;
92 union u u2;
93
94 u1.d = op1;
95 if (cnt >= 64 || cnt <= -64) {
96 u2.i[0] = u2.i[1] = 0;
97 } else if (cnt == 0) { /* 0 == cnt */
98 _MSH(u2) = _MSH(u1);
99 _LSH(u2) = _LSH(u1);
100 } else if (cnt >= 32) { /* 32 <= cnt <= 63 */
101 _MSH(u2) = _LSH(u1) << (cnt - 32);
102 _LSH(u2) = 0;
103 } else if (cnt > 0) { /* 0 < cnt <= 31 */
104 _MSH(u2) = _MSH(u1) << cnt;
105 _MSH(u2) |= (unsigned)_LSH(u1) >> (32 - cnt);
106 _LSH(u2) = _LSH(u1) << cnt;
107 } else if (cnt <= -32) { /* -63 <= cnt <= -32 */
108 _MSH(u2) = 0;
109 _LSH(u2) = (unsigned)_MSH(u1) >> ((-cnt) - 32);
110 } else /* if (cnt < 0) */ { /* -31 <= cnt < 0 */
111 int acnt = -cnt;
112 _MSH(u2) = (unsigned)_MSH(u1) >> acnt;
113 _LSH(u2) = (unsigned)_LSH(u1) >> acnt;
114 _LSH(u2) |= (unsigned)_MSH(u1) << (cnt + 32);
115 }
116 return u2.d;
117}
