154/*
155 * Before returning see if we need to set any of the processor
156 * exception flags.
157 *
158 * Domain error: a is negative, and b is a finite noninteger
159 * we need to raise the Invalid-Operation flag. This can be done by
160 * taking the square root of a negative number.
161 *
162 * Pole error: a is zero and b is negative we need to raise the
163 * divide by zero flag. This can be done by dividing by zero.
164 */
165
166 if (a_is_neg_flag && (!b_is_integer_flag)) {
167 __m512 volatile invop = _mm512_sqrt_ps(a);
168 }
169
170 if (a_is_any_zero_flag && b_is_lt_zero_flag) {
171 __m512 volatile divXzero = _mm512_div_ps(ONE_F_VEC,ZERO_F_VEC);
172 }
173
174 return res_exp;
