27{
28#if (defined EXP_C7)

30 zz = fmaf(zz, z, EXP_C6);
31 zz = fmaf(zz, z, EXP_C5);
32#elif (defined EXP_C6)

51static INLINE void __exp_scalar_kernel(float a, float * poly, int * scale)
52{
53 *scale = 0;
54
55 // clamp range of x to over/underflow bounds to avoid errors in
56 // range reduction procedure leading to unbounded polynomial.
57 // NOTE: we let the NaNs fall through
58 // a >= 3.0f*EXP_HI overflow cannot be recovered by sin/cos
59
60 #define EXP_HI (128.0f * logf(2.0f))
61 if ( a >= 3.0f*EXP_HI )
62 {
63 a = 3.0f*EXP_HI;
64 }
65 if (a <= -3.0f*EXP_HI) //underflow, cannot recover with sincos.
66 {
67 a = -3.0f*EXP_HI;
68 }
69
70 float t = fmaf(a, L2E, 0x1.8p23);
71 float tt = t - 0x1.8p23;
72 // FMA is essential here. If no FMA, need to provide exact multiplication by
73 // LN2_0, and this constant shall be changed to have 10 trailing zeros
74 // so that the product can absorb the 10 bits of tt.
75 float z = fmaf(tt, -LN2_0, a);
76 z = fmaf(tt, -LN2_1, z);
77
78 int exp = F2I(t);
79 // sign-extend integer exp:
80 // wipe dummy FP sign, FP exponent field and two leading FP mantissa bits
81 // (1 bit implicit), which are leftovers from right-shifter
82 exp <<= 10;
83 exp >>= 10;
84
85 // compute polynomial approximation, it shall be on the order
86 // of exp() in the reduced range [-ln2/2, ln2/2], so zz is in [1/sqrt(2), sqrt(2)] in (1/2, 2)
87 float zz = __exp_poly(z);
88
89 // exp scaling factor is now somewhere in [-128 - 24; 128 + 128 + 24]
90 // Subtract 64 in order to
91 // compensate for denormals in sin(), also make exp range symmetric.
92 // Adjust poly accordingly too.
93 exp -= 64;
94 // Here we multiply by a FP constant instead of integer addition to exp bits
95 // - to preserve the NaNs
96 *poly = zz * I2F((64 + 127) << 23);
97 // Now exp is in [-128 - 24 - 64; 128 + 24 + 64]
98 // less than 256 in abs value, so takes 9 bits with sign
99 // and new poly is in (2^63, 2^65)
100 *scale = exp;
101 return;
102}

105{
106 vfloat zz = vSETf(EXP_C6);

108 zz = vfma_vf_vf_vf_vf(zz, z, vSETf(EXP_C4));
109 zz = vfma_vf_vf_vf_vf(zz, z, vSETf(EXP_C3));
110 zz = vfma_vf_vf_vf_vf(zz, z, vSETf(EXP_C2));
111 zz = vfma_vf_vf_vf_vf(zz, z, vSETf(EXP_C1));

116static void INLINE
117__vexp_kernel(vfloat vx, vfloat * vpoly, vfloat * vscale)
118{
119 // This algorithm computes exp(vx) in a form of
120 // 2^(vscale) * vpoly, unevaluated. vscale is an integer.
121 // The intended use of this form is for subsequent
122 // multiplication of vpoly by sin/cos, which can be small.
123 // We don't know the values of sin/cos apriori, so need
124 // to compute exp() with the extended range, thus the need
125 // to hold the scale bits in a separate integer, wider than
126 // 8 bits provided by the IEEE binary32 format.
127 // To avoid potential loss of accuracy in denormals we
128 // make sure that vpoly * sin() is a normal number - for
129 // that we shift some of the scaling from vscale to vpoly.
130 // Later scaling by 2^(vscale) may still result in a denormal
131 // and the loss of accuracy, but in this case it will be
132 // bound by ~1 ulp, which is tolerable for the implementation.
133
134 // exp algorithm outline: we reduce argument to +-ln2/2 interval
135 // by representing x = N*ln2 + z, in this case exp(x) = 2^N * exp(z).
136 // We want N to be integer and thus it is obtained as:
137 // a) N = round_to_nearest_int(x * 1/ln2)
138 // And reduced argument z is:
139 // b) z = x - N*ln2
140 // We need to guarantee that |z| < ln2/2 and we need to estimate
141 // the error introduced by reduction too.
142 // exp(x) can quickly over/underflow, so given the bounds on argument
143 // x in which we want to compute exp(), we can infer the bounds on
144 // N and decide on the precision needs in finite approximations
145 // of 1/ln2 and ln2 constants.
146 //
147 // Once the reduced argument z is known, we compute the exp(z) as
148 // a polynomial approximation. We would like exp(0) to
149 // be exactly 1, so we chose the polynomial coefficients accordingly.
150
151 // FIXME: compute overflow threshold more accurately
152
153 // clamp range of x to over/underflow bounds to avoid errors in
154 // range reduction procedure leading to unbounded polynomial.
155 vopmask mover = vgt_vo_vf_vf(vx, vSETf(3.0f*EXP_HI));
156 vx = vsel_vf_vo_vf_vf(mover, vSETf(3.0f*EXP_HI), vx);
157 // exp underflows for x <= EXP_LO, it cannot be recovered with later
158 // multiply by sincos, which is <= 1
159 vopmask munder = vle_vo_vf_vf(vx, vSETf(-3.0*EXP_HI));
160 vx = vsel_vf_vo_vf_vf(munder, vSETf(-3.0*EXP_HI), vx);
161 // tt = [x / ln2] = [x * log2(e)], convert to integer using right-shifter
162 const vfloat vRS = vSETf(0x1.8p23);
163 // least significant bits of t now contain an integer rounded
164 // according to current rounding mode, default: to nearest
165 // TODO: this algorithm will fail in directed rounding mode because
166 // of the over/underestimate in t and thus tt.
167 vfloat t = vfma_vf_vf_vf_vf(vx, vSETf(L2E), vRS); PRINT(t);
168 // subtract right-shifter to obtain the integer as a normalized FP number
169 vfloat tt= vsub_vf_vf_vf(t, vRS);
170
171 // FMA is essential here. If no FMA, need to provide exact multiplication by
172 // LN2_0, and this constant shall be changed to have e.g. 10 trailing zeros
173 // so that the product can absorb the 10 bits of tt.
174 vfloat z = vfma_vf_vf_vf_vf(tt, vSETf(-LN2_0), vx);
175 z = vfma_vf_vf_vf_vf(tt, vSETf(-LN2_1), z);
176
177 vint2 exponent = vF2I(t);
178 // sign-extend integer exp:
179 // wipe dummy FP sign, FP exponent field and two leading FP mantissa
180 // bits (1 bit implicit), which are leftovers from right-shifter.
181 exponent = vsll_vi2_vi2_i(exponent, 10); PRINT(exponent);
182 exponent = vsra_vi2_vi2_i(exponent, 10); PRINT(exponent);
183
184 // compute polynomial approximation, it shall be on the order
185 // of exp() in the reduced range [-ln2/2, ln2/2], so zz is
186 // in [1/sqrt(2), sqrt(2)] or in (1/2, 2)
187 vfloat zz = __vexp_poly(z); PRINT(zz);
188
189 // exponent scaling factor is now somewhere in [-128 - 24; 128 + 128 + 24].
190 // Subtract 64 from scaling and add it back to polynomial so that later
191 // polynomial * sin() always results in normalized numbers.
192 // Plus it also makes exponent range symmetric:
193 // [-128 - 24 - 64; 128 + 24 + 64], only 9 bits of
194 // storage together with the sign.
195 exponent = vsub_vi2_vi2_vi2(exponent, vSETi(64)); PRINT(exponent);
196 // new poly is in (2^63, 2^65)
197 zz = vmul_vf_vf_vf(zz, vI2F(vSETi((64+127)<<23))); PRINT(zz);
198
199 *vpoly = zz;
200 *vscale = vI2F(exponent);
201 return;
202}
