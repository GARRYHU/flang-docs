17/*
18 * Hash of new uses. We were previously using a linear search of opt.useb each
19 * time a new use was added for O(N^2), which was impacting compile times.

58/*
59 * Turn the fields of the use into a hash_key_t. Hash function requires to
60 * avoid clustering in lower bits of hash.
61 *
62 * high: In the backend is ilix, frontend is addrx
63 * mid: In the backend is nmex, frontend is nmex
64 * low: In the backend is iltx, frontend is stdx
65 */
66static hash_key_t
67use_hash_key(bool high_bit, bool use_high, int high, int mid, int low)
68{
69 long long key;
70 // lower USE_HASH_BITS bits of mask set, upper bits cleared.
71 long long mask = ((long long)1 << USE_HASH_BITS)-1;
72
73 // This only works if the number of nme, ili, and ilt are each less than 2^USE_HASH_BITS
74 if (use_high) {
75 assert(high < (int)mask, "too many ILIs to hash", high, ERR_Fatal);
76 }
77 assert(mid < (int)mask, "too many NMEs to hash", mid, ERR_Fatal);
78 assert(low < (int)mask, "too many ILTs to hash", low, ERR_Fatal);
79
80 key = 0;
81 if (high_bit)
82 key |= (long long)1; // top bit high_bit flag
83 key <<= 1;
84 if (use_high && !high_bit)
85 key |= ((long long)high & mask);
86 key <<= USE_HASH_BITS;
87 key |= (mid & mask);
88 key <<= USE_HASH_BITS;
89 key |= (low & mask);
90
91 assert((void*)key != NULL, "NULL key computed", key, ERR_Fatal);
92 assert((void*)key != (void*)~0UL, "~0UL key computed", key, ERR_Fatal);
93
94 return (hash_key_t)key;
95} /* use_hash_key */

97/*
98 * From the fields of a new use, create a hash key, and insert the use into the
99 * use_hash with the key.
100 */
101void
102use_hash_insert(int usex, bool high_bit, bool use_match_ili, int ilix, int nmex, int iltx)
103{
104 hash_key_t key = use_hash_key(high_bit, use_match_ili, ilix, nmex, iltx);
105 hash_data_t data = INT2HKEY(usex);
106 hashmap_replace(use_hash, key, &data);
107} /* use_hash_insert */

109/*
110 * From the fields of a potential new use, create a hash key, and check the
111 * use_hash to see of that potential use has already been inserted. Return the
112 * existing use, if it exists.
113 *
114 * We were previously using a linear search of opt.useb each time a new use was
115 * added for O(N^2), which was impacting compile times. Replacing the linear
116 * search with a hash lookup results in O(N).
117 */
118int
119use_hash_lookup(bool high_bit, bool use_match_ili, int ilix, int nmex, int iltx)
120{
121 hash_data_t usex = NULL;
122 hash_key_t key = use_hash_key(high_bit, use_match_ili, ilix, nmex, iltx);
123 if (hashmap_lookup(use_hash, key, &usex)) {
124 return HKEY2INT(usex);
125 }
126 return 0;
127} /* use_hash_lookup */

151/*
152 * use_hash utility function to compare the results of the hash search (usex)
153 * to the results of the older and slower linear search (lusex).
154 */
155void
156use_hash_check(int usex, int lusex, bool found)
157{
158 if (found && (usex == lusex)) {
159 // Success. Both methods found the same use.
160 } else if (found && !usex) {
161 // Failure. Linear search found a use, but the hash lookup found nothing.
162 assert(0, "use_hash failed should be found", lusex, ERR_Fatal);
163 } else if (!found && usex) {
164 // Failure. Linear search found nothing, but hash lookup found a use.
165 assert(0, "use_hash failed shouldn't be found", usex, ERR_Fatal);
166 } else {
167 // Success. Boh methods found no use.
168 }
169} /* use_hash_check */
170#endif /* DEBUG */
