56FN_PROTOTYPE(mth_i_dtanh)(double x)
57{
58 /*
59 The definition of tanh(x) is sinh(x)/cosh(x), which is also equivalent
60 to the following three formulae:
61 1. (exp(x) - exp(-x))/(exp(x) + exp(-x))
62 2. (1 - (2/(exp(2*x) + 1 )))
63 3. (exp(2*x) - 1)/(exp(2*x) + 1)
64 but computationally, some formulae are better on some ranges.
65 */
66 static const double thirtytwo_by_log2 =
67 4.61662413084468283841e+01, /* 0x40471547652b82fe */
68 log2_by_32_lead = 2.16608493356034159660e-02, /* 0x3f962e42fe000000 */
69 log2_by_32_tail = 5.68948749532545630390e-11, /* 0x3dcf473de6af278e */
70 large_threshold = 20.0; /* 0x4034000000000000 */
71
72 __UINT8_T ux, aux, xneg;
73 double y, z, p, z1, z2;
74 int m;
75
76 /* Special cases */
77
78 GET_BITS_DP64(x, ux);
79 aux = ux & ~SIGNBIT_DP64;
80 if (aux < 0x3e30000000000000) /* |x| small enough that tanh(x) = x */
81 {
82 if (aux == 0)
83 return x; /* with no inexact */
84 else
85 return val_with_flags(x, AMD_F_INEXACT);
86 } else if (aux > 0x7ff0000000000000) /* |x| is NaN */
87 return x + x;
88
89 xneg = (aux != ux);
90
91 y = x;
92 if (xneg)
93 y = -x;
94
95 if (y > large_threshold) {
96 /* If x is large then exp(-x) is negligible and
97 formula 1 reduces to plus or minus 1.0 */
98 z = 1.0;
99 } else if (y <= 1.0) {
100 double y2;
101 y2 = y * y;
102 if (y < 0.9) {
103 /* Use a [3,3] Remez approximation on [0,0.9]. */
104 z = y +
105 y * y2 * (-0.274030424656179760118928e0 +
106 (-0.176016349003044679402273e-1 +
107 (-0.200047621071909498730453e-3 -
108 0.142077926378834722618091e-7 * y2) *
109 y2) *
110 y2) /
111 (0.822091273968539282568011e0 +
112 (0.381641414288328849317962e0 +
113 (0.201562166026937652780575e-1 +
114 0.2091140262529164482568557e-3 * y2) *
115 y2) *
116 y2);
117 } else {
118 /* Use a [3,3] Remez approximation on [0.9,1]. */
119 z = y +
120 y * y2 * (-0.227793870659088295252442e0 +
121 (-0.146173047288731678404066e-1 +
122 (-0.165597043903549960486816e-3 -
123 0.115475878996143396378318e-7 * y2) *
124 y2) *
125 y2) /
126 (0.683381611977295894959554e0 +
127 (0.317204558977294374244770e0 +
128 (0.167358775461896562588695e-1 +
129 0.173076050126225961768710e-3 * y2) *
130 y2) *
131 y2);
132 }
133 } else {
134 /* Compute p = exp(2*y) + 1. The code is basically inlined
135 from exp_amd. */
136
137 splitexp(2 * y, 1.0, thirtytwo_by_log2, log2_by_32_lead, log2_by_32_tail,
138 &m, &z1, &z2);
139 p = scaleDouble_2(z1 + z2, m) + 1.0;
140
141 /* Now reconstruct tanh from p. */
142 z = (1.0 - 2.0 / p);
143 }
144
145 if (xneg)
146 z = -z;
147 return z;
148}
