57/* macros to put character length arguments in their place.
58 DCHAR declares a character pointer argument.
59 DCLEN declares a character length argument. Since DCLEN may have an
60 empty definition, no commas should be used before or after a DCLEN
61 reference in a dummy argument list.
62 CADR gets the character pointer.
63 CLEN gets the character length. */
64
65#define __CLEN_T size_t
66#define DCHAR(ARG) char *ARG##_adr
67#define DCLEN(ARG) , int ARG##_len
68#define DCLEN64(ARG) , __CLEN_T ARG##_len
69#define CADR(ARG) (ARG##_adr)
70#define CLEN(ARG) (ARG##_len)
