44/** \brief Add an ilt
45 *
46 * Add an ilt inserting it after the ilt "after"; ilix locates
47 * the ili which represents the root of the ili tree
48 */
49int
50addilt(int after, int ilix)
51{
52 int i;
53 ILT *p;
54 ILTY_KIND type;
55 ILI_OP opc;
56
57 i = STG_NEXT_FREELIST(iltb);
58 p = iltb.stg_base + i;
59 p->flags.all = 0;
60 p->prev = after;
61 p->next = ILT_NEXT(after);
62 p->lineno = gbl.lineno;
63 p->order = -1;
64 ILT_NEXT(after) = i;
65 ILT_PREV(p->next) = i;
66 p->ilip = ilix;
67 opc = ILI_OPC(ilix);
68 type = IL_TYPE(opc);
69 if (type == ILTY_STORE)
70 p->flags.bits.st = 1;
71 else if (type == ILTY_BRANCH)
72 p->flags.bits.br = 1;
73 p->flags.bits.ex = iltb.callfg;
74 bihb.callfg |= iltb.callfg;
75 bihb.ldvol |= iltb.ldvol;
76 bihb.stvol |= iltb.stvol;
77 bihb.qjsrfg |= iltb.qjsrfg;
78 if (after == 0 && p->next == 0) {
79 /* this is the first ILT for this block */
80 fihb.currfindex = fihb.nextfindex;
81 fihb.currftag = fihb.nextftag;
82 }
83 p->findex = fihb.currfindex;
84 iltb.callfg = 0;
85 iltb.ldvol = 0;
86 iltb.stvol = 0;
87 iltb.qjsrfg = false;
88 return (i);
89}

91/** \brief Delete an ilt from a block which is "read" and reuse it
92 */
93void
94delilt(int iltx)
95{
96 int prev, next, ignore, ilip;
97
98 next = ILT_NEXT(iltx);
99 prev = ILT_PREV(iltx);
100 /* preserve the ILT_ILIP field, ILT_IGNORE flag, set ILT_FREE flag */
101 ignore = ILT_IGNORE(iltx);
102 ilip = ILT_ILIP(iltx);
103 ILT_PREV(next) = prev;
104 ILT_NEXT(prev) = next;
105 STG_ADD_FREELIST(iltb, iltx);
106 ILT_IGNORE(iltx) = ignore;
107 ILT_ILIP(iltx) = ilip;
108 ILT_FREE(iltx) = 1;
109}

111/** \brief delete an ilt where the block may not be "read", and possibly reuse
112 * it
113 *
114 * \param iltx ilt to be deleted
115 * \param bihx bih of block from which ilt is deleted (0 => read)
116 * \param reuse true if ilt is to be reused
117 */
118void
119unlnkilt(int iltx, int bihx, bool reuse)
120{
121 int i, j;
122
123 if (bihx) {
124 i = ILT_PREV(iltx);
125 j = ILT_NEXT(iltx);
126 if (j)
127 ILT_PREV(j) = i;
128 else
129 BIH_ILTLAST(bihx) = i;
130 if (i)
131 ILT_NEXT(i) = j;
132 else
133 BIH_ILTFIRST(bihx) = j;
134 } else {
135 j = ILT_NEXT(iltx);
136 i = ILT_PREV(j) = ILT_PREV(iltx);
137 ILT_NEXT(i) = j;
138 }
139 if (reuse) {
140 /* preserve the ILT_ILIP field, ILT_IGNORE flag */
141 int ignore = ILT_IGNORE(iltx);
142 int ilip = ILT_ILIP(iltx);
143 STG_ADD_FREELIST(iltb, iltx);
144 ILT_IGNORE(iltx) = ignore;
145 ILT_ILIP(iltx) = ilip;
146 }
147 ILT_FREE(iltx) = 1;
148 /* else: hopefully, scans will still work if we start with an ilt which
149 * was removed but not reused
150 */
151
152}

154/*
155 * move an ilt to this BIH after removing it with unlnkilt
156 * iltx = ilt to be added
157 * bihx = bih of block to which ilt is to be added
158 */
159void
160relnkilt(int iltx, int bihx)
161{
162 int j;
163 j = BIH_ILTLAST(bihx);
164 ILT_PREV(iltx) = j;
165 if (j)
166 ILT_NEXT(j) = iltx;
167 else
168 BIH_ILTFIRST(bihx) = iltx;
169 BIH_ILTLAST(bihx) = iltx;
170 ILT_FREE(iltx) = 0;
171} /* relnkilt */

175/** \brief Move an ilt before another ilt
176 */
177void
178moveilt(int iltx, int before)
179{
180 int i, j;
181
182 /** remove iltx from list **/
183 i = ILT_PREV(iltx);
184 ILT_NEXT(i) = j = ILT_NEXT(iltx);
185 ILT_PREV(j) = i;
186 /** insert iltx before 'before' **/
187 ILT_PREV(iltx) = i = ILT_PREV(before);
188 ILT_NEXT(i) = iltx;
189 ILT_PREV(before) = iltx;
190 ILT_NEXT(iltx) = before;
191}

195/**
196 search the ili subtree located by ilix for functions and creating an ilt for
197 each one found. The static variable iltcur (local to this module indicates
198 where an ilt is added. iltcur is updated to locate the new ilt
199 */
200static void
201srcfunc(int ilix)
202{
203 int noprs; /* number of lnk operands in ilix */
204 int i; /* index variable */
205 ILI_OP opc; /* ili opcode of ilix */
206
207 if (IL_TYPE(opc = ILI_OPC(ilix)) == ILTY_PROC && opc >= IL_JSR) {
208 iltb.callfg = 1;
209 iltcur = addilt(iltcur, ilix); /* create a function ilt */
210 } else if (opc == IL_DFRDP && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
211 iltb.callfg = 1;
212 iltcur = addilt(iltcur, ad1ili(IL_FREEDP, ilix));
213 } else if (opc == IL_DFRSP && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
214 iltb.callfg = 1;
215 iltcur = addilt(iltcur, ad1ili(IL_FREESP, ilix));
216 }
217 else if (opc == IL_DFRCS && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
218 iltb.callfg = 1;
219 iltcur = addilt(iltcur, ad1ili(IL_FREECS, ilix));
220 }
221#ifdef LONG_DOUBLE_FLOAT128
222 else if (opc == IL_FLOAT128RESULT && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
223 iltb.callfg = 1;
224 iltcur = addilt(iltcur, ad1ili(IL_FLOAT128FREE, ilix));
225 }
226#endif /* LONG_DOUBLE_FLOAT128 */
227 else {
228 noprs = ilis[opc].oprs;
229 for (i = 1; i <= noprs; i++) {
230 if (IL_ISLINK(opc, i))
231 srcfunc((int)(ILI_OPND(ilix, i)));
232 }
233 }
234}

238/** \brief Reduce an ilt to a sequence of function ilts
239 *
240 * Reduces the ili tree located by ilix producing ilts which locate function
241 * ilis occuring in ilix. This routine sets iltucr with iltx which indicates
242 * where ilts are to be added and calls srcfunc.
243 */
244int
245reduce_ilt(int iltx, int ilix)
246{
247 iltcur = iltx; /* avoid passing iltx recursively */
248 srcfunc(ilix);
249 return (iltcur);
250}

427 fprintf(ff, "\t iff [%d]", ILT_GUARD(p));
428 }
429#endif
430 if (ILT_INV(p))
431 fprintf(ff, " INV");
432 fprintf(ff, "\n");
433#if DEBUG
434 if (DBGBIT(10, 128)) {
435 dmpilitree((int)ILT_ILIP(p));
436 }
437#endif
438 }
439 }
440 assert(q == BIH_ILTLAST(bihx), "dmpilt: bad end of block", bihx, ERR_Severe);
441 iltb.privtmp = 0;
442}

456/** \brief Write out an ilt/ili block given its bih
457 *
458 * Write out ilts for the block denoted by bih. Various flags have already
459 * been set in bih. This routine buffers up the ilt block in the bih area.
460 */
461void
462wrilts(int bihx)
463{
464 BIH_ILTFIRST(bihx) = ILT_NEXT(0);
465 BIH_ILTLAST(bihx) = ILT_PREV(0);
466 bihb.callfg = 0;
467 bihb.ldvol = 0;
468 bihb.stvol = 0;
469 bihb.qjsrfg = 0;
470 if (bihx != gbl.entbih)
471 BIH_FINDEX(bihx) = fihb.currfindex;
472#ifdef BIH_FTAG
473 if (bihx != gbl.entbih) {
474 BIH_FINDEX(bihx) = fihb.currfindex;
475 BIH_FTAG(bihx) = fihb.currftag;
476 ++fihb.currftag;
477 if ((fihb.currftag > fihb.nextftag) && (fihb.currfindex == fihb.nextfindex))
478 fihb.nextftag = fihb.currftag;
479 }
480#endif
481#if DEBUG
482 if (flg.dbg[8] & 1)
483 dmpilt(bihx);
484#endif
485}

487/** \brief Read in an ilt/ili block
488 *
489 * "Read" in the block specified by bihx. the 0th entry in the ILT area (next
490 * and prev) are set to the first and last ilts respectively
491 */
492void
493rdilts(int bihx)
494{
495 if (BIH_RD(bihx) != 0) {
496 ILT_NEXT(0) = BIH_ILTFIRST(bihx);
497 ILT_PREV(0) = BIH_ILTLAST(bihx);
498 }
499 bihb.callfg = BIH_EX(bihx);
500 bihb.ldvol = BIH_LDVOL(bihx);
501 bihb.stvol = BIH_STVOL(bihx);
502 bihb.qjsrfg = BIH_QJSR(bihx);
503#ifdef BIH_FTAG
504 fihb.nextfindex = fihb.currfindex = BIH_FINDEX(bihx);
505 fihb.nextftag = fihb.currftag = BIH_FTAG(bihx);
506#endif
507}

509/*
510 * save one message
511 * call ccff_info

513void *
514ccff_ilt_info(int msgtype, const char *msgid, int iltx, int bihx, const char *message,
515 ...)
516{
517 va_list argptr;
518 int fihx, lineno;
519 va_start(argptr, message);
520
521 fihx = -1;
522 lineno = -1;
523 if (iltx > 0) {
524 fihx = ILT_FINDEX(iltx);
525 lineno = ILT_LINENO(iltx);
526 }
527 if (fihx <= 0 || lineno <= 0) {
528 fihx = BIH_FINDEX(bihx);
529 lineno = BIH_LINENO(bihx);
530 }
531 if (fihx == 0)
532 return NULL; /* no info */
533 return _ccff_info(msgtype, msgid, fihx, lineno, NULL, NULL, NULL, message,
534 argptr);
535} /* ccff_ilt_info */
