22// handles large cases as well as special cases such as infinities and NaNs
23__m128d __pgm_exp_d_vec128_slowpath(__m128d const a, __m128i const i, __m128d const t, __m128d const z)
24{
25#if defined(__clang__) && defined(TARGET_ARM64)
26 __m128d const INF = (__m128d)((long double)_mm_set1_epi64x(INF_D));
27#else
28 __m128d const INF = (__m128d)_mm_set1_epi64x(INF_D);
29#endif
30 __m128d const ZERO = _mm_set1_pd(ZERO_D);
31 __m128i const HI_ABS_MASK = _mm_set1_epi64x(HI_ABS_MASK_D);
32#if defined(__clang__) && defined(TARGET_ARM64)
33 __m128d const UPPERBOUND_1 = (__m128d)((long double)_mm_set1_epi64x(UPPERBOUND_1_D));
34 __m128d const UPPERBOUND_2 = (__m128d)((long double)_mm_set1_epi64x(UPPERBOUND_2_D));
35#else
36 __m128d const UPPERBOUND_1 = (__m128d)_mm_set1_epi64x(UPPERBOUND_1_D);
37 __m128d const UPPERBOUND_2 = (__m128d)_mm_set1_epi64x(UPPERBOUND_2_D);
38#endif
39 __m128i const MULT_CONST = _mm_set1_epi64x(MULT_CONST_D);
40
41#if defined(__clang__) && defined(TARGET_ARM64)
42 __m128d abs_lt = (__m128d)((long double)_mm_and_si128((__m128i)((long double)a), HI_ABS_MASK));
43#else
44 __m128d abs_lt = (__m128d)_mm_and_si128((__m128i)a, HI_ABS_MASK);
45#endif
46
47 __m128d slowpath_mask = (__m128d)_mm_cmp_pd(abs_lt, UPPERBOUND_1, _CMP_LT_OS);
48 __m128d lt_zero_mask = _mm_cmp_pd(a, ZERO, _CMP_LT_OS); // compute a < 0.0
49
50 __m128d a_plus_inf = _mm_add_pd(a, INF); // check if a is too big
51 __m128d zero_inf_blend = _mm_blendv_pd(a_plus_inf, ZERO, lt_zero_mask);
52
53 __m128d accurate_scale_mask = (__m128d)_mm_cmp_pd(abs_lt, UPPERBOUND_2, _CMP_LT_OS);
54
55 // compute accurate scale
56 __m128i k = _mm_srli_epi64(i, 1); // k = i / 2
57 __m128i i_scale_acc = _mm_slli_epi64(k, SCALE_D); // shift to HI and shift 20
58
59 k = _mm_sub_epi32(i, k); // k = i - k
60 __m128i i_scale_acc_2 = _mm_slli_epi64(k, SCALE_D); // shift to HI and shift 20
61#if defined(__clang__) && defined(TARGET_ARM64)
62 __m128d multiplier = (__m128d)((long double)_mm_add_epi64(i_scale_acc_2, MULT_CONST));
63 __m128d res = (__m128d)((long double)_mm_add_epi32(i_scale_acc, (__m128i)((long double)t)));
64#else
65 __m128d multiplier = (__m128d)_mm_add_epi64(i_scale_acc_2, MULT_CONST);
66 __m128d res = (__m128d)_mm_add_epi32(i_scale_acc, (__m128i)t);
67#endif
68 res = _mm_mul_pd(res, multiplier);
69
70 __m128d slowpath_blend = _mm_blendv_pd(zero_inf_blend, res, accurate_scale_mask);
71 return _mm_blendv_pd(slowpath_blend, z, slowpath_mask);
72}

75{
76 __m128d const L2E = _mm_set1_pd(L2E_D);
77 __m128d const NEG_LN2_HI = _mm_set1_pd(NEG_LN2_HI_D);
78 __m128d const NEG_LN2_LO = _mm_set1_pd(NEG_LN2_LO_D);
79 __m128d const ZERO = _mm_set1_pd(ZERO_D);
80#if defined(__clang__) && defined(TARGET_ARM64)

82#else

84#endif

88 __m128d const EXP_POLY_9 = _mm_set1_pd(EXP_POLY_9_D);

93 __m128d const EXP_POLY_4 = _mm_set1_pd(EXP_POLY_4_D);
94 __m128d const EXP_POLY_3 = _mm_set1_pd(EXP_POLY_3_D);
95 __m128d const EXP_POLY_2 = _mm_set1_pd(EXP_POLY_2_D);
96 __m128d const EXP_POLY_1 = _mm_set1_pd(EXP_POLY_1_D);

98
99 __m128d const DBL2INT_CVT = _mm_set1_pd(DBL2INT_CVT_D);
100#if defined(__clang__) && defined(TARGET_ARM64)
101 __m128d const UPPERBOUND_1 = (__m128d)((long double)_mm_set1_epi64x(UPPERBOUND_1_D));
102 __m128d const UPPERBOUND_2 = (__m128d)((long double)_mm_set1_epi64x(UPPERBOUND_2_D));
103#else
104 __m128d const UPPERBOUND_1 = (__m128d)_mm_set1_epi64x(UPPERBOUND_1_D);
105 __m128d const UPPERBOUND_2 = (__m128d)_mm_set1_epi64x(UPPERBOUND_2_D);
106#endif
107
108 __m128i const MULT_CONST = _mm_set1_epi64x(MULT_CONST_D);
109 __m128i const HI_ABS_MASK = _mm_set1_epi64x(HI_ABS_MASK_D);
110
111 // calculating exponent; stored in the LO of each 64-bit block
112#if defined(__clang__) && defined(TARGET_ARM64)
113 __m128i i = (__m128i) ((long double)_mm_fmadd_pd(a, L2E, DBL2INT_CVT));
114#else
115 __m128i i = (__m128i) _mm_fmadd_pd(a, L2E, DBL2INT_CVT);
116#endif
117
118 // calculate mantissa
119 //fast mul rint
120 __m128d t = _mm_sub_pd (_mm_fmadd_pd(a, L2E, DBL2INT_CVT), DBL2INT_CVT);
121 __m128d m = _mm_fmadd_pd (t, NEG_LN2_HI, a);
122 m = _mm_fmadd_pd (t, NEG_LN2_LO, m);
123
124 // evaluate highest 8 terms of polynomial with estrin, then switch to horner
125 __m128d z10 = _mm_fmadd_pd(EXP_POLY_11, m, EXP_POLY_10);
126 __m128d z8 = _mm_fmadd_pd(EXP_POLY_9, m, EXP_POLY_8);
127 __m128d z6 = _mm_fmadd_pd(EXP_POLY_7, m, EXP_POLY_6);
128 __m128d z4 = _mm_fmadd_pd(EXP_POLY_5, m, EXP_POLY_4);
129
130 __m128d m2 = _mm_mul_pd(m, m);
131 z8 = _mm_fmadd_pd(z10, m2, z8);
132 z4 = _mm_fmadd_pd(z6, m2, z4);
133
134 __m128d m4 = _mm_mul_pd(m2, m2);
135 z4 = _mm_fmadd_pd(z8, m4, z4);
136
137 t = _mm_fmadd_pd(z4, m, EXP_POLY_3);
138 t = _mm_fmadd_pd(t, m, EXP_POLY_2);
139 t = _mm_fmadd_pd(t, m, EXP_POLY_1);
140 t = _mm_fmadd_pd(t, m, EXP_POLY_0);
141
142 // fast scale
143 __m128i i_scale = _mm_slli_epi64(i, SCALE_D);
144#if defined(__clang__) && defined(TARGET_ARM64)
145 __m128d z = (__m128d)((long double)_mm_add_epi32(i_scale, (__m128i)((long double)t)));
146 __m128d abs_a = (__m128d)((long double)_mm_and_si128((__m128i)((long double)a), HI_ABS_MASK));
147#else
148 __m128d z = (__m128d)_mm_add_epi32(i_scale, (__m128i)t);
149 __m128d abs_a = (__m128d)_mm_and_si128((__m128i)a, HI_ABS_MASK);
150#endif
151
152#if defined(TARGET_LINUX_POWER)
153 int exp_slowmask = _vec_any_nz((__m128i)_mm_cmpgt_epi64((__m128i)abs_a, (__m128i)UPPERBOUND_1));
154#else
155#if defined(__clang__) && defined(TARGET_ARM64)
156 int exp_slowmask = _mm_movemask_epi8(_mm_cmpgt_epi64((__m128i)((long double)abs_a), (__m128i)((long double)UPPERBOUND_1)));
157#else
158 int exp_slowmask = _mm_movemask_epi8(_mm_cmpgt_epi64((__m128i)abs_a, (__m128i)UPPERBOUND_1));
159#endif
160#endif
161
162// if (exp_slowmask) {
163// return __pgm_exp_d_vec128_slowpath(a, i, t, z);
164// }
165
166
167// int exp_slowmask = _mm_movemask_pd(_mm_cmp_pd(abs_a, UPPERBOUND_1, _CMP_GE_OS));
168
169 if (__builtin_expect(exp_slowmask, 0)) {
170 return __pgm_exp_d_vec128_slowpath(a, i, t, z);
171 }
172
173 return z;
174}
