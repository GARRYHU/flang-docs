29typedef unsigned char epoch_t;

31/* A visit_info holds information about an ILI node */
32typedef struct {
33 /* This record is considered absent unless epoch==current_epoch. */
34 epoch_t epoch;
35} visit_info;
36
37static visit_info *visited_ili;
38static int visited_size; /* Number of elements pointed to by visited_ili. */
39static epoch_t current_epoch;
40static int walk_depth;
41
42/* If doing deep ILI verification, prepare for remembering which ILI have
43 already been walked. Must be paired with end_walk(level).
44 No-op if level does not require any remembering. */
45static void
46begin_walk(VERIFY_LEVEL level)
47{
48 if (level >= VERIFY_ILI_DEEP) {
49 if (++walk_depth == 1) {
50 ++current_epoch;
51 if (current_epoch == 0) {
52 int i;
53 /* epoch wrapped - wipe the slate clean */
54 for (i = 0; i < visited_size; ++i)
55 visited_ili[i].epoch = 0;
56 current_epoch = 1;
57 }
58 }
59 }
60}
61
62/* If at outermost walk, forget which ILI have already been walked.
63 The level *must* match the level in the corresponding call to begin_walk().
64 */
65static void
66end_walk(VERIFY_LEVEL level)
67{
68 if (level >= VERIFY_ILI_DEEP) {
69 VERIFY(walk_depth >= 1, "walk not started?");
70 --walk_depth;
71 }
72}

74static void
75ili_mark_first_visit(int ilix)
76{
77 int old_size;
78 visit_info *v;
79 DEBUG_ASSERT(walk_depth > 0, "internal error in verifier: not in a walk");
80 DEBUG_ASSERT(0 < ilix, "internal error in verifier itself");
81 old_size = visited_size;
82 if (ilix >= old_size) {
83 int i;
84 int new_size = 1;
85 while (ilix >= new_size)
86 new_size *= 2;
87 NEED(ilix + 1, visited_ili, visit_info, visited_size, new_size);
88 for (i = old_size; i < visited_size; ++i)
89 visited_ili[i].epoch = 0;
90 }
91 DEBUG_ASSERT(ilix < visited_size, "internal error in verifier itself");
92 v = &visited_ili[ilix];
93 VERIFY(v->epoch != current_epoch, "ILI already visited!");
94 v->epoch = current_epoch;
95}

97/** NULL if ilix has not been visited yet. */
98static visit_info *
99ili_visit_info(int ilix)
100{
101 DEBUG_ASSERT(walk_depth > 0, "internal error in verifier: not in a walk");
102 if (0 < ilix && ilix < visited_size &&
103 visited_ili[ilix].epoch == current_epoch)
104 return &visited_ili[ilix];
105 else
106 return NULL;
107}

109/** Return true if jth operand of opc is allowed to have operation j_opc
110 contrary to expectations of IL_OPRFLAG(opc, j). This routine
111 is called infrequently so consider readability over speed when
112 adding another case. */
113static bool
114is_known_bug(ILI_OP opc, int j, ILI_OP j_opc)
115{
116 /* r is the kind of operand supplied. */
117 ILIA_RESULT r = IL_RES(j_opc);
118 /* o is the kind of operand expected. */
119 ILIO_KIND o = IL_OPRFLAG(opc, j);
120 if (opc == IL_KNEG && o == ILIO_KRLNK && r == ILIA_AR)
121 return true;
122 if (opc == IL_KADD && o == ILIO_KRLNK && r == ILIA_IR)
123 return true;
124#ifdef TARGET_X86
125 if ((opc == IL_DASPSP || opc == IL_MVSPSP) && o == ILIO_DPLNK && r == ILIA_CS)
126 return true;
127#endif
128 if (opc == IL_STKR && o == ILIO_KRLNK && (r == ILIA_IR || r == ILIA_AR) &&
129 j == 1)
130 return true;
131 if ((opc == IL_FREEIR || opc == IL_CSEIR) && o == ILIO_IRLNK && r == ILIA_KR)
132 return true;
133 if (opc == IL_UKADD && o == ILIO_KRLNK && r == ILIA_IR)
134 return true;
135 if (opc == IL_ST && o == ILIO_IRLNK && r == ILIA_KR && j == 1)
136 return true;
137 if (opc == IL_IMUL && o == ILIO_IRLNK && r == ILIA_KR && j == 2)
138 return true;
139 if (opc == IL_IDIV && o == ILIO_IRLNK && r == ILIA_KR && j == 2)
140 return true;
141 if ((opc == IL_UKCMP || opc == IL_UKCJMP) && o == ILIO_KRLNK && r == ILIA_AR)
142 return true;
143 if (opc == IL_STKR && o == ILIO_KRLNK && r == ILIA_IR && j == 1)
144 return true;
145 if (opc == IL_AADD && o == ILIO_ARLNK && r == ILIA_IR && j == 1)
146 return true;
147 if (opc == IL_ST && o == ILIO_IRLNK && r == ILIA_AR && j == 1)
148 return true;
149 if (opc == IL_UKNEG && o == ILIO_KRLNK && r == ILIA_IR && j == 1)
150 return true;
151 if ((opc == IL_KMUL || opc == IL_IADD || opc == IL_IKMV) && j_opc == IL_ACCLDSYM)
152 return true;
153 if ((opc == IL_ACMPZ || opc == IL_ACJMPZ || opc == IL_LDA) && o == ILIO_ARLNK && r == ILIA_KR && j == 1)
154 return true;
155 if (opc == IL_IMUL && o == ILIO_IRLNK && r == ILIA_KR && j == 1)
156 return true;
157 if (opc == IL_DAIR && o == ILIO_IRLNK && j_opc == IL_KCON)
158 return true;
159 if ((opc == IL_IADD || opc == IL_IAMV) && o == ILIO_IRLNK && j_opc==IL_DFRKR)
160 return true;
161 if (opc == IL_KXOR && o == ILIO_KRLNK && j_opc == IL_ICMPZ)
162 return true;
163 if ((opc == IL_KCMP || opc == IL_KCJMP) && o == ILIO_KRLNK && j_opc == IL_LDDP)
164 return true;
165 if (opc == IL_PI8MV_LOW && o == ILIO_KRLNK && r == ILIA_AR)
166 return true;
167#ifdef IL_PI8BROADCAST
168 if (opc == IL_PI8BROADCAST && o == ILIO_KRLNK && r == ILIA_AR)
169 return true;
170#endif
171 if (opc == IL_IMUL && o == ILIO_IRLNK && j_opc == IL_KCON)
172 return true;
173 if (opc == IL_IKMV && o == ILIO_IRLNK && j_opc == IL_KCON)
174 return true;
175 if (opc == IL_IADD && o == ILIO_IRLNK && j_opc == IL_LDKR)
176 return true;
177 if (opc == IL_KIMV && o == ILIO_KRLNK && r == ILIA_IR)
178 return true;
179 return false;
180}

182/** Check that operation opc can have jth operand that has operation operand_opc. */
183static void
184verify_compatible(ILI_OP opc, int j, ILI_OP j_opc)
185{
186 ILIA_RESULT r = IL_RES(j_opc);
187 ILIO_KIND o = IL_OPRFLAG(opc, j);
188 ILIA_RESULT expected = (ILIA_RESULT)(-1);
189 if (j_opc == IL_ACCLDSYM)
190 return; /* satisfies any kind of link. used in device code only */
191 switch (o) {
192 case ILIO_LNK:
193 /* Any kind of link allowed. */
194 return;
195 case ILIO_IRLNK:
196 expected = ILIA_IR;
197 break;
198 case ILIO_SPLNK:
199 expected = ILIA_SP;
200 break;
201 case ILIO_DPLNK:
202 expected = ILIA_DP;
203 break;
204 case ILIO_KRLNK:
205 expected = ILIA_KR;
206 break;
207 case ILIO_ARLNK:
208 expected = ILIA_AR;
209 break;
210 case ILIO_QPLNK:
211 expected = ILIA_QP;
212 break;
213 case ILIO_CSLNK:
214 expected = ILIA_CS;
215 break;
216 case ILIO_CDLNK:
217 expected = ILIA_CD;
218 break;
219 case ILIO_CQLNK:
220 expected = ILIA_CQ;
221 break;
222 case ILIO_128LNK:
223 expected = ILIA_128;
224 break;
225 case ILIO_256LNK:
226 expected = ILIA_256;
227 break;
228 case ILIO_512LNK:
229 expected = ILIA_512;
230 break;
231 case ILIO_FLOAT128LNK:
232 expected = ILIA_FLOAT128;
233 break;
234 case ILIO_X87LNK:
235 expected = ILIA_X87;
236 break;
237 case ILIO_DOUBLEDOUBLELNK:
238 expected = ILIA_DOUBLEDOUBLE;
239 break;
240 default:
241 VERIFY(false, "unexpected ILIO_KIND");
242 break;
243 }
244 VERIFY(r == expected || is_known_bug(opc, j, j_opc),
245 "IL_RES of ILI is incompatible with context");
246}

248/** \brief Ad-hoc operand checks.
249
250 This routine does ad-hoc checking of the operands that needs
251 to be specific to an operation. This routine is called from
252 verify_ili_aux after generic checks have been done. */
253static void
254verify_ili_ad_hoc(int ilix)
255{
256 ILI_OP opc = ILI_OPC(ilix);
257 switch (opc) {
258 default:
259 break;
260 case IL_STSP:
261 VERIFY(ILI_OPND(ilix, 4) == MSZ_F4, "4th operand to STSP must be MSZ_F4");
262 break;
263 case IL_STDP:
264 VERIFY(ILI_OPND(ilix, 4) == MSZ_F8, "4th operand to STDP must be MSZ_F8");
265 break;
266#ifdef IL_STSPSP
267 case IL_STSPSP: {
268 ILI_OP opc1 = ILI_OPC(ILI_OPND(ilix, 1));
269 VERIFY(opc1 == IL_DFRDP || opc1 == IL_DPDF, "1st operand to STSPSP must be DFRDP or DPDF");
270 VERIFY(ILI_OPND(ilix, 4) == MSZ_F8, "4th operand to STSPSP must be MSZ_F8");
271 break;
272 }
273#endif
274#ifdef LONG_DOUBLE_FLOAT128
275 case IL_FLOAT128ST:
276 VERIFY(ILI_OPND(ilix, 4) == MSZ_F16, "4th operand to FLOAT128ST must be MSZ_16");
277 break;
278 case IL_FLOAT128LD:
279 VERIFY(ILI_OPND(ilix, 3) == MSZ_F16, "3rd operand to FLOAT128LD must be MSZ_F16");
280 break;
281#endif /* LONG_DOUBLE_FLAOT128 */
282 }
283}

285/** Recursive helper routine for verifying ILI. Because it is recursive,
286 please try to keep it relatively uncluttered so that the control flow
287 remains clear. */
288static void
289verify_ili_aux(int ilix, ILIO_KIND context, VERIFY_LEVEL level)
290{
291 ILI_OP opc;
292 int noprs, j;
293
294 VERIFY(1 <= ilix, "unexpected zero or negative ili index");
295 VERIFY(ilix < ilib.stg_avail, "out of bounds ili index");
296 if (level < VERIFY_ILI_SHALLOW)
297 return;
298
299 if (level >= VERIFY_ILI_DEEP && ili_visit_info(ilix))
300 /* Already checked this node. */
301 return;
302
303 /* General operand checks */
304 opc = ILI_OPC(ilix);
305 noprs = ilis[opc].oprs;
306 for (j = 1; j <= noprs; ++j) {
307 if (ILIO_ISLINK(IL_OPRFLAG(opc, j))) {
308 int operand = ILI_OPND(ilix, j);
309 verify_compatible(opc, j, ILI_OPC(operand));
310 if (level >= VERIFY_ILI_DEEP)
311 verify_ili_aux(operand, IL_OPRFLAG(opc, j), level);
312 }
313 }
314
315 verify_ili_ad_hoc(ilix);
316
317 if (level >= VERIFY_ILI_DEEP)
318 ili_mark_first_visit(ilix);
319}

322verify_ili(int ilix, VERIFY_LEVEL level)

324 begin_walk(level);
325 verify_ili_aux(ilix, ILIO_LNK, level);
326 end_walk(level);

330/** Check if iltx is store of first result of a pair of results returned by a
331 JSR. This is a helper for verify_ilt, and assumes that iltx is index of
332 ILT of type ILTY_STORE and ili_throw_label(iltx)!=0. */
333static bool
334is_first_store_of_pair(int iltx1)
335{
336 int iltx2 = ILT_NEXT(iltx1);
337 if (iltx2) {
338 int ilix1 = ILT_ILIP(iltx1);
339 int ilix2 = ILT_ILIP(iltx2);
340 if (IL_TYPE(ILI_OPC(ilix2)) == ILTY_STORE) {
341 if (ili_throw_label(ilix2)) {
342 /* At this point, we know that ilix and ilix2 are both stores of results
343 from JSR/JSRA operations that can throw. Extract the JSR/JSRA
344 "pointers" and check that they are equal. */
345 int throw_jsr1 = ILI_OPND(ILI_OPND(ilix1, 1), 1);
346 int throw_jsr2 = ILI_OPND(ILI_OPND(ilix2, 1), 1);
347 /* there are cases where the following is not necessarily true; change to
348 only check the equality of throw labels for now */
349 /* return throw_jsr1 == throw_jsr2; */
350 return ili_throw_label(ilix1) == ili_throw_label(ilix2);
351 }
352 }
353 }
354 return false;
355}

358void
359verify_ilt(int iltx, VERIFY_LEVEL level)
360{
361 int ilix, throw_label;
362 VERIFY(0 < iltx && iltx < iltb.stg_size, "invalid ILT index");
363
364 if (level < VERIFY_ILT)
365 return;
366
367 begin_walk(level);
368
369 ilix = ILT_ILIP(iltx);
370 verify_ili_aux(ilix, ILIO_LNK, level);
371
372 /* Check that ILT_CAN_THROW is set correctly. */
373 switch (IL_TYPE(ILI_OPC(ilix))) {
374 case ILTY_STORE:
375 throw_label = ili_throw_label(ilix);
376 if (!throw_label) {
377 VERIFY(!ILT_CAN_THROW(iltx), "ILT_CAN_THROW should be false for "
378 "ILTY_STORE that does not store result of "
379 "JSR that can throw");
380 } /* is_first_store_of_pair() does not always reliably return the correct
381 answer. We need to sort that out before invoking the following
382 else if (is_first_store_of_pair(iltx)) {
383 VERIFY(!ILT_CAN_THROW(iltx), "ILT_CAN_THROW should be false for "
384 "ILTY_STORE that stores first of a pair of "
385 "results from a JSR that can throw");
386
387 } else {
388 VERIFY(ILT_CAN_THROW(iltx), "ILT_CAN_THROW should be true for ILTY_STORE "
389 "that stores sole or second result of a JSR "
390 "that can throw");
391 } */
392 break;
393 case ILTY_PROC:
394 throw_label = ili_throw_label(ilix);
395 if (throw_label)
396 VERIFY(ILT_CAN_THROW(iltx),
397 "ILT_CAN_THROW should be true for ILTY_PROC that can throw");
398 else
399 VERIFY(!ILT_CAN_THROW(iltx),
400 "ILT_CAN_THROW should be false for ILTY_PROC that cannot throw");
401 break;
402 default:
403 VERIFY(!ILT_CAN_THROW(iltx),
404 "ILT_CAN_THROW can be true only for store or call");
405 break;
406 }
407
408 /* With non-extended basic blocks, an ILT that can throw terminates the
409 block. */
410 if (flg.opt >= 2 && ILT_NEXT(iltx)) {
411 VERIFY(!ILT_CAN_THROW(iltx), "ILT_CAN_THROW should be false at -O2 "
412 "if ILT is not last in block");
413 }
414
415 end_walk(level);
416}
