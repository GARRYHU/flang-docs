68/**
69 * Initialize symbol table for new user program unit.
70 */
71void
72sym_init(void)
73{
74 int i;
75 INT tmp[2], res[2];
76 static const char *npname = "hpf_np$";
77 int sptr;
78
79 /* allocate symbol table and name table space: */
80 sym_init_first();
81
82 init_chartab(); /* see dtypeutl.c */
83
84 STG_RESET(stb.dt);
85 STG_NEXT_SIZE(stb.dt, DT_MAX);
86 for (i = 0; i <= DT_MAX; ++i)
87 DTY(i) = pd_dtype[i];
88
89 if (XBIT(49, 0x800000)) {
90 DT_REAL = DT_REAL8;
91 DT_CMPLX = DT_CMPLX16;
92 if (XBIT(49, 0x200)) {
93 DT_DBLE = DT_REAL8;
94 DT_DCMPLX = DT_CMPLX16;
95 } else {
96 DT_DBLE = DT_QUAD;
97 DT_DCMPLX = DT_QCMPLX;
98 }
99 }
100 if (XBIT(49, 0x80000000)) {
101 DT_INT = DT_INT8;
102 DT_LOG = DT_LOG8;
103 }
104 if (XBIT(49, 0x100) && !XBIT(125, 0x2000)) {
105 DT_PTR = DT_INT8;
106 }
107 if (!XBIT(124, 0x10)) {
108 stb.user.dt_int = DT_INT;
109 stb.user.dt_log = DT_LOG;
110 } else {
111 /* -i8 */
112 stb.user.dt_int = DT_INT8;
113 stb.user.dt_log = DT_LOG8;
114 }
115 if (!XBIT(124, 0x8)) {
116 stb.user.dt_real = DT_REAL;
117 stb.user.dt_cmplx = DT_CMPLX;
118 } else {
119 /* -r8 */
120 stb.user.dt_real = DT_DBLE;
121 stb.user.dt_cmplx = DT_DCMPLX;
122 }
123 /*
124 * Set up initial implicit types. All are real except for the letters i
125 * thru n:
126 */
127 init_implicit();
128
129/*
130 * now initialize symbol table. There are 2 cases: The first case occurs
131 * within the utility symini - we start with a totally empty symbol
132 * table. The second case occurs within PGHPF - the initial symbol table is
133 * copied from some arrays set up by symini.
134 */
135
136#if DEBUG
137 assert(stb.stg_size >= INIT_SYMTAB_SIZE, "sym_init:INIT_SYMTAB_SIZE",
138 INIT_SYMTAB_SIZE, 0);
139#endif
140 BCOPY(stb.stg_base, init_sym, SYM, INIT_SYMTAB_SIZE);
141 stb.stg_avail = INIT_SYMTAB_SIZE;
142 stb.stg_cleared = INIT_SYMTAB_SIZE;
143#if DEBUG
144 assert(stb.n_size >= INIT_NAMES_SIZE, "sym_init:INIT_NAMES_SIZE",
145 INIT_NAMES_SIZE, 0);
146#endif
147 BCOPY(stb.n_base, init_names, char, INIT_NAMES_SIZE);
148 stb.namavl = INIT_NAMES_SIZE;
149
150 BCOPY(stb.hashtb, init_hashtb, int, HASHSIZE);
151
152 if (XBIT(124, 0x10)) {
153 /* -i8 */
154 cng_inttyp(intast_sym[I_ICHAR], DT_INT8);
155 }
156
157 /* change the specific intrinsics whose types are DT_QUAD and DT_QCMPLX
158 * if the the '-dp' switch is present. This could be within a #ifdef C90,
159 * but we do build the T3E/C90 pghpfc on solaris.
160 */
161 if (XBIT(49, 0x200))
162 for (i = 1; i < INIT_SYMTAB_SIZE; i++)
163 if (STYPEG(i) == ST_INTRIN) {
164 if (INTTYPG(i) == DT_QUAD)
165 INTTYPP(i, DT_REAL8);
166 else if (INTTYPG(i) == DT_QCMPLX)
167 INTTYPP(i, DT_CMPLX16);
168 if (ARGTYPG(i) == DT_QUAD)
169 ARGTYPP(i, DT_REAL8);
170 else if (ARGTYPG(i) == DT_QCMPLX)
171 ARGTYPP(i, DT_CMPLX16);
172 }
173
174 /*
175 * enter constants into symbol table:
176 *
177 * * * * * * * * * * N O T E * * * * * * * * *
178 * * * * * * * * * * N O T E * * * * * * * * *
179 * DO NOT CHANGE the order of entering these predefined symbols.
180 * To add a predefined, insert it between the last getcon/getsym
181 * and where stb.firstosym is set.
182 */
183
184 /* int 0, 1 */
185 tmp[0] = tmp[1] = (INT)0;
186 stb.i0 = getcon(tmp, DT_INT);
187 if (DT_INT == DT_INT8)
188 stb.k0 = stb.i0;
189 else if (!XBIT(57, 0x2))
190 stb.k0 = getcon(tmp, DT_INT8);
191 else
192 stb.k0 = 0;
193 tmp[1] = (INT)1;
194 stb.i1 = getcon(tmp, DT_INT);
195 if (DT_INT == DT_INT8)
196 stb.k1 = stb.i1;
197 else if (!XBIT(57, 0x2))
198 stb.k1 = getcon(tmp, DT_INT8);
199 else
200 stb.k1 = 0;
201
202 add_fp_constants();
203 /*
204 * * * * * * * * * * N O T E * * * * * * * * *
205 * NO MORE predefined contants until immediately before
206 * stb.firstosym is set ...
207 */
208
209 /* create symbol hpf_np$ */
210 sptr = getsymbol(npname);
211 STYPEP(sptr, ST_UNKNOWN);
212 DTYPEP(sptr, DT_INT);
213 DCLDP(sptr, 1);
214 SCP(sptr, SC_LOCAL);
215 NODESCP(sptr, 1);
216 gbl.sym_nproc = sptr;
217 if (XBIT(70, 0x80000000)) {
218 int bsym;
219 SCP(sptr, SC_BASED);
220 bsym = getsymf("%sp", npname); /* hpf_np$p */
221 STYPEP(bsym, ST_UNKNOWN);
222 DTYPEP(bsym, DT_PTR);
223 DCLDP(bsym, 1);
224 SCP(bsym, SC_LOCAL);
225 NODESCP(bsym, 1);
226 MIDNUMP(sptr, bsym);
227 }
228
229 /* allocate space for auxiliary symtab structures: */
230
231 if (aux.dpdsc_size <= 0) {
232 aux.dpdsc_size = 100;
233 NEW(aux.dpdsc_base, int, aux.dpdsc_size);
234 }
235 aux.dpdsc_avl = 1; /* 0 => null DPDSC field */
236 aux.dpdsc_base[0] = 0;
237
238 if (aux.arrdsc_size <= 0) {
239 aux.arrdsc_size = 200;
240 NEW(aux.arrdsc_base, int, aux.arrdsc_size);
241 }
242 aux.arrdsc_base[0] = 0; /* numdim=0 if array descriptor is zero */
243 aux.arrdsc_avl = 1;
244
245 if (aux.nml_size <= 0) {
246 aux.nml_size = 200;
247 NEW(aux.nml_base, NMLDSC, aux.nml_size);
248 }
249 aux.nml_avl = 1;
250 aux.nml_base[0].sptr = 0;
251 aux.nml_base[0].next = 0;
252 aux.nml_base[0].lineno = 0;
253
254 if (!XBIT(7, 0x100000)) {
255 if (aux.dvl_size <= 0) {
256 aux.dvl_size = 32;
257 NEW(aux.dvl_base, DVL, aux.dvl_size);
258 }
259 aux.dvl_avl = 0;
260 }
261
262 if (aux.symi_size <= 0) {
263 aux.symi_size = 100;
264 NEW(aux.symi_base, SYMI, aux.symi_size);
265 }
266 aux.symi_avl = 1; /* 0 => end of list */
267 aux.symi_base[0].sptr = 0;
268 aux.symi_base[0].next = 0;
269
270 /*
271 * * * * * * * * * * N O T E * * * * * * * * *
272 * More predefined constants after ORIGINAL set; note the
273 * value of NXTRA (number of post-original symbols.
274 */
275 tmp[0] = 0;
276 res[0] = 0;
277 tmp[1] = CONVAL2G(stb.flt0);
278 xfneg(tmp[1], &res[1]);
279 stb.fltm0 = getcon(res, DT_REAL4);
280 tmp[0] = CONVAL1G(stb.dbl0);
281 tmp[1] = CONVAL2G(stb.dbl0);
282 xdneg(tmp, res);
283 stb.dblm0 = getcon(res, DT_DBLE);
284#define NXTRA 2
285
286 aux.curr_entry = &onlyentry;
287 stb.firstusym = stb.firstosym = stb.stg_avail;
288 stb.lbavail = 99999;
289
290 for (i = 0; i < ST_MAX; i++)
291 aux.list[i] = NOSYM; /* 'empty' list for each stype */
292
293 soc.avail = 1;
294 if ((gbl.internal == 0) && (flg.ipa & 0x100) == 0) {
295 /* clear at outer routines, but not for static$init routines */
296 if (gbl.currmod == 0) {
297 dtimplicitstack = 0;
298 } else {
299 dtimplicitstack = 1;
300 }
301 }
302 symutl.sc = SC_LOCAL;
303
304 llmp_reset_uplevel();
305}

316/**
317 * Set up initial implicit types. All are real except for the letters i
318 * thru n:
319 */
320void
321init_implicit(void)
322{
323 int i;
324 int default_int;
325
326 for (i = 0; i < 54; i++) {
327 dtimplicit[i].dtype = stb.user.dt_real;
328 dtimplicit[i].set = FALSE;
329 dtimplicit[i].anyset = FALSE; /* not explicit set anywhere */
330 if (XBIT(124, 0x8) && !XBIT(49, 0x800000)) {
331 dtimplicit[i].typ8 = TRUE;
332 } else {
333 dtimplicit[i].typ8 = FALSE;
334 }
335 }
336
337 default_int = flg.i4 ? stb.user.dt_int : DT_SINT;
338 implicit_int(default_int);
339}

356void
357save_implicit(LOGICAL reset)
358{
359 if (save_dtimplicit == NULL) {
360 dtimplicitsize = 4;
361 NEW(save_dtimplicit, DTIMPL, dtimplicitsize);
362 BZERO(save_dtimplicit, DTIMPL, dtimplicitsize);
363 } else {
364 NEED(dtimplicitstack + 1, save_dtimplicit, DTIMPL, dtimplicitsize,
365 dtimplicitsize * 2);
366 }
367 BCOPY(save_dtimplicit[dtimplicitstack], dtimplicit, DTIMPL, 1);
368 ++dtimplicitstack;
369 if (reset) {
370 int i;
371 for (i = 0; i < 54; i++) {
372 dtimplicit[i].set = FALSE; /* not explicitly set at this level */
373 }
374 }
375}

381 interr("IMPLICIT stack too shallow", dtimplicitstack, 3);
382 --dtimplicitstack;

441 * initialize the new entry, and return pointer to it. If there is already

444int

446{
447 return getsym(name, strlen(name));

450/** \brief Enter symbol with indicated name into symbol table, initialize
451 the new entry.
452 \param name symbol name
453 \param olength number of characters in the symbol name
454 \return pointer to new symbol
455
456 If there is already such a symbol, just return pointer to the existing
457 symbol table entry.
458 */
459int
460getsym(const char *name, int olength)
461{
462 int sptr; /* pointer to symbol table entry */
463
464 sptr = installsym(name, olength);
465 switch (STYPEG(sptr)) {
466 case ST_ISOC:
467 case ST_IEEEARITH:
468 case ST_IEEEEXCEPT:
469 case ST_ISOFTNENV:
470 case ST_CRAY:
471 /* predefined symbol is the name of an HPF library procedure; since
472 * the stype is ST_HL, this implies that a 'USE HPF_LIBRARY' statement
473 * was not seen. Consequently, this symbol is a user symbol and
474 * a new (ST_UNKNOWN) symbol must be entered into the symbol table.
475 */
476 sptr = insert_sym(sptr);
477 setimplicit(sptr);
478 if (gbl.internal > 1)
479 INTERNALP(sptr, 1);
480 SCOPEP(sptr, stb.curr_scope);
481 IGNOREP(sptr, 0);
482 break;
483 case ST_UNKNOWN:
484 setimplicit(sptr);
485 if (gbl.internal > 1)
486 INTERNALP(sptr, 1);
487 SCOPEP(sptr, stb.curr_scope);
488 IGNOREP(sptr, 0);
489 break;
490 default:
491 break;
492 }
493 return sptr;
494}

516/** \brief Enter constant of given dtype and value into the symbol table and
517 return
518 pointer to it.
519
520 If an entry for the constant already exists, return pointer to the existing
521 entry instead.
522 */
523SPTR
524getcon(INT *value, DTYPE dtype)
525{
526 int sptr; /* symbol table pointer */
527 int hashval; /* index into hashtb */
528
529 /*
530 * First loop thru the appropriate hash link list to see if this constant
531 * is already in the symbol table:
532 */
533
534 hashval = HASH_CON(value);
535 if (hashval < 0)
536 hashval = -hashval;
537 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
538 if (DTY(dtype) == TY_QUAD) {
539 if (DTYPEG(sptr) != dtype || STYPEG(sptr) != ST_CONST ||
540 CONVAL1G(sptr) != value[0] || CONVAL2G(sptr) != value[1] ||
541 CONVAL3G(sptr) != value[2] || CONVAL4G(sptr) != value[3])
542 continue;
543
544 /* Matching entry has been found. Return it: */
545 return (sptr);
546 }
547 if (DTYPEG(sptr) != dtype || STYPEG(sptr) != ST_CONST ||
548 CONVAL1G(sptr) != value[0] || CONVAL2G(sptr) != value[1])
549 continue;
550
551 /* Matching entry has been found. Return it: */
552 return (sptr);
553 }
554
555 /* Constant not found. Create a new symbol table entry for it: */
556
557 ADDSYM(sptr, hashval);
558 STYPEP(sptr, ST_CONST);
559 CONVAL1P(sptr, value[0]);
560 CONVAL2P(sptr, value[1]);
561 if (DTY(dtype) == TY_QUAD) {
562 CONVAL3P(sptr, value[2]);
563 CONVAL4P(sptr, value[3]);
564 }
565 DTYPEP(sptr, dtype);
566 SCOPEP(sptr, 1);
567
568 return (sptr);
569}

571/* constant value (value[1] if 1 word) */
572int
573hashcon(INT *value, int dtype, int sptr)
574{
575 int sptr1; /* symbol table pointer */
576 int hashval; /* index into hashtb */
577
578 /*
579 * First loop thru the appropriate hash link list to see if this constant
580 * is already in the symbol table:
581 */
582
583 hashval = HASH_CON(value);
584 if (hashval < 0)
585 hashval = -hashval;
586 for (sptr1 = stb.hashtb[hashval]; sptr1 != 0; sptr1 = HASHLKG(sptr1)) {
587
588 if (sptr1 == sptr)
589 return (sptr);
590 }
591
592 /* sptr not found. */
593
594 HASHLKP(sptr, stb.hashtb[hashval]);
595 stb.hashtb[hashval] = sptr;
596
597 return (sptr);
598}

607/*
608 * BIGOBJects are supported, need an acon-specific getcon
609 */
610SPTR
611get_acon3(SPTR sym, ISZ_T off, DTYPE dtype)
612{
613 INT value[2];
614 int sptr; /* symbol table pointer */
615 int hashval; /* index into stb.hashtb */
616
617 /*
618 * First loop thru the appropriate hash link list to see if this constant
619 * is already in the symbol table:
620 */
621
622 bgitoi64(off, value);
623 value[0] = sym;
624 hashval = HASH_CON(value);
625 if (hashval < 0)
626 hashval = -hashval;
627 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
628 if (DTYPEG(sptr) != dtype || STYPEG(sptr) != ST_CONST ||
629 CONVAL1G(sptr) != sym || ACONOFFG(sptr) != off)
630 continue;
631
632 /* Matching entry has been found. Return it: */
633
634 return (sptr);
635 }
636
637 /* Constant not found. Create a new symbol table entry for it: */
638
639 ADDSYM(sptr, hashval);
640 CONVAL1P(sptr, sym);
641 ACONOFFP(sptr, off);
642 STYPEP(sptr, ST_CONST);
643 DTYPEP(sptr, dtype);
644 SCOPEP(sptr, 1);
645
646 return (sptr);
647}

697/**

699 * returned is sign extended with respect to the host's int type.

704 /* 32-bit INT */

709 w = 32 - width;
710 return ARSHIFT(LSHIFT(val, w), w);

715/** \brief Enter character constant into symbol table and return
716 pointer to it.
717 \param value character string value
718 \param length length of character string
719
720 If the constant is already in the table,
721 return pointer to the existing entry instead.
722 */
723int
724getstring(const char *value, int length)
725{
726 int sptr; /* symbol table pointer */
727 int hashval; /* index into hashtb */
728 char *np; /* pointer to string characters */
729 const char *p;
730 int i, clen;
731 int dtype;
732 /*
733 * first loop thru the appropriate hash link list to see if symbol is
734 * already in the table:
735 */
736 HASH_STR(hashval, value, length);
737 /* Ensure hash value is positive. '\nnn' can cause negative hash values */
738 if (hashval < 0)
739 hashval = -hashval;
740 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
741 if (STYPEG(sptr) != ST_CONST)
742 continue;
743 i = DTYPEG(sptr);
744 if (DTY(i) == TY_CHAR) {
745 clen = DTY(i + 1);
746 if (clen > 0 && A_ALIASG(clen)) {
747 clen = A_ALIASG(clen);
748 clen = A_SPTRG(clen);
749 clen = CONVAL2G(clen);
750 if (clen == length) {
751 /* now match the characters in the strings: */
752
753 np = stb.n_base + CONVAL1G(sptr);
754 p = value;
755 i = length;
756 while (i--)
757 if (*np++ != *p++)
758 goto Continue;
759 /* Matching entry has been found in symtab. Return it: */
760 return sptr;
761 }
762 }
763 }
764 Continue:;
765 }
766
767 /* String not found. Create a new symtab entry for it: */
768
769 dtype = get_type(2, TY_CHAR, mk_cval(length, DT_INT4));
770 ADDSYM(sptr, hashval);
771 STYPEP(sptr, ST_CONST);
772 CONVAL1P(sptr, putsname(value, length));
773 DTYPEP(sptr, dtype);
774 SCOPEP(sptr, 1);
775 return (sptr);
776}

940/**

942 */
943const char *
944getprint(int sptr)
945{
946 int len; /* length of character string */
947 static char *b = NULL;
948 char *from, *end, *to;
949 int c;
950 INT num[4];
951 int dtype;
952
953 if (sptr == 0)
954 return ".0.";
955 if (sptr < 0)
956 return ".neg.";
957 if (sptr >= stb.stg_avail)
958 return ".toobig.";
959 if (STYPEG(sptr) != ST_CONST)
960 return SYMNAME(sptr);
961
962 if (b == NULL) {
963 NEW(b, char, 100);
964 }
965 dtype = DTYPEG(sptr);
966 switch (DTY(dtype)) {
967 case TY_WORD:
968 sprintf(b, "%08X", CONVAL2G(sptr));
969 break;
970 case TY_DWORD:
971 sprintf(b, "%08X%08X", CONVAL1G(sptr), CONVAL2G(sptr));
972 break;
973 case TY_BINT:
974 sprintf(b, "%d_1", CONVAL2G(sptr));
975 break;
976 case TY_SINT:
977 sprintf(b, "%d_2", CONVAL2G(sptr));
978 break;
979 case TY_INT:
980 sprintf(b, "%d", CONVAL2G(sptr));
981 break;
982 case TY_INT8:
983 num[0] = CONVAL1G(sptr);
984 num[1] = CONVAL2G(sptr);
985 ui64toax(num, b, 22, 0, 10);
986 break;
987 case TY_BLOG:
988 sprintf(b, "%s_1", __log_print(CONVAL2G(sptr)));
989 break;
990 case TY_SLOG:
991 sprintf(b, "%s_2", __log_print(CONVAL2G(sptr)));
992 break;
993 case TY_LOG:
994 case TY_LOG8:
995 sprintf(b, "%s", __log_print(CONVAL2G(sptr)));
996 break;
997 case TY_REAL:
998 num[0] = CONVAL2G(sptr);
999 cprintf(b, "%17.10e", (INT*)(size_t)(num[0]));
1000 break;
1001
1002 case TY_DBLE:
1003 num[0] = CONVAL1G(sptr);
1004 num[1] = CONVAL2G(sptr);
1005 cprintf(b, "%24.17le", num);
1006 break;
1007 case TY_QUAD:
1008 num[0] = CONVAL1G(sptr);
1009 num[1] = CONVAL2G(sptr);
1010 num[2] = CONVAL3G(sptr);
1011 num[3] = CONVAL4G(sptr);
1012 cprintf(b, "%44.37qd", num);
1013 break;
1014
1015 case TY_CMPLX:
1016 num[0] = CONVAL1G(sptr);
1017 cprintf(b, "%17.10e", (INT*)(size_t)(num[0]));
1018 b[17] = ',';
1019 b[18] = ' ';
1020 num[0] = CONVAL2G(sptr);
1021 cprintf(&b[19], "%17.10e", (INT*)(size_t)(num[0]));
1022 break;
1023
1024 case TY_DCMPLX:
1025 num[0] = CONVAL1G(CONVAL1G(sptr));
1026 num[1] = CONVAL2G(CONVAL1G(sptr));
1027 cprintf(b, "%24.17le", num);
1028 b[24] = ',';
1029 b[25] = ' ';
1030 num[0] = CONVAL1G(CONVAL2G(sptr));
1031 num[1] = CONVAL2G(CONVAL2G(sptr));
1032 cprintf(&b[26], "%24.17le", num);
1033 break;
1034
1035 case TY_NCHAR:
1036 sptr = CONVAL1G(sptr); /* sptr to char string constant */
1037 dtype = DTYPEG(sptr);
1038 goto like_char;
1039 case TY_HOLL:
1040 sptr = CONVAL1G(sptr);
1041 dtype = DTYPEG(sptr);
1042 FLANG_FALLTHROUGH;
1043 case TY_CHAR:
1044 like_char:
1045 from = stb.n_base + CONVAL1G(sptr);
1046 len = string_length(dtype);
1047 end = b + 93;
1048 *b = '\"';
1049 for (to = b + 1; len-- && to < end;) {
1050 c = *from++ & 0xff;
1051 if (c == '\"' || c == '\'' || c == '\\') {
1052 *to++ = '\\';
1053 *to++ = c;
1054 } else if (c >= ' ' && c <= '~') {
1055 *to++ = c;
1056 } else if (c == '\n') {
1057 *to++ = '\\';
1058 *to++ = 'n';
1059 }
1060 else {
1061 *to++ = '\\';
1062 /* Mask off 8 bits worth of unprintable character */
1063 sprintf(to, "%03o", (c & 255));
1064 to += 3;
1065 }
1066 }
1067 *to++ = '\"';
1068 *to = '\0';
1069 break;
1070
1071 case TY_PTR:
1072 strcpy(b, "address constant");
1073 break;
1074
1075 default:
1076 interr("getprint:bad const dtype", sptr, 1);
1077 }
1078 return b;
1079}

1720/**
1721 * create (or possibly reuse) a compiler created symbol whose name is of the
1722 * form . 'letter' dddd where dddd is the decimal representation of n and
1723 * 'letter' is the character specified in the letter argument.
1724 */
1725int
1726getccsym(int letter, int n, SYMTYPE stype)
1727{
1728 char name[32];
1729 int sptr, i;
1730 const char *suffix = IPA_RECOMPILATION_SUFFIX;
1731
1732 sprintf(name, ".%c%04d%s", letter, n, suffix); /* at least 4, could be more */
1733 i = 0;
1734 do {
1735 sptr = getsymbol(name);
1736 if (STYPEG(sptr) == ST_UNKNOWN) {
1737 STYPEP(sptr, stype);
1738 CCSYMP(sptr, 1);
1739 IGNOREP(sptr, 0);
1740 SCOPEP(sptr, stb.curr_scope);
1741 return sptr;
1742 }
1743 if (SCOPEG(sptr) == stb.curr_scope && STYPEG(sptr) == stype)
1744 return sptr;
1745 /* make up a new name */
1746 ++i;
1747 sprintf(&name[2], "%04d%03d%s", n, i, suffix);
1748 } while (1);
1749}

2441/**
2442 * \brief Compares two symbols by returning true if they both have equivalent
2443 * interfaces. Otherwise, return false.
2444 *
2445 * If flag is set, then we also make sure that sym1 and sym2 have the same
2446 * symbol name.
2447 */
2448bool
2449cmp_interfaces(int sym1, int sym2, int flag)
2450{
2451
2452 int i, paramct, paramct2, dpdsc, dpdsc2, psptr, psptr2;
2453 int iface1, iface2;
2454
2455 if (sym1 <= NOSYM)
2456 return false;
2457
2458 /* It's OK for the argument procedure pointer to point to NOSYM as long as
2459 * the formal procedure pointer points to a valid symtab entry.
2460 *
2461 * We assume the following:
2462 *
2463 * sym1 is the formal procedure pointer dummy argument
2464 * sym2 is the actual procedure pointer argument
2465 */
2466 if (sym2 <= NOSYM)
2467 return true;
2468
2469 if (STYPEG(sym1) != ST_PROC) {
2470 int scope, alt_iface;
2471 int hash, hptr, len;
2472 char *symname;
2473 symname = SYMNAME(sym1);
2474 len = strlen(symname);
2475 HASH_ID(hash, symname, len);
2476 for (hptr = stb.hashtb[hash]; hptr; hptr = HASHLKG(hptr)) {
2477 if (STYPEG(hptr) == ST_PROC && strcmp(symname, SYMNAME(hptr)) == 0) {
2478 alt_iface = hptr;
2479 if (alt_iface && (scope = test_scope(alt_iface))) {
2480 if (scope <= test_scope(sym1)) {
2481 sym1 = alt_iface;
2482 break;
2483 }
2484 }
2485 }
2486 }
2487 }
2488 if (STYPEG(sym2) != ST_PROC) {
2489 int scope, alt_iface;
2490 int hash, hptr, len;
2491 char *symname;
2492 symname = SYMNAME(sym2);
2493 len = strlen(symname);
2494 HASH_ID(hash, symname, len);
2495 for (hptr = stb.hashtb[hash]; hptr; hptr = HASHLKG(hptr)) {
2496 if (STYPEG(hptr) == ST_PROC && strcmp(symname, SYMNAME(hptr)) == 0) {
2497 alt_iface = hptr;
2498 if (alt_iface && (scope = test_scope(alt_iface))) {
2499 if (scope <= test_scope(sym2)) {
2500 sym2 = alt_iface;
2501 break;
2502 }
2503 }
2504 }
2505 }
2506 }
2507
2508 iface1 = iface2 = paramct = paramct2 = dpdsc = dpdsc2 = 0;
2509 proc_arginfo(sym1, &paramct2, &dpdsc2, &iface1);
2510 proc_arginfo(sym2, &paramct, &dpdsc, &iface2);
2511 if (!iface1 || !iface2)
2512 return false;
2513 if (flag && strcmp(SYMNAME(iface1), SYMNAME(iface2)) != 0)
2514 return false;
2515 if (paramct != paramct2)
2516 return false;
2517 if (iface1 && iface1 == iface2)
2518 return true;
2519 if (!eq_dtype2(DTYPEG(FVALG(iface1)), DTYPEG(FVALG(iface2)), 0))
2520 return false; /* result types differ */
2521 for (i = 0; i < paramct; ++dpdsc, ++dpdsc2, ++i) {
2522 psptr2 = *(aux.dpdsc_base + dpdsc2);
2523 psptr = *(aux.dpdsc_base + dpdsc);
2524 if (!psptr || !psptr2 || STYPEG(psptr) != STYPEG(psptr2) ||
2525 strcmp(SYMNAME(psptr), SYMNAME(psptr2)) != 0)
2526 return false;
2527 if (STYPEG(psptr) == ST_PROC && STYPEG(psptr2) == ST_PROC) {
2528 if (!cmp_interfaces(psptr, psptr2, flag)) {
2529 return false;
2530 }
2531 } else if (!eq_dtype2(DTYPEG(psptr), DTYPEG(psptr2), 0)) {
2532 return false;
2533 }
2534 }

2539 * \brief Tests the characteristics between two interfaces.
2540 *
2541 * \param psptr is the first interface.
2542 *
2543 * \param pstr2 is the second interface.
2544 *
2545 * \param flag is a bit mask that enforces/relaxes certain checks (see
2546 * cmp_interface_flags enum in symtab.c).
2547 *
2548 * \return true if the two characteristics are compatible, else false.
2549 */
2550bool
2551compatible_characteristics(int psptr, int psptr2, cmp_interface_flags flag)
2552{
2553
2554 if (!psptr || !psptr2) {
2555 return false;
2556 }
2557
2558 if ( (((flag & RELAX_INTENT_CHK) == 0) &&
2559 INTENTG(psptr) != INTENTG(psptr2)) ||
2560 (((flag & CMP_OPTARG) != 0) && OPTARGG(psptr) != OPTARGG(psptr2)) ||
2561 ALLOCATTRG(psptr) != ALLOCATTRG(psptr2) ||
2562 PASSBYVALG(psptr) != PASSBYVALG(psptr2) ||
2563 ASYNCG(psptr) != ASYNCG(psptr2) || VOLG(psptr) != VOLG(psptr2) ||
2564 CLASSG(psptr) != CLASSG(psptr2) ||
2565 (((flag & RELAX_POINTER_CHK) == 0) &&
2566 POINTERG(psptr) != POINTERG(psptr2)) ||
2567 TARGETG(psptr) != TARGETG(psptr2) ||
2568 CONTIGATTRG(psptr) != CONTIGATTRG(psptr2)) {
2569 if (flag & CMP_SUBMOD_IFACE)
2570 generate_type_mismatch_errors(psptr, psptr2);
2571
2572 return false;
2573 }
2574
2575 if ((flag & RELAX_STYPE_CHK) == 0 && STYPEG(psptr) != STYPEG(psptr2)) {
2576 return false;
2577 }
2578
2579 if (strcmp(SYMNAME(psptr), SYMNAME(psptr2)) != 0) {
2580 if (flag & CMP_SUBMOD_IFACE) {
2581 /* function may use itself name as a return variable, so no name
2582 comparison for function return variables.
2583 */
2584 if (!RESULTG(psptr) && !RESULTG(psptr2))
2585 error(1057, ERR_Severe, gbl.lineno, SYMNAME(psptr2),SYMNAME(psptr));
2586 }
2587 if ((flag & IGNORE_ARG_NAMES) == 0 && (flag & CMP_SUBMOD_IFACE) == 0)
2588 return false;
2589 }
2590
2591 if (STYPEG(psptr) == ST_PROC && STYPEG(psptr2) == ST_PROC &&
2592 (flag & DEFER_IFACE_CHK) == 0) {
2593 if (!cmp_interfaces_strict(psptr, psptr2, (flag | CMP_OPTARG))) {
2594 return false;
2595 }
2596 } else if (DTY(DTYPEG(psptr)) == DTY(DTYPEG(psptr2)) &&
2597 (DTY(DTYPEG(psptr)) == TY_CHAR
2598 || DTY(DTYPEG(psptr)) == TY_NCHAR
2599 )) {
2600 /* check character objects only when they both
2601 * have constant lengths or at least one is assumed shape.
2602 */
2603 int d1 = DTYPEG(psptr);
2604 int a1 = DTY(d1+1);
2605 int d2 = DTYPEG(psptr2);
2606 int a2 = DTY(d2+1);
2607 if ((a1 == 0 || a2 == 0 ||
2608 (A_TYPEG(a1) == A_CNST && A_TYPEG(a2) == A_CNST)) &&
2609 !eq_dtype2(d1, d2, 0)) {
2610 return FALSE;
2611 }
2612 } else if (!eq_dtype2(DTYPEG(psptr), DTYPEG(psptr2), 0)) {
2613 if (flag & CMP_SUBMOD_IFACE) {
2614 /* check whether variable type matches for:
2615 1. argument type
2616 2. function return type
2617 */
2618 if ((DTY(DTYPEG(psptr)) != DTY(DTYPEG(psptr2))) ||
2619 (DDTG(DTYPEG(psptr)) != DDTG(DTYPEG(psptr2)) &&
2620 DTYG(DTYPEG(psptr)) != DTYG(DTYPEG(psptr2))))
2621 generate_type_mismatch_errors(psptr, psptr2);
2622 }
2623 return FALSE;
2624 } else if (DTY(DTYPEG(psptr)) == TY_ARRAY &&
2625 DTY(DTYPEG(psptr2)) == TY_ARRAY) {
2626 /* Check extents of array dimensions. Note: the call to eq_dtype2()
2627 * above checks type and rank.
2628 */
2629 ADSC *ad, *ad2;
2630 int i, ast, ast2, numdim;
2631
2632 ad = AD_DPTR(DTYPEG(psptr));
2633 numdim = AD_NUMDIM(ad);
2634
2635 ad2 = AD_DPTR(DTYPEG(psptr2));
2636
2637 for(i=0; i < numdim; ++i) {
2638 ast = AD_EXTNTAST(ad, i);
2639 ast2 = AD_EXTNTAST(ad2, i);
2640 if (A_TYPEG(ast) == A_CNST && A_TYPEG(ast2) == A_CNST &&
2641 CONVAL2G(A_SPTRG(ast)) != CONVAL2G(A_SPTRG(ast2))) {
2642 return false;
2643 }
2644 }
2645 }
2646
2647 return true;
2648}

2650/**
2651 * \brief Same as cmp_interfaces() except we also compare the characteristics as
2652 * defined in "12.2 Characteristics of procedures" in F2003 Spec.
2653 */
2654bool
2655cmp_interfaces_strict(SPTR sym1, SPTR sym2, cmp_interface_flags flag)
2656{
2657 int i, paramct, paramct2, dpdsc, dpdsc2, psptr, psptr2;
2658 int iface1, iface2, j;
2659 bool relax1, relax2;
2660
2661 iface1 = iface2 = paramct = paramct2 = dpdsc = dpdsc2 = 0;
2662 proc_arginfo(sym1, &paramct, &dpdsc, &iface1);
2663 proc_arginfo(sym2, &paramct2, &dpdsc2, &iface2);
2664
2665 if (FVALG(sym1) && FVALG(sym2) && dpdsc > 0 && dpdsc2 > 0) {
2666 /* Check characteristics of results if applicable. We do this here
2667 * to handle the case where one symbol will have its result in argument
2668 * 1 and another symbol will not. This occurs when one symbol is a
2669 * function and another symbol is a function interface (i.e., we do not
2670 * put the function result into argument 1 for interfaces). We then
2671 * adjust parameter counts and argument descriptors when the result is
2672 * in an argument so parameter counts are consistent between the two
2673 * symbols.
2674 */
2675 if (paramct > 0) {
2676 psptr = aux.dpdsc_base[dpdsc];
2677 if (FVALG(sym1) == psptr) {
2678 paramct--;
2679 dpdsc++;
2680 }
2681 }
2682 if (paramct2 > 0) {
2683 psptr2 = aux.dpdsc_base[dpdsc2];
2684 if (FVALG(sym2) == psptr2) {
2685 paramct2--;
2686 dpdsc2++;
2687 }
2688 }
2689 psptr = FVALG(sym1);
2690 psptr2 = FVALG(sym2);
2691 if (!compatible_characteristics(psptr, psptr2, flag)) {
2692 return false;
2693 }
2694 }
2695
2696 /* we may have added descriptors such as type descriptors to the
2697 * argument descriptor. Do not count them.
2698 */
2699
2700 for (j = i = 0; i < paramct; ++i) {
2701 psptr = aux.dpdsc_base[dpdsc + i];
2702 if (CCSYMG(psptr) && CLASSG(psptr)) {
2703 ++j;
2704 }
2705 }
2706 paramct -= j;
2707
2708 for (j = i = 0; i < paramct2; ++i) {
2709 psptr2 = aux.dpdsc_base[dpdsc2 + i];
2710 if (CCSYMG(psptr2) && CLASSG(psptr2)) {
2711 ++j;
2712 }
2713 }
2714 paramct2 -= j;
2715
2716 if (PUREG(sym1) != PUREG(sym2) || IMPUREG(sym1) != IMPUREG(sym2)) {
2717 if (flag & CMP_SUBMOD_IFACE)
2718 error(1060, ERR_Severe, gbl.lineno, "PURE function prefix",SYMNAME(sym1));
2719
2720 relax1 = (flag & RELAX_PURE_CHK_1) != 0;
2721 relax2 = (flag & RELAX_PURE_CHK_2) != 0;
2722
2723 if (!relax1 && !relax2) {
2724 /* both arguments must have matching pure/impure attributes */
2725 return false;
2726 }
2727 if (relax1 != relax2 && PUREG(sym1) != PUREG(sym2)) {
2728 if (!relax1 && PUREG(sym1)) {
2729 /* argument 1 has pure but argument 2 does not. */
2730 return false;
2731 }
2732 if (!relax2 && PUREG(sym2)) {
2733 /* argument 2 has pure but argument 1 does not */
2734 return false;
2735 }
2736 }
2737 }
2738 if (paramct != paramct2) {
2739 if (flag & CMP_SUBMOD_IFACE)
2740 error(1059, ERR_Severe, gbl.lineno, SYMNAME(sym1), NULL);
2741 return false;
2742 }
2743 if (CFUNCG(sym1) != CFUNCG(sym2)){
2744 if (flag & CMP_SUBMOD_IFACE)
2745 error(1060, ERR_Severe, gbl.lineno, "BIND attribute", SYMNAME(sym1));
2746 return false;
2747 }
2748 if (ELEMENTALG(sym1) != ELEMENTALG(sym2)){
2749 if (flag & CMP_SUBMOD_IFACE)
2750 error(1060, ERR_Severe, gbl.lineno, "ELEMENTAL function prefix",SYMNAME(sym1));
2751 return false;
2752 }
2753
2754 if ((FVALG(sym1) && !FVALG(sym2)) || (FVALG(sym2) && !FVALG(sym1))) {
2755 if (flag & CMP_SUBMOD_IFACE)
2756 error(1058, ERR_Severe, gbl.lineno, SYMNAME(sym1), NULL);
2757 return false;
2758 }
2759
2760 if (!iface1 || !iface2)
2761 return false;
2762 if ( ((flag & CMP_IFACE_NAMES) != 0) && strcmp(SYMNAME(iface1),
2763 SYMNAME(iface2)) != 0)
2764 return false;
2765 if (iface1 && iface1 == iface2)
2766 return true;
2767
2768 for (i = 0; i < paramct; ++dpdsc, ++dpdsc2, ++i) {
2769 psptr2 = aux.dpdsc_base[dpdsc2];
2770 psptr = aux.dpdsc_base[dpdsc];
2771
2772 if (!compatible_characteristics(psptr, psptr2, flag)) {
2773 return false;
2774 }
2775
2776 }
2777 return true;
2778}

3041/**
3042 * create a distinct mangled name
3043 */
3044char *
3045mangle_name(const char *basename, const char *purpose)
3046{
3047 int length, i, j;
3048 int sptr, hashval;
3049 int tail_index;
3050 static char name[MAXIDLEN + 1];
3051 int max_idlen = MAXIDLEN;
3052
3053/* we use the convention: basename$purpose%d
3054 * if purpose is absent, just basename%d
3055 * to deal with the length issue:
3056 * + if the length of the name exceeds the max allowed, truncate.
3057 * + if a clash occurs, append to or replace the 'tail' of the name
3058 * with a digit string (max 5 digits).
3059) */
3060
3061 if (flg.standard) {
3062 max_idlen = STANDARD_MAXIDLEN;
3063 }
3064 length = strlen(basename);
3065 if (length > max_idlen)
3066 length = max_idlen;
3067 memcpy(name, basename, length);
3068 if (purpose) {
3069 j = length + 1 + strlen(purpose); /* basname$purpose */
3070 if (j > max_idlen)
3071 j = max_idlen;
3072 j -= length; /* room for $purpose */
3073 if (j > 0) {
3074 name[length] = '$';
3075 memcpy(&name[length + 1], purpose, j - 1);
3076 length += j;
3077 }
3078 }
3079 name[length] = '\0';
3080 tail_index = length; /* append digit string */
3081 if ((max_idlen - length) < 5)
3082 tail_index = max_idlen - 5; /* no room replace last 5 characters */
3083 for (i = 0;;) {
3084 length = strlen(name);
3085 HASH_ID(hashval, name, length);
3086 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
3087 if (IGNOREG(sptr) && stb.curr_scope == SCOPEG(sptr))
3088 continue;
3089 if (strcmp(name, SYMNAME(sptr)) == 0)
3090 break;
3091 }
3092 if (sptr == 0)
3093 break;
3094 ++i;
3095 ++manglecount;
3096 if (manglecount >= 10000)
3097 manglecount = 0;
3098 sprintf(&name[tail_index], "%d", manglecount);
3099 assert(i < 10000, "mangle_name: too many temps", 0, 4);
3100 }
3101 return name;
3102}
