55/* Target Info is the main struct which keeps all the information about target
56 * or target data region. Each outlined function must have target info, if there
57 * several nested outlining their information are kept hierarchically. Also,
58 * each target data construct creates a target info. */
59typedef struct _OMPACCEL_TARGET OMPACCEL_TINFO;
60
61struct _OMPACCEL_TARGET{
62 SPTR func_sptr; /* Kernel or device function sptr */
63 OMPACCEL_SYM *symbols; /* Keeps host and device symbols along with map-type */
64 int n_symbols; /* Number of parameters */
65 int sz_symbols; /* Size of symbols array */
66 OMPACCEL_SYM *quiet_symbols; /* Keeps sc_based symbols. They don't be passed to the device */
67 int n_quiet_symbols; /* Number of quiet_symbols */
68 int sz_quiet_symbols; /* Size of quite_symbols */
69 OMP_TARGET_MODE mode; /* Combined construct mode */
70 OMPACCEL_TINFO* parent_tinfo; /* Parent tinfo is used for nested outlining in device. */
71 bool nowait; /* async */
72 int n_reduction_symbols; /* Number of reduction symbols */
73 OMPACCEL_RED_SYM *reduction_symbols; /* Reduction symbols along with the reduction operator */
74 OMPACCEL_RED_FUNCS reduction_funcs; /* Auxiliary functions for reduction */
75};
