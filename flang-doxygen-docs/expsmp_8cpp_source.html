245sectionCreateBlock(int nextLabel, SPTR lb, SPTR ub, int myVal)
246{
247 int ili, ubVal, lbVal;
248
249 myVal = ad_icon(myVal);
250 lbVal = ad3ili(IL_LD, ad_acon(lb, 0), addnme(NT_VAR, lb, 0, 0), MSZ_WORD);
251 ubVal = ad3ili(IL_LD, ad_acon(ub, 0), addnme(NT_VAR, ub, 0, 0), MSZ_WORD);

254 RFCNTI(nextLabel);

257 ili = ad4ili(IL_UICJMP, myVal, lbVal, CC_LT, nextLabel);
258 RFCNTI(nextLabel);
259
260 return ili;
261}

331 wrilts(mppgbih[mppgcnt - eampp]);
332 expb.curbih = savebih;
333 rdilts(expb.curbih);
334 expb.curilt = ILT_PREV(0);
335 flg.x[14] = savex14;
336 }

393 * The runtime for instance, _mp_copypriv_kmpc, expects size_t* for size.

395 * 'bytes' is the actual byte size and not an sptr or ili.

397static int
398genSizeAcon(int size_ili)
399{
400 int ili;
401 SPTR tmp;
402 int nme;
403 const DTYPE dtype = (TARGET_PTRSIZE == 8) ? DT_INT8 : DT_INT;
404
405 tmp = getPrivateTemp(dtype);
406 SCP(tmp, SC_AUTO);
407
408 ili = ad_acon(tmp, 0);
409 nme = addnme(NT_VAR, tmp, 0, 0);
410 ADDRTKNP(tmp, 1);
411
412 if (TARGET_PTRSIZE == 8) {
413 ili = ad4ili(IL_STKR, size_ili, ili, nme, MSZ_I8);
414 } else {
415 size_ili = ad1ili(IL_KIMV, size_ili);
416 ili = ad4ili(IL_ST, size_ili, ili, nme, MSZ_WORD);
417 }
418 chk_block(ili);
419
420 return ad_acon(tmp, 0);

423/* Given a sptr list, create an array of pairs:
424 * (size, address) where:
425 * 'size' - Pointer to a temporary variable containing the byte size of
426 * sptr. (size_t *)
427 * 'address' - Address of sptr. (void *).
428 *
429 * These pairs are represented in an array where
430 * the even indices are the size pointers and the odd indices the
431 * addresses. The sentinel/terminator is the all-zero pair.
432 * [(sz0,addr0), (sz1,addr1), ... (0x0, 0x0)].
433 *
434 * We represent these as an array, which is more convenient to manage
435 * internally. The runtime routine _mp_copypriv_kmpc expects this format.
436 *
437 * Returns: The sptr of this majestic array that we so masterfully create here.
438 */
439static SPTR
440makeCopyprivArray(const sptrListT *list, bool pass_size_addresses)
441{
442 int i, ili, nme, n_elts;
443 SPTR array;
444 DTYPE dtype;
445 int basenme;
446 static int id;
447 const sptrListT *node;
448
449 /* Count the number of items in the list */
450 n_elts = 0;
451 for (node = list; node; node = node->next)
452 ++n_elts;
453
454 /* We represent each entry as a pair for each private variable (each node in
455 * sptr_list): (size, sptr)
456 *
457 * +2 for the last node, the sentinel (null node), which tells the
458 * runtime it has reached the end of the array. Each node is 2 array elts.
459 */
460 n_elts = (n_elts * 2) + 2;
461
462 /* Create the array dtype: each element is word size */
463 array = getnewccsym('a', ++id, ST_ARRAY);
464 {
465 ADSC *adsc;
466 INT con[2] = {0, n_elts};
467
468 dtype = get_array_dtype(1, DT_CPTR);
469 adsc = AD_DPTR(dtype);
470 AD_LWBD(adsc, 0) = stb.i1;
471 AD_UPBD(adsc, 0) = getcon(con, DT_INT);
472 AD_NUMELM(adsc) = AD_UPBD(adsc, 0);
473 }
474
475 DTYPEP(array, dtype);
476 SCP(array, SC_AUTO);
477
478 /* Build the list: (size, sptr) pairs. */
479 basenme = addnme(NT_VAR, array, 0, 0);
480 for (node = list, i = 0; node; node = node->next, ++i) {
481 int sptr_nme, sptr_ili;
482
483 if (node->is_common_block || THREADG(node->sptr)) {
484/* This is thread private so obtain address from the TP vector */
485 if (node->is_common_block)
486 ref_threadprivate(node->sptr, &sptr_ili, &sptr_nme);
487 else
488 ref_threadprivate_var(node->sptr, &sptr_ili, &sptr_nme, 1);
489 } else {
490 /* Else, this is not thread private */
491 sptr_nme = addnme(NT_VAR, node->sptr, 0, 0);
492 sptr_ili = mk_address(node->sptr);
493 }
494
495 /* array[i] = size */
496 nme = add_arrnme(NT_ARR, array, basenme, 0, ad_icon(i), false);
497 if (pass_size_addresses) { /* why do I need to pass address? */
498 ili = genSizeAcon(node->size_ili);
499 ili = ad3ili(IL_STA, ili, ad_acon(array, i * TARGET_PTRSIZE), nme);
500 } else {
501 ili = ad4ili(IL_ST, node->size_ili, ad_acon(array, i * TARGET_PTRSIZE),
502 nme, TARGET_PTRSIZE == 8 ? MSZ_I8 : MSZ_WORD);
503 }
504 chk_block(ili);
505
506 /* array[i+1] = local (stack based) sptr */
507 ++i;
508 nme = add_arrnme(NT_ARR, array, basenme, 0, ad_icon(i), false);
509 ili = ad3ili(IL_STA, sptr_ili, ad_acon(array, i * TARGET_PTRSIZE), nme);
510 chk_block(ili);
511 }
512
513 /* Terminate the array with a sentinel that the runtime will recognize */
514 nme = add_arrnme(NT_ARR, array, basenme, 0, ad_icon(i), false);
515 ili = ad3ili(IL_STA, ad_aconi(0), ad_acon(array, i * TARGET_PTRSIZE), nme);
516 chk_block(ili);
517
518 ++i;
519 nme = add_arrnme(NT_ARR, array, basenme, 0, ad_icon(i), false);
520 ili = ad3ili(IL_STA, ad_aconi(0), ad_acon(array, i * TARGET_PTRSIZE), nme);
521 chk_block(ili);
522
523 return array;
524}

536/**
537 \brief ...
538 \param arglist
539 \param opc IL_DAIR/IL_DAAR/IL_DADP/IL_DASP/IL_ARGxx (x86)
540 \param argili
541
542 Add argument expression \p argili to existing argument list \p arglist using
543 opcode \p opc. If \p arglist = 0, begin a new list.
544 */
545static int
546jsrAddArg(int arglist, ILI_OP opc, int argili)
547{
548 int rg;
549 int ili;
550
551 if (arglist == 0) {
552 arglist = ad1ili(IL_NULL, 0);
553 availIreg = 0;
554 availFreg = 0;
555 }
556 /*
557 * WARNING: For the x86, this implies that the standard call mechanism is
558 * being used. If there are multiple arguments, they need to be pushed
559 * on the stack in reverse order (first jsrAddArg() call is for the last
560 * argument, ...).
561 */
562 switch (opc) {
563 case IL_ARGAR:
564 ili = ad3ili(IL_ARGAR, argili, arglist, 0);
565 return ili;
566 case IL_ARGIR:
567 case IL_ARGKR:
568 case IL_ARGSP:
569 case IL_ARGDP:
570#ifdef TARGET_SUPPORTS_QUADFP
571 case IL_ARGQP:
572#endif
573 ili = ad2ili(opc, argili, arglist);
574 return ili;
575 default:
576 /* allow arguments to be passed in registers and on the stack */
577 break;
578 }
579 assert(is_daili_opcode(opc), "jsrAddArg: invalid opcode", opc, ERR_Fatal);
580 if (opc == IL_DAIR || opc == IL_DAAR || opc == IL_DAKR) {
581 rg = IR(availIreg++);
582 } else {
583 if (opc == IL_DADP && (availFreg & 1))
584 availFreg++;
585 rg = SP(availFreg);
586 availFreg++;
587 if (opc == IL_DADP)
588 availFreg++;
589 }
590
591 ili = ad3ili(opc, argili, rg, arglist);
592 return ili;
593}

595/** \brief Return the ili of a call to a function with name fname, and argument
596 * list argili. If argili = 0, argument list is empty.
597 *
598 * \param fname function name
599 * \param opc IL_QJSR/IL_JSR
600 * \param argili argument list
601 */
602static int
603makeCall(const char *fname, ILI_OP opc, int argili)
604{
605 int ili;
606 bool old_share_proc, old_share_qjsr;
607
608 if (argili == 0) {
609 argili = ad1ili(IL_NULL, 0);
610 availIreg = 0;
611 availFreg = 0;
612 }
613
614 old_share_proc = share_proc_ili;
615 old_share_qjsr = share_qjsr_ili;
616 share_proc_ili = false;
617 share_qjsr_ili = false;
618 ili = ad2ili(opc, mkfunc(fname), argili);
619 share_proc_ili = old_share_proc;
620 share_qjsr_ili = old_share_qjsr;
621
622 if (availFreg > 0 && availFreg < 4)
623 availFreg = 4;
624 if (availIreg > maxIreg)
625 maxIreg = availIreg;
626 if (availFreg > maxFreg)
627 maxFreg = availFreg;
628
629 return ili;
630}

632static void
633addCopyinInplace(const sptrListT *list)
634{
635 int i, ili, n_elts, dest_nme, argili, call;
636 int master_ili;
637 SPTR lab;
638 int altili, func;
639 SPTR sptr;
640 int indirect_load;
641 const sptrListT *node;
642
643 n_elts = 0;
644 lab = getlab();
645 for (node = list, i = 0; node; node = node->next, ++i) {
646 int sptr_nme, sptr_ili;
647
648 sptr = node->o_sptr;
649 indirect_load = 0;
650 if (STYPEG(sptr) == ST_CMBLK) {
651 sptr = CMEMFG(sptr);
652 if (!sptr)
653 continue;
654 } else if (SCG(sptr) == SC_BASED && POINTERG(sptr)) {
655 if (ALLOCATTRG(sptr)) {
656 indirect_load = 1;
657 }
658 sptr = MIDNUMG(sptr);
659 }
660 /* what we have here it TPxx, need to find the symbol it points to */
661 /* master copy - should be passed from previous region */
662 master_ili = mk_address(sptr);
663
664 /* current threadprivate copy */
665 sptr_ili = llGetThreadprivateAddr(node->sptr);
666 if (indirect_load == 1) {
667 sptr_nme = addnme(NT_VAR, sptr, 0, 0);
668 sptr_ili = ad2ili(IL_LDA, sptr_ili, sptr_nme);
669 master_ili = ad2ili(IL_LDA, master_ili, sptr_nme);
670 }
671 dest_nme = ILI_OPND(sptr_ili, 2);
672
673 if (n_elts == 0) {
674 ili = ad4ili(IL_ACJMP, sptr_ili, master_ili, CC_EQ, lab);
675 RFCNTI(lab);
676 chk_block(ili);
677 n_elts = 1;
678 }
679
680 /* now do a copy */
681 altili = 0;
682 {
683 func = mkMemcpy();

701 cr_block();

793static void
794setTaskloopVars(SPTR lb, SPTR ub, SPTR stride, SPTR lastitr)
795{
796 int nme, basenm, baseili, ili;
797 SPTR arg;
798 int oldbih;
799 ILI_OP ld, st;
800 MSZ msz;
801
802 oldbih = expb.curbih;
803 /* This code is in an outlined taskloop routine.
804 * Load taskloop vars from arg1 to local/private vars.
805 */
806 arg = ll_get_hostprog_arg(GBL_CURRFUNC, 2);
807 basenm = addnme(NT_VAR, arg, 0, 0);
808 baseili = ad_acon(arg, 0);
809 baseili = mk_address(arg);
810 arg = mk_argasym(arg);
811 basenm = addnme(NT_VAR, arg, 0, (INT)0);
812 baseili = ad2ili(IL_LDA, baseili, basenm);
813 nme = addnme(NT_IND, lb, basenm, 0);
814 ili = ad3ili(IL_AADD, baseili, ad_aconi(TASK_LPVAR_OFFSET), 0);
815 ldst_msz(DT_INT8, &ld, &st, &msz);
816 ili = ad3ili(ld, ili, nme, msz);
817 ldst_msz(DTYPEG(lb), &ld, &st, &msz);
818 if (msz != MSZ_I8)
819 ili = kimove(ili);
820 ili = ad4ili(st, ili, mk_address(lb), addnme(NT_VAR, lb, 0, 0), msz);
821 chk_block(ili);
822
823 nme = addnme(NT_IND, ub, basenm, 0);
824 ili = ad3ili(IL_AADD, baseili,
825 ad_aconi(TASK_LPVAR_OFFSET + zsize_of(DT_INT8)), 0);
826 ldst_msz(DT_INT8, &ld, &st, &msz);
827 ili = ad3ili(ld, ili, nme, msz);
828 ldst_msz(DTYPEG(ub), &ld, &st, &msz);
829 if (msz != MSZ_I8)
830 ili = kimove(ili);
831 ili = ad4ili(st, ili, mk_address(ub), addnme(NT_VAR, ub, 0, 0), msz);
832 chk_block(ili);
833
834 if (STYPEG(stride) != ST_CONST) {
835 nme = addnme(NT_IND, stride, basenm, 0);
836 ili = ad3ili(IL_AADD, baseili,
837 ad_aconi(TASK_LPVAR_OFFSET + (zsize_of(DT_INT8) * 2)), 0);
838 ldst_msz(DT_INT8, &ld, &st, &msz);
839 ili = ad3ili(ld, ili, nme, msz);
840 ldst_msz(DTYPEG(stride), &ld, &st, &msz);
841 if (msz != MSZ_I8)
842 ili = kimove(ili);
843 ili =
844 ad4ili(st, ili, mk_address(stride), addnme(NT_VAR, stride, 0, 0), msz);
845 chk_block(ili);
846 }
847
848 if (lastitr && STYPEG(lastitr) != ST_CONST) {
849 nme = addnme(NT_IND, lastitr, basenm, 0);
850 ldst_msz(DT_INT, &ld, &st, &msz);
851 ili = ad3ili(IL_AADD, baseili,
852 ad_aconi(TASK_LPVAR_OFFSET + (zsize_of(DT_INT8) * 3)), 0);
853 ili = ad3ili(ld, ili, nme, msz);
854 ldst_msz(DTYPEG(lastitr), &ld, &st, &msz);
855 ili = ad4ili(st, ili, ad_acon(lastitr, 0), addnme(NT_VAR, lastitr, 0, 0),
856 msz);
857 chk_block(ili);
858 }
859 if (oldbih == expb.curbih) {
860 wr_block();
861 cr_block();
862 }
863}

912void
913exp_smp(ILM_OP opc, ILM *ilmp, int curilm)
914{
915#ifdef IM_BPAR
916 int ili = 0;
917 SPTR sym;
918 SPTR sptr;
919 int offset, savebih;
920 int off;
921 int addr, stili;
922 int argilm;
923 SPTR tpv;
924 int pv;
925 int savex14;
926 const char *doschedule;
927 int semaphore;
928#if defined(OMP_OFFLOAD_LLVM) || defined(OMP_OFFLOAD_PGI)
929 int dotarget;
930 SPTR beg_label;
931#endif

946 static int target_ili_thread_limit= 0;
947#endif
948 int sz;
949 ISZ_T size, num_elements;
950 static int isTargetDevice = 0;
951 switch (opc) {
952 case IM_BPAR:
953 case IM_BPARN:
954 case IM_BPARD:
955 case IM_BPARA:
956 case IM_EPAR:
957 case IM_EPARD:
958 case IM_BTASK:
959 case IM_BTASKLOOP:
960 case IM_ETASK:
961 case IM_ETASKLOOP:
962 case IM_BTARGET:
963 case IM_ETARGET:
964 case IM_BTEAMS:
965 case IM_BTEAMSN:
966 case IM_ETEAMS:

973 case IM_ETASKLOOPREG:
974 case IM_BTASKDUP:
975 case IM_ETASKDUP:
976 break;
977 default:
978 ll_rewrite_ilms(-1, curilm, 0);
979 break;
980 }
981 switch (opc) {
982 case IM_BMPSCOPE:
983 if (ll_ilm_is_rewriting())
984 break;
985 scopeSptr = ILM_SymOPND(ilmp, 1);
986#ifdef PARUPLEVELG
987 uplevel_sptr = PARUPLEVELG(scopeSptr);
988#else
989 uplevel_sptr = 0;
990#endif
991 break;
992 case IM_EMPSCOPE:
993 break;
994 case IM_BPAR:
995 case IM_BPARN:
996 case IM_BPARA:
997#ifdef OMP_OFFLOAD_LLVM
998 if (flg.omptarget && gbl.ompaccel_intarget) {
999 exp_ompaccel_bpar(ilmp, curilm, uplevel_sptr, scopeSptr, incrOutlinedCnt);
1000 break;
1001 }
1002#endif
1003 if (flg.opt != 0) {
1004 wr_block();
1005 cr_block();
1006 }
1007 incrOutlinedCnt();
1008 BIH_FT(expb.curbih) = true;
1009 BIH_QJSR(expb.curbih) = true;
1010 BIH_NOMERGE(expb.curbih) = true;
1011 if (gbl.outlined)
1012 expb.sc = SC_PRIVATE;
1013 if (outlinedCnt == 1) {
1014 int isPar = ILI_OF(ILM_OPND(ilmp, 1));
1015 SPTR par_label, end_label;
1016 int iliarg, nthreads, proc_bind;
1017 {
1018 sptr = ll_make_outlined_func_wopc((SPTR)uplevel_sptr, scopeSptr, opc);
1019 if (!PARENCLFUNCG(scopeSptr))
1020 PARENCLFUNCP(scopeSptr, sptr);
1021 }
1022 ll_write_ilm_header(sptr, curilm);
1023 iliarg = ll_load_outlined_args(scopeSptr, sptr, gbl.outlined);
1024
1025 /* if (isPar == 0)
1026 call omp_do_func(.....)
1027 goto do_end;

1029 call kmpc_fork_call (....., omp_do_func,.... )
1030 do_label:
1031 */

1034 end_label = getlab();
1035

1038 chk_block(isPar);
1039
1040 ili = ll_make_kmpc_serialized_parallel();
1041 iltb.callfg = 1;
1042 chk_block(ili);
1043
1044 ili = ll_make_outlined_call2(sptr, iliarg);
1045 iltb.callfg = 1;
1046 chk_block(ili);
1047
1048 ili = ll_make_kmpc_end_serialized_parallel();
1049 iltb.callfg = 1;
1050 chk_block(ili);
1051
1052 ili = ad1ili(IL_JMP, end_label);
1053 RFCNTI(end_label);
1054 chk_block(ili);
1055
1056 wr_block();
1057 cr_block();
1058 exp_label(par_label);
1059 proc_bind = 0;
1060 if (opc == IM_BPARA) {
1061 int flag = ILM_OPND(ilmp, 3);
1062 if (flag & 0x2) {
1063 nthreads = ILI_OF(ILM_OPND(ilmp, 2));
1064 ili = ll_make_kmpc_push_num_threads(nthreads);
1065 iltb.callfg = 1;
1066 chk_block(ili);
1067 }
1068 if (flag & 0x1) {
1069 proc_bind = ILM_OPND(ilmp, 4);
1070 }
1071 }
1072 if (opc == IM_BPARN) {
1073 nthreads = ILI_OF(ILM_OPND(ilmp, 2));
1074 ili = ll_make_kmpc_push_num_threads(nthreads);
1075 iltb.callfg = 1;
1076 chk_block(ili);
1077 }
1078 if (proc_bind) {
1079 proc_bind = ad_icon(proc_bind);
1080 ili = ll_make_kmpc_push_proc_bind(proc_bind);
1081 iltb.callfg = 1;
1082 chk_block(ili);
1083 }
1084 ili = ll_make_kmpc_fork_call(sptr, 1, &iliarg, OPENMP, -1);
1085 iltb.callfg = 1;
1086 chk_block(ili);
1087
1088 exp_label(end_label);
1089
1090 ccff_info(MSGOPENMP, "OMP001", gbl.findex, gbl.lineno,
1091 "Parallel region activated", NULL);
1092
1093 } else if (outlinedCnt > 1) {
1094 ll_rewrite_ilms(-1, curilm, 0);
1095 }
1096
1097 break;
1098 case IM_BPARD:
1099 /* lexically nested begin parallel */
1100 incrOutlinedCnt();
1101 if (outlinedCnt > 1) {
1102 ll_rewrite_ilms(-1, curilm, 0);
1103 break;
1104 }
1105 if (flg.opt != 0) {
1106 wr_block();
1107 cr_block();
1108 }
1109 BIH_FT(expb.curbih) = true;
1110 BIH_QJSR(expb.curbih) = true;
1111 BIH_NOMERGE(expb.curbih) = true;
1112 if (outlinedCnt == 1) {
1113 int isPar = ILI_OF(ILM_OPND(ilmp, 1));
1114 SPTR par_label, end_label;
1115 int iliarg, proc_bind;
1116 sptr = ll_make_outlined_func((SPTR)uplevel_sptr, scopeSptr);
1117 if (!PARENCLFUNCG(scopeSptr))
1118 PARENCLFUNCP(scopeSptr, sptr);
1119 ll_write_ilm_header(sptr, curilm);
1120 iliarg = ll_load_outlined_args(scopeSptr, sptr, gbl.outlined);
1121
1122 /* if (isPar == 0)
1123 call omp_do_func(.....)
1124 goto do_end;
1125 par_label:
1126 call kmpc_fork_call (....., omp_do_func,.... )
1127 do_label:
1128 */
1129
1130 par_label = getlab();
1131 end_label = getlab();
1132
1133 isPar = ad3ili(IL_ICJMPZ, isPar, CC_EQ, par_label);
1134 RFCNTI(par_label);
1135 chk_block(isPar);
1136
1137 ili = ll_make_kmpc_serialized_parallel();
1138 iltb.callfg = 1;
1139 chk_block(ili);
1140
1141 ili = ll_make_outlined_call2(sptr, iliarg);
1142 iltb.callfg = 1;
1143 chk_block(ili);
1144
1145 ili = ll_make_kmpc_end_serialized_parallel();
1146 iltb.callfg = 1;
1147 chk_block(ili);
1148
1149 ili = ad1ili(IL_JMP, end_label);
1150 RFCNTI(end_label);
1151 chk_block(ili);
1152
1153 proc_bind = ILM_OPND(ilmp, 2);
1154 if (proc_bind) {
1155 proc_bind = ad_icon(proc_bind);
1156 ili = ll_make_kmpc_push_proc_bind(proc_bind);
1157 iltb.callfg = 1;
1158 chk_block(ili);
1159 }
1160
1161 wr_block();
1162 cr_block();
1163 exp_label(par_label);
1164 ili = ll_make_kmpc_fork_call(sptr, 1, &iliarg, OPENMP, -1);
1165 iltb.callfg = 1;
1166 chk_block(ili);
1167
1168 wr_block();
1169 cr_block();
1170 exp_label(end_label);
1171 }
1172 ccff_info(MSGOPENMP, "OMP001", gbl.findex, gbl.lineno,
1173 "Parallel region activated", NULL);
1174 break;
1175 case IM_BTEAMS:
1176 case IM_BTEAMSN:
1177#ifdef OMP_OFFLOAD_LLVM
1178 if(flg.omptarget && gbl.ompaccel_intarget) {
1179 exp_ompaccel_bteams(ilmp, curilm, outlinedCnt, uplevel_sptr, scopeSptr, incrOutlinedCnt);
1180 break;
1181 }
1182#endif
1183 if (flg.opt != 0) {
1184 wr_block();
1185 cr_block();
1186 }
1187 incrOutlinedCnt();
1188 BIH_FT(expb.curbih) = true;
1189 BIH_QJSR(expb.curbih) = true;
1190 BIH_NOMERGE(expb.curbih) = true;
1191 if (gbl.outlined)
1192 expb.sc = SC_PRIVATE;
1193 if (outlinedCnt == 1) {
1194 SPTR par_label;
1195 int iliarg, nteams, n_limit;
1196 {
1197 sptr = ll_make_outlined_func_wopc((SPTR)uplevel_sptr, scopeSptr, opc);
1198 }
1199 if (!PARENCLFUNCG(scopeSptr))
1200 PARENCLFUNCP(scopeSptr, sptr);
1201 ll_write_ilm_header(sptr, curilm);
1202 iliarg = ll_load_outlined_args(scopeSptr, sptr, gbl.outlined);
1203
1204 par_label = getlab();
1205

1207 cr_block();
1208 exp_label(par_label);

1222
1223 } else if (outlinedCnt > 1) {
1224 ll_rewrite_ilms(-1, curilm, 0);
1225 }
1226
1227 break;
1228
1229 case IM_BTARGET:
1230#ifdef OMP_OFFLOAD_LLVM

1232 exp_ompaccel_btarget(ilmp, curilm, uplevel_sptr, scopeSptr, incrOutlinedCnt, &targetfunc_sptr, &isTargetDevice);
1233 break;

1235#endif
1236 /* lexically nested begin parallel */
1237 incrOutlinedCnt();

1240 break;
1241 }
1242 if (flg.opt != 0) {

1244 cr_block();

1247 BIH_QJSR(expb.curbih) = true;

1250 isTargetDevice = ILI_OF(ILM_OPND(ilmp, 1));

1257 "Target region activated", NULL);
1258 break;
1259 case IM_ETARGET:

1262 }
1263 decrOutlinedCnt();
1264 if (outlinedCnt >= 1) {

1297 }
1298 targetfunc_sptr = SPTR_NULL;
1299 ccff_info(MSGOPENMP, "OMP021", gbl.findex, gbl.lineno,
1300 "Target region terminated", NULL);
1301 break;

3112/** \brief C and Fortran threadprivate : for simple POD */
3113static int
3114allocThreadprivate(SPTR sym, int *tmpthr)
3115{
3116 SPTR cm;
3117 int size;
3118 int adr_vector;
3119 int adr_cm;
3120
3121 cm = MIDNUMG(sym); /* corresponding common block or threadprivate var */
3122 if (STYPEG(cm) == ST_CMBLK) {
3123 adr_cm = ad_acon(CMEMFG(cm), 0); /* &cmn_block */
3124 size = ad_icon((INT)SIZEG(cm));
3125 }
3126 else if (SCG(cm) == SC_BASED && POINTERG(cm)) {
3127 /*
3128 * Cannot rely on the SYMLK chain appearing as
3129 * $p -> $o -> $sd
3130 * Apparently, these links only occur for the
3131 * pointer's internal variables if the pointer
3132 * does not have the SAVE attribute. Without
3133 * these fields, the correct size of the threads'
3134 * copies cannot be computed.
3135 * Just explicitly look for the internal pointer
3136 * and descriptor. If the descriptor is present,
3137 * can assume that there is an offest variable which
3138 * only needs to be accounted for in the size
3139 * computation of the threads' copies.
3140 * Setup up the MIDNUM fields as follows where
3141 * foo is the symtab entry which has the POINTER
3142 * flag set:
3143 * foo -> foo$p
3144 * TPpfoo -> foo
3145 * foo$p -> TPpfoo
3146 * foo$sd -> TPpfoo
3147 * Note that foo's SDSC -> foo$sd.
3148 * Before we had:
3149 * foo -> TPpfoo
3150 * TPpfoo -> foo$p
3151 * which is a problem for computing the size
3152 * when starting with TPpfoo.
3153 */
3154 SPTR tptr;
3155 int sdsptr;
3156 tptr = MIDNUMG(cm);
3157 adr_cm = ad_acon(tptr, 0); /* &tp_var */
3158 size = size_of(DTYPEG(tptr));
3159 sdsptr = SDSCG(cm); /* $sd */
3160 if (sdsptr) {
3161 size += size_of(DT_ADDR); /* $o */
3162 size += size_of(DTYPEG(sdsptr)); /* $sd */
3163 }
3164 size = ad_icon(size);
3165 }
3166 else if (DTY(DTYPEG(cm)) == TY_PTR) {
3167 /*
3168 * Given the above code for POINTER, this code is
3169 * probably dead, but leave it just in case.
3170 */
3171 adr_cm = ad_acon(cm, 0); /* &tp_var */
3172 size = size_of(DTYPEG(cm));
3173 if (SYMLKG(cm) != NOSYM) {
3174 size += size_of(DTYPEG(SYMLKG(cm))); /* $o */
3175 if (SYMLKG(SYMLKG(cm)) != NOSYM) {
3176 size += size_of(DTYPEG(SYMLKG(SYMLKG(cm)))); /* $sd */
3177 }
3178 }
3179 size = ad_icon(size);
3180 } else if (SCG(sym) == SC_BASED) {
3181 adr_cm = ad_acon(cm, 0); /* &tp_var */
3182 size = ad_icon(size_of(DTYPEG(cm)));
3183 } else {
3184 adr_cm = ad_acon(cm, 0); /* &tp_var */
3185 size = ad_icon(size_of(DTYPEG(cm)));
3186 }
3187 adr_vector = ad_acon(sym, 0); /* &cmn_vector/tp_vector */
3188
3189 if (tmpthr) {
3190 int tili;
3191 SPTR tsym;
3192 ll_set_new_threadprivate(sym);
3193 tsym = THPRVTOPTG(sym);
3194 if (XBIT(69, 0x80)) { /* experiment flag */
3195 tili = _make_mp_get_threadprivate(adr_cm, size, adr_vector);
3196 *tmpthr = ad3ili(IL_STA, tili, adr_vector, addnme(NT_VAR, sym, 0, 0));
3197 } else {
