21F_VISIBILITY_VEC
22vfloat cexpf_vec(vfloat x)
23{
24 // Algorithm description for cexp(x)
25 // We follow the mathematical definition of the cexp
26 // cexp(re + I*im) = exp(re)*(cos(im) + I*sin(im))
27 // and also handle C99 special cases separately
28 //
29 // sin and cos will be computed in parallel and returned
30 // in the same SIMD register interleaved and placed properly
31 // for real and imaginary. We will use single precision for sin/cos.
32 //
33 // We will compute exp() as a pair of (poly, scale), where integer
34 // scale will give us an extended range for exp = 2^(scale) * poly
35 // exp result will be delivered in a pair of SIMD registers, real and
36 // imaginary positions will be duplicates.
37 //
38 // We multiply poly by sin/cos - this incurs one roundoff in
39 // every component and then we do ldexp to carefully multiply by
40 // 2^(scale).
41
42 vfloat rx = vmoveldup_vf_vf(x); PRINT(rx);
43 vfloat ix = vmovehdup_vf_vf(x); PRINT(ix);
44 // sign of resulting poly is 0, except maybe if NaN
45 vfloat poly, scale; __vexp_kernel(rx, &poly, &scale); PRINT(poly); PRINT(scale);
46 // cis(Inf & NaN) --> NaN
47 vfloat rcis = __vcis_kernel(ix); PRINT(rcis);
48 // store sign of cis result
49 vint2 signcis = vand_vi2_vi2_vi2(vF2I(rcis), vSETi(FL_SIGN_BIT)); PRINT(signcis);
50 // sign of NaN may be lost here in favor of sign of NaN coming from poly
51 vfloat polycis = vmul_vf_vf_vf(rcis, poly); PRINT(polycis);
52 // NaN sign fixup, perhaps not worth the effort
53 polycis = vI2F(vandnot_vi2_vi2_vi2(vSETi(FL_SIGN_BIT), vF2I(polycis))); PRINT(polycis);
54 polycis = vI2F(vor_vi2_vi2_vi2(vF2I(polycis), signcis)); PRINT(polycis);
55
56 // if creal(x) == +Inf, then creal(result) is +Inf
57 // if cimag(x) == 0.0 then fixup product to
58 // the same zero, even if poly were NaN
59 // NOTE: cimag(x) may be denormal under DAZ flag
60 // subsequent computation in ldexp will flush
61 // it to zero if done under the same DAZ condition
62 vopmask reset = veq_vo_vf_vf(x, vI2F(vSETLLi(0x0, FL_PINF))); PRINT(reset);
63 polycis = vsel_vf_vo_vf_vf(reset, x, polycis); PRINT(polycis);
64
65 // if creal(x) == -Inf, then result is +0 * sign_of_cis()
66 // NOTE: this fixup is only needed in case cimag(x)=Inf/NaN.
67 // Finite cases would deliver proper zero thanks to ldexp.
68 vopmask zeromask = veq_vo_vf_vf(rx, vI2F(vSETi(FL_NINF))); PRINT(zeromask);
69 polycis = vsel_vf_vo_vf_vf(zeromask, vI2F(signcis), polycis); PRINT(polycis);
70
71 // careful polycis * 2^(scale)
72 vfloat vcexp = __vldexpf_kernel(polycis, scale); PRINT(vcexp);
73 return vcexp;
74}

78F_VISIBILITY_SCALAR
79float _Complex cexpf_scalar(float _Complex a)
80{
81#if defined _MEASURE_BASELINE_
82return cexpf(a);
83#endif
84#if defined DO_PRINT
85 feclearexcept(FE_ALL_EXCEPT);
86#endif
87 float ra = crealf(a); PRINT(ra);
88 float ia = cimagf(a); PRINT(ia);
89
90#if !(defined __USE_PORTABLE_CODE)
91 // optimize for performance by calling 2-simd function
92 // fill in unused simd slots with safe values
93
94 unsigned long long int ua = F2I(ra) | (((unsigned long long int)F2I(ia)) << 32);
95 // load zeroes out the upper slots, zero is a safe value for cexp
96 vfloat va = _mm_castpd_ps(_mm_load_sd((double const*)(&ua))); PRINT(va);
97 vfloat vr = cexpf_vec(va); PRINT(vr);
98 float _Complex res = *(float _Complex *)(&vr); PRINT(res);
99 return res;
100#else
101
102 float poly;
103 int scale;
104 // This exp clamps input and doesn't over/underflow
105 __exp_scalar_kernel(ra, &poly, &scale); PRINT( poly ); PRINT( scale );
106 assert(((poly > 0x1.p63) && (poly < 0x1.p65)) || isnanf(poly));
107
108 float _Complex cmplx_cis = __cis_scalar_kernel(ia);
109 float rsin = cimagf(cmplx_cis); PRINT( rsin );
110 float rcos = crealf(cmplx_cis); PRINT( rcos );
111 // cis(Inf/NaN) results in NaN
112 assert((isinff(ia) || isnanf(ia)) ^ !(isnanf(rsin) && isnanf(rcos)));
113
114 // sign and payload of NaN from cis may be lost here
115 // in favor of sign/payload of NaN coming from poly
116 float polycos = poly * rcos; PRINT( polycos );
117 float polysin = poly * rsin; PRINT( polysin );
118 // restore sign of NaN coming from cis only to pass
119 // symmetry test, perhaps not worth the effort
120 polycos = copysignf(polycos, rcos);
121 polysin = copysignf(polysin, rsin);
122
123 if ( ia == 0.0f )
124 {
125 // if cimag(x) == 0.0 then fixup product to
126 // the same zero, even if poly were NaN
127 polysin = ia;
128 // NOTE: ia may be denormal under DAZ flag
129 // subsequent computation in ldexp will flush
130 // it to zero if done under the same DAZ condition
131 }
132 if ( ra == I2F(FL_PINF) )
133 {
134 // if creal(x) == +Inf, then creal(result) is +Inf
135 polycos = ra;
136 }
137 if ( ra == I2F(FL_NINF) )
138 {
139 // if creal(x) == -Inf, then result is +0 * sign_of_cis()
140 // NOTE: this fixup is only needed in case cimag(x)=Inf/NaN.
141 // Finite cases would deliver proper zero thanks to ldexp.
142 polycos = copysignf(0.0f, rcos);
143 polysin = copysignf(0.0f, rsin);
144 }
145
146 // This scaling shall not produce new NaNs
147 float cexp_real = __ldexpf_scalar_kernel(polycos, scale); PRINT( cexp_real );
148 float cexp_imag = __ldexpf_scalar_kernel(polysin, scale); PRINT( cexp_imag );
149
150 return set_cmplx(cexp_real, cexp_imag);
151#endif
152}
