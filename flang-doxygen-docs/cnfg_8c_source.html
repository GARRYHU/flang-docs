16/* WARNING: cnfg.c generates two objects, cnfg.o and pgfcnfg.o
17 * define global variable which contains the configurable items:
18 *
19 * default_name - sprintf string used to construct the default name of a
20 * file upon an open of unit which does not contain a file
21 * specifier and is not a scratch file.
22 *
23 * true_mask - mask value used to determine if a value is true or false;
24 * defined so that if the expression (true_mask & val) is
25 * non-zero, the value is true.
26 * 1 => bottom bit is tested (odd => true)
27 * -1 => non-zero value is true
28 * Default is 1 (VAX-style)
29 */
30
31FIO_CNFG __fortio_cnfg_ = { /* ending '_' so it can be accessed by user */
32 "fort.%d", /* default_name */
33 /* vax-style */
34 1, /* odd => true */
35 -1, /* internal value of .TRUE. */
36};
