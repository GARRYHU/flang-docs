28static void
29fill_recurse(int ast)
30{
31 char *s;
32 int l;
33 switch (A_TYPEG(ast)) {
34 case A_ID:
35 s = SYMNAME(A_SPTRG(ast));
36 l = strlen(s);
37 if (Len + l < LEN) {
38 strcpy(Name + Len, s);
39 Len += l;
40 }
41 break;
42 case A_MEM:
43 fill_name(A_LOPG(ast));
44 s = SYMNAME(A_SPTRG(A_MEMG(ast)));
45 l = strlen(s);
46 if (Len + l + 1 < LEN) {
47 strcpy(Name + Len, "%");
48 strcpy(Name + Len + 1, s);
49 Len += l + 1;
50 }
51 break;
52 case A_SUBSCR:
53 fill_name(A_LOPG(ast));
54 break;
55 case A_SUBSTR:
56 fill_name(A_LOPG(ast));
57 break;
58 default:
59 ast_error("unknown ast type for checks", ast);
60 break;
61 }
62} /* fill_recurse */

64static int
65fill_name(int ast)
66{
67 int sname;
68 Len = 0;
69 Name[0] = '\0';
70 fill_recurse(ast);
71 Name[Len] = '\0';
72 Name[Len + 1] = '\0';
73 sname = getstring(Name, Len + 1);
74 return sname;
75} /* fill_name */

77/** \brief Generate a pointer check.
78
79 Generate a call:
80 <pre>
81 ftn_ptrchk(%val(pointer),%val(lineno),%ref("varname"),%ref("filename"))
82 </pre>
83 */
84void
85lower_check_pointer(int ast, int ilm)
86{
87 int sname, ilm1, ilmline, ilmname, ilmfile;
88
89 /* make up the symbol name, lineno, filename */
90 sname = fill_name(ast);
91 ilmline = plower("oS", "ICON", lower_getintcon(lower_line));
92 ilmline = plower("oi", "DPVAL", ilmline);
93 ilmname = plower("oS", "BASE", sname);
94 ilmfile = plower("oS", "BASE", lowersym.sym_chkfile);
95
96 /* make up the call */
97 ilm1 = lower_typeload(DT_ADDR, ilm);
98 ilm1 = plower("oi", "DPVAL", ilm1);
99 plower("onsiiiiC", "CALL", 4, lowersym.sym_ptrchk, ilm1, ilmline, ilmname,
100 ilmfile, lowersym.sym_ptrchk);
101} /* lower_check_pointer */

103/** \brief Generate a subscript check.
104
105 Generate a call for each subscript
106 <pre>
107 ftn_ptrchk(%val(subscript),%val(lower),%val(upper),%val(dimension),
108 %val(lineno),%ref("varname"),%ref("filename"))
109 </pre>
110
111 \a ast is a subscript ast
112
113 If \a sym is zero, derive the name of the array from the subscript ast;
114 otherwise, use the name of sym.
115 */
116
117void
118lower_check_subscript(int sym, int ast, int ndim, int *ilm, int *lower,
119 int *upper)
120{
121 int i, sname, ilm1, ilm2, ilm3, ilm4, ilmline, ilmname, ilmfile;
122
123 /* make up the symbol name, lineno, filename */
124 if (sym == 0)
125 sname = fill_name(ast);
126 else {
127 char *s;
128 s = SYMNAME(sym);
129 Len = strlen(s);
130 if (Len > LEN)
131 Len = LEN;
132 strncpy(Name, s, Len);
133 Name[Len] = '\0';
134 Name[Len + 1] = '\0';
135 sname = getstring(Name, Len + 1);
136 }
137 ilmline = plower("oS", "ICON", lower_getintcon(lower_line));
138 ilmline = plower("oi", "DPVAL", ilmline);
139 ilmname = plower("oS", "BASE", sname);
140 ilmfile = plower("oS", "BASE", lowersym.sym_chkfile);
141
142 for (i = 0; i < ndim; ++i) {
143 /* make up the call */
144 ilm1 = plower("oi", "DPVAL", ilm[i]);
145 if (lower[i] == 0) {
146 lower[i] = plower("oS", lowersym.bnd.con, lowersym.bnd.one);
147 }
148 ilm2 = plower("oi", "DPVAL", lower[i]);
149 if (upper[i] == 0) {
150 upper[i] = plower("oS", lowersym.bnd.con, lowersym.bnd.max);
151 }
152 ilm3 = plower("oi", "DPVAL", upper[i]);
153 ilm4 = plower("oS", "ICON", lower_getintcon(i + 1));
154 ilm4 = plower("oi", "DPVAL", ilm4);
155 plower("onsiiiiiiiC", "CALL", 7, lowersym.sym_subchk, ilm1, ilm2, ilm3,
156 ilm4, ilmline, ilmname, ilmfile, lowersym.sym_subchk);
157 }
158} /* lower_check_subscript */
