68void
69dumpregs(gregset_t *mc)
70{
71 int i;
72 int j;
73 char *pc;
74 uint64_t *regs;
75
76 if (mc == NULL)
77 return; // Not sure if this is possible
78
79 regs = mc;
80
81/*
82 * Output has the following format:
83 * <REG> <HEXADECIMAL> <DECIMAL> <ASCII>
84 * Example:
85 * x0 0x0000000000000001 1 ........
86 * x1 0x0000000000000000 0 ........
87 * x2 0x000003ffffffe9c0 4398046505408 ........
88 * x3 0x000003ffb7bc8254 4396834128468 T.......
89 * ...
90 * x29 0x000003ffffffe750 4398046504784 P.......
91 * x30 0x000003ffb7bc84ac 4396834129068 ........
92 * sp 0x000003ffffffe750 4398046504784 P.......
93 * pc 0x000003ffb7bfd0e0 4396834345184 ........
94 * pstate 0x0000000080000000 2147483648 ........
95 * fault_addr 0x0000000000000000 0 ........
96 */
97
98 for (i = 0 ; (size_t)i < sizeof(xregs) / sizeof(*xregs); ++i) {
99 fprintf(__io_stderr(), "%-8s 0x%016" PRIx64 " %20" PRId64 "\t",
100 xregs[i].s, regs[xregs[i].ro], regs[xregs[i].ro]);
101 pc = (char *)&(regs[xregs[i].ro]);
102 for(j = 0 ; j < 8 ; ++j) {
103 fputc(isprint(pc[j]) ? pc[j] : '.', __io_stderr());
104 }
105 fputs("\n", __io_stderr());
106 }
107
108}
