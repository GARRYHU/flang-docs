140/*
141 * Types are uniqued within a module to save memory and to allow type
142 * equivalence to be tested by pointer comparison.
143 *
144 * Named structs are identified by name. All other types are uniqued by their
145 * structure which is simple to do since cyclic types can only be created by
146 * using named structs.
147 *
148 * The functions types_equal and types_hash below don't expect to be called
149 * with a named struct, they will treat it like an anonynous struct.
150 */
151
152/* How many elements in the sub_types array? */
153static BIGUINT64
154get_num_subtypes(const struct LL_Type_ *type)
155{
156 switch (type->data_type) {
157 case LL_PTR:
158 case LL_ARRAY:
159 case LL_VECTOR:
160 return 1;
161 default:
162 return type->sub_elements;
163 }
164}

166static hash_value_t
167types_hash(hash_key_t key)
168{
169 const struct LL_Type_ *t = (const struct LL_Type_ *)key;
170 hash_accu_t hacc = HASH_ACCU_INIT;
171 unsigned i, nsubtypes;
172
173 HASH_ACCU_ADD(hacc, t->data_type);
174 HASH_ACCU_ADD(hacc, t->flags);
175 HASH_ACCU_ADD(hacc, t->addrspace);
176 HASH_ACCU_ADD(hacc, (unsigned long)t->sub_elements);
177
178 /* Subtypes have already been uniqued, so just hash pointer values. */
179 nsubtypes = get_num_subtypes(t);
180 for (i = 0; i != nsubtypes; i++)
181 HASH_ACCU_ADD(hacc, (unsigned long)t->sub_types[i]);
182
183 HASH_ACCU_FINISH(hacc);
184 return HASH_ACCU_VALUE(hacc);
185}

187static int
188types_equal(hash_key_t key_a, hash_key_t key_b)
189{
190 const struct LL_Type_ *a = (const struct LL_Type_ *)key_a;
191 const struct LL_Type_ *b = (const struct LL_Type_ *)key_b;
192 unsigned i, n;
193
194 if (a->data_type != b->data_type)
195 return false;
196 if (a->flags != b->flags)
197 return false;
198 if (a->addrspace != b->addrspace)
199 return false;
200 if (a->sub_elements != b->sub_elements)
201 return false;
202
203 n = get_num_subtypes(a);
204 for (i = 0; i != n; i++)
205 if (a->sub_types[i] != b->sub_types[i])
206 return false;
207
208 /* FIXME: We really shouldn't be adding named structs to the uniquing
209 * tables, but at the moment we may be creating multiple versions of a
210 * named struct for different address spaces.
211 *
212 * The correct way of handling this is to store the address space in
213 * pointer types only, just like LLVM does it.
214 */
215 if (a->data_type == LL_STRUCT && (a->flags & LL_TYPE_IS_NAMED_STRUCT))
216 return a->str == b->str;
217
218 return true;
219}

223/*
224 * Create a unique name based on a printf-like template.
225 *
226 * 1. Pick a name based on format + ap that isn't already in 'names'.
227 * 2. Copy the name into malloc'ed memory.
228 * 3. Insert new pointer into names.
229 * 4. Return malloced pointer.
230 */
231static char *
232unique_name(hashset_t names, char prefix, const char *format, va_list ap)
233{
234 char buffer[256] = {prefix, 0};
235 size_t prefix_length;
236 unsigned count = 0;
237 int reseeded = 0;
238 char *unique_name;
239
240 /* The return value from vsnprintf() is useless because Microsoft Visual
241 * Studio doesn't follow the standard. */
242 vsnprintf(buffer + 1, sizeof(buffer) - 1, format, ap);
243 buffer[sizeof(buffer) - 1] = '\0';
244 prefix_length = strlen(buffer);
245
246 /* Make room for a ".%u" suffix. */
247 if (prefix_length > sizeof(buffer) - 12)
248 prefix_length = sizeof(buffer) - 12;
249
250 /* Search for a not previously used name. */
251 while (hashset_lookup(names, buffer)) {
252 /* Try a pretty .1, .2, ... .9 suffix sequence at first, but then
253 * switch to a scheme that isn't quadratic time. */
254 if (++count == 10 && !reseeded) {
255 count = 10 * hashset_size(names);
256 reseeded = 1;
257 }
258 sprintf(buffer + prefix_length, ".%u", count);
259 }
260
261 unique_name = strdup(buffer);
262 hashset_insert(names, unique_name);
263
264 return unique_name;
265}

267/*
268 * Hash functions for interned constants.
269 *
270 * LLVM constants are identified by their uniqued type pointer and textual
271 * representation.
272 */
273
274static hash_value_t
275constants_hash(hash_key_t key)
276{
277 const LL_Value *t = (const LL_Value *)key;
278 const unsigned char *p = (const unsigned char *)t->data;
279 hash_accu_t hacc = HASH_ACCU_INIT;
280
281 HASH_ACCU_ADD(hacc, (unsigned long)t->type_struct);
282 for (; *p; p++)
283 HASH_ACCU_ADD(hacc, *p);
284
285 HASH_ACCU_FINISH(hacc);
286 return HASH_ACCU_VALUE(hacc);
287}

301/*
302 * Metadata nodes.
303 *
304 * A metadata node is a tuple of MDRefs, structurally uniqued by a hashmap. The
305 * node header and element array is allocated in contiguous memory.
306 */
307
308static hash_value_t
309mdnode_hash(hash_key_t key)
310{
311 const LL_MDNode *t = (const LL_MDNode *)key;
312 unsigned i;
313 hash_accu_t hacc = HASH_ACCU_INIT;
314
315 HASH_ACCU_ADD(hacc, t->num_elems);
316 for (i = 0; i < t->num_elems; i++) {
317 HASH_ACCU_ADD(hacc, LL_MDREF_kind(t->elem[i]));
318 HASH_ACCU_ADD(hacc, LL_MDREF_value(t->elem[i]));
319 }
320
321 HASH_ACCU_FINISH(hacc);
322 return HASH_ACCU_VALUE(hacc);
323}

325static int
326mdnode_equal(hash_key_t key_a, hash_key_t key_b)
327{
328 const LL_MDNode *a = (const LL_MDNode *)key_a;
329 const LL_MDNode *b = (const LL_MDNode *)key_b;
330 unsigned i;
331
332 if (a->num_elems != b->num_elems)
333 return false;
334
335 for (i = 0; i < a->num_elems; i++)
336 if (a->elem[i] != b->elem[i])
337 return false;
338
339 return true;
340}

383/**
384 \brief Compute the set of IR features in module when generating IR for the
385 specified LLVM version.
386 */
387static void
388compute_ir_feature_vector(LLVMModuleRef module, enum LL_IRVersion vers)
389{
390 if (strncmp(module->target_triple, "nvptx", 5) == 0)
391 module->ir.is_nvvm = 1;
392 if (strncmp(module->target_triple, "spir", 4) == 0)
393 module->ir.is_spir = 1;
394
395 module->ir.version = vers;
396 InitializeDIFlags(&module->ir);
397 if (XBIT(120, 0x200)) {
398 module->ir.dwarf_version = LL_DWARF_Version_2;
399 } else if (XBIT(120, 0x4000)) {
400 module->ir.dwarf_version = LL_DWARF_Version_3;
401 } else if (XBIT(120, 0x1000000)) {
402 module->ir.dwarf_version = LL_DWARF_Version_4;
403 } else if (XBIT(120, 0x2000000)) {
404 module->ir.dwarf_version = LL_DWARF_Version_5;
405 } else { // DWARF 4 is the default
406 module->ir.dwarf_version = LL_DWARF_Version_4;
407 }
408
409 if (ll_feature_versioned_dw_tag(&module->ir)) {
410 /* LLVMDebugVersion 12 was used by LLVM versions 3.1 through 3.5, and we
411 * don't support LLVM versions older than 3.1, so the version as always 12.
412 */
413 module->ir.debug_info_version = 12;
414 } else {
415 /* LLVM 3.6 onwards encodes the debug info version in a module flag
416 * metadata node, and the numbering sequence started over.
417 *
418 * LLVM 3.6 used v2 which had the stringified header fields in metadata
419 * nodes. This format was abandoned in 3.7, and we don't support it.
420 */
421 if (vers >= LL_Version_3_7)
422 module->ir.debug_info_version = 3;
423 else
424 module->ir.debug_info_version = 1;
425 }
426}

481void
482ll_destroy_module(LLVMModuleRef module)
483{
484 struct LL_Function_ *current;
485 struct LL_Function_ *next;
486 struct LL_ManagedMallocs_ *cur_m;
487 struct LL_ManagedMallocs_ *next_m;
488 unsigned i;
489
490 current = module->first;
491 while (current != NULL) {
492 next = current->next;
493 ll_destroy_function(current);
494 current = next;
495 }
496
497 cur_m = module->first_malloc;
498 while (cur_m != NULL) {
499 next_m = cur_m->next;
500 ll_destroy_mem(cur_m);
501 cur_m = next_m;
502 }
503 free(module->module_vars.values);
504 free(module->user_structs.values);
505 hashmap_free(module->user_structs_byid);
506 hashset_free(module->used_type_names);
507 hashset_free(module->used_global_names);
508 hashset_free(module->anon_types);
509
510 free(module->constants);
511 hashmap_free(module->constants_map);
512
513#pragma GCC diagnostic push
514#pragma GCC diagnostic warning "-Wcast-qual"
515 for (i = 0; i < module->mdstrings_count; i++)
516 free((char *)module->mdstrings[i]);
517#pragma GCC diagnostic pop
518 free(module->mdstrings);
519 hashmap_free(module->mdstrings_map);
520
521 for (i = 0; i < module->mdnodes_count; i++)
522 free(module->mdnodes[i]);
523 free(module->mdnodes);
524 hashmap_free(module->mdnodes_map);
525
526 hashmap_free(module->global_debug_map);
527 hashmap_free(module->module_debug_map);
528 hashmap_free(module->common_debug_map);
529 hashmap_free(module->modvar_debug_map);
530
531 for (i = 0; i < MD_NUM_NAMES; i++)
532 free(module->named_mdnodes[i]);
533
534 lldbg_free(module->debug_info);
535 free(module);
536}

583#ifdef _WIN64

592void
593add_linker_directives(LLVMModuleRef module) {
594 if (get_llvm_version() < LL_Version_5_0) {
595 LLMD_Builder mdb = llmd_init(module);
596 char* linker_directive;
597 for (int i = 0; (linker_directive = flg.linker_directives[i]); ++i) {
598 LLMD_Builder submdb = llmd_init(module);
599
600 llmd_add_string(submdb, linker_directive);
601 LL_MDRef submd = llmd_finish(submdb);
602
603 llmd_add_md(mdb, submd);
604 }
605 LL_MDRef md = llmd_finish(mdb);
606
607 LLMD_Builder boilerplate_mdb = llmd_init(module);
608

611 llmd_add_md(boilerplate_mdb, md);
612
613 LL_MDRef boilerplate_md = llmd_finish(boilerplate_mdb);
614 ll_extend_named_md_node(module, MD_llvm_module_flags, boilerplate_md);
615
616 LLMD_Builder debug_mdb = llmd_init(module);
617
618 const int mdVers = ll_feature_versioned_dw_tag(&module->ir) ? 1 :
619 module->ir.debug_info_version;
620
621 llmd_add_i32(debug_mdb, 1);
622 llmd_add_string(debug_mdb, "Debug Info Version");
623 llmd_add_i32(debug_mdb, mdVers);
624
625 LL_MDRef debug_md = llmd_finish(debug_mdb);
626
627 ll_extend_named_md_node(module, MD_llvm_module_flags, debug_md);
628 } else {
629 int i;
630 char *linker_directive;
631 LLMD_Builder mdb = llmd_init(module);
632 for (i = 0; (linker_directive = flg.linker_directives[i]); ++i) {
633 llmd_add_string(mdb, linker_directive);
634 }
635 LL_MDRef linker_md = llmd_finish(mdb);
636 ll_extend_named_md_node(module, MD_llvm_linker_options, linker_md);
637 }

820/**
821 \brief Get the size of a type in bytes without checks
822 \param type The type to be examined
823 \param noSize Set if not null and \p type isn't concrete [output]
824 \return size of an object of type \p type or 0
825 */
826static ISZ_T
827LLTypeGetBytesUnchecked(LL_Type *type, int *noSize)
828{
829 if (noSize)
830 *noSize = 0;
831 switch (type->data_type) {
832 case LL_I1:
833 case LL_I8:
834 return 1;
835 case LL_I16:
836 return 2;
837 case LL_I24:
838 return 3;
839 case LL_I32:
840 return 4;
841 case LL_I40:
842 return 5;
843 case LL_I48:
844 return 6;
845 case LL_I56:
846 return 7;
847 case LL_I64:
848 return 8;
849 case LL_I128:
850 return 16;
851 case LL_I256:
852 return 32;
853 case LL_HALF:
854 return 2;
855 case LL_FLOAT:
856 return 4;
857 case LL_DOUBLE:
858 return 8;
859 case LL_X86_FP80:
860 case LL_FP128:
861 case LL_PPC_FP128:
862 return 16;
863 case LL_PTR:
864 /* FIXME: Use the data layout so we can cross compile. */
865 return sizeof(void *); // 8 * TARGET_PTRSIZE
866 case LL_ARRAY:
867 case LL_VECTOR:
868 return type->sub_elements * ll_type_bytes(type->sub_types[0]);
869 case LL_STRUCT: {
870 unsigned i;
871 ISZ_T sum;
872 if (type->sub_offsets) {
873 /* This should be the size in memory sans assumptions */
874 return type->sub_offsets[type->sub_elements];
875 }
876 sum = 0;
877 /* Note: We're assuming the struct has no implicit padding. */
878 for (i = 0; i < type->sub_elements; i++)
879 sum += ll_type_bytes(type->sub_types[i]);
880 return sum;
881 }
882 default:
883 if (noSize)
884 *noSize = 1;
885 break;
886 }
887 return 0;
888}

1071/*
1072 * Get a uniqued copy of type.
1073 *
1074 * If we already have a type identical to t, return that. Otherwise create a
1075 * copy of t and remember that.
1076 *
1077 * The copy will be shallow - the str and sub_types pointers will be copied
1078 * directly from type.
1079 */
1080static struct LL_Type_ *
1081unique_type(LLVMModuleRef module, const struct LL_Type_ *type)
1082{
1083#pragma GCC diagnostic push
1084#pragma GCC diagnostic warning "-Wcast-qual"
1085 hash_key_t existing = hashset_lookup(module->anon_types, type);
1086 if (existing)
1087 return (struct LL_Type_ *)existing;
1088#pragma GCC diagnostic pop
1089
1090 /* No such type exists. Save a copy. */
1091 struct LL_Type_ *copy =
1092 (struct LL_Type_ *)ll_manage_malloc(module, sizeof(struct LL_Type_));
1093 memcpy(copy, type, sizeof(*copy));
1094 copy->module = module;
1095 hashset_insert(module->anon_types, copy);
1096 return copy;
1097}

1343 * The struct type is identified by the LL_Value returned from

1428 */
1429LL_Type *
1430ll_create_named_struct_type(LLVMModuleRef module, int id, bool unique,
1431 const char *format, ...)
1432{
1433 va_list ap;
1434 struct LL_Type_ *new_type;
1435 LL_Value *struct_value;
1436
1437 if (!unique) {
1438 va_start(ap, format);
1439 struct_value = ll_named_struct_type_exists(module, id, format, ap);
1440 va_end(ap);
1441 if (struct_value)
1442 return struct_value->type_struct;
1443 ll_remove_struct_type(module, id);
1444 }
1445 va_start(ap, format);
1446 new_type = (struct LL_Type_ *)calloc(1, sizeof(struct LL_Type_));
1447 new_type->str = unique_name(module->used_type_names, '%', format, ap);
1448 va_end(ap);
1449
1450 new_type->module = module;
1451 new_type->data_type = LL_STRUCT;
1452 new_type->flags = LL_TYPE_IS_NAMED_STRUCT;
1453
1454 /* Maintain a list of types in the module so we can print them later. */
1455 struct_value = ll_create_value_from_type(module, new_type, new_type->str);
1456 ll_create_sym(&(module->user_structs), module->num_user_structs,
1457 struct_value);
1458 module->num_user_structs++;
1459
1460#pragma GCC diagnostic push
1461#pragma GCC diagnostic warning "-Wcast-qual"
1462 if (id > 0) {
1463 hash_key_t old_id = hashmap_replace(module->user_structs_byid, INT2HKEY(id),
1464 (hash_data_t *)&struct_value);
1465 assert(!old_id, "Duplicate structs created for id.", id, ERR_Fatal);
1466 }
1467#pragma GCC diagnostic pop
1468
1469 return new_type;
1470}

1509/**
1510 \brief Get an anonymous struct type.
1511
1512 Anonymous structs are identified by their contents, they don't have a name.
1513 This will either return an existing anonymous struct type with the required
1514 elements, or create a new type with a copy of the elements array.
1515
1516 A packed struct has alignment 1 and no padding between members.
1517 */
1518LL_Type *
1519ll_create_anon_struct_type(LLVMModuleRef module, LL_Type *elements[],
1520 unsigned num_elements, bool is_packed, int addrspace)
1521{
1522 struct LL_Type_ new_type;
1523 struct LL_Type_ *ret_type;
1524
1525 new_type.str = NULL;
1526 new_type.data_type = LL_STRUCT;
1527 new_type.flags = is_packed ? LL_TYPE_IS_PACKED_STRUCT : 0;
1528 new_type.sub_types = elements;
1529 new_type.sub_offsets = (unsigned *)calloc(num_elements + 1, sizeof(unsigned));
1530 new_type.sub_elements = num_elements;
1531 new_type.sub_padding = NULL;
1532 new_type.addrspace = 0; //addrspace;
1533
1534 unsigned i, offset;
1535 for (i = 0, offset = 0; i < num_elements; ++i) {
1536 new_type.sub_offsets[i] = offset;
1537 offset += ll_type_bytes(new_type.sub_types[i]);
1538 }
1539 new_type.sub_offsets[num_elements] = offset;
1540
1541 ret_type = unique_type(module, &new_type);
1542
1543 if (!ret_type->str) {
1544 if (num_elements == 0) {
1545 ret_type->str = is_packed ? "<{}>" : "{}";
1546 ret_type->sub_types = NULL;
1547 } else {
1548 char *new_str;
1549 unsigned i, len, pos;
1550
1551 /* Build new_str = "{i32, i8}". First compute length to allocate. */
1552 len = 3;
1553 for (i = 0; i < num_elements; i++)
1554 len += strlen(elements[i]->str) + 2;
1555
1556 new_str = (char *)ll_manage_malloc(module, len);
1557 sprintf(new_str, is_packed ? "<{%s" : "{%s", elements[0]->str);
1558 pos = strlen(new_str);
1559 for (i = 1; i < num_elements; i++) {
1560 sprintf(new_str + pos, ", %s", elements[i]->str);
1561 pos += strlen(new_str + pos);
1562 }
1563 strcat(new_str + pos, is_packed ? "}>" : "}");
1564 ret_type->str = new_str;
1565 ret_type->sub_types = (LL_Type **)ll_manage_malloc(
1566 module, num_elements * sizeof(struct LL_Type_ *));
1567 memcpy(ret_type->sub_types, elements,
1568 num_elements * sizeof(struct LL_Type_ *));
1569 }
1570 }
1571
1572 return ret_type;
1573}

1672/*
1673 * Interned constants.
1674 *
1675 * Constants are uniqued by their type_struct pointer (which is already
1676 * uniqued) and the text of their data.
1677 */
1678
1679/* Find or create an LL_Value of the given type and textual representation.
1680 * Like ll_create_value_from_type() with uniquing.
1681 * Return an index into module->constants */
1682static unsigned
1683intern_constant(LLVMModuleRef module, LL_Type *type, const char *data)
1684{
1685 LL_Value temp;
1686 hash_data_t oldval;
1687 LL_Value *newval;
1688 unsigned slot;
1689
1690 memset(&temp, 0, sizeof(temp));
1691 temp.data = data;
1692 temp.type_struct = type;
1693
1694 /* Was this constant seen before? */
1695 if (hashmap_lookup(module->constants_map, &temp, &oldval))
1696 return HKEY2INT(oldval);
1697
1698 /* First time we see this constant. */
1699 newval = ll_create_value_from_type(module, type, data);
1700 slot = module->constants_count;
1701 if (++module->constants_count > module->constants_alloc) {
1702 module->constants_alloc *= 2;
1703 module->constants = (LL_Value **)realloc(module->constants,
1704 module->constants_alloc *
1705 sizeof(module->constants[0]));
1706 }
1707 module->constants[slot] = newval;
1708 hashmap_insert(module->constants_map, newval, INT2HKEY(slot));
1709
1710 return slot;
1711}

1779/* Return the type corresponding to applying one gep index */
1780static LL_Type *
1781apply_gep_index(LL_Type *type, unsigned idx)
1782{
1783 switch (type->data_type) {
1784 case LL_PTR:
1785 case LL_VECTOR:
1786 case LL_ARRAY:
1787 return type->sub_types[0];

1804/**
1805 \brief Get a shared \ref LL_Value representing a constant getelementptr of
1806 the provided pointer value.
1807
1808 This creates a gep-expression that may be used to initialize globals and
1809 metadata. It does not create an LL_GEP instruction.
1810
1811 The num_idx argument is the number of index operands on the gep
1812 expression. It should be at least 1.
1813
1814 The gep indices following num_idx should be ints.
1815
1816 Note: element type is the exected element type, this does not get elements of
1817 any different type.
1818 */
1819LL_Value *
1820ll_get_const_gep(LLVMModuleRef module, LL_Value *ptr, unsigned num_idx, ...)
1821{
1822 va_list ap;
1823 unsigned slot;
1824 char *pointee;
1825 /* Space for getelementptr(<ptr>, i32 <idx0>, i32 <idx1>, ...) */
1826 char *name = (char *)malloc(19 + strlen(ptr->type_struct->str) +
1827 2 * strlen(ptr->data) + 16 * num_idx);
1828 char *p = name;
1829 LL_Type *type = ptr->type_struct;
1830
1831 /*** getelementpointer can only be used on pointers ***/
1832 assert(num_idx >= 1, "ll_get_const_gep: Need at least one index.", num_idx,
1833 ERR_Fatal);
1834 assert(type->data_type == LL_PTR,
1835 "ll_get_const_gep: "
1836 "Expected a pointer type.",
1837 type->data_type, ERR_Fatal);
1838
1839 /*** Compose pointee type string ****/
1840 pointee = (char *)malloc(3 + strlen(ptr->type_struct->sub_types[0]->str));
1841 pointee[0] = '\0';
1842
1843 /* Not every version of LLVM requires pointee type for GEP */
1844 if (ll_feature_explicit_gep_load_type(&module->ir))
1845 sprintf(pointee, "%s, ", ptr->type_struct->sub_types[0]->str);
1846
1847 /*** Put everything together ***/
1848 sprintf(p, "getelementptr(%s%s %s", pointee, ptr->type_struct->str,
1849 ptr->data);
1850 p += strlen(p);
1851
1852 va_start(ap, num_idx);
1853 while (num_idx--) {
1854 int idx = va_arg(ap, int);
1855 sprintf(p, ", i32 %d", idx);
1856 p += strlen(p);
1857 type = apply_gep_index(type, idx);
1858 }
1859 va_end(ap);
1860 sprintf(p, ")");
1861
1862 /* getelementptr produces a pointer in the same address space as
1863 * the original pointer. */
1864 type =
1865 ll_get_addrspace_type(type, ll_get_pointer_addrspace(ptr->type_struct));
1866 type = ll_get_pointer_type(type);
1867
1868 slot = intern_constant(module, type, name);
1869
1870 free(name);
1871 free(pointee);
1872 return module->constants[slot];
1873}

1984#define NEEDS_ESC(C) ((C) < 32 || (C) >= 127 || (C) == '\\' || (C) == '"')
1985
1986/**
1987 \brief Get an LL_MDRef representing a raw binary string.
1988
1989 The string can contain arbitrary bytes, including nuls; appropriate escaping
1990 will be added.
1991 */
1992LL_MDRef
1993ll_get_md_rawstring(LLVMModuleRef module, const void *rawstr, size_t length)
1994{
1995 const unsigned char *ustr = (const unsigned char *)rawstr;
1996 unsigned num_escapes = 0;
1997 unsigned slot;
1998 size_t i;
1999 char *str, *p;
2000 hash_data_t oldval;
2001 LL_MDRef mdref = LL_MDREF_INITIALIZER(MDRef_String, 0);
2002
2003 /* Count the number of bytes that need escaping. */
2004 for (i = 0; i < length; i++)
2005 if (NEEDS_ESC(ustr[i]))
2006 ++num_escapes;
2007
2008 /* Make a copy with escaped bytes, a !" prefix and a "\0 suffix. */
2009 p = str = (char *)malloc(length + 3 * num_escapes + 4);
2010 *p++ = '!';
2011 *p++ = '"';
2012 for (i = 0; i < length; i++) {
2013 if (NEEDS_ESC(ustr[i])) {
2014 sprintf(p, "\\%02x", ustr[i]);
2015 p += 3;
2016 } else {
2017 *p++ = ustr[i];
2018 }
2019 }
2020 *p++ = '"';
2021 *p++ = 0;
2022
2023 /* Is this a known string? */
2024 if (hashmap_lookup(module->mdstrings_map, str, &oldval)) {
2025 mdref = LL_MDREF_ctor(mdref, HKEY2INT(oldval));
2026 free(str);
2027 return mdref;
2028 }
2029
2030 /* This string hasn't been seen before. Allocate a new slot. */
2031 slot = module->mdstrings_count;
2032 if (++module->mdstrings_count > module->mdstrings_alloc) {
2033 module->mdstrings_alloc *= 2;
2034 module->mdstrings = (const char **)realloc(
2035 module->mdstrings,
2036 module->mdstrings_alloc * sizeof(module->mdstrings[0]));
2037 }
2038 module->mdstrings[slot] = str;
2039 hashmap_insert(module->mdstrings_map, str, INT2HKEY(slot));
2040
2041 mdref = LL_MDREF_ctor(mdref, slot);
2042 assert(LL_MDREF_value(mdref) == slot, "Metadata string table overflow", 0,
2043 ERR_Fatal);
2044 return mdref;
2045}

2069/* Allocate a mdnode an initialize its content array. */
2070static LL_MDNode *
2071alloc_mdnode(LLVMModuleRef module, enum LL_MDClass mdclass,
2072 const LL_MDRef *elems, unsigned nelems, int is_distinct)
2073{
2074 LL_MDNode *node =
2075 (LL_MDNode *)malloc(sizeof(LL_MDNode) + nelems * sizeof(LL_MDRef));
2076
2077 node->num_elems = nelems;
2078 node->mdclass = mdclass;
2079 node->is_distinct = is_distinct;
2080 node->is_flexible = false;
2081 memcpy(node->elem, elems, nelems * sizeof(LL_MDRef));
2082
2083 /* Check for bitfield overflow. */
2084 assert(node->num_elems == nelems, "MDNode overflow", nelems, ERR_Fatal);

2089#define MIN_FLEX_SIZE 4
2090
2091/* Allocate a flexible mdnode and initialize its contents. */
2092static LL_MDNode *
2093alloc_flexible_mdnode(LLVMModuleRef module, const LL_MDRef *elems,
2094 unsigned nelems)
2095{
2096 LL_MDNode *node;
2097 unsigned size;
2098
2099 /* Flexible nodes always have an allocated capacity that is a power of two
2100 * and at least 4. */
2101 if (nelems <= MIN_FLEX_SIZE) {
2102 size = MIN_FLEX_SIZE;
2103 } else {
2104 /* Size is the smallest power of two >= nelems. */
2105 size = nelems - 1;
2106 size |= size >> 1;
2107 size |= size >> 2;
2108 size |= size >> 4;
2109 size |= size >> 8;
2110 size |= size >> 16;
2111 size += 1;
2112 }
2113
2114 node = (LL_MDNode *)malloc(sizeof(LL_MDNode) + size * sizeof(LL_MDRef));
2115 node->num_elems = nelems;
2116 node->mdclass = LL_PlainMDNode;
2117 node->is_distinct = false;
2118 node->is_flexible = true;
2119 if (nelems)
2120 memcpy(node->elem, elems, nelems * sizeof(LL_MDRef));
2121
2122 assert(node->num_elems == nelems, "MDNode overflow", nelems, ERR_Fatal);
2123
2124 return node;
2125}

2127/**
2128 \brief Append element to flexible mdnode
2129 */
2130static void
2131mdnode_append(LLVMModuleRef module, LL_MDNode **pnode, LL_MDRef elem)
2132{
2133 unsigned nelems = (*pnode)->num_elems;
2134
2135 assert((*pnode)->is_flexible, "Not a flexible metadata node", 0, ERR_Fatal);
2136
2137 /* The allocated capacity will be a power of two >= 4.
2138 * Check if the new element will fit. */
2139 if (nelems >= MIN_FLEX_SIZE && (nelems & (nelems - 1)) == 0) {
2140 /* Node is full. Reallocate to the next power of two. */
2141 *pnode = (LL_MDNode *)realloc(*pnode, sizeof(LL_MDNode) +
2142 2 * nelems * sizeof(LL_MDRef));
2143 }
2144 (*pnode)->elem[(*pnode)->num_elems++] = elem;
2145}
