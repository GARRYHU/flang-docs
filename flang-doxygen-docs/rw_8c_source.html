74
75void
76__fort_rbcstl(int src, void *adr, long cnt, long str, int typ, long ilen)
77{
78 register int in; /* increment or decrement for other cpu */
79 register int me; /* this cpu's adjusted number */
80 register int cp; /* other cpu's real number */
81 struct ents s;
82 struct ent e;
83 int lcpu, tcpus;
84
85 if (cnt <= 0) {
86 return;
87 }
88 s.beg = &e;
89 s.avl = (&e) + 1;
90 s.end = s.avl;
91 lcpu = GET_DIST_LCPU;
92 e.adr = adr;
93 e.str = (str == 0 ? 1 : str);
94 e.cnt = cnt;
95 e.typ = typ;
96 e.ilen = ilen;
97 e.len = cnt * e.ilen;
98
99 in = 1;
100 tcpus = GET_DIST_TCPUS;
101 while (in < tcpus) {
102 in <<= 1;
103 }
104 in >>= 1;
105
106 me = lcpu - src;
107 me = (me < 0 ? me + tcpus : me);
108 if (me != 0) {
109 while (((in - 1) & me) != 0) {
110 in >>= 1;
111 }
112 cp = (me - in) + src;
113 cp = (cp >= tcpus ? cp - tcpus : cp);
114 __fort_erecv(cp, &s);
115 in >>= 1;
116 }
117 while (in > 0) {
118 if ((me + in) < tcpus) {
119 cp = (me + in) + src;
120 cp = (cp >= tcpus ? cp - tcpus : cp);
121 __fort_esend(cp, &s);
122 }
123 in >>= 1;
124 }
125}
