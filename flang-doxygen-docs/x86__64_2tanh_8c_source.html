56FN_PROTOTYPE(mth_i_tanh)(float x)
57{
58 /*
59 The definition of tanh(x) is sinh(x)/cosh(x), which is also equivalent
60 to the following three formulae:
61 1. (exp(x) - exp(-x))/(exp(x) + exp(-x))
62 2. (1 - (2/(exp(2*x) + 1 )))
63 3. (exp(2*x) - 1)/(exp(2*x) + 1)
64 but computationally, some formulae are better on some ranges.
65 */
66 static const float thirtytwo_by_log2 = 4.6166240692e+01F, /* 0x4238aa3b */
67 log2_by_32_lead = 2.1659851074e-02F, /* 0x3cb17000 */
68 log2_by_32_tail = 9.9831822808e-07F, /* 0x3585fdf4 */
69 large_threshold = 10.0F; /* 0x41200000 */
70
71 unsigned int ux, aux;
72 float y, z, p, z1, z2, xneg;
73 int m;
74
75 /* Special cases */
76
77 GET_BITS_SP32(x, ux);
78 aux = ux & ~SIGNBIT_SP32;
79 if (aux < 0x39000000) /* |x| small enough that tanh(x) = x */
80 {
81 if (aux == 0)
82 return x; /* with no inexact */
83 else
84 return valf_with_flags(x, AMD_F_INEXACT);
85 } else if (aux > 0x7f800000) /* |x| is NaN */
86 return x + x;
87
88 xneg = 1.0F - 2.0F * (aux != ux);
89
90 y = xneg * x;
91
92 if (y > large_threshold) {
93 /* If x is large then exp(-x) is negligible and
94 formula 1 reduces to plus or minus 1.0 */
95 z = 1.0F;
96 } else if (y <= 1.0F) {
97 float y2;
98 y2 = y * y;
99
100 if (y < 0.9F) {
101 /* Use a [2,1] Remez approximation on [0,0.9]. */
102 z = y +
103 y * y2 *
104 (-0.28192806108402678e0F +
105 (-0.14628356048797849e-2F + 0.4891631088530669873e-4F * y2) *
106 y2) /
107 (0.845784192581041099e0F + 0.3427017942262751343e0F * y2);
108 } else {
109 /* Use a [2,1] Remez approximation on [0.9,1]. */
110 z = y +
111 y * y2 *
112 (-0.24069858695196524e0F +
113 (-0.12325644183611929e-2F + 0.3827534993599483396e-4F * y2) *
114 y2) /
115 (0.72209738473684982e0F + 0.292529068698052819e0F * y2);
116 }
117 } else {
118 /* Compute p = exp(2*y) + 1. The code is basically inlined
119 from exp_amd. */
120
121 splitexpf(2 * y, 1.0F, thirtytwo_by_log2, log2_by_32_lead, log2_by_32_tail,
122 &m, &z1, &z2);
123 p = scaleFloat_2(z1 + z2, m) + 1.0F;
124 /* Now reconstruct tanh from p. */
125 z = (1.0F - 2.0F / p);
126 }
127
128 return xneg * z;
129}
