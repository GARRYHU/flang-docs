10_ULONGLONG_T
11ftn_i_kishft(_ULONGLONG_T op, int count)
12{
13 /*
14 logical shift:
15 cnt < 0 ==> shift op1 left by |cnt|
16 cnt > 0 ==> shift op1 right by cnt
17 |cnt| >= 64 ==> result is 0
18 */
19
20 if (count >= 0) {
21 if (count >= 64)
22 return 0;
23 return op << count;
24 }
25 if (count <= -64)
26 return 0;
27 return op >> -count;
28}
