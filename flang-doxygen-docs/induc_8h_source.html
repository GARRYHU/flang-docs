30/***** Induction Table Entry *****/
31
32typedef struct {
33 int nm; /* names entry of induction variable */
34 int load; /* ast of the load of the induction var */
35 int init; /* initial value of basic induction var:
36 * will either be the load ast or the value
37 * of a single def which reaches this head
38 * of the loop.
39 */
40 int family; /* "family of" (ind index) */
41 union {
42 uint16_t all;
43 struct {
44 uint16_t omit : 1; /* names not an induction variable */
45 uint16_t ptr : 1; /* induction variable is a pointer */
46 uint16_t delete : 1; /* induction variable can be deleted */
47 uint16_t niu : 1; /* iv has non-induction use */
48 uint16_t gone : 1; /* iv has been deleted */
49 uint16_t midf : 1; /* multiple defs for iv */
50 uint16_t rmst : 1; /* found by removest (def in nested blocks) */
51 uint16_t alias : 1; /* var is an induction alias of "family" */
52 } bits;
53 } flags;
54 INT16 opc; /* ast opcode of skip expr */
55 int skip; /* ast of skip (0 if skips are not the same)*/
56 int mult; /* the multiplier (an ast) which is applied to
57 * iv's "family of" variable.
58 */
59 int initdef; /* def entry if there exists a def which
60 * provides biv's initial value; 0 if it
61 * doesn't exist.
62 */
63 Q_ITEM *bivl; /* list of basic induction definitions -
64 * The info field is the def pointer.
65 * The flag field has the format:
66 * 15 11 7 0
67 * +--------+--------+----------------+
68 * |IDT_type| opn# | '+' or '-' |
69 * +--------+--------+----------------+
70 * NOTE:
71 * when an induction definition is deleted
72 * its corresponding info field is relaced
73 * with the def's store ast.
74 */
75 IUSE *usel; /* induction use list; first item is actually
76 * the list head. Its use field is the number
77 * of pseudo store uses.
78 */
79 Q_ITEM *astl; /* list of ast assigned to this variable */
80 int derived; /* the induction variable (ind index) from
81 * which this induction variable is derived
82 */
83} IND;
