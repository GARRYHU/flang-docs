72/*
73
74 The bit vector representation for a set of size N is a sequence of
75 bits ordered right to left. For i, 1<=i<=N,
76 w = (i-1) / #bits in a BV unit
77 r = (i-1) % #bits in a BV unit
78 then the rth bit in the wth BV unit represents the set membership
79 for i.
80
81 Bit vector support routines:
82 bv_zero - a = 0
83 bv_copy - a = b
84 bv_union - a = a U b
85 bv_sub - a = a - b
86 bv_set - add element to a
87 bv_off - remove element from a
88 bv_notequal - a != b
89 bv_mem - is element a member of a
90*/

134void
135bv_set(BV *a, int elem)
136{
137 int w, r;
138
139 w = (elem - 1) / BV_BITS;
140 r = (elem - 1) - w * BV_BITS;
141 *(a + w) |= 1 << r;
142}

144void
145bv_off(BV *a, int elem)
146{
147 int w, r;
148
149 w = (elem - 1) / BV_BITS;
150 r = (elem - 1) - w * BV_BITS;
151 *(a + w) &= (BV) ~(1 << r);
152}

163LOGICAL
164bv_mem(BV *a, int elem)
165{
166 int w, r;
167
168#if DEBUG
169 assert(a != NULL, "bv_mem: bv null", 0, 3);
170#endif
171 w = (elem - 1) / BV_BITS;
172 r = (elem - 1) - w * BV_BITS;
173 if (*(a + w) & (BV)(1 << r))
174 return (TRUE);
175 return (FALSE);

213LOGICAL
214is_sym_optsafe(int nme, int lpx)
215{
216 int sym;
217 /*
218 * a symbol is not safe (potential "side-effect" or alias conflicts
219 * exist) if:
220 * 1. for C and Fortran, the symbol is volatile
221 *
222 * 2. for Fortran, the symbol is equivalenced
223 *
224 * 3. loop contains a call AND
225 * a. sym's address has been taken, or
226 * b. sym's storage class is not auto (it is static or extern), or
227 * c. for c++, sym is used in two or more nested scope levels
228 *
229 * 4. loop contains a store or load via a pointer AND the sym could
230 * conflict with a pointer (see optutil.c:is_sym_ptrsafe()).
231 *
232 * 6. the symbol is private and the loop is not a parallel loop.
233 *
234 * 7. the symbol is not a private variable and the loop contains
235 * a parallel section.
236 *
237 * 8. the symbol is threadprivate and the current region is 0; in
238 * region 0, its address can't computed before the call to mp_cdecl().
239 *
240 * WARNING - Any new constraints may need to be added to
241 * invar.c:is_sym_invariant_safe() as well.
242 *
243 */
244 sym = NME_SYM(nme);
245
246 if (VOLG(sym))
247 return (FALSE);
248
249 if (!XBIT(19, 0x1) && SOCPTRG(sym)) /* noeqvchk => XBIT(19,0x1) set */
250 return (FALSE);
251
252 if (LP_CALLFG(lpx) && (ADDRTKNG(sym)
253 || !IS_LCL_OR_DUM(sym))) {
254 return (FALSE);
255 }
256
257 if ((LP_PTR_STORE(lpx) || LP_PTR_LOAD(lpx)) && !is_sym_ptrsafe(sym))
258 return (FALSE);
259
260 if (IS_PRIVATE(sym) && !LP_PARLOOP(lpx))
261 return (FALSE);
262
263 if ((LP_CSECT(lpx) || LP_PARREGN(lpx)) && !IS_PRIVATE(sym)) {
264 Q_ITEM *q;
265 for (q = LP_STL_PAR(lpx); q != NULL; q = q->next)
266 if (q->info == nme)
267 return (FALSE);
268 }
269
270 if (LP_PARSECT(lpx) && !IS_PRIVATE(sym)) {
271 return (FALSE);
272 }
273
274 if (lpx == 0 && THREADG(sym)) /* not going to qualify any further */
275 return FALSE;
276
277 return (TRUE);
278}

280LOGICAL
281is_sym_live_safe(int nme, int lpx)
282{
283 int sym;
284/*
285 * Less conservative form of is_sym_optsafe() for flow.c:is_live_out()/
286 * is_live_in(). The is_live routines used to call is_sym_optsafe()
287 * with a loop value of 0 to catch cases such as passing the address
288 * of a variable to a subroutine (i.e., its ADDRTKN flag is set and
289 * LP_CALLFG(0) is set). Using 0 had the effect of returning FALSE
290 * for any variable appearing in a critical section; however, it's
291 * sufficient to use the actual loop index for which the is_live
292 * inquiry. In this function, explicitly use a loop value of 0 (LPZ)
293 * where it's necesary to be convservative; flow.c will call this
294 * function with the actual loop index.
295 */
296#define LPZ 0
297 /*
298 * a symbol is not safe (potential "side-effect" or alias conflicts
299 * exist) if:
300 * 1. for C and Fortran, the symbol is volatile
301 *
302 * 2. for Fortran, the symbol is equivalenced
303 *
304 * 3. loop contains a call AND
305 * a. sym's address has been taken, or
306 * b. sym's storage class is not auto (it is static or extern), or
307 * c. for c++, sym is used in two or more nested scope levels
308 *
309 * 4. loop contains a store or load via a pointer AND the sym could
310 * conflict with a pointer (see optutil.c:is_sym_ptrsafe()).
311 *
312 * 6. the symbol is private and the loop is not a parallel loop.
313 *
314 * 7. the symbol is not a private variable and the loop contains
315 * a parallel section.
316 *
317 * 8. the symbol is threadprivate and the current region is 0; in
318 * region 0, its address can't computed before the call to mp_cdecl().
319 *
320 * WARNING - Any new constraints may need to be added to
321 * invar.c:is_sym_invariant_safe() as well.
322 *
323 */
324 sym = NME_SYM(nme);
325
326 if (VOLG(sym))
327 return (FALSE);
328
329 if (!XBIT(19, 0x1) && SOCPTRG(sym)) /* noeqvchk => XBIT(19,0x1) set */
330 return (FALSE);
331
332 if (LP_CALLFG(LPZ) && (ADDRTKNG(sym)
333 || !IS_LCL_OR_DUM(sym))) {
334 return (FALSE);
335 }
336
337 if ((LP_PTR_STORE(LPZ) || LP_PTR_LOAD(LPZ)) && !is_sym_ptrsafe(sym))
338 return (FALSE);
339
340 if (IS_PRIVATE(sym) && !LP_PARLOOP(lpx))
341 return (FALSE);
342
343 if ((LP_CSECT(lpx) || LP_PARREGN(lpx)) && !IS_PRIVATE(sym)) {
344 Q_ITEM *q;
345 for (q = LP_STL_PAR(lpx); q != NULL; q = q->next)
346 if (q->info == nme)
347 return (FALSE);
348 }
349
350 if (LP_PARSECT(lpx) && !IS_PRIVATE(sym)) {
351 return (FALSE);
352 }
353
354 return (TRUE);
355}

357LOGICAL
358is_call_safe(int nme)
359{
360 int sym;
361 /*
362 * a symbol can be modified by a call if:
363 * 1. for C, the symbol is extern or file static
364 *
365 * 2. for Fortran, the symbol is extern (common block)
366 *
367 * 3. its address is taken
368 *
369 * 4. for C++, sym is accessed from separate scoping units
370 *
371 * 5. for Fortran, the call is to an contained subprogram
372 * Unfortunately, we don't distinguish between internal and
373 * external calls as yet, so if there are contained subprograms,
374 * outer-block symbols are marked as not call-safe.
375 */
376 sym = NME_SYM(nme);
377
378 if (IS_CMNBLK(sym))
379 return FALSE;
380 if ((POINTERG(sym) || ALLOCATTRG(sym)) && MIDNUMG(sym) &&
381 IS_CMNBLK(MIDNUMG(sym))) {
382 return FALSE;
383 }
384 if (gbl.internal && !INTERNALG(sym))
385 return FALSE;
386 if (ADDRTKNG(sym))
387 return FALSE;
388
389 return TRUE;
390}

392/*
393 * determine, given an pointer nme (type NT_IND), if it's safe to ignore
394 * any pointer conflicts. This is based on a combination of the storage
395 * class of the pointer variable involved in the reference and the value
396 * of the -x 2 flag.
397 */
398LOGICAL
399is_ptr_safe(int nme)
400{
401 int sym;
402 int sc;
403
404#if DEBUG
405 assert(NME_TYPE(nme) == NT_IND || NME_TYPE(nme) == NT_VAR ||
406 NME_TYPE(nme) == NT_MEM,
407 "is_ptr_safe: nme not ind", nme, 3);
408#endif
409 if (0 != (sym = basesym_of(nme))) {
410 sc = SCG(sym);
411 if ((XBIT(2, 0x1) && sc == SC_DUMMY) || (XBIT(2, 0x2) && sc == SC_LOCAL) ||
412 (XBIT(2, 0x4) && sc == SC_STATIC) ||
413 (XBIT(2, 0x8) && (sc == SC_CMBLK || sc == SC_EXTERN)))
414 return TRUE;
415 }
416
417 return FALSE;
418}

420/*
421 * Determine if a symbol does not conflict with a pointer reference.
422 * A pointer can conflict with a symbol if:
423 * 1. the symbol's addrtkn flag is set,
424 *
425 * 2. for C, the symbol is extern or file static,
426 *
427 * 3. for fortran, the symbol is in common or is an external,
428 *
429 * 4. for C++, sym is accessed from separate scoping units
430 *
431 * The storage class tests may be inhibited by the -x 2 flag or the
432 * presence of -Mnodepchk.
433 */
434LOGICAL
435is_sym_ptrsafe(int sym)
436{
437 int sc;
438
439 if (ADDRTKNG(sym))
440 return FALSE;
441 sc = SCG(sym);
442 if (flg.depchk && (sc == SC_CMBLK || sc == SC_EXTERN) && !XBIT(2, 0x8))
443 return FALSE;
444
445 return TRUE;
446}

448/*
449 * find the flowgraph node which is the predecessor of the indicated loop.
450 * If there is more than 1, 0 is returned.
451 */
452int
453pred_of_loop(int lpx)
454{
455 PSI_P pred;
456 int fgx;
457
458 fgx = 0;
459 pred = FG_PRED(LP_HEAD(lpx));
460 for (; pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
461 if (FG_LOOP(PSI_NODE(pred)) == lpx)
462 continue;
463 if (fgx) {
464 fgx = 0;
465 break;
466 }
467 fgx = PSI_NODE(pred);
468 }
469
470 return fgx;
471}

473/*
474 * find a single definition for the symbol indicated by its names entry
475 * which reaches the beginning or the end of the flowgraph node, fgx.
476 * If a definition isn't found or if more than one def reaches the node,
477 * 0 is returned.
478 */
479/* TRUE if def is to reach the beginning */
480int
481find_rdef(int nme, int fgx, LOGICAL begin)
482{
483 int def;
484 int rdef;
485 BV *inout;
486
487 /* ensure that the variable is an "optimizable" symbol */
488
489 if (!is_optsym(nme))
490 return 0;
491
492 inout = begin ? FG_IN(fgx) : FG_OUT(fgx);
493 /*
494 * if the node doesn't have a IN/OUT set (node added after flow analysis),
495 * return "no def".
496 */
497 if (inout == NULL)
498 return 0;
499 /*
500 * scan the defs for the induction variable for a single reaching
501 * definition.
502 */
503 rdef = 0;
504 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
505 if (bv_mem(inout, def)) {
506 if (rdef) {
507 rdef = 0;
508 break;
509 }
510 rdef = def;
511 }
512 }
513
514 return (rdef);
515}

517/*
518 * determine if a symbol is live upon exit from a function/
519 * subprogram.
520 */
521LOGICAL
522is_sym_exit_live(int nme)
523{
524 int sym;
525
526#if DEBUG
527 assert(nme > 0 && nme <= nmeb.stg_avail, "is_sym_exit_live, bad nme", nme, 3);
528#endif
529 sym = NME_SYM(nme);
530#if DEBUG
531 assert(sym <= stb.stg_avail, "is_sym_exit_live, bad sym", nme, 3);
532#endif
533 if (IS_STATIC(sym) || IS_EXTERN(sym))
534 return TRUE;
535 if (SAVEG(sym))
536 return TRUE;
537 if (IS_DUM(sym) && INTENTG(sym) != INTENT_IN)
538 return TRUE;
539 /* if this is the result variable */
540 if (RESULTG(sym))
541 return TRUE;
542 /* in a program that contains others, or in a contained program,
543 * outer-block variables are live */
544 if (gbl.internal && !INTERNALG(sym))
545 return TRUE;
546 if (SCG(sym) == SC_LOCAL && DINITG(sym) && (ADDRTKNG(sym) || ASSNG(sym)) &&
547 gbl.rutype != RU_PROG)
548 /*
549 * if a local variable is data initialized, disallow it if it
550 * has been stored; the thinking is that it falls into the
551 * same category of a saved variable -- someday, may want
552 * to override this if XBIT(124,0x80) is set (also expreg.c)
553 */
554 return TRUE;
555 if (SCG(sym) == SC_BASED) {
556 int s;
557 for (s = MIDNUMG(sym); s; s = MIDNUMG(s))
558 if (IS_DUM(s) || IS_STATIC(s) || IS_EXTERN(s) || DINITG(s) || SAVEG(s))
559 return TRUE;
560 }
561 if (THREADG(sym)) /* not going to qualify any further */
562 return TRUE;
563
564 return FALSE;
565}

568 * determine if a symbol is live because of implicit uses -- live upon
569 * exit from a function, has its address taken, is volatile, etc.
570 */
571LOGICAL
572is_sym_imp_live(int nme)
573{
574 int sym;
575
576#if DEBUG
577 assert(nme > 0 && nme <= nmeb.stg_avail, "is_sym_imp_live, bad nme", nme, 3);
578#endif
579 if (is_sym_exit_live(nme))
580 return TRUE;
581 sym = NME_SYM(nme);
582 if (ADDRTKNG(sym) || VOLG(sym))
583 return TRUE;
584 if (ARGG(sym))
585 return TRUE;
586#ifdef PTRSTOREP

588 return TRUE;
589#endif
590 if (!XBIT(19, 0x1) && SOCPTRG(sym)) /* noeqvchk => XBIT(19,0x1) set */
591 return TRUE;
592 if (SCG(sym) == SC_BASED && MIDNUMG(sym))
593 return TRUE;
594 if (PTRVG(sym))
595 return TRUE;
596 if (STYPEG(sym) == ST_VAR && HCCSYMG(sym) && !VCSYMG(sym))
597 /* ...implicit uses of compiler-created bounds variables. */
598 return TRUE;
599 return FALSE;
600}

693/*
694 * determine if it's safe to copy the value of the def to either the
695 * beginning of the flowgraph node or to the end of the flowgraph
696 * node. We've already determined that the def reaches the flowgraph node.
697 */
698LOGICAL
699can_copy_def(int def, int fgx, LOGICAL begin)
700{
701 int end_ilt;
702
703#if DEBUG
704 assert(def, "can_copy_def: def is 0", 0, 3);
705#endif
706 if (OPTDBG(9, 16384))
707 fprintf(gbl.dbgfil, "can_copy_def trace for %s, def %d, expr %d, to fg %d",
708 getprint(basesym_of(DEF_NM(def))), def, DEF_RHS(def), fgx);
709 if (begin)
710 end_ilt = BIH_ILTFIRST(FG_TO_BIH(fgx));
711 else
712 end_ilt = BIH_ILTLAST(FG_TO_BIH(fgx));
713 if (OPTDBG(9, 16384))
714 fprintf(gbl.dbgfil, ", ilt %d\n", end_ilt);
715
716 if (!def_ok(def, fgx, end_ilt))
717 return FALSE;
718
719 if (DEF_CONST(def)) {
720 if (OPTDBG(9, 16384))
721 fprintf(gbl.dbgfil, "can copy const def %d\n", def);
722 return TRUE;
723 }
724
725 if (XBIT(7, 1)) {
726 if (OPTDBG(9, 16384))
727 fprintf(gbl.dbgfil, "can't copy def %d, inhibited\n", def);
728 return FALSE;
729 }
730
731 if (DEF_DOINIT(def))
732 srch_ae.start.stmt = FG_STDLAST(DEF_FG(def));
733 else
734 srch_ae.start.stmt = DEF_STD(def);
735 srch_ae.start.fg = DEF_FG(def);
736 srch_ae.end.stmt = end_ilt;
737 srch_ae.end.fg = fgx;
738 srch_ae.eob = TRUE;
739 return (avail_expr((int)DEF_RHS(def)));
740}

742/*
743 * determine if it's safe to copy the value of the def to a statement
744 * by checking attributes about the def.
745 * We've already determined that the def reaches the flowgraph node.
746 */
747LOGICAL
748def_ok(int def, int fgx, int stmt)
749{
750 int nme;
751 int sym;
752 int def_fg, def_std;
753 int iltx;
754
755#if DEBUG
756 assert(def, "def_ok: def is 0", 0, 3);
757#endif
758 if (OPTDBG(9, 16384))
759 fprintf(gbl.dbgfil, "def_ok trace for def %d of %s\n", def,
760 getprint(basesym_of(DEF_NM(def))));
761
762 /* does the storing expr contain a use of the symbol being defined */
763
764 if (DEF_SELF(def)) {
765 if (OPTDBG(9, 16384))
766 fprintf(gbl.dbgfil, "def %d not ok, self.\n", def);
767 return FALSE;
768 }
769
770 def_fg = DEF_FG(def);
771
772 if (def_fg != fgx && BIH_CS(FG_TO_BIH(fgx))) {
773 if (OPTDBG(9, 16384))
774 fprintf(gbl.dbgfil, "def %d not ok, use in critical sec\n", def);
775 return FALSE;
776 }
777
778 nme = DEF_NM(def);
779 sym = NME_SYM(nme);
780 if (is_sym_entry_live(nme) && !is_dominator(def_fg, fgx)) {
781 if (OPTDBG(9, 16384))
782 fprintf(gbl.dbgfil, "def %d not ok, not dom.\n", def);
783 return FALSE;
784 }
785 def_std = DEF_STD(def);
786 if (STD_EX(def_std)) {
787 if (OPTDBG(9, 16384))
788 fprintf(gbl.dbgfil, "def %d not, ilt_ex\n", def);
789 return FALSE;
790 }
791
792 if (VOLG(sym)) {
793 if (OPTDBG(9, 16384))
794 fprintf(gbl.dbgfil, "def %d not ok, sym %d VOL\n", def, sym);
795 return FALSE;
796 }
797 if (!XBIT(19, 0x1) && SOCPTRG(sym)) { /* noeqvchk => XBIT(19,0x1) set */
798 if (OPTDBG(9, 16384))
799 fprintf(gbl.dbgfil, "def %d not ok, sym %d EQUIV\n", def, sym);
800 return FALSE;
801 }
802
803 /* If sym is a common block variable (and potentially aliased), do not propagate
804 * its definition, like how variables in an equivalent statement are handled.
805 */
806 if (SCG(sym) == SC_CMBLK && MODCMNG(CMBLKG(sym)) == 0) {
807 if (OPTDBG(9, 16384))
808 fprintf(gbl.dbgfil, "def %d not ok, sym %d is common block "
809 "variable\n", def, sym);
810 return FALSE;
811 }
812
813 if (FG_IN(fgx) == NULL) {
814 if (OPTDBG(9, 16384))
815 fprintf(gbl.dbgfil, "can't copy def %d, no in of fg %d\n", def, fgx);
816 return FALSE;
817 }
818
819 /* does stmt precede the def?
820 *
821 * if def and stmt are in the same block, it's necessary to
822 * scan backwards from the def.
823 *
824 * if they aren't in the same block, the dominator test is
825 * sufficient.
826 *
827 */
828 if (def_fg == fgx) {
829 if (DEF_DOINIT(def))
830 /* defs which initialize do variables never precedes a stmt */
831 ;
832 else
833 for (iltx = (def_std); iltx; iltx = STD_PREV(iltx)) {
834 if (iltx == stmt) {
835 if (OPTDBG(9, 16384))
836 fprintf(gbl.dbgfil, "def %d not ok, ilt %d precedes def\n", def,
837 stmt);
838 return FALSE;
839 }
840 }
841 } else if (!is_dominator(def_fg, fgx)) {
842 if (OPTDBG(9, 16384))
843 fprintf(gbl.dbgfil, "def %d not ok, def does not dominator ilt %d\n", def,
844 stmt);
845 return FALSE;
846 }
847
848 /*
849 * If variable is "call" unsafe, then:
850 * 1. a call cannot exist between the def and the point.
851 * 2. If the variable has its address taken, then a store via a ptr
852 * cannot exist between the def and the point.
853 */
854 if (!is_call_safe(nme)) {
855 if (is_call_in_path(def_std, def_fg, stmt, fgx)) {
856 if (OPTDBG(9, 16384))
857 fprintf(gbl.dbgfil, "def %d not ok, call in path (%d, %d), (%d, %d)\n",
858 def, def_fg, def_std, fgx, stmt);
859 return FALSE;
860 }
861 if (!is_sym_ptrsafe(sym)) {
862 if (is_ptr_in_path(def_std, def_fg, stmt, fgx)) {
863 if (OPTDBG(9, 16384))
864 fprintf(gbl.dbgfil, "def %d not ok, ptr in path (%d, %d), (%d, %d)\n",
865 def, def_fg, def_std, fgx, stmt);
866 return FALSE;
867 }
868 }
869 }
870
871 return TRUE;
872}

874/*
875 * determine if an expression is available along a path which is defined as
876 * two points in a flowgraph node, (start_ilt, start_fg) and (end_ilt, end_ilt)
877 * NOTES:
878 * 1. starting point is always the <fg, ilt> of a def
879 * 2. the end of the path "follows" the def (i.e., def_ok has been called
880 * to ensure that the end of the path does not precede the def).
881 */
882LOGICAL
883is_avail_expr(int expr, int start_ilt, int start_fg, int end_ilt, int end_fg)
884{
885
886 if (OPTDBG(9, 16384))
887 fprintf(gbl.dbgfil, "is_avail_expr(expr:%d, s:%d, sfg:%d, e:%d, efg:%d)\n",
888 expr, start_ilt, start_fg, end_ilt, end_fg);
889
890 if (A_TYPEG(expr) == A_CNST) {
891 if (OPTDBG(9, 16384))
892 fprintf(gbl.dbgfil, "const expr is avail.\n");
893 return TRUE;
894 }
895 srch_ae.start.stmt = start_ilt;
896 srch_ae.start.fg = start_fg;
897 srch_ae.end.stmt = end_ilt;
898 srch_ae.end.fg = end_fg;
899 srch_ae.eob = FALSE;
900
901 return (avail_expr(expr));
902}

907 * determine if an expression reaches the beginning or end of a block
908 * starting at the end of a block containing a given point.
909 * Note that this recurses thru the original expression passed to
910 * is_avail_expr.
911
912 * For each "variable" found in the expression, it's determined if its
913 * value is available at the end of the path.
914 */

917{

919 ast_visit(1, 1);
920 av = TRUE;
921 ast_traverse(expr, _avail, NULL, &av);
922 ast_unvisit();
923 if (OPTDBG(9, 16384) && av)
924 fprintf(gbl.dbgfil, "expr avail.\n");
925 return av;

928static LOGICAL
929_avail(int expr, LOGICAL *av_p)
930{
931 int opc;
932 int nme, sym;
933 int i;
934 int def;
935 int iltx;
936
937 if (!*av_p)
938 return TRUE; /* don't continue if already not available */
939 if (OPTDBG(9, 16384))
940 fprintf(gbl.dbgfil, "_avail(%d)\n", expr);
941 opc = A_TYPEG(expr);
942 switch (opc) {
943 case A_CNST:
944 if (OPTDBG(9, 16384))
945 fprintf(gbl.dbgfil, "const expr is avail.\n");
946 return TRUE; /* stop */
947 case A_ID:
948 sym = A_SPTRG(expr);
949 if (!ST_ISVAR(STYPEG(sym)))
950 return TRUE; /* ignore ST_PROC, ST_INTRIN, etc. */
951 nme = A_NMEG(expr);
952 if (OPTDBG(9, 16384))
953 fprintf(gbl.dbgfil, "avail_expr. nme %d, load %d\n", nme, expr);
954 if (!is_optsym(nme)) {
955 switch (NME_TYPE(nme)) {
956 case NT_VAR:
957 if ((SCG(NME_SYM(nme)) == SC_BASED || POINTERG(NME_SYM(nme))) &&
958 isptr_in_path()) {
959 *av_p = FALSE;
960 return TRUE;
961 }
962 FLANG_FALLTHROUGH;
963 case NT_ARR:
964 case NT_MEM:
965 case NT_UNK:
966 if (OPTDBG(9, 16384))
967 fprintf(gbl.dbgfil, "expr not avail %d, nme.\n", expr);
968 *av_p = FALSE;
969 return TRUE;
970 case NT_IND:
971 if (isptr_in_path()) {
972 *av_p = FALSE;
973 return TRUE;
974 }
975 break;
976 default:
977 interr("avail_expr: unk nme", expr, 3);
978 break;
979 }
980 break; /* recurse */
981 }
982 /*
983 * A load of an optimizable symbol has been found.
984 * Step 1. If variable is "call" unsafe, then:
985 * 1. a call cannot be in the path defined from start to end
986 * (precise analysis).
987 * 2. If the variable has its address taken, then a store via a ptr
988 * cannot be in the path defined from start to end
989 */
990 sym = NME_SYM(nme);
991#ifdef RESHAPEDG
992 if (SCG(sym) == SC_BASED && RESHAPEDG(sym)) {
993 *av_p = FALSE;
994 return TRUE;
995 }
996#endif
997 if (!is_call_safe(nme)) {
998 if (iscall_in_path()) {
999 *av_p = FALSE;
1000 return TRUE;
1001 }
1002 if (ADDRTKNG(sym)) {
1003 if (isptr_in_path()) {
1004 *av_p = FALSE;
1005 return TRUE;
1006 }
1007 }
1008 }
1009 /*
1010 * Step2. determine if there are any defs of the variable after the
1011 * point in the flowgraph node to the end of the same node.
1012 * Note that if the path is in one block, we'll terminate the
1013 * search at the end of the path; otherwise, we check until the
1014 * end of the block.
1015 */
1016 if (srch_ae.eob)
1017 iltx = 0; /* ==> end of block */
1018 else if (srch_ae.start.fg == srch_ae.end.fg)
1019 iltx = srch_ae.end.stmt;
1020 else
1021 iltx = 0; /* ==> end of block */
1022 /*
1023 * scan the defs in the node to find the stmt containing the expression.
1024 * NOTE: works only because we've defined the point in the node to
1025 * be a def of an optimizable symbol.
1026 */
1027 def = FG_FDEF(srch_ae.start.fg);
1028 for (; def; def = DEF_LNEXT(def))
1029 if (DEF_STD(def) == srch_ae.start.stmt) {
1030 if (DEF_NM(def) == nme && DEF_SELF(def)) {
1031 /* At the starting point, there is a def which defines
1032 * the variable and includes itself; normally, this is
1033 * caught by def_ok(), but is_avail_expr() may be called
1034 * without first calling def_ok().
1035 */
1036 if (OPTDBG(9, 16384))
1037 fprintf(gbl.dbgfil, "expr not avail. DEF_SELF(%d) for nme %d\n",
1038 def, nme);
1039 *av_p = FALSE;
1040 return TRUE;
1041 }
1042 break;
1043 }
1044
1045 /* scan all the defs after the point */
1046 rdilts(FG_TO_BIH(srch_ae.start.fg));
1047 for (def = DEF_LNEXT(def); def; def = DEF_LNEXT(def))
1048 if (DEF_NM(def) == nme) {
1049 for (i = STD_NEXT(srch_ae.start.stmt);; i = STD_NEXT(i)) {
1050 if (i == iltx)
1051 break;
1052 if (DEF_STD(def) == i) {
1053 if (OPTDBG(9, 16384))
1054 fprintf(gbl.dbgfil, "expr not avail. def %d after\n", def);
1055 wrilts(FG_TO_BIH(srch_ae.start.fg));
1056 *av_p = FALSE;
1057 return TRUE;
1058 }
1059 }
1060 }
1061 wrilts(FG_TO_BIH(srch_ae.start.fg));
1062
1063 /*
1064 * Step 3. scan all defs of the variable. If one reaches the end
1065 * of the path, then the expression is unavailble.
1066 */
1067 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
1068 if (DEF_FG(def) == srch_ae.start.fg)
1069 continue; /* def is in same block -- it must precede */
1070 /*
1071 * if def is in the flow graph node containing end point, we need
1072 * to ensure that the def does not precede end.stmt. WARNING:
1073 * steps should have already been taken to ensure that start
1074 * dominates end.
1075 */
1076 if (DEF_FG(def) == srch_ae.end.fg) {
1077 if (DEF_DOINIT(def) && srch_ae.eob) {
1078 if (OPTDBG(9, 16384))
1079 fprintf(gbl.dbgfil, "expr not avail. dodef %d bef\n", def);
1080 *av_p = FALSE;
1081 return TRUE;
1082 }
1083 if (srch_ae.eob)
1084 iltx = 0; /* ==> end of block */
1085 else
1086 iltx = srch_ae.end.stmt;
1087 rdilts(FG_TO_BIH(srch_ae.end.fg));
1088 for (iltx = STD_PREV(iltx); iltx; iltx = STD_PREV(iltx))
1089 if (DEF_STD(def) == iltx) {
1090 if (OPTDBG(9, 16384))
1091 fprintf(gbl.dbgfil, "expr not avail. def %d bef\n", def);
1092 wrilts(FG_TO_BIH(srch_ae.end.fg));
1093 *av_p = FALSE;
1094 return TRUE;
1095 }
1096 wrilts(FG_TO_BIH(srch_ae.end.fg));
1097 continue;
1098 }
1099 if (is_dominator((int)DEF_FG(def), srch_ae.start.fg))
1100 continue; /* if def dominates point, it's ok */
1101 if (bv_mem(FG_IN(srch_ae.end.fg), def)) {
1102 if (srch_ae.start.fg == srch_ae.end.fg &&
1103 !bv_mem(FG_OUT(srch_ae.end.fg), def)) {
1104 /*
1105 * start and end are in the same block. A def is in its
1106 * IN set but is killed. check where the def is killed.
1107 * if it's at the start or before, then all is ok.
1108 */
1109 int d;
1110 d = FG_FDEF(srch_ae.end.fg);
1111 while (d) {
1112 if (DEF_NM(d) == nme) {
1113 for (iltx = srch_ae.start.stmt; iltx; iltx = STD_PREV(iltx)) {
1114 if (DEF_STD(d) == iltx)
1115 goto next_def;
1116 }
1117 }
1118 d = DEF_LNEXT(d);
1119 }
1120 if (OPTDBG(9, 16384))
1121 fprintf(gbl.dbgfil, "expr not avail. def %d IN.1\n", def);
1122 *av_p = FALSE;
1123 return TRUE;
1124 }
1125 /*
1126 * although def is in IN(end), is it in the path (start, end).
1127 */
1128 if (is_dominator(srch_ae.start.fg, srch_ae.end.fg) &&
1129 !is_in_path((int)DEF_FG(def)))
1130 ;
1131 else {
1132 if (OPTDBG(9, 16384))
1133 fprintf(gbl.dbgfil, "expr not avail. def %d IN.2\n", def);
1134 *av_p = FALSE;
1135 return TRUE;
1136 }
1137 }
1138 next_def:;
1139 }
1140
1141 if (OPTDBG(9, 16384))
1142 fprintf(gbl.dbgfil, "expr avail.\n");
1143 return (TRUE);
1144
1145 case A_CALL:
1146 case A_FUNC:
1147 if (OPTDBG(9, 16384))
1148 fprintf(gbl.dbgfil, "expr %d not avail. contains call\n", expr);
1149 *av_p = FALSE;
1150 return TRUE;
1151
1152 default:
1153 break;
1154 }
1155
1156 return FALSE; /* recurse on expr's operands */
1157}

1159/*
1160 * is node fg in the path (srch_ae.start.fg, srch_ae.end.fg) where
1161 * start dominates end.
1162 */
1163static LOGICAL
1164is_in_path(int fg)
1165{
1166 LOGICAL ret;
1167
1168 if (srch_ae.start.fg == fg)
1169 return TRUE;
1170
1171 /* use the "natnxt" field to link together nodes visited during in_path */
1172 visit_list = srch_ae.start.fg;
1173 FG_NATNXT(visit_list) = 0;
1174 FG_VISITED(srch_ae.start.fg) = 1;
1175
1176 ret = in_path(srch_ae.end.fg, fg);
1177
1178 /* unvisit the nodes visited during in_path */
1179 while (visit_list) {
1180 FG_VISITED(visit_list) = 0;
1181 visit_list = FG_NATNXT(visit_list);
1182 }
1183
1184 return ret;
1185}

1187static LOGICAL
1188in_path(int cur, int fg)
1189{
1190 PSI_P pred;
1191 int pred_fg;
1192
1193 if (FG_VISITED(cur))
1194 return FALSE;
1195 if (cur == fg)
1196 return TRUE;
1197
1198 FG_VISITED(cur) = 1;
1199 FG_NATNXT(cur) = visit_list;
1200 visit_list = cur;
1201
1202 for (pred = FG_PRED(cur); pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
1203 pred_fg = PSI_NODE(pred);
1204 if (in_path(pred_fg, fg))
1205 return TRUE;
1206 }
1207 return FALSE;
1208}

1210/*
1211 * is a call in the path <srch_ae.start.fg, srch_ae.start.stmt>,
1212 * <srch_ae.end.fg, srch_ae.end.stmt>.
1213 */
1214static LOGICAL
1215iscall_in_path(void)
1216{
1217 int iltx;
1218 int save_ex;
1219 LOGICAL ret;
1220 int term_std;
1221 /*
1222 * examine the ilts from the start stmt to the end of the start node.
1223 */
1224 if (srch_ae.start.fg == srch_ae.end.fg) {
1225 /* nodes are the same; check ilts after the start and before the
1226 * end stmt.
1227 */
1228 if (srch_ae.start.stmt == srch_ae.end.stmt)
1229 return FALSE;
1230 iltx = STD_NEXT(srch_ae.start.stmt);
1231 while (iltx) { /* WARNING: end.stmt could have been deleted */
1232 if (iltx == srch_ae.end.stmt)
1233 break;
1234 if (STD_EX(iltx)) {
1235 if (OPTDBG(9, 16384))
1236 fprintf(gbl.dbgfil, "call is in path, after def in fg %d, ilt %d\n",
1237 srch_ae.start.fg, iltx);
1238 return TRUE;
1239 }
1240 iltx = STD_NEXT(iltx);
1241 }
1242 return FALSE;
1243 }
1244 /*
1245 * nodes are not the same; check ilts after the start and thru the
1246 * end of the block.
1247 */
1248 iltx = srch_ae.start.stmt;
1249 term_std = FG_STDLAST(srch_ae.start.fg);
1250 while (term_std != iltx) {
1251 iltx = STD_NEXT(iltx);
1252 if (iltx == 0)
1253 break;
1254 if (STD_EX(iltx)) {
1255 if (OPTDBG(9, 16384))
1256 fprintf(gbl.dbgfil, "call is in path, after def in fg %d, ilt %d\n",
1257 srch_ae.start.fg, iltx);
1258 return TRUE;
1259 }
1260 }
1261
1262 /* nodes are different, examine ilts between start of the end node and
1263 * the end statement.
1264 */
1265 term_std = FG_STDFIRST(srch_ae.end.fg);
1266 for (iltx = (srch_ae.end.stmt); iltx; iltx = STD_PREV(iltx)) {
1267 if (STD_EX(iltx)) {
1268 if (OPTDBG(9, 16384))
1269 fprintf(gbl.dbgfil, "call is in path, before end of fg %d, ilt %d\n",
1270 srch_ae.start.fg, iltx);
1271 return TRUE;
1272 }
1273 if (iltx == term_std)
1274 break;
1275 }
1276
1277 /* if start does not dominate end, check if a call (the call "def")
1278 * reaches then end node.
1279 */
1280 if (!is_dominator(srch_ae.start.fg, srch_ae.end.fg)) {
1281 if (bv_mem(FG_IN(srch_ae.end.fg), CALL_DEF)) {
1282 if (OPTDBG(9, 16384))
1283 fprintf(gbl.dbgfil, "call is in path, calldef IN fg %d\n",
1284 srch_ae.end.fg);
1285 return TRUE;
1286 }
1287 return FALSE;
1288 }
1289
1290 /*
1291 * nodes are not the same & start dominates the end.
1292 * if the end node is in a loop and start is NOT in the same loop
1293 * then it's necessary to check the block's * external flag.
1294 * Even if the call exists after the end stmt, it can
1295 * still reach the statement.
1296 */
1297 if (FG_LOOP(srch_ae.end.fg) &&
1298 FG_LOOP(srch_ae.end.fg) != FG_LOOP(srch_ae.start.fg) &&
1299 BIH_EX(FG_TO_BIH(srch_ae.end.fg))) {
1300 if (OPTDBG(9, 16384))
1301 fprintf(gbl.dbgfil, "call is in path, fg %d in loop %d contains call\n",
1302 srch_ae.end.fg, FG_LOOP(srch_ae.end.fg));
1303 return TRUE;
1304 }
1305
1306 /* traverse the path start to end bottom-up, checking if a call exists
1307 * in each node. To setup, start's visit flag is set (=> if a call exists
1308 * at the beginning of the block, it will be ignored), and the "EX" flag
1309 * of the end's bih must be cleared then restored.
1310 */
1311 /* use the "natnxt" field to link together nodes visited during in_path */
1312 visit_list = srch_ae.start.fg;
1313 FG_NATNXT(visit_list) = 0;
1314 FG_VISITED(srch_ae.start.fg) = 1;
1315 save_ex = BIH_EX(FG_TO_BIH(srch_ae.end.fg));
1316 BIH_EX(FG_TO_BIH(srch_ae.end.fg)) = 0;
1317
1318 ret = call_in_path(srch_ae.end.fg);
1319
1320 /* unvisit the nodes visited during in_path */
1321 while (visit_list) {
1322 FG_VISITED(visit_list) = 0;
1323 visit_list = FG_NATNXT(visit_list);
1324 }
1325
1326 BIH_EX(FG_TO_BIH(srch_ae.end.fg)) = save_ex;
1327
1328 return ret;
1329}

1331static LOGICAL
1332call_in_path(int cur)
1333{
1334 PSI_P pred;
1335 int pred_fg;
1336
1337 if (FG_VISITED(cur))
1338 return FALSE;
1339 if (BIH_EX(FG_TO_BIH(cur)))
1340 return TRUE;
1341
1342 FG_VISITED(cur) = 1;
1343 FG_NATNXT(cur) = visit_list;
1344 visit_list = cur;
1345
1346 for (pred = FG_PRED(cur); pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
1347 pred_fg = PSI_NODE(pred);
1348 if (call_in_path(pred_fg)) {
1349 if (OPTDBG(9, 16384))
1350 fprintf(gbl.dbgfil, "call is in path, call in fg %d\n", pred_fg);
1351 return TRUE;
1352 }
1353 }
1354 return FALSE;
1355}

1373/*
1374 * is a ptr store in the path <srch_ae.start.fg, srch_ae.start.stmt>,
1375 * <srch_ae.end.fg, srch_ae.end.stmt>.
1376 */
1377static LOGICAL
1378isptr_in_path(void)
1379{
1380 int iltx;
1381 int term_std;
1382 /*
1383 * examine the ilts from the start stmt to the end of the start node.
1384 * only search if there exists a store via a pointer in the node.
1385 */
1386 if (srch_ae.start.fg == srch_ae.end.fg) {
1387 if (srch_ae.start.stmt == srch_ae.end.stmt)
1388 return FALSE;
1389 if (FG_PTR_STORE(srch_ae.start.fg)) {
1390 /* nodes are the same; check ilts after the start and before the
1391 * end stmt.
1392 */
1393 iltx = STD_NEXT(srch_ae.start.stmt);
1394 while (iltx) { /* WARNING: end.stmt could have been deleted */
1395 if (iltx == srch_ae.end.stmt)
1396 break;
1397 if (is_store_via_ptr((int)STD_AST(iltx))) {
1398 if (OPTDBG(9, 16384))

1613/*
1614 * a def's uses have been removed from a loop. check if there are any
1615 * other uses of the def; if none and it meets other criteria, delete
1616 * the definition.
1617 */
1618void
1619rm_def_rloop(int def, int lpx)
1620{
1621 int def_fg, def_std;
1622 int use;
1623 int count; /* # of uses of def in same block */
1624 int i;
1625 DU *du;
1626
1627#if DEBUG
1628 assert(def && def < opt.defb.stg_avail, "rm_def_rloop: bad def", def, 3);
1629 assert(lpx, "rm_def_rloop: lpx", 0, 3);
1630#endif
1631 /*
1632 * first, if the def is of a symbol which is live upon exit (i.e., due to
1633 * storage class), it can't be deleted if it reaches the exit from the
1634 * function.
1635 */
1636 if (is_def_imp_live(def)) {
1637 if (OPTDBG(9, 8192))
1638 fprintf(gbl.dbgfil, "rm_def_rloop - def %d is live at exit\n", def);
1639 return;
1640 }
1641
1642 def_fg = DEF_FG(def);
1643 def_std = DEF_STD(def);
1644 count = 0;
1645 for (du = DEF_DU(def); du != NULL; du = du->next) {
1646 int use_fg;
1647 use_fg = USE_FG(use = du->use);
1648 if (FG_LOOP(use_fg) == lpx)
1649 continue;
1650 if (use_fg != def_fg) {
1651 if (OPTDBG(9, 8192))
1652 fprintf(gbl.dbgfil, "rm_def_rloop - def %d has other uses\n", def);
1653 return;
1654 }
1655 /* use in same block as def: scan ilts before use, searching for def */
1656 for (i = (USE_STD(use)); i; i = STD_PREV(i)) {
1657 if (i == def_std) {
1658 count++;
1659 goto next_use;
1660 }
1661 }
1662 if (OPTDBG(9, 8192))
1663 fprintf(gbl.dbgfil, "rm_def_rloop - def %d has uses bef\n", def);
1664 return;
1665 next_use:;
1666 }
1667
1668 if (count) {
1669 STD_DELETE(def_std) = 1;
1670 DEF_DELETE(def) = 1;
1671 if (OPTDBG(9, 8192))
1672 fprintf(gbl.dbgfil, "rm_def_rloop - def %d, ilt %d marked deleted\n", def,
1673 def_std);
1674 return;
1675 }
1676
1677 DEF_DELETE(def) = 1;
1678 unlnkilt(def_std, (int)FG_TO_BIH(def_fg), FALSE);
1679 if (OPTDBG(9, 8192))
1680 fprintf(gbl.dbgfil, "rm_def_rloop - def %d, ilt %d deleted\n", def,
1681 def_std);
1682}

1697/*
1698 * propagate the definitions of any variables in 'expr' which is
1699 * considered to be at the point immediately preceding the loop.
1700 */
1701int
1702copy_to_loop(int tree, int lp)
1703{
1704 int i;
1705 LOGICAL changes;
1706
1707 copyb.lp = lp;
1708 copyb.new = tree;
1709 if (OPTDBG(9, 65536)) {
1710 fprintf(gbl.dbgfil, "copy_to_loop %d, ast %d:\n", lp, tree);
1711 dbg_print_ast(tree, gbl.dbgfil);
1712 }
1713 OPT_ALLOC(copyb, int, 32);
1714 while (TRUE) {
1715 /*
1716 * Recursively search 'tree' and collect all loads which are
1717 * candidates for copy propagation.
1718 */
1719 copyb.stg_avail = 0;
1720 ast_visit(1, 1);
1721 ast_traverse(copyb.new, collect_loads, NULL, NULL);
1722 ast_unvisit();
1723 changes = FALSE;
1724 /*
1725 * For all candidate loads, attempt to copy their values. cp_loop()
1726 * updates copyb.new if replacement occurs.
1727 */
1728 for (i = 0; i < copyb.stg_avail; i++) {
1729 if (cp_loop(copyb.stg_base[i])) {
1730 if (OPTDBG(9, 65536)) {
1731 fprintf(gbl.dbgfil, "recur copy_to_loop %d, ast: %d\n", lp,
1732 copyb.new);
1733 dbg_print_ast(copyb.new, gbl.dbgfil);
1734 }
1735 changes = TRUE;
1736 }
1737 }
1738 if (!changes)
1739 break;
1740 }
1741
1742 OPT_FREE(copyb);
1743 return copyb.new;
1744}

1775static LOGICAL
1776cp_loop(int expr)
1777{
1778 int nme;
1779 int i;
1780 int rdef;
1781 PSI_P pred;
1782
1783 nme = A_NMEG(expr);
1784 if (OPTDBG(9, 65536))
1785 fprintf(gbl.dbgfil, " cp_loop. nme %d, load %d\n", nme, expr);
1786 /*
1787 * A load of an optimizable symbol has been found.
1788 * for the predecessors of the loop, determine if there is one
1789 * and only one def of the variable which reaches the end of
1790 * the predecessors. This def must be live for all paths to the
1791 * predecessors.
1792 */
1793 rdef = 0;
1794 pred = FG_PRED(LP_HEAD(copyb.lp));
1795 for (; pred != PSI_P_NULL; pred = PSI_NEXT(pred)) {
1796 if (FG_LOOP(PSI_NODE(pred)) == copyb.lp)
1797 continue;
1798 i = find_rdef(nme, PSI_NODE(pred), FALSE /* end of block */);
1799 if (i == 0) {
1800 rdef = 0;
1801 if (OPTDBG(9, 65536))
1802 fprintf(gbl.dbgfil, " cp_loop: rdef nfnd to end of fg %d\n",
1803 PSI_NODE(pred));
1804 break;
1805 }
1806 if (rdef) {
1807 if (i != rdef) {
1808 if (OPTDBG(9, 65536))
1809 fprintf(gbl.dbgfil, " cp_loop: mult rdefs fnd, %d & %d\n", rdef,
1810 i);
1811 rdef = 0;
1812 break;
1813 }
1814 } else
1815 rdef = i;
1816
1817 if (!can_copy_def(rdef, PSI_NODE(pred), FALSE /* end of block */)) {
1818 if (OPTDBG(9, 65536))
1819 fprintf(gbl.dbgfil,
1820 " cp_loop: def %d cannot be copied to end of %d\n", rdef,
1821 PSI_NODE(pred));
1822 rdef = 0;
1823 break;
1824 }
1825 }
1826 if (rdef) {
1827 if (OPTDBG(9, 65536))
1828 fprintf(gbl.dbgfil, " cp_loop: copy def %d\n", rdef);
1829 i = DEF_RHS(rdef);
1830 if (A_TYPEG(i) == A_CONV) {
1831 switch (DTY(A_DTYPEG(i))) {
1832 case TY_INT:
1833 case TY_SINT:
1834 case TY_BINT:
1835 if (OPTDBG(9, 65536))
1836 fprintf(gbl.dbgfil, " cp_loop: def %d - I_INT(rhs %d)\n", rdef, i);
1837 i = ast_intr(I_INT, DT_INT, 1, A_LOPG(i));
1838 break;
1839 case TY_REAL:
1840 if (OPTDBG(9, 65536))
1841 fprintf(gbl.dbgfil, " cp_loop: def %d - I_REAL(rhs %d)\n", rdef,
1842 i);
1843 i = ast_intr(I_REAL, DT_REAL4, 1, A_LOPG(i));
1844 break;
1845 case TY_DBLE:
1846 if (OPTDBG(9, 65536))
1847 fprintf(gbl.dbgfil, " cp_loop: def %d - I_DBLE(rhs %d)\n", rdef,
1848 i);
1849 i = ast_intr(I_DBLE, DT_DBLE, 1, A_LOPG(i));
1850 break;
1851 default:
1852 if (OPTDBG(9, 65536))
1853 fprintf(gbl.dbgfil,
1854 " cp_loop: def %d cannot be copied - ugly A_CONV %d\n",
1855 rdef, i);
1856 return FALSE;
1857 }
1858 }
1859 if (OPTDBG(9, 65536))
1860 fprintf(gbl.dbgfil, " cp_loop: replace ast %d with %d\n", expr, i);
1861 ast_visit(1, 1);
1862 ast_replace(expr, i);
1863 ast_rewrite(copyb.new);
1864 copyb.new = A_REPLG(copyb.new);
1865 ast_unvisit();
1866 return TRUE; /* replacement occurred */
1867 }
1868 return FALSE; /* replacement did not occur */
1869}

2391/* check if all def of this ast is in allocate statement and it is safe */
2392LOGICAL
2393alldefs_allocsafe(int ast, int stmt)
2394{
2395 int def = 0;
2396 int fgx = STD_FG(stmt);
2397 int std = 0;
2398 int hasalloc = 0;
2399 BV *bv = NULL;
2400 LOGICAL is_inited = FALSE;
2401
2402 int nme = nme_of_ast(ast);
2403
2404 if (!fgx)
2405 return FALSE;
2406 if (nme) {
2407 bv = FG_UNINITED(STD_FG(stmt));
2408 is_inited = is_initialized(bv, nme);
2409 if (!is_inited) /* must be defined in this routine */
2410 return FALSE;
2411 while ((def = find_next_reaching_def(nme, fgx, def))) {
2412 std = DEF_STD(def);
2413 if (std == stmt) {
2414 continue;
2415 }
2416 if (is_alloc_std(std)) {
2417 int allocast = STD_AST(std);
2418
2419 if (!is_parentof_ast(A_SRCG(allocast), ast))
2420 continue;
2421
2422 hasalloc = 1;
2423 /* check if there is a pointer def or a call with this pointer */
2424 if (is_ptrdef_in_path(std, STD_FG(std), stmt, STD_FG(stmt), nme, ast)) {
2425 return FALSE;
2426 }
2427 } else {
2428 /* an assignment of a parent will cause an unsafe. */
2429 if (A_ASN == A_TYPEG(STD_AST(std))) {
2430 int destast = STD_AST(std);
2431 if (is_parentof_ast(A_DESTG(destast), ast)) {
2432 if (A_TYPEG(destast) == A_SUBSCR) {
2433 destast = A_LOPG(destast);
2434 if (ast == destast) /* normal assignment */
2435 continue;
2436 }
2437 return FALSE;
2438 }
2439 continue;
2440 } else if (A_TYPEG(STD_AST(std)) == A_ICALL &&
2441 A_OPTYPEG(STD_AST(std)) == I_PTR2_ASSIGN) {
2442 /* pointer associate constructs are ok; we already processed them,
2443 e.g., in ptrdefs_has_lhsconflict */
2444 hasalloc = 1;
2445 continue;
2446 } else
2447 return FALSE;
2448 }
2449 }
2450 /* all pointer defs are in allocate stmt and it is safe */
2451 if (hasalloc)
2452 return TRUE;
2453 }
2454 return FALSE;
2455}
