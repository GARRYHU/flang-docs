74double FN_PROTOTYPE(mth_i_dexp2)(double x)
75{
76 static const double max_exp2_arg = 1024.0, /* 0x4090000000000000 */
77 min_exp2_arg = -1074.0, /* 0xc090c80000000000 */
78 log2 = 6.931471805599453094178e-01, /* 0x3fe62e42fefa39ef */
79 log2_lead = 6.93147167563438415527E-01, /* 0x3fe62e42f8000000 */
80 log2_tail = 1.29965068938898869640E-08, /* 0x3e4be8e7bcd5e4f1 */
81 one_by_32_lead = 0.03125;
82
83 double y, z1, z2, z, hx, tx, y1, y2;
84 int m;
85 __UINT8_T ux, ax;
86
87 /*
88 Computation of exp2(x).
89
90 We compute the values m, z1, and z2 such that
91 exp2(x) = 2**m * (z1 + z2), where exp2(x) is 2**x.
92
93 Computations needed in order to obtain m, z1, and z2
94 involve three steps.
95
96 First, we reduce the argument x to the form
97 x = n/32 + remainder,
98 where n has the value of an integer and |remainder| <= 1/64.
99 The value of n = x * 32 rounded to the nearest integer and
100 the remainder = x - n/32.
101
102 Second, we approximate exp2(r1 + r2) - 1 where r1 is the leading
103 part of the remainder and r2 is the trailing part of the remainder.
104
105 Third, we reconstruct exp2(x) so that
106 exp2(x) = 2**m * (z1 + z2).
107 */
108
109 GET_BITS_DP64(x, ux);
110 ax = ux & (~SIGNBIT_DP64);
111
112 if (ax >= 0x4090000000000000) /* abs(x) >= 1024.0 */
113 {
114 if (ax >= 0x7ff0000000000000) {
115 /* x is either NaN or infinity */
116 if (ux & MANTBITS_DP64)
117 /* x is NaN */
118 return x + x; /* Raise invalid if it is a signalling NaN */
119 else if (ux & SIGNBIT_DP64)
120 /* x is negative infinity; return 0.0 with no flags. */
121 return 0.0;
122 else
123 /* x is positive infinity */
124 return x;
125 }
126 if (x > max_exp2_arg)
127 /* Return +infinity with overflow flag */
128 return retval_errno_erange_overflow(x);
129 else if (x < min_exp2_arg)
130 /* x is negative. Return +zero with underflow and inexact flags */
131 return retval_errno_erange_underflow(x);
132 }
133
134 /* Handle small arguments separately */
135 if (ax < 0x3fb7154764ee6c2f) /* abs(x) < 1/(16*log2) */
136 {
137 if (ax < 0x3c00000000000000) /* abs(x) < 2^(-63) */
138 return 1.0 + x; /* Raises inexact if x is non-zero */
139 else {
140 /* Split x into hx (head) and tx (tail). */
141 __UINT8_T u;
142 hx = x;
143 GET_BITS_DP64(hx, u);
144 u &= 0xfffffffff8000000;
145 PUT_BITS_DP64(u, hx);
146 tx = x - hx;
147 /* Carefully multiply x by log2. y1 is the most significant
148 part of the result, and y2 the least significant part */
149 y1 = x * log2_lead;
150 y2 = (((hx * log2_lead - y1) + hx * log2_tail) + tx * log2_lead) +
151 tx * log2_tail;
152
153 y = y1 + y2;
154 z = (9.99564649780173690e-1 +
155 (1.61251249355268050e-5 +
156 (2.37986978239838493e-2 + 2.68724774856111190e-7 * y) * y) *
157 y) /
158 (9.99564649780173692e-1 +
159 (-4.99766199765151309e-1 +
160 (1.070876894098586184e-1 +
161 (-1.189773642681502232e-2 + 5.9480622371960190616e-4 * y) * y) *
162 y) *
163 y);
164 z = ((z * y1) + (z * y2)) + 1.0;
165 }
166 } else {
167 /* Find m, z1 and z2 such that exp2(x) = 2**m * (z1 + z2) */
168
169 splitexp(x, log2, 32.0, one_by_32_lead, 0.0, &m, &z1, &z2);
170
171 /* Scale (z1 + z2) by 2.0**m */
172 if (m > EMIN_DP64 && m < EMAX_DP64)
173 z = scaleDouble_1((z1 + z2), m);
174 else
175 z = scaleDouble_2((z1 + z2), m);
176 }
177 return z;
178}
