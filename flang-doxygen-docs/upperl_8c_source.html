518void
519addcombination()
520/* see if this combination of operands has been seen before */
521{
522 int i, l, lc;
523 char c[MAXSTRING];
524 if (numoperands == 0)
525 return;
526 strcpy(c, operand[0]);
527 lc = strlen(c);
528 for (i = 1; i < numoperands; ++i) {
529 l = strlen(operand[i]);
530 if (lc + l >= MAXSTRING) {
531 fprintf(stderr, "line %d: too many operands; increase MAXSTRING\n",
532 linenum);
533 exit(1);
534 }
535 lc += l;
536 strcat(c, operand[i]);
537 }
538 for (i = 0; i < combinations; ++i) {
539 if (numoperands != combolen[i])
540 continue;
541 if (strcmp(c, combo[i]) == 0)
542 return; /* found it! */
543 }
544 /* didn't find it */
545 ++combinations;
546 if (combinations >= MAXCOMBO) {
547 fprintf(stderr,
548 "line %d: too many operand combinations; increase MAXCOMBO\n",
549 linenum);
550 exit(1);
551 }
552 combolen[combinations] = numoperands;
553 combo[combinations] = (char *)malloc(lc + 1);
554 strcpy(combo[combinations], c);
555
556 fprintf(outfile, "int OP%s[] = {", c);
557 for (i = 0; i < numoperands; ++i) {
558 if (i) {
559 fprintf(outfile, ",");
560 }
561 fprintf(outfile, " p%s", operand[i]);
562 }
563 fprintf(outfile, " };\n");
564 ++combinations;
565} /* addcombination */
