48/** \brief Go through and tag the communications for this statement.
49 \return 0 if any unstructured communication
50 */
51int
52tag_forall_comm(int ast)
53{
54 int std;
55 int l, r, o;
56 int a;
57 int i, nargs, argt;
58 int ndim, asd;
59 int src;
60 int arref;
61 int sptr;
62
63 a = ast;
64 if (!a)
65 return 1;
66 std = comminfo.std;
67 switch (A_TYPEG(ast)) {
68 /* expressions */
69 case A_BINOP:
70 o = A_OPTYPEG(a);
71 l = tag_forall_comm(A_LOPG(a));
72 r = tag_forall_comm(A_ROPG(a));
73 if (l == 0 || r == 0)
74 return 0;
75 return 1;
76 case A_UNOP:
77 o = A_OPTYPEG(a);
78 l = tag_forall_comm(A_LOPG(a));
79 if (l == 0)
80 return 0;
81 return 1;
82 case A_CONV:
83 l = tag_forall_comm(A_LOPG(a));
84 if (l == 0)
85 return 0;
86 return 1;
87 case A_PAREN:
88 l = tag_forall_comm(A_LOPG(a));
89 if (l == 0)
90 return 0;
91 return 1;
92 case A_MEM:
93 l = tag_forall_comm(A_PARENTG(a));
94 if (l == 0)
95 return 0;
96 return 1;
97 case A_SUBSTR:
98 return 1;
99 case A_INTR:
100 case A_FUNC:
101 nargs = A_ARGCNTG(a);
102 argt = A_ARGSG(a);
103 for (i = 0; i < nargs; ++i) {
104 l = tag_forall_comm(ARGT_ARG(argt, i));
105 if (l == 0)
106 return 0;
107 }
108 if (A_OPTYPEG(a) == I_CSHIFT) {
109 src = ARGT_ARG(argt, 0);
110 arref = A_RFPTRG(src);
111 assert(ARREF_CLASS(arref) == OVERLAP,
112 "tag_forall_comm: CSHIFT must be overlap", a, 2);
113 }
114 if (A_OPTYPEG(a) == I_EOSHIFT) {
115 src = ARGT_ARG(argt, 0);
116 arref = A_RFPTRG(src);
117 assert(ARREF_CLASS(arref) == OVERLAP,
118 "tag_forall_comm: EOSHIFT must be overlap", a, 2);
119 }
120
121 return 1;
122 case A_CNST:
123 case A_CMPLXC:
124 return 1;
125 case A_ID:
126 return 1;
127 case A_SUBSCR:
128 if (A_SHAPEG(a))
129 return 1;
130 sptr = sptr_of_subscript(a);
131 if (!ALIGNG(sptr)) {
132 asd = A_ASDG(a);
133 ndim = ASD_NDIM(asd);
134 for (i = 0; i < ndim; i++) {
135 l = tag_forall_comm(ASD_SUBS(asd, i));
136 if (l == 0)
137 return 0;
138 }
139 l = tag_forall_comm(A_LOPG(a));
140 return l;
141 }
142 /* It is distributed */
143 tag_comms(a);
144 return 1;
145 case A_CALL:
146 case A_ICALL:
147 nargs = A_ARGCNTG(a);
148 argt = A_ARGSG(a);
149 for (i = 0; i < nargs; ++i) {
150 if (!ARGT_ARG(argt, i))
151 continue;
152 l = tag_forall_comm(ARGT_ARG(argt, i));
153 if (l == 0)
154 return 0;
155 }
156 return 1;
157 default:
158 interr("tag_forall_comm: unknown expression", std, 2);
159 return 0;
160 }
161}

163void
164process_rhs_sub(int a)
165{
166 int ndim1, sptr;
167 int asd, ndim;
168 int subinfo;
169 int i;
170 int arref;
171 int align;
172
173 assert(A_TYPEG(a) == A_SUBSCR, "process_rhs_sub: not SUBSCR", a, 4);
174 asd = A_ASDG(a);
175 ndim = ASD_NDIM(asd);
176 sptr = sptr_of_subscript(a);
177 align = ALIGNG(sptr);
178 ndim1 = ASD_NDIM(A_ASDG(comminfo.sub));
179
180 arref = trans.arrb.stg_avail++;
181 TRANS_NEED(trans.arrb, ARREF, trans.arrb.stg_size + 100);
182 A_RFPTRP(a, arref);
183
184 subinfo = trans.subb.stg_avail;
185 trans.subb.stg_avail += ndim;
186 assert(ndim < 100, "transform_forall: ndim huge?", 0, 4);
187 TRANS_NEED(trans.subb, SUBINFO, trans.subb.stg_size + 1000);
188 /* make list of rhs references */
189 ARREF_SUB(arref) = subinfo;
190 ARREF_NDIM(arref) = ndim;
191 ARREF_NEXT(arref) = trans.rhsbase;
192 trans.rhsbase = arref;
193 ARREF_ARRSYM(arref) = sptr;
194 ARREF_ARR(arref) = a;
195 ARREF_TEMP(arref) = 0;
196 ARREF_CLASS(arref) = NO_CLASS;
197 for (i = 0; i < ndim; ++i) {
198 SUBI_DUPL(subinfo + i) = 0;
199 SUBI_NOP(subinfo + i) = 0;
200 SUBI_POP(subinfo + i) = 0;
201 SUBI_SUB(subinfo + i) = ASD_SUBS(asd, i);
202 SUBI_DSTT(subinfo + i) = 0;
203 SUBI_COMMT(subinfo + i) = COMMT_NOTAG;
204 process_sub(subinfo + i, A_LISTG(comminfo.forall));
205 }
206 matched_dim(a);
207}

209/* try to find communication class for rhs array a
210* communication class can be NO_COMM, OVERLAP,
211* COLLECTIVE, COPY_SECTION and IRREGULAR.
212* This is the beginning of communication detection algorithm.
213* It works for any alignment and distribution. Good luck!
214*/
215static void
216tag_comms(int a)
217{
218 int ndim1, sptr;
219 int asd, ndim;
220 int arref;
221 int align;
222 int forall;
223 int lhs;
224
225 assert(A_TYPEG(a) == A_SUBSCR, "tag_comms: not SUBSCR", a, 4);
226 sptr = sptr_of_subscript(a);
227 asd = A_ASDG(a);
228 ndim = ASD_NDIM(asd);
229 align = ALIGNG(sptr);
230 ndim1 = ASD_NDIM(A_ASDG(comminfo.sub));
231 forall = comminfo.forall;
232 lhs = A_DESTG(A_IFSTMTG(forall));
233
234 process_rhs_sub(a);
235 arref = A_RFPTRG(a);
236
237 ARREF_CLASS(arref) = NO_CLASS;
238 if (is_structured(a)) {
239 if (ARREF_CLASS(arref) == NO_CLASS) {
240 no_comm_class(a);
241 }
242 if (ARREF_CLASS(arref) == NO_CLASS) {
243 overlap_class(a);
244 }
245 }
246 if (ARREF_CLASS(arref) == NO_CLASS) {
247 ARREF_CLASS(arref) = NO_COMM;
248 }
249}

549/* This return check the sign of structured communication.
550 * lhs and rhs has to be same template.
551 * each rhs subscripts has to be linear.
552 * matched dimension has to have same forall index not like lhs(j)=rhs(i)
553 */
554static int
555is_structured(int a)
556{
557 int sptr;
558 int asd, ndim;
559 int subinfo;
560 int i, j;
561 int arref;
562
563 asd = A_ASDG(a);
564 ndim = ASD_NDIM(asd);
565 sptr = sptr_of_subscript(a);
566 arref = A_RFPTRG(a);
567 subinfo = ARREF_SUB(arref);
568 for (i = 0; i < ndim; ++i) {
569 /* if rhs index is not linear */
570 if (SUBI_BASE(subinfo + i) == 0)
571 return 0;
572 /* if matched, has to have same index */
573 j = SUBI_LDIM(subinfo + i);
574 if (j != -1) {
575 if ((SUBI_IDX(subinfo + i) != 0) &&
576 (SUBI_IDX(comminfo.subinfo + j) != 0) &&
577 (ASTLI_SPTR(SUBI_IDX(subinfo + i)) !=
578 ASTLI_SPTR(SUBI_IDX(comminfo.subinfo + j))))
579 return 0;
580 }
581 }
582 return 1;
583}

585/* Algorithm:
586 * This routine will only work for BLOCK and GEN_BLOCK distribution.
587 * It only mark array with OVERLAP,
588 * This only marks dimension for COMMT_SHIFTC or COMMT_SHIFTV for BLOCK
589 * If it marked before as COMMT_NONE, COMMT_CONST, it respect that.
590 * iff each dimension is taged COMMT_SHIFTC, COMMT_NONE or COMMT_CONST.
591 */
592static void
593overlap_class(int a)
594{
595 int ndim1, sptr;
596 int asd, ndim, l;
597 int subinfo;
598 int i, j;
599 int arref;
600 int align;
601 int count;
602 int diff;
603 int shdw;
604
605 asd = A_ASDG(a);
606 ndim = ASD_NDIM(asd);
607 l = A_LOPG(a);
608 sptr = sptr_of_subscript(a);
609 if (POINTERG(sptr))
610 return;
611 align = 0;
612 shdw = 0;
613 ndim1 = ASD_NDIM(A_ASDG(comminfo.sub));
614 arref = A_RFPTRG(a);
615 subinfo = ARREF_SUB(arref);
616 for (i = 0; i < ndim; ++i) {
617 j = SUBI_LDIM(subinfo + i);
618 if (j != -1) {
619 if (SUBI_COMMT(subinfo + i) == COMMT_NONE ||
620 SUBI_COMMT(subinfo + i) == COMMT_CONST)
621 continue;
622 if (SUBI_CNST(subinfo + i)) {
623 diff = SUBI_DIFF(subinfo + i);
624 if (A_TYPEG(diff) == A_CNST) {
625 SUBI_COMMT(subinfo + i) = COMMT_SHIFTC;
626 SUBI_COMMV(subinfo + i) = diff;
627 }
628 }
629 }
630 }
631 count = 0;
632 for (i = 0; i < ndim; ++i)
633 if (SUBI_COMMT(subinfo + i) == COMMT_SHIFTC ||
634 SUBI_COMMT(subinfo + i) == COMMT_NONE ||
635 SUBI_COMMT(subinfo + i) == COMMT_CONST)
636 count++;
637 if (ndim == count)
638 ARREF_CLASS(arref) = OVERLAP;
639}

641/* Algorithm:
642 * This routine does not care about distribution types.
643 * It only cares about alignment.
644 * If array is not distributed, it will be marked NO_COMM class.
645 * If lhs and rhs array is in different template,
646 * this routine can not guarantee NO_COMM class.
647 * If each dimension of rhs array marked as COMMT_NONE or COMMT_CONST.
648 * then the rhs array is marked as NO_COMM class.
649 */
650static void
651no_comm_class(int a)
652{
653 int sptr;
654 int asd, ndim;
655 int subinfo;
656 int i;
657 int arref;
658 int align;
659 int no_comm;
660 int zero = astb.bnd.zero;
661 int forall, lhs;
662 int align1, sptr1;
663 LOGICAL single_ok[7];
664
665 asd = A_ASDG(a);
666 ndim = ASD_NDIM(asd);
667 sptr = sptr_of_subscript(a);
668 align = ALIGNG(sptr);
669 arref = A_RFPTRG(a);
670 subinfo = ARREF_SUB(arref);
671 /* simple case */
672 forall = comminfo.forall;
673 lhs = dist_ast(A_DESTG(A_IFSTMTG(forall)));
674 sptr1 = sptr_of_subscript(lhs);
675 align1 = ALIGNG(sptr1);
676 if (lhs == a) {
677 ARREF_CLASS(arref) = NO_COMM;
678 return;
679 }
680 for (i = 0; i < 7; i++)
681 single_ok[i] = FALSE;
682
683 no_comm = 0;
684 for (i = 0; i < ndim; ++i) {
685 {
686 if (SUBI_STRIDE(subinfo + i) == zero)
687 SUBI_COMMT(subinfo + i) = COMMT_CONST;
688 else
689 SUBI_COMMT(subinfo + i) = COMMT_NONE;
690 no_comm++;
691 continue;
692 }
693 }
694
695 if (ndim == no_comm)
696 ARREF_CLASS(arref) = NO_COMM;
697}

699#ifdef FLANG_DETECT_UNUSED
700/* Algorithm:
701 * This routine does not cares about neither template nor distribution types.
702 * It looks the subscripts of lhs and rhs:
703 * -All subscripts have to be linear.
704 * -No diagonal accesses.
705 * -No transpose accesses.
706 * -All array assignments are okay except indirection.
707 */
708
709static void
710copy_section_class(int a)
711{
712 int ndim1;
713 int asd, ndim;
714 int subinfo, subinfo_lhs;
715 int j;
716 int arref;
717 int forall;
718 int list;
719 int lhs;
720
721 arref = A_RFPTRG(a);
722 forall = comminfo.forall;
723 list = A_LISTG(forall);
724 lhs = comminfo.sub;
725
726 /* if it was array assignment, It is okay */
727 /* Except indirection array */
728
729 if (A_ARRASNG(forall) && !is_indirection_in_it(a)) {
730 ARREF_CLASS(arref) = COPY_SECTION;
731 return;
732 }
733
734 /* rhs: has to have linear subscripts */
735 asd = A_ASDG(a);
736 ndim = ASD_NDIM(asd);
737 subinfo = ARREF_SUB(arref);
738 for (j = 0; j < ndim; ++j)
739 if (SUBI_BASE(subinfo + j) == 0)
740 return;
741
742 /* lhs: has to have linear subscripts */
743 ndim1 = ASD_NDIM(A_ASDG(comminfo.sub));
744 subinfo_lhs = comminfo.subinfo;
745 for (j = 0; j < ndim1; ++j)
746 if (SUBI_BASE(subinfo_lhs + j) == 0)
747 return;
748
749 /* rhs: no diagonal access */
750 if (is_duplicate(a, list))
751 return;
752
753 /* lhs: no diagonal access */

789/* This routine checks whether array a has nice subscripts for
790 * scatterx and gatherx routines.
791 * It can have duplicate.
792 * It can have vector subscript
793 * but vector subscript has to has scalar or idx.
794 * other subscript can have a*i+c
795 */
796static LOGICAL
797is_scatterx_gatherx_subscript(int ast, int forall)
798{
799 int a, list;
800 /* test that vector subscript must not be diagonal access */
801 list = A_LISTG(forall);
802 a = ast;
803 while (A_TYPEG(a) != A_ID) {
804 if (A_TYPEG(a) == A_MEM) {
805 a = A_PARENTG(a);
806 } else if (A_TYPEG(a) == A_SUBSCR) {
807 int i, ndim, sptr, dtype;
808 int asd;
809 sptr = sptr_of_subscript(a);
810 dtype = DTYPEG(sptr);
811 assert(DTY(dtype) == TY_ARRAY,
812 "is_gatherx_scatterx_subscript: must be array", sptr, 4);
813
814 asd = A_ASDG(a);
815 ndim = ASD_NDIM(asd);
816 for (i = 0; i < ndim; ++i) {
817 int ss;
818 ss = ASD_SUBS(asd, i);
819 if (is_scalar(ss, list) || is_idx(ss, list)) {
820 /* ok */
821 } else if (is_vector_subscript(ss, list)) {
822 int lb;
823 if (is_duplicate(ss, list))
824 return FALSE;
825 lb = ADD_LWBD(dtype, i);
826 if (XBIT(58, 0x22) && lb != 0 && lb != astb.bnd.one)

828 * for gather/scatter to work, array must really be
829 * indexed 1:extent */
830 return FALSE;
831 } else {
832 /* not ok */
833 return FALSE;
834 }
835 }
836 a = A_LOPG(a);
837 } else {
838 interr("is_gatherx_scatterx_subscript: not member or subscript",
839 A_TYPEG(a), 3);

841 }
842 }

1344void
1345search_idx(int ast, int list, int *astli, int *base, int *stride)
1346{
1347 int i;
1348 int base1, base2, stride1, stride2;
1349 int zero = astb.bnd.zero;
1350 int opc;
1351 int astli_a, nidx;
1352
1353 astli_a = 0;
1354 nidx = 0;
1355 search_forall_idx(ast, list, &astli_a, &nidx);
1356 if (nidx == 0 && astli_a == 0) {
1357 *stride = zero;
1358 *base = ast;
1359 return;
1360 }
1361
1362 switch (A_TYPEG(ast)) {
1363 case A_ID:
1364 for (i = list; i != 0; i = ASTLI_NEXT(i))
1365 if (ASTLI_SPTR(i) == A_SPTRG(ast))
1366 goto found;
1367 *base = ast;
1368 *stride = zero;
1369 return;
1370 found:
1371 if (*astli && *astli != i) {
1372 /* too many index variables */
1373 *base = 0;
1374 *stride = 0;
1375 return;
1376 }
1377 *astli = i;
1378 *stride = mk_isz_cval(1, astb.bnd.dtype);
1379 *base = zero;
1380 return;
1381
1382 case A_BINOP:
1383 search_idx(A_LOPG(ast), list, astli, &base1, &stride1);
1384 if (base1 == 0) {
1385 *base = *stride = 0;
1386 return;
1387 }
1388 search_idx(A_ROPG(ast), list, astli, &base2, &stride2);
1389 if (base2 == 0) {
1390 *base = *stride = 0;
1391 return;
1392 }
1393 if (stride1 == zero && stride2 == 0) {
1394 *base = ast;
1395 *stride = zero;
1396 return;
1397 }
1398 switch (opc = A_OPTYPEG(ast)) {
1399 case OP_ADD:
1400 case OP_SUB:
1401 *base = opt_binop(opc, base1, base2, A_DTYPEG(ast));
1402 *stride = opt_binop(opc, stride1, stride2, A_DTYPEG(ast));
1403 return;
1404
1405 case OP_MUL:
1406 if (stride1 == zero) {
1407 /* invar * induc */
1408 *base = opt_binop(OP_MUL, base2, base1, A_DTYPEG(ast));
1409 *stride = opt_binop(OP_MUL, stride2, base1, A_DTYPEG(ast));
1410 } else if (stride2 == zero) {
1411 /* induc * invar */
1412 *base = opt_binop(OP_MUL, base1, base2, A_DTYPEG(ast));
1413 *stride = opt_binop(OP_MUL, stride1, base2, A_DTYPEG(ast));
1414 } else {
1415 /* classic nonlinear */
1416 *base = *stride = 0;
1417 }
1418 return;
1419 default:
1420 /* unknown binary op */
1421 *base = *stride = 0;
1422 return;
1423 }
1424
1425 case A_UNOP:
1426 search_idx(A_LOPG(ast), list, astli, &base1, &stride1);
1427 if (base1 == 0) {
1428 *base = *stride = 0;
1429 return;
1430 }
1431 if (stride1 == zero) {
1432 *base = ast;
1433 *stride = zero;
1434 return;
1435 }
1436 switch (opc = A_OPTYPEG(ast)) {
1437 case OP_ADD:
1438 *base = base1;
1439 *stride = stride1;
1440 return;
1441 case OP_SUB:
1442 *base = opt_unop(OP_SUB, base1, A_DTYPEG(ast));
1443 *stride = opt_unop(OP_SUB, stride1, A_DTYPEG(ast));
1444 return;
1445 default:
1446 /* unknown binary op */
1447 *base = *stride = 0;
1448 return;
1449 }
1450 case A_CONV:
1451 search_idx(A_LOPG(ast), list, astli, &base1, &stride1);
1452 if (base1 == 0) {
1453 *base = *stride = 0;
1454 return;
1455 }
1456 *base = mk_convert(base1, A_DTYPEG(ast));
1457 *stride = mk_convert(stride1, A_DTYPEG(ast));
1458 return;
1459
1460 case A_CMPLXC:
1461 case A_CNST:
1462 *stride = zero;
1463 *base = ast;
1464 return;
1465
1466 case A_PAREN:
1467 search_idx(A_LOPG(ast), list, astli, &base1, &stride1);
1468 if (base1 == 0) {
1469 *base = *stride = 0;
1470 return;
1471 }
1472 *base = base1;
1473 *stride = stride1;
1474 return;
1475
1476 case A_SUBSCR:
