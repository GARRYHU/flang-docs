27{
28 // Quick exit if argument is +/-0.0
29 const uint32_t a_u32 = *reinterpret_cast<const uint32_t *>(&a);
30 if ((a_u32 << 1) == 0) return 1.0;
31
32 if (a != a)
33 return a;
34 if (a >= EXP_HI)
35 return itf(INF);
36 if (a <= EXP_LO)
37 return 0.0f;
38 float t = FMAF(a, L2E, FLT2INT_CVT);
39 float tt = t - FLT2INT_CVT;
40 float z = FMAF(tt, -LN2_0, a);
41 z = FMAF(tt, -LN2_1, z);
42
43 int exp = fti(t);
44 exp <<= 23;
45
46 float zz = EXP_C7;
47 zz = FMAF(zz, z, EXP_C6);

51 zz = FMAF(zz, z, EXP_C2);
52 zz = FMAF(zz, z, EXP_C1);
53 zz = FMAF(zz, z, EXP_C0);
54
55 if (a <= EXP_DN) {
56 int dnrm = exp > DNRM_THR ? DNRM_THR : exp;
57 dnrm = dnrm + DNRM_SHFT;
58 exp = exp > DNRM_THR ? exp : DNRM_THR;
59 float res = itf(exp + fti(zz));
60 res = res * itf(dnrm);
61 return res;
62 } else {

64 }
