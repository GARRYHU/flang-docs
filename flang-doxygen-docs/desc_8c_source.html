83#define TEMPLATE(dd, i, lb, ub, lbase, gsize) \
84 { \
85 __INT_T __extent, u, l; \
86 DECL_DIM_PTRS(_dd); \
87 SET_DIM_PTRS(_dd, dd, i - 1); \
88 l = lb; \
89 u = ub; \
90 __extent = u - l + 1; \
91 if (__extent < 0) { \
92 __extent = 0; \
93 u = l - 1; \
94 } \
95 F90_DPTR_LBOUND_P(_dd, l); \
96 DPTR_UBOUND_P(_dd, u); \
97 F90_DPTR_SSTRIDE_P(_dd, 1); \
98 F90_DPTR_SOFFSET_P(_dd, 0); \
99 lbase -= gsize * l; \
100 F90_DPTR_LSTRIDE_P(_dd, gsize); \
101 gsize *= __extent; \
102 }
