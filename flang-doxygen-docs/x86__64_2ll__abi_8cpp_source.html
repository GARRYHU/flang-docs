27/* Parameter classes from the AMD64 ABI Draft 0.99.6 section on parameter
28 * passing. */
29enum amd64_class {
30 AMD64_NO_CLASS = 0,
31 AMD64_INTEGER,
32 AMD64_SSE,
33 AMD64_SSEUP,
34 AMD64_X87,
35 AMD64_X87UP,
36 AMD64_COMPLEX_X87,
37 AMD64_MEMORY
38};

40/* Merge two argument classes for the case when there are multiple objects in
41 * one eight-byte. */
42static enum amd64_class
43amd64_merge(enum amd64_class c1, enum amd64_class c2)
44{
45 /* (a) If both classes are equal, this is the resulting class. */
46 if (c1 == c2)
47 return c1;
48
49 /* (b) If one of the classes is NO_CLASS, the resulting class is the other
50 * class. */
51 if (c1 == AMD64_NO_CLASS)
52 return c2;
53 if (c2 == AMD64_NO_CLASS)
54 return c1;
55
56 /* (c) If one of the classes is MEMORY, the result is the MEMORY class. */
57 if (c1 == AMD64_MEMORY || c2 == AMD64_MEMORY)
58 return AMD64_MEMORY;
59
60 /* (d) If one of the classes is INTEGER, the result is the INTEGER. */
61 if (c1 == AMD64_INTEGER || c2 == AMD64_INTEGER)
62 return AMD64_INTEGER;
63
64 /* (e) If one of the classes is X87, X87UP, COMPLEX_X87 class, MEMORY is used
65 * as class. */
66 if (c1 == AMD64_X87 || c2 == AMD64_X87 || c1 == AMD64_X87UP ||
67 c2 == AMD64_X87UP || c1 == AMD64_COMPLEX_X87 || c2 == AMD64_COMPLEX_X87)
68 return AMD64_MEMORY;
69
70 /* (f) Otherwise class SSE is used. */
71 return AMD64_SSE;
72}

74/* The maximum size of a single argument is 4 eightbytes, but that is only when
75 * passing a single __m256 in a %ymm register. In all other cases, the argument
76 * must be at most two eightbytes.
77 *
78 * We'll special-case the __mm256 case and only classify the first two
79 * eightbytes.
80 *
81 * LLVM: we want to simply pass vectors as arguments in LLVM.
82 */
83
84/* Update classification in eightbyte[] when adding a dtype at offset, or
85 * return 1 to indicate that everything needs to be passed in memory.
86 */
87static int
88amd64_update_class(void *context, DTYPE dtype, unsigned address,
89 int member_sptr)
90{
91 enum amd64_class *ebc = (enum amd64_class *)context;
92
93 /* Special case for __mm256 as mentioned above. */
94 if (DTY(dtype) == TY_256 && address == 0) {
95 ebc[0] = amd64_merge(ebc[0], AMD64_SSE);
96 ebc[1] = amd64_merge(ebc[1], AMD64_SSEUP);
97 return 0;
98 }
99
100 /* Since this is LLVM, then we want to pass vectors by value and not as
101 temps on the stack. */
102 if (DT_ISVECT(dtype)) {
103 ebc[0] = AMD64_SSE; /* force amd64_classify() */
104 ebc[1] = AMD64_SSEUP;
105 return 0;
106 }
107
108
109 /* Check if this overflows our two ebcs. */
110 unsigned eight_num = address / 8;
111 unsigned size = zsize_of(dtype);
112 if (!size || address + size > 16)
113 return 1;
114
115 /* Unaligned fields cause the whole argument to be passed in memory. */
116 unsigned offset = address % 8;
117 if (offset & alignment(dtype))
118 return 1;
119
120 /* Handle member arrays by recursively calling amd64_update_class() with
121 * one or two representative array elements. */
122 if (DTY(dtype) == TY_ARRAY) {
123 /* TY_ARRAY dtype dim */
124 DTYPE ddtype = DTySeqTyElement(dtype);
125 int retval = amd64_update_class(context, ddtype, address, 0);
126 /* Does this array extend into the next 8-byte segment? */
127 if (retval == 0 && address < 8 && address + size > 8)
128 retval = amd64_update_class(context, ddtype, address + 8, 0);
129 return retval;
130 }
131
132 if (size <= 8) {
133 bool is_ptr = DTY(dtype) == TY_PTR;
134 enum amd64_class cls = AMD64_MEMORY;
135 if (DT_ISINT(dtype) || is_ptr)
136 cls = AMD64_INTEGER;
137 else if (DT_ISREAL(dtype) || DT_ISCMPLX(dtype))
138 cls = AMD64_SSE;
139 ebc[eight_num] = amd64_merge(ebc[eight_num], cls);
140 return ebc[eight_num] == AMD64_MEMORY;
141 }
142
143 /* This type is larger than 8 bytes. It must be aligned. */
144 if (address != 0)
145 return 1;
146
147 enum amd64_class cls[2] = {AMD64_MEMORY, AMD64_MEMORY};
148 switch (DTY(dtype)) {
149 default:
150 break;
151
152 case TY_VECT:
153 case TY_128:
154 case TY_FLOAT128:
155 cls[0] = AMD64_SSE;
156 cls[1] = AMD64_SSEUP;
157 break;
158
159 case TY_INT128:
160 cls[0] = AMD64_INTEGER;
161 cls[1] = AMD64_INTEGER;
162 break;
163
164 case TY_DCMPLX:
165 cls[0] = AMD64_SSE;
166 cls[1] = AMD64_SSE;
167 break;
168 }
169 ebc[0] = amd64_merge(ebc[0], cls[0]);
170 ebc[1] = amd64_merge(ebc[1], cls[1]);
171
172 return ebc[0] == AMD64_MEMORY;
173}

175/* Classify dtype for passing as an argument or return value.
176 *
177 * Return 1 if argument must be passed in memory, otherwise return 0 and update
178 * the eightbyte classes.
179 */
180static int
181amd64_classify(enum amd64_class ebc[2], DTYPE dtype)
182{
183 unsigned i;
184 for (i = 0; i != 2; i++)
185 ebc[i] = AMD64_NO_CLASS;
186 if (visit_flattened_dtype(amd64_update_class, (void *)ebc, dtype, 0, 0))
187 return 1;
188
189 /* Post-merge cleanup.
190 (a) If one of the classes is MEMORY, the whole argument is passed in
191 memory. */
192 if (ebc[0] == AMD64_MEMORY || ebc[1] == AMD64_MEMORY)
193 return 1;
194 /* (b) If X87UP is not preceded by X87, the whole argument is passed in
195 memory. */
196 if (ebc[1] == AMD64_X87UP && ebc[0] != AMD64_X87)
197 return 1;
198 /* (c) If the size of the aggregate exceeds two eightbytes and the first
199 eightbyte isn’t SSE or any other eightbyte isn’t SSEUP, the whole
200 argument is passed in memory. */
201 if (zsize_of(dtype) > 16 && (ebc[0] != AMD64_SSE || ebc[1] != AMD64_SSEUP))
202 return 1;
203
204 /* This is a register argument, assuming there are free registers available.
205 */
206 return 0;
207}

209/* Compute coercion types for passing or returning dtype in registers, using
210 * ebc from amd64_classify(). */
211static void
212amd64_coerce(LL_Module *module, LL_ABI_ArgInfo *arg,
213 const enum amd64_class ebc[2], DTYPE dtype)
214{
215 unsigned i;
216 LL_Type *types[2] = {NULL, NULL};
217 ISZ_T size = zsize_of(dtype);
218
219 arg->kind = LL_ARG_COERCE;
220
221 /* This is a single vector register. */
222 if (ebc[0] == AMD64_SSE && ebc[1] == AMD64_SSEUP) {
223 /* Possible coercion types: <2 x double>, <4 x float>, <4 x i32>
224 Also 256-bit vectors: <4 x double>, ...
225 TODO: Create a coercion type that better matches the original. */
226 LL_Type *ltype = ll_create_basic_type(module, LL_FLOAT, 0);
227 unsigned lanes = (size > 16) ? 8 : 4;
228 arg->type = ll_get_vector_type(ltype, lanes);
229 return;
230 }
231
232 if (ebc[0] == AMD64_X87 && ebc[1] == AMD64_X87UP) {
233 /* A single 16-byte fp80 value. */
234 arg->type = ll_create_basic_type(module, LL_X86_FP80, 0);
235 return;
236 }
237
238 /* This is one or two registers. */
239 for (i = 0; i != 2; ++i) {
240 switch (ebc[i]) {
241 case AMD64_NO_CLASS:
242 break;
243 case AMD64_INTEGER:
244 if (size < 8 * i + 8)
245 types[i] = ll_create_int_type(module, 8 * (size - 8 * i));
246 else
247 types[i] = ll_create_basic_type(module, LL_I64, 0);
248 break;
249 case AMD64_SSE:
250 /* Possibilities: float, double, <2 x float> */
251 if (dtype == DT_CMPLX) {
252 LL_Type *ftype = ll_create_basic_type(module, LL_FLOAT, 0);
253 LL_Type *pair[2] = {ftype, ftype};
254 types[i] = ll_create_anon_struct_type(module, pair, 2, true, 0);
255 }
256 else if (size == 8 * i + 4)
257 types[i] = ll_create_basic_type(module, LL_FLOAT, 0);
258 else
259 types[i] = ll_create_basic_type(module, LL_DOUBLE, 0);
260 break;
261 default:
262 assert(0, "Unexpected eightbyte class", ebc[i], ERR_Fatal);
263 break;
264 }
265 }
266
267 /* An undefined struct shows up with size = 0 and ebc = { NO_CLASS,
268 * NO_CLASS }. We can't compute the correct handling of that struct here,
269 * so just use an i64. This wil be OK for function pointers, not for
270 * actually making a call. */
271 if (size == 0)
272 types[0] = ll_create_int_type(module, 64);
273 else
274 assert(types[0], "No class for first register", 0, ERR_Fatal);
275
276 /* Build an anonymous struct if we have two registers. */
277 if (types[1]) {
278 arg->type = ll_create_anon_struct_type(module, types, 2, false, LL_AddrSp_Default);
279 } else {
280 arg->type = types[0];
281 }
282}

284/* Classify an integer type for return or arg. */
285static enum LL_ABI_ArgKind
286classify_int(DTYPE dtype)
287{
288 /* Integer types smaller than a register must be sign/zero extended. */
289 if (size_of(dtype) < 8)
290 return DT_ISUNSIGNED(dtype) ? LL_ARG_ZEROEXT : LL_ARG_SIGNEXT;
291
292 return LL_ARG_DIRECT;
293}

295void
296ll_abi_classify_return_dtype(LL_ABI_Info *abi, DTYPE dtype)
297{
298 enum amd64_class ebc[2];
299
300 if (dtype == DT_VOIDNONE) {
301 abi->arg[0].kind = LL_ARG_DIRECT;
302 return;
303 }
304
305
306 if (amd64_classify(ebc, dtype)) {
307 /* Must be returned in memory via an sret pointer argument. */
308 abi->used_iregs = 1;
309 abi->arg[0].kind = LL_ARG_INDIRECT;
310 return;
311 }
312
313 /* Integer types can be returned in registers, possibly sign/zero extended. */
314 if (DT_ISINT(dtype)) {
315 abi->arg[0].kind = classify_int(dtype);
316 return;
317 }
318
319 /* Basic types can be returned in registers directly. */
320 if ((DT_ISSCALAR(dtype) && !DT_ISCMPLX(dtype)) || DT_ISVECT(dtype)) {
321 abi->arg[0].kind = LL_ARG_DIRECT;
322 return;
323 }
324
325 /* Other types need to be coerced into one or two arguments corresponding
326 to the registers. */
327 amd64_coerce(abi->module, &abi->arg[0], ebc, dtype);
328}

330void
331ll_abi_classify_arg_dtype(LL_ABI_Info *abi, LL_ABI_ArgInfo *arg, DTYPE dtype)
332{
333 enum amd64_class ebc[2];
334 bool inregs;
335
336 inregs = amd64_classify(ebc, dtype) == 0;
337
338 /* If inregs is true, this argument can be passed in registers, but only if
339 enough are available. */
340 unsigned need_iregs = (ebc[0] == AMD64_INTEGER) + (ebc[1] == AMD64_INTEGER);
341 unsigned need_fregs = (ebc[0] == AMD64_SSE) + (ebc[1] == AMD64_SSE);
342
343 /* %rdi, %rsi, %rdx, %rcx, %r8 and %r9. */
344 if (abi->used_iregs + need_iregs > 6)
345 inregs = 0;
346 /* %xmm0 - %xmm7 */
347 if (abi->used_fregs + need_fregs > 8)
348 inregs = 0;
349
350 if (inregs) {
351 /* Registers are available, update the counters. */
352 abi->used_iregs += need_iregs;
353 abi->used_fregs += need_fregs;
354 }
355
356 /* Integer types can be passed in registers, possibly sign/zero extended. */
357 if (DT_ISINT(dtype)) {
358 arg->kind = classify_int(dtype);
359 return;
360 }
361
362 /* For simple types, LLVM figures out whether they go on the stack or in
363 registers. */
364 if ((DT_ISSCALAR(dtype) && !DT_ISCMPLX(dtype)) || DT_ISVECT(dtype)) {
365 arg->kind = LL_ARG_DIRECT;
366 return;
367 }
368
369 /* This disables coercion for Fortran... for now, including ISO-C.
370 * We probably want to lift the ISO-C restriction in the future.
371 */
372 if (inregs)
373 amd64_coerce(abi->module, arg, ebc, dtype);
374 else
375 arg->kind = LL_ARG_BYVAL;
376}

378/* Map values for __builtin_va_genarg(). See rte/pgc/hammer/src/va_arg.c. */
379#define GP_XM 0
380#define XM_GP 1
381#define XM_XM 2
382
383unsigned
384ll_abi_classify_va_arg_dtype( LL_Module* module, DTYPE dtype,
385 unsigned *num_gp, unsigned *num_fp)
386{
387 enum amd64_class ebc[2];
388
389 if (amd64_classify(ebc, dtype) == 0) {
390 *num_gp = (ebc[0] == AMD64_INTEGER) + (ebc[1] == AMD64_INTEGER);
391 *num_fp = (ebc[0] == AMD64_SSE) + (ebc[1] == AMD64_SSE);
392
393 if (ebc[0] == AMD64_INTEGER && ebc[1] == AMD64_SSE)
394 return GP_XM;
395 if (ebc[0] == AMD64_SSE && ebc[1] == AMD64_INTEGER)
396 return XM_GP;
397 } else {
398 *num_fp = 0;
399 *num_gp = 0;
400 }
401 /* Return value is only used when we end up calling __builtin_va_genarg(). */
402 return XM_XM;
403}
