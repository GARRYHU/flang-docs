88extern FIO_FCB *
89__fortio_alloc_fcb(void)
90{
91 FIO_FCB *p;
92
93 if (fcb_avail) { /* return item from avail list */
94 p = fcb_avail;
95 fcb_avail = p->next;
96 } else { /* call malloc for some new space */
97 int i;
98 p = (FIO_FCB *)malloc(CHUNKSZ * sizeof(FIO_FCB));
99 assert(p);
100 /*
101 * Waste the first element of the chunk; the first element is
102 * used to link all of the chunks together so that they can
103 * be freed upon program termination.
104 */
105 for (i = 2; i < CHUNKSZ - 1; i++) /* create avail list */
106 p[i].next = &p[i + 1];
107 p[CHUNKSZ - 1].next = NULL; /* end of avail list */
108 fcb_avail = &p[2];
109
110 p[0].next = fcb_chunks;
111 fcb_chunks = p;
112
113 p++;
114 }
115
116 memset(p, 0, sizeof(FIO_FCB));
117 p[0].next = fioFcbTbls.fcbs; /* add new FCB to front of list */
118 fioFcbTbls.fcbs = p;
119 return p;
120}

122extern void
123__fortio_free_fcb(FIO_FCB *p)
124{
125 if (p->name) {
126 free(p->name);
127 p->name = NULL;
128 }
129
130 if (fioFcbTbls.fcbs == p) /* delete p from list */
131 fioFcbTbls.fcbs = p->next;
132 else {
133 FIO_FCB *q;
134 for (q = fioFcbTbls.fcbs; q; q = q->next) /* find predecessor of p */
135 if (q->next == p)
136 break;
137 assert(q != NULL); /* trying to free unallocated block */
138 q->next = p->next;
139 }
140
141 p->next = fcb_avail; /* add to front of avail list */
142 fcb_avail = p;
143}

145extern void
146__fortio_cleanup_fcb()
147{
148 FIO_FCB *p, *p_next;
149 for (p = fcb_chunks; p; p = p_next) {
150 p_next = p->next;
151 free(p);
152 }
153 fcb_avail = NULL;
154 fcb_chunks = NULL;
155}

159extern FIO_FCB *__fortio_rwinit(
160 int unit, int form, /* FIO_FORMATTED, FIO_UNFORMATTED */
161 __INT_T *recptr, /* ptr to record number, may be present or NULL */
162 int optype) /* 0,1,2 - read, write, endfile */
163{
164 FIO_FCB *f;
165 int errflag;
166 seekoffx_t pos; /* typedef in pgstdio.h */
167 seekoffx_t rec;
168 bool rec_specified;
169
170#define ERR(code) \
171 { \
172 (void)__fortio_error(code); \
173 return NULL; \
174 }
175
176 if (recptr == 0 || !ISPRESENT(recptr)) {
177 rec = 0;
178 rec_specified = FALSE;
179 } else {
180 rec = *recptr;
181 rec_specified = TRUE;
182 }
183
184 f = __fortio_find_unit(unit);
185 if (f == NULL) { /* unit not connected */
186 int status = FIO_UNKNOWN;
187
188 if (optype == 0) /* if READ, error if file does not exist */
189 status = FIO_OLD;
190 if (!fioFcbTbls.pos_present) {
191 errflag = __fortio_open(unit, FIO_READWRITE, status, FIO_KEEP,
192 FIO_SEQUENTIAL, FIO_NULL, form, FIO_NONE,
193 FIO_ASIS, FIO_YES, 0, NULL /*name*/, 0);
194 if (errflag != 0)
195 return NULL;
196 f = __fortio_find_unit(unit);
197 assert(f && f->acc == FIO_SEQUENTIAL);
198 } else {
199 errflag = __fortio_open(unit, FIO_READWRITE, status, FIO_KEEP, FIO_STREAM,
200 FIO_NULL, form, FIO_NONE, FIO_ASIS, FIO_YES, 0,
201 NULL /*name*/, 0);
202 if (errflag != 0)
203 return NULL;
204 f = __fortio_find_unit(unit);
205 assert(f && f->acc == FIO_STREAM);
206 if (f->form == FIO_UNFORMATTED)
207 f->binary = TRUE;
208 fioFcbTbls.pos_present = FALSE;
209 pos = fioFcbTbls.pos;
210 /*
211 * spec says a pos of 1 => beginning of file
212 * INQUIRE always adds one.
213 */
214 if (pos > 0)
215 --pos;
216 else
217 ERR(FIO_EPOSV);
218 if (__io_fseek(f->fp, (seekoffx_t)pos, SEEK_SET) != 0)
219 ERR(__io_errno());
220 f->coherent = 0;
221 }
222 } else { /* unit is already connected: */
223
224 /* check for outstanding async i/o */
225
226 if (f->asy_rw) { /* stop any async i/o */
227 f->asy_rw = 0;
228 if (Fio_asy_disable(f->asyptr) == -1) {
229 return (NULL);
230 }
231 }
232
233 if (fioFcbTbls.pos_present) {
234 fioFcbTbls.pos_present = FALSE;
235 if (f->acc != FIO_STREAM) {
236 ERR(FIO_EPOS);
237 } else {
238 pos = fioFcbTbls.pos;
239 /*
240 * spec says a pos of 1 => beginning of file
241 * INQUIRE always adds one.
242 */
243 if (pos > 0)
244 --pos;
245 else
246 ERR(FIO_EPOSV);
247 if (__io_fseek(f->fp, (seekoffx_t)pos, SEEK_SET) != 0)
248 ERR(__io_errno());
249 f->coherent = 0;
250 f->eof_flag = FALSE; /* clear it for the ensuing test */
251 }
252 }
253
254 if (optype != 0 && f->action == FIO_READ)
255 ERR(FIO_EREADONLY);
256 if (optype == 0 && f->action == FIO_WRITE)
257 ERR(FIO_EWRITEONLY);
258
259 /* error if attempt operation while positioned past end of file.
260 For write or endfile statements, ignore error for compatibility
261 with other Fortran compilers. */
262 if (f->eof_flag && optype == 0 /*read*/)
263 ERR(FIO_EEOFERR);
264 f->eof_flag = FALSE;
265
266 if (form != f->form && optype != 2)
267 ERR(FIO_EFORM);
268
269 if (f->acc == FIO_DIRECT) {
270 assert(f->reclen > 0);
271
272 if (!rec_specified || rec == 0)
273 /* since rec not specified, assume the next record */
274 rec = f->nextrec;
275 else if (rec < 1)
276 ERR(FIO_EDIRECT);
277 if (optype == 0 && rec > f->maxrec) {
278 seekoffx_t len;
279 seekoffx_t sav_pos;
280
281 sav_pos = __io_ftell(f->fp);
282 if (__io_fseek(f->fp, (seekoffx_t)0, SEEK_END) != 0)
283 ERR(__io_errno());
284 len = __io_ftell(f->fp);
285 f->partial = len % f->reclen;
286 if (form == FIO_UNFORMATTED && f->partial) {
287 /* allow read of partial record */
288 if (__io_fseek(f->fp, sav_pos, SEEK_SET) != 0)
289 ERR(__io_errno());
290 } else {
291
292 /* Add simple check to see if maxrec has been
293 changed by another process before bailing out */
294 f->maxrec = len / f->reclen;
295
296 /* Now check with recomputed maxrec */
297 if (rec > f->maxrec) {
298 f->nextrec = rec + 1; /* make error info come out correct */
299 ERR(FIO_EDREAD); /* read of non-existing record */
300 }
301
302 /* We recovered, so seek to the right point */
303 pos = f->reclen * (rec - 1);
304 if (__io_fseek(f->fp, (seekoffx_t)pos, SEEK_SET) != 0)
305 ERR(__io_errno());
306 f->coherent = 0;
307 }
308 }
309
310 if (f->nextrec != rec) {
311 /* FS 3662 Add simple check to see if maxrec has been
312 changed by another process before bailing out.
313 Certainly need to recompute it before bb is calculated
314 below, as multiple writers can cause a file to grow
315 enormously. If it has changed, and the record is not
316 past the end, we can just use the normal seek and reset
317 coherent section of code below.
318 */
319 if (rec > f->maxrec + 1) {
320 seekoffx_t len;
321 if (__io_fseek(f->fp, 0L, SEEK_END) != 0)
322 ERR(__io_errno());
323 len = __io_ftell(f->fp);
324 f->maxrec = len / f->reclen;
325 } /* Now go to next if-check with recomputed maxrec */
326
327 if (rec <= f->maxrec + 1) {
328 pos = f->reclen * (rec - 1);
329 if (__io_fseek(f->fp, (seekoffx_t)pos, SEEK_SET) != 0)
330 ERR(__io_errno());
331 f->coherent = 0;
332 } else {
333 /* pad with (rec-maxrec-1)*reclen bytes: */
334
335 seekoffx_t bb = (rec - f->maxrec - 1) * f->reclen;
336 /*
337 * It has been reported that extending the file by writing
338 * to the file is very slow when the number of bytes is
339 * large. Rather than use writes to pad the file, use fseek
340 * to extend the file 1 less the number of bytes and
341 * complete the padding by writing a single byte. A write
342 * is necessary after the fseek to ensure that the file's
343 * physical size is increased.
344 if (__io_fseek(f->fp, 0L, SEEK_END) != 0)
345 ERR(__io_errno());
346 errflag =
347 __fortio_zeropad(f->fp, (rec-f->maxrec-1) * f->reclen);
348 */
349 /* With multiple writers, there is a chance that
350 this could clobber a byte of data, but a very
351 small chance now that we've added the
352 recomputation of f->maxrec above
353 */
354 if (__io_fseek(f->fp, (seekoffx_t)(bb - 1), SEEK_END) != 0)
355 ERR(__io_errno());
356 errflag = __fortio_zeropad(f->fp, 1);
357 if (errflag != 0)
358 ERR(errflag);
359 f->coherent = 1;
360 }
361 }
362 if (optype == 0 /*read*/ && form == FIO_FORMATTED)
363 f->nextrec = rec;
364 else
365 f->nextrec = rec + 1;
366
367 if (rec > f->maxrec)
368 f->maxrec = rec;
369 }
370 }
371
372 /* for write into sequential file, may need to truncate file: */
373
374 assert(f->form == FIO_FORMATTED || f->form == FIO_UNFORMATTED);
375 assert(f->acc == FIO_DIRECT || f->acc == FIO_SEQUENTIAL ||
376 f->acc == FIO_STREAM);
377 if (f->acc == FIO_SEQUENTIAL) {
378 if (f->form == FIO_UNFORMATTED)
379 f->skip = 0;
380 if (rec_specified)
381 ERR(FIO_ECOMPAT);
382 if (optype != 0 && f->truncflag) {
383 pos = __io_ftell(f->fp);
384 if (__io_fseek(f->fp, 0L, SEEK_END) != 0)
385 ERR(__io_errno());
386 f->coherent = 0;
387 /* if not currently positioned at end of file, need to trunc: */
388 if (pos != __io_ftell(f->fp)) {
389 if (__io_fseek(f->fp, (seekoffx_t)pos, SEEK_SET) != 0)
390 ERR(__io_errno());
391 errflag = __fortio_trunc(f, pos);
392 if (errflag != 0)
393 return NULL;
394 }
395 f->truncflag = FALSE;
396 }
397 if (optype == 0 /*read*/) {
398 if (f->ispipe) {
399 f->truncflag = FALSE;
400 f->nextrec = 1;
401 if (f->coherent == 1)
402 /* last operation was a write */
403 fflush(f->fp);
404 f->coherent = 0;
405 f->skip = 0;
406 return f;
407 }
408 f->truncflag = TRUE;
409 } else
410 f->nextrec++; /* endfile or write */
411 } else /* FIO_DIRECT */
412 f->skip = 0;
413
414 /* coherent flag of fcb indicates how the file was last accessed
415 (read/write). If an operation occurs which is not identical with
416 the previous operation, the file's buffer must be
417 flushed. NOTE: coherent is set to 0 by open and rewind. */
418
419 if (optype != 2) {
420 if (f->coherent && (f->coherent != 2 - optype)) {
421 (void)__io_fseek(f->fp, 0L, SEEK_CUR);
422 f->skip = 0;
423 }
424 f->coherent = 2 - optype; /* write ==> 1, read ==> 2*/
425 } else
426 f->skip = 0;
427
428 return f;
429}

449extern int
450__fortio_zeropad(FILE *fp, long len)
451{
452#define BFSZ 512L
453 static struct { /* (double aligned buff may be faster) */
454 char b[BFSZ];
455 double dalign;
456 } b = {{0}, 0};
457
458 while (len >= BFSZ) {
459 if (FWRITE(b.b, BFSZ, 1, fp) != 1)
460 return __io_errno();
461 len -= BFSZ;
462 }
463
464 if (len > 0)
465 if (FWRITE(b.b, len, 1, fp) != 1)
466 return __io_errno();
467
468 return 0;
469}

473/* return TRUE if string 'str' of length 'len' is equal to 'pattern'. */
474extern bool __fortio_eq_str(
475 char *str, /* user-specified string, not null terminated */
476 __CLEN_T len, /* maximum number of characters to comprae */
477 const char *pattern) /* upper case, null terminated string */
478{
479 char c1, c2;
480
481 if (str == NULL || len <= 0)
482 return FALSE;
483
484 while (1) {
485 c1 = *str++;
486 c2 = *pattern++;
487
488 if (len == 0)
489 break;
490 len--;
491
492 if (c1 >= 'a' && c1 <= 'z') /* convert to upper case */
493 c1 = c1 + ('A' - 'a');
494
495 if (c2 == '\0' || c1 != c2)
496 break;
497 }
498
499 if (c2 != 0)
500 return FALSE;
501
502 if (len == 0)
503 return TRUE;
504
505 /* verify that remaining characters of str are blank: */
506
507 while (len--)
508 if (*str++ != ' ')
509 return FALSE;
510
511 return TRUE;
512}

518#define SWAPB(p, b1, b2, tmp) \
519 { \
520 tmp = p[b1]; \
521 p[b1] = p[b2]; \
522 p[b2] = tmp; \
523 }
524
525void __fortio_swap_bytes(
526 /*
527 * swap bytes where the value located by p is in the wrong endian order.
528 * swapping is endian-independent.
529 */
530 char *p, /* locates first byte of items */
531 int type, /* data type of item */
532 long cnt) /* number of 'unit_sz' items to be swapped */
533{
534 char btmp;
535 int unit_sz; /* basic size of item to be swapped */
536
537 switch (type) {
538 case __STR:
539 return;
540 case __CPLX8:
541 unit_sz = FIO_TYPE_SIZE(__REAL4);
542 cnt <<= 1;
543 break;
544 case __CPLX16:
545 unit_sz = FIO_TYPE_SIZE(__REAL8);
546 cnt <<= 1;
547 break;
548 case __CPLX32:
549 unit_sz = FIO_TYPE_SIZE(__REAL16);
550 cnt <<= 1;
551 break;
552 default:
553 unit_sz = FIO_TYPE_SIZE(type);
554 break;
555 }
556 while (cnt--) {
557 switch (unit_sz) {
558 case 1: /* byte */
559 return;
560 case 2: /* half-word */
561 SWAPB(p, 0, 1, btmp);
562 break;
563 case 4: /* word */
564 SWAPB(p, 0, 3, btmp);
565 SWAPB(p, 1, 2, btmp);
566 break;
567 case 8: /* double-word */
568 SWAPB(p, 0, 7, btmp);
569 SWAPB(p, 1, 6, btmp);
570 SWAPB(p, 2, 5, btmp);
571 SWAPB(p, 3, 4, btmp);
572 break;
573 default: /* error */
574 assert(0);
575 return;
576 }
577 p += unit_sz;
578 }
579}

584__fortio_trunc(FIO_FCB *p, seekoffx_t length)
585{

590 /*

597 p->eof_flag = FALSE;
598 }
