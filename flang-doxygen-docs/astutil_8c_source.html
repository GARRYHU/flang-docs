104static void
105proc_sym()
106{
107 int len;
108 int i;
109 int cursyms[20], cursym;
110 FILE *outf;
111 FILE *tempfp;
112 int pdoffs;
113
114 tempfp = NULL;
115 outf = OUT1;
116 lt = get_line1(IN1, elt, outf);
117 fieldnum = 0;
118 symnum = 0;
119 cursym = 0;
120 pdoffs = 0;
121 LOOP
122 { /* once for each line */
123 switch (lt) {
124 case LT_SF:
125 /* shared field */
126 get_token(tok, &len); /* field name */
127 if (len <= 1)
128 break;
129 (void)addfield(TRUE, FALSE);
130 break;
131
132 case LT_ST:
133 /* ast type (used to be symbol type) */
134 get_token(tok, &len);
135 if (len <= 1)
136 break;
137 (void)findsym();
138 break;
139
140 case LT_SM:
141 /* ast type-specific fields */
142 get_token(tok, &len);
143 flushsym(cursyms, cursym, tempfp); /* flush info for current sym */
144 cursym = 0;
145 if (len == 1) {
146 /* end of SM definitions */
147 outf = OUT1;
148 goto again;
149 }
150 while (len > 0) {
151 cursyms[cursym++] = findsym();
152 get_token(tok, &len);
153 }
154 /* open file to hold up to next SM */
155 if ((tempfp = fopen(ASTTMPFILE, "wb+")) == NULL) {
156 put_error(4, "Can't open temp file ASTTMPFILE");
157 }
158 outf = tempfp;
159 flush_line(OUT1);
160 goto again;
161
162 case LT_SI:
163 get_token(tok, &len);
164 i = cursyms[0];
165 strncpy(symbols[i].sname, tok, 31);
166 symbols[i].sname[31] = 0;
167
168 get_token(tok, &len);
169 while (len > 0) {
170 int jj;
171 jj = addattr();
172 symbols[i].attr[symbols[i].nattr++] = jj;
173 get_token(tok, &len);
174 }
175 break;
176
177 case LT_SE:
178 get_token(tok, &len);
179 i = addfield(FALSE, FALSE);
180 addfieldtosym(cursyms, cursym, i);
181 break;
182
183 case LT_FL:
184 get_token(tok, &len);
185 i = addfield(FALSE, TRUE);
186 addfieldtosym(cursyms, cursym, i);
187 break;
188
189 case LT_EOF:
190 goto done;
191 default:
192 put_err(4, "Unknown LT: can't happen\n");
193 }
194 flush_line(outf);
195 again:
196 lt = get_line1(IN1, elt, outf);
197 }
198done:;
199}

201static void
202write_ast()
203{
204 int fptrs[MAXFIELDS];
205 int i, j;
206 int k;
207 char buff[32];
208 char buff1[32];
209
210 for (i = 0; i < fieldnum; ++i)
211 fptrs[i] = i;
212 qsort((char *)fptrs, fieldnum, sizeof(int), qscmp);
213 /* read the ast.h boilerplate file and write stuff */
214 lt = get_line1(IN2, elt, OUT2);
215 LOOP
216 {
217 switch (lt) {
218 case LT_ST: /* print symbol types */
219 for (i = 0; i < symnum; ++i)
220 fprintf(OUT2, "#define A_%s %d\n", symbols[i].stype, i);
221 fprintf(OUT2, "#define AST_MAX %d\n", symnum - 1);
222 fprintf(OUT2, "\n");
223 break;
224 case LT_SE: /* print fields access macros */
225 for (i = 0; i < fieldnum; ++i) {
226 j = fptrs[i];
227 if (fields[j].flag)
228 sprintf(buff, "f%d", fields[j].offs);
229 else if (fields[j].shared) {
230 if (strcmp("flags", fields[j].name) == 0)
231 continue;
232 strcpy(buff1, fields[j].name);
233 for (k = 0; buff1[k] != '\0'; ++k)
234 if (isupper(buff1[k]))
235 buff1[k] = tolower(buff1[k]);
236 sprintf(buff, "%s", buff1);
237 } else {
238 switch (fields[j].size) {
239 case 1:
240 put_error(2, "byte field not supported");
241 sprintf(buff, "b%d", fields[j].offs + 1);
242 break;
243 case 2:
244 sprintf(buff, "hw%d", fields[j].offs / 2 + 1);
245 break;
246 case 4:
247 sprintf(buff, "w%d", fields[j].offs / 4 + 1);
248 break;
249 default:
250 put_err(2, "Field not b,h,w in macro");
251 strcpy(buff, "w0");
252 break;
253 }
254 }
255 if (!checkmode || fields[j].shared) {
256 fprintf(OUT2, "#define A_%sG(s) astb.stg_base[s].%s\n", fields[j].name,
257 buff);
258 fprintf(OUT2, "#define A_%sP(s,v) (astb.stg_base[s].%s = (v))\n",
259 fields[j].name, buff);
260 } else {
261 /* output code to check that the type is ok for this access */
262 fprintf(OUT2, "#define A_%sG(s) "
263 "(astb.stg_base[(ast_type_check[astb.stg_base[s].type][%d]?("
264 "interr(\"bad A_%sG access, "
265 "A_TYPE=\",astb.stg_base[s].type,3),0):0),(s)].%s)\n",
266 fields[j].name, i, fields[j].name, buff);
267 fprintf(OUT2, "#define A_%sP(s,v) "
268 "(astb.stg_base[(ast_type_check[astb.stg_base[s].type][%d]?("
269 "interr(\"bad A_%sP access, "
270 "A_TYPE=\",astb.stg_base[s].type,3),0):0),(s)].%s = (v))\n",
271 fields[j].name, i, fields[j].name, buff);
272 }
273 }
274 if (checkmode) {
275 fprintf(OUT2, "extern char ast_type_check[AST_MAX+1][%d];\n", fieldnum);
276 }
277 break;
278 case LT_TA: /* print type attributes */
279 for (i = 0; i < attrnum; ++i)
280 fprintf(OUT2, "#define __A_%s %d\n", attrnames[i], 1 << i);
281 fprintf(OUT2, "\n");
282 break;
283 case LT_EOF:
284 goto done;
285 default:
286 put_error(2, "Unknown line type");
287 break;
288 }
289 lt = get_line1(IN2, elt, OUT2);
290 }
291done:;
292 copy_file(IN4, OUT3); /* copy intast_sym inits created by symtab */
293 /* write dinit for ASTB */
294 fprintf(OUT3, "\n#pragma GCC diagnostic push\n");
295 fprintf(OUT3, "#pragma GCC diagnostic ignored \"-Wmissing-field-initializers\"\n");
296 fprintf(OUT3, "ASTB astb = {\n");
297 fprintf(OUT3, " {");
298 /* char *atypes[AST_MAX + 1]; */
299 j = 6;
300 for (i = 0; i < symnum; ++i) {
301 if ((j += (k = strlen(symbols[i].sname) + 3)) > 80) {
302 fprintf(OUT3, "\n ");
303 j = 6 + k;
304 }
305 fprintf(OUT3, "\"%s\",", symbols[i].sname);
306 }
307 fprintf(OUT3, "},\n");
308
309 fprintf(OUT3, " {0,\n");
310 for (i = 1; i < symnum; ++i) {
311 k = symbols[i].nattr;
312 if (k == 0)
313 fprintf(OUT3, " 0,\n");
314 else {
315 fprintf(OUT3, " __A_%s", attrnames[symbols[i].attr[0]]);
316 for (j = 1; j < k; ++j)
317 fprintf(OUT3, "|__A_%s", attrnames[symbols[i].attr[j]]);
318 fprintf(OUT3, ",\n");
319 }
320 }
321 fprintf(OUT3, " },\n");
322 fprintf(OUT3, "};\n");
323 fprintf(OUT3, "#pragma GCC diagnostic pop\n");
324
325 if (checkmode) {
326 char ch;
327 int s, i, j, sf, x;
328 fprintf(OUT3, "\n\nchar ast_type_check[AST_MAX+1][%d] = {\n", fieldnum);
329 for (s = 0; s < symnum; ++s) {
330 ch = ' ';
331 for (i = 0; i < fieldnum; ++i) {
332 j = fptrs[i];
333 x = 1;
334 if (fields[j].shared) {
335 x = 0;
336 } else {
337 for (sf = 0; sf < symbols[s].nfields; ++sf) {
338 if (symbols[s].fields[sf] == j)
339 break;
340 }
341 if (sf < symbols[s].nfields) {
342 x = 0;
343 }
344 }
345 fprintf(OUT3, "%c%d", ch, x);
346 ch = ',';
347 }
348 if (s < symnum - 1) {
349 fprintf(OUT3, ", /* A_%s */\n", symbols[s].stype);
350 } else {
351 fprintf(OUT3, "}; /* A_%s */\n", symbols[s].stype);
352 }
353 }
354 for (i = 0; i < fieldnum; ++i) {
355 j = fptrs[i];
356 fprintf(OUT3, "/* field %2d = A_%sG */\n", i, fields[j].name);
357 }
358 }
359}

361/*
362 * I made sure the sort is stable, so it will give the same
363 * results regardless of the host machine. It was returning
364 * '0' (no order) for fields with the same offset, which different
365 * qsort implementations would order differently, so the ast.h file
366 * would get updated for no good reason; my solution was to
367 * compare the field names in that case
368 */
369static int
370qscmp(const void *a1, const void *a2)
371{
372 int r;
373 int *f1, *f2;
374 f1 = (int *)a1;
375 f2 = (int *)a2;
376 if (fields[*f1].flag && fields[*f2].flag) {
377 r = fields[*f1].offs - fields[*f2].offs;
378 if (r == 0)
379 r = strcmp(fields[*f1].name, fields[*f2].name);
380 } else if (fields[*f1].flag)
381 r = -1;
382 else if (fields[*f2].flag)
383 r = 1;
384 else {
385 r = fields[*f1].offs - fields[*f2].offs;
386 if (r == 0)
387 r = strcmp(fields[*f1].name, fields[*f2].name);
388 }
389 return r;
390}

394static void
395flushsym(int *cursyms, int cursym, FILE *tempf)
396{
397 int i, j, k;
398 int indir;
399 int addit;
400 int *p;
401 int offs;
402 int output;
403 int last;
404
405 if (cursym == 0 || tempf == NULL)
406 return;
407 /* add shared fields not already present */
408 for (i = 0; i < cursym; ++i) {
409 indir = cursyms[i];
410 for (j = 0; j < fieldnum; ++j) {
411 if (!fields[j].shared)
412 continue;
413 addit = TRUE;
414 for (k = 0; k < symbols[indir].nfields; ++k)
415 if (chk_overlap(j, symbols[indir].fields[k], FALSE)) {
416 addit = FALSE;
417 break;
418 }
419 if (addit)
420 symbols[indir].fields[symbols[indir].nfields++] = j;
421 }
422 qsort((char *)symbols[indir].fields, symbols[indir].nfields, sizeof(int),
423 (int (*)())qscmp);
424 }
425
426 /* write symbol picture */
427 indir = cursyms[0];
428 fputs(".sz -4\n", OUT1);
429 fputs(".TS\n", OUT1);
430 fputs("tab(%);\n", OUT1);
431 j = 0;
432 p = symbols[indir].fields;
433 k = symbols[indir].nfields;
434 while (j < k && fields[*p].flag) {
435 ++j;
436 ++p;
437 }
438 offs = 0;
439 last = 1;
440 /* write the table format lines */
441 for (i = 0; i < SYMLEN; ++i) {
442 fputs("n cw(1.0i) sw(1.0i) sw(1.0i) sw(1.0i)\n", OUT1);
443 fputs("n ", OUT1);
444 output = 0;
445 last = 1;
446 while (output < 4) {
447 if (j < k && fields[*p].offs == offs) {
448 switch (fields[*p].size) {
449 case 1:
450 fputs("| cw(1.0i) ", OUT1);
451 break;
452 case 2:
453 fputs("| cw(1.0i) sw(1.0i) ", OUT1);
454 break;
455 case 3:
456 fputs("| cw(1.0i) sw(1.0i) sw(1.0i) ", OUT1);
457 break;
458 case 4:
459 fputs("| cw(1.0i) sw(1.0i) sw(1.0i) sw(1.0i) ", OUT1);
460 break;
461 default:
462 put_err(4, "Bad size in field");
463 return;
464 }
465 last = 1;
466 offs += fields[*p].size;
467 output += fields[*p].size;
468 ++j;
469 ++p;
470 } else {
471 if (last)
472 fputs("| ", OUT1);
473 fputs("cw(1.0i) ", OUT1);
474 ++output;
475 ++offs;
476 last = 0;
477 }
478 }
479 fputs("|\n", OUT1);
480 }
481 fputs("n cw(1.0i) sw(1.0i) sw(1.0i) sw(1.0i) .\n", OUT1);
482
483 j = 0;
484 p = symbols[indir].fields;
485 k = symbols[indir].nfields;
486 while (j < k && fields[*p].flag) {
487 ++j;
488 ++p;
489 }
490 offs = 0;
491 /* write the data lines */
492 for (i = 0; i < SYMLEN; ++i) {
493 fputs("%_\n", OUT1);
494 fprintf(OUT1, "%d", i + 1);
495 output = 0;
496 while (output < 4) {
497 if (j < k && fields[*p].offs == offs) {
498 fprintf(OUT1, "%%%s", fields[*p].name);
499 offs += fields[*p].size;
500 output += fields[*p].size;
501 ++j;
502 ++p;
503 } else {
504 fputc('%', OUT1);
505 ++output;
506 ++offs;
507 }
508 }
509 fputc('\n', OUT1);
510 }
511 fputs("%_\n", OUT1);
512 fputs(".TE\n", OUT1);
513 fputs(".sz +4\n", OUT1);
514
515 /* append temp file contents to troff output */
516 rewind(tempf);
517 {
518 char buffer[133];
519 buffer[132] = 0;
520 while (fgets(buffer, 132, tempf))
521 fputs(buffer, OUT1);
522 }
523
524 /* close temp file */
525 fclose(tempf);
526}

528static void
529addfieldtosym(int *cursyms, int cursym, int field)
530{
531 int i, indir, j, k;
532
533 for (i = 0; i < cursym; ++i) {
534 indir = cursyms[i]; /* symbol number */
535 for (j = 0; j < symbols[indir].nfields; ++j) {
536 if (field == (k = symbols[indir].fields[j])) {
537 put_err(2, "Field already specified for this sym");
538 goto again;
539 }
540 if (chk_overlap(field, k, TRUE))
541 goto again;
542 }
543 symbols[indir].fields[symbols[indir].nfields++] = field;
544 again:;
545 }
546}

548static int
549chk_overlap(int f1, int f2, int flag)
550{
551 if (fields[f1].flag && fields[f2].flag) {
552 if (fields[f1].offs == fields[f2].offs) {
553 if (flag)
554 put_err(2, "Flag overlaps one already defined");
555 return TRUE;
556 }
557 } else if (fields[f1].flag || fields[f2].flag)
558 return FALSE;
559 /* check for field overlap */
560 if (fields[f1].offs + fields[f1].size <= fields[f2].offs ||
561 fields[f2].offs + fields[f2].size <= fields[f1].offs)
562 return FALSE; /* they're disjoint */
563 if (flag)
564 put_err(2, "Field overlaps one already defined");
565 return TRUE;
566}

585static int
586addfield(int sharedflag, int flagflag)
587{
588 int i;
589 int size, offs;
590 int len;
591 char *aftp; /* position after w<d> */
592
593 for (i = 0; i < fieldnum; ++i)
594 if (strcmp(tok, fields[i].name) == 0)
595 return i;
596
597 /* add it */
598 if (fieldnum >= MAXFIELDS) {
599 put_error(2, "Too many symbol fields");
600 return 0;
601 }
602 strncpy(fields[fieldnum].name, tok, 31);
603 fields[fieldnum].name[31] = 0;
604 get_token(tok, &len);
605 if (len <= 0) {
606 put_error(2, "Field location not specified");
607 goto fixup;
608 }
609 /* parse location */
610 if (flagflag) {
611 offs = 0;
612 size = 0;
613 if (*tok != 'f')
614 goto badloc;
615 sscanf(tok + 1, "%d", &offs);
616 if (offs == 0)
617 goto badloc;
618 goto done;
619 }
620 if (*tok != 'w' || tok[1] < '1')
621 goto badloc;
622 offs = 0;
623 sscanf(tok + 1, "%d", &offs);
624 if (offs == 0 || offs > SYMLEN)
625 goto badloc;
626 aftp = tok + 2;
627 if (offs > 9)
628 aftp++;
629 offs = (offs - 1) * 4;
630 if (aftp[0] == 0) {
631 size = 4;
632 } else if (aftp[0] != ':') {
633 put_error(2, ": must follow word spec");
634 size = 4;
635 } else if (aftp[1] != 'h' && aftp[1] != 'b') {
636 put_error(2, "Bad subfield spec");
637 size = 4;
638 } else if (aftp[1] == 'h') {
639 size = 2;
640 if (aftp[2] < '1' || aftp[2] > '2')
641 put_error(2, "Bad halfword spec");
642 else
643 offs += (aftp[2] - '1') * 2;
644 } else if (aftp[1] == 'b') {
645 size = 1;
646 if (aftp[2] < '1' || aftp[2] > '4')
647 put_error(2, "Bad byte spec");
648 else
649 offs += aftp[2] - '1';
650 if (aftp[3] == '-') {
651 if (strcmp(fields[fieldnum].name, "flags") != 0 || aftp[4] < '1' ||
652 aftp[4] > '4')
653 put_error(2, "Bad flags spec");
654 else
655 size = (aftp[4] - aftp[2]) + 1;
656 }
657 } else {
658 size = 0;
659 }
660done:
661 fields[fieldnum].size = size;
662 fields[fieldnum].offs = offs;
663 fields[fieldnum].shared = sharedflag;
664 fields[fieldnum].flag = flagflag;
665 return fieldnum++;
666
667badloc:
668 put_error(2, "Bad field location");
669/**** fall thru ****/
670fixup:
671 fields[fieldnum].size = fields[fieldnum].offs = 0;
672 fields[fieldnum].shared = FALSE;
673 fields[fieldnum].flag = FALSE;
674 return fieldnum++;
675}
