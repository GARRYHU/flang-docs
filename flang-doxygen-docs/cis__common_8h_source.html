13/* Payne-Hanek style argument reduction. */
14static float
15__attribute__ ((noinline))
16//__attribute__ ((always_inline)) inline
17__reduction_slowpath_pio2(float const a, int32_t *h, float *w)
18{
19 uint2 m;
20 uint32_t ia = float_as_int(a);
21 uint32_t result[7];
22 uint32_t hi, lo;
23 uint32_t e;
24 int32_t idx;
25 int32_t q;
26
27 *w = 0.0f;
28 if ((unsigned)float_as_int(a) >= 0x7f800000) return a * 0.0f;
29
30 e = ((ia >> 23) & 0xff) - 127;
31 ia = (ia << 8) | 0x80000000;
32
33 /* compute x * 1/pi */
34 idx = 4 - ((e >> 5) & 3);
35
36 hi = 0;
37 for (q = 0; q < 6; q++) {
38 m = umad32wide(i1opi_f[q], ia, hi);
39 lo = m.x;
40 hi = m.y;
41 result[q] = lo;
42 }
43 result[q] = hi;
44
45 e = e & 31;
46 /* shift result such that hi:lo<63:63> is the least significant
47 integer bit, and hi:lo<62:0> are the fractional bits of the result
48 */
49
50 uint64_t p;
51 p = (uint64_t)result[idx + 2] << 32;
52 p |= result[idx + 1];
53 // 32-int right shift by >=32 is undefined, so need to
54 // promote to 64 bits first
55 p = (p << e) | ((uint64_t)(result[idx]) >> (32 - e));
56
57 /* fraction */
58 q = (p >> 32) & 0xE0000000; //3 selector bits
59
60 // eliminate r > pi/4 by subtracting r from pi/2
61 if (p & 0x2000000000000000ULL)
62 {
63 p = ~p;
64 }
65 // discard selector bits
66 p &= 0x1fffffffffffffffULL;
67
68 *h = q;
69 double d = (double)(int64_t)p;
70 d *= PI_2_M63;
71
72 float r = (float)d;
73 *w = (float)(d - (double)r);
74
75 return r;
76}

79static float _Complex __attribute__((always_inline)) inline
80__cis_scalar_kernel(float x)
81{
82 float p, k, r, s, t, w;
83 float x2, x3, ps, pc;
84 int h = 0;
85 int sign = 0, csign = 0, ssign = 0;
86 int selector = 0;
87
88 // p = |x|
89 p = I2F(F2I(x) & FL_ABS_MASK); PRINT(p);
90#undef THRESHOLD_F
91#define THRESHOLD_F 0x1.7c3e58p17
92#define DELTA 0.005
93 if (__builtin_expect(p > THRESHOLD_F, 0))
94 {
95 // huge arguments reduction
96 p = __reduction_slowpath_pio2(p, &h, &w); PRINT(p); PRINT(h); PRINT(w);
97 // h contains: selector bits for [pi, pi/2, extra swap]
98 ssign = h & FL_SIGN_BIT; PRINT(ssign);
99 ssign = ssign ^ (F2I(x) & FL_SIGN_BIT); PRINT(ssign);
100 csign = ((h + (1<<30)) & FL_SIGN_BIT); PRINT(csign);
101 selector = ((h << 1) + (h << 2)) >> 31; PRINT(selector);
102 } else {
103 // Here we follow the reduction process presented in
104 // [1] S. Boldo, M. Daumas & R-C. Li,
105 // Formally Verified Argument Reduction with a Fused-Multiply-Add,
106 // IEEE Transactions on Computers (IF: 1.680), 58(8), pp. 1139-1145, 2009.
107 // i.e. we compute (x - N*pi/2) using 3-double representation of
108 // pi/2 and obtain result as an unevaluated sum of two doubles: p + w.
109
110 // NOTE: this process becomes inaccurate because of the first step
111 // where we compute x / (pi/2) via multiplication of x by R,
112 // rounded 2/pi. So when x grows that rounding error in the 2/pi
113 // may propagate into integer part of the quotient and the resulting
114 // reduced argument will fall out of desired [-pi/4, pi/4] range.
115 // Here's what happens:
116 // R = 2/pi + eps
117 // where eps is an absolute error in the rounded constant.
118 // We may always mathematically represent x as
119 // x = N*pi/2 + m*pi/2
120 // with N being nearest integer and m < 1/2.
121 // Now the multiplication x*R results in:
122 // x*R = x*2/pi + x*eps = N + m + x*eps
123 // In order for this to round correctly to nearest integer N
124 // x*eps shall be small enough, so that it doesn't bump m over 1/2.
125 // Suppose m = 1/2 - delta, delta is positive since m < 1/2.
126 // x*eps must be less than delta in order for the rounding to be
127 // correct and result in N.
128 // eps is known to us by definition of R. Acceptable range for the
129 // input |x| < T is given to us by the paper [1]. So the error will
130 // appear when T*eps > delta.
131 // For T = pi/2 * (2^(p-2) - 1) and eps ~= 2/pi*2^(-p-1) we get
132 // T*eps ~= 1/8
133 // Now it is easy to find inputs around integer
134 // multiples of pi/16 that will have delta <= 1/8:
135 // e.g. N*pi/2 + 1/2*pi/2 - 1/8 * pi/2
136 // So what happens when the rounding goes wrong by 1? The reduction
137 // r = x - (N+1)*pi/2 = m*pi/2 - pi/2 = pi/2*(m - 1)
138 // = -pi/2*(1/2 + delta) = -pi/4*(1 + 2*delta)
139 // produces reduced argument r, which is off from the target interval
140 // proportionally to 2*delta. Approximation error for the minimax
141 // polynomial oscillates near the interval bounds and
142 // grows significantly outside the interval, so we may end up with
143 // computed results that will be off by too much.
144 // We may also notice that small delta doesn't actually pose the same
145 // threat as the large delta in terms of being off from the
146 // approximation interval, so we are not interested in finding
147 // the lower limit on delta in the given precision.
148 // Instead we accept that the problem with reduction comes from the
149 // arguments 'not too close' to multiples of pi/4. And the larger
150 // |x| is, the larger x*eps becomes and thus we need to accomodate
151 // for larger deltas. We may reduce the effect of the interval error
152 // by allowing our approximation interval to be wider by some
153 // margin, perhaps grow it until we need to add an extra degree to
154 // the polynomial.
155 // But we also do not want delta to be too large since the
156 // reduced argument may end up in a different binade and that would
157 // affect the polynomial evaluation error: instead of reducing
158 // to pi/4 < 2^0 we may end up with r >= 2^0.
159 // With bad delta capped by DELTA from the choice of
160 // approximation interval bounds:
161 // pi/4*(1 + 2*DELTA)
162 // we shall work from the other end and ensure x*eps < DELTA
163 // by lowering the threshold for |x| < min(T, DELTA/eps).
164
165#undef _1_OVER_PI_F
166#define _1_OVER_PI_F I2F(0x3ea2f983)
167 k = fmaf(p, _1_OVER_PI_F * 2.0f, 0x1.8p23); PRINT(k);
168 ssign = (F2I(k) & 0x2) << 30; PRINT(ssign);
169 ssign = ssign ^ (F2I(x) & FL_SIGN_BIT); PRINT(ssign);
170 csign = ((float_as_int(k) + 1) & 0x2) << 30; PRINT(csign);
171 //sign extend to obtain bitmask
172 selector = 0 - (float_as_int(k) & 0x1); PRINT(selector);
173 k -= 0x1.8p23; PRINT(k);
174#undef PI_HI_F
175#undef PI_MI_F
176#undef PI_LO_F
177#define PI_HI_F I2F(0x40490fdc)
178#define PI_MI_F I2F(0xb4aeef48)
179#define PI_LO_F I2F(0xa9e7b967)
180
181 // first reduction step
182 float u = fmaf(k, -PI_HI_F/2.0f, p); PRINT(u);
183 // second reduction step
184 float v1 = fmaf(k, -PI_MI_F/2.0f, u); PRINT(v1);
185 float p1, p2;
186 fast2mul(k, -PI_MI_F/2.0f, &p1, &p2); PRINT(p1); PRINT(p2);
187 float t1, t2;
188 fast2sum(u, p1, &t1, &t2); PRINT(t1); PRINT(t2);
189 float v2 = (t1 - v1);
190 v2 = v2 + t2;
191 v2 = v2 + p2; PRINT(v1); PRINT(v2);
192 // third reduction step
193 w = fmaf(k, -PI_LO_F/2.0f, v2); PRINT(w);
194 // now v1 + w give us the reduced argument in the double-float form
195 // redistribute bits between high and low parts
196 fast2sum(v1, w, &p, &w); PRINT(p); PRINT(w);
197 }
198
199 // assume that the range reduction produced argument
200 // suitable for further minimax evaluation or NaN
201 assert( (p != p) || ((fabsf(p) / (PI_HI_F/4.0f) - 1.0f) < 2.0f*DELTA) );
202
203 x2 = fmaf(p,p, (2.0f*w*p)); PRINT(x2);
204
205// 2^(-26), interval: [+-pi/4*(1 + 2*delta)], delta=0.005
206#define S1 I2F(0x3f800000)
207#define S3 I2F(0xbe2aaaa9)
208#define S5 I2F(0x3c0885e1)
209#define S7 I2F(0xb94d4ec3)
210
211// 2^(-31), interval: [+-pi/4*(1 + 2*delta)], delta=0.005
212#define C0 I2F(0x3f800000)
213#define C2 I2F(0xbf000000)
214#define C4 I2F(0x3d2aaaa7)
215#define C6 I2F(0xbab60727)
216#define C8 I2F(0x37cd39b8)
217
218 ps = S7;
219 ps = fmaf(ps, x2, S5);
220 ps = fmaf(ps, x2, S3);
221 x3 = p * x2;
222 ps = fmaf(ps, x3, w); PRINT(ps);
223 ps = ps + p; PRINT(ps);
224
225 pc = C8;
226 pc = fmaf(pc, x2, C6);
227 pc = fmaf(pc, x2, C4);
228 pc = fmaf(pc, x2, C2);
229 pc = fmaf(pc, x2, C0); PRINT(pc);
230
231 float rsin, rcos;
232 rsin = I2F((selector & F2I(pc)) | ((~selector) & F2I(ps))); PRINT(rsin);
233 rsin = I2F(F2I(rsin) ^ ssign); PRINT(rsin);
234 rcos = I2F((selector & F2I(ps)) | ((~selector) & F2I(pc))); PRINT(rcos);
235 rcos = I2F(F2I(rcos) ^ csign); PRINT(rcos);
236
237 return set_cmplx(rcos, rsin);
238}

240static vfloat __attribute__((noinline)) __vcis_slowpath(vfloat x, vopmask m)
241{
242 vfloat in = x;
243 vfloat res = vSETf(0.0f);
244 int i;
245 const int vlen = sizeof(vfloat) / sizeof(float _Complex);
246 for (i = 0; i < vlen; i++)
247 {
248 // only work on even elements as odd ones are duplicates
249 float a = *(2*i + (float *)(&in));
250 // FIXME: here it might be more robust if we read the mask and decide,
251 // instead of duplicating the real condition
252 if (__builtin_expect(!(I2F(F2I(a) & FL_ABS_MASK) > THRESHOLD_F), 0)) continue;
253 *(i + (float _Complex *)(&res)) = __cis_scalar_kernel(a);
254 }
255 return res;
256}

258//static vfloat __attribute((noinline))
259static vfloat INLINE
260__vcis_kernel(vfloat va)
261{
262 // This function will compute cos and sin in parallel
263 // over the SIMD register filled with duplicate input
264 // values: x x y y z z, etc
265 // Will use the same range reduction for every value.
266 // If we need to take huge reduction slow path, then
267 // we will recall about duplicates.
268
269 const vfloat vRS = vSETf(0x1.8p23);
270 vfloat vabsa = vI2F(vand_vi2_vi2_vi2(vF2I(va), vSETi(FL_ABS_MASK))); PRINT(vabsa);
271 vfloat k = vfma_vf_vf_vf_vf(vabsa, vSETf(_1_OVER_PI_F * 2.0f), vRS); PRINT(k);
272 // Need k in sin positions and k+1 in cos positions
273 const vint2 _0_1 = vSETLLi(0x00000000, 0x00000001);
274 // cssign merged sign
275 vint2 cssign = vand_vi2_vi2_vi2(vadd_vi2_vi2_vi2(vF2I(k), _0_1), vSETi(0x00000002));
276 cssign = vsll_vi2_vi2_i(cssign, 30);
277 cssign = vxor_vi2_vi2_vi2(cssign, vand_vi2_vi2_vi2(vF2I(va), vSETLLi(FL_SIGN_BIT, 0x00000000)));
278 // parity bit of k is the selector
279 vopmask selector = veq_vo_vi2_vi2(vand_vi2_vi2_vi2(vF2I(k), vSETi(1)), vSETi(1));
280 // k rounded to integer, now in normalized FP format
281 k = vsub_vf_vf_vf(k, vRS); PRINT(k);
282 // first reduction step
283 vfloat u = vfma_vf_vf_vf_vf(k, vSETf(-PI_HI_F/2.0f), vabsa); PRINT(u);
284 // second reduction step
285 vfloat v1 = vfma_vf_vf_vf_vf(k, vSETf(-PI_MI_F/2.0f), u); PRINT(v1);
286 vfloat p1, p2;
287 vfast2mul(k, vSETf(-PI_MI_F/2.0f), &p1, &p2); PRINT(p1); PRINT(p2);
288 vfloat t1, t2;
289 vfast2sum(u, p1, &t1, &t2);
290 vfloat v2 = vsub_vf_vf_vf(t1, v1);
291 v2 = vadd_vf_vf_vf(v2, t2);
292 v2 = vadd_vf_vf_vf(v2, p2);
293 // third reduction step
294 vfloat w = vfma_vf_vf_vf_vf(k, vSETf(-PI_LO_F/2.0f), v2);
295 vfloat p;
296 // now v1 + w give us the reduced argument in the double-float form
297 // redistribute bits between high and low parts
298 vfast2sum(v1, w, &p, &w);
299 vfloat x2 = vfma_vf_vf_vf_vf(p, p, vmul_vf_vf_vf(vmul_vf_vf_vf(p, w), vSETf(2.0f)));
300
301 // compute polynomials: minimax on +-pi/4.0*1.01
302 //sin coef, cos coef
303 vfloat c8s7 = vI2F(vSETLLi(F2I(S7), F2I(C8)));
304 vfloat c6s5 = vI2F(vSETLLi(F2I(S5), F2I(C6)));
305 vfloat c4s3 = vI2F(vSETLLi(F2I(S3), F2I(C4)));
306
307 vfloat x2_0 = vI2F(vand_vi2_vi2_vi2(vF2I(x2),vSETLLi(0x00000000, 0xffffffff)));
308 vfloat x2_1 = vI2F(vor_vi2_vi2_vi2(vF2I(x2_0),vSETLLi(FL_ONE, 0x00000000)));
309 vfloat _0p = vI2F(vand_vi2_vi2_vi2(vF2I(p), vSETLLi(0xffffffff, 0x00000000)));
310 vfloat _0w = vI2F(vand_vi2_vi2_vi2(vF2I(w), vSETLLi(0xffffffff, 0x00000000)));
311 vfloat _1p = vI2F(vor_vi2_vi2_vi2(vF2I(_0p),vSETLLi(0x00000000, FL_ONE)));
312 vfloat c2w = vI2F(vor_vi2_vi2_vi2(vF2I(_0w),vSETLLi(0x00000000, F2I(C2))));
313 vfloat x2x3 = vmul_vf_vf_vf(x2, _1p);
314
315 vfloat pcps = c8s7;
316 pcps = vfma_vf_vf_vf_vf(pcps, x2, c6s5);
317 pcps = vfma_vf_vf_vf_vf(pcps, x2, c4s3);
318 pcps = vfma_vf_vf_vf_vf(pcps, x2x3, c2w);
319 pcps = vfma_vf_vf_vf_vf(pcps, x2_1, _1p);
320
321 // swap cos and sin in SIMD register
322 vfloat pspc = vrev21_vf_vf(pcps);
323 // select cos or sin
324 vfloat vres = vsel_vf_vo_vf_vf(selector, pspc, pcps);
325 // fixup sign
326 vres = vI2F(vxor_vi2_vi2_vi2(vF2I(vres), cssign)); PRINT(vres);
327
328 vopmask m = vgt_vo_vf_vf(vabsa, vSETf(THRESHOLD_F)); PRINT(m);
329 if (__builtin_expect(!vtestz_i_vo(m), 0))
330 {
331 // blend slow and fast path results
332 vfloat vslowfres = __vcis_slowpath(va, m);
333 vres = vsel_vf_vo_vf_vf(m, vslowfres, vres);
334 }
335
336 return vres;
337}
