74static void
75__llcrit_atfork(void)
76{
77 is_init_nest = 0;
78 is_init_nest_red = 0;
79 /* The atfork handlers are inherited by the sub-processes,
80 * see https://elias.rhi.hi.is/libc/Threads-and-Fork.html
81 */
82 if (!is_atfork_registered)
83 fprintf(__io_stderr(), "The atfork not registered when it should be!\n");

86void
87_mp_bcs_nest(void)
88{
89 if (!is_init_nest) {
90 _mp_p(&nest_sem);
91 if (!is_init_nest) {
92#ifndef _WIN64
93 if (!is_atfork_registered) {
94 if (pthread_atfork(NULL, NULL, __llcrit_atfork))
95 fprintf(__io_stderr(), "Could not register atfork handler!\n");
96 else
97 is_atfork_registered = 1;
98 }
99#endif
100 omp_init_nest_lock(&nest_lock);
101 is_init_nest = 1;
102 }
103 _mp_v(&nest_sem);
104 }
105 omp_set_nest_lock(&nest_lock);
106}

178/* C/C++: copy a private stack or other other variable */
179void
180_mp_copypriv(char *adr, long len, int thread)
181{
182 if (thread == 0) {
183 singadr = adr;
184 singlen = len;
185 }
186 __kmpc_barrier(0, __kmpc_global_thread_num(0));
187 if (thread)
188 memcpy(adr, singadr, singlen);
189 __kmpc_barrier(0, __kmpc_global_thread_num(0));
190}

313/* C++: copy data from the master's block to the other threads blocks
314 using the assignment operator
315 vector_size is 1 for non arrays
316 n for array[n]
317 * Don't use: keep for backward compatibility
318 */
319
320void
321_mp_copyin_move_cpp(void *blk_tp, int off, int class_size, int vector_size,
322 assign_func_ptr assign_op)
323{
324 int lcpu;
325 char *to, *fr;
326 char *garbage = 0;
327 int i;
328
329 lcpu = __kmpc_global_thread_num(0);
330
331 __kmpc_barrier(0, lcpu);
332 if (lcpu != 0) {
333 fr = __kmpc_threadprivate_cached(
334 0, 0, garbage, (size_t)(class_size * vector_size), blk_tp);
335 to = __kmpc_threadprivate_cached(
336 0, lcpu, garbage, (size_t)(class_size * vector_size), blk_tp);
337
338 for (i = 0; i < vector_size; i++) {
339 if (to != fr)
340 (*assign_op)(to, fr);
341 to += class_size;
342 fr += class_size;
343 }
344 }
345 __kmpc_barrier(0, lcpu);
346}

403_mp_copyin_move_multiple(int n_entries, void *data)
404{

450/* Handler for __kmpc_copyprivate 'cpy_func'
451 * See how we marshall data in make_copypriv_array() in expsmp.c.
452 */
453void
454_mp_copypriv_kmpc(void *dest, void *src)
455{
456 struct pair_t {
457 size_t *size;
458 void *data;
459 };
460 const struct pair_t *to = (struct pair_t *)dest;
461 const struct pair_t *from = (struct pair_t *)src;
462
463 for (; from->size; ++from, ++to) {
464 if (to->data != from->data)
465 memcpy(to->data, from->data, *from->size);
466 }
467}
