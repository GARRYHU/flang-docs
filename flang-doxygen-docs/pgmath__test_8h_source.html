304static void
305build_masks(bool gray_code)
306{
307 int32_t i;
308 int32_t j;
309 int32_t k;
310
311 if (verbose) {
312 printf("%s: %s mask vectors\n",
313 __func__, gray_code ? "Gray code" : "binary");
314 }
315
316 memset(mask_sp, 0, sizeof mask_sp);
317 memset(mask_dp, 0, sizeof mask_dp);
318
319
320 for (j = 0; j < 1<<VLD; j++) {
321 k = gray_code ? j ^ (j>>1) : j;
322 for (i = 0; i < VLD; i++) {
323 mask_dp[j][i] = (k&0x1) * -1;
324 k = k>>1;
325 }
326 }
327
328 for (j = 0; j < 1<<VLS; j++) {
329 k = gray_code ? j ^ (j>>1) : j;
330 for (i = 0; i < VLS; i++) {
331 mask_sp[j][i] = (k&0x1) * -1;
332 k = k>>1;
333 }
334 }
335
336 if (verbose) {
337 for (j = 0; j < 1<<VLD; j++) {
338 for (i = 0; i < VLD; i++) {
339 printf(" %2lld", mask_dp[j][i]);
340 }
341 puts("");
342 }
343 for (j = 0; j < 1<<VLS; j++) {
344 for (i = 0; i < VLS; i++) {
345 printf(" %2d", mask_sp[j][i]);
346 }
347 puts("");
348 }
349 }
350
351 return;
352
353}

356static int
357checkfltol1(float res, float exp, float ltol)
358{
359 int tests_passed = 0;
360 int tests_failed = 0;
361
362 if (exp == res) {
363 tests_passed ++;
364 }else if( exp != 0.0 && (fabsf((exp-res)/exp)) <= ltol ){
365 tests_passed ++;
366 }else if( exp == 0.0 && exp <= ltol ){
367 tests_passed ++;
368 } else {
369 tests_failed ++;
370 if (verbose) {
371 printf("test FAILED. res %f exp %f\n", res, exp);
372 }
373 }
374
375 if (verbose) {
376 if (tests_failed == 0) {
377 printf("1 test completed. %d tests PASSED. %d tests failed.\n",
378 tests_passed, tests_failed);
379 } else {
380 printf("1 test completed. %d tests passed. %d tests FAILED.\n",
381 tests_passed, tests_failed);
382 }
383 }
384
385 return(tests_failed);
386}
