34 N2Rst(const std::vector<std::string> &args) : verbose(false)

36 for (std::vector<std::string>::const_iterator it = args.begin(),

47 /**
48 \brief Implements the logic of the program.
49 */
50 int
51 run()
52 {
53 if (input_filenames.empty()) {
54 std::cerr << "No input files." << std::endl;
55 return 1;
56 }
57 for (std::vector<std::string>::const_iterator it = input_filenames.begin(),
58 E = input_filenames.end();
59 it != E; ++it) {
60 std::ifstream ifs(it->c_str());
61 if (!ifs) {
62 std::cerr << "Can't open file " << *it << std::endl;
63 return 1;
64 }
65 if (verbose) {
66 std::cout << *it << " -> " << get_output_filename(*it) << std::endl;
67 }
68 sphinx.setOutputFile(get_output_filename(*it));
69 for (std::string line; std::getline(ifs, line);) {
70 sphinx.process(line);
71 }
72 }
73 return 0;
74 }
75
76private:
77 /**
78 \brief Generate the name of an output file given the name of an input file.
79 */
80 std::string
81 get_output_filename(const std::string &s)
82 {
83 auto pos = s.find_last_of('/');
84 std::string filename = pos != std::string::npos ? s.substr(pos + 1) : s;
85 pos = filename.find_last_of('.');
86 if (pos != std::string::npos) {
87 return filename.substr(0, pos + 1) + "rst";
88 }
89 return filename + ".rst";
90 }
