85void
86addcombination()
87/* see if this combination of operands has been seen before */
88{
89 int i, l, lc;
90 char c[MAXSTRING];
91 if( numoperands == 0 ) return;
92 strcpy( c, operand[0] );
93 lc = strlen( c );
94 for( i = 1; i < numoperands; ++i ){
95 l = strlen( operand[i] );
96 if( lc+l >= MAXSTRING ){
97 fprintf( stderr, "line %d: too many operands; increase MAXSTRING\n",
98 linenum );
99 exit( 1 );
100 }
101 lc += l;
102 strcat( c, operand[i] );
103 }
104 for( i = 0; i < combinations; ++i ){
105 if( numoperands != combolen[i] ) continue;
106 if( strcmp( c, combo[i] ) == 0 ) return; /* found it! */
107 }
108 /* didn't find it */
109 ++combinations;
110 if( combinations >= MAXCOMBO ){
111 fprintf( stderr, "line %d: too many operand combinations; increase MAXCOMBO\n",
112 linenum );
113 exit( 1 );
114 }
115 combolen[combinations] = numoperands;
116 combo[combinations] = (char*)malloc( lc + 1 );
117 strcpy( combo[combinations], c );
118
119 fprintf( outfile, "int OP%s[] = {", c );
120 for( i = 0; i < numoperands; ++i ){
121 if( i ){
122 fprintf( outfile, "," );
123 }
124 fprintf( outfile, " p%s", operand[i] );
125 }
126 fprintf( outfile, " };\n" );
127 ++combinations;
128}/* addcombination */
