25/** \brief Initialize the scope stack.
26
27 It starts with one frame representing the outer scope,
28 with no associated symbol.
29 */
30void
31scopestack_init()
32{
33 SCOPESTACK *scope;
34 if (sem.scope_stack == NULL) {
35 sem.scope_size = 10;
36 NEW(sem.scope_stack, SCOPESTACK, sem.scope_size);
37 }
38 sem.scope_level = 0;
39 sem.next_unnamed_scope = 6; // probably some arbitrary small value > 1 or 2
40 scope = curr_scope();
41 BZERO(scope, SCOPESTACK, 1);
42 scope->kind = SCOPE_OUTER;
43 scope->closed = true;
44 scope->symavl = stb.stg_avail;
45}

54/** \brief Get an entry in the scope stack.
55 \param level Level of the entry to return: 0 means top and negative is
56 relative to the top.
57 */
58SCOPESTACK *
59get_scope(int level)
60{
61 if (level <= 0) {
62 level += sem.scope_level;
63 }
64 if (level < 0 || level >= sem.scope_size) {
65#if DEBUG
66 dumpscope(gbl.dbgfil);
67#endif
68 interr("bad scope stack level", level, ERR_Fatal);
69 }
70 return &sem.scope_stack[level];
71}

73/** \brief Return the level of this entry in the scope stack
74 or -1 if \a scope is null.
75 */
76int
77get_scope_level(SCOPESTACK *scope)
78{
79 if (scope == 0) {
80 return -1;
81 } else {
82 int level = scope - sem.scope_stack;
83 assert(level >= 0 && level <= sem.scope_level, "bad scope stack level",
84 level, ERR_Fatal);
85 return level;
86 }
87}

89/** \brief Return the next entry below this one in the scope stack; 0 if none.
90 If scope is 0, return the top of the stack.
91 */
92SCOPESTACK *
93next_scope(SCOPESTACK *scope)
94{
95 int sl = get_scope_level(scope);
96 if (sl < 0) {
97 return curr_scope();
98 } else if (sl == 0) {
99 return 0;
100 } else {
101 return scope - 1;
102 }
103}

162/** \brief Return the USE module scope for the module associated with this
163 symbol, or -1 if none.
164 */
165int
166have_use_scope(int sptr)
167{
168 SCOPESTACK *scope = 0;
169 if (sem.scope_stack == NULL)
170 return -1;
171 while ((scope = next_scope(scope)) != 0) {
172 if (scope->kind == SCOPE_USE && scope->sptr == sptr)
173 return get_scope_level(scope);
174 if (scope->closed)
175 return -1;
176 }
177 return -1;
178}

197/** \brief Push a slot on the scope stack.
198 * \param sptr For most scopes, scope symbol. For interface and parallel
199 scopes, a small (unnamed) numeric index.
200 * \param kind Scope kind.
201 */
202void
203push_scope_level(int sptr, SCOPEKIND kind)
204{
205 SCOPESTACK *scope;
206
207 if (sem.scope_stack == NULL)
208 return;
209
210 scope = push_scope();
211 scope->kind = kind;
212 scope->sptr = sptr;
213 scope->symavl = stb.stg_avail;
214
215 switch (kind) {
216 case SCOPE_NORMAL:
217 case SCOPE_MODULE:
218 case SCOPE_BLOCK:
219 case SCOPE_USE:
220 case SCOPE_PAR:
221 break;
222 case SCOPE_SUBPROGRAM:
223 if (sem.which_pass == 1)
224 setfile(1, SYMNAME(sptr), 0);
225 break;
226 case SCOPE_INTERFACE:
227 scope->closed = TRUE;
228 break;
229 default:
230 interr("push_scope_level: invalid scope kind", kind, ERR_Warning);
231 }
232
233 /* When entering a parallel scope, the current scope is left as the
234 * enclosing scope of the parallel region. */
235 if (kind != SCOPE_PAR)
236 stb.curr_scope = sptr;
237
238#if DEBUG
239 if (DBGBIT(5, 0x200)) {
240 fprintf(gbl.dbgfil, "\n++++++++ push_%ccope_level(%s) pass=%d line=%d\n",
241 sem.which_pass ? 'S' : 's', kind_to_string(kind), sem.which_pass,
242 gbl.lineno);
243 dumpscope(gbl.dbgfil);
244 }
245#endif
246}

248/* \brief Pop scope stack up to and including a slot of the given kind. */
249void
250pop_scope_level(SCOPEKIND kind)
251{
252 SCOPEKIND curr_kind;
253 SPTR sptr;
254
255 if (sem.scope_stack == NULL)
256 return;
257
258 if (sem.scope_level == 0) {
259 interr("pop_scope_level: stack underflow", sem.scope_level, ERR_Severe);
260 return;
261 }
262
263 do {
264 SCOPESTACK *scope = curr_scope();
265 int top = scope->symavl;
266 int scope_id = scope->sptr;
267 curr_kind = scope->kind;
268 switch (curr_kind) {
269 case SCOPE_BLOCK:
270 for (sptr = top; sptr < stb.stg_avail; ++sptr) {
271 if (!CONSTRUCTSYMG(sptr) || HIDDENG(sptr))
272 continue;
273 if (ST_ISVAR(STYPEG(sptr))) {
274 // Some secondary syms are marked as construct syms when created.
275 // Assuming that isn't guaranteed for all cases, re/do that here.
276 SPTR parent = ENCLFUNCG(sptr), sptr2;
277 if ((sptr2 = MIDNUMG(sptr)) != 0) {
278 CONSTRUCTSYMP(sptr2, true);
279 ENCLFUNCP(sptr2, parent);
280 }
281 if ((sptr2 = SDSCG(sptr)) != 0) {
282 CONSTRUCTSYMP(sptr2, true);
283 ENCLFUNCP(sptr2, parent);
284 }
285 if ((sptr2 = PTROFFG(sptr)) != 0) {
286 CONSTRUCTSYMP(sptr2, true);
287 ENCLFUNCP(sptr2, parent);
288 }
289 }
290 // Avoid repeat processing in an ancestor block.
291 HIDDENP(sptr, true);
292 // Secondary syms must remain on hash chains to avoid conflicts with
293 // same-named syms in other blocks. Remove primary syms.
294 if (!HCCSYMG(sptr))
295 pop_sym(sptr);
296 }
297 break;
298 case SCOPE_INTERFACE: {
299 int parent_scope_id = get_scope(-1)->sptr;
300 if (parent_scope_id != scope_id)
301 for (sptr = stb.stg_avail - 1; sptr >= top; --sptr)
302 if (SCOPEG(sptr) == scope_id)
303 SCOPEP(sptr, parent_scope_id); // rehost to enclosing scope
304 break;
305 }
306 default:
307 if (sem.interface && STYPEG(scope_id) != ST_MODULE)
308 for (sptr = stb.stg_avail - 1; sptr >= top; --sptr)
309 if (SCOPEG(sptr) == scope_id)
310 IGNOREP(sptr, 1); // remove interface scope symbol
311 break;
312 }
313 pop_scope();
314 } while (curr_kind != kind);
315
316 /*
317 * When leaving a parallel scope, the current scope doesn't need to be
318 * reset since it should always be the scope of the nonparallel region
319 * containing the parallel region.
320 */
321 if (kind != SCOPE_PAR && sem.scope_stack[sem.scope_level].kind != SCOPE_PAR)
322 stb.curr_scope = sem.scope_level > 0 ?
323 curr_scope()->sptr : sem.scope_stack[1].sptr;
324
325#if DEBUG
326 if (DBGBIT(5, 0x200)) {
327 fprintf(gbl.dbgfil, "\n-------- pop_%ccope_level(%s) pass=%d line=%d\n",
328 sem.which_pass ? 'S' : 's', kind_to_string(kind), sem.which_pass,
329 gbl.lineno);
330 dumpscope(gbl.dbgfil);
331 }
332#endif
333}

335static SCOPESTACK saved_scope_stack[1];
336static int count_scope_saved = 0;

338/** \brief Pop the current scope into a save area; restore with
339 * restore_scope_level() */
340void
341save_scope_level(void)
342{
343 if (count_scope_saved >= 1) {
344 interr("trying to save too many scope levels", count_scope_saved, 3);
345 return;
346 }
347 saved_scope_stack[count_scope_saved++] = *curr_scope();
348 pop_scope();
349 stb.curr_scope = curr_scope()->sptr;
350#if DEBUG
351 if (DBGBIT(5, 0x200)) {
352 fprintf(gbl.dbgfil, "\n-------- save_%ccope_level pass=%d line=%d\n",
353 sem.which_pass ? 'S' : 's', sem.which_pass, gbl.lineno);
354 dumpscope(gbl.dbgfil);
355 }
356#endif
357}
358
359/** \brief Restore the scope that was saved by save_scope_level() */
360void
361restore_scope_level(void)
362{
363 if (count_scope_saved <= 0) {
364 interr("trying to restore too many scope levels", count_scope_saved, 3);
365 return;
366 }
367 *push_scope() = saved_scope_stack[--count_scope_saved];
368 stb.curr_scope = curr_scope()->sptr;
369#if DEBUG
370 if (DBGBIT(5, 0x200)) {
371 fprintf(gbl.dbgfil, "\n++++++++ restore_%ccope_level pass=%d line=%d\n",
372 sem.which_pass ? 'S' : 's', sem.which_pass, gbl.lineno);
373 dumpscope(gbl.dbgfil);
374 }
375#endif
376}

378void
379par_push_scope(LOGICAL bind_to_outer)
380{
381 SCOPESTACK *scope, *next_scope;
382 SC_KIND prev_sc = sem.sc;
383 if (curr_scope()->kind != SCOPE_PAR && sem.parallel >= 1) {
384 sem.sc = SC_PRIVATE;
385 } else if (sem.task) {
386 sem.sc = SC_PRIVATE;
387 }
388 else if (sem.teams >= 1) {
389 sem.sc = SC_PRIVATE;
390 } else if (sem.target && sem.parallel >= 1) {
391 sem.sc = SC_PRIVATE;
392 }
393 push_scope_level(sem.next_unnamed_scope++, SCOPE_PAR);
394 scope = curr_scope();
395 next_scope = get_scope(-1); /* next to top of stack */
396 if (!bind_to_outer || next_scope->kind != SCOPE_PAR) {
397 scope->rgn_scope = sem.scope_level;
398 scope->par_scope = PAR_SCOPE_SHARED;
399 } else {
400 scope->rgn_scope = next_scope->rgn_scope;
401 scope->par_scope = next_scope->par_scope;
402 scope->end_prologue = next_scope->end_prologue;
403 }
404 scope->di_par = sem.doif_depth;
405 scope->shared_list = NULL;
406 scope->prev_sc = prev_sc;
407 enter_lexical_block(flg.debug && !XBIT(123, 0x400));
408}

410void
411par_pop_scope(void)
412{
413 SCOPE_SYM *symp;
414 /*
415 * Restore the scope of any symbols that appeared in a SHARED clause.
416 * This is only needed if the DEFAULT scope is 'PRIVATE' or 'NONE".
417 */
418 for (symp = curr_scope()->shared_list; symp != NULL; symp = symp->next) {
419 SCOPEP(symp->sptr, symp->scope);
420 }
421 if (BLK_SYM(sem.scope_level)) {
422 exit_lexical_block(flg.debug && !XBIT(123, 0x400));
423 }
424
425 sem.sc = curr_scope()->prev_sc;
426 pop_scope_level(SCOPE_PAR);
427 if (curr_scope()->kind != SCOPE_PAR) {
428 sem.sc = SC_LOCAL;
429 }
430}

432static SCOPESTACK *
433push_scope(void)
434{
435 ++sem.scope_level;
436 NEED(sem.scope_level + 1, sem.scope_stack, SCOPESTACK, sem.scope_size,
437 sem.scope_size + 10);
438 BZERO(sem.scope_stack + sem.scope_level, SCOPESTACK, 1);
439 return curr_scope();
440}

442static void
443pop_scope(void)
444{
445 --sem.scope_level;
446 assert(sem.scope_level >= 0, "scope stack underflow", sem.scope_level,
447 ERR_Fatal);
448}

467void
468dump_one_scope(int sl, FILE *f)
469{
470 SCOPESTACK *scope;
471 SPTR sptr;
472 if (f == NULL) {
473 f = stderr;
474 }
475 if (sl < 0 || sl >= sem.scope_size) {
476 interr("dump_one_scope: bad scope stack level", sl, ERR_Warning);
477 return;
478 }
479 scope = sem.scope_stack + sl;
480 sptr = scope->sptr;
481 fprintf(f, "%ccope %2d. %-11s %-7s %-8s symavl=%3d %d=%s\n",
482 sem.which_pass ? 'S' : 's', sl, kind_to_string(scope->kind),
483 scope->closed ? "closed" : "open",
484 scope->Private ? "private" : "public",
485 scope->symavl, sptr,
486 sptr >= stb.firstosym ? SYMNAME(sptr) : "");
487 if (scope->except) {
488 int ex;
489 fprintf(f, "+ except");
490 for (ex = scope->except; ex; ex = SYMI_NEXT(ex)) {
491 fprintf(f, " %d(%s)", SYMI_SPTR(ex), SYMNAME(SYMI_SPTR(ex)));
492 }
493 fprintf(f, "\n");
494 }
495 if (scope->import) {
496 int im;
497 fprintf(f, "+ import");
498 for (im = scope->import; im; im = SYMI_NEXT(im)) {
499 fprintf(f, " %d(%s)", SYMI_SPTR(im), SYMNAME(SYMI_SPTR(im)));
500 }
501 fprintf(f, "\n");
502 }
