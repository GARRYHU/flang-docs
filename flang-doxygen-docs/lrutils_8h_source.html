59/* Macros */
60/**
61 * INSET In Set Macro
62 *
63 * INSET receives two (2) numeric parameters. The first (set)
64 * is a bit map for a set, with each bit being a flag which
65 * indicates if the number corresponding to the bit's position
66 * in the word is in the set. The second parameter (number)
67 * is a number which must be in the range 0 to 31. The macro
68 * returns 1 if 'number' is in 'set', 0 otherwise.
69 */
70#define INSET(set, number) (((set) >> (number)) & 1)
