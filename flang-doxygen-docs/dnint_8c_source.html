43double
44__mth_i_dnint(double f)
45{
46 double x = f; /* Cases where f == 0.0, or f == NaN */
47 union ieee754_double *u = (union ieee754_double *)&x;
48
49 /*
50 * Depending on the default rounding mode of the processor, the logic
51 * below with modf(f + 0.5) can result in a bogus rounding when 0.5
52 * is normalized such that it falls within the guard or round bits.
53 *
54 * Fast return if the exponent guarantees that the floating point number
55 * is a whole integer.
56 *
57 * This quick exit also catches infinities and NaNs.
58 */
59
60 if (u->ieee.exponent >= IEEE754_DOUBLE_BIAS+52) return x;
61
62 if (f > 0.0)
63 (void)modf(f + 0.5, &x);
64 else if (f < 0.0)
65 (void)modf(f - 0.5, &x);
66 return x;
67}
