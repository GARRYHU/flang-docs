27typedef struct {

106static void I8(copy_xfer_loop)(copy_parm *z, copy_sect *ly, __INT_T offset,
107 __INT_T dim)
108{
109 DECL_HDR_PTRS(lc);
110 DECL_DIM_PTRS(lcd);
111 char *adr;
112 int lx;
113 __INT_T cnt, str;
114
115 lc = ly->sect;
116 cnt = str = 1;
117 if (dim > 0) {
118 if (dim > z->span) {
119 lx = ly->axis_map[dim - 1];
120 SET_DIM_PTRS(lcd, lc, lx - 1);
121 cnt = ly->extent[lx - 1];
122 str = ly->stride[lx - 1] * F90_DPTR_SSTRIDE_G(lcd) *
123 F90_DPTR_LSTRIDE_G(lcd);
124 if (dim > z->unit) {
125 for (; --cnt >= 0; offset += str)
126 I8(copy_xfer_loop)(z, ly, offset, dim - 1);
127 return;
128 }
129 } else
130 cnt = z->cnt;
131 }
132 adr = ly->base + offset * F90_LEN_G(lc);
133#if defined(DEBUG)
134 if (__fort_test & DEBUG_COPY) {
135 printf("%d copy %s cpu=%d offset=%d adr=%x cnt=%d str=%d ", GET_DIST_LCPU,
136 z->xdir, z->cpu, offset, adr, cnt, str);
137 __fort_show_scalar(adr, F90_KIND_G(lc));
138 printf("\n");
139 }
140#endif
141 z->xfer(z->cc, z->cpu, adr, cnt, str, F90_KIND_G(lc), F90_LEN_G(lc));
142}

146static void I8(copy_xfer)(copy_parm *z, /* parameter struct */
147 copy_sect *ly, /* local parameters */
148 copy_sect *ry, /* remote parameters */
149 __INT_T offset, /* local array offset */
150 __INT_T cnt) /* transfer length */
151{
152 DECL_HDR_PTRS(lc);
153 DECL_HDR_PTRS(rc);
154 DECL_DIM_PTRS(lcd);
155 DECL_DIM_PTRS(rcd);
156 proc *p;
157 procdim *pd;
158 int dfmt, lx, px, rx;
159 __INT_T i, m, pc = 0, rg0, rp0;
160 __INT_T rpc[MAXDIMS];
161
162 /* Determine the maximum span of dimensions that can be
163 transferred as one contiguous block and compute the contiguous
164 block length */
165
166 lc = ly->sect;
167 rc = ry->sect;
168 for (m = i = 1; i <= F90_RANK_G(lc); ++i) {
169 if ((z->permuted >> (i - 1)) & 1)
170 break;
171 lx = ly->axis_map[i - 1];
172 SET_DIM_PTRS(lcd, lc, lx - 1);
173 rx = ry->axis_map[i - 1];
174 SET_DIM_PTRS(rcd, rc, rx - 1);
175 if (ly->stride[lx - 1] * F90_DPTR_SSTRIDE_G(lcd) *
176 F90_DPTR_LSTRIDE_G(lcd) !=
177 m ||
178 ry->stride[rx - 1] * F90_DPTR_SSTRIDE_G(rcd) *
179 F90_DPTR_LSTRIDE_G(rcd) !=
180 m)
181 break;
182 m *= ly->extent[lx - 1];
183 }
184 for (; i <= F90_RANK_G(lc); ++i) {
185 lx = ly->axis_map[i - 1];
186 if (ly->extent[lx - 1] != 1)
187 break;
188 }
189 z->span = i - 1;
190 z->cnt = m;
191
192 /* Determine the maximum span of unit dimensions, i.e.
193 dimensions that can be transferred with a stride between
194 single elements. */
195
196 for (i = 1; i <= F90_RANK_G(lc); ++i) {
197 lx = ly->axis_map[i - 1];
198 if (ly->extent[lx - 1] != 1)
199 break;
200 }
201 z->unit = i;
202
203#if defined(DEBUG)
204 if (__fort_test & DEBUG_COPY) {
205 printf("%d copy %s l(", GET_DIST_LCPU, z->xdir);
206 for (i = 0; i < F90_RANK_G(lc); ++i) {
207 if (i > 0)
208 printf(",");
209 printf("%d:%d:%d", ly->lower[i], ly->upper[i], ly->stride[i]);
210 }
211 printf(") r(");
212 for (i = 0; i < F90_RANK_G(rc); ++i) {
213 if (i > 0)
214 printf(",");
215 printf("%d:%d:%d", ry->lower[i], ry->upper[i], ry->stride[i]);
216 }
217 printf(") cnt=%d span=%d unit=%d\n", z->cnt, z->span, z->unit);
218 }
219#endif
220
221 if (LOCAL_MODE) {
222
223 /* only local communication */
224
225 z->cpu = 0;
226 } else {
227
228 /* Determine remote prime owner (low processor number) and
229 replication group index. */
230
231 rp0 = DIST_PBASE_G(rc);
232 dfmt = DIST_DFMT_G(rc);
233 for (rx = 0; rx < F90_RANK_G(rc); ++rx, dfmt >>= DFMT__WIDTH) {
234 switch (dfmt & DFMT__MASK) {
235 case DFMT_COLLAPSED:
236 rpc[rx] = 0;
237 continue;
238 default:
239 __fort_abort("copy: unsupported dist-format");
240 }
241 rp0 += pc * DIST_DPTR_PSTRIDE_G(rcd); /* owning processor number */
242 rpc[rx] = pc;
243 }
244
245 if (DIST_REPLICATED_G(lc) | DIST_REPLICATED_G(rc)) {
246
247 /* source or dest replication */
248
249 p = DIST_DIST_TARGET_G(rc);
250 m = DIST_REPLICATED_G(rc);
251 for (px = 0; m != 0; m >>= 1, ++px) {
252 pd = &p->dim[px];
253 if (m & 1 && pd->coord > 0)
254 rp0 -= pd->coord * pd->stride;
255 }
256
257 rg0 = 0;
258 for (rx = F90_RANK_G(rc); --rx >= 0;)
259 rg0 += rpc[rx] * ry->repl.gstr[rx]; /* replication group index */
260
261 if (rp0 == ry->repl.plow)
262 z->cc = z->mych; /* source & dest = my group */
263 else
264 z->cc = ry->ch[rg0];
265 if (z->cc == NULL) {
266 if (z->xfer == __fort_sendl) {
267 z->cc = __fort_chn_1toN(NULL, ry->repl.ndim, rp0, ry->repl.pcnt,
268 ry->repl.pstr, ly->repl.ndim, ly->repl.plow,
269 ly->repl.pcnt, ly->repl.pstr);
270 } else {
271 z->cc = __fort_chn_1toN(NULL, ly->repl.ndim, ly->repl.plow,
272 ly->repl.pcnt, ly->repl.pstr, ry->repl.ndim,
273 rp0, ry->repl.pcnt, ry->repl.pstr);
274 }
275 if (rp0 == ry->repl.plow)
276 z->mych = z->cc;
277 ry->ch[rg0] = z->cc;
278 }
279#if defined(DEBUG)
280 if (__fort_test & DEBUG_COPY)
281 printf("%d copy %s cc=%x rg0=%d rp0=%d\n", GET_DIST_LCPU, z->xdir,
282 z->cc, rg0, rp0);
283 if (rg0 < 0 || rg0 >= ry->repl.ngrp)
284 __fort_abort("copy_xfer: bad remote group (internal error)");
285#endif
286 } else /* no replication */
287 z->cpu = rp0; /* remote processor number */
288 }
289 I8(copy_xfer_loop)(z, ly, offset, F90_RANK_G(lc));
290}

292/* loop over local section, find maximal blocks to transfer to remote
293 processors. Set up localized descriptors. */
294
295static void I8(copy_loop)(copy_parm *z, /* parameter struct */
296 copy_sect *ly, /* local section */
297 copy_sect *ry, /* remote section */
298 __INT_T offset, /* local array offset */
299 __INT_T cnt, /* transfer length */
300 int dim) /* dimension */
301{
302 DECL_HDR_PTRS(lc);
303 DECL_HDR_PTRS(rc);
304 DECL_DIM_PTRS(lcd);
305 DECL_DIM_PTRS(rcd);
306 int lx, rx;
307 __INT_T cl, cn, cs, clof, clos, i, ll, ln, lu, n, off, rl, rn, ru;
308
309 lc = ly->sect;
310 lx = ly->axis_map[dim - 1] - 1;
311 SET_DIM_PTRS(lcd, lc, lx);
312
313 rc = ry->sect;
314 rx = ry->axis_map[dim - 1] - 1;
315 SET_DIM_PTRS(rcd, rc, rx);
316
317 if (DIST_MAPPED_G(lc) >> lx & 1) {
318
319 /* local is mapped */
320
321 cl = DIST_DPTR_CL_G(lcd); /* cyclic parameters */
322 cs = DIST_DPTR_CS_G(lcd);
323 cn = DIST_DPTR_CN_G(lcd);
324 clof = DIST_DPTR_CLOF_G(lcd);
325 clos = DIST_DPTR_CLOS_G(lcd);
326
327 if (DIST_DPTR_CNO_G(lcd) > 1 &&
328 DIST_DPTR_BLOCK_G(lcd) == DIST_DPTR_TSTRIDE_G(lcd)) {
329
330 /* local is mapped cyclic, one element per block */
331
332 if (DIST_MAPPED_G(rc) >> rx & 1) {
333
334#if defined(DEBUG)
335 if (__fort_test & DEBUG_COPY)
336 printf("%d lx %d cyclic : mapped rx %d (skipped)\n", GET_DIST_LCPU,
337 lx + 1, rx + 1);
338#endif
339
340 } else {
341
342/* remote is unmapped */
343
344#if defined(DEBUG)
345 if (__fort_test & DEBUG_COPY)
346 printf("%d lx %d cyclic : unmapped rx %d\n", GET_DIST_LCPU, lx + 1,
347 rx + 1);
348#endif
349
350 ly->lower[lx] = cl - DIST_DPTR_TOFFSET_G(lcd);
351 if (DIST_DPTR_TSTRIDE_G(lcd) != 1)
352 ly->lower[lx] /= DIST_DPTR_TSTRIDE_G(lcd);
353 ly->upper[lx] = ly->lower[lx] + cn - 1;
354 ly->extent[lx] = cn;
355
356 ry->lower[rx] =
357 F90_DPTR_LBOUND_G(rcd) + ly->lower[lx] - F90_DPTR_LBOUND_G(lcd);
358 ry->upper[rx] = ry->lower[rx] + (cn - 1) * DIST_DPTR_PSHAPE_G(lcd);
359 ry->stride[rx] = DIST_DPTR_PSHAPE_G(lcd);
360 ry->extent[rx] = cn;
361
362 off = offset +
363 (F90_DPTR_SSTRIDE_G(lcd) * ly->lower[lx] +
364 F90_DPTR_SOFFSET_G(lcd) - clof) *
365 F90_DPTR_LSTRIDE_G(lcd);
366
367 if (dim > 1)
368 I8(copy_loop)(z, ly, ry, off, cnt * cn, dim - 1);
369 else
370 I8(copy_xfer)(z, ly, ry, off, cnt * cn);
371 return;
372 }
373 }
374 } else {
375
376 /* local is unmapped */
377
378 cl = clof = clos = 0; /* unmapped cyclic same as block */
379 cn = cs = 1;
380
381 if (DIST_DPTR_CNO_G(rcd) > 1 &&
382 DIST_DPTR_BLOCK_G(rcd) == DIST_DPTR_TSTRIDE_G(rcd)) {
383
384/* remote is mapped cyclic, one element per block */
385
386#if defined(DEBUG)
387 if (__fort_test & DEBUG_COPY)
388 printf("%d lx %d unmapped : cyclic rx %d\n", GET_DIST_LCPU, lx + 1,
389 rx + 1);
390#endif
391
392 rn = DIST_DPTR_PSHAPE_G(rcd);
393 ly->lower[lx] = F90_DPTR_LBOUND_G(lcd); /* no-op ? */
394 ly->upper[lx] = DPTR_UBOUND_G(lcd); /* no-op ? */
395 ly->stride[lx] = rn;
396
397 off = offset +
398 (F90_DPTR_SSTRIDE_G(lcd) * ly->lower[lx] + F90_DPTR_SOFFSET_G(lcd) -
399 clof) *
400 F90_DPTR_LSTRIDE_G(lcd);
401
402 ry->lower[rx] = F90_DPTR_LBOUND_G(rcd); /* no-op ? */
403 ry->stride[rx] = 1;
404 for (i = rn; i > 0; --i) {
405 n = (DPTR_UBOUND_G(rcd) - ry->lower[rx] + rn) / rn;
406 if (n <= 0)
407 break;
408 ry->upper[rx] = ry->lower[rx] + n - 1;
409 ly->extent[lx] = ry->extent[rx] = n;
410 if (dim > 1)
411 I8(copy_loop)(z, ly, ry, off, cnt * cn, dim - 1);
412 else
413 I8(copy_xfer)(z, ly, ry, off, cnt * cn);
414 ly->lower[lx] += 1;
415 ry->lower[rx] += 1;
416 off += F90_DPTR_SSTRIDE_G(lcd) * F90_DPTR_LSTRIDE_G(lcd);
417 }
418 return;
419 }
420 }
421
422/* general case copy */
423
424#if defined(DEBUG)
425 if (__fort_test & DEBUG_COPY)
426 printf("%d lx %d mapped : mapped rx %d\n", GET_DIST_LCPU, lx + 1, rx + 1);
427#endif
428
429 while (cn > 0) {
430
431 /* set up local block bounds */
432
433 if (DIST_MAPPED_G(lc) >> lx & 1) {
434 ln = I8(__fort_block_bounds)(lc, lx + 1, cl, &ll, &lu);
435 } else {
436 /* unmapped case */
437 ll = F90_DPTR_LBOUND_G(lcd);
438 lu = DPTR_UBOUND_G(lcd);
439 ln = F90_DPTR_EXTENT_G(lcd);
440 }
441
442 off = offset +
443 (F90_DPTR_SSTRIDE_G(lcd) * ll + F90_DPTR_SOFFSET_G(lcd) - clof) *
444 F90_DPTR_LSTRIDE_G(lcd);
445
446#if defined(DEBUG)
447
448 if (__fort_test & DEBUG_COPY) {
449
450 printf("%d copy_loop: Local Array dist format is ", GET_DIST_LCPU);
451 I8(print_dist_format)(lc);
452 printf(" block size=%d tstride=%d tlb=%d tub=%d", DIST_DPTR_BLOCK_G(lcd),
453 DIST_DPTR_TSTRIDE_G(lcd), DIST_DPTR_TLB_G(lcd), DIST_DPTR_TUB_G(lcd));
454
455 printf("\n%d copy_loop: Remote Array dist format is ", GET_DIST_LCPU);
456 I8(print_dist_format)(rc);
457 printf(" block size=%d tstride=%d tlb=%d tub=%d", DIST_DPTR_BLOCK_G(rcd),
458 DIST_DPTR_TSTRIDE_G(rcd), DIST_DPTR_TLB_G(rcd), DIST_DPTR_TUB_G(rcd));
459
460 printf("\n");
461 }
462#endif
463
464 if ((DFMT(rc, rx + 1) == DFMT_GEN_BLOCK)) {
465
466 /*
467 * For the case of the remote array being a gen_block
468 * distribution, we cannot use the elegant math for
469 * finding the largest overlapping block. We need
470 * to consult the gen_block array for the block size since
471 * the block size can vary across processors.
472 */
473
474 __INT_T pshape;
475
476 __INT_T i, j;
477
478 __INT_T trl, tru, lLbound, rLbound;
479 __INT_T direction;
480 __INT_T blockSize;
481 __INT_T tExtent, aExtent;
482
483 __INT_T *gb;
484
485 __INT_T *tempGB;
486
487 lLbound = F90_DPTR_LBOUND_G(lcd);
488 rLbound = F90_DPTR_LBOUND_G(rcd);
489
490 pshape = DIST_DPTR_PSHAPE_G(rcd);
491
492 /* If we have a negative template stride, then we need to
493 * traverse the gen_block array from right to left rather
494 * than left to right...otherwise, we may send some array
495 * elements out of order ...
496 */
497
498 direction = (DIST_DPTR_TSTRIDE_G(rcd) < 0);
499
500 tExtent = (DIST_DPTR_TUB_G(rcd) - DIST_DPTR_TLB_G(rcd)) + 1;
501 aExtent = (DPTR_UBOUND_G(rcd) - rLbound) + 1;
502
503 if (tExtent != aExtent) {
504
505 tempGB = I8(__fort_new_gen_block)(rc, rx);
506 gb = tempGB;
507
508 } else {
509
510 tempGB = 0;
511 gb = DIST_DPTR_GEN_BLOCK_G(rcd);
512 }
513
514 gb += ((!direction) ? 0 : (pshape - 1));
515
516 rl = rLbound;
517 ru = rl - 1;
518
519 for (i = 0; i < pshape; ++i) {
520
521 blockSize = *gb;
522
523 /* check each element in gen_block array for overlapping
524 * block...
525 */
526
527 if (blockSize) {
528
529 ru += blockSize;
530 trl = tru = 0;
531
532 /* find overlapping bounds */
533
534 for (j = ((ll - lLbound) + 1); j <= ((lu - lLbound) + 1); j++) {
535
536 if (j == ((rl - rLbound) + 1))
537 trl = j;
538
539 if (j >= ((rl - rLbound) + 1) && j <= ((ru - rLbound) + 1))
540 tru = j;
541 }
542
543 if (trl || tru) {
544
545 if (!trl) {
546 if (ll > rl)
547 trl = ((ll - lLbound) + 1);
548 else
549 trl = ((rl - rLbound) + 1);
550
551 } else if (!tru) {
552
553 if (lu > ru)
554 tru = ((ru - rLbound) + 1);
555 else
556 tru = ((lu - lLbound) + 1);
557 }
558
559 /* we got an overlapping boundary.
560 * let's send the request ...
561 */
562
563 n = tru - trl + 1;
564
565 ly->lower[lx] = ((trl + lLbound) - 1);
566 ly->upper[lx] = (ll + n) - 1;
567 ly->extent[lx] = n;
568 ry->lower[rx] = ((trl + rLbound) - 1);
569 ry->upper[rx] = (rl + n) - 1;
570 ry->extent[rx] = n;
571
572 if (dim > 1)
573 I8(copy_loop)(z, ly, ry, off, cnt * n, dim - 1);
574 else
575 I8(copy_xfer)(z, ly, ry, off, cnt * n);
576
577 off += n * F90_DPTR_SSTRIDE_G(lcd) * F90_DPTR_LSTRIDE_G(lcd);
578 }
579 }
580
581 rl += *gb;
582
583 if (!direction)
584 ++gb;
585 else
586 --gb;
587 }
588
589 cl += cs;
590 clof += clos;
591 --cn;
592
593 if (tempGB)
594 __fort_free(tempGB);
595
596 continue; /*jump to while(cn > 0) loop head*/
597 }
598
599 i = ll - F90_DPTR_LBOUND_G(lcd);
600 rl = F90_DPTR_LBOUND_G(rcd) + i;
601 ru = DPTR_UBOUND_G(rcd);
602
603 /* find largest overlapping blocks of local and remote
604 sections, i.e. subdivide the local block at remote block
605 boundaries */
606
607 while (ln > 0) {
608
609 rn = ru - rl + 1;
610 n = Min(ln, rn);
611
612 if (n <= 0) {
613 char error[100];
614 sprintf(error, "copy_loop: empty block (internal error)");
615 __fort_abort(error);
616 }
617 ly->lower[lx] = ll;
618 ly->upper[lx] = ll + n - 1;
619 ly->extent[lx] = n;
620 ry->lower[rx] = rl;
621 ry->upper[rx] = rl + n - 1;
622 ry->extent[rx] = n;
623
624 /* transfer one overlapping block */
625
626 if (dim > 1)
627 I8(copy_loop)(z, ly, ry, off, cnt * n, dim - 1);
628 else
629 I8(copy_xfer)(z, ly, ry, off, cnt * n);
630
631 rl += n;
632 ll += n;
633 off += n * F90_DPTR_SSTRIDE_G(lcd) * F90_DPTR_LSTRIDE_G(lcd);
634 ln -= n;
635 }
636
637 cl += cs;
638 clof += clos;
639 --cn;
640 }
641}

643void I8(copy_setup)(copy_sect *y, char *b, F90_Desc *c, int *axis_map)
644{
645 int cx, i;
646 DECL_DIM_PTRS(cd);
647
648 y->base = b;
649 y->sect = c;
650
651 y->axis_map = axis_map;
652
653 I8(__fort_cycle_bounds)(c);
654
655 y->islocal = (~F90_FLAGS_G(c) & __OFF_TEMPLATE) && (F90_LSIZE_G(c) > 0);
656
657 for (i = F90_RANK_G(c); i > 0; --i) {
658 cx = axis_map[i - 1];
659 SET_DIM_PTRS(cd, c, cx - 1);
660 y->lower[cx - 1] = F90_DPTR_LBOUND_G(cd);
661 y->upper[cx - 1] = DPTR_UBOUND_G(cd);
662 y->stride[cx - 1] = 1;
663 y->extent[cx - 1] = F90_DPTR_EXTENT_G(cd);
664 y->islocal &= (DIST_DPTR_CN_G(cd) > 0);
665 }
666}

668/* base addresses already adjusted for scalar subscripts */
669
670chdr *I8(__fort_copy)(void *db, void *sb, F90_Desc *dc, F90_Desc *sc,
671 int *src_axis_map)
672{
673 DECL_DIM_PTRS(dcd);
674 DECL_DIM_PTRS(scd);
675 chdr *ch;
676 copy_parm z;
677 int dbogus, dg, dl, dn, ds, du, dx, sbogus, sg, sl, sn, ss, su, sx;
678 int i, lcpu, j, n, ng;
679 __INT_T offset;
680 int *tcpus_addr;
681
682 if (src_axis_map == NULL)
683 src_axis_map = identity_map;
684
685 lcpu = GET_DIST_LCPU;
686
687#if defined(DEBUG)
688 if (F90_TAG_G(dc) != __DESC)

704 __fort_abort("copy: type/len mismatch");
705#endif

710 sbogus = F90_FLAGS_G(sc) & __BOGUSBOUNDS;

715 F90_FLAGS_P(sc, F90_FLAGS_G(sc) & ~__BOGUSBOUNDS);

718
719 for (dx = 1; dx <= F90_RANK_G(dc); ++dx) {
720 sx = src_axis_map[dx - 1];

723 SET_DIM_PTRS(scd, sc, sx - 1);

725 /* compute number of strides to bring section lbound up to
726 target array lbound */

730 dl = ds > 0 ? DIST_DPTR_TLB_G(dcd) - 1 : DIST_DPTR_TUB_G(dcd) + 1;
731 dn = dl - F90_DPTR_LBOUND_G(dcd) + ds;
732 if (ds != 1)
733 dn /= ds;
734 if (dn < 0)
735 dn = 0;
736 } else
737 dn = 0;
738
739 if (sbogus) {
740 ss = F90_DPTR_SSTRIDE_G(scd);
741 sl = ss > 0 ? DIST_DPTR_TLB_G(scd) - 1 : DIST_DPTR_TUB_G(scd) + 1;
742 sn = sl - F90_DPTR_LBOUND_G(scd) + ss;
743 if (ss != 1)
744 sn /= ss;
745 if (sn < 0)
746 sn = 0;
747 } else

750 /* maximize lbound increment */
751
752 n = Max(dn, sn);
753
754 /* adjust lbound, compute extents */
755
756 if (dbogus) {
757 dl = F90_DPTR_LBOUND_G(dcd) + n * ds;
758 du = ds > 0 ? Min(DPTR_UBOUND_G(dcd), DIST_DPTR_TUB_G(dcd))
759 : Max(DPTR_UBOUND_G(dcd), DIST_DPTR_TLB_G(dcd));
760 dn = du - dl + ds;
761 if (ds != 1)
762 dn /= ds;
763 if (dn < 0)
764 dn = 0;
765 } else if (n != 0)
766 __fort_abort("copy: can't adjust dst lbound");
767 else
768 dn = F90_DPTR_EXTENT_G(dcd);
769
770 if (sbogus) {
771 sl = F90_DPTR_LBOUND_G(scd) + sn * ss;
772 su = ss > 0 ? Min(DPTR_UBOUND_G(scd), DIST_DPTR_TUB_G(scd))
773 : Max(DPTR_UBOUND_G(scd), DIST_DPTR_TLB_G(scd));
774 sn = su - sl + ss;
775 if (ss != 1)
776 sn /= ss;
777 if (sn < 0)
778 sn = 0;
779 } else if (n != 0)
780 __fort_abort("copy: can't adjust src lbound");
781 else
782 sn = F90_DPTR_EXTENT_G(scd);
783
784 /* minimize extent */
785
786 n = Min(dn, sn);
787
788 if (n <= 0)

794 du = dl + (n - 1) * ds;

796 DIST_DPTR_TAXIS_G(dcd), dl, du, ds);
797 } else if (n != dn)
798 __fort_abort("copy: can't adjust dst ubound");
799
800 if (sbogus) {

803 DIST_DPTR_TAXIS_G(scd), sl, su, ss);
804 } else if (n != sn)
805 __fort_abort("copy: can't adjust src ubound");
806 }

810 I8(__fort_finish_section)(dc);
811 }

817 }

820 return NULL;

836#if defined(DEBUG)

849 } else if (DIST_REPLICATED_G(dc) | DIST_REPLICATED_G(sc)) {

856#if defined(DEBUG)

883 } else {

894 if (F90_RANK_G(sc) > 0)

903#endif

915 well-defined order across all processors */

917 dx = z.sy.repl.grpi; /* dest = *, src = me */
918 sx = z.dy.repl.grpi * ng; /* dest = me, src = * */

922 dg = z.dy.repl.grpi;
923 sg = i;
924 ch = z.sy.ch[sg];
925 ++i;
926 ++sx;
927 } else if (dx == sx) {
928 dg = z.dy.repl.grpi;
929 sg = z.sy.repl.grpi;
930 ch = z.mych;
931 ++i;

933 ++j;
934 dx += ng;
935 } else {
936 dg = j;
937 sg = z.sy.repl.grpi;
938 ch = z.dy.ch[dg];
939 ++j;
940 dx += ng;
941 }
942 if (ch != NULL) {
943 z.cc = __fort_chain_em_up(z.cc, ch);
944 }
945 }
946 __fort_free(z.dy.ch);
947 } else
948 __fort_chn_prune(z.cc);
949
950 return z.cc;
951}
