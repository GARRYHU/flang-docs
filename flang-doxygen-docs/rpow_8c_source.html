57/* Deal with errno for out-of-range result */
58static inline float
59retval_errno_erange_overflow(float x __attribute__((unused)), float y __attribute__((unused)), int sign)
60{
61 if (sign == 1)
62 return infinityf_with_flags(AMD_F_OVERFLOW);
63 else /* sign == -1 */
64 return -infinityf_with_flags(AMD_F_OVERFLOW);
65}

90float FN_PROTOTYPE(mth_i_rpowr)(float x, float y)
91{
92 unsigned int ux, ax, uy, ay, mask;
93 int yexp, inty, xpos, ypos, negateres;
94 double dx, dy, dw, dlog2, dr;
95
96 /* Largest float, stored as a double */
97 const double large = 3.40282346638528859812e+38; /* 0x47efffffe0000000 */
98
99 /* Smallest float, stored as a double */
100 const double tiny = 1.40129846432481707092e-45; /* 0x36a0000000000000 */
101
102 GET_BITS_SP32(x, ux);
103 ax = ux & (~SIGNBIT_SP32);
104 xpos = ax == ux;
105 GET_BITS_SP32(y, uy);
106 ay = uy & (~SIGNBIT_SP32);
107 ypos = ay == uy;
108
109 if (ux == 0x3f800000) {
110 /* x = +1.0. Return +1.0 for all y, even NaN,
111 raising invalid only if y is a signalling NaN */
112 if (y + 1.0F == 2.0F)
113 dummy = 1;
114 return 1.0F;
115 } else if (uy == 0x3fc00000) {
116 /* y is 1.5. */
117 /* Return x * sqrt(x), even if x is infinity or NaN */
118 return (x * sqrt(x));
119 } else if (uy == 0x3f000000) {
120 /* y is 0.5. */
121 /* Return sqrt(x), even if x is infinity or NaN */
122 return (sqrt(x));
123 } else if (uy == 0x3e800000) {
124 /* y is 0.25. */
125 /* Return sqrt(sqrt(x)), even if x is infinity or NaN */
126 return (sqrt(sqrt(x)));
127 } else if (ay == 0) {
128 /* y is zero. */
129 /* y is zero. Return 1.0, even if x is infinity or NaN,
130 raising invalid only if x is a signalling NaN */
131 if (x + 1.0F == 2.0F)
132 dummy = 1;
133 return 1.0F;
134 } else if (((ax & EXPBITS_SP32) == EXPBITS_SP32) && (ax & MANTBITS_SP32)) {
135 /* x is NaN. Return NaN, with invalid exception if it's
136 a signalling NaN. */
137 return x + x;
138 } else if (((ay & EXPBITS_SP32) == EXPBITS_SP32) && (ay & MANTBITS_SP32)) {
139 /* y is NaN. Return NaN, with invalid exception if y
140 is a signalling NaN. */
141 return y + y;
142 } else if (uy == 0x3f800000)
143 /* y is 1.0; return x */
144 return x;
145 else if ((ay & EXPBITS_SP32) > 0x4f000000) {
146 /* y is infinite or so large that the result would
147 overflow or underflow. Flags should be raised
148 unless y is an exact infinity. */
149 int yinf = (ay == EXPBITS_SP32);
150 if (ypos) {
151 /* y is +ve */
152 if (ax == 0)
153 /* abs(x) = 0.0. */
154 return 0.0F;
155 else if (ax < 0x3f800000) {
156 /* abs(x) < 1.0 */
157 if (yinf)
158 return 0.0F;
159 else
160 return retval_errno_erange_underflow(x, y, 1);
161 } else if (ax == 0x3f800000) {
162 /* abs(x) = 1.0. */
163 return 1.0F;
164 } else {
165 /* abs(x) > 1.0 */
166 if (yinf)
167 return infinityf_with_flags(0);
168 else
169 return retval_errno_erange_overflow(x, y, 1);
170 }
171 } else {
172 /* y is -ve */
173 if (ax == 0) {
174 /* abs(x) = 0.0. Return +infinity. */
175 return retval_errno_edom(x, y, 1);
176 } else if (ax < 0x3f800000) {
177 /* abs(x) < 1.0; return +infinity. */
178 if (yinf)
179 return infinityf_with_flags(0);
180 else
181 return retval_errno_erange_overflow(x, y, 1);
182 } else if (ax == 0x3f800000) {
183 /* abs(x) = 1.0. */
184 return 1.0F;
185 } else {
186 /* abs(x) > 1.0 */
187 if (yinf)
188 return 0.0F;
189 else
190 return retval_errno_erange_underflow(x, y, 1);
191 }
192 }
193 }
194
195 /* See whether y is an integer.
196 inty = 0 means not an integer.
197 inty = 1 means odd integer.
198 inty = 2 means even integer.
199 */
200 yexp = ((uy & EXPBITS_SP32) >> EXPSHIFTBITS_SP32) - EXPBIAS_SP32 + 1;
201 if (yexp < 1)
202 inty = 0;
203 else if (yexp > 24)
204 inty = 2;
205 else /* 1 <= yexp <= 24 */
206 {
207 /* Mask out the bits of r that we don't want */
208 mask = (1 << (24 - yexp)) - 1;
209 if ((uy & mask) != 0)
210 inty = 0;
211 else if (((uy & ~mask) >> (24 - yexp)) & 0x00000001)
212 inty = 1;
213 else
214 inty = 2;
215 }
216
217 if ((ax & EXPBITS_SP32) == EXPBITS_SP32) {
218 /* x is infinity (NaN was already ruled out). */
219 if (xpos) {
220 /* x is +infinity */
221 if (ypos)
222 /* y > 0.0 */
223 return x;
224 else
225 return 0.0F;
226 } else {
227 /* x is -infinity */
228 if (inty == 1) {
229 /* y is an odd integer */
230 if (ypos)
231 /* Result is -infinity */
232 return x;
233 else
234 return -0.0F;
235 } else {
236 if (ypos)
237 /* Result is +infinity */
238 return -x;
239 else
240 return 0.0F;
241 }
242 }
243 } else if (ax == 0) {
244 /* x is zero */
245 if (xpos) {
246 /* x is +0.0 */
247 if (ypos)
248 /* y is positive; return +0.0 for all cases */
249 return x;
250 else
251 /* y is negative; return +infinity with div-by-zero
252 for all cases */
253 return retval_errno_edom(x, y, 1);
254 } else {
255 /* x is -0.0 */
256 if (ypos) {
257 /* y is positive */
258 if (inty == 1)
259 /* -0.0 raised to a positive odd integer returns -0.0 */
260 return x;
261 else
262 /* Return +0.0 */
263 return -x;
264 } else {
265 /* y is negative */
266 if (inty == 1) {
267 /* -0.0 raised to a negative odd integer returns -infinity
268 with div-by-zero */
269 return retval_errno_edom(x, y, 2);
270 } else {
271 /* Return +infinity with div-by-zero */
272 return retval_errno_edom(x, y, 1);
273 }
274 }
275 }
276 }
277
278 negateres = 0;
279 if (!xpos) {
280 /* x is negative */
281 if (inty) {
282 /* It's OK because y is an integer. */
283 ux = ax;
284 PUT_BITS_SP32(ux, x); /* x = abs(x) */
285 /* If y is odd, the result will be negative */
286 negateres = (inty == 1);
287 } else
288 /* y is not an integer. Return a NaN. */
289 return retval_errno_edom(x, y, 3);
290 }
291
292 if (ay < 0x2e800000) /* abs(y) < 2^(-34) */
293 {
294 /* y is close enough to zero for the result to be 1.0
295 no matter what the size of x */
296 return 1.0F + y;
297 }
298
299 /* Simply use double precision for computation of log2(x),
300 y*log2(x) and exp2(y*log2(x)) */
301 dx = x;
302 dy = y;
303 dlog2 = FN_PROTOTYPE(mth_i_dlog2)(dx);
304 dw = dy * dlog2;
305 dr = FN_PROTOTYPE(mth_i_dexp2)(dw);
306
307 /* If dr overflowed or underflowed we need to deal with errno */
308 if (dr > large) {
309 /* Double dr has overflowed range of float. */
310 if (negateres)
311 return retval_errno_erange_overflow(x, y, -1);
312 else
313 return retval_errno_erange_overflow(x, y, 1);
314 } else if (dr < tiny) {
315 /* Double dr has underflowed range of float. */
316 if (negateres)
317 return retval_errno_erange_underflow(x, y, -1);
318 else
319 return retval_errno_erange_underflow(x, y, 1);
320 } else {
321 if (negateres)
322 return (float)-dr;
323 else
324 return (float)dr;
325 }
326}
