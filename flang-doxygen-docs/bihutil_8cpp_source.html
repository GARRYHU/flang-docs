31/** \brief Initialize BIH area

34bih_init(void)
35{
36 STG_ALLOC(bihb, 128);
37 STG_SET_FREELINK(bihb, BIH, next);
38 BIH_LPCNTFROM(0) = 0;
39 BIH_NEXT(bihb.stg_size - 1) = 0;
40 BIH_FLAGS(0) = 0;
41 BIH_NEXT(0) = 0;
42 BIH_PREV(0) = 0;
43 BIH_AVLPCNT(0) = 0;
44 BIH_GUARDEE(0) = 0;

68/** \brief Add a new block during the expand phase.
69 *
70 * Create a new bih entry and add it after the one specified by the formal
71 * argument set the FINDEX/FTAG from fihb structure; used during expand
72 *
73 * \param after BIH entry to add after
74 * \return new BIH entry
75 */
76int
77exp_addbih(int after)
78{
79
80 int i;
81 BIH *p;
82
83 i = STG_NEXT_FREELIST(bihb);
84 p = bihb.stg_base + i;
85 p->prev = after;
86 p->next = BIH_NEXT(after);
87 BIH_NEXT(after) = i;
88 BIH_PREV(p->next) = i;
89 p->label = SPTR_NULL;
90 p->lineno = 0;
91 p->flags.all = 0;
92 p->flags2.all = 0;
93 p->flags.bits.rd = 1;
94 p->first = 0;
95 p->last = 0;
96 p->assn = 0;
97 p->lpcntFrom = 0;
98 p->rgset = 0;
99#ifdef BIH_FTAG
100 p->findex = fihb.nextfindex;
101 p->ftag = fihb.nextftag;
102#endif
103 p->blkCnt = UNKNOWN_EXEC_CNT;
104 p->aveLpCnt = 0;
105 if (i > bihb.stg_max)
106 bihb.stg_max = i;
107
108 return (i);
109} /* exp_addbih */

111/** \brief Universal function to create new BIH entry
112 *
113 * \param after BIH anery to the new one after
114 * \param flags BIH entry to take flags from
115 * \param fih BIH entry to take FIH information from
116 *
117 * \return new BIH entry
118 */
119int
120addnewbih(int after, int flags, int fih)
121{
122 int i, next;
123 BIH *p;
124
125 i = STG_NEXT_FREELIST(bihb);
126 p = bihb.stg_base + i;
127 p->prev = after;
128 next = BIH_NEXT(after);
129 BIH_NEXT(after) = i;
130 if (next >= 0) {
131 p->next = next;
132 BIH_PREV(next) = i;
133 }
134 p->label = SPTR_NULL;
135 p->lineno = 0;
136 p->flags.all = 0;
137 p->flags2.all = 0;
138 p->flags.bits.rd = 1;
139 p->first = 0;
140 p->last = 0;
141 p->assn = 0;
142 p->lpcntFrom = 0;
143 p->rgset = 0;
144#ifdef BIH_FTAG
145 p->findex = 0;
146 p->ftag = 0;
147#endif
148 p->blkCnt = 0;
149#ifdef BIH_FTAG
150 p->ftag = BIH_FTAG(fih);
151#endif
152 p->lineno = BIH_LINENO(fih);
153 p->findex = BIH_FINDEX(fih);
154 if (flags) {
155 p->flags.bits.par = BIH_PAR(flags);
156 p->flags.bits.parsect = BIH_PARSECT(flags);
157 p->flags2.bits.task = BIH_TASK(flags);
158 p->blkCnt = BIH_BLKCNT(flags);
159 p->aveLpCnt = BIH_AVLPCNT(flags);
160 }
161 if (i > bihb.stg_max)
162 bihb.stg_max = i;
163 return i;
164} /* addnewbih */

180/** \brief Delete a bih entry from the BIH list
181 *
182 * \param bihx BIH entry to delete
183 */
184void
185delbih(int bihx)
186{
187 int prev, next;
188 BIH bb;
189 next = BIH_NEXT(bihx);
190 prev = BIH_PREV(bihx);
191 BIH_PREV(next) = prev;
192 BIH_NEXT(prev) = next;
193 /* STG_ADD_FREELIST clears the fields;
194 * instead we want to preserve the fields,
195 * except the freelist link in BIH_NEXT */
196 bb = bihb.stg_base[bihx];
197 STG_ADD_FREELIST(bihb, bihx);
198 bb.next = BIH_NEXT(bihx);
199 bihb.stg_base[bihx] = bb;
200}

204void
205merge_bih_flags(int to_bih, int fm_bih)
206{
207 /* after merging two blocks, merge their BIH flags */
208
209 BIH_FT(to_bih) = BIH_FT(fm_bih);
210 /* BIH_EX(to_bih) |= BIH_EX(fm_bih); ***causes SUN cc to error */
211 BIH_EX(to_bih) = BIH_EX(to_bih) | BIH_EX(fm_bih);
212 BIH_ZTRP(to_bih) = BIH_ZTRP(to_bih) | BIH_ZTRP(fm_bih);
213 BIH_SMOVE(to_bih) = BIH_SMOVE(to_bih) | BIH_SMOVE(fm_bih);
214 BIH_NOBLA(to_bih) = BIH_NOBLA(to_bih) | BIH_NOBLA(fm_bih);
215 BIH_QJSR(to_bih) = BIH_QJSR(to_bih) | BIH_QJSR(fm_bih);
216 BIH_INVIF(to_bih) = BIH_INVIF(to_bih) | BIH_INVIF(fm_bih);
217 BIH_NOINVIF(to_bih) = BIH_NOINVIF(to_bih) | BIH_NOINVIF(fm_bih);
218 BIH_SIMD(to_bih) = BIH_SIMD(to_bih) | BIH_SIMD(fm_bih);
219 BIH_NOSIMD(to_bih) = BIH_NOSIMD(to_bih) | BIH_NOSIMD(fm_bih);
220 BIH_RESID(to_bih) = BIH_RESID(to_bih) | BIH_RESID(fm_bih);
221 BIH_VCAND(to_bih) = BIH_VCAND(to_bih) | BIH_VCAND(fm_bih);
222 BIH_MIDIOM(to_bih) = BIH_MIDIOM(to_bih) | BIH_MIDIOM(fm_bih);
223 BIH_COMBST(to_bih) = BIH_COMBST(to_bih) | BIH_COMBST(fm_bih);
224 BIH_ASM(to_bih) = BIH_ASM(to_bih) | BIH_ASM(fm_bih);
225 BIH_LDVOL(to_bih) = BIH_LDVOL(to_bih) | BIH_LDVOL(fm_bih);
226 BIH_STVOL(to_bih) = BIH_STVOL(to_bih) | BIH_STVOL(fm_bih);
227 BIH_NODEPCHK(to_bih) = BIH_NODEPCHK(to_bih) | BIH_NODEPCHK(fm_bih);
228 BIH_UNROLL(to_bih) = BIH_UNROLL(to_bih) | BIH_UNROLL(fm_bih);
229 BIH_UNROLL_COUNT(to_bih) = BIH_UNROLL_COUNT(to_bih) | BIH_UNROLL_COUNT(fm_bih);
230 BIH_NOUNROLL(to_bih) = BIH_NOUNROLL(to_bih) | BIH_NOUNROLL(fm_bih);
231
232 if (BIH_TAIL(fm_bih))
233 BIH_TAIL(to_bih) = 1;
234 if (BIH_LAST(fm_bih))
235 BIH_LAST(to_bih) = 1;

238/** \brief Merge a block with its successor
239 *
240 * \param curbih BIH block to merge with its successor
241 * \return BIH of the merged block if merging occurred; otherwise, return 0.
242 */
243int
244merge_bih(int curbih)
245{
246 int nextbih, label;
247 int firstilt, lastilt, iltx;
248
249 if (XBIT(8, 0x80000000))
250 return 0;
251
252 nextbih = BIH_NEXT(curbih);
253
254 if (BIH_EN(curbih) || BIH_EN(nextbih) || BIH_XT(curbih) || BIH_XT(nextbih) ||
255 BIH_NOMERGE(curbih) || BIH_NOMERGE(nextbih) || BIH_ENLAB(curbih) ||
256 BIH_ENLAB(nextbih)
257#ifdef BIH_GASM
258 || BIH_GASM(curbih) || BIH_GASM(nextbih)
259#endif
260 ) {
261 return 0;
262 }
263 if (BIH_EN(nextbih))
264 return 0;
265
266 if (BIH_ASSN(curbih) != BIH_ASSN(nextbih) ||
267 BIH_PAR(curbih) != BIH_PAR(nextbih) ||
268 BIH_PARSECT(curbih) != BIH_PARSECT(nextbih) ||
269 BIH_PL(curbih) != BIH_PL(nextbih) || BIH_CS(curbih) != BIH_CS(nextbih) ||
270 BIH_TASK(curbih) != BIH_TASK(nextbih)) {
271 return 0;
272 }
273
274 if (BIH_COMBST(curbih) && !BIH_COMBST(nextbih)) {
275 return 0;
276 }
277
278 label = BIH_LABEL(nextbih);
279 if (label) {
280 if (RFCNTG(label) || VOLG(label)) {
281 return 0;
282 }
283 ILIBLKP(label, 0);
284 BIH_LABEL(nextbih) = SPTR_NULL;
285 }
286
287 firstilt = BIH_ILTFIRST(nextbih);
288 if (ILT_DBGLINE(firstilt)) /* watch out for debugger call */
289 firstilt = ILT_NEXT(firstilt);
290 if (firstilt == 0) {
291 if (BIH_LINENO(nextbih) && !BIH_LINENO(curbih))
292 BIH_LINENO(curbih) = BIH_LINENO(nextbih);
293 delbih(nextbih); /* remove empty block */
294 return curbih;
295 }
296
297 lastilt = BIH_ILTLAST(curbih);
298 if (lastilt && IL_TYPE(ILI_OPC(ILT_ILIP(lastilt))) == ILTY_BRANCH)
299 return 0;
300
301 /*
302 * Add the "first" ilt of the block to the end of the current
303 * block.
304 */
305 rdilts(curbih);
306 ILT_NEXT(lastilt) = firstilt;
307 ILT_PREV(firstilt) = lastilt;
308 ILT_PREV(0) = BIH_ILTLAST(nextbih);
309#if DEBUG
310 if (EXPDBG(9, 32))
311 fprintf(gbl.dbgfil, " block %d merged, label %d\n",
312 nextbih, label);
313#endif
314
315 iltx = lastilt;
316 while (iltx) {
317 if (ILT_DELEBB(iltx)) {
318 ILT_DELETE(iltx) = 1;
319 ILT_DELEBB(iltx) = 0;
320#if DEBUG
321 if (EXPDBG(9, 32))
322 fprintf(gbl.dbgfil,
323 " ilt %d: ILT_DELEBB->ILT_DELETE\n", iltx);
324#endif
325 }
326 iltx = ILT_PREV(iltx);
327 }
328
329 if (BIH_LINENO(nextbih) && !BIH_LINENO(curbih))
330 BIH_LINENO(curbih) = BIH_LINENO(nextbih);
331 /* update the BIH flags of the current block */
332 merge_bih_flags(curbih, nextbih);
333
334#if DEBUG
335 assert((BIH_PARSECT(curbih) ^ BIH_PARSECT(nextbih)) == 0,
336 "merge_bih:parsect,nonparsect", curbih, ERR_Severe);
337#endif
338
339 wrilts(curbih);
340
341 merge_rgset(curbih, nextbih, false);
342
343 /* remove the block from the BIH list */
344
345 delbih(nextbih);
346
347 return curbih;
348}

365/** \brief Merge block b2 into b1
366 */
367void
368merge_blks(int b1, int b2)
369{
370 int i;
371 int j;
372
373 rdilts(b1);
374 i = ILT_PREV(0); /* last ilt in first block */
375 j = BIH_ILTFIRST(b2); /* first ilt in second block */
376 if (i == 0) {
377 /* first block is empty: just copy contents of second block */
378 ILT_NEXT(0) = j;
379 ILT_PREV(0) = BIH_ILTLAST(b2);
380 } else if (j) {
381 /* first & second blocks are not empty */
382 ILT_NEXT(i) = j; /* link last ilt to first */
383 ILT_PREV(j) = i;
384 ILT_PREV(0) = BIH_ILTLAST(b2);
385 }
386 /* else first block nonempty & second block empty -- nothing to do */
387
388 wrilts(b1);
389
390 /* update necessary BIH flags */
391
392 BIH_FT(b1) = BIH_FT(b2);
393 BIH_EX(b1) = BIH_EX(b1) | BIH_EX(b2);
394 BIH_QJSR(b1) = BIH_QJSR(b1) | BIH_QJSR(b2);
395}

416/*
417 * get rid of useless splits between blocks;
418 * if we have two blocks b1 and its lexical successor b2
419 * b1 is fall-through, does not end in a jump
420 * neither has BIH_XT BIH_PL BIH_NOMERGE BIH_ENLAB
421 * BIH_GASM BIH_LAST BIH_PAR BIH_CS BIH_PARSECT
422 * merge the two blocks
423 */
424void
425unsplit(void)
426{
427 int bihx;
428#if DEBUG
429 /* convenient place for a segfault */
430 if (XBIT(4, 0x2000)) {
431 if (!XBIT(4, 0x1000) || gbl.func_count > 2) {
432 /* store to null pointer */
433 *badpointer1 = 99;
434 }
435 }
436 if (XBIT(4, 0x4000)) {
437 if (!XBIT(4, 0x1000) || gbl.func_count > 2) {
438 /* divide by zero */
439 badnumerator = badnumerator / baddenominator;
440 }
441 }
442 if (XBIT(4, 0x8000)) {
443 if (!XBIT(4, 0x1000) || gbl.func_count > 2) {
444 /* infinite loop */
445 while (badnumerator) {
446 badnumerator = (badnumerator < 1) | 3;
447 }
448 }
449 }
450#endif
451 for (bihx = gbl.entbih; bihx; bihx = BIH_NEXT(bihx)) {
452 int b;
453 if (BIH_LAST(bihx))
454 break;
455 for (b = bihx; b;) {
456 /* if 'bihx' ends in a branch or jump or call that can throw, stop here */
457 bihx = b;
458 if (!BIH_FT(bihx))
459 break;
460 if (BIH_ILTLAST(bihx) && ILT_BR_OR_CAN_THROW(BIH_ILTLAST(bihx)))
461 break;
462 b = merge_bih(bihx);
463 }
464 }
465} /* unsplit */

467/*
468 * Split blocks after an internal jump; that is, eliminate extended basic blocks
469 */
470void
471split_extended()
472{
473 int bihx, iltx, iltnext;
474 for (bihx = gbl.entbih; bihx; bihx = BIH_NEXT(bihx)) {
475 if (BIH_LAST(bihx))
476 break;
477 for (iltx = BIH_ILTFIRST(bihx); iltx; iltx = iltnext) {
478 int ilix, opc;
479 iltnext = ILT_NEXT(iltx);
480 ilix = ILT_ILIP(iltx);
481 opc = ILI_OPC(ilix);
482 if (iltnext && IL_TYPE(opc) == ILTY_BRANCH) {
483 /* split this block just after the branch.
484 * move ILTs after this one to the new block. */
485 int newbihx;
486 newbihx = addbih(bihx);
487 BIH_ILTFIRST(newbihx) = iltnext;
488 BIH_ILTLAST(newbihx) = BIH_ILTLAST(bihx);
489 BIH_ILTLAST(bihx) = iltx;
490 ILT_NEXT(iltx) = 0;
491 ILT_PREV(iltnext) = 0;
492 BIH_FT(newbihx) = BIH_FT(bihx);
493 BIH_FT(bihx) = 1;
494 BIH_EX(newbihx) = BIH_EX(bihx);
495 BIH_LDVOL(newbihx) = BIH_LDVOL(bihx);
496 BIH_STVOL(newbihx) = BIH_STVOL(bihx);
497 BIH_QJSR(newbihx) = BIH_QJSR(bihx);
498 BIH_SMOVE(newbihx) = BIH_SMOVE(bihx);
499 BIH_NOMERGE(newbihx) = BIH_NOMERGE(bihx);
500 BIH_PAR(newbihx) = BIH_PAR(bihx);
501 iltnext = 0;
502 }
503 }
504 }
