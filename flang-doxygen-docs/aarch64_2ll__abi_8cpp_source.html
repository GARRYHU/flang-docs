31/* AAPCS has the concept of a homogeneous aggregrate. It is an aggregate type
32 * where all the fundamental types are the same after flattening all structs
33 * and arrays. */
34struct arm_homogeneous_aggr {
35 LL_Module *module;
36 LL_Type *base_type;
37 unsigned base_bytes;
38};

55/* Return 1 if dtype is inconsistent with the homogeneous aggregate
56 * information pointed to by context. */
57static int
58update_homogeneous(void *context, DTYPE dtype, unsigned address,
59 int member_sptr)
60{
61 struct arm_homogeneous_aggr *ha = (struct arm_homogeneous_aggr *)context;
62 unsigned size;
63 LL_Type *llt;
64
65 dtype = DT_BASETYPE(dtype);
66
67 if (DTY(dtype) == TY_ARRAY)
68 dtype = (DTYPE)DTY(dtype + 1); // ???
69
70 switch (dtype) {
71 default:
72 break;
73 case DT_CMPLX:
74 dtype = DT_FLOAT;
75 break;
76 case DT_DCMPLX:
77 dtype = DT_DBLE;
78 break;
79 }
80
81 size = size_of(dtype);
82 llt = ll_convert_dtype(ha->module, dtype);
83
84 if (!ha->base_type) {
85 if (address != 0)
86 return 1;
87 ha->base_type = llt;
88 ha->base_bytes = size;
89 return 0;
90 }
91
92 /* Check if dtype is consistent with the existing base type. */
93 if (size != ha->base_bytes)
94 return 1;
95
96 if (!size || address % size != 0)
97 return 1;
98
99 /* Vector types just need matching sizes. Elements don't need to match. */
100 if (ha->base_type->data_type == LL_VECTOR && llt->data_type == LL_VECTOR)
101 return 0;
102
103 /* Other base types must be identical. */
104 return ha->base_type != llt;
105}

107/* Check if dtype is a VFP register candidate. Return the coercion type or NULL.
108 */
109static LL_Type *
110check_vfp(LL_Module *module, DTYPE dtype, struct arm_homogeneous_aggr *aggr)
111{
112 ISZ_T size = size_of(dtype);
113
114 /* Check if dtype is a homogeneous aggregate. */
115 if (visit_flattened_dtype(update_homogeneous, aggr, dtype, 0, 0))
116 return NULL;
117 if (!aggr->base_type)
118 return NULL;
119
120 /* A non-aggregated scalar will simply be copied to base_type. */
121 switch (aggr->base_type->data_type) {
122 case LL_FLOAT:
123 case LL_DOUBLE:
124 break;
125 case LL_VECTOR:
126 /* Only 64-bit or 128-bit vectors supported. */
127 if (aggr->base_bytes != 8 && aggr->base_bytes != 16)
128 return NULL;
129 break;
130 default:
131 return NULL;
132 }
133
134 /* We have a scalar or a homogeneous aggregate of the right type. The ABI
135 * supports one to four elements of the base type. */
136 if (size > 4 * aggr->base_bytes)
137 return NULL;
138
139 /* Single-element aggregate? */
140 if (size == aggr->base_bytes)
141 return aggr->base_type;
142
143 /* Multiple elements coerced to an array type. */
144 return ll_get_array_type(aggr->base_type, size / aggr->base_bytes, 0);
145}

147/* Classify an integer type for return or arg. */
148static enum LL_ABI_ArgKind
149classify_int(DTYPE dtype)
150{
151 /* Integer types smaller than a register must be sign/zero extended. */
152 if (size_of(dtype) < 4)
153 return DT_ISUNSIGNED(dtype) ? LL_ARG_ZEROEXT : LL_ARG_SIGNEXT;
154
155 return LL_ARG_DIRECT;
156}

164/* Classify common to args and return values. */
165static bool
166classify_common(LL_Module *module, LL_ABI_Info *abi, ARM_ABI_ArgInfo *arg,
167 DTYPE dtype)
168{
169 if (DT_ISINT(dtype)) {
170 arg->kind = classify_int(dtype);
171 return true;
172 }
173
174 /* Basic types can be returned in registers directly. Complex types also
175 * get handled correctly. */
176 if (dtype == DT_VOIDNONE || DT_ISSCALAR(dtype)) {
177 arg->kind = LL_ARG_DIRECT;
178 return true;
179 }
180
181 struct arm_homogeneous_aggr aggr = {module, NULL, 0};
182 LL_Type *haggr_lltype = check_vfp(module, dtype, &aggr);
183 if (haggr_lltype) {
184 arg->kind = LL_ARG_COERCE;
185 arg->type = haggr_lltype;
186 return true;
187 }
188
189 // AAPCS64: Arm 64 bit Architecture Procedure Call Standard
190 if (DTY(dtype) == TY_STRUCT || DTY(dtype) == TY_UNION) {
191 ISZ_T size = size_of(dtype);
192 if (size > 16) {
193 // AAPCS64: Stage B3
194 // If the argument is a composite type that is larger than 16 bytes, then
195 // the argument is copied to memory by the caller and the argument is
196 // replaced by a pointer to the copy
197 if (arg->is_return_val) {
198 arg->kind = LL_ARG_INDIRECT;
199 } else {
200 arg->kind = LL_ARG_INDIRECT_BUFFERED;
201 }
202 } else {
203 arg->kind = LL_ARG_COERCE;
204 if (arg->is_return_val) {
205 // Whether directly or indirectly, returned values are always passed
206 // through registers. However, unlike input arguments whose size has to
207 // be rounded up to the nearest 8 bytes (see LLObj_LocalBuffered), the
208 // type of the returned value has to match the size of the actual LHS.
209 // This is because the returned value is immediately stored into the
210 // local variable for the LHS and that store has to match the size of
211 // the local otherwise it will spill over the next local in the stack.
212 // We use a coercion type to signify that :
213 // - returned value is to be passed through registers
214 // - return value size matches that of LHS
215 // Example:
216 // {i8,i8,i8,i8,i32,i8} -> {i64, i8}
217 arg->type = ll_coercion_type(abi->module, dtype, size, 8);
218 } else {
219 // AAPCS64: Stage B4
220 // If the argument is a composite type then the size of the argument is
221 // rounded up to the nearest multiple of 8 bytes
222 //
223 // AAPCS64: Stage C14
224 // If the size of the argument is less than 8 bytes then the size of the
225 // argument is set to 8 bytes
226 arg->type = ll_create_basic_type(abi->module, LL_I64, 0);
227 if (size > 8) {
228 arg->type = ll_get_array_type(arg->type, ll_abi_num_regs(size), 0);
229 }
230 }
231 }
232 return true;
233 }
234 return false;
235}

237void
238ll_abi_classify_return_dtype(LL_ABI_Info *abi, DTYPE dtype)
239{
240 enum LL_BaseDataType bdt = LL_NOTYPE;
241 ARM_ABI_ArgInfo tmp_arg_info = {LL_ARG_UNKNOWN, NULL, true};
242
243 dtype = DT_BASETYPE(dtype);
244
245 if (classify_common(abi->module, abi, &tmp_arg_info, dtype)) {
246 update_arg_info(&(abi->arg[0]), &tmp_arg_info);
247 return;
248 }
249 /* Small structs can be returned in r0.
250 * FIXME: can also be returned in register pair of floating-point registers.
251 */
252 switch (size_of(dtype)) {
253 case 1:
254 bdt = LL_I8;
255 break;
256 case 2:
257 bdt = LL_I16;
258 break;
259 case 3:
260 case 4:
261 bdt = LL_I32;
262 break;
263 case 8:
264 bdt = LL_I64;
265 break;
266 }
267 if (bdt != LL_NOTYPE) {
268 abi->arg[0].kind = LL_ARG_COERCE;
269 abi->arg[0].type = ll_create_basic_type(abi->module, bdt, 0);
270 return;
271 }
272
273 /* Large types must be returned in memory via an sret pointer argument. */
274 abi->arg[0].kind = LL_ARG_INDIRECT;
275}

277void
278ll_abi_classify_arg_dtype(LL_ABI_Info *abi, LL_ABI_ArgInfo *arg, DTYPE dtype)
279{
280 ISZ_T size;
281 ARM_ABI_ArgInfo tmp_arg_info = {LL_ARG_UNKNOWN, NULL, false};
282
283 dtype = DT_BASETYPE(dtype);
284
285 if (classify_common(abi->module, abi, &tmp_arg_info, dtype)) {
286 update_arg_info(arg, &tmp_arg_info);
287 return;
288 }
289
290 /* All other arguments are coerced into an array of 32-bit registers. */
291 size = size_of(dtype);
292 arg->kind = LL_ARG_COERCE;
293 if (alignment(dtype) > 4 && size % 8 == 0) {
294 /* The coercion type needs to have the same alignment as the original type.
295 */
296 arg->type = ll_create_basic_type(abi->module, LL_I64, 0);
297 if (size > 8)
298 arg->type = ll_get_array_type(arg->type, size / 8, 0);
299 } else {
300 arg->type = ll_create_basic_type(abi->module, LL_I32, 0);
301 if (size > 4)
302 arg->type = ll_get_array_type(arg->type, (size + 3) / 4, 0);
303 }
304}

306unsigned
307ll_abi_classify_va_arg_dtype(LL_Module *module, DTYPE dtype, unsigned *num_gp,
308 unsigned *num_fp)
309{
310 ISZ_T size;
311 struct arm_homogeneous_aggr aggr = {module, NULL, 0};
312 LL_Type *haggr_lltype;
313
314 size = size_of(dtype);
315 *num_gp = 0;
316 *num_fp = 0;
317
318 haggr_lltype = check_vfp(module, dtype, &aggr);
319 if (haggr_lltype) {
320 /*
321 Only one member per register. a struct of 4 32-bit floats is scattered
322 over 4 128-bit registers. Recomputing the number of members as the size
323 of the whole type / the size of a member.
324 __builtin_va_fpargs gathers these register back into a temporary that
325 matches the original layout */
326 *num_fp = size / aggr.base_bytes;
327 return aggr.base_bytes;
328 }
329
330 if (DT_ISINT(dtype) || DTY(dtype) == TY_PTR) {
331 *num_gp = ll_abi_num_regs(size);
332 return 0;
333 }
334
335 if (DTY(dtype) == TY_STRUCT || DTY(dtype) == TY_UNION) {
336 if (size > 16) {
337 } else {
338 *num_gp = ll_abi_num_regs(size);
339 }
340 return 0;
341 }
342
343 return 0;
