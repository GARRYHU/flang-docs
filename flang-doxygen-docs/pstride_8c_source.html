104/*
105 * determine whether we have pointer target info for this
106 * symbol at this statement; if so, merge that into what we have
107 */
108static void
109do_targets(int stdx, int sptr)
110{
111 if (!SEEN_UNK_TARGET(sptr)) {
112 int c, tag, tid;
113 c = 0;
114 while (pta_target(stdx, sptr, &tag, &tid)) {
115 int t;
116 ++c;
117 if (tag == FIST_UNK) {
118#if DEBUG
119 if (DBGBIT(10, 0x200000)) {
120 fprintf(gbl.dbgfil, " set UNKTARGET for %d:%s at std %d\n", sptr,
121 SYMNAME(sptr), stdx);
122 }
123#endif
124 SET_UNK_TARGET(sptr);
125 return;
126 }
127 for (t = info[sptr].fistptr; t; t = FIST_NEXT(t)) {
128 if (FIST_TYPE(t) == tag && FIST_ID(t) == tid)
129 break;
130 }
131 if (!t) {
132 t = fist.stg_avail++;
133 OPT_NEED(fist, FIST, 100);
134 FIST_TYPE(t) = tag;
135 FIST_ID(t) = tid;
136 FIST_NEXT(t) = info[sptr].fistptr;
137 info[sptr].fistptr = t;
138 }
139 }
140 if (c) {
141 SET_TARGET(sptr);
142 } else {
143 SET_UNK_TARGET(sptr);
144#if DEBUG
145 if (DBGBIT(10, 0x200000)) {
146 fprintf(gbl.dbgfil, " set UNKTARGET (no targets) for %d:%s at std %d\n",
147 sptr, SYMNAME(sptr), stdx);
148 }
149#endif
150 }
151 }
152} /* do_targets */

154/*
155 * analyze one symbol referenced in this statement
156 */
157static void
158analyze(int sptr, int stdx)
159{
160 if (sptr && deref) {
161 switch (STYPEG(sptr)) {
162 case ST_VAR:
163 case ST_ARRAY:
164 if (ALLOCATTRG(sptr)) {
165 /* allocatables are always aligned */
166 if (!VISITG(sptr)) {
167 info[sptr].flags = 0;
168 info[sptr].fistptr = 0;
169 info[sptr].next = first;
170 first = sptr;
171 VISITP(sptr, 1);
172 }
173 SET_QALN(sptr);
174 SET_STRIDE1(sptr);
175 if (XBIT(53, 2))
176 do_targets(stdx, sptr);
177 } else if (deref && POINTERG(sptr) && XBIT(53, 2)) {
178#if DEBUG
179 if (DBGBIT(10, 0x200000)) {
180 fprintf(dfile, "--deref=%d sptr=%d:%s std=%d ", deref, sptr,
181 SYMNAME(sptr), stdx);
182 printast(STD_AST(stdx));
183 fprintf(dfile, "\n");
184 putstdpta(stdx);
185 }
186#endif
187 if (!VISITG(sptr)) {
188 info[sptr].flags = 0;
189 info[sptr].fistptr = 0;
190 info[sptr].next = first;
191 first = sptr;
192 VISITP(sptr, 1);
193 }
194 if (!SEEN_NON_QALN(sptr)) {
195 if (pta_aligned(stdx, sptr)) {
196 SET_QALN(sptr);
197 } else {
198 SET_NON_QALN(sptr);
199 }
200 }
201 if (!SEEN_NON_STRIDE1(sptr)) {
202 if (pta_stride1(stdx, sptr)) {
203 SET_STRIDE1(sptr);
204 } else {
205 SET_NON_STRIDE1(sptr);
206 }
207 }
208 if (XBIT(53, 2))
209 do_targets(stdx, sptr);
210 }
211 break;
212 case ST_MEMBER:
213 if (ALLOCATTRG(sptr)) {
214 /* allocatables are always aligned */
215 if (!VISITG(sptr)) {
216 info[sptr].flags = 0;
217 info[sptr].fistptr = 0;
218 info[sptr].next = first;
219 first = sptr;
220 VISITP(sptr, 1);
221 }
222 SET_QALN(sptr);
223 SET_STRIDE1(sptr);
224 }
225 else if (deref && POINTERG(sptr) && XBIT(53, 2)) {
226 if (!VISITG(sptr)) {
227 info[sptr].flags = 0;
228 info[sptr].fistptr = 0;
229 info[sptr].next = first;
230 first = sptr;
231 VISITP(sptr, 1);
232 }
233 if (!SEEN_NON_QALN(sptr)) {
234 if (pta_aligned(stdx, sptr)) {
235 SET_QALN(sptr);
236 } else {
237 SET_NON_QALN(sptr);
238 }
239 }
240 if (!SEEN_NON_STRIDE1(sptr)) {
241 if (pta_stride1(stdx, sptr)) {
242 SET_STRIDE1(sptr);
243 } else {
244 SET_NON_STRIDE1(sptr);
245 }
246 }
247 if (XBIT(53, 2))
248 do_targets(stdx, sptr);
249 }
250 break;
251 default:;
252 }
253 }
254} /* analyze */

256/*
257 * recursive, depth-first (parent, then all children) traversal of expression
258 * tree.
259 * special handling for some types of expressions, like procedure calls.
260 */
261static int
262_findrefs(int astx, int *pstdx)
263{
264 int savederef;
265 int asd, ndim, i;
266 int args, argcnt, a;
267
268 savederef = deref;
269
270 switch (A_TYPEG(astx)) {
271 case A_ID:
272 analyze(A_SPTRG(astx), *pstdx);
273 break;
274 case A_ALLOC:
275 /* the object being allocated/deallocated is not interesting */
276 deref = 0;
277 ast_traverse(A_SRCG(astx), _findrefs, NULL, pstdx);
278 /* the status, if given, is interesting */
279 deref = 1;
280 if (A_LOPG(astx))
281 ast_traverse(A_LOPG(astx), _findrefs, NULL, pstdx);
282 deref = savederef;
283 return 1;
284 case A_SUBSCR:
285 ast_traverse(A_LOPG(astx), _findrefs, NULL, pstdx);
286 /* all subscripts are interesting */
287 asd = A_ASDG(astx);
288 ndim = ASD_NDIM(asd);
289 deref = 1;
290 for (i = 0; i < ndim; ++i)
291 ast_traverse(ASD_SUBS(asd, i), _findrefs, NULL, pstdx);
292 deref = savederef;
293 return 1;
294 case A_SUBSTR:
295 ast_traverse(A_LOPG(astx), _findrefs, NULL, pstdx);
296 /* all substring arguments are interesting */
297 deref = 1;
298 if (A_LEFTG(astx))
299 ast_traverse(A_LEFTG(astx), _findrefs, NULL, pstdx);
300 if (A_RIGHTG(astx))
301 ast_traverse(A_RIGHTG(astx), _findrefs, NULL, pstdx);
302 deref = savederef;
303 return 1;
304 case A_ICALL:
305 /* intrinsic call, see if it is ptr assignment */
306 if (A_OPTYPEG(astx) == I_PTR2_ASSIGN) {
307 /* pointer assignment, 1st argument is not interesting */
308 args = A_ARGSG(astx);
309 deref = 0;
310 ast_traverse(ARGT_ARG(args, 0), _findrefs, NULL, pstdx);
311 deref = 1;
312 ast_traverse(ARGT_ARG(args, 1), _findrefs, NULL, pstdx);
313 deref = savederef;
314 return 1;
315 } else if (A_OPTYPEG(astx) == I_NULLIFY) {
316 /* pointer nullify */
317 args = A_ARGSG(astx);
318 deref = 0;
319 ast_traverse(ARGT_ARG(args, 0), _findrefs, NULL, pstdx);
320 deref = savederef;
321 return 1;
322 }
323 break;
324 case A_CALL:
325 case A_FUNC:
326 /* look at any expression arguments */
327 args = A_ARGSG(astx);
328 argcnt = A_ARGCNTG(astx);
329 for (a = 0; a < argcnt; ++a) {
330 deref = 0;
331 ast_traverse(ARGT_ARG(args, a), _findrefs, NULL, pstdx);
332 }
333 deref = savederef;
334 return 1;
335 }
336
337 deref = 1;
338 return 0;
339} /* _findrefs */

341/*
342 * go through all statements, visit all expressions
343 * for any expression, find references to any pointers.
344 * determine whether any of those references might NOT be to
345 * aligned memory, or contiguous memory, or leftmost-stride-1 memory.
346 */
347void
348pstride_analysis(void)
349{
350 int stdx, astx, sptr;
351#if DEBUG
352 dfile = gbl.dbgfil ? gbl.dbgfil : stderr;
353#endif
354 NEW(info, INFO, stb.stg_avail);
355 if (XBIT(53, 2)) {
356 points_to(); /* pointsto.c */
357 OPT_ALLOC(fist, FIST, 100);
358 fist.stg_avail = 1;
359 }
360 first = 0;
361 for (sptr = stb.firstosym; sptr < stb.stg_avail; ++sptr)
362 VISITP(sptr, 0);
363#if DEBUG
364 if (DBGBIT(10, 0x400000)) {
365 dstdpa();
366 }
367#endif
368 /* go through all statements, go through all expressions */
369 for (stdx = STD_NEXT(0); stdx != 0; stdx = STD_NEXT(stdx)) {
370 astx = STD_AST(stdx);
371 ast_visit(1, 1);
372 ast_traverse(astx, _findrefs, NULL, &stdx);
373 ast_unvisit();
374 }
375 if (XBIT(53, 2)) {
376 f90_fini_pointsto(); /* pointsto.c */
377 }
378 count = 0;
379 for (sptr = first; sptr > 0; sptr = info[sptr].next) {
380 int ptr, sdsc;
381 VISITG(sptr) = 0;
382 if (SEEN_QALN(sptr) && !SEEN_NON_QALN(sptr)) {
383 ptr = MIDNUMG(sptr);
384 if (ptr)
385 TQALNP(ptr, 1);
386 }
387 if (SEEN_STRIDE1(sptr) && !SEEN_NON_STRIDE1(sptr)) {
388 SDSCS1P(sptr, 1);
389 sdsc = SDSCG(sptr);
390 if (sdsc)
391 SDSCS1P(sdsc, 1);
392 }
393 if (SEEN_TARGET(sptr) && !SEEN_UNK_TARGET(sptr)) {
394 ++count;
395 }
396#if DEBUG
397 if (DBGBIT(10, 0x200000)) {
398 dsym(sptr);
399 }
400#endif
401 }
402 if (XBIT(53, 2) && count) {
403 OPT_ALLOC(fistlist, FISTLIST, count + 1);
404 fistlist.stg_avail = 1;
405 for (sptr = first; sptr > 0; sptr = info[sptr].next) {
406 if (SEEN_TARGET(sptr) && !SEEN_UNK_TARGET(sptr)) {
407 FLIST_SPTR(fistlist.stg_avail) = sptr;
408 FLIST_PTR(fistlist.stg_avail) = info[sptr].fistptr;
409#if DEBUG
410 if (DBGBIT(10, 0x200000)) {
411 int t;
412 fprintf(gbl.dbgfil, "FIS targets for %s :", SYMNAME(sptr));
413 for (t = FLIST_PTR(fistlist.stg_avail); t; t = FIST_NEXT(t)) {
414 switch (FIST_TYPE(t)) {
415 case FIST_UNK:
416 fprintf(gbl.dbgfil, " unk");
417 break;
418 case FIST_LDYN:
419 fprintf(gbl.dbgfil, " ldyn(%d)", FIST_ID(t));
420 break;
421 case FIST_GDYN:
422 fprintf(gbl.dbgfil, " gdyn(%d)", FIST_ID(t));
423 break;
424 case FIST_NLOC:
425 fprintf(gbl.dbgfil, " nloc(%d)", FIST_ID(t));
426 break;
427 case FIST_PSYM:
428 fprintf(gbl.dbgfil, " %d:%s", FIST_ID(t), SYMNAME(FIST_ID(t)));
429 break;
430 case FIST_ISYM:
431 fprintf(gbl.dbgfil, " %d:%s?", FIST_ID(t), SYMNAME(FIST_ID(t)));
432 break;
433 default:
434 fprintf(gbl.dbgfil, " ??%d", FIST_ID(t));
435 break;
436 }
437 }
438 fprintf(gbl.dbgfil, "\n");
439 }
440#endif
441 ++fistlist.stg_avail;
442 }
443 }
444 }
445 FREE(info);
446 first = 0;
447 count = 0;
448} /* pstride_analysis */

450/*
451 * put any discovered flow-insensitive pointer target information
452 * out to the information file
453 */
454void
455lower_pstride_info(FILE *lowerfile)
456{
457 int f, t, sptr;
458 if (fistlist.stg_base == NULL || fist.stg_base == NULL) {
459 return;
460 }
461 for (f = 1; f < fistlist.stg_avail; ++f) {
462 for (t = FLIST_PTR(f); t; t = FIST_NEXT(t)) {
463 sptr = FLIST_SPTR(f);
464 if (sptr && MIDNUMG(sptr)) {
465 switch (FIST_TYPE(t)) {
466 case FIST_UNK:
467 default:
468 fprintf(lowerfile, "info:%d T type:1 id:0\n", MIDNUMG(sptr));
469 break;
470 case FIST_LDYN:
471 fprintf(lowerfile, "info:%d T type:2 id:%d\n", MIDNUMG(sptr),
472 FIST_ID(t));
473 break;
474 case FIST_GDYN:
475 fprintf(lowerfile, "info:%d T type:3 id:%d\n", MIDNUMG(sptr),
476 FIST_ID(t));
477 break;
478 case FIST_NLOC:
479 fprintf(lowerfile, "info:%d T type:4 id:%d\n", MIDNUMG(sptr),
480 FIST_ID(t));
481 break;
482 case FIST_PSYM:
483 fprintf(lowerfile, "info:%d T type:5 id:%d\n", MIDNUMG(sptr),
484 FIST_ID(t));
485 break;
486 case FIST_ISYM:
487 fprintf(lowerfile, "info:%d T type:6 id:%d\n", MIDNUMG(sptr),
488 FIST_ID(t));
489 break;
490 }
491 }
492 }
493 }
494} /* lower_pstride_info */
