17void ENTFTN(CSHIFTS, cshifts)(void *rb, /* result base */
18 void *ab, /* array base */
19 __INT_T *sb, /* shift base */
20 __INT_T *db, /* dimension */
21 F90_Desc *rs, /* result descriptor */
22 F90_Desc *as, /* array descriptor */
23 F90_Desc *ss, /* shift descriptor */
24 F90_Desc *ds) /* dim descriptor */
25{
26 DECL_HDR_VARS(ac);
27 DECL_HDR_VARS(rc);
28 DECL_DIM_PTRS(ad);
29 DECL_DIM_PTRS(rd);
30 char *ap, *rp;
31 chdr *c1, *c2;
32 __INT_T aflags, albase, apbase, arepli, ascoff;
33 __INT_T rflags, rlbase, rpbase, rrepli, rscoff;
34 __INT_T aolb[MAXDIMS], aoub[MAXDIMS];
35 __INT_T rolb[MAXDIMS], roub[MAXDIMS];
36 __INT_T dim, extent, i, sabs, shift;
37
38 shift = *sb;
39 dim = *db;
40
41#if defined(DEBUG)
42 if (__fort_test & DEBUG_CSHF) {
43 printf("%d r", GET_DIST_LCPU);
44 I8(__fort_show_section)(rs);
45 printf("@%x = CSHIFT(a", rb);
46 I8(__fort_show_section)(as);
47 printf("@%x, shift=%d, dim=%d)\n", ab, shift, dim);
48 }
49#endif
50
51 /* compute the net positive (left) shift amount */
52
53 SET_DIM_PTRS(ad, as, dim - 1);
54 extent = F90_DPTR_EXTENT_G(ad);
55 if (extent < 0)
56 return;
57
58 sabs = shift % extent;
59 if (sabs < 0)
60 sabs += extent;
61
62 /* copy straight across if net shift amount is zero */
63
64 if (sabs == 0) {
65 rp = (char *)rb + DIST_SCOFF_G(rs) * F90_LEN_G(rs);
66 ap = (char *)ab + DIST_SCOFF_G(as) * F90_LEN_G(as);
67 c1 = I8(__fort_copy)(rp, ap, rs, as, NULL);
68 __fort_doit(c1);
69 __fort_frechn(c1);
70 return;
71 }
72
73 /* form section descriptors to describe shift. */
74
75 __DIST_INIT_SECTION(ac, F90_RANK_G(as), as);
76 __DIST_INIT_SECTION(rc, F90_RANK_G(as), rs);
77
78 for (i = 1; i <= F90_RANK_G(as); ++i) {
79 if (i == dim)
80 continue;
81 SET_DIM_PTRS(ad, as, i - 1);
82 I8(__fort_set_section)(ac, i, as, i, F90_DPTR_LBOUND_G(ad),
83 DPTR_UBOUND_G(ad), 1);
84 SET_DIM_PTRS(rd, rs, i - 1);
85 I8(__fort_set_section)(rc, i, rs, i, F90_DPTR_LBOUND_G(rd),
86 DPTR_UBOUND_G(rd), 1);
87 }
88
89 /* save descriptor fields affected by set/finish_section */
90
91 aflags = F90_FLAGS_G(ac);
92 albase = F90_LBASE_G(ac);
93 apbase = DIST_PBASE_G(ac);
94 arepli = DIST_REPLICATED_G(ac);
95 ascoff = DIST_SCOFF_G(ac);
96 for (i = F90_RANK_G(ac); --i >= 0;) {
97 aolb[i] = DIST_DIM_OLB_G(ac, i);
98 aoub[i] = DIST_DIM_OUB_G(ac, i);
99 }
100 rflags = F90_FLAGS_G(rc);
101 rlbase = F90_LBASE_G(rc);
102 rpbase = DIST_PBASE_G(rc);
103 rrepli = DIST_REPLICATED_G(rc);
104 rscoff = DIST_SCOFF_G(rc);
105 for (i = F90_RANK_G(rc); --i >= 0;) {
106 rolb[i] = DIST_DIM_OLB_G(rc, i);
107 roub[i] = DIST_DIM_OUB_G(rc, i);
108 }
109
110 /* move upper section to lower */
111
112 SET_DIM_PTRS(ad, as, dim - 1);
113 SET_DIM_PTRS(rd, rs, dim - 1);
114
115 I8(__fort_set_section)(ac, dim, as, dim, F90_DPTR_LBOUND_G(ad) + sabs,
116 DPTR_UBOUND_G(ad), 1);
117 I8(__fort_finish_section)((ac));
118 I8(__fort_set_section)(rc, dim, rs, dim, F90_DPTR_LBOUND_G(rd),
119 DPTR_UBOUND_G(rd) - sabs, 1);
120 I8(__fort_finish_section)((rc));
121
122 rp = (char *)rb + DIST_SCOFF_G(rc) * F90_LEN_G(rc);
123 ap = (char *)ab + DIST_SCOFF_G(ac) * F90_LEN_G(ac);
124 c1 = I8(__fort_copy)(rp, ap, rc, ac, NULL);
125
126 /* restore descriptor fields */
127
128 F90_FLAGS_P(ac, aflags);
129 F90_LBASE_P(ac, albase);
130 DIST_PBASE_P(ac, apbase);
131 DIST_REPLICATED_P(ac, arepli);
132 DIST_SCOFF_P(ac, ascoff);
133 for (i = F90_RANK_G(ac); --i >= 0;) {
134 DIST_DIM_OLB_P(ac, i, aolb[i]);
135 DIST_DIM_OUB_P(ac, i, aoub[i]);
136 }
137 DIST_CACHED_P(ac, 0);
138
139 F90_FLAGS_P(rc, rflags);
140 F90_LBASE_P(rc, rlbase);
141 DIST_PBASE_P(rc, rpbase);
142 DIST_REPLICATED_P(rc, rrepli);
143 DIST_SCOFF_P(rc, rscoff);
144 for (i = F90_RANK_G(rc); --i >= 0;) {
145 DIST_DIM_OLB_P(rc, i, rolb[i]);
146 DIST_DIM_OUB_P(rc, i, roub[i]);
147 }
148 DIST_CACHED_P(rc, 0);
149
150 /* move lower section to upper */
151
152 I8(__fort_set_section)(ac, dim, as, dim, F90_DPTR_LBOUND_G(ad),
153 DPTR_UBOUND_G(ad) - (extent - sabs), 1);
154 I8(__fort_finish_section)((ac));
155 I8(__fort_set_section)(rc, dim, rs, dim,
156 F90_DPTR_LBOUND_G(rd) + (extent - sabs),
157 DPTR_UBOUND_G(rd), 1);
158 I8(__fort_finish_section)((rc));
159
160 rp = (char *)rb + DIST_SCOFF_G(rc) * F90_LEN_G(rc);
161 ap = (char *)ab + DIST_SCOFF_G(ac) * F90_LEN_G(ac);
162 c2 = I8(__fort_copy)(rp, ap, rc, ac, NULL);
163
164 c1 = __fort_chain_em_up(c1, c2);
165 __fort_doit(c1);
166 __fort_frechn(c1);
167}

199/* loop over shift array, constructing rank 1 descriptors for each
200 corresponding vector section of the array and shifting them the
201 selected amount */
202
203void I8(cshift_loop)(void *rb, /* result base */
204 void *ab, /* array base */
205 __INT_T *sb, /* shift base */
206 __INT_T shift_dim, /* dimension to shift */
207 F90_Desc *rs, /* result descriptor */
208 F90_Desc *as, /* array descriptor */
209 F90_Desc *ss, /* shift descriptor */
210 F90_Desc *rc, /* result subsection descr. */
211 F90_Desc *ac, /* array subsection descr. */
212 __INT_T soff, /* shift index */
213 __INT_T loop_dim) /* current shift dim */
214{
215 DECL_DIM_PTRS(ad);
216 DECL_DIM_PTRS(rd);
217 DECL_DIM_PTRS(sd);
218 chdr *c1, *c2;
219 char *ap, *rp;
220 __INT_T aflags, albase, apbase, arepli, ascoff;
221 __INT_T rflags, rlbase, rpbase, rrepli, rscoff;
222 __INT_T aolb[MAXDIMS], aoub[MAXDIMS];
223 __INT_T rolb[MAXDIMS], roub[MAXDIMS];
224 __INT_T aflags2, albase2, apbase2, arepli2, ascoff2;
225 __INT_T rflags2, rlbase2, rpbase2, rrepli2, rscoff2;
226 __INT_T aolb2[MAXDIMS], aoub2[MAXDIMS];
227 __INT_T rolb2[MAXDIMS], roub2[MAXDIMS];
228 __INT_T array_dim, extent, i, sabs, shift;
229 __INT_T ai, al, au, ri, rl, ru, sstr;
230
231 /* shift rank is one dimension less than result/array rank */
232
233 array_dim = loop_dim; /* result/array dim */
234 if (loop_dim >= shift_dim)
235 ++array_dim;
236
237 SET_DIM_PTRS(rd, rs, array_dim - 1);
238 SET_DIM_PTRS(ad, as, array_dim - 1);
239 SET_DIM_PTRS(sd, ss, loop_dim - 1);
240
241 ri = F90_DPTR_LBOUND_G(rd);
242 ai = F90_DPTR_LBOUND_G(ad);
243 sstr = F90_DPTR_SSTRIDE_G(sd) * F90_DPTR_LSTRIDE_G(sd);
244 soff += (F90_DPTR_SSTRIDE_G(sd) * F90_DPTR_LBOUND_G(sd) +
245 F90_DPTR_SOFFSET_G(sd)) *
246 F90_DPTR_LSTRIDE_G(sd);
247
248 /* save descriptor fields affected by set_single */
249
250 aflags = F90_FLAGS_G(ac);
251 albase = F90_LBASE_G(ac);
252 apbase = DIST_PBASE_G(ac);
253 arepli = DIST_REPLICATED_G(ac);
254 ascoff = DIST_SCOFF_G(ac);
255 for (i = F90_RANK_G(ac); --i >= 0;) {
256 aolb[i] = DIST_DIM_OLB_G(ac, i);
257 aoub[i] = DIST_DIM_OUB_G(ac, i);
258 }
259 rflags = F90_FLAGS_G(rc);
260 rlbase = F90_LBASE_G(rc);
261 rpbase = DIST_PBASE_G(rc);
262 rrepli = DIST_REPLICATED_G(rc);
263 rscoff = DIST_SCOFF_G(rc);
264 for (i = F90_RANK_G(rc); --i >= 0;) {
265 rolb[i] = DIST_DIM_OLB_G(rc, i);
266 roub[i] = DIST_DIM_OUB_G(rc, i);
267 }
268
269 for (; ri <= DPTR_UBOUND_G(rd); ++ri, ++ai, soff += sstr) {
270 I8(__fort_set_single)(rc, rs, array_dim, ri, __SCALAR);
271 I8(__fort_set_single)(ac, as, array_dim, ai, __SCALAR);
272
273 if (loop_dim > 1) {
274 I8(cshift_loop)(rb, ab, sb, shift_dim,
275 rs, as, ss, rc, ac, soff, loop_dim-1);
276 } else {
277 shift = sb[soff];
278
279 al = F90_DIM_LBOUND_G(as, shift_dim - 1);
280 au = DIM_UBOUND_G(as, shift_dim - 1);
281 rl = F90_DIM_LBOUND_G(rs, shift_dim - 1);
282 ru = DIM_UBOUND_G(rs, shift_dim - 1);
283
284 /* compute the net positive (left) shift amount */
285
286 extent = au - al + 1;
287 if (extent < 0)
288 extent = 0;
289
290 sabs = shift % extent;
291 if (sabs < 0)
292 sabs += extent;
293
294 /* copy directly if net shift amount is zero */
295
296 if (sabs == 0) {
297 I8(__fort_set_section)(ac, 1, as, shift_dim, al, au, 1);
298 I8(__fort_finish_section)(ac);
299 I8(__fort_set_section)(rc, 1, rs, shift_dim, rl, ru, 1);
300 I8(__fort_finish_section)(rc);
301
302 rp = (char *)rb + DIST_SCOFF_G(rc) * F90_LEN_G(rs);
303 ap = (char *)ab + DIST_SCOFF_G(ac) * F90_LEN_G(as);
304 c1 = I8(__fort_copy)(rp, ap, rc, ac, NULL);
305 } else {
306
307 /* save descriptor fields affected by set_section */
308
309 aflags2 = F90_FLAGS_G(ac);
310 albase2 = F90_LBASE_G(ac);
311 apbase2 = DIST_PBASE_G(ac);
312 arepli2 = DIST_REPLICATED_G(ac);
313 ascoff2 = DIST_SCOFF_G(ac);
314 for (i = F90_RANK_G(ac); --i >= 0;) {
315 aolb2[i] = DIST_DIM_OLB_G(ac, i);
316 aoub2[i] = DIST_DIM_OUB_G(ac, i);
317 }
318 rflags2 = F90_FLAGS_G(rc);
319 rlbase2 = F90_LBASE_G(rc);
320 rpbase2 = DIST_PBASE_G(rc);
321 rrepli2 = DIST_REPLICATED_G(rc);
322 rscoff2 = DIST_SCOFF_G(rc);
323 for (i = F90_RANK_G(rc); --i >= 0;) {
324 rolb2[i] = DIST_DIM_OLB_G(rc, i);
325 roub2[i] = DIST_DIM_OUB_G(rc, i);
326 }
327
328 /* move upper section to lower */
329
330 I8(__fort_set_section)(ac, 1, as, shift_dim, al + sabs, au, 1);
331 I8(__fort_finish_section)(ac);
332 I8(__fort_set_section)(rc, 1, rs, shift_dim, rl, ru - sabs, 1);
333 I8(__fort_finish_section)(rc);
334
335 rp = (char *)rb + DIST_SCOFF_G(rc) * F90_LEN_G(rs);
336 ap = (char *)ab + DIST_SCOFF_G(ac) * F90_LEN_G(as);
337 c1 = I8(__fort_copy)(rp, ap, rc, ac, NULL);
338
339 /* restore descriptor fields */
340
341 F90_FLAGS_P(ac, aflags2);
342 F90_LBASE_P(ac, albase2);
343 DIST_PBASE_P(ac, apbase2);
344 DIST_REPLICATED_P(ac, arepli2);
345 DIST_SCOFF_P(ac, ascoff2);
346 for (i = F90_RANK_G(ac); --i >= 0;) {
347 DIST_DIM_OLB_P(ac, i, aolb2[i]);
348 DIST_DIM_OUB_P(ac, i, aoub2[i]);
349 }
350 DIST_CACHED_P(ac, 0);
351
352 F90_FLAGS_P(rc, rflags2);
353 F90_LBASE_P(rc, rlbase2);
354 DIST_PBASE_P(rc, rpbase2);
355 DIST_REPLICATED_P(rc, rrepli2);
356 DIST_SCOFF_P(rc, rscoff2);
357 for (i = F90_RANK_G(rc); --i >= 0;) {
358 DIST_DIM_OLB_P(rc, i, rolb2[i]);
359 DIST_DIM_OUB_P(rc, i, roub2[i]);
360 }
361 DIST_CACHED_P(rc, 0);
362
363 /* move lower section to upper */
364
365 I8(__fort_set_section)(ac, 1, as, shift_dim,
366 al, au - (extent - sabs), 1);
367 I8(__fort_finish_section)(ac);
368 I8(__fort_set_section)(rc, 1, rs, shift_dim,
369 rl + (extent - sabs), ru, 1);
370 I8(__fort_finish_section)(rc);
371
372 rp = (char *)rb + DIST_SCOFF_G(rc) * F90_LEN_G(rs);
373 ap = (char *)ab + DIST_SCOFF_G(ac) * F90_LEN_G(as);
374 c2 = I8(__fort_copy)(rp, ap, rc, ac, NULL);
375
376 c1 = __fort_chain_em_up(c1, c2);
377 }
378 __fort_doit(c1);
379 __fort_frechn(c1);
380 }
381
382 /* restore descriptor fields */
383
384 F90_FLAGS_P(ac, aflags);
385 F90_LBASE_P(ac, albase);
386 DIST_PBASE_P(ac, apbase);
387 DIST_REPLICATED_P(ac, arepli);
388 DIST_SCOFF_P(ac, ascoff);
389 for (i = F90_RANK_G(ac); --i >= 0;) {
390 DIST_DIM_OLB_P(ac, i, aolb[i]);
391 DIST_DIM_OUB_P(ac, i, aoub[i]);
392 }
393 DIST_CACHED_P(ac, 0);
394
395 F90_FLAGS_P(rc, rflags);
396 F90_LBASE_P(rc, rlbase);
397 DIST_PBASE_P(rc, rpbase);
398 DIST_REPLICATED_P(rc, rrepli);
399 DIST_SCOFF_P(rc, rscoff);
400 for (i = F90_RANK_G(rc); --i >= 0;) {
401 DIST_DIM_OLB_P(rc, i, rolb[i]);
402 DIST_DIM_OUB_P(rc, i, roub[i]);
403 }
404 DIST_CACHED_P(rc, 0);
405 }
