8/**
9 \file
10 \brief Functions for dealing with lexical scopes and the lifetimes
11 of scoped variables.
12
13
14 A lexical scope is represented by an ST_BLOCK symbol table entry. The
15 scope's ENCLFUNC field points to either the enclosing scope or the
16 containing function. Scopes have two pairs of start/end labels:
17
18 STARTLAB / ENDLAB: These labels are only inserted when generating debug
19 info. The are tagged as volatile so they won't be removed, and they can be
20 used to generate complete DWARF lexical scope information.
21
22 BEGINSCOPELAB / ENDSCOPELAB: These labels are inserted in-stream as ilts
23 with an IL_LABEL instruction. The corresponding ST_LABEL symbol table
24 entries are tagged with the corresponding BEGINSCOPE / ENDSCOPE flags, and
25 the ENCLFUNC field points back to the ST_BLOCK entry.
26
27 Every local variable has a scope indicated by its ENCLFUNC field. Variables
28 with function scope point to the function's symbol table entry instead of an
29 ST_BLOCK entry.
30*/
31

48/**
49 \brief Is sptr a symbol table entry that can be the root of a scope
50 tree? For C, this is functions. For Fortran, this includes entries
51 and procedures.
52 */
53static bool
54is_scope_root(SPTR sptr)
55{
56 switch (STYPEG(sptr)) {
57 case ST_PROC:
58 case ST_ENTRY:
59 return true;
60 default:
61 return false;
62 }
63}
64
65/**
66 \brief Return true if the scope 'outer' contains the scope 'inner'.
67
68 Both outer and inner must be valid ST_BLOCK or ST_FUNC symbol table
69 entries.
70 */
71bool
72scope_contains(SPTR outer, SPTR inner)
73{
74 ICHECK(VALIDSYM(outer) &&
75 (is_scope_root(outer) || STYPEG(outer) == ST_BLOCK));
76 ICHECK(VALIDSYM(inner) &&
77 (is_scope_root(inner) || STYPEG(inner) == ST_BLOCK));
78
79 while (STYPEG(inner) == ST_BLOCK) {
80 if (inner == outer)
81 return true;
82 inner = ENCLFUNCG(inner);
83 }
84
85 return inner == outer;
86}

88/**
89 \brief Return true if label_sptr is a scope label.
90 */
91bool
92is_scope_label(int label_sptr)
93{
94 if (!VALIDSYM(label_sptr) || STYPEG(label_sptr) != ST_LABEL)
95 return false;
96
97 return BEGINSCOPEG(label_sptr) || ENDSCOPEG(label_sptr);
98}

126/**
127 \brief Verify the integrity of an ST_BLOCK symbol table entry and
128 its associated labels.
129 */
130static void
131verify_block(int block_sptr)
132{
133 int lab;
134
135 ICHECK(VALIDSYM(block_sptr));
136 ICHECK(STYPEG(block_sptr) == ST_BLOCK);
137
138 /* STARTLAB and ENDLAB are not required to be present. */
139 if (STARTLABG(block_sptr)) {
140 lab = STARTLABG(block_sptr);
141 ICHECK(VALIDSYM(lab));
142 ICHECK(STYPEG(lab) == ST_LABEL);
143 ICHECK(ENCLFUNCG(lab) == block_sptr);
144
145 /* These flags should go on the BEGINSCOPELAB / ENDSCOPELAB
146 labels, not here. */
147 ICHECK(!BEGINSCOPEG(lab));
148 ICHECK(!ENDSCOPEG(lab));
149 /* FIXME: Check ILIBLK. */
150 }
151
152 if (ENDLABG(block_sptr)) {
153 lab = ENDLABG(block_sptr);
154 ICHECK(VALIDSYM(lab));
155 ICHECK(STYPEG(lab) == ST_LABEL);
156 ICHECK(ENCLFUNCG(lab) == block_sptr);
157 ICHECK(!BEGINSCOPEG(lab));
158 ICHECK(!ENDSCOPEG(lab));
159 /* FIXME: Check ILIBLK. */
160 }
161
162 /* All blocks should have delineating labels. */
163 lab = BEGINSCOPELABG(block_sptr);
164 ICHECK(VALIDSYM(lab));
165 ICHECK(STYPEG(lab) == ST_LABEL);
166 ICHECK(ENCLFUNCG(lab) == block_sptr);
167 ICHECK(BEGINSCOPEG(lab));
168 ICHECK(!ENDSCOPEG(lab));
169 /* FIXME: Check ILIBLK. */
170
171 lab = ENDSCOPELABG(block_sptr);
172 ICHECK(VALIDSYM(lab));
173 ICHECK(STYPEG(lab) == ST_LABEL);
174 ICHECK(ENCLFUNCG(lab) == block_sptr);
175 ICHECK(!BEGINSCOPEG(lab));
176 ICHECK(ENDSCOPEG(lab));
177 /* FIXME: Check ILIBLK. */
178}

180static bool
181is_local_variable(int sptr)
182{
183 switch (STYPEG(sptr)) {
184 case ST_PLIST:
185 case ST_VAR:
186 case ST_ARRAY:
187 case ST_STRUCT:
188 case ST_UNION:
189 switch (SCG(sptr)) {
190 case SC_LOCAL:
191 /* Somebody who understands Fortran should probably revisit this
192 and check DINITG, SAVEG etc. */
193 return true;
194 /* Ignore SC_DUMMY. It's going to have function scope anyway. */
195 default:
196 return false;
197 }
198 default:
199 return false;
200 }
201}

203/**
204 \brief Collect the set of locals referenced by the ILI tree starting at ilix.
205
206 Loads and stores to local variables are always going to use an ACON
207 to compute the address, so just find all the symbols referenced by
208 ACONs.
209 */
210static void
211collect_referenced_locals(hashset_t symbs, int ilix)
212{
213 ILI_OP opc = ILI_OPC(ilix);
214
215 if (opc == IL_ACON) {
216 /* ACON opnd 1 is an ST_CONST sptr with DT_CPTR type. */
217 int sptr = CONVAL1G(ILI_OPND(ilix, 1));
218 if (sptr && is_local_variable(sptr))
219 hashset_replace(symbs, INT2HKEY(sptr));
220 } else {
221 int i;
222 for (i = IL_OPRS(opc); i > 0; i--)
223 if (IL_ISLINK(opc, i))
224 collect_referenced_locals(symbs, ILI_OPND(ilix, i));
225 }
226}

228/* Info about a single open scope. */
229struct scope_info {
230 /* ST_BLOCK for scope, or ST_FUNC/ST_PROC/ST_ENTRY for the function-level
231 scope. */
232 SPTR block;
233 /* The IL_LABEL ilt that closes this scope. */
234 int closing_ilt;
235};
236
237/* A stack of open scopes. */
238struct scope_stack {
239 /* scope[0..curr] are in use. */
240 int curr;
241 /* scope[0..outer] are perfectly nested. */
242 int outer;
243 /* Allocated entries in scope. */
244 int allocated;
245 struct scope_info *scope;
246 /* scope[outer].block from before closed scopes were popped. */
247 SPTR prev_top;
248};
249
250/**
251 \brief Initialize a scope stack with func as the outermost scope.
252 */
253static void
254init_scope_stack(struct scope_stack *ss, SPTR func)
255{
256 ss->curr = ss->outer = 0;
257 ss->allocated = 100;
258 NEW(ss->scope, struct scope_info, ss->allocated);
259 memset(&ss->scope[0], 0, sizeof(ss->scope[0]));
260 ss->scope[0].block = func;
261 ss->prev_top = SPTR_NULL;
262}
263
264static void
265push_block(struct scope_stack *ss, SPTR block)
266{
267 ++ss->curr;
268 NEED(ss->curr + 1, ss->scope, struct scope_info, ss->allocated,
269 ss->allocated + 100);
270 memset(&ss->scope[ss->curr], 0, sizeof(ss->scope[0]));
271 ss->scope[ss->curr].block = block;
272 dbgprintf(" Pushing scope %d (%s) at level %d: enclfunc=%d (%s)\n", block,
273 SYMNAME(block), ss->curr, ENCLFUNCG(block),
274 SYMNAME(ENCLFUNCG(block)));
275}
276
277static void
278pop_block(struct scope_stack *ss)
279{
280 dbgprintf(" Popping scope %d (%s) at level %d\n", ss->scope[ss->curr].block,
281 SYMNAME(ss->scope[ss->curr].block), ss->curr);
282 ICHECK(ss->curr > 0);
283 if (--ss->curr < ss->outer)
284 ss->outer = ss->curr;
285}

287/**
288 \brief Check labels opening scopes in bih, and collect all symbols
289 referenced in this block.
290
291 Add opened scopes to ss.
292 */
293static void
294verify_beginscopes(int bih, hashset_t locals, struct scope_stack *ss)
295{
296 int ilt;
297
298 /* All of the scopes pushed in this block must be sub-scopes of the
299 current top. */
300 SPTR outer_scope = ss->scope[ss->curr].block;
301
302 for (ilt = BIH_ILTFIRST(bih); ilt; ilt = ILT_NEXT(ilt)) {
303 int ilix = ILT_ILIP(ilt);
304 SPTR lab, blk;
305
306 if (ILI_OPC(ilix) != IL_LABEL) {
307 collect_referenced_locals(locals, ilix);
308 continue;
309 }
310
311 /* This is an IL_LABEL ilt. */
312 lab = ILI_SymOPND(ilix, 1);
313 ICHECK(VALIDSYM(lab) && STYPEG(lab) == ST_LABEL);
314 if (!BEGINSCOPEG(lab))
315 continue;
316
317 /* This is a scope-opening label. */
318 blk = ENCLFUNCG(lab);
319 verify_block(blk);
320 ICHECK(BEGINSCOPELABG(blk) == lab);
321
322 /* If multiple scopes are pushed in the same BIH, we'll allow any
323 order. They all have to be proper sub-scopes of outer_scope,
324 though. */
325 push_block(ss, blk);
326 ICHECK((blk != outer_scope) && "Can't open the same scope twice");
327 ICHECK(scope_contains(outer_scope, blk) &&
328 "New scope is not properly nested");
329 }
330}

332/**
333 \brief Verify a local that was referenced in the current function.
334
335 This function conforms to the prototype expected by hashset_iterate.
336 */
337static void
338verify_local(hash_key_t key, void *context)
339{
340 int sptr = HKEY2INT(key);
341 struct scope_stack *ss = (struct scope_stack *)context;
342 SPTR encl = ENCLFUNCG(sptr);
343 int i;
344
345/* The Fortran frontend doesn't bother setting ENCLFUNC on local
346 variables. Fortran doesn't have scopes anyway, except through
347 inlining. */
348 if (!encl)
349 return;
350
351 /* Compiler-created scalar temporaries don't always have associated
352 scopes. They will be treated as function-scope. */
353 if (!encl && CCSYMG(sptr)) {
354 dbgprintf(" Referenced local compiler-created temporary with no "
355 "scope: %d (%s)\n",
356 sptr, SYMNAME(sptr));
357 return;
358 }
359
360 dbgprintf(" Referenced local %d (%s) from scope %d. At block %d level "
361 "%d-%d, prev %d.\n",
362 sptr, SYMNAME(sptr), encl, ss->scope[ss->curr].block, ss->outer,
363 ss->curr, ss->prev_top);
364 ICHECK(VALIDSYM(encl) && "Local variable does not have a valid scope");
365
366 /* Now verify that the 'encl' scope is accessible in one of the
367 active scopes in the current block.
368
369 It is possible that multiple non-overlapping scopes were pushed
370 and popped in the current basic block. When that happens, these
371 scopes will appear on the scope stack from ss->outer to
372 ss->curr. These scopes are pushed even though they are not
373 nested. They will be popped again by verify_endscopes().
374
375 We are satisfied if encl contains any scope from this set. */
376 for (i = ss->outer; i <= ss->curr; i++) {
377 if (scope_contains(encl, ss->scope[i].block))
378 return;
379 }
380
381 /* Finally, it is also possible that encl overlaps one of the scopes
382 that ended in the current block. ss->prev_top records the
383 innermost active scope on entry to the block. The assertion here
384 covers the above scope containment checks too. */
385 ICHECK(scope_contains(encl, ss->prev_top) &&
386 "Local variable referenced by code outside its scope");
387}

389/**
390 \brief Process all the scope closing labels in bih.
391
392 Pop the closed scopes off the stack.
393 */
394static void
395verify_endscopes(int bih, struct scope_stack *ss)
396{
397 int ilt, i;
398 int ilix;
399 int lab, blk;
400
401 for (ilt = BIH_ILTFIRST(bih); ilt; ilt = ILT_NEXT(ilt)) {
402 ilix = ILT_ILIP(ilt);
403 if (ILI_OPC(ilix) != IL_LABEL)
404 continue;
405 lab = ILI_OPND(ilix, 1);
406 if (!ENDSCOPEG(lab))
407 continue;
408
409 /* This is a scope-closing label. */
410 blk = ENCLFUNCG(lab);
411 verify_block(blk);
412 ICHECK(ENDSCOPELABG(blk) == lab);
413
414 /* Find the scope that is being closed. It doesn't have to be the
415 topmost one. */
416 for (i = ss->curr; i > 0; i--) {
417 if (ss->scope[i].block == blk) {
418 /* We can't have duplicate labels closing the same scope. */
419 ICHECK(ss->scope[i].closing_ilt == 0);
420 ss->scope[i].closing_ilt = ilt;
421 break;
422 }
423 }
424 }
425
426 /* Pop all the closed scopes. */
427 while (ss->scope[ss->curr].closing_ilt != 0)
428 pop_block(ss);
429}

431/**
432 \brief Verify all of the locals referenced in the current function
433 and check the scope labels.
434 */
435static void
436verify_all_blocks(void)
437{
438 hashset_t locals = hashset_alloc(hash_functions_direct);
439 struct scope_stack ss = {-1, 0, 0, NULL, SPTR_NULL};
440 int bih;
441
442 dbgprintf("\nVerifying scopes\n");
443 init_scope_stack(&ss, GBL_CURRFUNC);
444
445 /* Visit basic blocks in program order. */
446 for (bih = gbl.entbih; bih; bih = BIH_NEXT(bih)) {
447 int i, j;
448
449 ss.prev_top = ss.scope[ss.curr].block;
450 dbgprintf("block:%d in scope %d (%s)\n", bih, ss.prev_top,
451 SYMNAME(ss.prev_top));
452
453 /* First get rid of any open scopes that are closed in this block. */
454 verify_endscopes(bih, &ss);
455
456 /* Check labels opening scopes, collect all symbols referenced. */
457 hashset_clear(locals);
458 verify_beginscopes(bih, locals, &ss);
459
460 /* The scope stack is now as deep as it gets in bih. Check that
461 all referenced locals belong to a pushed scope block. */
462 hashset_iterate(locals, verify_local, &ss);
463
464 /* Check labels closing scopes again. This takes care of scopes
465 that were opened and closed in the same block. */
466 verify_endscopes(bih, &ss);
467
468 /* Check that new open scope are perfectly nested with respect to
469 other open scopes. */
470 for (i = ss.curr; i > ss.outer; i--) {
471 if (ss.scope[i].closing_ilt == 0) {
472 for (j = i - 1; j > ss.outer; j--)
473 if (ss.scope[j].closing_ilt == 0)
474 break;
475 ICHECK(ENCLFUNCG(ss.scope[i].block) == ss.scope[j].block);
476 }
477 }
478 ss.outer = ss.curr;
479 }
480
481 FREE(ss.scope);
482 hashset_free(locals);
483}

500/**
501 \brief Insert a begin-scope label ILM for the scope blksym which
502 much be an ST_BLOCK symbol table entry.
503
504 \return the new ST_LABEL symbol table entry.
505 */
506int
507insert_begin_scope_label(int block_sptr)
508{
509 int lab;
510
511 ICHECK(VALIDSYM(block_sptr) && STYPEG(block_sptr) == ST_BLOCK);
512 ICHECK(BEGINSCOPELABG(block_sptr) == 0 && "Scope already has a begin label.");
513
514 if (!XBIT_USE_SCOPE_LABELS)
515 return 0;
516
517 lab = getlab();
518 BEGINSCOPEP(lab, 1);
519 ENCLFUNCP(lab, block_sptr);
520 BEGINSCOPELABP(block_sptr, lab);
521 addlabel(lab);
522
523 return lab;
524}

552/*
553 * Scope tracking.
554 *
555 * During inlining, as we're scanning the current function ILMs top to bottom,
556 * call track_scope_label(label_sptr) whenever an IM_LABEL is seen in order to
557 * keep track of the current scope.
558 *
559 * The global variable current_scope points to the currently open scope.
560 */
561int current_scope = 0;
562
563/**
564 \brief Reset scope tracking, start from the outer function scope.
565 */
566void
567track_scope_reset()
568{
569 current_scope = GBL_CURRFUNC;
570 dbgprintf("Scope: Reset to current function %d:%s\n", current_scope,
571 SYMNAME(current_scope));
572}

574/**
575 \brief Update curent_scope after scanning over an IM_LABEL ilmx.
576 The label sptr is the first and only ILM operand on IM_LABEL.
577 */
578void
579track_scope_label(int label)
580{
581 assert(STYPEG(label) == ST_LABEL, "track_scope_label: Bogus label", label,
582 ERR_Severe);
583
584 /* If scope labels are not being inserted, it is impossible to keep
585 track of the current scope. */
586 if (!XBIT_USE_SCOPE_LABELS)
587 return;
588
589 /* Non-scope labels are simply ignored. */
590 if (!BEGINSCOPEG(label) && !ENDSCOPEG(label))
591 return;
592
593 /* Labels left over after cancel_lexical_block() are also ignored. */
594 if (!ENCLFUNCG(label))
595 return;
596
597 if (BEGINSCOPEG(label)) {
598 dbgprintf("Scope: Enter scope %d:%s from %d:%s. Parent %d:%s.\n",
599 ENCLFUNCG(label), SYMNAME(ENCLFUNCG(label)), current_scope,
600 SYMNAME(current_scope), ENCLFUNCG(ENCLFUNCG(label)),
601 SYMNAME(ENCLFUNCG(ENCLFUNCG(label))));
602 current_scope = ENCLFUNCG(label);
603 /* A label can't refer to a function scope. */
604 assert(STYPEG(current_scope) == ST_BLOCK,
605 "track_scope_label: Scope label does not refer to an ST_BLOCK",
606 label, ERR_Severe);
607 }
608
609 if (ENDSCOPEG(label)) {
610 dbgprintf("Scope: Leave scope %d:%s from %d:%s. Parent %d:%s.\n",
611 ENCLFUNCG(label), SYMNAME(ENCLFUNCG(label)), current_scope,
612 SYMNAME(current_scope), ENCLFUNCG(ENCLFUNCG(label)),
613 SYMNAME(ENCLFUNCG(ENCLFUNCG(label))));
614 /* This is the scope that's ending: */
615 current_scope = ENCLFUNCG(label);
616 assert(STYPEG(current_scope) == ST_BLOCK,
617 "track_scope_label: Scope label does not refer to an ST_BLOCK",
618 label, ERR_Severe);
619 /* This is the active scope after leaving the current scope: */
620 current_scope = ENCLFUNCG(current_scope);
621 }
622
623 if (!flg.inliner || XBIT(117, 0x10000)) {
624 assert(current_scope == GBL_CURRFUNC || STYPEG(current_scope) == ST_BLOCK,
625 "track_scope_label: Invalid scope label", label, ERR_Severe);
626 }
627}

629/**
630 \brief Find all the scope labels in the current ILM block and pass
631 them to track_scope_label. This is used for blocks that don't
632 contain any calls.
633
634 This function is looking at numilms ILMs in the ilmb.ilm_base
635 array. We don't have any ILM_* macros defined to access that area.
636
637 Compare find_bpar() in inliner.c
638 */
639void
640find_scope_labels(int numilms)
641{
642 int ilmx, len;
643
644 if (!XBIT_USE_SCOPE_LABELS)
645 return;
646
647 for (ilmx = BOS_SIZE; ilmx < numilms; ilmx += len) {
648 int opc = ilmb.ilm_base[ilmx];
649 if (opc == IM_LABEL) {
650 int label = ilmb.ilm_base[ilmx + 1];
651 track_scope_label(label);
652 }
653
654 len = ilms[opc].oprs + 1; /* length for opcode and fixed operands */
655 if (IM_VAR(opc))
656 len += ilmb.ilm_base[ilmx + 1]; /* include the variable opnds */
657 }
658}

660/*
661 * Inliner support.
662 *
663 * The functions and global variables below are used by the C and Fortran
664 * inliners.
665 */
666
667int new_callee_scope = 0;
668
669/* Stack of simultaneously open callee scopes.
670 *
671 * This stack contains scopes whose begin label has been inserted, and whose
672 * end label has yet to be inserted.
673 *
674 * The stack top is equal to new_callee_scope only after calling
675 * begin_inlined_scope().
676 */
677static int *open_callee_scope = NULL;
678static int open_callee_scope_count = 0;
679static int open_callee_scope_capacity = 0;
680
681/* Check if new_callee_scope is at the top of the open_callee_scope stack.
682 *
683 * This is false when new_callee_scope is 0 or when it hasn't been pushed yet
684 * by begin_inlined_scope().
685 */
686#define NEW_CALLEE_SCOPE_IS_STACK_TOP \
687 (open_callee_scope_count > 0 && \
688 new_callee_scope == open_callee_scope[open_callee_scope_count - 1])
689
690/**
691 \brief The original function scope of the callee being inlined is
692 represented as a new ST_BLOCK scope, stored in new_callee_scope.
693
694 Call create_inlined_scope() to create new_callee_scope as a subscope of the
695 currently tracked scope. Then call begin_inlined_scope() and
696 end_inlined_scope() to insert the begin/end labels.
697 */
698void
699create_inlined_scope(int callee_sptr)
700{
701 if (!XBIT_USE_SCOPE_LABELS) {
702 /* Without scope labels, we don't know the scope containing the
703 call site. Conservatively map scope to the entire function
704 scope. */
705 new_callee_scope = GBL_CURRFUNC;
706 return;
707 }
708
709 if (!flg.inliner || XBIT(117, 0x10000)) {
710 assert(current_scope, "create_inlined_scope: No current scope?", 0,
711 ERR_Severe);
712 }
713
714 /* An existing callee scope should have been saved or cancelled. */
715 if (new_callee_scope) {
716 assert(NEW_CALLEE_SCOPE_IS_STACK_TOP,
717 "create_inlined_scope: Previous callee scope wasn't saved",
718 new_callee_scope, ERR_Fatal);
719 }
720
721 NEWSYM(new_callee_scope);
722 STYPEP(new_callee_scope, ST_BLOCK);
723 NMPTRP(new_callee_scope, NMPTRG(callee_sptr));
724 SCOPEP(new_callee_scope, 1);
725 ENCLFUNCP(new_callee_scope, current_scope);
726
727 dbgprintf("Scope: Create %d:%s for inlining %d:%s.\n", new_callee_scope,

731/**
732 \brief Immediately before inlining a function, create a new
733 ST_BLOCK to represent the callee's function scope in the current
734 function.
735
736 Insert an IM_LABEL ILM to open the new scope, update the
737 new_callee_scope and current_scope variables to refer to the newly
738 created scope.
739 */
740void
741begin_inlined_scope(int func_sptr)
742{
743 int label;
744
745 if (!XBIT_USE_SCOPE_LABELS)
746 return;
747
748 /* Make a new scope unless create_inlined_scope() has already been called.
749 * Don't attempt to reuse an open scope that has already been pushed. */
750 if (!new_callee_scope || NEW_CALLEE_SCOPE_IS_STACK_TOP)
751 create_inlined_scope(func_sptr);
752
753 assert(new_callee_scope && STYPEG(new_callee_scope) == ST_BLOCK,
754 "begin_inlined_scope: No inlined scope was created", new_callee_scope,
755 ERR_Severe);
756
757 /* Push scope onto the stack of open callee scopes. */
758 NEED(open_callee_scope_count + 1, open_callee_scope, int,
759 open_callee_scope_capacity, open_callee_scope_capacity + 16);
760 open_callee_scope[open_callee_scope_count++] = new_callee_scope;
761
762 /* Create begin label here, end label in end_inlined_scope(). */
763 label = insert_begin_scope_label(new_callee_scope);
764 dbgprintf("Scope: Label %d:%s created, nest=%d.\n", label, SYMNAME(label),
765 open_callee_scope_count);
766
767 track_scope_label(label);
768}
769
770/**
771 \brief After inlining a function, emit a scope-ending label.
772 */
773void
774end_inlined_scope(void)
775{
776 int label;
777
778 if (!XBIT_USE_SCOPE_LABELS)
779 return;
780
781 assert(new_callee_scope && STYPEG(new_callee_scope) == ST_BLOCK,
782 "end_inlined_scope: Not currently in an inlined scope",
783 new_callee_scope, ERR_Severe);
784
785 label = insert_end_scope_label(new_callee_scope);
786 dbgprintf("Scope: Label %d:%s created, next=%d.\n", label, SYMNAME(label),
787 open_callee_scope_count);
788
789 /* Pop this scope off the stack. Make new_callee_scope refer to the
790 * previous open callee scope. */
791 assert(NEW_CALLEE_SCOPE_IS_STACK_TOP,
792 "end_inlined_scope: Scope stack corrupted", new_callee_scope,
793 ERR_Fatal);
794 if (--open_callee_scope_count)
795 new_callee_scope = open_callee_scope[open_callee_scope_count - 1];
796 else
797 new_callee_scope = 0;
798
799 track_scope_label(label);
800}
801
802/**
803 \brief End currently open callee scopes until
804 open_callee_scope_count is new_open_count.
805 */
806void
807end_inlined_scopes(int new_open_count)
808{
809 while (open_callee_scope_count > new_open_count) {
810 new_callee_scope = open_callee_scope[open_callee_scope_count - 1];
811 end_inlined_scope();
812 }
813}

815/**
816 \brief If create_inlined_scope() was called, but the inlining was
817 abandoned before any labels were inserted, call
818 cancel_inlined_scope() to clean up.
819 */
820void
821cancel_inlined_scope(void)
822{
823 if (!XBIT_USE_SCOPE_LABELS)
824 return;
825
826 if (!new_callee_scope)
827 return;
828
829 /* If the begin label was already inserted, this scope can't be cancelled. */
830 if (NEW_CALLEE_SCOPE_IS_STACK_TOP)
831 return;
832
833 /* Go back to the scope we came from before calling
834 * begin_inlined_scope(). */
835 current_scope = ENCLFUNCG(new_callee_scope);
836 dbgprintf("Scope: Reverted to %d:%s scope.\n", current_scope,
837 SYMNAME(current_scope));
838
839 /* Restore the old stack top into new_callee_scope. */
840 if (open_callee_scope_count)
841 new_callee_scope = open_callee_scope[open_callee_scope_count - 1];
842 else
843 new_callee_scope = 0;
844}

848{
849 new_callee_scope = 0;
