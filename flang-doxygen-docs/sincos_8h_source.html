75{
76 uint64_t result[4];
77 uint64_t ia = double_as_ll(a);
78 uint64_t s = ia & 0x8000000000000000ULL;
79 uint64_t e = ((ia >> 52) & 0x7ff) - 1022;
80 int32_t idx = 15 - (e >> 6);
81 int32_t q;
82 ia = ((ia << 11) | 0x8000000000000000ULL) >> 1;
83 e = e & 63;
84
85 __uint128_t acc = 0;
86 for (q = idx; q < idx + 4; q++) {
87 acc += (__uint128_t)ia * i1opi_f[q];
88 result[q - idx] = (uint64_t)acc;
89 acc >>= 64;
90 }
91
92 uint64_t p = result[3];
93 if (e) {
94 p = (p << e) | (result[2] >> (64 - e));
95 result[2] = (result[2] << e) | (result[1] >> (64 - e));
96 }
