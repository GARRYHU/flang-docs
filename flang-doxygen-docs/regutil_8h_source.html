57/**
58 \brief For restricting the assignment of registers to constants per RTYPE.
59 Used in machreg.c to init reg[...].const_flag; checked where assignments
60 occur (i.e., globalreg.c).

62typedef enum RegConstFlags_t {
63 RCF_NONE = 0, /**< constants not assigned */
64 RCF_ALL = 1, /**< no restriction on value */
65 RCF_NOT0 = 2, /**< not if value is 0 */
66 RCF_NOT32K = 4, /**< not if int value in [-32768, 32767] */
67 RCF_NOTL16 = 8, /**< not if high 16 bits of mask value == 0 */
68 RCF_NOTH16 = 0x10, /**< not if low 16 bits of mask value == 0 */
69 RCF_NOTA8B = 0x20, /**< not if abs(v) <= 255 (8 bits) */
70 RCF_NOTA9B = 0x40 /**< not if abs(v) <= 511 (9 bits) */

194/***** Register Candidate Table *****/
195
196typedef struct {
197 char atype; /* type of candidate:
198 * RATA_NME, RATA_CONST, RATA_ILI,
199 * RATA_TEMP, RATA_IND, RATA_ARR,
200 * RATA_UPL
201 */
202 char rtype; /* register type: */
203 /* RATA_IR, RATA_SP, RATA_DP, RATA_AR */
204
205 union {
206 unsigned all;
207 struct {
208 unsigned confl : 1; /* ILI usage conflicts with the existing
209 * msize
210 */
211 unsigned store : 1; /* variable was stored */
212 unsigned cse : 1; /* candidate is an induction cse
213 * use - for opt 2 only
214 */
215 unsigned ok : 1; /* ok to assign register to const cand */
216 unsigned noreg : 1; /* do not assign register to non-const cand */
217 unsigned ignore : 1; /* ignore this candidate */
218 unsigned eint : 1; /* extended int (ST100) */
219 unsigned vint : 1; /* vectorial int (ST140) */
220 unsigned inv : 1; /* this candidate is for an invariant */
221 unsigned tinv : 1; /* this candidate is for a transitive invariant */
222 } bits;
223 } flags;
224 INT msize; /* memory size of register; also, the dtype
225 * record if vector
226 */
227 int val; /* index of candidate (depends on atype)*/
228 int temp; /* register temp if RATA_TEMP */
229 int next; /* next candidate */
230 int count; /* number of uses */
231 int oload; /* other load ili for the same variable */
232 int ocand; /* other load candidate that produces the same value */
233 int rat; /* RAT for this candidate */
234} RCAND;
