666void
667__fort_init_consts()
668{
669 int i, j, k;
670 char *m, *t;
671
672/* Compute max value for N bits: 2**(N-1)-1 can overflow so use 2**(N-2) - 1 + 2**(N-2) */
673#define MAX_FOR_INT_TYPE(type) \
674 ((type)1 << (8*sizeof(type) - 2)) - 1 + ((type)1 << (8*sizeof(type) - 2));
675 max_int1 = MAX_FOR_INT_TYPE(__INT1_T);
676 max_int2 = MAX_FOR_INT_TYPE(__INT2_T);
677 max_int4 = MAX_FOR_INT_TYPE(__INT4_T);
678 max_int8 = MAX_FOR_INT_TYPE(__INT8_T);
679#undef MAX_FOR_INT_TYPE
680
681 max_str = (__STR_T) 255;
682
683 min_int1 = -max_int1 - 1;
684 min_int2 = -max_int2 - 1;
685 min_int4 = -max_int4 - 1;
686 min_int8 = -max_int8 - 1;
687 min_str = 0;
688
689 __fort_shifts[__NONE] = 0;
690
691 for (i = __NONE + 1; i < __NTYPES; ++i) {
692
693 /* initialize __fort_shifts */
694
695 for (j = 0, k = 1; k < __fort_size_of[i]; ++j, k <<= 1)
696 ;
697#if defined(DEBUG)
698 if (k != __fort_size_of[i])
699 __fort_abort("init_consts: type size not a power of two");
700#endif
701 __fort_shifts[i] = j;
702
703 /* initialize logical trues */
704
705 m = (char *)GET_FIO_CNFG_FTN_TRUE_ADDR;
706 t = __fort_trues[i];
707 if (t != (void *)0) {
708 for (j = 0; j < k; ++j)
709 t[j] = m[1];
710 t[0] |= m[0];
711 t[k - 1] |= m[sizeof(GET_FIO_CNFG_FTN_TRUE) - 1];
712 }
713
714 /* initialize logical masks */
715
716 m = (char *)GET_FIO_CNFG_TRUE_MASK_ADDR;
717 t = __fort_masks[i];
718 if (t != (void *)0) {
719 for (j = 0; j < k; ++j)
720 t[j] = m[1];
721 t[0] |= m[0];
722 t[k - 1] |= m[sizeof(GET_FIO_CNFG_TRUE_MASK) - 1];
723 }
724 }
725 __fort_true_log = *(__LOG_T *)__fort_trues[__LOG];
726 __fort_mask_log = *(__LOG_T *)__fort_masks[__LOG];
727
728#if defined(DEBUG)
729
730/* check compiler-runtime descriptor interface constants */
731
732 if (sizeof(__POINT_T) != sizeof(char *))
733 __fort_abort("init_consts: __POINT_T is not pointer size");
734
735 if (sizeof(F90_Desc) !=
736 (F90_DESC_HDR_INT_LEN * sizeof(__INT_T) +
737 F90_DESC_HDR_PTR_LEN * sizeof(__POINT_T) +
738 MAXDIMS * (F90_DESC_DIM_INT_LEN * sizeof(__INT_T) +
739 F90_DESC_DIM_PTR_LEN * sizeof(__POINT_T))))
740 __fort_abort("init_consts: F90_DESC_HDR INT/PTR_LEN incorrect");
741
742 if (sizeof(DIST_Desc) !=
743 (DIST_DESC_HDR_INT_LEN * sizeof(__INT_T) +
744 DIST_DESC_HDR_PTR_LEN * sizeof(__POINT_T) +
745 MAXDIMS * (DIST_DESC_DIM_INT_LEN * sizeof(__INT_T) +
746 DIST_DESC_DIM_PTR_LEN * sizeof(__POINT_T))))
747 __fort_abort("init_consts: DIST_DESC_HDR INT/PTR_LEN incorrect");
748
749 /* check reciprocal operations */
750
751 for (j = 1; j <= 10; ++j) {
752 __INT_T j_recip = RECIP(j);
753 for (i = 0; i < 100; ++i) {
754 int quo, rem;
755 RECIP_DIV(&quo, i, j);
756 if (quo != i / j)
757 __fort_abort("init_consts: RECIP_DIV failed");
758 RECIP_MOD(&rem, i, j);
759 if (rem != i % j)
760 __fort_abort("init_consts: RECIP_MOD failed");
761 RECIP_DIVMOD(&quo, &rem, i, j);
762 if (quo != i / j || rem != i % j)
763 __fort_abort("init_consts: RECIP_DIVMOD failed");
764 }
765 }
