36// double-double "fma"
37inline
38__m128d_2 __internal_ddfma (__m128d_2 x, __m128d_2 y, __m128d_2 z)
39{
40 __m128d e;
41 __m128d_2 t, m;
42 t.y = _mm_mul_sd(x.y, y.y);
43 t.x = _mm_fmsub_sd(x.y, y.y, t.y);
44 t.x = _mm_fmadd_sd(x.y, y.x, t.x);
45 t.x = _mm_fmadd_sd(x.x, y.y, t.x);
46
47 m.y = _mm_add_sd (z.y, t.y);
48 e = _mm_sub_sd (z.y, m.y);
49 m.x = _mm_add_sd (_mm_add_sd(_mm_add_sd (e, t.y), t.x), z.x);
50
51 return m;
52}

54// special case of double-double addition, where |x| > |y| and y is a double.

56__m128d_2 __internal_ddadd_yisdouble(__m128d_2 x, __m128d_2 y)

62 z.x = _mm_add_sd(_mm_add_sd (e, y.y), x.x);

69__m128d __internal_fast_int2dbl(__m128i a)

80// slowpath for exp. used to improve accuracy on larger inputs
81__m128d __attribute__ ((noinline)) __pgm_exp_d_scalar_slowpath(__m128i const i, __m128d const t, __m128d const bloga, __m128d z, __m128d prodx)
82{
83 __m128d const UPPERBOUND_1 = (__m128d)_mm_set1_epi64x(UPPERBOUND_1_D);
84 __m128d const UPPERBOUND_2 = (__m128d)_mm_set1_epi64x(UPPERBOUND_2_D);
85 __m128d const ZERO = _mm_set1_pd(ZERO_D);
86 __m128d const INF = (__m128d)_mm_set1_epi64x(INF_D);
87 __m128i const HI_ABS_MASK = _mm_set1_epi64x(HI_ABS_MASK_D);
88 __m128i const ABS_MASK = _mm_set1_epi64x(ABS_MASK_D);
89 __m128i const MULT_CONST = _mm_set1_epi64x(MULT_CONST_D);
90
91 __m128i abs_bloga = _mm_and_si128((__m128i)bloga, ABS_MASK);
92 __m128d abs_lt = (__m128d)_mm_and_si128((__m128i)abs_bloga, HI_ABS_MASK);
93 __m128d lt_zero_mask = _mm_cmp_sd(bloga, ZERO, _CMP_LT_OS); // compute bloga < 0.0
94 __m128d slowpath_mask = _mm_cmp_sd(abs_lt, UPPERBOUND_1, _CMP_LT_OS);
95
96 __m128d a_plus_inf = _mm_add_sd(bloga, INF); // check if bloga is too big
97 __m128d zero_inf_blend = _mm_blendv_pd(a_plus_inf, ZERO, lt_zero_mask);
98
99 __m128d accurate_scale_mask = (__m128d)_mm_cmp_sd(abs_lt, UPPERBOUND_2, _CMP_LT_OS);
100
101 // compute accurate scale
102 __m128i k = _mm_srli_epi64(i, 1); // k = i / 2
103 __m128i i_scale_acc = _mm_slli_epi64(k, D52_D); // shift to HI and shift 20
104 k = _mm_sub_epi32(i, k); // k = i - k
105 __m128i i_scale_acc_2 = _mm_slli_epi64(k, D52_D); // shift to HI and shift 20
106 __m128d multiplier = (__m128d)_mm_add_epi64(i_scale_acc_2, MULT_CONST);
107 multiplier = _mm_blendv_pd(ZERO, multiplier, accurate_scale_mask); // quick fix for overflows in case they're being trapped
108
109 __m128d res = (__m128d)_mm_add_epi32(i_scale_acc, (__m128i)t);
110 res = _mm_mul_sd(res, multiplier);
111
112 __m128d slowpath_blend = _mm_blendv_pd(zero_inf_blend, res, accurate_scale_mask);
113 z = _mm_blendv_pd(slowpath_blend, z, slowpath_mask);
114
115 __m128i isinf_mask = _mm_cmpeq_epi64((__m128i)INF, (__m128i)z); // special case for inf
116 __m128d z_fixed = _mm_fmadd_sd(z, prodx, z); // add only if not inf
117 z = _mm_blendv_pd(z_fixed, z, (__m128d)isinf_mask);
118
119 return z;
120}

122// special cases for pow
123// implementation derived from cudart/device_functions_impl.c
124__m128d __attribute__ ((noinline)) __pgm_pow_d_scalar_special_cases(__m128d const a, __m128d const b, __m128d t)
125{
126 __m128i const HI_MASK = _mm_set1_epi64x(HI_MASK_D);
127 __m128d const SGN_EXP_MASK = (__m128d)_mm_set1_epi64x(SGN_EXP_MASK_D);
128 __m128i const ABS_MASK = _mm_set1_epi64x(ABS_MASK_D);
129 __m128d const NEG_ONE = _mm_set1_pd(NEG_ONE_D);
130 __m128d const ZERO = _mm_set1_pd(ZERO_D);
131 __m128d const ONE = _mm_set1_pd(ONE_D);
132 __m128d const HALF = _mm_set1_pd(HALF_D);
133 __m128d const SGN_MASK = (__m128d)_mm_set1_epi64x(SGN_MASK_D);
134 __m128d const INF = (__m128d)_mm_set1_epi64x(INF_D);
135 __m128i const INF_FAKE = _mm_set1_epi64x(INF_FAKE_D);
136 __m128d const NAN_MASK = (__m128d)_mm_set1_epi64x(NAN_MASK_D);
137 __m128d const NEG_ONE_CONST = (__m128d)_mm_set1_epi64x(NEG_ONE_CONST_D);
138
139 __m128i const TEN_23 = _mm_set1_epi64x(TEN_23_D);
140 __m128i const ELEVEN = _mm_set1_epi64x(ELEVEN_D);
141
142 __m128i a_sign, b_sign, bIsOddInteger;
143 __m128d abs_a, abs_b;
144 __m128i shiftb;
145
146 a_sign = (__m128i)_mm_and_pd(a, SGN_MASK);
147 b_sign = (__m128i)_mm_and_pd(b, SGN_MASK);
148 abs_a = _mm_and_pd(a, (__m128d)ABS_MASK);
149 abs_b = _mm_and_pd(b, (__m128d)ABS_MASK);
150
151 // determining if b is an odd integer, since there are special cases for it
152 shiftb = (__m128i)_mm_and_pd (b, SGN_EXP_MASK);
153 shiftb = _mm_srli_epi64(shiftb, D52_D);
154 shiftb = _mm_sub_epi64(shiftb, TEN_23);
155 shiftb = _mm_add_epi64(shiftb, ELEVEN);
156
157 __m128i b_is_half = _mm_cmpeq_epi64((__m128i)abs_b, (__m128i)HALF);
158 bIsOddInteger = _mm_sllv_epi64((__m128i)b, shiftb);
159 bIsOddInteger = _mm_cmpeq_epi64((__m128i)SGN_MASK, bIsOddInteger);
160 // fix for b = +/-0.5 being incorrectly identified as an odd integer
161 bIsOddInteger = _mm_andnot_si128(b_is_half, bIsOddInteger);
162
163 // corner cases where a <= 0
164 // if ((ahi < 0) && bIsOddInteger)
165 __m128d ahi_lt_0 = (__m128d)_mm_cmpeq_epi64(a_sign, (__m128i)ZERO);
166 __m128d ahilt0_and_boddint_mask = _mm_andnot_pd(ahi_lt_0, (__m128d)bIsOddInteger);
167
168 t = _mm_blendv_pd(t, neg(t), ahilt0_and_boddint_mask);
169
170 // else if ((ahi < 0) && (b != trunc(b)))
171 __m128d b_ne_trunc = _mm_cmp_sd(b, _mm_round_pd(b, _MM_FROUND_TO_ZERO), _CMP_NEQ_UQ);
172 __m128d nan_mask = _mm_andnot_pd(ahi_lt_0, b_ne_trunc);
173 t = _mm_blendv_pd(t, NAN_MASK, nan_mask);
174
175 // if (a == 0.0)
176 __m128d a_is_0_mask = (__m128d)_mm_cmpeq_epi64((__m128i)abs_a, (__m128i)ZERO);
177 __m128d thi_when_ais0 = ZERO;
178
179 // if (bIsOddInteger && a == 0.0)
180 thi_when_ais0 = _mm_blendv_pd(thi_when_ais0, (__m128d)a, (__m128d)bIsOddInteger);
181
182 // if (bhi < 0 && a == 0.0)
183 __m128d bhi_lt_0 = (__m128d)_mm_cmpeq_epi64(b_sign, (__m128i)ZERO); //this mask is inverted
184 __m128d thi_or_INF = _mm_or_pd(thi_when_ais0, INF);
185 thi_when_ais0 = _mm_blendv_pd(thi_or_INF, thi_when_ais0, bhi_lt_0);
186 __m128d t_when_ais0 = _mm_and_pd(thi_when_ais0, (__m128d)HI_MASK);
187
188 t = _mm_blendv_pd(t, t_when_ais0, a_is_0_mask);
189
190 // else if (a is INF)
191 __m128d a_inf_mask = (__m128d)_mm_cmpeq_epi64((__m128i)abs_a, (__m128i)INF);
192 // use bhi_lt_0 backwards to evaluate bhi >= 0
193 __m128d thi_when_aisinf = _mm_blendv_pd(thi_when_aisinf, INF, bhi_lt_0);
194
195 // now evaluate ((ahi < 0) && bIsOddInteger)
196 __m128d thi_xor_sgn = _mm_xor_pd(thi_when_aisinf, SGN_MASK);
197 thi_when_aisinf = _mm_blendv_pd(thi_when_aisinf, thi_xor_sgn, ahilt0_and_boddint_mask);
198
199 t = _mm_blendv_pd(t, thi_when_aisinf, a_inf_mask);
200
201 // else if (abs(b) is INF)
202 __m128d b_inf_mask = (__m128d)_mm_cmpeq_epi64((__m128i)abs_b, (__m128i)INF);
203 __m128d thi_when_bisinf = ZERO;
204 __m128d absa_gt_one = _mm_cmp_sd(abs_a, ONE, _CMP_GT_OS); // evaluating (abs(a) > 1)
205 thi_when_bisinf = _mm_blendv_pd(thi_when_bisinf, INF, absa_gt_one);
206
207 __m128d thi_xor_inf = _mm_xor_pd(thi_when_bisinf, INF);
208 thi_when_bisinf = _mm_blendv_pd(thi_xor_inf, thi_when_bisinf, bhi_lt_0); // bhi < 0
209
210 __m128d a_is_negone = (__m128d)_mm_cmpeq_epi64((__m128i)a, (__m128i)NEG_ONE);
211 thi_when_bisinf = _mm_blendv_pd(thi_when_bisinf, NEG_ONE_CONST, a_is_negone); //a == -1
212
213 t = _mm_blendv_pd(t, thi_when_bisinf, b_inf_mask);
214
215 // if(a is NAN || B is NAN) <=> !(a is a number && b is a number)
216 __m128i a_nan_mask = _mm_cmpeq_epi64(_mm_max_epu32((__m128i)abs_a, INF_FAKE), INF_FAKE);
217 __m128i b_nan_mask = _mm_cmpeq_epi64(_mm_max_epu32((__m128i)abs_b, INF_FAKE), INF_FAKE);
218 __m128i aorb_nan_mask = _mm_and_si128(a_nan_mask, b_nan_mask); //this mask is inverted
219 t = _mm_blendv_pd(_mm_add_sd(a,b), t, (__m128d)aorb_nan_mask);
220
221 // if a == 1 or b == 0, answer = 1
222 __m128d a_is_one_mask = (__m128d)_mm_cmpeq_epi64((__m128i)a, (__m128i)ONE);
223 __m128d b_is_zero_mask = (__m128d)_mm_cmpeq_epi64((__m128i)abs_b, (__m128i)ZERO);
224 __m128d ans_equals_one_mask = _mm_or_pd(a_is_one_mask, b_is_zero_mask);
225 t = _mm_blendv_pd(t, ONE, ans_equals_one_mask);
226 // ******************************************************************************************
227//
228//
229 __m128i a_is_neg = (__m128i)_mm_cmp_pd(a, ZERO, _CMP_LT_OS);
230 int a_is_neg_flag = _mm_movemask_epi8((__m128i)a_is_neg);
231
232 __m128i b_is_integer = (__m128i)_mm_cmp_pd(b, _mm_floor_pd(b), _CMP_EQ_OQ);
233 int b_is_integer_flag = _mm_movemask_epi8((__m128i)b_is_integer);
234
235 __m128i b_is_lt_zero = (__m128i)_mm_cmp_pd(b, ZERO, _CMP_LT_OS);
236 int b_is_lt_zero_flag = _mm_movemask_epi8((__m128i)b_is_lt_zero);
237
238 __m128d const MINUS_ZERO = _mm_set1_pd((double)-0.0);
239 __m128i a_is_pos_zero = _mm_cmpeq_epi64( (__m128i)a, (__m128i)ZERO);
240 __m128i a_is_neg_zero = _mm_cmpeq_epi64( (__m128i)a, (__m128i)MINUS_ZERO);
241 __m128i a_is_any_zero = _mm_or_si128(a_is_pos_zero, a_is_neg_zero);
242 int a_is_any_zero_flag = _mm_movemask_epi8((__m128i)a_is_any_zero);
243
244/*
245 * Before returning see if we need to set any of the processor
246 * exception flags.
247 *
248 * Domain error: a is negative, and b is a finite noninteger
249 * we need to raise the Invalid-Operation flag. This can be done by
250 * taking the square root of a negative number.
251 *
252 * Pole error: a is zero and b is negative we need to raise the
253 * divide by zero flag. This can be done by dividing by zero.
254 */
255
256 if (a_is_neg_flag && (!b_is_integer_flag)) {
257 __m128d volatile invop = _mm_sqrt_pd(a);
258 }
259
260 if (a_is_any_zero_flag && b_is_lt_zero_flag) {
261 __m128d volatile divXzero = _mm_div_pd(ONE,ZERO);
262 }
263
264 return t;
265}

267double __fsd_pow_fma3(double const a_in, double const b_in)
268{
269 //////////////////////////////////////////////////// log constants
270 __m128d const LOG_POLY_6 = _mm_set1_pd(LOG_POLY_6_D);
271 __m128d const LOG_POLY_5 = _mm_set1_pd(LOG_POLY_5_D);
272 __m128d const LOG_POLY_4 = _mm_set1_pd(LOG_POLY_4_D);
273 __m128d const LOG_POLY_3 = _mm_set1_pd(LOG_POLY_3_D);
274 __m128d const LOG_POLY_2 = _mm_set1_pd(LOG_POLY_2_D);
275 __m128d const LOG_POLY_1 = _mm_set1_pd(LOG_POLY_1_D);
276 __m128d const LOG_POLY_0 = _mm_set1_pd(LOG_POLY_0_D);
277
278 __m128d const CC_CONST_Y = _mm_set1_pd(CC_CONST_Y_D);
279 __m128d const CC_CONST_X = _mm_set1_pd(CC_CONST_X_D);
280
281 __m128i const EXPO_MASK = _mm_set1_epi64x(EXPO_MASK_D);
282 __m128d const HI_CONST_1 = (__m128d)_mm_set1_epi64x(HI_CONST_1_D);
283 __m128d const HI_CONST_2 = (__m128d)_mm_set1_epi64x(HI_CONST_2_D);
284 __m128i const HALFIFIER = _mm_set1_epi64x(HALFIFIER_D);
285 __m128i const HI_THRESH = _mm_set1_epi64x(HI_THRESH_D); //~sqrt(2)
286 __m128d const ONE_F = _mm_set1_pd(ONE_F_D);
287 __m128d const TWO = _mm_set1_pd(TWO_D);
288 __m128d const ZERO = _mm_set1_pd(ZERO_D);
289
290 __m128d const LN2_HI = _mm_set1_pd(LN2_HI_D);
291 __m128d const LN2_LO = _mm_set1_pd(LN2_LO_D);
292 //////////////////////////////////////////////////////////////////
293
294 //////////////////////////////////////////////////// exp constants
295 __m128d const L2E = _mm_set1_pd(L2E_D);
296 __m128d const NEG_LN2_HI = _mm_set1_pd(NEG_LN2_HI_D);
297 __m128d const NEG_LN2_LO = _mm_set1_pd(NEG_LN2_LO_D);
298
299 __m128d const EXP_POLY_B = _mm_set1_pd(EXP_POLY_B_D);
300 __m128d const EXP_POLY_A = _mm_set1_pd(EXP_POLY_A_D);
301 __m128d const EXP_POLY_9 = _mm_set1_pd(EXP_POLY_9_D);
302 __m128d const EXP_POLY_8 = _mm_set1_pd(EXP_POLY_8_D);
303 __m128d const EXP_POLY_7 = _mm_set1_pd(EXP_POLY_7_D);
304 __m128d const EXP_POLY_6 = _mm_set1_pd(EXP_POLY_6_D);
305 __m128d const EXP_POLY_5 = _mm_set1_pd(EXP_POLY_5_D);
306 __m128d const EXP_POLY_4 = _mm_set1_pd(EXP_POLY_4_D);
307 __m128d const EXP_POLY_3 = _mm_set1_pd(EXP_POLY_3_D);
308 __m128d const EXP_POLY_2 = _mm_set1_pd(EXP_POLY_2_D);
309 __m128d const EXP_POLY_1 = _mm_set1_pd(EXP_POLY_1_D);
310 __m128d const EXP_POLY_0 = _mm_set1_pd(EXP_POLY_0_D);
311
312 __m128d const DBL2INT_CVT = _mm_set1_pd(DBL2INT_CVT_D);
313
314 __m128d const UPPERBOUND_1 = (__m128d)_mm_set1_epi64x(UPPERBOUND_1_D);
315 __m128i const HI_ABS_MASK = _mm_set1_epi64x(HI_ABS_MASK_D);
316
317 //////////////////////////////////////////////////// pow constants
318 __m128i const HI_MASK = _mm_set1_epi64x(HI_MASK_D);
319 __m128d const SGN_EXP_MASK = (__m128d)_mm_set1_epi64x(SGN_EXP_MASK_D);
320 __m128i const ABS_MASK = _mm_set1_epi64x(ABS_MASK_D);
321 __m128i const ONE = _mm_set1_epi64x(ONE_D);
322
323 __m128i const TEN_23 = _mm_set1_epi64x(TEN_23_D);
324 __m128i const ALL_ONES_EXPONENT = _mm_set1_epi64x(ALL_ONES_EXPONENT_D);
325
326 __m128d abs_a, a_mut;
327 __m128d_2 loga;
328 __m128d t_hi, t_lo, tmp, e, bloga, prodx;
329
330 __m128d_2 qq, cc, uu, tt;
331 __m128d f, g, u, v, q, ulo, m;
332 __m128i ihi, expo, expo_plus1;
333 __m128d thresh_mask;
334
335 __m128d a = _mm_set1_pd(a_in);
336 __m128d b = _mm_set1_pd(b_in);
337
338
339 /*
340 * Check for exponent(b) being 1.0 and take a quick exit.
341 */
342 if (b_in == 1.0) {
343 return a_in;
344 }
345
346 // *****************************************************************************************
347 // computing log(abs(a))
348 abs_a = _mm_and_pd(a, (__m128d)ABS_MASK);
349 a_mut = _mm_and_pd(abs_a, HI_CONST_1);
350 a_mut = _mm_or_pd(a_mut, HI_CONST_2);
351 m = (__m128d)_mm_sub_epi32((__m128i)a_mut, HALFIFIER); // divide by 2
352
353 ihi = _mm_and_si128((__m128i)a_mut, HI_MASK);
354 thresh_mask = _mm_cmp_sd((__m128d)ihi, (__m128d)HI_THRESH,_CMP_GT_OS);
355 m = _mm_blendv_pd(a_mut, m, thresh_mask);
356
357 expo = _mm_srli_epi64((__m128i)abs_a, D52_D);
358 expo = _mm_sub_epi64(expo, TEN_23);
359 expo_plus1 = _mm_add_epi64(expo, ONE); // add one to exponent instead
360 expo = (__m128i)_mm_blendv_pd((__m128d)expo, (__m128d)expo_plus1, thresh_mask);
361
362 // begin computing log(m)
363 f = _mm_sub_sd(m, ONE_F);
364 g = _mm_add_sd(m, ONE_F);
365 g = _mm_div_sd(ONE_F, g);
366
367 u = _mm_mul_sd(_mm_mul_sd(TWO, f), g);
368
369 // u = 2.0 * (m - 1.0) / (m + 1.0)
370 v = _mm_mul_sd(u, u);
371
372 // polynomial is used to approximate atanh(v)
373 // an estrin evaluation scheme is used.
374 __m128d c0 = _mm_fmadd_sd(LOG_POLY_1, v, LOG_POLY_0);
375 __m128d c2 = _mm_fmadd_sd(LOG_POLY_3, v, LOG_POLY_2);
376 __m128d c4 = _mm_fmadd_sd(LOG_POLY_5, v, LOG_POLY_4);
377 __m128d v2 = _mm_mul_sd(v, v);
378 __m128d v4 = _mm_mul_sd(v2, v2);
379
380 c0 = _mm_fmadd_sd(c2, v2, c0);
381 c4 = _mm_fmadd_sd(LOG_POLY_6, v2, c4);
382 q = _mm_fmadd_sd(c4, v4, c0);
383 q = _mm_mul_sd(q, v);
384
385 tmp = _mm_mul_sd(TWO, _mm_sub_sd(f, u));
386 tmp = _mm_fmadd_sd(neg(u), f, tmp);
387 ulo = _mm_mul_sd(g, tmp);
388
389 // double-double computation begins
390 qq.y = q;
391 qq.x = ZERO;
392 uu.y = u;
393 uu.x = ulo;
394 cc.y = CC_CONST_Y;
395 cc.x = CC_CONST_X;
396
397 qq = __internal_ddadd_yisdouble(cc, qq);
398
399 // computing log(m) in double-double format
400 cc.y = _mm_mul_sd(uu.y, uu.y);
401 cc.x = _mm_fmsub_sd(uu.y, uu.y, cc.y);
402 cc.x = _mm_fmadd_sd(uu.y,
403 (__m128d)_mm_add_epi32((__m128i)uu.x, HALFIFIER),
404 cc.x); // u ** 2
405
406 tt.y = _mm_mul_sd(cc.y, uu.y);
407 tt.x = _mm_fmsub_sd(cc.y, uu.y, tt.y);
408 tt.x = _mm_fmadd_sd(cc.y, uu.x, tt.x);
409 tt.x = _mm_fmadd_sd(cc.x, uu.y, tt.x); // u ** 3 un-normalized
410
411 uu = __internal_ddfma(qq, tt, uu);
412
413 // computing log a = log(m) + log(2)*expo
414 f = __internal_fast_int2dbl (expo);
415 q = _mm_fmadd_sd(f, LN2_HI, uu.y);
416 tmp = _mm_fmadd_sd(neg(f), LN2_HI, q);
417 tmp = _mm_sub_sd(tmp, uu.y);
418 loga.y = q;
419 loga.x = _mm_sub_sd(uu.x, tmp);
420 loga.x = _mm_fmadd_sd(f, LN2_LO, loga.x);
421
422 // finish log(a)
423 // ******************************************************************************************
424
425 // compute b * log(a)
426 t_hi = _mm_mul_sd(loga.y, b);
427 t_lo = _mm_fmsub_sd(loga.y, b, t_hi);
428 t_lo = _mm_fmadd_sd(loga.x, b, t_lo);
429 bloga = e = _mm_add_sd(t_hi, t_lo);
430 prodx = _mm_add_sd(_mm_sub_sd(t_hi, e), t_lo);
431
432 // ***************************************************************************************
433 // computing exp(b * log(a))
434 // calculating exponent; stored in the LO of each 64-bit block
435 __m128i i = (__m128i) _mm_fmadd_sd(bloga, L2E, DBL2INT_CVT);
436
437 // calculate mantissa
438 __m128d t = _mm_sub_sd ((__m128d)i, DBL2INT_CVT);
439 __m128d z = _mm_fmadd_sd (t, NEG_LN2_HI, bloga);
440 z = _mm_fmadd_sd (t, NEG_LN2_LO, z);
441
442 // use polynomial to calculate exp
443 // mixed estrin/horner scheme: estrin on the higher 8 coefficients, horner on the lowest 4.
444 // provided speedup without loss of precision compared to full horner
445 __m128d t4 = _mm_fmadd_sd(EXP_POLY_5, z, EXP_POLY_4);
446 __m128d t6 = _mm_fmadd_sd(EXP_POLY_7, z, EXP_POLY_6);
447 __m128d t8 = _mm_fmadd_sd(EXP_POLY_9, z, EXP_POLY_8);
448 __m128d t10 = _mm_fmadd_sd(EXP_POLY_B, z, EXP_POLY_A);
449
450 __m128d z2 = _mm_mul_sd(z, z);
451 __m128d z4 = _mm_mul_sd(z2, z2);
452
453 t4 = _mm_fmadd_sd(t6, z2, t4);
454 t8 = _mm_fmadd_sd(t10, z2, t8);
455 t4 = _mm_fmadd_sd(t8, z4, t4);
456
457 t = _mm_fmadd_sd(t4, z, EXP_POLY_3);
458 t = _mm_fmadd_sd(t, z, EXP_POLY_2);
459 t = _mm_fmadd_sd(t, z, EXP_POLY_1);
460 t = _mm_fmadd_sd(t, z, EXP_POLY_0);
461
462 // fast scale
463 __m128i i_scale = _mm_slli_epi64(i,D52_D);
464 __m128d ztemp = z = (__m128d)_mm_add_epi32(i_scale, (__m128i)t);
465
466 // slowpath detection for exp
467 __m128d abs_bloga = (__m128d)_mm_and_si128((__m128i)bloga, HI_ABS_MASK);
468#if defined(TARGET_LINUX_POWER)
469 int exp_slowmask = _vec_any_nz((__m128i)_mm_cmp_sd(abs_bloga, UPPERBOUND_1, _CMP_GE_OS));
470#else
471 int exp_slowmask = _mm_movemask_pd(_mm_cmp_sd(abs_bloga, UPPERBOUND_1, _CMP_GE_OS));
472#endif
473
474 z = _mm_fmadd_sd(z, prodx, z);
475
476 if (__builtin_expect(exp_slowmask, 0)) {
477 z = __pgm_exp_d_scalar_slowpath(i, t, bloga, ztemp, prodx);
478 }
479 // finished exp(b * log (a))
480 // ************************************************************************************************
481
482 // compute if we have special cases (inf, nan, etc). see man pow for full list of special cases
483 __m128i detect_inf_nan = (__m128i)_mm_add_sd(a, b); // check for inf/nan
484 __m128i overridemask = _mm_cmpeq_epi64( (__m128i)a, (__m128i)ONE_F); // if a == 1
485 __m128i overridemask2 = _mm_cmpeq_epi64(_mm_and_si128(detect_inf_nan, ALL_ONES_EXPONENT), ALL_ONES_EXPONENT);
486 overridemask = _mm_or_si128(overridemask, (__m128i)_mm_cmp_sd(a, ZERO, _CMP_LE_OQ)); // if a < 0
487
488#if defined(TARGET_LINUX_POWER)
489 int specMask = _vec_any_nz((__m128i)_mm_or_si128(overridemask, overridemask2));
490#else
491 int specMask = _mm_movemask_pd((__m128d)_mm_or_si128(overridemask, overridemask2));
492#endif
493 if(__builtin_expect(specMask, 0)) {
494 return _mm_cvtsd_f64(__pgm_pow_d_scalar_special_cases(a, b, z));
495 }
496 return _mm_cvtsd_f64(z);
497}
