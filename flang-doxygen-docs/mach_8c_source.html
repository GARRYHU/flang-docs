26void
27set_mach(MACHTYPE *mach, int machtype)
28{
29 int has_fma3 = 0;
30 int has_fma4 = 0;
31 memset(&mach->type, 0, sizeof(mach->type));
32 memset(&mach->feature, 0, sizeof(mach->feature));
33 mach->tpval = machtype;
34/*
35 * -tp may not be appropriate for non-x86; so machtype may refer to
36 * machine attriubutes rather than a TP_ value.
37 */
38#if defined(TARGET_X86) /* { */
39 mach->cachesize = flg.x[32]; /* this applies to all 'machtype's */
40
41 switch (machtype) {
42 case TP_ZEN:
43 /* AMD Zen microarchitecture, e.g. EPYC and Ryzen processors.
44 */
45 mach->type[MACH_AMD_ZEN] = 1;
46 mach->feature[FEATURE_AVX2] = 1;
47 FLANG_FALLTHROUGH;
48
49 case TP_PILEDRIVER:
50 /* AMD piledriver
51 */
52 mach->type[MACH_AMD_PILEDRIVER] = 1;
53 mach->feature[FEATURE_FMA3] = 1;
54 has_fma3 = 1;
55 mach->feature[FEATURE_LD_VMOVUPD] = 1; /* added on 14 Dec 2015 */
56 mach->feature[FEATURE_ST_VMOVUPD] = 1; /* " " " " */
57 mach->feature[FEATURE_ST_MOVUPD] = 1; /* " " " " */
58 FLANG_FALLTHROUGH;
59
60 case TP_BULLDOZER:
61 /* AMD bulldozer
62 */
63 mach->type[MACH_AMD_BULLDOZER] = 1;
64 mach->feature[FEATURE_SSE41] = 1; /* JHM: added on 2 Feb 2017 */
65 mach->feature[FEATURE_SSE42] = 1; /* " " " " " */
66 mach->feature[FEATURE_AVX] = 1;
67 if (machtype == TP_PILEDRIVER || machtype == TP_BULLDOZER) {
68 mach->feature[FEATURE_SIMD128] = 1;
69 mach->feature[FEATURE_FMA4] = 1;
70 has_fma4 = 1;
71 mach->feature[FEATURE_XOP] = 1;
72 }
73 mach->feature[FEATURE_ALIGNLOOP8] = 1;
74 mach->feature[FEATURE_ALIGNJMP8] = 1;
75 FLANG_FALLTHROUGH;
76
77 case TP_ISTANBUL:
78 /* AMD instanbul
79 */
80 mach->type[MACH_AMD_ISTANBUL] = 1;
81 FLANG_FALLTHROUGH;
82
83 case TP_SHANGHAI:
84 /* AMD shanghai, like greyhound but with a larger cache.
85 */
86 mach->type[MACH_AMD_SHANGHAI] = 1;
87 mach->feature[FEATURE_MULTI_ACCUM] = 1;
88 if (mach->cachesize == 0)
89 mach->cachesize = (6 * 1024 * 1024);
90 FLANG_FALLTHROUGH;
91
92 case TP_GH:
93 /* AMD greyhound
94 */
95 mach->type[MACH_AMD] = 1;
96 mach->type[MACH_AMD_HAMMER] = 1;
97 mach->type[MACH_AMD_GH] = 1;
98 mach->feature[FEATURE_SSE] = 1;
99 mach->feature[FEATURE_SSE2] = 1;
100 mach->feature[FEATURE_SSE3] = 1;
101 mach->feature[FEATURE_MISALIGNEDSSE] = 1;
102 mach->feature[FEATURE_LD_MOVUPD] = 1;
103 mach->feature[FEATURE_UNROLL_16] = 1;
104 mach->feature[FEATURE_DOUBLE_UNROLL] = 1;
105 mach->feature[FEATURE_PEEL_SHUFFLE] = 1;
106 mach->feature[FEATURE_PREFETCHNTA] = 1;
107 mach->feature[FEATURE_PDSHUF] = 1;
108 mach->feature[FEATURE_GHLIBS] = 1;
109 mach->feature[FEATURE_SSEMISALN] = 1;
110 mach->feature[FEATURE_DAZ] = 0; /* cf. 1 for Intel */
111 mach->feature[FEATURE_PREFER_MOVLPD] = 0;
112 mach->feature[FEATURE_USE_INC] = 1; /* cf. 0 for Intel */
113 mach->feature[FEATURE_USE_MOVAPD] = 1;
114 mach->feature[FEATURE_MERGE_DEPENDENT] = 1;
115 mach->feature[FEATURE_SCALAR_NONTEMP] = 1;
116 mach->feature[FEATURE_SSE4A] = 1;
117 mach->feature[FEATURE_SSEIMAX] = 1;
118 mach->feature[FEATURE_ABM] = 1;
119 if (machtype == TP_ISTANBUL ||
120 machtype == TP_SHANGHAI ||
121 machtype == TP_GH)
122 {
123 if (XBIT(135, 0x400000))
124 mach->feature[FEATURE_ALIGNLOOP32] = 1;
125 else
126 mach->feature[FEATURE_ALIGNLOOP16] = 1;
127 mach->feature[FEATURE_ALIGNJMP16] = 1;
128 }
129 if (mach->cachesize == 0)
130 mach->cachesize = (2 * 1024 * 1024);
131 break;
132
133 case TP_K8E:
134 /* AMD hammer
135 */
136 mach->feature[FEATURE_SSE3] = 1;
137 FLANG_FALLTHROUGH;
138
139 case TP_K8:
140 /* AMD hammer
141 */
142 mach->type[MACH_AMD] = 1;
143 mach->type[MACH_AMD_HAMMER] = 1;
144 mach->feature[FEATURE_SSE] = 1;
145 mach->feature[FEATURE_SSE2] = 1;
146 mach->feature[FEATURE_DAZ] = 0; /* cf. 1 for Intel */
147 mach->feature[FEATURE_PREFER_MOVLPD] = 1;
148 mach->feature[FEATURE_USE_INC] = 1;
149 mach->feature[FEATURE_ALIGNLOOP16] = 1;
150 mach->feature[FEATURE_ALIGNJMP16] = 1;
151 if (mach->cachesize == 0)
152 mach->cachesize = (1024 * 1024);
153 break;
154
155 case TP_LARRABEE:
156 mach->type[MACH_INTEL] = 1;
157 mach->type[MACH_INTEL_PENTIUM4] = 1;
158 mach->type[MACH_INTEL_LARRABEE] = 1;
159 mach->feature[FEATURE_SSE] = 1;
160 mach->feature[FEATURE_SSE2] = 1;
161 mach->feature[FEATURE_SSE3] = 1;
162 mach->feature[FEATURE_SSE41] = 1;
163 mach->feature[FEATURE_SSE42] = 1;
164 mach->feature[FEATURE_USE_INC] = 0;
165 mach->feature[FEATURE_LD_MOVUPD] = 1;
166 mach->feature[FEATURE_USE_MOVAPD] = 1;
167 mach->feature[FEATURE_MNI] = 1;
168 mach->feature[FEATURE_DAZ] = 1;
169 mach->feature[FEATURE_SSEIMAX] = 1;
170 mach->feature[FEATURE_SSEPMAX] = 1;
171 mach->feature[FEATURE_LRBNI] = 1;
172 mach->feature[FEATURE_NOPREFETCH] = 1;
173 mach->feature[FEATURE_ALIGNLOOP8] = 1;
174 mach->feature[FEATURE_ALIGNJMP8] = 1;
175 if (mach->cachesize == 0)
176 mach->cachesize = 262144;
177 break;
178
179 case TP_SKYLAKE:
180 if (! DONT_GENERATE_AVX512) {
181 mach->type[MACH_INTEL_SKYLAKE] = 1;
182 mach->feature[FEATURE_AVX512VL] = 1;
183 }
184 FLANG_FALLTHROUGH;
185
186 case TP_KNIGHTS_LANDING:
187 if (! DONT_GENERATE_AVX512) {
188 if (machtype == TP_KNIGHTS_LANDING) {
189 mach->type[MACH_INTEL_KNIGHTS_LANDING] = 1;
190 }
191 mach->feature[FEATURE_AVX512F] = 1;
192 }
193 FLANG_FALLTHROUGH;
194
195 case TP_HASWELL:
196 mach->type[MACH_INTEL_HASWELL] = 1;
197 mach->feature[FEATURE_AVX2] = 1;
198 mach->feature[FEATURE_FMA3] = 1;
199 has_fma3 = 1;
200 mach->feature[FEATURE_LD_VMOVUPD] = 1;
201 mach->feature[FEATURE_ST_VMOVUPD] = 1;
202 FLANG_FALLTHROUGH;
203
204 case TP_IVYBRIDGE:
205 case TP_SANDYBRIDGE:
206 mach->type[MACH_INTEL_SANDYBRIDGE] = 1;
207 mach->feature[FEATURE_AVX] = 1;
208 mach->feature[FEATURE_ST_MOVUPD] = 1;
209 mach->feature[FEATURE_MULTI_ACCUM] = 1;
210 FLANG_FALLTHROUGH;
211
212 case TP_NEHALEM:
213 mach->type[MACH_INTEL_NEHALEM] = 1;
214 mach->feature[FEATURE_SSE42] = 1;
215 mach->feature[FEATURE_LD_MOVUPD] = 1;
216 mach->feature[FEATURE_SSEIMAX] = 1;
217 mach->feature[FEATURE_SSEPMAX] = 1;
218 if (mach->cachesize == 0)
219 mach->cachesize = (8 * 1024 * 1024);
220 FLANG_FALLTHROUGH;
221
222 case TP_PENRYN:
223 mach->type[MACH_INTEL_PENRYN] = 1;
224 mach->feature[FEATURE_SSE41] = 1;
225 if (mach->cachesize == 0)
226 mach->cachesize = (6 * 1024 * 1024);
227 FLANG_FALLTHROUGH;
228
229 case TP_CORE2:
230 mach->type[MACH_INTEL_CORE2] = 1;
231 mach->feature[FEATURE_SSE3] = 1;
232 mach->feature[FEATURE_MNI] = 1;
233 if (mach->cachesize == 0)
234 mach->cachesize = (4 * 1024 * 1024);
235 FLANG_FALLTHROUGH;
236
237 case TP_P7:
238 /* Intel P7 Pentium IV
239 */
240 mach->type[MACH_INTEL] = 1;
241 mach->type[MACH_INTEL_PENTIUM4] = 1;
242 mach->feature[FEATURE_SSE] = 1;
243 mach->feature[FEATURE_SSE2] = 1;
244 mach->feature[FEATURE_USE_INC] = 0; /* cf. 1 for AMD */
245 mach->feature[FEATURE_USE_MOVAPD] = 1;
246 mach->feature[FEATURE_DAZ] = 1; /* cf. 0 for AMD */
247 mach->feature[FEATURE_ALIGNLOOP8] = 1;
248 mach->feature[FEATURE_ALIGNJMP8] = 1;
249 if (XBIT(80, 0x4000000)) {
250 mach->feature[FEATURE_SSE3] = 1;
251 }
252 if (machtype == TP_PENRYN || machtype == TP_CORE2 || machtype == TP_P7) {
253 mach->feature[FEATURE_NOPREFETCH] = 1;
254 }
255 if (mach->cachesize == 0)
256 mach->cachesize = (1024 * 1024);
257 break;
258
259#ifdef TARGET_X8664
260 case TP_PY:
261 case TP_PX:
262 /* we know all x86-64 have at least SSE and SSE2 */
263 /* more or less the same as p7 without sse3, without -Mdaz */
264 mach->type[MACH_INTEL] = 1;
265 mach->type[MACH_INTEL_PENTIUM4] = 1;
266 mach->feature[FEATURE_SSE] = 1;
267 mach->feature[FEATURE_SSE2] = 1;
268 mach->feature[FEATURE_USE_INC] = 0;
269 mach->feature[FEATURE_USE_MOVAPD] = 1;
270 mach->feature[FEATURE_NOPREFETCH] = 1;
271 mach->feature[FEATURE_ALIGNLOOP8] = 1;
272 mach->feature[FEATURE_ALIGNJMP8] = 1;
273 if (mach->cachesize == 0)
274 mach->cachesize = (1024 * 1024);
275 break;
276#else
277 case TP_PY: /* for 32-bit, treat like generic */
278 case TP_PX: /* " " " " " */
279#endif
280 default:
281 mach->type[MACH_GENERIC] = 1;
282 mach->feature[FEATURE_SSE] = 1;
283 mach->feature[FEATURE_SSE2] = 1;
284 mach->feature[FEATURE_USE_INC] = 0;
285 mach->feature[FEATURE_NOPREFETCH] = 1;
286 mach->feature[FEATURE_ALIGNLOOP4] = 1;
287 mach->feature[FEATURE_ALIGNJMP8] = 1;
288 if (XBIT(129, 4))
289 mach->feature[FEATURE_DAZ] = 1;
290 mach->tpval = TP_PX;
291 if (mach->cachesize == 0)
292 mach->cachesize = 262144;
293 break;
294 } /* end switch (machtype) */
295#endif /* defined(TARGET_X86) } */
296
297#if defined(TARGET_LLVM_ARM)
298 mach->feature[FEATURE_SCALAR_NEON] = 1;
299 mach->feature[FEATURE_NEON] = 1;
300 mach->feature[FEATURE_FMA] = 1;
301 has_fma3 = 1;
302#elif defined(TARGET_LLVM_POWER)
303 mach->feature[FEATURE_SCALAR_VSX] = 1;
304 mach->feature[FEATURE_VSX] = 1;
305 mach->feature[FEATURE_FMA] = 1;
306 has_fma3 = 1;
307#elif defined(X86_64)
308 /* new cg or 64-bit cg */
309 mach->feature[FEATURE_SCALAR_SSE] = 1;
310#endif

315 else if (XBIT(129, 0x400))

319

321 if (XBIT(39, 1))
322 mach->feature[FEATURE_NOPREFETCH] = 1;
323
324 /* -Mvect=simd:128 */
325 if (XBIT(56, 0x40))
326 mach->feature[FEATURE_SIMD128] = 1;
327 /* -Mvect=simd:256 or -Mvect=simd:512 */
328 if (XBIT(56, 0x100) || XBIT(56, 0x800))
329 mach->feature[FEATURE_SIMD128] = 0;
330
331 /* align 16 before loops */
332 if (XBIT(135, 0x2000) && !XBIT(135, 0x4000)) {
333 mach->feature[FEATURE_ALIGNLOOP4] = 0;
334 mach->feature[FEATURE_ALIGNLOOP8] = 0;
335 mach->feature[FEATURE_ALIGNLOOP16] = 1;
336 mach->feature[FEATURE_ALIGNLOOP32] = 0;
337 }
338 /* align 8 before loops */
339 if (XBIT(135, 0x2000) && XBIT(135, 0x4000)) {
340 mach->feature[FEATURE_ALIGNLOOP4] = 0;
341 mach->feature[FEATURE_ALIGNLOOP8] = 1;
342 mach->feature[FEATURE_ALIGNLOOP16] = 0;
343 mach->feature[FEATURE_ALIGNLOOP32] = 0;
344 }
345 /* no align before loops */
346 if (XBIT(135, 0x8000) || (XBIT(135, 0x4000) && !XBIT(135, 0x2000))) {
347 mach->feature[FEATURE_ALIGNLOOP4] = 0;
348 mach->feature[FEATURE_ALIGNLOOP8] = 0;
349 mach->feature[FEATURE_ALIGNLOOP16] = 0;
350 mach->feature[FEATURE_ALIGNLOOP32] = 0;
351 }
352 /* align 16 after jmp */
353 if (XBIT(135, 0x10000) && XBIT(135, 0x8000)) {
354 mach->feature[FEATURE_ALIGNJMP8] = 0;
355 mach->feature[FEATURE_ALIGNJMP16] = 1;
356 }
357 /* align 8 after jmp */
358 if (!XBIT(135, 0x10000) && XBIT(135, 0x8000)) {
359 mach->feature[FEATURE_ALIGNJMP8] = 1;
360 mach->feature[FEATURE_ALIGNJMP16] = 0;
361 }
362 /* no align after jmp */
363 if (XBIT(135, 0x10000) && !XBIT(135, 0x8000)) {
364 mach->feature[FEATURE_ALIGNJMP8] = 0;
365 mach->feature[FEATURE_ALIGNJMP16] = 0;
366 }

370 mach->feature[FEATURE_SCALAR_SSE] = 0;
371 else if (XBIT(172, 1))
372 mach->feature[FEATURE_SCALAR_SSE] = 1;
373 if (XBIT(171, 2))
374 mach->feature[FEATURE_SSE] = 0;
375 else if (XBIT(172, 2))
376 mach->feature[FEATURE_SSE] = 1;

379 else if (XBIT(172, 4))
380 mach->feature[FEATURE_SSE2] = 1;
381 if (XBIT(171, 8))
382 mach->feature[FEATURE_SSE3] = 0;
383 else if (XBIT(172, 8))
384 mach->feature[FEATURE_SSE3] = 1;
385 if (XBIT(171, 0x10))
386 mach->feature[FEATURE_SSE41] = 0;
387 else if (XBIT(172, 0x10))
388 mach->feature[FEATURE_SSE41] = 1;
389 if (XBIT(171, 0x20))
390 mach->feature[FEATURE_SSE42] = 0;
391 else if (XBIT(172, 0x20))
392 mach->feature[FEATURE_SSE42] = 1;
393 if (XBIT(171, 0x40))
394 mach->feature[FEATURE_SSE4A] = 0;
395 else if (XBIT(172, 0x40))
396 mach->feature[FEATURE_SSE4A] = 1;
397 if (XBIT(171, 0x80))
398 mach->feature[FEATURE_SSE5] = 0;

406 mach->feature[FEATURE_DAZ] = 0;
407 else if (XBIT(172, 0x200))
408 mach->feature[FEATURE_DAZ] = 1;
409 if (XBIT(171, 0x400))
