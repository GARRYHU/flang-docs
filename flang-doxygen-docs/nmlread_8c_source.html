417/** \param nmldesc namelist group descriptor */
418static int
419_f90io_nml_read(NML_GROUP *nmldesc)
420{
421 int err;
422
423 /* first check for errors: */
424 if (fioFcbTbls.eof)
425 return EOF_FLAG;
426 if (fioFcbTbls.error)
427 return ERR_FLAG;
428
429 assert(nmldesc);
430
431 err = find_group(nmldesc->group, nmldesc->nlen);
432 if (err != 0)
433 return err; /* error or eof condition */
434
435 /* -------- file is now positioned immediately after group name: */
436
437 live_token = 0;
438 vrf.size = 32;
439 vrf.base = (VRF *)malloc(sizeof(VRF) * vrf.size);
440 tri.size = 32;
441 tri.base = (TRIPLE *)malloc(sizeof(TRIPLE) * tri.size);
442
443 /* at this point we should call a routine that will fill
444 * the lower/upper/stride for array for array pointer/allocatable
445 * nmldesc should contain enough information
446 */
447
448 while (TRUE) { /* loop once for each namelist group item */
449 err = do_parse(nmldesc);
450 if (err == -1) /* end of group token encountered */
451 break;
452 if (err != 0)
453 goto return_err; /* error or end of file */
454 err = assign_values();
455 if (err != 0)
456 goto return_err; /* error or end of file */
457 }
458 err = 0;
459return_err:
460 free(vrf.base);
461 free(tri.base);
462 return err;
463}

465/** \brief transfer data to other processes
466 * \param nmldesc namelist group descriptor
467 *
468 */
469static void
470xfer(NML_GROUP *nmldesc)
471{
472 int num_consts;
473 int i;
474 NML_DESC *descp;
475 int pn;
476
477 /* ------ cycle through namelist entities */
478
479 /* point to first descriptor: */
480 descp = (NML_DESC *)((char *)nmldesc + sizeof(NML_GROUP));
481
482 i = 0;
483 while (TRUE) {
484
485 /* count up number of constants ( == 1 unless array): */
486 num_consts = nelems_of(descp);
487
488 /* transfer data */
489
490 pn = (descp->len > 0 ? descp->len * num_consts : num_consts);
491 if (!LOCAL_MODE) {
492 if (descp->type != __DERIVED)
493 DIST_RBCST(GET_DIST_IOPROC, descp->addr, pn, 1, descp->type);
494 else
495 DIST_RBCST(GET_DIST_IOPROC, descp->addr, pn, 1, __STR);
496 }
497
498 i++;
499 if (i >= nmldesc->ndesc)
500 break;
501
502 descp = skip_to_next(descp);
503 }
504}

506static NML_DESC *
507skip_to_next(NML_DESC *descp)
508{
509 NML_DESC *next_descp;
510 int k;
511 __POINT_T actual_ndims;
512
513 /* compute number of bytes to add to reach next descriptor: */
514 ACTUAL_NDIMS(actual_ndims);
515 if (actual_ndims >= 0)
516 k = sizeof(NML_DESC) + (actual_ndims * sizeof(__POINT_T) * 2);
517 else
518 k = sizeof(NML_DESC) + (sizeof(__POINT_T) * 2);
519 next_descp = (NML_DESC *)((char *)descp + k);
520
521 if (descp->ndims == -2 || descp->ndims >= MAX_DIM) {
522 return skip_dtio_datainit(next_descp);
523 } else if (descp->type == __DERIVED) {
524 int level = 0;
525 /* skip over all members and the members of any contained
526 * derived types.
527 */
528 while (TRUE) {
529 if (next_descp->nlen) {
530 if (next_descp->type == __DERIVED)
531 level++;
532 } else {
533 next_descp = (NML_DESC *)((char *)next_descp + sizeof(__POINT_T));
534 if (level <= 0)
535 break;
536 level--;
537 continue;
538 }
539 actual_ndims = next_descp->ndims >= MAX_DIM ? next_descp->ndims - 30
540 : next_descp->ndims;
541 if (actual_ndims >= 0) {
542 k = sizeof(NML_DESC) + (actual_ndims * sizeof(__POINT_T) * 2);
543 } else {
544 k = sizeof(NML_DESC) + (sizeof(__POINT_T) * 2);
545 }
546 next_descp = (NML_DESC *)((char *)next_descp + k);
547 }
548 }
549 return next_descp;
550}

627/** \brief search file for line which begins with '&<groupname>' */
628static int
629find_group(char *str, int nlen)
630{
631 int c;
632 int i;
633 int ret_err;
634
635 while (TRUE) {
636 ret_err = read_record();
637 if (ret_err) {
638 if (ret_err == FIO_EEOF)
639 return __fortio_eoferr(FIO_ENOGROUP);
640 return __fortio_error(ret_err);
641 }
642 while ((c = *currc++) == ' ')
643 ;
644 if (c != '$' && c != '&')
645 continue; /* eat record */
646
647 for (i = 0; i < nlen; i++) { /* compare letters of group name */
648 c = *currc++;
649 if (c >= 'A' && c <= 'Z')
650 c = c + ('a' - 'A');
651 if (str[i] != c)
652 goto eat_record;
653 }
654 c = *currc++;
655 if (IS_SPACE(c)) { /* group name matched */
656 currc--;
657 break;
658 }
659 eat_record:;
660 }
661 return 0;
662}

666static int
667get_token(void)
668{
669 static int recur = 0;
670 int i, c;
671 char delim;
672 int ret_err;
673
674 if (live_token) { /* token exists from previous call to get_token */
675 assert(live_token > 0);
676 live_token--;
677 return 0;
678 }
679
680/* scan past white space: */
681 while (TRUE) {
682 c = *currc++;
683 if (c == '\n') {
684 ret_err = read_record();
685 if (ret_err) {
686 if (ret_err == FIO_EEOF)
687 return __fortio_eoferr(FIO_ENMLEOF);
688 return __fortio_error(ret_err);
689 }
690 continue;
691 }
692 if (IS_SPACE(c))
693 continue;
694 if (c == '!') {
695 /* comment; skip to end of line:*/
696 ret_err = read_record();
697 if (ret_err) {
698 if (ret_err == FIO_EEOF)
699 return __fortio_eoferr(FIO_ENMLEOF);
700 return __fortio_error(ret_err);
701 }
702 continue;
703 }
704 break;
705 }
706
707 /* switch based on first character of token: */
708
709 switch (c) {
710 case 'A':
711 case 'B':
712 case 'C':
713 case 'D':
714 case 'E':
715 case 'F':
716 case 'G':
717 case 'H':
718 case 'I':
719 case 'J':
720 case 'K':
721 case 'L':
722 case 'M':
723 case 'N':
724 case 'O':
725 case 'P':
726 case 'Q':
727 case 'R':
728 case 'S':
729 case 'T':
730 case 'U':
731 case 'V':
732 case 'W':
733 case 'X':
734 case 'Y':
735 case 'Z':
736 c = c + ('a' - 'A');
737 FLANG_FALLTHROUGH;
738 case 'a':
739 case 'b':
740 case 'c':
741 case 'd':
742 case 'e':
743 case 'f':
744 case 'g':
745 case 'h':
746 case 'i':
747 case 'j':
748 case 'k':
749 case 'l':
750 case 'm':
751 case 'n':
752 case 'o':
753 case 'p':
754 case 'q':
755 case 'r':
756 case 's':
757 case 't':
758 case 'u':
759 case 'v':
760 case 'w':
761 case 'x':
762 case 'y':
763 case 'z':
764 case '_':
765 for (i = 0; i < MAX_TOKEN_LEN;) { /* copy ident into buffer */
766 token_buff[i] = c;
767 c = *currc++;
768 if (c >= 'A' && c <= 'Z') /* convert to lower case */
769 c = c + ('a' - 'A');
770 i++;
771 if (!((c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || c == '.' ||
772 c == '_' || c == '$'))
773 break;
774 }
775 token_buff[i] = '\0';
776
777 if (token_buff[0] == 't' || token_buff[0] == 'f') {
778 /* determine if this is a logical constant or an identifier: */
779 while (TRUE) {
780 if (c == '=' || c == '(' || c == '%')
781 goto return_ident;
782 if (!IS_SPACE(c))
783 break;
784 if (c == '\n') {
785 ret_err = read_record();
786 if (ret_err) {
787 if (ret_err == FIO_EEOF)
788 return __fortio_eoferr(FIO_ENMLEOF);
789 return __fortio_error(ret_err);
790 }
791 }
792 c = *currc++;
793 }
794
795 /* token is logical constant: */
796 constval.dtype = __BIGLOG;
797 constval.val.i = FTN_FALSE;
798 if (token_buff[0] == 't')
799 constval.val.i = FTN_TRUE;
800 currc--;
801 token = TK_CONST;
802 break;
803 }
804
805 return_ident:
806 currc--;
807 token = TK_IDENT;
808 break;
809
810 case '\'':
811 case '\"':
812 constval.dtype = __STR;
813 token = TK_CONST; /* string constant */
814 delim = c; /* delim matches first */
815 for (i = 0; i < MAX_TOKEN_LEN;) { /* copy string into buffer */
816 c = *currc++;
817 if (c == delim) {
818 c = *currc++;
819 if (c != delim) {
820 currc--; /* put back char following string */
821 break; /* exit loop */
822 }
823 } else {
824 if (c == '\r' && EOR_CRLF) {
825 c = *currc++;
826 if (c != '\n') {
827 currc--;
828 c = '\r';
829 }
830 }
831 if (c == '\n') {
832 /*
833 * do not discard 1st character of a rew record;
834 * WAS (void) __io_fgetc(fp)
835 */
836 ret_err = read_record();
837 if (ret_err) {
838 if (ret_err == FIO_EEOF)
839 return __fortio_eoferr(FIO_ENMLEOF);
840 return __fortio_error(ret_err);
841 }
842 continue; /* ignore end of line char */
843 }
844 }
845 token_buff[i++] = c;
846 }
847 token_buff[i + 1] = '\0';
848 constval.val.c.len = i;
849 constval.val.c.str = token_buff;
850 break;
851
852 case '(':
853 if (lparen_is_token) {
854 token = TK_LPAREN;
855 break;
856 }
857
858 /* else return a complex constant. Call get_token recursively to
859 process constant -which must be of form: ( TK_CONST , TK_CONST ) */
860
861 if (recur > 1) /* error if get_token is being called recursively */
862 return __fortio_error(FIO_ESYNTAX);
863 recur = 2;
864
865 GET_TOKEN(i);
866 if (token != TK_CONST || constval.dtype == __STR ||
867 constval.dtype == __NCHAR)
868 return __fortio_error(FIO_ESYNTAX);
869 cmplxval[0] = constval;
870
871 GET_TOKEN(i);
872 if (gbl->decimal == FIO_COMMA) {
873 if (token != TK_SEMICOLON)
874 return __fortio_error(FIO_ESYNTAX);
875 } else {
876 if (token != TK_COMMA)
877 return __fortio_error(FIO_ESYNTAX);
878 }
879 GET_TOKEN(i);
880 if (token != TK_CONST || constval.dtype == __STR ||
881 constval.dtype == __NCHAR)
882 return __fortio_error(FIO_ESYNTAX);
883 cmplxval[1] = constval;
884
885 GET_TOKEN(i);
886 if (token != TK_RPAREN)
887 return __fortio_error(FIO_ESYNTAX);
888
889 recur = 0;
890 token = TK_CONST;
891 constval.dtype = __BIGCPLX;
892 constval.val.cmplx = cmplxval;
893 break;
894
895 case ')':
896 token = TK_RPAREN;
897 break;
898
899 case ',':
900 if (comma_is_token) {
901 token = TK_COMMA;
902 break;
903 }
904 if (gbl->decimal != FIO_COMMA) {
905 token = TK_COMMA;
906 break;
907 }
908 c = *currc++;
909 if (c == 't' || c == 'T' || c == 'f' || c == 'F') {
910 token = TK_CONST;
911 constval.dtype = __BIGLOG;
912 constval.val.i = FTN_FALSE;
913 if (c == 't' || c == 'T')
914 constval.val.i = FTN_TRUE;
915 /* read and discard remaining alphabetic characters in token: */
916 do {
917 c = *currc++;
918 } while ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == ',' ||
919 c == '_' || c == '$');
920 currc--;
921 break;
922 }
923
924 /* else, treat this ',' as beginning of numeric constant: */
925 currc--;
926 c = ',';
927
928 i = 0;
929 goto do_numeric_token;
930
931 case ':':
932 token = TK_COLON;
933 break;
934 case ';':
935 if (gbl->decimal == FIO_COMMA)
936 token = TK_SEMICOLON;
937 break;
938 case '%':
939 token = TK_PERCENT;
940 break;
941
942 case '=':
943 token = TK_EQUALS;
944 break;
945
946 case '0':
947 case '1':
948 case '2':
949 case '3':
950 case '4':
951 case '5':
952 case '6':
953 case '7':
954 case '8':
955 case '9':
956 token_buff[0] = c;
957 for (i = 1; i < MAX_TOKEN_LEN; i++) {
958 c = *currc++;
959 ;
960 if (c < '0' || c > '9')
961 break;
962 token_buff[i] = c;
963 }
964 token_buff[i] = '\0';
965
966 if (c != '*') /* no repeat count present */
967 goto do_numeric_token;
968
969 if (c == '*') { /* REPEAT COUNT */
970 long k = atol(token_buff);
971 if (recur)
972 return __fortio_error(FIO_ELEX); /* unknown token */
973 assert(live_token == 0);
974
975 /* check character after '*' for blank, comma or e.o.g.: */
976 c = *currc++;
977 currc--;
978 if (c == ',' || IS_SPACE(c) || c == '$' || c == '&') {
979 token = TK_SKIP;
980 live_token = k - 1;
981 } else {
982 recur = 1;
983 GET_TOKEN(i);
984 recur = 0;
985 if (token != TK_CONST)
986 return __fortio_error(FIO_ESYNTAX); /* syntax error */
987 live_token = k - 1;
988 }
989 }
990 break;
991 case '.':
992 c = *currc++;
993 if (c == 't' || c == 'T' || c == 'f' || c == 'F') {
994 token = TK_CONST;
995 constval.dtype = __BIGLOG;
996 constval.val.i = FTN_FALSE;
997 if (c == 't' || c == 'T')
998 constval.val.i = FTN_TRUE;
999 /* read and discard remaining alphabetic characters in token: */
1000 do {
1001 c = *currc++;
1002 } while ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '.' ||
1003 c == '_' || c == '$');
1004 currc--;
1005 break;
1006 }
1007
1008 /* else, treat this '.' as beginning of numeric constant: */
1009 currc--;
1010 c = '.';
1011 FLANG_FALLTHROUGH;
1012 case '+':
1013 case '-':
1014 i = 0;
1015 do_numeric_token:
1016 /* scan for as long as we see characters which may be part of
1017 a numeric constant: */
1018 for (; i < MAX_TOKEN_LEN; i++) { /* copy number into buffer */
1019 char decimal = '.';
1020 if (gbl->decimal == FIO_COMMA)
1021 decimal = ',';
1022 if (!((c >= '0' && c <= '9') || c == decimal || c == 'e' || c == 'E' ||
1023 c == 'd' || c == 'D' || c == 'q' || c == 'Q' || c == '-' || c == '+'))
1024 break;
1025 token_buff[i] = c;
1026 c = *currc++;
1027 }
1028 token_buff[i] = '\0';
1029 currc--;
1030 {
1031 int type; /* 0 - integer 1 - __BIGREAL */
1032 union {
1033 __BIGINT_T i;
1034 __BIGREAL_T d;
1035 __INT8_T i8v;
1036 } val;
1037 int len, errcode;
1038
1039 if (gbl->decimal == FIO_COMMA)
1040 errcode = __fortio_getnum(token_buff, &type, &val, &len, TRUE);
1041 else
1042 errcode = __fortio_getnum(token_buff, &type, &val, &len, FALSE);
1043 if (errcode != 0)
1044 return __fortio_error(errcode);
1045 if (len != i) /* token not entirely used up */
1046 return __fortio_error(FIO_ELEX); /* unknown token */
1047 token = TK_CONST;
1048 if (type == 0) {
1049 constval.dtype = __BIGINT;
1050 constval.val.i = val.i;
1051 }
1052 else if (type == 2) {
1053 constval.dtype = __INT8;
1054 constval.val.i8v = val.i8v;
1055 }
1056 else if (type == 3) {
1057 if (!REAL_ALLOWED(VRF_DESCP(vrf_cur)->type)) {
1058 return __fortio_error(FIO_EERR_DATA_CONVERSION);
1059 } else {
1060 constval.dtype = __BIGINT;
1061 constval.val.i = val.i;
1062 }
1063 } else {
1064 constval.dtype = __BIGREAL;
1065 constval.val.d = val.d;
1066 }
1067 }
1068 break;
1069
1070 case '/': /* f90 */
1071 case '$': /* f77 */
1072 case '&': /* extension */
1073 token = TK_ENDGROUP;
1074 break;
1075
1076 default: /* no possible legal token: */
1077 return __fortio_error(FIO_ELEX); /* unknown token */
1078 }
1079
1080 return 0;
1081}

1085static int
1086do_parse(NML_GROUP *nmldesc)
1087{
1088 int i;
1089 int err;
1090 NML_DESC *descp;
1091
1092 /* The syntax for a reference is:
1093 * <ref> ::= <id> |
1094 * <ref> % <id>
1095 * <ref> ( <ss list> )
1096 * <ss list> ::= <ss list> , <ss> |
1097 * <ss>
1098 * <ss> ::= <ct> |
1099 * <opt ct> : <opt ct> <opt stride>
1100 * <opt ct> ::= |
1101 * <ct>
1102 * <opt stride> := |
1103 * : <ct>
1104 */
1105
1106 /* Begin by retrieving an identifier */
1107 GET_TOKEN(i);
1108 if (token != TK_IDENT) {
1109 if (token == TK_ENDGROUP)
1110 return -1;
1111 return __fortio_error(FIO_ENONAME); /* syntax error */
1112 }
1113
1114 /* find the matching namelist item descriptors */
1115
1116 /* point to the first item descriptor: */
1117 descp = (NML_DESC *)((char *)nmldesc + sizeof(NML_GROUP));
1118
1119 i = 0;
1120 while (TRUE) {
1121 if ((long)strlen(token_buff) == descp->nlen &&
1122 strncmp(descp->sym, token_buff, (int)descp->nlen) == 0)
1123 break;
1124
1125 i++;
1126 if (i >= nmldesc->ndesc)
1127 break;
1128
1129 descp = skip_to_next(descp);
1130 }
1131 if (i == nmldesc->ndesc) /* match not found */
1132 return NML_ERROR(FIO_ENOTMEM);
1133
1134 /* Setup for the main parsing loop: */
1135 vrf.avl = 0;
1136 tri.avl = 1;
1137 lparen_is_token = TRUE; /* want get_token to recognize '(' as a token */
1138 substring.present = FALSE;
1139 vrf_cur = add_vrf(VRF_ID, descp);
1140
1141 if (descp->ndims == -1 || descp->ndims == -2)
1142 I8(fillup_sb)(vrf_cur, descp, descp->addr);
1143
1144 err = parse_ref(descp);
1145
1146 /* set pread */
1147 if (descp->ndims == -2 || descp->ndims >= 30) {
1148 if (!internal_file)
1149 f->pread = currc;
1150 }
1151
1152 /* Cleanup after the main parsing loop */
1153 lparen_is_token = FALSE;
1154
1155 return err;
1156}

1183static int
1184parse_ref(NML_DESC *gdescp)
1185{
1186 int i, k;
1187 NML_DESC *descp, *next_descp;
1188 __POINT_T new_ndims;
1189 __POINT_T actual_ndims;
1190
1191 descp = gdescp;
1192
1193/* Enter a parsing loop searching for subobject designators */
1194
1195ref_loop:
1196 GET_TOKEN(i);
1197 ACTUAL_NDIMS(actual_ndims);
1198 switch (token) {
1199 case TK_EQUALS:
1200 return 0;
1201
1202 case TK_PERCENT:
1203 GET_TOKEN(i);
1204 if (token != TK_IDENT)
1205 return NML_ERROR(FIO_ESYNTAX);
1206 if (descp->type != __DERIVED)
1207 return NML_ERROR(FIO_ESYNTAX);
1208
1209 /* -- scan item descriptors to find one with name that matches: */
1210 /* compute number of bytes to add to reach next descriptor: */
1211 if (actual_ndims >= 0)
1212 k = sizeof(NML_DESC) + (actual_ndims * sizeof(__POINT_T) * 2);
1213 else
1214 k = sizeof(NML_DESC) + (sizeof(__POINT_T) * 2);
1215 next_descp = (NML_DESC *)((char *)descp + k);
1216 if (descp->ndims >= 30) {
1217 /* need to skip dtio data init section */
1218 next_descp = skip_dtio_datainit(descp);
1219 }
1220 while (TRUE) {
1221 if (next_descp->nlen == 0) {
1222 return NML_ERROR(FIO_ESYNTAX);
1223 }
1224 if ((long)strlen(token_buff) == next_descp->nlen &&
1225 strncmp(next_descp->sym, token_buff, next_descp->nlen) == 0)
1226 break;
1227 next_descp = skip_to_next(next_descp);
1228 }
1229 descp = next_descp;
1230 vrf_cur = add_vrf(VRF_MEMBER, descp);
1231 break;
1232
1233 case TK_LPAREN:
1234 new_ndims = *(__POINT_T *)((char *)descp + sizeof(NML_DESC));
1235 if (actual_ndims > 0 &&
1236 (VRF_TYPE(vrf_cur) == VRF_ID || VRF_TYPE(vrf_cur) == VRF_MEMBER)) {
1237 i = I8(parse_subscripts)(descp);
1238 if (i)
1239 return i;
1240 } else if ((descp->ndims == -1 || descp->ndims == -2) && new_ndims > 0) {
1241 i = I8(parse_subscripts)(descp);
1242 if (i)
1243 return i;
1244
1245 } else if (descp->type == __STR) {
1246 i = I8(parse_substring)(descp);
1247 if (i)
1248 return i;
1249 }
1250 else
1251 return NML_ERROR(FIO_ESYNTAX);
1252 break;
1253 default:
1254 return NML_ERROR(FIO_ESYNTAX);
1255 }
1256 goto ref_loop;
1257}

1271static int
1272I8(parse_subscripts)(NML_DESC *descp)
1273{
1274 int i, k, v;
1275 __POINT_T *desc_dims; /* base pointer to 2-dim descriptor array */
1276 __BIGINT_T val, upb, stride;
1277 bool is_section;
1278 __POINT_T new_ndims = 0;
1279 __POINT_T actual_ndims = 0;
1280 F90_Desc *sd;
1281 DECL_DIM_PTRS(acd);
1282
1283 ACTUAL_NDIMS(actual_ndims);
1284 desc_dims = (__POINT_T *)((char *)descp + sizeof(NML_DESC));
1285 k = -1;
1286 is_section = FALSE;
1287 new_ndims = actual_ndims;
1288 if (descp->ndims == -1 || descp->ndims == -2) {
1289 new_ndims = *(__POINT_T *)((char *)descp + sizeof(NML_DESC));
1290 sd = get_descriptor(descp);
1291 }
1292 v = add_triple(new_ndims);
1293
1294 do {
1295 k++;
1296 if (k >= new_ndims)
1297 goto subscript_error; /* too many subscripts */
1298 /* examine first token for the current dimension */
1299 comma_is_token = TRUE;
1300 GET_TOKEN(i);
1301 if (descp->ndims == -1 || descp->ndims == -2)
1302 SET_DIM_PTRS(acd, sd, k);
1303 if (token == TK_CONST && constval.dtype == __BIGINT) {
1304 val = constval.val.i;
1305
1306 if (descp->ndims == -1 || descp->ndims == -2) {
1307 if (val < F90_DPTR_LBOUND_G(acd) || val > F90_DPTR_EXTENT_G(acd))
1308 goto subscript_error; /* subscr out of range */
1309 } else if (val < desc_dims[2 * k] || val > desc_dims[2 * k + 1]) {
1310 goto subscript_error; /* subscr out of range */
1311 }
1312
1313 /* Is this constant a subscipt or the lower bound of a section? */
1314
1315 GET_TOKEN(i);
1316 if (token == TK_COMMA || token == TK_RPAREN) {
1317 /* the constant is a subscript */
1318 TRI_LWB(v, k) = val;
1319 TRI_UPB(v, k) = val;
1320 TRI_STRIDE(v, k) = 1;
1321 if (descp->ndims == -1 || descp->ndims == -2) {
1322 sb.sect[k].lwb = val;
1323 sb.sect[k].upb = val;
1324 sb.sect[k].stride = 1;
1325 }
1326 continue;
1327 }
1328
1329 /* Expect to see a : */
1330
1331 if (token != TK_COLON)
1332 goto subscript_error;
1333 } else if (token == TK_COLON) {
1334 if (descp->ndims == -1 || descp->ndims == -2)
1335 val = F90_DPTR_LBOUND_G(acd);
1336 else
1337 val = desc_dims[2 * k]; /* default lower bound */
1338 } else
1339 goto subscript_error;
1340
1341 /* So far, we've parsed
1342 * <c> :
1343 * or
1344 * :
1345 * in which case we now have a lower bound of a section - determine
1346 * the upper bound.
1347 */
1348 is_section = TRUE;
1349 if (descp->ndims == -1 || descp->ndims == -2) {
1350 upb = F90_DPTR_EXTENT_G(acd);
1351 } else {
1352 upb = desc_dims[2 * k + 1]; /* default upper bound */
1353 }
1354 stride = 1; /* default stride */
1355 GET_TOKEN(i);
1356 if (token == TK_CONST && constval.dtype == __BIGINT) {
1357 upb = constval.val.i;
1358 if (descp->ndims == -1 || descp->ndims == -2) {
1359 if (upb < F90_DPTR_LBOUND_G(acd) || upb > F90_DPTR_EXTENT_G(acd))
1360 goto subscript_error; /* subscr out of range */
1361 } else if (upb < desc_dims[2 * k] || upb > desc_dims[2 * k + 1])
1362 goto subscript_error; /* subscr out of range */
1363
1364 /* Found <c> as the upper bound; check for stride. */
1365
1366 GET_TOKEN(i);
1367 if (token == TK_COLON) {
1368 /* expect a constant */
1369 GET_TOKEN(i);
1370 if (token != TK_CONST || constval.dtype != __BIGINT)
1371 goto subscript_error;
1372 stride = constval.val.i;
1373 if (stride < 0)
1374 goto subscript_error; /* subscr out of range */
1375 GET_TOKEN(i);
1376 }
1377 } else if (token != TK_COMMA && token != TK_RPAREN)
1378 goto subscript_error;
1379
1380 TRI_LWB(v, k) = val;
1381 TRI_UPB(v, k) = upb;
1382 TRI_STRIDE(v, k) = stride;
1383 if (descp->ndims == -1 || descp->ndims == -2) {
1384 sb.sect[k].lwb = val;
1385 sb.sect[k].upb = upb;
1386 sb.sect[k].stride = stride;
1387 }
1388
1389 } while (token != TK_RPAREN);
1390 comma_is_token = FALSE;
1391
1392 if (descp->ndims == -1 || descp->ndims == -2) {
1393 VRF_SUBSCRIPT(vrf_cur) = v;
1394 if (k != new_ndims - 1)
1395 goto subscript_error;
1396 } else if (k != actual_ndims - 1)
1397 goto subscript_error;
1398
1399 if (is_section)
1400 vrf_cur = add_vrf(VRF_SECTION, descp);
1401 else
1402 vrf_cur = add_vrf(VRF_ELEMENT, descp);
1403 VRF_SUBSCRIPT(vrf_cur) = v;
1404
1405 return 0; /* no errors encountered */
1406
1407subscript_error:
1408 return NML_ERROR(FIO_ESUBSC);
1409}

1411static int
1412I8(parse_substring)(NML_DESC *descp)
1413{
1414 int i;
1415 __BIGINT_T val, end;
1416 F90_Desc *sd;
1417
1418 val = 1; /* default starting value */
1419 GET_TOKEN(i);
1420 if (token == TK_CONST && constval.dtype == __BIGINT) {
1421 val = constval.val.i;
1422 GET_TOKEN(i); /* expect a : */
1423 }
1424 if (token != TK_COLON) /* illegal substring spec */
1425 return NML_ERROR(FIO_ESUBSC);
1426
1427 /* At this point, ":" of substring descriptor has been read and
1428 * 'val' contains the value of the starting position.
1429 */
1430 if (descp->ndims == -1 || descp->ndims == -2) {
1431 sd = get_descriptor(descp);

1513/*
1514 * Recursively compute the index space given a set of subscripts for n
1515 * dimensions. The evaluation begins by iterating over the last dimension,
1516 * recursively evaluating the subscripts of the next (to the left) for
1517 * each iteration. For a given dimension d's index, subscripts to the left
1518 * are recursively computed. When the first dimension is reached, the address
1519 * of the element represented by the current subscript values is passed to
1520 * the 'eval' function.
1521 */

1523I8(eval_sb)(int d)
1524{
1525 int j, err;
1526 __BIGINT_T offset;
1527 char *new_addr;
1528 NML_DESC *descp;
1529 __POINT_T *desc_dims;
1530 F90_Desc *sd;
1531

1535 desc_dims = (__POINT_T *)((char *)descp + sizeof(NML_DESC));

1540 if (err)
1541 return err;
1542 return 0;
1543 }
1544 }
1545 if (d == 0) {

1547 * Reached the first dimension; iterate over the first dimension,
1548 * compute the address of each element, and pass each address to
1549 * 'eval'.
1550 */

1552 for (sb.idx[0] = sb.sect[0].lwb; sb.idx[0] <= sb.sect[0].upb;
1553 sb.idx[0] += sb.sect[0].stride) {

1574 * subscripts in the dimensions to the left.

1685static int
1686eval(int v, char *loc_addr)
1687{
1688 NML_DESC *descp;
1689 int i, k;
1690 __BIGINT_T offset, mm;
1691 char *new_addr;
1692 __POINT_T *desc_dims; /* base pointer to 2-dim descriptor array */
1693 __POINT_T actual_ndims;
1694
1695 if (v > vrf_cur) {
1696
1697 descp = VRF_DESCP(v - 1);
1698 if (descp->ndims == -1)
1699 return I8(eval_sb)(sb.ndims - 1);
1700 else if (descp->ndims == -2)
1701 return I8(eval_dtio_sb)(sb.ndims - 1);
1702 switch (VRF_TYPE(v - 1)) {
1703 case VRF_ELEMENT:
1704 /* subscripted reference, assign() stores a scalar but there
1705 * may be additional values.
1706 */
1707 return assign(VRF_DESCP(v - 1), loc_addr, NULL, FALSE, TRUE);
1708 case VRF_SECTION:
1709 /* for this type, we're already iterating over an index space;
1710 * just have assign store a scalar.
1711 */
1712 return assign(VRF_DESCP(v - 1), loc_addr, NULL, FALSE, FALSE);
1713 default:
1714 break;
1715 }
1716 return assign(VRF_DESCP(v - 1), loc_addr, NULL, TRUE, FALSE);
1717 }
1718
1719 new_addr = loc_addr;
1720 descp = VRF_DESCP(v);
1721 ACTUAL_NDIMS(actual_ndims);
1722 switch (VRF_TYPE(v)) {
1723 case VRF_ID:
1724 if (descp->ndims == -1) /* scalar pointer - getting $p */
1725 new_addr = *(char **)descp->addr;
1726 else
1727 new_addr = descp->addr;
1728 break;
1729
1730 case VRF_ELEMENT:
1731 k = VRF_SUBSCRIPT(v);
1732 desc_dims = (__POINT_T *)((char *)descp + sizeof(NML_DESC));
1733 if (descp->ndims != -1 && descp->ndims != -2) {
1734 offset = TRI_LWB(k, 0) - desc_dims[0];
1735 mm = 1; /* multiplier for each dimension */
1736 for (i = 1; i < actual_ndims; i++) {
1737 mm *= desc_dims[2 * (i - 1) + 1] - desc_dims[2 * (i - 1)] + 1;
1738 offset += (TRI_LWB(k, i) - desc_dims[2 * i]) * mm;
1739 }
1740 offset *= I8(siz_of)(descp);
1741 new_addr += offset;
1742 }
1743 break;
1744
1745 case VRF_SECTION:
1746 k = VRF_SUBSCRIPT(v);
1747
1748 desc_dims = (__POINT_T *)((char *)descp + sizeof(NML_DESC));
1749 /*
1750 * Copy the section information into the looping sect structure.
1751 * Compute the multipliers for each dimension.
1752 */
1753 if (descp->ndims != -1 && descp->ndims != -2) {
1754 sb.mult[0] = 1;
1755 sb.sect[0] = TRI_SECT(k, 0);
1756 sb.lwb[0] = desc_dims[0];
1757 sb.ndims = actual_ndims;
1758 sb.loc_addr = loc_addr;
1759 for (i = 1; i < actual_ndims; i++) {
1760 sb.lwb[i] = desc_dims[2 * i];
1761 sb.mult[i] = sb.mult[i - 1] *
1762 (desc_dims[2 * (i - 1) + 1] - desc_dims[2 * (i - 1)] + 1);
1763 sb.sect[i] = TRI_SECT(k, i);
1764 }
1765 }
1766 sb.v = v;
1767 sb.elemsz = I8(siz_of)(descp);
1768 if (descp->ndims == -2) {
1769 return I8(eval_dtio_sb)(sb.ndims - 1);
1770 } else if (descp->ndims >= MAX_DIM) {
1771 return I8(eval_dtio_sb)(sb.ndims - 1);
1772 }
1773 return I8(eval_sb)(sb.ndims - 1);
1774
1775 case VRF_MEMBER:
1776 new_addr = loc_addr + (long)descp->addr;
1777 break;
1778 }
1779
1780 return eval(v + 1, new_addr);
1781}
