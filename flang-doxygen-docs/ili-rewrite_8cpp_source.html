23/* Determines whether an ILI that has been passed to addili() was stored
24 * without change. If this be not the case, addili() can be assumed to
25 * have rewritten the expression, possibly to improve it by having
26 * recognized and replaced an expression (but possibly also just to
27 * canonicalize it).
28 */
29static bool
30addili_changed_ili(int ili, const ILI *saved)
31{
32 int j, opc = ILI_OPC(ili), opnds = IL_OPRS(opc);
33 if (opc != saved->opc)
34 return true;
35 for (j = 1; j <= opnds; ++j)
36 if (ILI_OPND(ili, j) != saved->opnd[j - 1])
37 return true;
38 return ILI_ALT(ili) != saved->alt;
39}

41/* JSRs have special semantics: their ILIs must be treated as
42 * common subexpressions and evaluated at most once in any
43 * non-extended basic block.
44 */
45static bool
46is_jsr(int opc)
47{
48 switch (opc) {
49 case IL_JSR:
50 case IL_JSRA:
51/* case IL_QJSR: no, QJSR is discretionary */
52 case IL_GJSR:
53 case IL_GJSRA:
54#ifdef LONG_DOUBLE_FLOAT128
55 case IL_FLOAT128RESULT:
56#endif
57 return true;
58 }
59 return false;
60}

73/* Visits ILI that are referenced as subscripts in name table (NME)
74 * operands of load/store ILIs.
75 */
76static int
77visit_nme_subscript(int nme, bool *operand_cse, ILI_coordinates *at,
78 struct hidden_state *state)
79{
80 if (nme > NME_VOL /* && state->is_context_insensitive */) {
81 if (NME_TYPE(nme) == NT_IND || NME_TYPE(nme) == NT_MEM) {
82 int base = NME_NM(nme);
83 int new_base = visit_nme_subscript(base, operand_cse, at, state);
84 if (new_base != base)
85 nme = addnme(NME_TYPE(nme), NME_SYM(nme), new_base, NME_CNST(nme));
86 } else if (NME_TYPE(nme) == NT_ARR) {
87 int base = NME_NM(nme);
88 int sub = NME_SUB(nme);
89 int new_base = visit_nme_subscript(base, operand_cse, at, state);
90 int new_sub;
91 at->ili = sub;
92 at->original_ili = sub;
93 new_sub = visit_ili_operands(operand_cse, at, state);
94 if (new_base != base || new_sub != sub)
95 nme = add_arrnme(NT_ARR, NME_SYM(nme), new_base, NME_CNST(nme), new_sub,
96 NME_INLARR(nme));
97 }
98 }
99 return nme;
100}

102/* Recursive traversal of ILI expression trees with a visitation callback.
103 * Returns the ILI index of the replacement operand.
104 */
105static int
106visit_ili_operands(bool *result_cse, const ILI_coordinates *at,
107 struct hidden_state *state)
108{
109 ILI_OP opc = ILI_OPC(at->ili);
110 int opnds = IL_OPRS(opc);
111 ILI new_ili;
112 bool any_change = false;
113 ILI_coordinates here = *at;
114 int orig_ili = at->original_ili;
115 int result = state->ili_map[orig_ili];
116 bool operand_cse = false;
117 int j;
118
119 if (DEBUGGING)
120 dbgprintf("visit_ili_operands: begin ili %d original %d, ilt %d, "
121 "bih %d\n",
122 at->ili, at->original_ili, at->ilt, at->bih);
123
124 CHECK(orig_ili >= 0 && orig_ili < state->limit);
125
126 /* Respect JSR and CSE block semantics if this expression has already
127 * been visited in this block.
128 * 1) Each distinct JSR ILI number produces at most one call per block.
129 * JSRs are automatically common subexpressions. Distinct JSR ILIs
130 * are distinct calls.
131 * 2) CSE ILIs re-use the value produced by the most recent evaluation
132 * of their operand ILI in the same block. If the operand ILI appears
133 * outside a CSE ILI, it is re-evaluated. Expressions that are used
134 * as CSEs often appear in FREE ILIs but are not required to do so.
135 * The expander and optimizer should take care to avoid potential
136 * ambiguity, such as having the same ILI appear within and without
137 * CSE ILIs in the same statement.
138 */
139 if (result >= 0) {
140 if (state->is_context_insensitive) {
141 if (DEBUGGING)
142 dbgprintf("visit_ili_operands: end ili %d (orig %d), "
143 "cached result %d\n",
144 at->ili, orig_ili, result);
145 return result;
146 } else if (state->ili_bih[orig_ili] != at->bih) {
147 result = -1; /* cached result no longer valid */
148 } else if (is_jsr(opc)) {
149 if (DEBUGGING)
150 dbgprintf("visit_ili_operands: end JSR ili %d (orig %d), "
151 "forced block CSE %d\n",
152 at->ili, orig_ili, result);
153 CHECK(result > 0 && result < state->limit);
154 CHECK(state->ili_backmap[result] == orig_ili);
155 *result_cse = true;
156 return result;
157 } else if (at->parent && at->parent_opnd == 1 &&
158 is_cseili_opcode(ILI_OPC(at->parent->ili))) {
159 /* This ILI is the operand of a CSE ILI. */
160 if (DEBUGGING)
161 dbgprintf("visit_ili_operands: end ili %d (orig %d), "
162 "CSE result %d\n",
163 at->ili, orig_ili, result);
164 *result_cse = true;
165 return result;
166 }
167 }
168
169 /* Check for circular dependence */
170 CHECK(state->ili_map[orig_ili] >= -1);
171 state->ili_map[orig_ili] = -2;
172
173 /* Recursively visit operands */
174 here.parent = at;
175 for (j = 1; j <= opnds; ++j) {
176 int opd = ILI_OPND(at->ili, j), original_opd = opd;
177 if (IL_ISLINK(opc, j)) {
178 here.ili = opd;
179 here.original_ili = opd;
180 here.parent_opnd = j;
181 opd = visit_ili_operands(&operand_cse, &here, state);
182 any_change |= opd != original_opd;
183 } else if (IL_OPRFLAG(opc, j) == ILIO_NME) {
184 here.parent_opnd = j;
185 opd = visit_nme_subscript(opd, &operand_cse, &here, state);
186 any_change |= opd != original_opd;
187 }
188 new_ili.opnd[j - 1] = opd;
189 }
190
191 /* Visit alt code, too */
192 here.ili = ILI_ALT(at->ili);
193 if (here.ili > 0) {
194 here.original_ili = here.ili;
195 here.parent_opnd = 0;
196 new_ili.alt = visit_ili_operands(&operand_cse, &here, state);
197 any_change |= new_ili.alt != here.ili;
198 } else {
199 new_ili.alt = 0;
200 }
201
202 if (!any_change) {
203 result = at->visitor(at);
204 if (DEBUGGING)
205 dbgprintf("visit_ili_operands: ili %d, no operand changes, "
206 "visitor returned result %d\n",
207 at->ili, result);
208 } else {
209 /* At least one of the operands to this ILI has changed.
210 * Construct a new ILI and see whether it enjoys any improvement
211 * in addili().
212 */
213 new_ili.opc = opc;
214 here = *at;
215 here.original_ili = at->ili;
216 if (is_jsr(opc)) {
217 /* Observe special semantics for JSRs, which are *always* CSEs
218 * in the blocks that contain them. Create a new distinct JSR
219 * so that we don't accidentally alias another JSR in the same
220 * block.
221 */
222 here.ili = get_ili_ns(&new_ili);
223 CHECK(here.ili > 0);
224 if (new_ili.alt != 0) {
225 if (ILI_ALT(here.ili) == 0)
226 ILI_ALT(here.ili) = new_ili.alt;
227 else
228 CHECK(ILI_ALT(here.ili) == new_ili.alt);
229 }
230 } else {
231 here.ili = addili(&new_ili);
232 }
233 if (here.ili == 0) {
234 /* conditional jump was nullified */
235 if (DEBUGGING)
236 dbgprintf("visit_ili_operands: ili %d, result is 0 after "
237 "addili() with changed operands\n",
238 at->ili);
239 result = 0;
240 } else {
241 /* Improvement test:
242 * Traversal has visited the operands of this ILI instance, at least
243 * one of which has changed, and the ILI has been reconstituted.
244 * It will be characterized as having been "improved" if the
245 * expression finally stored by addili() differs from what was
246 * passed into it, and so it may be assumed that a local pattern
247 * recognition and replacement transformation has affected the
248 * expression. If an operand changed was a forced move due to
249 * JSR or CSE semantics, that is also conveyed.
250 */
251 if (new_ili.alt > 0 && new_ili.alt != ILI_ALT(at->ili)) {
252 here.ili = get_ili_ns(&new_ili);
253 if (ILI_ALT(here.ili) == 0)
254 ILI_ALT(here.ili) = new_ili.alt;
255 else
256 CHECK(ILI_ALT(here.ili) == new_ili.alt);
257 }
258 here.this_ili_improved = addili_changed_ili(here.ili, &new_ili);
259 here.has_cse = operand_cse;
260 result = at->visitor(&here);
261 if (DEBUGGING)
262 dbgprintf("visit_ili_operands: ili %d, operands changed; "
263 "addili() returned %d (improved? %d forced? %d), "
264 "visitor result %d\n",
265 at->ili, here.ili, here.this_ili_improved, operand_cse,
266 result);
267 }
268 }
269
270 /* Bookkeeping of results */
271 if (result >= state->limit) {
272 int new_limit = 2 * result;
273 int dummy = state->limit; /* so NEED can overwrite it */
274 NEED(new_limit, state->ili_map, int, dummy, new_limit);
275 dummy = state->limit;
276 NEED(new_limit, state->ili_bih, int, dummy, new_limit);
277 dummy = state->limit;
278 NEED(new_limit, state->ili_backmap, int, dummy, new_limit);
279 for (j = state->limit; j < new_limit; ++j) {
280 state->ili_map[j] = -1;
281 state->ili_bih[j] = -1;
282 state->ili_backmap[j] = -1;
283 }
284 state->limit = new_limit;
285 }
286
287 if (DEBUGGING)
288 dbgprintf("visit_ili_operands: end ili %d original %d, result %d\n",
289 at->ili, orig_ili, result);
290 state->ili_map[orig_ili] = result;
291 state->ili_bih[orig_ili] = at->bih;
292 if (result >= 0)
293 state->ili_backmap[result] = orig_ili;
294 *result_cse |= operand_cse;
295 return result;
296}

298/* Driver for context-sensitive ILI traversal.
299 * Returns true if any ILI changed.
300 */
301bool
302visit_ilis(ILI_visitor visitor, void *visitor_context,
303 bool is_context_insensitive)
304{
305 int j;
306 bool any = false;
307 ILI_coordinates at;
308 struct hidden_state state;
309
310 memset(&state, 0, sizeof state);
311 state.limit = 2 * ilib.stg_avail;
312 NEW(state.ili_map, int, state.limit);
313 NEW(state.ili_bih, int, state.limit);
314 NEW(state.ili_backmap, int, state.limit);
315 state.is_context_insensitive = is_context_insensitive;
316 for (j = 0; j < state.limit; ++j) {
317 state.ili_map[j] = -1;
318 state.ili_bih[j] = -1;
319 state.ili_backmap[j] = -1;
320 }
321
322 at.visitor = visitor;
323 at.context = visitor_context;
324 at.parent = NULL;
325 at.parent_opnd = 0;
326 at.this_ili_improved = false;
327 for (at.bih = gbl.entbih; at.bih > 0; at.bih = BIH_NEXT(at.bih)) {
328 int next_ilt;
329 rdilts(at.bih);
330 for (at.ilt = BIH_ILTFIRST(at.bih); at.ilt > 0; at.ilt = next_ilt) {
331 int new_ili;
332 bool has_cse = false;
333 next_ilt = ILT_NEXT(at.ilt);
334 at.ili = ILT_ILIP(at.ilt);
335 at.original_ili = at.ili;
336 new_ili = visit_ili_operands(&has_cse, &at, &state);
337 if (new_ili != at.original_ili) {
338 /* ILI has changed for this statement */
339 any = true;
340 if (new_ili == 0) {
341 /* conditional branch was nullified */
342 new_ili = ad1ili(IL_NULL, 0);
343 ILT_BR(at.ilt) = 0;
344 if (DEBUGGING)
345 dbgprintf("visit_ilis: ilt %d nullified\n", at.ilt);
346 } else {
347 if (ILI_OPC(new_ili) == IL_JMP)
348 BIH_FT(at.bih) = 0;
349 if (DEBUGGING)
350 dbgprintf("visit_ilis: ilt %d: ili %d -> %d "
351 "(cses? %d)\n",
352 at.ilt, at.ili, new_ili, has_cse);
353 }
354 ILT_ILIP(at.ilt) = new_ili;
355 }
356 }
357 wrilts(at.bih);
358 if (BIH_LAST(at.bih))
359 break;
360 }
361
362 FREE(state.ili_backmap);
363 FREE(state.ili_bih);
364 FREE(state.ili_map);
365 return any;
366}

368/* Create a new ILI that is a copy of another with a potential
369 * change at one operand index. This function is suitable for
370 * use by a visitation callback. "opnd_index" is one-based,
371 * and -1 refers to alt code.
372 */
373int
374update_ili_operand(int ili, int opnd_index, int new_opnd)
375{
376 if (ILI_OPND(ili, opnd_index) != new_opnd) {
377 int j;
378 ILI_OP opc = ILI_OPC(ili);
379 int opnds = IL_OPRS(opc);
380 ILI newili;
381 newili.opc = opc;
382 if (opnd_index == -1)
383 newili.alt = new_opnd;
384 else
385 newili.alt = ILI_ALT(ili);
386 for (j = 1; j <= opnds; ++j) {
387 if (j == opnd_index)
388 newili.opnd[j - 1] = new_opnd;
389 else
390 newili.opnd[j - 1] = ILI_OPND(ili, j);
391 }
392 ili = addili(&newili);
393 if (newili.alt != 0 && newili.alt != ILI_ALT(ili)) {
394 ili = get_ili_ns(&newili);
395 if (ILI_ALT(ili) == 0)
396 ILI_ALT(ili) = newili.alt;
397 }
398 }
399 return ili;
400}

402/* Collect the set of ILIs that are ILT expression roots. */
403void
404collect_root_ilis(fastset *root_ilis)
405{
406 int bih, ilt;
407
408 fastset_vacate(root_ilis);
409 for (bih = gbl.entbih; bih > 0; bih = BIH_NEXT(bih)) {
410 rdilts(bih);
411 for (ilt = BIH_ILTFIRST(bih); ilt > 0; ilt = ILT_NEXT(ilt)) {
412 int ili = ILT_ILIP(ilt);
413 fastset_add(root_ilis, ili);
414 }
415 wrilts(bih);
416 if (BIH_LAST(bih))
417 break;
418 }
419}

439bool
440scan_ili_tree(ILI_tree_scan_visitor visitor, void *visitor_context, int ili)
441{
442 int j;
443 ILI_OP opc;
444 int opnds;
445
446 if (visitor(visitor_context, ili))
447 return true;
448 if (is_cseili_opcode(ILI_OPC(ili)))
449 return false;
450 opc = ILI_OPC(ili);
451 opnds = IL_OPRS(opc);
452 for (j = 1; j <= opnds; ++j)
453 if (IL_ISLINK(opc, j))
454 if (scan_ili_tree(visitor, visitor_context, ILI_OPND(ili, j)))
455 return true;
456 if (ILI_ALT(ili))
457 return scan_ili_tree(visitor, visitor_context, ILI_ALT(ili));
458 return false;
459}
