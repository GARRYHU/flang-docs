124/*
125 * A direction vector is a 32-bit integer. It consists of up to
126 * 7 4-bit fields, together with a 4-bit type field.
127 * The 4-bit fields give the directions <, =, >, and reduction;
128 * the type field gives the information as to whether the all-equals
129 * direction is allowed.
130 *
131 * For a loop nest, the innermost loop corresponds to the rightmost bit
132 * field. This bit field is accessed with value 0. And so forth.
133 */
134
135typedef BIGUINT64 DIRVEC;
136
137#define DIRV_ENTSIZ 4
138#define DIRV_ENTMSK (DIRVEC)15
139
140/* get dirvec entry */
141#define DIRV_ENTRYG(dirv, idx) (((dirv) >> ((idx)*DIRV_ENTSIZ)) & DIRV_ENTMSK)
142/* put dirvec entry (assume clear) */
143#define DIRV_ENTRYP(dirv, idx, dirent) \
144 ((dirv) |= ((dirent & DIRV_ENTMSK) << ((idx)*DIRV_ENTSIZ)))
145/* clear dirvec entry */
146#define DIRV_ENTRYC(dirv, idx) ((dirv) &= ~(DIRV_ENTMSK << ((idx)*DIRV_ENTSIZ)))
147
148#define DIRV_ALLEQ ((DIRVEC)0x8000000000000000L)
149#define DIRV_MASK ((DIRVEC)0x0FFFFFFFFFFFFFFFL)
150
151/*
152 * Dirvec Bits
153 */
154#define DIRV_LT (DIRVEC)1L
155#define DIRV_EQ (DIRVEC)2L
156#define DIRV_GT (DIRVEC)4L
157#define DIRV_RD (DIRVEC)8L
158#define DIRV_NONE (DIRVEC)0L
159
160#define DIRV_STAR (DIRV_LT | DIRV_EQ | DIRV_GT)
161
162/*
163 * Macros to get dirvec portion and information portion
164 */
165#define DIRV_DIRPART(dirv) ((dirv) & (DIRV_MASK))
166#define DIRV_INFOPART(dirv) ((dirv) & ~(DIRV_MASK))
