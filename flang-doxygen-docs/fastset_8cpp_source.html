30/** \brief Verifies that each member of the set corresponds to a valid index
31 * entry. */
32void
33fastset_check(const fastset *set)
34{
35 if (set) {
36 unsigned j;
37 for (j = 0; j < set->members; ++j) {
38 CHECK(set->member[j] < set->limit);
39 CHECK(set->index[set->member[j]] == j);
40 }
41 }
42}

77void
78fastset_resize(fastset *set, unsigned limit)
79{
80 limit += FASTSET_GRANULARITY - 1;
81 limit &= -FASTSET_GRANULARITY;
82 if (limit > set->limit) {
83 unsigned i, n, *new_member, *new_index;
84
85 /* Do *not* use realloc, since it takes time O(set->limit) and we want
86 to take time O(set->members) when !DEBUG. */
87
88 /* Index array is redundant, so it can be freed now. */
89 free(set->index);
90
91 /* Reallocate the member array. */
92 n = set->members;
93 new_member = (unsigned *) malloc(limit * sizeof *set->member);
94 assert(new_member, "fastset_resize: out of memory", limit, ERR_Fatal);
95 memcpy(new_member, set->member, n * sizeof *set->member);
96 free(set->member);
97
98 /* Allocate the index array and rebuild the index. */
99 new_index = (unsigned *) malloc(limit * sizeof *set->index);
100 assert(new_index, "fastset_resize: out of memory", limit, ERR_Fatal);
101#if DEBUG+0
102 /* It's not necessary to initialize any of this allocated storage
103 * for correctness or safety, but the uninitialized index[] array
104 * produces warnings from valgrind. So overwrite it in DEBUG
105 * mode. The member[] entries will not be overindexed past
106 * "->members", so leave it uninitialized.
107 *
108 * Unfortunately this takes time O(set->limit).
109 */
110 memset(new_index, 0xaa, limit * sizeof *set->index);
111#endif
112 for (i = 0; i < n; ++i)
113 new_index[new_member[i]] = i;
114
115 set->member = new_member;
116 set->index = new_index;
117 set->limit = limit;
118 }
119}

121/** \brief Set union, destructive.
122 *
123 * Just call fastset_add() for every member of the second set argument.
124 */
125void
126fastset_union(fastset *xs, const fastset *ys)
127{
128 unsigned j;
129 for (j = 0; j < ys->members; ++j)
130 fastset_add(xs, ys->member[j]);
131}

133/** Internal utility: keep a set's member list contiguous by moving its
134 * last member into a vacated position.
135 */
136INLINE static void
137fastset_internal_drop_at(fastset *set, unsigned idx)
138{
139 unsigned last = set->member[--set->members];
140 set->member[set->index[last] = idx] = last;
141}

143/** \brief Set difference, destructive.
144 *
145 * Algorithm depends on the number of members in the two sets.
146 */
147void
148fastset_difference(fastset *xs, const fastset *ys)
149{
150 unsigned j = 0;
151 if (xs->members < ys->members) {
152 /* xs is small - sweep it and test membership in ys */
153 while (j < xs->members) {
154 if (fastset_contains(ys, xs->member[j]))
155 fastset_internal_drop_at(xs, j);
156 else
157 ++j;
158 }
159 } else if (xs == ys) {
160 fastset_vacate(xs);
161 } else {
162 /* ys is small - delete its members from xs */
163 while (j < ys->members)
164 fastset_remove(xs, ys->member[j++]);
165 }
166}

168/** \brief Set intersection, destructive.
169 *
170 * Algorithm depends on the number of members in the two sets.
171 */
172void
173fastset_intersection(fastset *xs, const fastset *ys)
174{
175 unsigned j;
176 if (xs->members <= ys->members) {
177 /* xs is small - sweep it and test membership in ys */
178 if (xs != ys) {
179 for (j = 0; j < xs->members;) {
180 if (fastset_contains(ys, xs->member[j]))
181 ++j;
182 else
183 fastset_internal_drop_at(xs, j);
184 }
185 }
186 } else {
187 /* ys is small - sweep it and move any common members to the front
188 * of xs->member[]
189 */
190 unsigned keep = 0;
191 for (j = 0; j < ys->members; ++j) {
192 unsigned y = ys->member[j];
193 if (y < xs->limit) {
194 unsigned idx = xs->index[y];
195 if (idx < xs->members && xs->member[idx] == y) {
196 unsigned tmp = xs->member[keep];
197 xs->member[xs->index[y] = keep++] = y;
198 xs->member[xs->index[tmp] = idx] = tmp;
199 }
200 }
201 }
202 xs->members = keep;
203 }
204}

206/** \brief Apply a function to each member of a set. */
207void *
208fastset_map(const fastset *xs, void *f(void *, int), void *pointer)
209{
210 if (xs) {
211 int j, count = xs->members;
212 for (j = 0; j < count; ++j)
213 pointer = f(pointer, xs->member[j]);
214 }
215 return pointer;
216}
