51/** Add action to action map */
52void
53add_action(action_map_t *map, const char *keyword, void (*action)(void))
54{
55 action_list_t *action_list = NULL;
56
57#pragma GCC diagnostic push
58#pragma GCC diagnostic warning "-Wcast-qual"
59 /* Check if there is something in the map already */
60 if (hashmap_lookup(map->actions, keyword, (hash_data_t *)&action_list)) {
61 /* Add one more element at the end of the list */
62 ++action_list->count;
63 action_list->actions = (void(**)())
64 realloc(action_list->actions, sizeof(action_t) * action_list->count);
65 action_list->actions[action_list->count - 1] = action;
66 } else {
67 /* Create brand new action list */
68 action_list = (action_list_t *)malloc(sizeof(action_list_t));
69 action_list->count = 1;
70 action_list->actions = (void(**)()) malloc(sizeof(action_t));
71 *(action_list->actions) = action;
72 /* Add it to the list */
73 hashmap_insert(map->actions, keyword, action_list);
74 }
75#pragma GCC diagnostic pop
76}

78/** Execute action(s) for a given keyword */
79void
80execute_actions_for_keyword(action_map_t *map, const char *keyword)
81{
82 action_list_t *action_list = NULL;
83
84#pragma GCC diagnostic push
85#pragma GCC diagnostic warning "-Wcast-qual"
86 /* Execute if there is anything for this keyword */
87 if (hashmap_lookup(map->actions, keyword, (hash_data_t *)&action_list)) {
88 size_t index;
89 for (index = 0; index < action_list->count; ++index) {
90 action_list->actions[index]();
91 }
92 }
93#pragma GCC diagnostic pop
94 /* Do nothing is there is no record */
95}

97/** Copy an action from one map to another */
98void
99copy_action(const action_map_t *from, const char *keyword_from,
100 action_map_t *to, const char *keyword_to)
101{
102 action_list_t *source_actions = NULL;
103 action_list_t *dest_actions = NULL;
104
105#pragma GCC diagnostic push
106#pragma GCC diagnostic warning "-Wcast-qual"
107 /* Silently exit if nothing is found in source dataset */
108 if (!hashmap_lookup(from->actions, keyword_from,
109 (hash_data_t *)&source_actions)) {
110 return;
111 }
112
113 const size_t count = source_actions->count;
114
115 /* Check if there is something in the destination map already */
116 if (hashmap_lookup(to->actions, keyword_to, (hash_data_t *)&dest_actions)) {
117 /* Copy the array of function pointers */
118 dest_actions->count = count;
119 dest_actions->actions = (void(**)())
120 realloc(dest_actions->actions, sizeof(action_t) * count);
121 memcpy(dest_actions->actions, source_actions->actions,
122 sizeof(action_t) * count);
123 } else {
124 /* Create new record identical to the old one */
125 dest_actions = (action_list_t*) malloc(sizeof(action_map_t));
126 dest_actions->count = count;
127 dest_actions->actions = (void(**)()) malloc(sizeof(action_t) * count);
128 memcpy(dest_actions->actions, source_actions->actions,
129 sizeof(action_t) * count);
130 /* Add it to the list */
131 hashmap_insert(to->actions, keyword_to, dest_actions);
132 }
133#pragma GCC diagnostic pop
134}
