56/* tan(x + xx) approximation valid on the interval [-pi/4,pi/4].
57 If recip is true return -1/tan(x + xx) instead. */
58static inline double
59tan_piby4(double x, double xx, int recip)
60{
61 double r, t1, t2, xl;
62 int transform = 0;
63 static const double piby4_lead =
64 7.85398163397448278999e-01, /* 0x3fe921fb54442d18 */
65 piby4_tail = 3.06161699786838240164e-17; /* 0x3c81a62633145c06 */
66
67 /* In order to maintain relative precision transform using the identity:
68 tan(pi/4-x) = (1-tan(x))/(1+tan(x)) for arguments close to pi/4.
69 Similarly use tan(x-pi/4) = (tan(x)-1)/(tan(x)+1) close to -pi/4. */
70
71 if (x > 0.68) {
72 transform = 1;
73 x = piby4_lead - x;
74 xl = piby4_tail - xx;
75 x += xl;
76 xx = 0.0;
77 } else if (x < -0.68) {
78 transform = -1;
79 x = piby4_lead + x;
80 xl = piby4_tail + xx;
81 x += xl;
82 xx = 0.0;
83 }
84
85 /* Core Remez [2,3] approximation to tan(x+xx) on the
86 interval [0,0.68]. */
87
88 r = x * x + 2.0 * x * xx;
89 t1 = x;
90 t2 = xx +
91 x * r * (0.372379159759792203640806338901e0 +
92 (-0.229345080057565662883358588111e-1 +
93 0.224044448537022097264602535574e-3 * r) *
94 r) /
95 (0.111713747927937668539901657944e1 +
96 (-0.515658515729031149329237816945e0 +
97 (0.260656620398645407524064091208e-1 -
98 0.232371494088563558304549252913e-3 * r) *
99 r) *
100 r);
101
102 /* Reconstruct tan(x) in the transformed case. */
103
104 if (transform) {
105 double t;
106 t = t1 + t2;
107 if (recip)
108 return transform * (2 * t / (t - 1) - 1.0);
109 else
110 return transform * (1.0 - 2 * t / (1 + t));
111 }
112
113 if (recip) {
114 /* Compute -1.0/(t1 + t2) accurately */
115 double trec, trec_top, z1, z2, t;
116 __UINT8_T u;
117 t = t1 + t2;
118 GET_BITS_DP64(t, u);
119 u &= 0xffffffff00000000;
120 PUT_BITS_DP64(u, z1);
121 z2 = t2 - (z1 - t1);
122 trec = -1.0 / t;
123 GET_BITS_DP64(trec, u);
124 u &= 0xffffffff00000000;
125 PUT_BITS_DP64(u, trec_top);
126 return trec_top + trec * ((1.0 + trec_top * z1) + trec_top * z2);
127
128 } else
129 return t1 + t2;
130}

132double FN_PROTOTYPE(mth_i_dtan)(double x)
133{
134 double r, rr;
135 int region, xneg;
136
137 __UINT8_T ux, ax;
138 GET_BITS_DP64(x, ux);
139 ax = (ux & ~SIGNBIT_DP64);
140 if (ax <= 0x3fe921fb54442d18) /* abs(x) <= pi/4 */
141 {
142 if (ax < 0x3f20000000000000) /* abs(x) < 2.0^(-13) */
143 {
144 if (ax < 0x3e40000000000000) /* abs(x) < 2.0^(-27) */
145 {
146 if (ax == 0x0000000000000000)
147 return x;
148 else
149 return val_with_flags(x, AMD_F_INEXACT);
150 } else {
151 return x + x * x * x * 0.333333333333333333;
152 }
153 } else
154 return tan_piby4(x, 0.0, 0);
155 } else if ((ux & EXPBITS_DP64) == EXPBITS_DP64) {
156 /* x is either NaN or infinity */
157 if (ux & MANTBITS_DP64)
158 /* x is NaN */
159 return x + x; /* Raise invalid if it is a signalling NaN */
160 else
161 /* x is infinity. Return a NaN */
162 return nan_with_flags(AMD_F_INVALID);
163 }
164 xneg = (ax != ux);
165
166 if (xneg)
167 x = -x;
168
169 if (x < 5.0e5) {
170 /* For these size arguments we can just carefully subtract the
171 appropriate multiple of pi/2, using extra precision where
172 x is close to an exact multiple of pi/2 */
173 static const double twobypi =
174 6.36619772367581382433e-01, /* 0x3fe45f306dc9c883 */
175 piby2_1 = 1.57079632673412561417e+00, /* 0x3ff921fb54400000 */
176 piby2_1tail = 6.07710050650619224932e-11, /* 0x3dd0b4611a626331 */
177 piby2_2 = 6.07710050630396597660e-11, /* 0x3dd0b4611a600000 */
178 piby2_2tail = 2.02226624879595063154e-21, /* 0x3ba3198a2e037073 */
179 piby2_3 = 2.02226624871116645580e-21, /* 0x3ba3198a2e000000 */
180 piby2_3tail = 8.47842766036889956997e-32; /* 0x397b839a252049c1 */
181 double t, rhead, rtail;
182 int npi2;
183 __UINT8_T uy, xexp, expdiff;
184 xexp = ax >> EXPSHIFTBITS_DP64;
185 /* How many pi/2 is x a multiple of? */
186 if (ax <= 0x400f6a7a2955385e) /* 5pi/4 */
187 {
188 if (ax <= 0x4002d97c7f3321d2) /* 3pi/4 */
189 npi2 = 1;
190 else
191 npi2 = 2;
192 } else if (ax <= 0x401c463abeccb2bb) /* 9pi/4 */
193 {
194 if (ax <= 0x4015fdbbe9bba775) /* 7pi/4 */
195 npi2 = 3;
196 else
197 npi2 = 4;
198 } else
199 npi2 = (int)(x * twobypi + 0.5);
200 /* Subtract the multiple from x to get an extra-precision remainder */
201 rhead = x - npi2 * piby2_1;
202 rtail = npi2 * piby2_1tail;
203 GET_BITS_DP64(rhead, uy);
204 expdiff = xexp - ((uy & EXPBITS_DP64) >> EXPSHIFTBITS_DP64);
205 if (expdiff > 15) {
206 /* The remainder is pretty small compared with x, which
207 implies that x is a near multiple of pi/2
208 (x matches the multiple to at least 15 bits) */
209 t = rhead;
210 rtail = npi2 * piby2_2;
211 rhead = t - rtail;
212 rtail = npi2 * piby2_2tail - ((t - rhead) - rtail);
213 if (expdiff > 48) {
214 /* x matches a pi/2 multiple to at least 48 bits */
215 t = rhead;
216 rtail = npi2 * piby2_3;
217 rhead = t - rtail;
218 rtail = npi2 * piby2_3tail - ((t - rhead) - rtail);
219 }
220 }
221 r = rhead - rtail;
222 rr = (rhead - r) - rtail;
223 region = npi2 & 3;
224 } else {
225 /* Reduce x into range [-pi/4,pi/4] */
226 __remainder_piby2_inline(x, &r, &rr, &region);
227 }
228
229 if (xneg)
230 return -tan_piby4(r, rr, region & 1);
231 else
232 return tan_piby4(r, rr, region & 1);
233}
