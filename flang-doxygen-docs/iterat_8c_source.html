24/* This routine is to check array subs whether it is array element.
25 * That can be confirmed if forall index does not appear any subscripts
26 * of array subs.
27 */
28int
29is_array_element_in_forall(int subs, int std)
30{
31 int ast;
32 int list;
33 int ndim;
34 int i;
35 int j;
36 int asd;
37 int sub_expr;
38
39 ast = STD_AST(std);
40 list = A_LISTG(ast);
41 asd = A_ASDG(subs);
42 ndim = ASD_NDIM(asd);
43 for (i = 0; i < ndim; i++)
44 for (j = list; j != 0; j = ASTLI_NEXT(j)) {
45 sub_expr = ASD_SUBS(asd, i);
46 if (is_name_in_expr(sub_expr, ASTLI_SPTR(j)))
47 return 0;
48 }
49 return 1;
50}

56LOGICAL
57is_name_in_expr(int expr, int isptr)
58{
59 int i, n, argt, asd, sptr;
60
61 if (expr == 0)
62 return FALSE;
63 switch (A_TYPEG(expr)) {
64 case A_ID:
65 sptr = A_SPTRG(expr);
66 if (isptr == sptr)
67 return TRUE;
68 if (is_equivalence(isptr, sptr))
69 return TRUE;
70 break;
71 case A_BINOP:
72 if (is_name_in_expr(A_LOPG(expr), isptr))
73 return TRUE;
74 if (is_name_in_expr(A_ROPG(expr), isptr))
75 return TRUE;
76 break;
77 case A_CONV:
78 case A_UNOP:
79 case A_PAREN:
80 if (is_name_in_expr(A_LOPG(expr), isptr))
81 return TRUE;
82 break;
83 case A_LABEL:
84 case A_CMPLXC:
85 case A_CNST:
86 break;
87 case A_MEM:
88 if (is_name_in_expr(A_PARENTG(expr), isptr))
89 return TRUE;
90 break;
91 case A_SUBSTR:
92 if (is_name_in_expr(A_LOPG(expr), isptr))
93 return TRUE;
94 if (is_name_in_expr(A_LEFTG(expr), isptr))
95 return TRUE;
96 if (is_name_in_expr(A_RIGHTG(expr), isptr))
97 return TRUE;
98 break;
99 case A_ICALL:
100 case A_INTR:
101 case A_FUNC:
102 argt = A_ARGSG(expr);
103 n = A_ARGCNTG(expr);
104 for (i = 0; i < n; ++i) {
105 if (is_name_in_expr(ARGT_ARG(argt, i), isptr))
106 return TRUE;
107 }
108 break;
109
110 case A_TRIPLE:
111 if (is_name_in_expr(A_LBDG(expr), isptr))
112 return TRUE;
113 if (is_name_in_expr(A_UPBDG(expr), isptr))
114 return TRUE;
115 if (is_name_in_expr(A_STRIDEG(expr), isptr))
116 return TRUE;
117 break;
118
119 case A_SUBSCR:
120 asd = A_ASDG(expr);
121 n = ASD_NDIM(asd);
122 for (i = 0; i < n; ++i) {
123 if (is_name_in_expr(ASD_SUBS(asd, i), isptr))
124 return TRUE;
125 }
126 if (is_name_in_expr(A_LOPG(expr), isptr))
127 return TRUE;
128 break;
129 default:
130 interr("is_name_in_expr: unknown ast type", expr, 3);
131 break;
132 }
133 return FALSE;
134} /* is_name_in_expr */

136/* This function return TRUE if it find isptr is in expression
137 * However, It does not count isptr if it appears as subscript.
138 * For example, expr=i.eq.a(i), isptr=i return TRUE;
139 * expr=0.eq.a(i), isptr=i return FALSE;
140 * expr=j.eq.i, isptr=i return TRUE;
141 */
142LOGICAL
143is_lonely_idx(int expr, int isptr)
144{
145 int i;
146 int n;
147 int find1;
148 int find2;
149 int argt;
150
151 if (expr == 0)
152 return FALSE;
153 switch (A_TYPEG(expr)) {
154 case A_ID:
155 if (isptr == A_SPTRG(expr))
156 return TRUE;
157 else
158 return FALSE;
159 case A_BINOP:
160 find1 = is_lonely_idx(A_LOPG(expr), isptr);
161 find2 = is_lonely_idx(A_ROPG(expr), isptr);
162 if (find1 || find2)
163 return TRUE;
164 else
165 return FALSE;
166 case A_CONV:
167 case A_UNOP:
168 case A_PAREN:
169 find1 = is_lonely_idx(A_LOPG(expr), isptr);
170 return find1;
171 case A_LABEL:
172 case A_CMPLXC:
173 case A_CNST:
174 return FALSE;
175 case A_MEM:
176 find1 = is_lonely_idx((int)A_PARENTG(expr), isptr);
177 return find1;
178 case A_SUBSTR:
179 if (is_lonely_idx((int)A_LOPG(expr), isptr))
180 return TRUE;
181 if (is_lonely_idx((int)A_LEFTG(expr), isptr))
182 return TRUE;
183 if (is_lonely_idx((int)A_RIGHTG(expr), isptr))
184 return TRUE;
185 return FALSE;
186 case A_ICALL:
187 case A_INTR:
188 case A_FUNC:
189 argt = A_ARGSG(expr);
190 n = A_ARGCNTG(expr);
191 for (i = 0; i < n; ++i) {
192 if (is_lonely_idx(ARGT_ARG(argt, i), isptr))
193 return TRUE;
194 }
195 return FALSE;
196
197 case A_TRIPLE:
198 if (is_lonely_idx(A_LBDG(expr), isptr))
199 return TRUE;
200 if (is_lonely_idx(A_UPBDG(expr), isptr))
201 return TRUE;
202 if (is_lonely_idx(A_STRIDEG(expr), isptr))
203 return TRUE;
204 return FALSE;
205
206 case A_SUBSCR:
207 return FALSE;
208 default:
209 interr("is_lonely_idx: unknown type", expr, 3);
210 return FALSE;
211 }
212}

223/*
224 * Given an AST tree (such as FORALL LHS), save the ID and member names on the
225 * LHS
226 * these will be used to determine if any RHS or subscript expression
227 * might be changed by this forall assignment
228 * forall(i=1:10) A(i)%MEM(10) = B(i)
229 * this will save A%MEM, so we can check whether B conflicts with A%MEM
230 */
231static void
232build_dlist(int depends, int forall_list, int lstdx, int rstdx)
233{
234 int d;
235 dep.forall_list = forall_list;
236 dep.dstdx = lstdx;
237 dep.estdx = rstdx;
238 dep.dexpr = depends;
239 /* build list of depends names */
240 if (dep.dsize == 0) {
241 dep.dsize = 20;
242 NEW(dep.dlist, int, dep.dsize);
243 }
244 dep.davl = 0;
245 for (d = depends; d;) {
246 switch (A_TYPEG(d)) {
247 case A_ID:
248 NEED(dep.davl + 1, dep.dlist, int, dep.dsize, dep.dsize * 2);
249 dep.dlist[dep.davl] = A_SPTRG(d);
250 ++dep.davl;
251 d = 0;
252 break;
253 case A_MEM:
254 NEED(dep.davl + 1, dep.dlist, int, dep.dsize, dep.dsize * 2);
255 dep.dlist[dep.davl] = A_SPTRG(A_MEMG(d));
256 ++dep.davl;
257 d = A_PARENTG(d);
258 break;
259 case A_SUBSTR:
260 case A_SUBSCR:
261 d = A_LOPG(d);
262 break;
263 default:
264 interr("build_dlist: unexpected ast", d, 3);
265 d = 0;
266 }
267 }
268} /* build_dlist */

272/*
273 * allocate and fill a name buffer
274 */
275static char *
276build_name_buffer(int *list, int avl)
277{
278 int i, len;
279 char *buffer;
280 len = 0;
281 for (i = 1; i <= avl; ++i) {
282 len += strlen(SYMNAME(list[avl - i])) + 1;
283 }
284 len += 1;
285 buffer = (char *)malloc(len);
286 len = 0;
287 for (i = 1; i <= avl; ++i) {
288 strcpy(buffer + len, SYMNAME(list[avl - i]));
289 len += strlen(SYMNAME(list[avl - i]));
290 if (i < avl) {
291 buffer[len++] = '%';
292 }
293 buffer[len] = '\0';
294 }
295 return buffer;
296} /* build_name_buffer */

298/*
299 * see whether the AST reference at expr conflicts with the previously
300 * saved names from build_dlist
301 */
302static LOGICAL
303name_dependent_check(int expr)
304{
305 int e, d;
306 int esptr, dsptr, edtype, ddtype;
307 LOGICAL etarget, dtarget;
308 /* expr must be A_MEM or A_ID */
309 if (expr == 0)
310 return FALSE;
311 /* build list of expr names */
312 if (dep.esize == 0) {
313 dep.esize = 20;
314 NEW(dep.elist, int, dep.esize);
315 }
316 dep.eavl = 0;
317 for (e = expr; e;) {
318 switch (A_TYPEG(e)) {
319 case A_ID:
320 NEED(dep.eavl + 1, dep.elist, int, dep.esize, dep.esize * 2);
321 dep.elist[dep.eavl] = A_SPTRG(e);
322 ++dep.eavl;
323 e = 0;
324 break;
325 case A_MEM:
326 NEED(dep.eavl + 1, dep.elist, int, dep.esize, dep.esize * 2);
327 dep.elist[dep.eavl] = A_SPTRG(A_MEMG(e));
328 ++dep.eavl;
329 e = A_PARENTG(e);
330 break;
331 case A_SUBSTR:
332 case A_SUBSCR:
333 e = A_LOPG(e);
334 break;
335 case A_FUNC:
336 e = 0;
337 break;
338 default:
339 interr("name_dependent: unexpected ast", e, 3);
340 e = 0;
341 }
342 }
343
344 if (dep.eavl == 0 || dep.davl == 0) {
345 interr("name_dependent: no names", 0, 3);
346 return FALSE;
347 }
348 /* compare elist to the previously built dlist */
349 etarget = FALSE;
350 for (e = dep.eavl; e > 0; --e) {
351 esptr = dep.elist[e - 1];
352 edtype = DDTG(DTYPEG(esptr));
353 if (TARGETG(esptr))
354 etarget = TRUE;
355 dtarget = FALSE;
356 for (d = dep.davl; d > 0; --d) {
357 dsptr = dep.dlist[d - 1];
358 ddtype = DDTG(DTYPEG(dsptr));
359 if (TARGETG(dsptr))
360 dtarget = TRUE;
361 /* can 'esptr' overlap with 'dsptr'? */
362 /* yes if they are the same variable,
363 * esptr is a pointer and dsptr is a pointer or target of same type,
364 * esptr is a target and dsptr is a pointer of same type,
365 * both variables and equivalenced (handled later)
366 */
367 if (edtype == ddtype) {
368 /* ### for F2003, allow extended datatypes as well */
369 int overlap;
370 overlap = 0;
371 if (esptr == dsptr && e == dep.eavl && d == dep.davl) {
372 overlap = 1;
373 } else if (POINTERG(esptr) && (POINTERG(dsptr) || dtarget)) {
374 overlap = 2;
375 if (flg.opt >= 2 && XBIT(53, 2)) {
376 /* see if we have any PTA information about
377 * the RHS in elist with respect to the LHS in dlist */
378 if (!pta_conflict(dep.estdx, dep.elist[dep.eavl - 1], dep.dstdx,
379 dep.dlist[dep.davl - 1], POINTERG(dsptr),
380 dtarget)) {
381 overlap = 0;
382 }
383 }
384 if (!flg.depchk)
385 overlap = 0;
386 } else if (etarget && POINTERG(dsptr)) {
387 overlap = 2;
388 if (flg.opt >= 2 && XBIT(53, 2)) {
389 if (!pta_conflict(dep.dstdx, dep.dlist[dep.davl - 1], dep.estdx,
390 dep.elist[dep.eavl - 1], POINTERG(esptr),
391 etarget)) {
392 overlap = 0;
393 }
394 }
395 if (!flg.depchk)
396 overlap = 0;
397 }
398 if (DTY(ddtype) == TY_UNION) {
399 /* assume overlap */
400 } else {
401 int ee, dd;
402 /* compare the rest of the list */
403 for (dd = d - 1, ee = e - 1; overlap > 0 && dd > 0 && ee > 0;
404 --dd, --ee) {
405 /* AAA%b%c%d%rest
406 * BBB%b%c%d%more
407 * compare 'rest' and 'more' */
408 int ddsptr, eesptr;
409 ddsptr = dep.dlist[dd - 1];
410 eesptr = dep.elist[ee - 1];
411 if (ddsptr != eesptr) {
412 overlap = 0;
413 } else {
414 int dddtype;
415 /* same member; if this is a union, stop here */
416 dddtype = DDTG(DTYPEG(ddsptr));
417 if (DTY(dddtype) == TY_UNION)
418 break;
419 }
420 }
421 if (overlap == 1 && dep.forall_list) {
422 /* allow (=,=,=) dependence */
423 if (dep.dexpr == expr) {
424 overlap = 0;
425 /* check subscripts, if any */
426 } else if (!dd_array_conflict(A_LISTG(dep.forall_list), dep.dexpr,
427 expr, -1)) {
428 overlap = 0;
429 } else {
430 /* flg.depchk MORE WORK HERE */
431 }
432 }
433 if (overlap > 0) {
434 int i, needmsg;
435 if (dep.eavl != dep.davl) {
436 needmsg = 1;
437 } else {
438 needmsg = 0;
439 for (i = 0; i < dep.davl; ++i) {
440 if (dep.dlist[i] != dep.elist[i]) {
441 needmsg = 1;
442 break;
443 }
444 }
445 }
446 if (needmsg) {
447 char *dname, *ename;
448 dname = build_name_buffer(dep.dlist, dep.davl);
449 ename = build_name_buffer(dep.elist, dep.eavl);
450 ccff_info(MSGFTN, "FTN019", 1, STD_LINENO(dep.dstdx),
451 "Conflict or overlap between %var1 and %var2",
452 "var1=%s", dname, "var2=%s", ename, NULL);
453 free(ename);
454 free(dname);
455 }
456 return TRUE;
457 }
458 }
459 }
460 }
461 }
462 esptr = dep.elist[dep.eavl - 1];
463 dsptr = dep.dlist[dep.davl - 1];
464 if (is_equivalence(dsptr, esptr))
465 return TRUE;
466
467 /* now check any subscripts */
468 if (subscr_dependent_check(expr))
469 return TRUE;
470 return FALSE;
471 /* do we need a special case for ST_ARRDSC? */
472} /* name_dependent_check */

474/*
475 * recursively visit the tree expr to see whether any name in the tree
476 * conflicts with the previously saved names from build_dlist
477 */
478static LOGICAL
479expr_dependent_check(int expr)
480{
481 int i, n, argt;
482
483 if (expr == 0)
484 return FALSE;
485 switch (A_TYPEG(expr)) {
486 case A_ID:
487 case A_MEM:
488 case A_SUBSCR:
489 if (name_dependent_check(expr))
490 return TRUE;
491 break;
492 case A_BINOP:
493 if (expr_dependent_check(A_LOPG(expr)))
494 return TRUE;
495 if (expr_dependent_check(A_ROPG(expr)))
496 return TRUE;
497 break;
498 case A_CONV:
499 case A_UNOP:
500 case A_PAREN:
501 if (expr_dependent_check(A_LOPG(expr)))
502 return TRUE;
503 break;
504 case A_LABEL:
505 case A_CMPLXC:
506 case A_CNST:
507 break;
508 case A_SUBSTR:
509 if (expr_dependent_check(A_LOPG(expr)))
510 return TRUE;
511 if (expr_dependent_check(A_LEFTG(expr)))
512 return TRUE;
513 if (expr_dependent_check(A_RIGHTG(expr)))
514 return TRUE;
515 break;
516 case A_ICALL:
517 case A_INTR:
518 case A_FUNC:
519 argt = A_ARGSG(expr);
520 n = A_ARGCNTG(expr);
521 for (i = 0; i < n; ++i) {
522 if (expr_dependent_check(ARGT_ARG(argt, i)))
523 return TRUE;
524 }
525 break;
526
527 case A_TRIPLE:
528 if (expr_dependent_check(A_LBDG(expr)))
529 return TRUE;
530 if (expr_dependent_check(A_UPBDG(expr)))
531 return TRUE;
532 if (expr_dependent_check(A_STRIDEG(expr)))
533 return TRUE;
534 break;
535 case A_MP_ATOMICREAD:
536 if (expr_dependent_check(A_SRCG(expr)))
537 return TRUE;
538 break;
539
540 default:
541 interr("expr_dependent_check: unknown type", expr, 3);
542 break;
543 }
544 return FALSE;
545} /* expr_dependent_check */

547/*
548 * recursively visit the tree expr to see whether any name in the tree
549 * conflicts with the previously saved names from build_dlist
550 * Like expr_dependent_check, but only really checks expressions in
551 * subscript or substring selectors.
552 */
553static LOGICAL
554subscr_dependent_check(int expr)
555{
556 int i, n, argt, asd;
557
558 if (expr == 0)
559 return FALSE;
560 switch (A_TYPEG(expr)) {
561 case A_ID:
562 break;
563 case A_MEM:
564 if (subscr_dependent_check(A_PARENTG(expr)))
565 return TRUE;
566 break;
567 case A_BINOP:
568 if (subscr_dependent_check(A_LOPG(expr)))
569 return TRUE;
570 if (subscr_dependent_check(A_ROPG(expr)))
571 return TRUE;
572 break;
573 case A_CONV:
574 case A_UNOP:
575 case A_PAREN:
576 if (subscr_dependent_check(A_LOPG(expr)))
577 return TRUE;
578 break;
579 case A_LABEL:
580 case A_CMPLXC:
581 case A_CNST:
582 break;
583 case A_SUBSTR:
584 if (subscr_dependent_check(A_LOPG(expr)))
585 return TRUE;
586 if (expr_dependent_check(A_LEFTG(expr)))
587 return TRUE;
588 if (expr_dependent_check(A_RIGHTG(expr)))
589 return TRUE;
590 break;
591 case A_ICALL:
592 case A_INTR:
593 case A_FUNC:
594 argt = A_ARGSG(expr);
595 n = A_ARGCNTG(expr);
596 for (i = 0; i < n; ++i) {
597 if (subscr_dependent_check(ARGT_ARG(argt, i)))
598 return TRUE;
599 }
600 break;
601
602 case A_TRIPLE:
603 if (subscr_dependent_check(A_LBDG(expr)))
604 return TRUE;
605 if (subscr_dependent_check(A_UPBDG(expr)))
606 return TRUE;
607 if (subscr_dependent_check(A_STRIDEG(expr)))
608 return TRUE;
609 break;
610
611 case A_SUBSCR:
612 asd = A_ASDG(expr);
613 n = ASD_NDIM(asd);
614 for (i = 0; i < n; ++i) {
615 if (expr_dependent_check(ASD_SUBS(asd, i)))
616 return TRUE;
617 }
618 break;
619 case A_MP_ATOMICREAD:
620 if (subscr_dependent_check(A_SRCG(expr)))
621 return TRUE;
622 break;
623 default:
624 interr("subscr_dependent_check: unknown type", expr, 3);
625 break;
626 }
627 return FALSE;
628} /* subscr_dependent_check */

653/* find isptr symbol in sub_expr and replace with expr
654 if indirection=0, it does not change array subscripts
655*/
656
657int
658replace_expr(int sub_expr, int isptr, int expr, int indirection)
659{
660 int i;
661 int asd;
662 int ndim;
663 int expr1;
664 int expr2;
665 int sptr;
666 int subs[7];
667 int nargs, argt;
668
669 if (sub_expr == 0)
670 return sub_expr;
671 switch (A_TYPEG(sub_expr)) {
672 case A_ID:
673 if (isptr == A_SPTRG(sub_expr))
674 return expr;
675 else
676 return sub_expr;
677 case A_BINOP:
678 expr1 = replace_expr(A_LOPG(sub_expr), isptr, expr, indirection);
679 expr2 = replace_expr(A_ROPG(sub_expr), isptr, expr, indirection);
680 return mk_binop(A_OPTYPEG(sub_expr), expr1, expr2, A_DTYPEG(sub_expr));
681 case A_UNOP:
682 expr1 = replace_expr(A_LOPG(sub_expr), isptr, expr, indirection);
683 return mk_unop(A_OPTYPEG(sub_expr), expr1, A_DTYPEG(sub_expr));
684 case A_CONV:
685 expr1 = replace_expr(A_LOPG(sub_expr), isptr, expr, indirection);
686 return mk_convert(expr1, A_DTYPEG(sub_expr));
687 case A_CMPLXC:
688 case A_CNST:
689 return sub_expr;
690 case A_PAREN:
691 expr1 = replace_expr(A_LOPG(sub_expr), isptr, expr, indirection);
692 return mk_paren(expr1, A_DTYPEG(sub_expr));
693 case A_SUBSCR:
694 if (indirection) {
695 sptr = sptr_of_subscript(sub_expr);
696 /*
697 assert(!ALIGNG(sptr),
698 "replace_expr: indirection should not distri", sptr, 999);
699 */
700 asd = A_ASDG(sub_expr);
701 ndim = ASD_NDIM(asd);
702 for (i = 0; i < ndim; i++) {
703 expr1 = replace_expr(ASD_SUBS(asd, i), isptr, expr, indirection);
704 subs[i] = expr1;
705 }
706 sub_expr = mk_subscr(A_LOPG(sub_expr), subs, ndim, A_DTYPEG(sub_expr));
707 }
708 return sub_expr;
709 case A_FUNC:
710 case A_INTR:
711 argt = A_ARGSG(sub_expr);
712 nargs = A_ARGCNTG(sub_expr);
713 for (i = 0; i < nargs; ++i)
714 if (ARGT_ARG(argt, i))
715 ARGT_ARG(argt, i) =
716 replace_expr(ARGT_ARG(argt, i), isptr, expr, indirection);
717 A_ARGSP(sub_expr, argt);
718 return sub_expr;
719 default:
720 return sub_expr;
721 }
722}

724LOGICAL
725is_equivalence(int sptr1, int sptr2)
726{
727 int stype1;
728 int stype2;
729 int p;
730
731 stype1 = STYPEG(sptr1);
732 stype2 = STYPEG(sptr2);
733 if (stype1 != ST_ARRAY && stype1 != ST_VAR)
734 return FALSE;
735 if (stype2 != ST_ARRAY && stype2 != ST_VAR)
736 return FALSE;
737
738 for (p = SOCPTRG(sptr1); p; p = SOC_NEXT(p)) {
739 if (SOC_SPTR(p) == sptr2)
740 return TRUE;
