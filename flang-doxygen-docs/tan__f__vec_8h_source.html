24{

26 vint2 ia_a, ia_b, p_a, p_b, hi_a, hi_b;

29 vmask i2opi_vec[] = {
30 vcast_vm_i_i(0, i2opi_f[0]),
31 vcast_vm_i_i(0, i2opi_f[1]),
32 vcast_vm_i_i(0, i2opi_f[2]),
33 vcast_vm_i_i(0, i2opi_f[3]),
34 vcast_vm_i_i(0, i2opi_f[4]),
35 vcast_vm_i_i(0, i2opi_f[5]),
36 };

39 s = vand_vi2_vi2_vi2(ia, vcast_vi2_i(0x80000000));
40 /* e = ((ia >> 23) & 0xff) - 127; */
41 e = vsrl_vi2_vi2_i(ia, 23);
42 e = vand_vi2_vi2_vi2(e, vcast_vi2_i(0xff));
43 e = vsub_vi2_vi2_vi2(e, vcast_vi2_i(127));
44 /* ia = (ia << 8) | 0x80000000; */
45 ia = vsll_vi2_vi2_i(ia, 8);
46 ia = vor_vi2_vi2_vi2(ia, vcast_vi2_i(0x80000000));
47
48 /* compute x * 1/pi */
49 /* idx = 6 - ((e >> 5) & 3); */
50 idx = vsrl_vi2_vi2_i(e, 5);
51 idx = vand_vi2_vi2_vi2(idx, vcast_vi2_i(3));
52 idx = vsub_vi2_vi2_vi2(vcast_vi2_i(6), idx);
53
54 ia_a = vsrl64_vi2_vi2_i(ia, 32);
55 ia_b = ia;
56 hi_a = vcast_vi2_i(0);
57 hi_b = vcast_vi2_i(0);
58
59 q = vcast_vi2_i(0);
60 for (int i = 0; i < 6; i++) {
61 p_a = vmulu_vi2_vi2_vi2((vint2)i2opi_vec[i], ia_a);
62 p_b = vmulu_vi2_vi2_vi2((vint2)i2opi_vec[i], ia_b);
63 p_a = vadd64_vi2_vi2_vi2(p_a, hi_a);
64 p_b = vadd64_vi2_vi2_vi2(p_b, hi_b);
65

67 hi_b = vsrl64_vi2_vi2_i(p_b, 32);
68
69 p_a = vsll64_vi2_vi2_i(p_a, 32);
70 p_b = vand_vi2_vi2_vi2(p_b, vcast_vm_i_i(0, 0xffffffff));
71
72 p = vor_vi2_vi2_vi2(p_a, p_b);
73
74 vopmask m = veq_vo_vi2_vi2(idx, q);
75 hi = vsel_vi2_vo_vi2_vi2(m, p, hi);
76 lo = vsel_vi2_vo_vi2_vi2(m, prev, lo);
77 ll = vsel_vi2_vo_vi2_vi2(m, prev2, ll);
78
79 prev2 = prev;
80 prev = p;
81
82 q = vadd_vi2_vi2_vi2(q, vcast_vi2_i(1));
83 }
84 p = vor_vi2_vi2_vi2(vsll64_vi2_vi2_i(hi_a, 32), hi_b);
85
86 vopmask m = veq_vo_vi2_vi2(idx, q);
87 hi = vsel_vi2_vo_vi2_vi2(m, p, hi);
88 lo = vsel_vi2_vo_vi2_vi2(m, prev, lo);
89 ll = vsel_vi2_vo_vi2_vi2(m, prev2, ll);
90
91 e = vand_vi2_vi2_vi2(e, vcast_vi2_i(31));
92
93 union {
94 vint2 v;
95 uint32_t t[sizeof(vint2) / sizeof(uint32_t)];
96 } ue, uhi, ulo, ull, uh, ur;
97 ue.v = e; uhi.v = hi; ulo.v = lo; ull.v = ll;
98 for (unsigned i = 0; i < sizeof(vint2) / sizeof(uint32_t); i++) {
99 uint32_t e = ue.t[i], q;
100 uint64_t p;
101 p = (uint64_t)uhi.t[i] << 32;
102 p |= ulo.t[i];
103
104 if (e) {
105 q = 32 - e;
106 p = (p << e) | (ull.t[i] >> q);
107 }
108
109 q = (uhi.t[i] << e) & 0x80000000;
110 p &= 0x7fffffffffffffffULL;
111
112 if (p & 0x4000000000000000ULL) {
113 p |= 0x8000000000000000ULL;
114 q ^= 0x80000000;
115 }
116 uh.t[i] = q;
117
118 double d = (double)(int64_t)p;
119 d *= PI_2_M64;
120 float r = (float)d;
121 ur.t[i] = float_as_int(r);
122 }
123 vstore_v_p_vf((float*)h, (vfloat)uh.v);
124 return (vfloat)vxor_vi2_vi2_vi2(ur.v, s);
125}
