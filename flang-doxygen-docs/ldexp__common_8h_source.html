12static float INLINE __ldexpf_scalar_kernel(float a, int scale)
13{
14#if (defined __AVX512F__)
15 float res = _mm_cvtss_f32(_mm_scalef_ss(_mm_set_ss(a), _mm_set_ss((float)scale)));
16 return res;
17#else
18 PRINT(a); PRINT(scale);
19 // Input is allowed to be such that signed |scale| < 256,
20 // |a| may be in {+-0} or +-[2^-149, 2^0] as it comes from sin/cos,
21 // but we took precaution outside this routine and normalized a,
22 // so that it is within +-[2^-149 + 64, 2^64] or zero.
23
24 // Zeros and Inf/NaNs are handled separately.
25 // Input denormals end up here too and yield incorrect result.
26 // FIXME: assert(this function assumes no denormals on input !!!);
27 unsigned exp_bits = F2I(a) & FL_EXP_MASK;
28 unsigned zeroinfnan_mask = ((exp_bits == FL_EXP_MASK) || (exp_bits == 0))
29 ? 0xffffffff : 0; PRINT(zeroinfnan_mask);
30 // Preserve sign of input, quiet NaN
31 float zeroinfnan_res = a + a; PRINT(zeroinfnan_res);
32
33 // biased exponent bits, shifted to least significant position
34 unsigned getexp_a = exp_bits >> (FL_PREC_BITS-1); PRINT(getexp_a);
35
36 // For a * 2^scale to fit in floats we need getexp(a) + scale
37 // to fit in exponents range of floats: bias + (FL_EXP_MIN-1, FL_EXP_MAX).
38 // FL_EXP_MIN-1 is less than the smallest denormal, but it may round up.
39 int sumexp = getexp_a + scale; PRINT(sumexp);
40
41 // Return Inf of correct sign if overflow
42 unsigned ovf_mask = ((sumexp > (signed int)(FL_EXP_MAX + FL_EXP_BIAS))
43 ? 0xffffffff : 0);
44 unsigned sign_a = F2I(a) & FL_SIGN_BIT; PRINT(sign_a);
45 unsigned ovf_res = (sign_a | FL_EXP_MASK); PRINT(ovf_res);
46
47 // If underflow, return zero of correct sign
48 unsigned udf_mask = (sumexp < (signed int)(FL_EXP_MIN-1 + FL_EXP_BIAS) )
49 ? 0xffffffff : 0;
50 unsigned udf_res = sign_a; PRINT(udf_res);
51
52 // Check if result is within denormalized numbers range
53 // and doesn't completely underflow
54 unsigned den_mask = ~udf_mask &
55 (((signed int)(sumexp) <= 0) ? 0xffffffff : 0);
56
57 // If scaling leads to denormals: we shall do it via FP multiplication
58 // 2^scale * a. But 2^scale alone may not be representable in FP, while
59 // the product is OK. Thus we would like the sum of exponents sumexp in
60 // range for FP. Since sumexp already contains the value of biased exponent
61 // of a, we will first compensate a by reducing its exponent to biased zero:
62 // a = a * 2^(-(getexp_a - bias)), or set exponent bits of a to FL_EXP_BIAS.
63 // Now we would like sumexp become positive, for that we may add as little
64 // as -(FL_EXP_MIN-2 + FL_EXP_BIAS). We'd have to compensate exponent of a
65 // by this same quantity, so in the end we'll be setting exponent of a to
66 // FL_EXP_BIAS + (FL_EXP_MIN-2 + FL_EXP_BIAS) = 2*FL_EXP_BIAS + FL_EXP_MIN-2
67 int new_scale = ((unsigned int)(sumexp -(FL_EXP_MIN-2 + FL_EXP_BIAS)))
68 << (FL_PREC_BITS-1); PRINT(new_scale);
69 float new_a = I2F((F2I(a) & (~FL_EXP_MASK)) |
70 ((2*FL_EXP_BIAS + FL_EXP_MIN-2) << (FL_PREC_BITS-1))); PRINT(new_a);
71 float den_res = new_a * I2F(new_scale); PRINT(den_res);
72
73 // normal case, just add scale to exponent bits
74 unsigned gen_res = F2I(a) + (((unsigned int)scale) << (FL_PREC_BITS-1)); PRINT(gen_res);
75 unsigned gen_mask = ~(ovf_mask | udf_mask | den_mask);
76
77 float result = I2F((F2I(zeroinfnan_res) & zeroinfnan_mask) |
78 ((~zeroinfnan_mask) & ((ovf_res & ovf_mask) |
79 (udf_res & udf_mask) |
80 (F2I(den_res) & den_mask) |
81 (gen_res & gen_mask)))); PRINT(result);
82
83 return result;
84#endif //#if (defined __AVX512F__)
85}

87static vfloat INLINE
88//static vfloat __attribute__((noinline))
89__vldexpf_manual(vfloat va, vfloat vscale)
90{
91 PRINT(va); PRINT(vscale);
92 // Input is allowed to be such that signed |scale| < 256,
93 // |a| may be in {+-0} or +-[2^-149, 2^0] as it comes from sin/cos,
94 // but we took precaution outside this routine and normalized a,
95 // so that it is within +-[2^-149 + 64, 2^64] or zero.
96
97 // Zeros and Inf/NaNs are handled separately.
98 // Input denormals end up here too and yield incorrect result.
99 // FIXME: assert(this function assumes no denormals on input !!!);
100 vint2 exp_bits = vand_vi2_vi2_vi2(vF2I(va), vSETi(FL_EXP_MASK));
101 vopmask zero_mask = veq_vo_vi2_vi2(exp_bits, vSETi(0));
102 vopmask infnan_mask = veq_vo_vi2_vi2(exp_bits, vSETi(FL_EXP_MASK));
103 vopmask zeroinfnan_mask = vor_vo_vo_vo(zero_mask, infnan_mask); PRINT(zeroinfnan_mask);
104
105 // Preserve sign of input, quiet NaN
106 vfloat zeroinfnan_res = vadd_vf_vf_vf(va, va); PRINT(zeroinfnan_res);
107
108 // biased exponent bits, shifted to least significant position
109 vint2 getexp_a = vsrl_vi2_vi2_i(exp_bits, FL_PREC_BITS-1); PRINT(getexp_a);
110
111 // For a * 2^scale to fit in floats we need getexp(a) + scale
112 // to fit in exponents range of floats: bias + (FL_EXP_MIN-1, FL_EXP_MAX).
113 // FL_EXP_MIN-1 is less than the smallest denormal, but it may round up.
114 vint2 sumexp = vadd_vi2_vi2_vi2(getexp_a, vF2I(vscale)); PRINT(sumexp);
115
116 // Return Inf of correct sign if overflow
117 vopmask ovf_mask = vgt_vo_vi2_vi2(sumexp, vSETi(FL_EXP_MAX + FL_EXP_BIAS));
118 vint2 sign_a = vand_vi2_vi2_vi2(vF2I(va), vSETi(FL_SIGN_BIT)); PRINT(sign_a);
119 vint2 ovf_res = vor_vi2_vi2_vi2(sign_a, vSETi(FL_EXP_MASK)); PRINT(ovf_res);
120
121 // If underflow, return zero of correct sign
122 vopmask udf_mask = vgt_vo_vi2_vi2(vSETi(FL_EXP_MIN-1 + FL_EXP_BIAS), sumexp);
123 vint2 udf_res = sign_a; PRINT(udf_res);
124
125 // Check if result is within denormalized numbers range
126 // and doesn't completely underflow
127 vopmask den_mask = vandnot_vo_vo_vo(udf_mask, vgt_vo_vi2_vi2(vSETi(1), sumexp));
128
129 // If scaling leads to denormals: we shall do it via FP multiplication
130 // 2^scale * a. But 2^scale alone may not be representable in FP, while
131 // the product is OK. Thus we would like the sum of exponents sumexp in
132 // range for FP. Since sumexp already contains the value of biased exponent
133 // of a, we will first compensate a by reducing its exponent to biased zero:
134 // a = a * 2^(-(getexp_a - bias)), or set exponent bits of a to FL_EXP_BIAS.
135 // Now we would like sumexp become positive, for that we may add as little
136 // as -(FL_EXP_MIN-2 + FL_EXP_BIAS). We'd have to compensate exponent of a
137 // by this same quantity, so in the end we'll be setting exponent of a to
138 // FL_EXP_BIAS + (FL_EXP_MIN-2 + FL_EXP_BIAS) = 2*FL_EXP_BIAS + FL_EXP_MIN-2
139 vint2 new_scale =
140 vsll_vi2_vi2_i(
141 vadd_vi2_vi2_vi2(sumexp, vSETi(-(FL_EXP_MIN-2 + FL_EXP_BIAS))),
142 FL_PREC_BITS-1); PRINT(new_scale);
143 vfloat new_a = vI2F(vor_vi2_vi2_vi2(
144 vand_vi2_vi2_vi2(vF2I(va), vSETi(~FL_EXP_MASK)),
145 vSETi((2*FL_EXP_BIAS + FL_EXP_MIN-2) << (FL_PREC_BITS-1)))); PRINT(new_a);
146 vfloat den_res = vmul_vf_vf_vf(new_a, vI2F(new_scale)); PRINT(den_res);
147
148 // normal case, just add scale to exponent bits
149 vint2 gen_res = vadd_vi2_vi2_vi2(vF2I(va),
150 vsll_vi2_vi2_i( vF2I(vscale), FL_PREC_BITS-1)); PRINT(gen_res);
151 vopmask ngen_mask =
152 vor_vo_vo_vo(vor_vo_vo_vo(ovf_mask, udf_mask), den_mask);
153
154 vfloat result = vI2F(
155 vor_vi2_vi2_vi2(
156 vand_vi2_vo_vi2(zeroinfnan_mask, vF2I(zeroinfnan_res)),
157 vandnot_vi2_vo_vi2(zeroinfnan_mask,
158 vor_vi2_vi2_vi2(
159 vand_vi2_vo_vi2(ovf_mask, ovf_res),
160 vor_vi2_vi2_vi2(
161 vand_vi2_vo_vi2(udf_mask, udf_res),
162 vor_vi2_vi2_vi2(
163 vand_vi2_vo_vi2(den_mask, vF2I(den_res)),
164 vandnot_vi2_vo_vi2(ngen_mask, gen_res))))))); PRINT(result);
165
166 return result;
167}

169static vfloat INLINE
170//static vfloat __attribute__((noinline))
171__vldexpf_kernel(vfloat va, vfloat vscale)
172{
173 PRINT(va); PRINT(vscale);
174#if (defined __AVX512F__) && ((defined __AVX512VL__) || (_VL == 8))
175 // use AVX512VL instruction for _VL < 8
176 // use AVX512F instruction in case of a full width
177 vfloat vfres = JOIN(__SIMD_TYPE,_scalef_ps)(va, vcast_vf_vi2(vF2I(vscale))); PRINT(vfres);
178 return vfres;
179#elif (defined __AVX512F__)
180 // AVX512VL not supported and _VL < 8
181 vfloat vscale_converted = vcast_vf_vi2(vF2I(vscale)); PRINT(vscale_converted);
182 __mmask16 mask = (__mmask16)((1 << (2*_VL)) - 1); PRINT(mask);
183 __m512 fullwidth_va = JOIN3(_mm512_castps,__SIMD_BITS,_ps512)(va); PRINT(fullwidth_va);
184 __m512 fullwidth_vscale = JOIN3(_mm512_castps,__SIMD_BITS,_ps512)(vscale_converted); PRINT(fullwidth_vscale);
185 __m512 fullwidth_vfres = _mm512_maskz_scalef_ps(mask, fullwidth_va, fullwidth_vscale); PRINT(fullwidth_vfres);
186 vfloat vfres = JOIN(_mm512_castps512_ps,__SIMD_BITS)(fullwidth_vfres); PRINT(vfres);
187 return vfres;
188#else
189 return __vldexpf_manual(va, vscale);
190#endif
191}
