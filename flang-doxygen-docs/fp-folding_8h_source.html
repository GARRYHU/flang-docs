7/** \file
8 * \brief API for floating-point folding
9 *
10 * Define an API for compile-time evaluation of floating-point
11 * operations and math library functions.
12 *
13 * The implementation use the host's floating-point arithmetic
14 * and libraries, but it could also implement these interfaces
15 * with software emulation, such as by using Berkeley SoftFloat.
16 *
17 * Note: All of these interfaces receive their actual arguments as
18 * pointers to constant floating-point types. This protocol is
19 * less convenient to use, due to its need for explicit intermediate
20 * temporary variables, but it reduces the risk of a silent
21 * numeric conversion of an argument value.
22 *
23 * Note: The types have names like "float32_t" to distinguish them
24 * from integers, while the routines use names like "real32" to
25 * distinguish them from complex operations that might be added.
26 */
27

40 * to identify the native "long double" type.

42#if DECIMAL_DIG == 21 && LDBL_MANT_DIG == 64
43# define FOLD_LDBL_X87 1 /* x87 FPU 80-bit extended precision */
44#elif DECIMAL_DIG == 33 && LDBL_MANT_DIG == 106
45# define FOLD_LDBL_DOUBLEDOUBLE 1 /* Power(PC) double-double */
46#elif DECIMAL_DIG > 33 && LDBL_MANT_DIG == 112
47# define FOLD_LDBL_128BIT 1 /* 128-bit IEEE */
48#elif DECIMAL_DIG == 36 && LDBL_MANT_DIG == 113
49# define FOLD_LDBL_128BIT 1 /* 128-bit IEEE */
50#elif DECIMAL_DIG == 17 && LDBL_MANT_DIG == 53
51# define FOLD_LDBL_JUST_DOUBLE /* long double is a synonym for IEEE double */
52#else
53# warning unrecognized long double configuration (DECIMAL_DIG, LDBL_MANT_DIG)
