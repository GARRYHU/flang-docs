8/*
9 * Math elemental function dispatch initialization.
10 *
11 * Dispatch initialization builds the runtime architecture specific dispatch
12 * jump tables used by the generic elemental function compiler interface.
13 *
14 * There are two methods used to initialize the dispatch tables:
15 * 1) C constructors - automatically call __math_dispatch() before main().
16 * The dispatch tables are undefined prior to calling __math_dispatch().
17 * This method is serialized by the linker support runtime library.
18 *
19 * 2) Construct on first elemental call.
20 * The dispatch table are fully defined (for all possible compiler
21 * generated entry points). The definitions for each entry point are of
22 * the form __[frp][sdcz]_<NAME>_<VL>_init(), example:
23 * __fs_acos_1_init().
24 * Upon entry to __fs_acos_1_init, a call is made to __math_dispatch_init().
25 * __math_dispatch_init then only allows the "master" thread to enter
26 * __math_dispatch(). Any other non-master (not-first) thread will be held
27 * spin waiting for the local initialization flag "__math_dispatch_is_init"
28 * to to indicate __math_dispatch() has completed setup.
29 *
30 * Environment variable MTH_I_DEBUG is a bitmask used to control certain aspects of
31 * dispatch initialization/shutdown.
32 *
33 * 0x1 Dump architecture specific dispatch table.
34 * 0x2 Display MTH_I_OVERRIDE overrides
35 * 0x4 Display MTH_I_OVERRIDE overrides - internal
36 * 0x100 Init-on-first-call: one second delay for "master" thread in
37 * __math_dispatch_init() before entering __math_dispatch().
38 * Also generate stderr message for any subsequent "non-master"
39 * thread waiting for __math_dispatch() to complete.
40 *
41 * All debug/waring/error messages are directed to stderr.
42 *
43 */

581 "\t<ALT_FRP> (alternative) is one of the character set \"F|R|P|S\"\n", stderr);

583
584static void
585mth_i_override()
586{
587 char *penv;
588 char *tbuf;
589 char *name;
590 char *sv;
591 char corigfrp;
592 char caltfrp;
593 int i;
594 int j;
595 sv_e s;
596 frp_e origfrp;
597 frp_e altfrp;
598 func_e f;
599 bool failure;
600
601 penv = getenv("MTH_I_OVERRIDE");
602 if (__mth_i_debug & 0x2) {
603 fprintf(stderr, "MTH_I_OVERRIDE=%s\n", penv);
604 }
605 if (penv == NULL) {
606 return;
607 }
608
609 j = strlen(penv) + 1;
610 tbuf = malloc(j);
611 for (i = 0; i < j; i++) {
612 tbuf[i] = tolower(penv[i]);
613 }
614
615 /*
616 * Real simple state machine.
617 * MTH_I_OVERRIDE=<INTRIN_NAME>+<SV>:<ORIG_FRP>=<ALT_FRP>[,...]
618 * Where:
619 * <INTRIN_NAME> is a known intrinsic name in the dispatch table.
620 * <SV> is the scalar/vector size designator/descriptor.
621 * <ORIG_FRP> (original) is one of the character set "F|R|P"
622 * <ALT_FRP> (alternative) is one of the character set "F|R|P"
623 *
624 * State machine does not ignore whitespace.
625 *
626 * Valid examples:
627 * MTH_I_OVERRIDE=acos+sd:f=p
628 * MTH_I_OVERRIDE=acos+sd:f=p,acos+vs4:r=f
629 * MTH_I_OVERRIDE=acos+sd:f=p,acos+vs4:r=f,
630 *
631 * Invalid examples:
632 * MTH_I_OVERRIDE=acos-sd:f=p
633 * MTH_I_OVERRIDE=acos-sd;f=p
634 * MTH_I_OVERRIDE=acos+sd;f-p
635 * MTH_I_OVERRIDE=acos-sd;f,
636 */
637
638 failure = true;
639 i = 0;
640 while (true) {
641 name = NULL;
642 sv = NULL;
643
644 /*
645 * <INTRIN_NAME>
646 */
647 if (isalpha(tbuf[i]) == 0) {
648 // fprintf(stderr, "tbuf[%d]=%c must start with a letter\n", i, tbuf[i]);
649 break;
650 }
651 name = &tbuf[i];
652 for (i++; isalnum(tbuf[i]) != 0; i++) {
653 }
654
655 /*
656 * Seperator "+".
657 */
658 if (tbuf[i] != '+') {
659 // fprintf(stderr, "tbuf[%d]=%c must be a '+'\n", i, tbuf[i]);
660 break;
661 }
662
663 /*
664 * <SV>
665 */
666 tbuf[i++] = '\0';
667 sv = &tbuf[i];
668 for (i++; isalnum(tbuf[i]) != 0; i++) {
669 }
670
671 /*
672 * Seperator ":".
673 */
674 if (tbuf[i] != ':') {
675 // fprintf(stderr, "tbuf[%d]=%c must be a ':'\n", i, tbuf[i]);
676 break;
677 }
678
679 /*
680 * <ORIG_FRP>
681 */
682 tbuf[i++] = '\0';
683 if (isalpha(tbuf[i])) {
684 corigfrp = tbuf[i++];
685 } else {
686 // fprintf(stderr, "tbuf[%d]=%c must be one of 'frp'\n", i, tbuf[i]);
687 break;
688 }
689
690 /*
691 * Seperator "=".
692 */
693 if (tbuf[i++] != '=') {
694 // fprintf(stderr, "tbuf[%d]=%c must be a '='\n", i, tbuf[i]);
695 break;
696 }
697
698 /*
699 * <ALT_FRP>
700 */
701 caltfrp = tbuf[i++];
702
703 /*
704 * End of buffer or seperator "," for more intrinsics.
705 */
706 if (tbuf[i] != '\0' && tbuf[i] != ',') {
707 break;
708 }
709
710 j = get_string_index(name, cfunc, sizeof cfunc / sizeof *cfunc);
711 if (j < 0) {
712 fprintf(stderr, "MTH_I_OVERRIDE: intrinsic name %s not valid\n", name);
713 break;
714 }
715 f = j;
716
717 j = get_string_index(sv, csv, sizeof csv / sizeof *csv);
718 if (j < 0) {
719 fprintf(stderr, "MTH_I_OVERRIDE: intrinsic class %s not valid\n", sv);
720 break;
721 }
722 s = j;
723
724 if (false == is_frp_char_valid(corigfrp)) {
725 fprintf(stderr,
726 "MTH_I_OVERRIDE: Original precision=%c must be one of 'frps'\n",
727 corigfrp);
728 break;
729 }
730
731 if (false == is_frp_char_valid(caltfrp)) {
732 fprintf(stderr,
733 "MTH_I_OVERRIDE: Alternative precision=%c must be one of 'frps'\n",
734 caltfrp);
735 break;
736 }
737
738 origfrp = frp2index(corigfrp);
739 altfrp = frp2index(caltfrp);
740
741 if (__mth_i_debug & (0x2 | 0x4)) {
742 if (__mth_i_debug & 0x4) {
743 fprintf(stderr, "MTH_I_OVERRIDE: name=%s,sv=%s,old=%c,new=%c\n", name,
744 sv, corigfrp, caltfrp);
745 fprintf(stderr, "MTH_I_OVERRIDE: f=%d, s=%d, iorigfrp=%d, ialtfrp=%d\n",
746 f, s, origfrp, altfrp);
747 }
748 fprintf(stderr, "MTH_I_OVERRIDE: Replacing %s(%s):%c=%s with %c=%s\n",
749 name, sv, corigfrp, fptr2char(__mth_rt_vi_ptrs_new[f][s][origfrp]),
750 caltfrp, fptr2char(__mth_rt_vi_ptrs_new[f][s][altfrp]));
751 }
752
753 /*
754 * Override dispatch entry.
755 */
756 __mth_rt_vi_ptrs_new[f][s][origfrp] = __mth_rt_vi_ptrs_new[f][s][altfrp];
757
758 if (tbuf[i] == '\0' || (tbuf[i] == ',' && tbuf[i + 1] == '\0')) {
759 failure = false;
760 break;
761 }
762 tbuf[i++] = '\0';
763 }
764
765 if (true == failure) {
766 mth_i_override_usage();
767 fprintf(stderr, "Errors in processing MTH_I_OVERRIDE");
768 exit(1);
769 }
770
771 free(tbuf);
772}

987void CONSTRUCTOR
988__math_dispatch()
989{
990
991 char *ptenv;
992 int i;
993 int j;
994 func_e f;
995 mth_intrins_defs_t *p;
996 sv_e s;
997
998 if (NULL != (ptenv = getenv("MTH_I_DEBUG"))) {
999 char *pend;
1000 __mth_i_debug = strtol(ptenv, &pend, 0);

1009 if (NULL != (ptenv = getenv("MTH_I_ARCH"))) {

1012 __math_target = text2archtype[i].parch;

1014 }
1015 }

1017 fputs("The only valid values for MTH_I_ARCH are:\n", stderr);
1018 for (i = 0; i < sizeof(text2archtype) / sizeof(*text2archtype); i++) {
1019 fprintf(stderr, " %s", text2archtype[i].pname);
1020 }
1021 fprintf(stderr, "\nDefaulting to " STR_ARCH_DEFAULT "\n");
1022 __math_target = ARCH_DEFAULT;
1023 }
1024
1025 } else if (NULL != (ptenv = getenv("PGI_FASTMATH_CPU"))) {
1026 for (i = 0; i < sizeof(text2archtype) / sizeof(*text2archtype); i++) {
1027 if (0 == strcasecmp(text2archtype[i].pname, ptenv)) {
1028 __math_target = text2archtype[i].parch;
1029 break;
1030 }
1031 }
1032 if (__math_target == arch_size) {
1033 fputs("The only valid values for MTH_I_ARCH are:\n", stderr);
1034 for (i = 0; i < sizeof(text2archtype) / sizeof(*text2archtype); i++) {
1035 fprintf(stderr, " %s", text2archtype[i].pname);
1036 }
1037 fprintf(stderr, "\nDefaulting to " STR_ARCH_DEFAULT "\n");
1038 __math_target = ARCH_DEFAULT;
1039 }
1040
1041 } else { /* Get processor architecture using CPUID information */
1042#if defined(TARGET_LINUX_X8664) || defined(TARGET_OSX_X8664) || defined(TARGET_WIN_X8664)
1043 if (X86IDFN(is_avx512vl)() == 1) {
1044 __math_target = arch_avx512;

1055 __math_target = arch_avxfma4;
1056 } else {
1057 __math_target = arch_sse4;
1058 }
1059 }
1060 } else {
1061 if ((X86IDFN(is_sse4a)() == 1) || (X86IDFN(is_sse41)() == 1)) {
1062 __math_target = arch_sse4;
1063 } else {
1064 __math_target = arch_em64t;
1065 }
1066 }
1067#endif
1068#ifdef TARGET_LINUX_POWER
1069 __math_target = ARCH_DEFAULT;
1070#endif
1071#ifdef TARGET_ARM64
1072 __math_target = ARCH_DEFAULT;
1073#endif
1074#ifdef TARGET_LINUX_GENERIC
1075 __math_target = ARCH_DEFAULT;
1076#endif
1077 }
1078
1079 /* Allow overriding of the default functions called for fast,
1080 * relaxed, and precise intrinsics. This is done by setting
1081 * the following environment variables:
1082 *
1083 * MTH_I_FAST = {fast,relaxed,precise,sleef}
1084 * MTH_I_RELAXED = {fast,relaxed,precise,sleef}
1085 * MTH_I_PRECISE = {fast,relaxed,precise,sleef}
1086 * MTH_I_SLEEF = {fast,relaxed,precise,sleef}
1087 *
1088 */
1089
1090 for (j = 0; j < sizeof(frp_env) / sizeof(*frp_env); j++) {
1091 if (NULL != (ptenv = getenv(frp_env[j].var))) {
1092 *frp_env[j].val = frp_size;
1093 for (i = 0; i < sizeof(frp2text) / sizeof(*frp2text); i++) {
1094 if (0 == strcasecmp(frp2text[i], ptenv)) {
1095 *frp_env[j].val = i;
1096 break;
1097 }
1098 }
1099 if (*frp_env[j].val == frp_size) {
1100 fprintf(stderr, "The only valid values for %s are:\n", frp_env[j].var);
1101 for (i = 0; i < sizeof(frp2text) / sizeof(*frp2text); i++) {
1102 fprintf(stderr, " %s", frp2text[i]);
1103 }
1104 *frp_env[j].val = frp_env[j].defval;
1105 fprintf(stderr, "\nDefaulting to %s\n", frp2text[*frp_env[j].val]);
1106 }
1107 }
1108 }
1109
1110 if (__mth_i_debug & 0x1) {
1111 fprintf(stderr, "__math_target: %s(%d)\n", carch[__math_target],
1112 __math_target);
1113 fprintf(stderr, "__mth_fast: %s(%d)\n", frp2text[__mth_fast], __mth_fast);
1114 fprintf(stderr, "__mth_relaxed: %s(%d)\n", frp2text[__mth_relaxed],
1115 __mth_relaxed);
1116 fprintf(stderr, "__mth_precise: %s(%d)\n", frp2text[__mth_precise],
1117 __mth_precise);
1118 fprintf(stderr, "__mth_sleef: %s(%d)\n", frp2text[__mth_sleef],
1119 __mth_sleef);
1120 fputs("__math_dispatch: built on " __DATE__ " " __TIME__ " with "
1121#if defined(__clang__)
1122 // Too much internal info "clang-" __clang_version__
1123 "clang-" STRINGIFY(__clang_major__ )
1124 "." STRINGIFY(__clang_minor__)
1125 "." STRINGIFY(__clang_patchlevel__)
1126#elif defined(__GNUC__)
1127 "gcc-" STRINGIFY(__GNUC__)
1128 "." STRINGIFY(__GNUC_MINOR__)
1129 "." STRINGIFY(__GNUC_PATCHLEVEL__)
1130#elif defined(__PGIC__)
1131 "pgcc-" STRINGIFY(__PGIC__)
1132 "." STRINGIFY(__PGIC_MINOR__)
1133 "." STRINGIFY(__PGIC_PATCHLEVEL__)
1134#else
1135 "Unknown build compiler"
1136#endif
1137 "\n", stderr);
1138 }
1139
1140 /*
1141 * 2 pass scan of table definitions.
1142 * 1st pass:
1143 * populate __mth_rt_vi_ptrs with all mth_intrins_defs entries that
1144 * have arch_any.
1145 * 2nd pass:
1146 * populate __mth_rt_vi_ptrs with entries whose architecture match
1147 * __math_target.
1148 *
1149 * We need two passes to build the dispatch table because there are not
1150 * ordering rules for where arch_any can be specified in the definition
1151 * files.
1152 *
1153 * Probably not the most efficient way to scan the definition table twice,
1154 * but the table is relatively speaking not that large.
1155 */
1156
1157 for (j = 0; j < 2; j++) {
1158 arch_e target_arch;
1159
1160 target_arch = j == 0 ? arch_any : __math_target;
1161 for (i = 0; i < sizeof(mth_intrins_defs) / sizeof(*mth_intrins_defs); i++) {
1162 p = &mth_intrins_defs[i];
1163
1164 /*
1165 * XXX - testing p->arch == __math_target is not sufficient,
1166 * since arch_any is 0, and all the undefined entries will have
1167 * arch == 0.
1168 * But defined entries will have one or more valid pointers to intrinsic
1169 * functions.
1170 *
1171 * Macro MTHINTRIN requires all 3 F/R/P fields to be defined, though
1172 * technically nothing prevents any one of those fields from using
1173 * "NULL".
1174 *
1175 * Here to check for a valid definition we only check the first field
1176 * F(ast) for a non NULL value.
1177 */
1178
1179 if (p->arch == target_arch && p->pf != NULL) {
1180 if (__mth_fast == frp_f) {
