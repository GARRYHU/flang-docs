81/* RPCT (runtime pointer conflict test) struct:
82 * The NMEs 'nme1' and 'nme2' in an RPCT record are "RPCT NMEs" that
83 * are created by 'add_rpct_nme()', and the RPCT record itself is
84 * created by 'add_rpct( nme1, nme2 )'. The existence of such a
85 * record indicates that the generated code contains a test which
86 * proves at runtime that this pair of references do not conflict, so
87 * 'conflict( nme1, nme2 )' will return NOCONFLICT.
88 */
89typedef struct {
90 int nme1;
91 int nme2;
92 int hshlnk;
93} RPCT;
