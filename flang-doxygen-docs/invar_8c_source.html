8/**
9 \file
10 \brief optimizer submodule responsible for invariant analysis
11 for a given loop. Used by the optimizer and vectorizer.
12
13 NOTE: During the detection phase of the analysis, stores are analyzed by
14 traversing the store lists of a loop and its children. A store is marked by
15 recording the store item in the corresponding name's stl field. The lifetime
16 of the names' stl fields extends after the analysis phase so that the general
17 query function, is_invariant, can be provided. The stl fields (and ili
18 marked during the analysis) are cleaned up in unmark/unmarkv.
19 */

73/*
74 * perform invariant analysis along with code motion.
75 */
76void
77invariant(int lp)
78{
79 int stdx, fgx, fg_tail, fg_tailnext;
80 invar_init(lp);
81
82 invar_mark(lp);
83 LP_HSTDF(lp) = 0;
84 LP_DSTDF(lp) = 0;
85
86 if (flg.opt >= 2 && !XBIT(2, 0x400000)) {
87 if (!LP_MEXITS(lp) && !LP_PARLOOP(lp) && !LP_CS(lp)) {
88 fg_tail = LP_TAIL(lp);
89 fg_tailnext = FG_LNEXT(fg_tail);
90 for (fgx = LP_HEAD(lp); fgx && fgx != fg_tailnext; fgx = FG_LNEXT(fgx)) {
91 for (stdx = FG_STDFIRST(fgx); stdx; stdx = STD_NEXT(stdx)) {
92 if (STD_ACCEL(stdx) || STD_KERNEL(stdx)) {
93 } else {
94 invar_motion((int)stdx);
95 }
96 if (stdx == FG_STDLAST(fgx))
97 break;
98 }
99 if (fgx == fg_tail)
100 break;
101 }
102 }
103 }
104
105 invar_end(lp);
106 LP_HSTDF(lp) = 0;
107 LP_DSTDF(lp) = 0;
108
109}

137/*
138 * cleanup invariant table after it's not needed (typically during the last
139 * step in induction analysis, must be used after invariant analysis when
140 * it includes code motion.
141 */
142void
143invariant_unmark(void)
144{
145 int i, ilix;
146 STL *stl;
147
148 for (i = opt.invb.stg_avail - 1; i; i--) {
149 ilix = opt.invb.stg_base[i];
150 if (!AST_ISINV_TEMP(ilix))
151 AST_INVP(ilix, 0);
152 }
153 /*
154 * clean up the names entries of the stores in the loop and in any nested
155 * loops
156 */
157 stl = LP_STL(lpx);
158 cleannames(stl);
159}

200static void
201invar_mark(int lp)
202{
203 /*
204 * If a loop contains a critical section, first traverse the blocks which
205 * have their FG_CS flags set; this ensures that any ili in the critical
206 * sections will not be considered to be invariant.
207 */
208 if (LP_CS(lp)) {
209 mark_value = NOT_INV;
210 mark_return = FALSE;
211 for (fgx = LP_FG(lp); fgx; fgx = FG_NEXT(fgx)) {
212 safe = TRUE;
213 if (FG_CS(fgx)) {
214 if (OPTDBG(9, 256))
215 fprintf(gbl.dbgfil, " flow graph node %d (LP_CS)\n", fgx);
216 for (stdx = FG_STDFIRST(fgx); stdx; stdx = STD_NEXT(stdx)) {
217 if (OPTDBG(9, 256))
218 fprintf(gbl.dbgfil, " ilt %d\n", stdx);
219 (void)invar_src((int)STD_AST(stdx));
220 if (stdx == FG_STDLAST(fgx))
221 break;
222 }
223 }
224 }
225 }
226 mark_value = INV;
227 mark_return = TRUE;
228
229 /*
230 * Next, traverse the blocks in the loop that are always executed.
231 * i.e, blocks which dominate the tail of the loop. Otherwise, could
232 * first discover an unsafe invariant expression which appears later
233 * in block that's always executed.
234 */
235 for (fgx = LP_FG(lp); fgx; fgx = FG_NEXT(fgx)) {
236 safe = is_dominator(fgx, (int)LP_TAIL(lp));
237 if (safe) {
238 if (OPTDBG(9, 256))
239 fprintf(gbl.dbgfil, " flow graph node %d (%s)\n", fgx,
240 safe ? "safe" : "unsafe");
241 for (stdx = FG_STDFIRST(fgx); stdx; stdx = STD_NEXT(stdx)) {
242 if (OPTDBG(9, 256))
243 fprintf(gbl.dbgfil, " std %d\n", stdx);
244 (void)invar_src((int)STD_AST(stdx));
245 if (stdx == FG_STDLAST(fgx))
246 break;
247 }
248 }
249 }
250 /*
251 * The last traversal is the blocks in the loop that are not always
252 * executed.
253 */
254 for (fgx = LP_FG(lp); fgx; fgx = FG_NEXT(fgx)) {
255 safe = is_dominator(fgx, (int)LP_TAIL(lp));
256 if (!safe) {
257 if (OPTDBG(9, 256))
258 fprintf(gbl.dbgfil, " flow graph node %d (%s)\n", fgx,
259 safe ? "safe" : "unsafe");
260 for (stdx = FG_STDFIRST(fgx); stdx; stdx = STD_NEXT(stdx)) {
261 if (OPTDBG(9, 256))
262 fprintf(gbl.dbgfil, " std %d\n", stdx);
263 (void)invar_src((int)STD_AST(stdx));
264 if (stdx == FG_STDLAST(fgx))
265 break;
266 }
267 }
268 }
269
270}

280/*
281 * initialize the names entries of the stores in the loop and in any nested
282 * loops
283 */
284static void
285initnames(STL *stlitem)
286{
287 STL *tmp;
288 int i, j, nme;
289
290 tmp = stlitem;
291 for (i = tmp->store; i; i = STORE_NEXT(i)) {
292 nme = STORE_NM(i);
293 j = NME_STL(nme);
294 if (j)
295 STORE_TYPE(j) |= STORE_TYPE(i);
296 else
297 NME_STL(nme) = i;
298 }
299 /* init names (recursively) for nested loops too. */
300 for (tmp = tmp->childlst; tmp != NULL; tmp = tmp->nextsibl) {
301 initnames(tmp);
302 }
303}
304
305/*
306 * clean up the names entries of the stores in the loop and in any nested
307 * loops
308 */
309static void
310cleannames(STL *stlitem)
311{
312 STL *tmp;
313 int i;
314
315 tmp = stlitem;
316 for (i = tmp->store; i; i = STORE_NEXT(i))
317 NME_STL(STORE_NM(i)) = 0;
318
319 /* clean up (recursively) names of any descendants, too */
320 for (tmp = tmp->childlst; tmp != NULL; tmp = tmp->nextsibl) {
321 cleannames(tmp);
322 }
323}

327 * the regular analysis has been done; i.e., safe and lpx which are
328 * static have been set.

330LOGICAL
331is_invariant(int ilix)
332{
333 LOGICAL invf;
334 if (OPTDBG(9, 256))
335 fprintf(gbl.dbgfil, "is_invariant of ili %d called\n", ilix);
336 invf = invar_src(ilix);

345static LOGICAL
346invar_src(int ast)
347{
348 ast_visit(1, 1);
349 ast_traverse(ast, NULL, compute_invariant, NULL);
350 ast_unvisit();
351 return AST_ISINV(ast);
352}

354static void
355compute_invariant(int ast, int *dummy)
356{
357 int atype;
358 int sym;
359 int nme;
360 int dtype;
361 int i, asd;
362 int argt;
363 int cnt;
364 LOGICAL invar_flag;
365
366 /* already done
367 if (AST_INVG(ast))
368 return;
369 */
370
371 switch (atype = A_TYPEG(ast)) {
372 case A_NULL:
373 case A_CNST:
374 case A_CMPLXC:
375 break; /* mark invariant */
376 case A_ID:
377 sym = A_SPTRG(ast);
378 if (ST_ISVAR(STYPEG(sym)) || (ST_ARRDSC == STYPEG(sym))) {
379 nme = A_NMEG(ast);
380 if (NME_STL(nme)) {
381 if (OPTDBG(9, 256))
382 fprintf(gbl.dbgfil,
383 " ast %5d-%s %s is not invariant--ST_NME %d\n", ast,
384 astb.atypes[atype], SYMNAME(sym), nme);
385 goto mark_variant;
386 }
387 if (NME_TYPE(nme) != NT_VAR)
388 goto mark_variant;
389 if (!is_sym_invariant_safe(nme, lpx)) {
390 if (OPTDBG(9, 256)) {
391 fprintf(gbl.dbgfil, " ast %5d-%s not invariant - %s unsafe\n",
392 ast, astb.atypes[atype], SYMNAME(sym));
393 }
394 goto mark_variant;
395 }
396 }
397 break;
398 case A_MEM:
399 if (!AST_ISINV(A_PARENTG(ast)))
400 goto markd_variant;
401 /* _ast_trav(A_MEMG(ast)) */
402 break;
403 case A_BINOP:
404 invar_flag = AST_ISINV(A_LOPG(ast)) && AST_ISINV(A_ROPG(ast));
405 if (!invar_flag)
406 goto markd_variant;
407 goto chk_fp_safe;
408 case A_UNOP:
409 case A_CONV:
410 case A_PAREN:
411 if (!AST_ISINV(A_LOPG(ast)))
412 goto markd_variant;
413 goto chk_fp_safe;
414 case A_SUBSCR:
415 asd = A_ASDG(ast);
416 if (!AST_ISINV(A_LOPG(ast)))
417 goto markd_variant;
418 for (i = 0; i < (int)ASD_NDIM(asd); i++)
419 if (!AST_ISINV(ASD_SUBS(asd, i)))
420 goto markd_variant;
421 break;
422 case A_SUBSTR:
423 if (!AST_ISINV(A_LOPG(ast)))
424 goto markd_variant;
425 if (A_LEFTG(ast) && !AST_ISINV(A_LEFTG(ast)))
426 goto markd_variant;
427 if (A_RIGHTG(ast) && !AST_ISINV(A_RIGHTG(ast)))
428 goto markd_variant;
429 break;
430 case A_TRIPLE:
431 /* [lb]:[ub][:stride] */
432 if (A_LBDG(ast) && !AST_ISINV(A_LBDG(ast)))
433 goto markd_variant;
434 if (A_UPBDG(ast) && !AST_ISINV(A_UPBDG(ast)))
435 goto markd_variant;
436 if (A_STRIDEG(ast) && !AST_ISINV(A_STRIDEG(ast)))
437 goto markd_variant;
438 break;
439 case A_FUNC:
440 goto markd_variant;
441 case A_ALLOC:
442
443/* currently enabled when STD_HSTBLE(stdx) is set , this flag is set
444 * when the allocate/deallocate is for forall temp array.
445 * might need to keep track of allo/deall.
446 */
447 if (flg.opt >= 2 && !XBIT(2, 0x400000)) {
448 if (STD_HSTBLE(stdx) && is_alloc_ast(ast)) {
449 int srcast = A_SRCG(ast);
450 if (A_TYPEG(srcast) == A_SUBSCR)
451 goto markd_variant;
452 compute_invariant(srcast, NULL);
453 if (!AST_ISINV(srcast))
454 goto markd_variant;
455 if (A_SHAPEG(srcast)) {
456 int asd, i, ndim, ss;
457 asd = A_ASDG(srcast);
458 ndim = ASD_NDIM(asd);
459 for (i = 0; i < ndim; ++i) {
460 ss = ASD_SUBS(asd, i);
461 compute_invariant((ss), NULL);
462 if (!AST_ISINV(ss))
463 goto markd_variant;
464 }
465 }
466 }
467 if (STD_HSTBLE(stdx) && is_dealloc_ast(ast)) {
468 int std = STD_HSTBLE(stdx);
469 /* check if ast of std is this ast and it is hoistable and it is
470 * invariant */
471
472 if (A_TYPEG(A_SRCG(ast)) == A_SUBSCR || !STD_HSTBLE(std))
473 goto markd_variant;
474 if (AST_ISINV(A_SRCG(ast)))
475 break;
476 }
477 }
478 goto markd_variant;
479 break;
480
481 case A_INTR:
482 switch (A_OPTYPEG(ast)) {
483 case I_RAN:
484 case I_RANDOM_NUMBER:
485 case I_RANDOM_SEED:
486 goto markd_variant;
487 }
488 cnt = A_ARGCNTG(ast);
489 argt = A_ARGSG(ast);
490 if (cnt) {
491 i = 0;
492 while (TRUE) {
493 /* lfm -- optional args */
494 if (ARGT_ARG(argt, i) != 0 && !AST_ISINV(ARGT_ARG(argt, i)))
495 goto markd_variant;
496 if (++i >= cnt)
497 break;
498 }
499 }
500 chk_fp_safe:
501 if (!safe) {
502 dtype = DDTG(A_DTYPEG(ast));
503 if (DT_ISREAL(dtype) || DT_ISCMPLX(dtype))
504 goto unsafe;
505 }
506 break;
507 default:
508 AST_INVP(ast, NOT_INV);
509 return; /* default; don't add to invb list */
510 }
511
512 invariant_mark(ast, mark_value);
513 if (OPTDBG(9, 256)) {
514 if (mark_value == INV || mark_value == T_INV) {
515 fprintf(gbl.dbgfil, " ast %5d-%s", ast, astb.atypes[atype]);
516 if (A_TYPEG(ast) == A_ID)
517 fprintf(gbl.dbgfil, " %s", SYMNAME(A_SPTRG(ast)));
518 fprintf(gbl.dbgfil, " is invariant\n");
519 } else
520 fprintf(gbl.dbgfil, " ast %5d-%s is not invariant -- csec\n", ast,
521 astb.atypes[atype]);
522 }
523 return;
524
525unsafe:
526 if (OPTDBG(9, 256))
527 fprintf(gbl.dbgfil, " ast %5d-%s -- unsafe\n", ast,
528 astb.atypes[atype]);
529/* fall thru */
530markd_variant:
531 if (OPTDBG(9, 256))
532 fprintf(gbl.dbgfil, " ast %5d-%s is not invariant\n", ast,
533 astb.atypes[atype]);
534mark_variant:
535 invariant_mark(ast, NOT_INV);
536
537}

540/*
541 * for any array nme's in nme, check their subscripts. Note that this only
542 * marks the ili if necessary. Detecting an invariant subscript is not
543 * enough for it to be moved; the subscript ili must actually appear in
544 * a tree rooted by an ilt. There are situations (e.g., a[i+1]) where
545 * the subscript ili does not actually appear in the reference due to
546 * algebraic simplification.
547 */
548static void
549invar_arrnme(int nme)
550{
551 int anme;
552
553 anme = nme;
554 while (NME_TYPE(anme) == NT_MEM || NME_TYPE(anme) == NT_IND)
555 anme = NME_NM(anme);
556 while (NME_TYPE(anme) == NT_ARR) {
557 if (NME_SUB(anme)) {
558 if (OPTDBG(9, 256))
559 fprintf(gbl.dbgfil, " arrnme %d, subili %d\n", anme,
560 NME_SUB(anme));
561 (void)invar_src((int)NME_SUB(anme));
562 }
563 anme = NME_NM(anme);
564 }
565}
566#endif

570/* allocate statement to move up, deallocate statement to move down */
571static void
572invar_motion(int std)
573{
574 int ast, astd;
575 LOGICAL hstable = FALSE;
576 /* only STD_HSTBLE for now */
577 if (STD_HSTBLE(std)) {
578 ast = STD_AST(std);
579 if (is_alloc_ast(ast)) {
580 hstable = is_std_hoistable(std, lpx);
581 if (hstable) {
582 hoist_stmt(std, LP_FG(lpx), lpx);
583 } else {
584 /* mark not hoistable so that the deallocate will not be hoist */
585 astd = STD_HSTBLE(std);
586 STD_HSTBLE(std) = 0;
587 STD_HSTBLE(astd) = 0;
588 }
589 } else if (is_dealloc_ast(ast) && STD_HSTBLE(std)) {
590 astd = STD_HSTBLE(std);
591 if (STD_VISIT(astd)) {
592 hoist_stmt(std, LP_FG(lpx), lpx);
593
594 /* only do one level of hoisting for now */
595 STD_HSTBLE(std) = 0;
596 STD_HSTBLE(astd) = 0;
597 }
598 }
599 }
600}

602static LOGICAL
603def_in_innerlp(int lpx, int def_lp)
604{
605 int lp;
606 for (lp = LP_CHILD(lpx); lp; lp = LP_SIBLING(lp)) {
607 if (lp == def_lp)
608 return TRUE;
609 }
610 return FALSE;
611}

613/* return a number of def in a loop
614 * relies on LP_STL
615 * for subscript x[3]:
616 * if there is a whole array ref (x) => no
617 * if there is a subscript but variable index (x[n]) => no
618 */
619static LOGICAL
620has_def_inlp(int ast, int lp, int std)
621{
622 int lop, def, nme, def_fg, asd, def_addr, ndim;
623 int def_count = 0;
624
625 switch (A_TYPEG(ast)) {
626
627 case A_SUBSCR:
628 lop = A_LOPG(ast);
629 if (A_TYPEG(lop) != A_ID) /* don't handle derived type yet */
630 return FALSE;
631 nme = A_NMEG(ast);
632 if (!nme)
633 return FALSE;
634 if (NME_TYPE(nme) != NT_ARR)
635 return FALSE;
636 nme = NME_NM(nme);
637
638 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
639 def_fg = DEF_FG(def);
640 def_addr = DEF_ADDR(def);
641 asd = A_ASDG(def_addr);
642 ndim = ASD_NDIM(asd);
643 if (FG_LOOP(def_fg) != lp && !def_in_innerlp(lp, FG_LOOP(def_fg)))
644 continue;
645 if (DEF_STD(def) == std) {
646 ++def_count;
647 } else {
648 if (def_addr == DEF_LHS(def)) /* whole array */
649 ++def_count;
650 else if (A_TYPEG(ASD_SUBS(asd, 0)) != A_CNST)
651 ++def_count; /* index is variable, don't do further */
652 else if (def_addr == ast) /* same index */
653 ++def_count;
654 }
655 if (def_count > 1)
656 return FALSE;
657 }
658 break;
659 case A_ID:
660 nme = A_NMEG(ast);
661 break;
662 default:
663 return FALSE;
664 break;
665 }
666
667 if (def_count > 1)
668 return FALSE;
669
670 return TRUE;
671}

673static LOGICAL
674 /* also do the hoist */
675 is_hoistable(int ast, int std, int lp)
676{
677 int lop, sym, nme, sptr, def_addr, ndim, def_std, i;
678 int asd, du_std;
679 int use_count, def_count;
680 int l, u, s, r, cnme, found_nme, def, def_fg, hoistme;
681 STL *stl;
682 DU *du;
683
684 switch (A_TYPEG(ast)) {
685 case A_BINOP:
686 r = is_hoistable(A_ROPG(ast), std, lp);
687 l = is_hoistable(A_LOPG(ast), std, lp);
688 if (l & r) {
689 return TRUE;
690 }
691 break;
692 case A_UNOP:
693 l = is_hoistable(A_LOPG(ast), std, lp);
694 if (l)
695 return TRUE;
696 break;
697 case A_TRIPLE:
698 l = is_hoistable(A_LBDG(ast), std, lp);
699 u = is_hoistable(A_UPBDG(ast), std, lp);
700 s = is_hoistable(A_STRIDEG(ast), std, lp);
701
702 return (l & u & s);
703 break;
704 case A_CNST:
705 return TRUE;
706 case A_ID:
707 return FALSE; /* for now */
708 break;
709 case A_SUBSCR:
710 if (A_SHAPEG(ast)) /* currently don't do too many nested subscripts */
711 return FALSE;
712 lop = A_LOPG(ast);
713 if (A_TYPEG(lop) == A_ID) {
714 sym = A_SPTRG(lop);
715 if (ADDRTKNG(sym) && (LP_CALLFG(lp) || LP_CALLINTERNAL(lp))) {
716 /* to do list: if call is external we can check if this is passed
717 * as an argument
718 * NOTE: should I also check inner loop calls?
719 */
720 return FALSE;
721 }
722 cnme = A_NMEG(lop);
723 if (NME_SYM(cnme) == -1)
724 cnme = NME_NM(cnme);
725 if (STYPEG(sym) == ST_DESCRIPTOR) {
726 stl = LP_STL(lp);
727 /* no need to use store list */
728 found_nme = 0;
729 for (i = stl->store; i; i = STORE_NEXT(i)) {
730 nme = STORE_NM(i);
731 if (NME_SYM(nme) == -1)
732 nme = NME_NM(nme);
733 sptr = NME_SYM(nme);
734 if (cnme != nme)
735 continue;
736 if (NME_TYPE(nme) != NT_VAR)
737 return FALSE;
738
739 found_nme = 1;
740 use_count = 0;
741 def_count = 0;
742 hoistme = 0;
743 for (def = NME_DEF(nme); def; def = DEF_NEXT(def)) {
744 def_addr = DEF_ADDR(def);
745 def_std = DEF_STD(def);
746
747 /* check if this def is in a loop or inner loop */
748 def_fg = DEF_FG(def);
749 if (lp != FG_LOOP(def_fg)) {
750 if (LP_CHILD(lp) && !def_in_innerlp(lp, FG_LOOP(def_fg)))
751 continue;
752 }
753 if (def_addr && A_TYPEG(def_addr) == A_SUBSCR) {
754 asd = A_ASDG(def_addr);
755 ndim = ASD_NDIM(asd);
756 if (ndim > 1)
757 return FALSE;
758 /* don't want to do more checking if subscript is not constant */
759 if (A_TYPEG(ASD_SUBS(asd, 0)) != A_CNST)
760 return FALSE;
761
762 if (def_addr == ast) {
763
764 /* handle with 1 def for now */
765 ++def_count;
766 if (def_count > 1)
767 return FALSE;
768
769 for (du = DEF_DU(def); du; du = du->next) {
770 du_std = USE_STD(du->use);
771 if (USE_STD(du->use) == std) {
772 use_count++;
773
774 /* this check(def_std != std) is unnessary but we
775 * want to make sure that infinite recursive
776 * won't happen here.
777 */
778 if (def_std != std) {
779 if (!is_std_hoistable(def_std, lp)) {
780 return FALSE;
781 } else {
782 hoistme = 1;
783 }
784 }
785 }
786 }
787 }
788
789 } else {
790 return FALSE;
791 }
792 if (def_count > 1)
793 return FALSE;
794
795 /* do the hoist of std */
796 if (hoistme) {
797 hoist_stmt(def_std, LP_FG(lp), lp);
798 return TRUE;
799 }
800 }
801
802 if (found_nme) {
803 return TRUE;
804 }
805 /* found nme */
806 break;
807 }
808 if (found_nme == 0) { /* not found in store list */
809 return TRUE;
810 }
811 }
812 }
813 break;
814 default:
815 break;
816 }
817 return FALSE;
818}

821static LOGICAL
822is_std_hoistable(int std, int lpx)
823{
824 LOGICAL canhoist = FALSE;
825 LOGICAL l, r, s, u, sym, tmplog;
826 int shape, n, i, astd;
827 int ast = STD_AST(std);
828
829 switch (A_TYPEG(ast)) {
830 case A_ALLOC:
831 /* assume temp alloc */
832 if (is_alloc_ast(ast) && STD_HSTBLE(std)) {
833 ast = A_SRCG(ast);
834 switch (A_TYPEG(ast)) {
835 case A_SUBSCR:
836 shape = A_SHAPEG(ast);
837 n = SHD_NDIM(shape);
838 tmplog = FALSE;
839 for (i = 0; i < n; ++i) {
840 l = is_hoistable(SHD_LWB(shape, i), std, lpx);
841 u = is_hoistable(SHD_UPB(shape, i), std, lpx);
842 s = is_hoistable(SHD_STRIDE(shape, i), std, lpx);
843 tmplog = l && u && s;
844 if (tmplog) {
845 int fast = A_FIRSTALLOCG(ast);
846 if (fast) {
847 if (!is_hoistable(fast, std, lpx))
848 return FALSE;
849 }
850 } else {
851 return FALSE;
852 }
853 }
854 return tmplog;
855 break;
856 case A_ID:
857 return FALSE; /* To do list */
858 break;
859 default:
860 return FALSE;
861 }
862 } else if (is_dealloc_ast(ast) && STD_HSTBLE(std)) {
863 /* this assume that allocate statement is already visited */
864 astd = STD_HSTBLE(ast);
865 if (STD_HSTBLE(astd) && STD_VISIT(astd))
866 return TRUE;
867 }
868 return FALSE;
869 break;
870
871 case A_ASN:
872 l = A_DESTG(ast);
873 r = A_SRCG(ast);
874
875 if (A_TYPEG(l) == A_SUBSCR) {
876 int a = A_LOPG(l);
877 if (A_TYPEG(a) != A_ID) {
878 return FALSE;
879 }
880 } else if (A_TYPEG(l) == A_ID) {
881 sym = A_SPTRG(ast);
882 if (ADDRTKNG(sym)) {
883 if (LP_CALLFG(lpx) || LP_CALLINTERNAL(lpx))
884 return FALSE;
885 }
886 } else {
887 return FALSE;
888 }
889
890 if (has_def_inlp(l, lpx, std) <= 1) {
891 /* only hoist this std if the lhs has only one def,
892 * address taken ok but no call
893 */
894 if (is_hoistable(r, std, lpx)) {
895 return TRUE;
896 } else {
897 return FALSE;
898 }
899 }
900
901 break;
902
903 /* everything else is not hoistable for now */
904 case A_FUNC:
905 case A_INTR:
906 case A_CALL:
907 case A_ICALL:
908 case A_ASNGOTO:
909 default:
910 return FALSE;
911 break;
912 }
913 return canhoist;
914}

958LOGICAL
959is_sym_invariant_safe(int nme, int lpx)
960{
961 int sym, sflag;
962 sflag = 0;
963
964 /*
965 * a symbol is not safe (potential "side-effect" conflicts exist) if:
966 * 1. for c and fortran, the symbol is volatile
967 *
968 * 2. for fortran, the symbol is equivalenced
969 *
970 * 3. loop contains a call AND
971 * a. sym's address has been taken, or
972 * b. sym's storage class is not auto (it is static or extern).
973 * c. or for pascal- sym is used in two or more nested scope levels
974 *
975 * 4. loop contains a store via a pointer AND the sym could
976 * conflict with a pointer (see optutil.c:is_sym_ptrsafe()).
977 *
978 * 5. the symbol is private and the loop is not a parallel region.
979 *
980 * 6. the symbol is not a private variable and is stored while in a
981 * critical section in this loop or any enclosed loop, or in a
982 * parallel region.
983 *
984 * 7. the symbol is not a private variable and the loop contains
985 * a parallel section.
986 *
987 * 8. for Fortran, the loop contains a call to an internal subprogram
988 * and this is a noninternal variable.
989 *
990 * NOTE that this differs from is_sym_optsafe in that a load via a pointer
991 * does not rule out loads of certain symbols as being invariant. This is
992 * because the symbol cannot be redefined with a load via a pointer.
993 */
994 sym = NME_SYM(nme);
995 if (VOLG(sym))
996 return (FALSE);
997
998 if (SOCPTRG(sym))
999 return (FALSE);
1000 if (LP_CALLFG(lpx) && (!IS_LCL_OR_DUM(sym)
1001 )) {
1002 return (FALSE);
1003 }
1004
1005 if (SOCPTRG(sym))
1006 return (FALSE);
1007
1008 if (LP_CALLINTERNAL(lpx) && !INTERNALG(sym))
1009 return FALSE;
1010
1011 if (IS_PRIVATE(sym) && !LP_PARREGN(lpx))
1012 return (FALSE);
1013
1014 if ((LP_CSECT(lpx) || LP_PARREGN(lpx)) && !IS_PRIVATE(sym)) {
1015 Q_ITEM *q;
1016 for (q = LP_STL_PAR(lpx); q != NULL; q = q->next)
1017 if (q->info == nme)
1018 return (FALSE);
1019 }
1020
1021 if (LP_PARSECT(lpx) && !IS_PRIVATE(sym)) {
1022 return (FALSE);
1023 }
1024
1025 if (!sflag && LP_PTR_STORE(lpx) && !is_sym_ptrsafe(sym))
1026 /* check ptr before argument */
1027 /* if (flg.opt >= 2 && XBIT (xxx)) {
1028 is_sym_invar_inlp();
1029 }*/
1030 return (FALSE);
1031
1032 if (!sflag && LP_CALLFG(lpx) && ADDRTKNG(sym)) {
1033 /* check argument in a call, if it is in then it is safe */
1034 return (FALSE);
1035 }
1036
1037 return (TRUE);
1038}
