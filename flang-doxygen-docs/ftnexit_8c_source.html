28static void
29_f90io_f2003_stop_with_ieee_warnings(int exc)
30{
31 if ((exc & FE_INVALID) == FE_INVALID)
32 fprintf(__io_stderr(), "Warning: ieee_invalid is signaling\n");
33#ifdef FE_DENORM
34 if ((exc & FE_DENORM) == FE_DENORM)
35 fprintf(__io_stderr(), "Warning: ieee_denorm is signaling\n");
36#endif
37 if ((exc & FE_DIVBYZERO) == FE_DIVBYZERO)
38 fprintf(__io_stderr(), "Warning: ieee_divide_by_zero is signaling\n");
39 if ((exc & FE_OVERFLOW) == FE_OVERFLOW)
40 fprintf(__io_stderr(), "Warning: ieee_overflow is signaling\n");
41 if ((exc & FE_UNDERFLOW) == FE_UNDERFLOW)
42 fprintf(__io_stderr(), "Warning: ieee_underflow is signaling\n");
43 if ((exc & FE_INEXACT) == FE_INEXACT)
44 fprintf(__io_stderr(), "Warning: ieee_inexact is signaling\n");
45}

47static void
48_f90io_stop(int exit_status, char *str, __CLEN_T str_siz, __LOG_T is_quiet, __LOG_T is_errorstop)
49{
50 int __fenv_fetestexcept(int);
51 int anyexc;
52 anyexc = __fenv_fetestexcept(FE_ALL_EXCEPT);
53 MP_P_STDIO;
54 if (str) {
55 if (is_quiet == 0)
56 _f90io_f2003_stop_with_ieee_warnings(anyexc);
57 if (is_errorstop)
58 fprintf(__io_stderr(), "ERROR STOP ");
59
60 fprintf(__io_stderr(), "%.*s\n", str_siz, str);
61 } else if (__fort_getenv("NO_STOP_MESSAGE") == 0
62 ) {
63 if (is_quiet == 0)
64 _f90io_f2003_stop_with_ieee_warnings(anyexc);
65 if (is_quiet == 0) {
66 if (is_errorstop)
67 fprintf(__io_stderr(), "ERROR STOP\n");
68 else
69 fprintf(__io_stderr(), "FORTRAN STOP\n");
70 }
71 }
72 MP_V_STDIO;
73 __fort_exit(exit_status);
74}

162static void
163_f90io_pause(char *str, __CLEN_T str_siz)
164{
165 MP_P_STDIO;
166 if (str)
167 fprintf(__io_stderr(), "FORTRAN PAUSE: %.*s\n", str_siz, str);
168 if (__fort_isatty(__fort_getfd(__io_stdin()))) {
169 fprintf(__io_stderr(),
170 "FORTRAN PAUSE: enter <return> or <ctrl>d to continue>");
171 while (1) {
172 char c = fgetc(__io_stdin());
173 if (c == '\n')
174 break;
175 if (__io_feof(__io_stdin())) {
176 fprintf(__io_stderr(), "\n");
177 break;
178 }
179 }
180 } else /* print message but don't pause: */
181 fprintf(__io_stderr(), "FORTRAN PAUSE: continuing...\n");
182 MP_V_STDIO;
183}
