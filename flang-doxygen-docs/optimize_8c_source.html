59/*
60 * Initialize and free the data structures that are shared by the
61 * vectorizer and optimizer. These data structures represent the
62 * information created by those modules that are shared by the vectorizer
63 * and optimizer. For C, this space is reused across functions in the
64 * source file during a given pass (vectorizer and optimizer); for Fortran,
65 * the space is allocated for each function. At the end of each pass, the
66 * space is freed.
67 *
68 * The shared modules are:
69 * flowgraph
70 * findloop
71 * flow
72 * invariant analysis (not code motion).
73 *
74 * Since it's possible for the vectorizer/optimizer to see more than one
75 * function, it's necessary to initialize and free certain data structures
76 * that are valid only for the lifetime of a function (for example,
77 * various getitem areas).
78 *
79 * A skeleton of the vectorizer/optimizer for using these routines is:
80 *
81 * optshrd_init();
82 * foreach function {
83 * optshrd_finit();
84 * ...
85 * optshrd_fend();
86 * }
87 * optshrd_end();
88 *
89 */
90
91void
92optshrd_init(void)
93{
94
95 STG_ALLOC(opt.fgb, 100); /* flowgraph space */
96 opt.fgb.stg_avail = 0; /* expected starting value */
97 gbl.entbih = 1; /* the first bih */
98 STG_ALLOC(opt.rteb, 32); /* retreating edges */
99
100 STG_ALLOC(opt.lpb, 50); /* loop table */
101 LP_PARENT(0) = 0; /* set the parent of region 0 to 0 -- this is
102 * for tests which look at the parent of a
103 * loop without looking at the loop index
104 */
105
106 STG_ALLOC(opt.storeb, 100); /* store area */
107 STG_ALLOC(opt.defb, 64); /* definition table */
108 STG_ALLOC(opt.useb, 64); /* use lists */
109
110 STG_ALLOC(opt.invb, 100); /* invariant expr area */
111 STG_ALLOC_SIDECAR(astb, opt.astb);
112
113 opt.sc = SC_AUTO; /* default storage class for opt-created temps */
114
115 nme_init();
116
117}

245/*
246 * given a linked list of flow edges;
247 * return the linked list with the flow edge to 'r' removed
248 */
249static PSI_P
250remove_flow_list(PSI_P link, int r)
251{
252 PSI_P head, tail, next;
253 head = NULL;
254 tail = NULL;
255 for (; link; link = next) {
256 next = PSI_NEXT(link);
257 PSI_NEXT(link) = NULL;
258 if (PSI_NODE(link) != r) {
259 if (head == NULL) {
260 head = link;
261 } else {
262 PSI_NEXT(tail) = link;
263 }
264 tail = link;
265 }
266 }
267 return head;
268} /* remove_flow_list */

270/*
271 * remove a loop;
272 * remove lpx from the LP_CHILD list of its parent.
273 * make the fnodes be simple fall-through fnodes
274 * add the fnodes to the fnode list of the parent
275 */
276static void
277remove_loop(int lpx)
278{
279 int parent, prev, head, tail;
280 parent = LP_PARENT(lpx);
281 if (LP_CHILD(parent) == lpx) {
282 LP_CHILD(parent) = LP_SIBLING(lpx);
283 /* does this make 'parent' an innermost loop? */
284 if (parent && LP_CHILD(parent) == 0) {
285 LP_INNERMOST(parent) = 1;
286 }
287 } else {
288 int lc;
289 for (lc = LP_CHILD(parent); lc && LP_SIBLING(lc) != lpx;
290 lc = LP_SIBLING(lc))
291 ;
292 if (lc && LP_SIBLING(lc) == lpx) {
293 LP_SIBLING(lc) = LP_SIBLING(lpx);
294 }
295 }
296 head = LP_HEAD(lpx);
297 tail = LP_TAIL(lpx);
298 FG_PRED(head) = remove_flow_list(FG_PRED(head), tail);
299 FG_SUCC(tail) = remove_flow_list(FG_SUCC(tail), head);
300
301 for (prev = FG_LPREV(head); prev; prev = FG_LPREV(prev)) {
302 if (FG_LOOP(prev) == parent)
303 break;
304 }
305 if (prev && LP_FG(lpx)) {
306 int next, fg;
307 next = FG_NEXT(prev);
308 FG_NEXT(prev) = LP_FG(lpx);
309 for (fg = LP_FG(lpx); fg && FG_NEXT(fg); fg = FG_NEXT(fg))
310 ;
311 FG_NEXT(fg) = next;
312 }
313 LP_FG(lpx) = 0;
314} /* remove_loop */

326void
327optimize(int whichpass)
328{
329 int lpx;
330 int i;
331
332 optimize_init();
333
334#if DEBUG
335 if (DBGBIT(38, 128)) {
336 optimize_end();
337 return;
338 }
339#endif
340 if (opt.fgb.stg_avail == 1) {
341 optimize_end();
342 return;
343 }
344/* optimize is called for each function */
345
346#if DEBUG
347 if (DBGBIT(10, 2)) {
348 fprintf(gbl.dbgfil, "STDs before optimizer\n");
349 dump_std();
350 }
351#endif
352
353 function_init();
354
355 flowgraph(); /* build the flowgraph for the function */
356#if DEBUG
357 if (DBGBIT(9, 1))
358 dump_flowgraph();
359#endif
360
361 findloop(HLOPT_ALL); /* find the loops */
362
363#if DEBUG
364 if (DBGBIT(9, 4)) {
365 dump_flowgraph();
366 dump_loops();
367 }
368#endif
369
370 if (XBIT(70, 0x2000)) /* enhanced analysis with HCCSYM variables */
371 flg.x[70] |= 0x1000;
372 flow(); /* do flow analysis on the loops */
373 flg.x[70] &= ~0x1000;
374 if (whichpass == 1) {
375 flow_end();
376 optimize_end();
377 return;
378 }
379
380#if DEBUG
381 if (DBGBIT(29, 128))
382 dump_loops();
383#endif
384
385 if (XBIT(0, 0x1000))
386 use_before_def(); /* after flow; check for uses before defs */
387
388 delete_stores(); /* find deleteable stores */
389
390 /*
391 * do optimizations for each loop. also, for each loop, mark
392 * the blocks which are the head and tail of the loop and mark
393 * the head block as innermost if it's the head of an innermost
394 * loop (this information is used by the scheduler).
395 */
396 for (i = 1; i <= opt.nloops; i++) {
397 int headfg;
398
399 lpx = LP_LOOP(i);
400 headfg = LP_HEAD(lpx);
401 gbl.lineno = FG_LINENO(headfg);
402
403 if (LP_PARLOOP(lpx))
404 opt.sc = SC_PRIVATE;
405
406 /* apply loop-scoped pragmas/directives available for this loop */
407 open_pragma(gbl.lineno);
408
409 if (LP_INNERMOST(lpx)) {
410 if (XBIT(70, 0x4000)) {
411 int fg, empty, dostd;
412 /* remove empty loops */
413 empty = 1;
414 dostd = 0;
415 for (fg = LP_FG(lpx); empty && fg; fg = FG_NEXT(fg)) {
416 int std;
417 for (std = FG_STDFIRST(fg); empty && std; std = STD_NEXT(std)) {
418 switch (A_TYPEG(STD_AST(std))) {
419 case A_DO:
420 if (dostd == 0) {
421 dostd = std;
422 } else {
423 empty = 0;
424 }
425 break;
426 case A_ENDDO:
427 case A_CONTINUE:
428 break;
429 default:
430 empty = 0;
431 break;
432 }
433 if (std == FG_STDLAST(fg))
434 break;
435 }
436 }
437 if (empty && dostd) {
438 /* if the DO variable is dead, remove all the statements */
439 int doast, dovar, donme;
440 doast = STD_AST(dostd);
441 dovar = A_SPTRG(A_DOVARG(doast));
442 donme = add_arrnme(NT_VAR, dovar, 0, (INT)0, 0, FALSE);
443 if (!is_live_out(donme, lpx) && no_functions(doast)) {
444 /* remove all statements, change all to continue */
445 for (fg = LP_FG(lpx); fg; fg = FG_NEXT(fg)) {
446 int std;
447 for (std = FG_STDFIRST(fg); std; std = STD_NEXT(std)) {
448 int ast;
449 ast = STD_AST(std);
450 A_TYPEP(ast, A_CONTINUE);
451 if (std == FG_STDLAST(fg))
452 break;
453 }
454 }
455 /* this loop no longer exists.
456 * remove the loop from the LP_CHILD list of its parent.
457 * if the parent has no other children, mark the parent as INNERMOST
458 */
459 remove_loop(lpx);
460 continue;
461 } else {
462 /* insert assignment from lower bound, upper bound, stride */
463 /* insert proper assignment to DO variable */
464 /* ### not done */
465 }
466 }
467 }
468 FG_INNERMOST(headfg) = 1;
469 }
470
471 if (LP_FG(lpx) == 0) {
472 continue;
473 }
474 FG_HEAD(headfg) = 1;
475 FG_MEXITS(headfg) = LP_MEXITS(lpx);
476 FG_TAIL(LP_TAIL(lpx)) = 1;
477 loop_init(lpx);
478
479 invariant(lpx);
480
481 induction(lpx);
482
483 add_loop_preheader(lpx);
484 add_loop_exit(lpx);
485
486 /* set done bit for all nodes in the loop */
487
488 close_pragma();
489
490 opt.sc = SC_AUTO;
491 }
492 if (opt.nloops == 0) {
493 ;
494 } else {
495
496 /* do region 0 */
497
498 opt.pre_fg = NEW_NODE(gbl.entbih);
499 opt.exit_fg = NEW_NODE(FG_LPREV(opt.exitfg));
500
501 if (OPTDBG(9, 1)) {
502 fprintf(gbl.dbgfil, "\n Loop init for region 0\n");
503 fprintf(gbl.dbgfil, " preheader: %d\n", opt.pre_fg);
504 fprintf(gbl.dbgfil, " exit: %d\n", opt.exit_fg);
505 }
506 }
507
508 br_to_br();
509
510 merge_blocks(); /* merge the blocks */
511
512 function_end();
513
514#if DEBUG
515 if (DBGBIT(10, 4)) {
516 fprintf(gbl.dbgfil, "STDs after optimizer\n");
517 dump_std();
518 }
519 if (DBGBIT(10, 64))
520 dmpnme();
521#endif
522
523 flow_end();
524 optimize_end();
525
526}

604/*
605 * After a loop has been processed, the flow graph is updated to include
606 * the preheader. This node is physically placed immediatedly before the
607 * loop head.
608 * The predecessors of head not in the loop are replaced by the preheader
609 * flow graph node. The predecessors of this node are the predecessors
610 * of head not in the loop.
611 * Also, this node replaces head as the successor of all the nodes not
612 * in the loop.
613 * If there is a path to the head node by branching (the node does not
614 * fall thru to the head), the label of head is replaced with a new label
615 * and all the branches to the head are modified. The original label
616 * is used to label the preheader.
617 * Also, the preheader is added to the region containing the loop.
618 */
619void
620add_loop_preheader(int lp)
621{
622 int i, v;
623 PSI_P p, q, prev;
624 int newlabel;
625
626 int head;
627
628 if (FG_STDFIRST(opt.pre_fg) == 0)
629 return;
630 if (OPTDBG(9, 1))
631 fprintf(gbl.dbgfil, "\n Preheader end for loop (%d)\n", lp);
632
633 /* Every node dominated by lp's dominator will be dominated by the
634 * new preheader. */
635 head = LP_HEAD(lp); /* loop's head */
636 i = FG_DOM(head); /* dominator of loop's head */
637 for (p = FG_SUCC(i); p != PSI_P_NULL; p = PSI_NEXT(p)) {
638 v = PSI_NODE(p);
639 if (FG_DOM(v) == i)
640 FG_DOM(v) = opt.pre_fg;
641 }
642 FG_DOM(opt.pre_fg) = i;
643 FG_DOM(head) = opt.pre_fg;
644
645 i = 0; /* number of predecessors not in the loop */
646 newlabel = 0; /* non-zero if a pred does fall thru to the head */
647 prev = PSI_P_NULL; /* the last predecessor to head that's kept */
648 for (p = FG_PRED(head); p != PSI_P_NULL; p = PSI_NEXT(p)) {
649 v = PSI_NODE(p);
650 if (FG_LOOP(v) != lp) { /* v in pred(head) and not in the loop */
651 i++;
652 if (FG_LNEXT(v) != opt.pre_fg) {
653 /* if pred. doesn't fall thru */
654 newlabel = 1;
655 }
656 /* search v's successor list for the item indicating head */
657
658 for (q = FG_SUCC(v); q != PSI_P_NULL; q = PSI_NEXT(q))
659 if (head == PSI_NODE(q))
660 break;
661 assert(q != PSI_P_NULL, "add_loop_pre:suc(q)nhd", head, 3);
662
663 /* Replace head in succ(v) with the preheader node */
664
665 PSI_NODE(q) = opt.pre_fg;
666 /*
667 * Replace v in pred(head) with the preheader node only if this
668 * is the first predecessor found which is not in the loop.
669 * Otherwise, this predecessor item (p) is deleted.
670 */
671 if (i == 1) {
672 PSI_NODE(p) = opt.pre_fg;
673 prev = p;
674 } else {
675 assert(prev != PSI_P_NULL, "add_loop_pre:prev", head, 3);
676 PSI_NEXT(prev) = PSI_NEXT(p);
677 }
678 /*
679 * make v a predecessor of the preheader and make head a
680 * successor of the preheader
681 */
682 (void)add_pred(opt.pre_fg, v);
683 q = add_succ(opt.pre_fg, head);
684 PSI_FT(q) = 1;
685#if DEBUG
686 if (OPTDBG(9, 1)) {
687 fprintf(gbl.dbgfil, " old pred of head :\n");
688 dump_node(v);
689 }
690#endif
691 } else /* v not in pred(head), remember this predecessor item */
692 prev = p;
693 }
694
695 if (newlabel) { /* have to create a new label for the loop head */
696 int label, newlab;
697
698 label = STD_LABEL(FG_STDFIRST(head)); /* original label */
699 newlab = getlab();
700 ILIBLKP(newlab, head); /* label the head with the new label */
701 STD_LABEL(FG_STDFIRST(head)) = newlab;
702
703 ILIBLKP(label, opt.pre_fg); /* label the preheader with the */
704 STD_LABEL(FG_STDFIRST(opt.pre_fg)) = label; /* original label */
705
706 /* find all nodes in the loop which are predecessors of head.
707 * the branches in these nodes are changed to access the new
708 * label
709 */
710 for (p = FG_PRED(head); p != PSI_P_NULL; p = PSI_NEXT(p)) {
711 v = PSI_NODE(p);
712 if (FG_LOOP(v) == lp)
713 replace_label((int)FG_TO_BIH(v), label, newlab);
714 }
715 }
716 /*
717 * set the loop field of the preheader to an ancestor of lp which
718 * contains blocks -- a loop could have been deleted (indicated by a null
719 * region)
720 */
721 for (i = lp; LP_FG(i = LP_PARENT(i)) == 0;)
722 ;
723
724 FG_LOOP(opt.pre_fg) = i;
725
726 /* add the preheader to the region for parent of lp */
727
728 FG_NEXT(opt.pre_fg) = LP_FG(i);
729 LP_FG(i) = opt.pre_fg;
730
731 /* link the preheader's stds into the std list */
732
733 {
734 int new;
735 int old;
736 int p;
737
738 /* search for the last std preceding the preheader */
739 for (p = FG_LPREV(opt.pre_fg); p; p = FG_LPREV(p))
740 if (FG_STDLAST(p))
741 break;
742 new = FG_STDFIRST(opt.pre_fg);
743 old = FG_STDLAST(p);
744 STD_NEXT(old) = new;
745 STD_PREV(new) = old;
746
747 /* search for the first std following the preheader */
748 for (p = FG_LNEXT(opt.pre_fg); p; p = FG_LNEXT(p))
749 if (FG_STDFIRST(p))
750 break;
751 new = FG_STDLAST(opt.pre_fg);
752 old = FG_STDFIRST(p);
753 STD_NEXT(new) = old;
754 STD_PREV(old) = new;
755 }
756
757#if DEBUG
758 if (OPTDBG(9, 1)) {
759 fprintf(gbl.dbgfil, " preheader:\n");
760 dump_node(opt.pre_fg);
761 fprintf(gbl.dbgfil, " head:\n");
762 dump_node(head);
763 }
764 if (OPTDBG(9, 4)) {
765 fprintf(gbl.dbgfil, "\n Region of preheader node %d\n", opt.pre_fg);
766 dump_region(i);
767 }
768#endif
769}

788/* Add a loop exit following the tail of loop lp. */
789void
790add_single_loop_exit(int lp)
791{
792 int fg, fgSucc, fgNew;
793 PSI_P psiSucc;
794 int lpSucc;
795 int sptrLbl, sptrNewLbl;

797 int ast;

800 assert(!LP_MEXITS(lp), "add_loop_exit: multiple exits", lp, 4);
801

803 add_to_parent(fgNew, LP_PARENT(lp));
804
805 for (fg = LP_FG(lp); fg; fg = FG_NEXT(fg)) {
806 /* Search for a node with a branch to the exit. */
807 for (psiSucc = FG_SUCC(fg); psiSucc; psiSucc = PSI_NEXT(psiSucc)) {
808 fgSucc = PSI_NODE(psiSucc);
809 lpSucc = FG_LOOP(fgSucc);
810 if (!lpSucc)
811 break;
812 if (lpSucc != lp && LP_PARENT(lpSucc) != lp)
813 break;
814 }
815 if (psiSucc)
816 /* ...node with exit branch found. */
817 break;
818 }
819 assert(fg, "add_loop_exit: branch not found", lp, 4);
820 FG_FT(fgNew) = PSI_FT(psiSucc);
821
822 /* Add a CONTINUE statement to the new loop trailer. */
823 ast = mk_stmt(A_CONTINUE, 0);
824 rdilts(fgNew);
825 std = add_stmt_after(ast, 0);
826 wrilts(fgNew);
827 FG_LINENO(fgNew) = STD_LINENO(std) = FG_LINENO(LP_TAIL(lp));
828
829 sptrLbl = FG_LABEL(fgSucc);
830 if (!PSI_FT(psiSucc) && sptrLbl) {
831 int astlab;
832 /* Add a GOTO statement to the new loop trailer. */
833 ast = mk_stmt(A_GOTO, 0);
834 astlab = mk_label(sptrLbl);
835 A_L1P(ast, astlab);
836 std = add_stmt_after(ast, std);
837
838 /* Create a new label for fgNew. */
839 sptrNewLbl = getlab();
840 ILIBLKP(sptrNewLbl, fgNew);
841 RFCNTI(sptrNewLbl);
842 FG_LABEL(fgNew) = sptrNewLbl;
843
844 /* Revise the branch to go to the new loop trailer. */
845 replace_label(FG_TO_BIH(fg), sptrLbl, sptrNewLbl);
846 }
847
848 /* Fix up the pred./succ. chains. */
849 add_succ(fg, fgNew); /* succ(fg) = fgNew. */
850 add_pred(fgNew, fg); /* pred(fgNew) = fg. */
851 add_succ(fgNew, fgSucc); /* succ(fgNew) = fgSucc. */
852 add_pred(fgSucc, fgNew); /* pred(fgSucc) = fgNew. */
853 rm_edge(fg, fgSucc);
854}

936/*
937 * look for conditional branches with constant conditions.
938 * remove the branch, remove unreachable code as well.
939 */
940void
941unconditional_branches(void)
942{
943 int stdx, nextstdx;
944 int astx, nest = 0, stdifx, stdstmtx, sptr, stdelsex, stdendx, astelsex, astendx;
945 int condx;
946 int *ifnest;
947 int ifnestsize;
948 ifnestsize = 50;
949 NEW(ifnest, int, ifnestsize);
950 /* initially, set STD_FG for A_IFTHEN, A_ELSEIF, A_ELSE to the
951 * matching following A_ELSEIF, A_ELSE, A_ENDIF, as appropriate */
952 for (stdx = STD_NEXT(0); stdx; stdx = STD_NEXT(stdx)) {
953 astx = STD_AST(stdx);
954 switch (A_TYPEG(astx)) {
955 case A_IFTHEN:
956 ++nest;
957 NEED(nest, ifnest, int, ifnestsize, ifnestsize + 50);
958 ifnest[nest] = stdx;
959 break;
960 case A_ELSEIF:
961 case A_ELSE:
962 if (nest <= 0) {
963 /* bad nesting */
964 return;
965 }
966 stdifx = ifnest[nest];
967 STD_FG(stdifx) = stdx;
968 ifnest[nest] = stdx;
969 break;
970 case A_ENDIF:
971 if (nest <= 0) {
972 /* bad nesting */
973 FREE(ifnest);
974 return;
975 }
976 stdifx = ifnest[nest];
977 STD_FG(stdifx) = stdx;
978 --nest;
979 break;
980 }
981 }
982 FREE(ifnest);
983 for (stdx = STD_NEXT(0); stdx; stdx = nextstdx) {
984 nextstdx = STD_NEXT(stdx);
985 astx = STD_AST(stdx);
986 switch (A_TYPEG(astx)) {
987 case A_IF:
988 /* if condition is .FALSE., remove if and its statement */
989 /* if condition is .TRUE., replace condition by the statement */
990 condx = A_IFEXPRG(astx);
991 if (A_ALIASG(condx))
992 condx = A_ALIASG(condx);
993 if (A_TYPEG(condx) == A_CNST && (sptr = A_SPTRG(condx)) > 0 &&
994 DT_ISLOG(DTYPEG(sptr))) {
995 if (CONVAL2G(sptr) == 0) {
996 /* .false. */
997 remove_block(stdx, stdx, FALSE, FALSE);
998 } else {
999 /* .true. */
1000 stdstmtx = A_IFSTMTG(astx);
1001 STD_AST(stdx) = stdstmtx;
1002 }
1003 }
1004 break;
1005 case A_IFTHEN:
1006 case A_ELSEIF:
1007 /* if condition is .FALSE., remove if and all statements up
1008 * to the ENDIF, ELSEIF, ELSE (change ELSEIF to IF, ELSE/ENDIF to
1009 * CONTINUE)
1010 * if the condition is .TRUE., remove this statement, and remove
1011 * all statements from the ELSEIF/ELSE up to and including the ENDIF */
1012 condx = A_IFEXPRG(astx);
1013 if (A_ALIASG(condx))
1014 condx = A_ALIASG(condx);
1015 if (A_TYPEG(condx) == A_CNST && (sptr = A_SPTRG(condx)) > 0 &&
1016 DT_ISLOG(DTYPEG(sptr))) {
1017 stdelsex = STD_FG(stdx);
1018 if (CONVAL2G(sptr) == 0) {
1019 /* .false. */
1020 /* remove_block changes the ELSEIF to IF, if necessary */
1021 astelsex = STD_AST(stdelsex);
1022 if (A_TYPEG(astelsex) == A_ELSEIF) {
1023 remove_block(stdx, stdelsex, TRUE, FALSE);
1024 } else {
1025 stdendx = STD_FG(stdelsex);
1026 astendx = STD_AST(stdendx);
1027 if (A_TYPEG(astendx) == A_ENDIF) {
1028 remove_block(stdx, stdelsex, TRUE, FALSE);
1029 remove_block(stdendx, stdendx, FALSE, FALSE);
1030 }
1031 }
1032 } else {
1033 /* .true. */
1034 astelsex = STD_AST(stdelsex);
1035 if (A_TYPEG(astelsex) == A_ENDIF) {
1036 /* simply change if/endif to continue/continue
1037 * or else/endif */
1038 if (A_TYPEG(astx) == A_IF) {
1039 remove_block(stdx, stdx, FALSE, FALSE);
1040 remove_block(stdelsex, stdelsex, FALSE, FALSE);
1041 } else {
1042 A_TYPEG(astx) = A_ELSE;
1043 }
1044 } else if (A_TYPEG(astelsex) == A_ELSE ||
1045 A_TYPEG(astelsex) == A_ELSEIF) {
1046 /* simply change if to continue or elseif to else,
1047 * remove elseif through endif */
1048 astendx = 0;
1049 for (stdendx = STD_FG(stdelsex); STD_FG(stdendx);
1050 stdendx = STD_FG(stdendx)) {
1051 astendx = STD_AST(stdendx);
1052 if (A_TYPEG(astendx) == A_ENDIF)
1053 break;
1054 }
1055 if (stdendx)
1056 astendx = STD_AST(stdendx);
1057 if (stdendx && astendx && A_TYPEG(astendx) == A_ENDIF) {
1058 if (A_TYPEG(astx) == A_IFTHEN) {
1059 remove_block(stdx, stdx, FALSE, FALSE);
1060 remove_block(stdelsex, stdendx, FALSE, FALSE);
1061 } else {
1062 A_TYPEG(astx) = A_ELSE;
1063 remove_block(stdelsex, stdendx, FALSE, TRUE);
1064 }
1065 }
1066 }
1067 }
1068 }
1069 break;
1070 }
1071 }
1072 /* clear STD_FG field */
1073 for (stdx = STD_NEXT(0); stdx; stdx = STD_NEXT(stdx)) {
1074 STD_FG(stdx) = 0;
1075 }
1076} /* unconditional_branches */
