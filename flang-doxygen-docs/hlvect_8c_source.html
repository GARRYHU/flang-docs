291int
292hlv_getvsym(int baseDtype)
293{
294 int sptr = getsymf("pgf_%s%04d", hlv_vtemps.pfx, hlv_vtemps.iavl++);
295 int dtype;
296 ADSC *ad;
297
298 if (hlv_vtemps.iavl > hlv_vtemps.iavl_max)
299 hlv_vtemps.iavl_max = hlv_vtemps.iavl;
300 STYPEP(sptr, ST_ARRAY);
301 dtype = get_array_dtype(1, baseDtype);
302 DCLDP(sptr, TRUE);
303 DTYPEP(sptr, dtype);
304 ad = AD_DPTR(dtype);
305 AD_NUMDIM(ad) = 1;
306 AD_LWBD(ad, 0) = AD_LWAST(ad, 0) = astb.i0;
307 AD_UPBD(ad, 0) = AD_UPAST(ad, 0) = AD_EXTNTAST(ad, 0) =
308 mk_cval(STRIPSIZE - 1, DT_INT);
309 AD_MLPYR(ad, 0) = astb.i1;
310 AD_NUMELM(ad) = mk_cval(STRIPSIZE, DT_INT);
311
312 /* needs to be static for VPU */
313 /* switchable? */
314 SCP(sptr, SC_STATIC);
315 return sptr;
316}

318/* Add bihx to the list of blocks that need to be deleted at
319 * the end of processing by hlvect. */
320void
321vdel_bih(int bihx)
322{
323 /* Use the BIH_RGSET field to construct a singly-linked list of
324 * deleted blocks. */
325 BIH_RGSET(bihx) = vdel_lst;
326 vdel_lst = bihx;
327}
328
329#ifdef FLANG_HLVECT_UNUSED
330/* Delete all bih's within the vdel_lst list, as well as all ilt's within
331 * those bih's. */
332static void
333vunlnk_bih(void)
334{
335 int bihx, nextbih;
336
337 for (bihx = vdel_lst; bihx; bihx = nextbih) {
338 nextbih = BIH_RGSET(bihx);
339 BIH_RGSET(bihx) = 0;
340 rdilts(bihx);
341 while (STD_NEXT(0) != 0)
342 delilt(STD_NEXT(0));
343 wrilts(bihx);
344 delbih(bihx);
345 }
346}
347#endif
