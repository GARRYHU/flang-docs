23static void
24list_ln(const char *beg, const char *txt)
25{
26 if (pgpos == 1 && !DBGBIT(14, 3)) {
27 if (!lf)
28 return; /* in case error msg written before file
29 * opened */
30 fprintf(lf, "\n\n\n%s(Version %8s) %s page %d\n\n",
31 version.product, version.vsn, gbl.datetime, pgno);
32 pgno++;
33 pgpos = 6;
34 }
35
36 fprintf(lf, "%s%s\n", beg, txt);
37#if DEBUG
38 if (DBGBIT(0, 4))
39 fprintf(gbl.dbgfil, "%s%s\n", beg, txt);
40#endif
41 pgpos++;
42
43 if (pgpos == LPP + 4 && !DBGBIT(14, 3)) {
44 fprintf(lf, "\n\n\n");
45 pgpos = 1;
46 }
47}
