125/*
126 * decoder, line at a time
127 */
128char *
129ulz(lzhandle *lzh)
130{
131#ifndef USE_GETLINE
132 int ch;
133#endif
134 lzh->bufflen = 0;
135#ifdef USE_GETLINE
136 int res = getline(&lzh->buff, &lzh->buffsize, lzh->file);
137 if (res > 0) {
138 if (lzh->buff[res-1] == '\n')
139 --res;
140 lzh->bufflen = res;
141 }
142#else
143 /* read in chars one by one to endline or EOF */
144 while ((ch = getc(lzh->file)) != '\n' && ch != EOF) {
145 if (lzh->bufflen + 3 >= lzh->buffsize) {
146 lzh->buffsize = lzh->buffsize * 2;
147 lzh->buff = realloc(lzh->buff, lzh->buffsize);
148 if (lzh->buff == NULL) {
149 fprintf(stderr, "Ran out of memory\n");
150 exit(1);
151 }
152 }
153 lzh->buff[lzh->bufflen++] = ch;
154 }
155#endif
156 lzh->buff[lzh->bufflen] = '\0';
157#ifdef ZDEBUGLZ
158 printf("ULZ:%d %s\n", lzh->compress, lzh->buff);
159#endif
160 return lzh->buff;
161} /* ulz */
