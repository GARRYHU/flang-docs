8/*
9 * Common set of interface routines to convert an intrinsic math library call
10 * using Intel AVX-512 vectors in to two calls of the corresponding AVX2
11 * implementation.
12 *
13 * Note: code is common to both AVX-512 and KNL architectures.
14 * Thus, have to use Intel intrinsics that are common to both systems.
15 */

25 rl = func((vrs8_t) _mm512_castps512_ps256(x));
26 return (vrs16_t) _mm512_insertf64x4(_mm512_castpd256_pd512((__m256d)rl),
27 (__m256d)ru, 1);
28}
