88char *
89__fortio_default_convert(char *item, int type,
90 int item_length, /* only if character */
91 int *lenp, bool dc_flag, bool plus_flag, int round)
92{
93 int width;
94 char *p;
95 DBLINT64 i8val;
96
97 switch (type) {
98 default:
99 assert(0);
100 width = 1;
101 strcpy(conv_bufp, " ");
102 break;
103 case __INT1:
104 width = 5;
105 (void) __fortio_fmt_i((__BIGINT_T)PP_INT1(item), width, 1, plus_flag);
106 break;
107 case __INT2:
108 width = 7;
109 (void) __fortio_fmt_i((__BIGINT_T)PP_INT2(item), width, 1, plus_flag);
110 break;
111 case __INT8:
112 width = 24;
113 (void) __fortio_fmt_i8(*(DBLINT64 *)(item), width, 1, plus_flag);
114 break;
115 case __WORD4:
116 width = 8;
117 buff_pos = conv_bufp + 8 - 1; /* start at right end of buffer */
118 (void) __fortio_fmt_z(*item);
119 (void) __fortio_fmt_z(*(item + 1));
120 (void) __fortio_fmt_z(*(item + 2));
121 (void) __fortio_fmt_z(*(item + 3));
122 break;
123 case __INT4:
124 width = 12;
125 (void) __fortio_fmt_i((__BIGINT_T)PP_INT4(item), width, 1, plus_flag);
126 break;
127 case __REAL4:
128 width = REAL4_W;
129 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL4(item), width, REAL4_D, REAL4_E,
130 1, __REAL4, plus_flag, TRUE, dc_flag, round, FALSE);
131 break;
132 case __WORD8:
133 width = 16;
134 buff_pos = conv_bufp + 16 - 1; /* start at right end of buffer */
135 (void) __fortio_fmt_z(*item);
136 (void) __fortio_fmt_z(*(item + 1));
137 (void) __fortio_fmt_z(*(item + 2));
138 (void) __fortio_fmt_z(*(item + 3));
139 (void) __fortio_fmt_z(*(item + 4));
140 (void) __fortio_fmt_z(*(item + 5));
141 (void) __fortio_fmt_z(*(item + 6));
142 (void) __fortio_fmt_z(*(item + 7));
143 break;
144 case __REAL8:
145 width = REAL8_W;
146 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL8(item), width, REAL8_D, REAL8_E,
147 1, __REAL8, plus_flag, TRUE, dc_flag, round, FALSE);
148 break;
149 case __REAL16:
150 width = G_REAL16_W;
151 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL16(item), width, G_REAL16_D,
152 REAL16_E, 1, __REAL16, plus_flag, TRUE, dc_flag, round,
153 TRUE);
154 break;
155 case __WORD16:
156 assert(0);
157 width = 0;
158 strcpy(conv_bufp, "");
159 break;
160 case __CPLX8:
161 p = cmplx_buf;
162 *p++ = '(';
163 width = REAL4_W;
164 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL4(item), width, REAL4_D, REAL4_E,
165 1, __REAL4, plus_flag, TRUE, dc_flag, round, FALSE);
166 p = strip_blnk(p, conv_bufp);
167 if (dc_flag == TRUE)
168 *p++ = ';';
169 else
170 *p++ = ',';
171 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL4(item + 4), width, REAL4_D,
172 REAL4_E, 1, __REAL4, plus_flag, TRUE, dc_flag, round,
173 FALSE);
174 p = strip_blnk(p, conv_bufp);
175 *p++ = ')';
176 *p++ = '\0';
177 *lenp = strlen(cmplx_buf);
178 return cmplx_buf;
179 case __CPLX16:
180 p = cmplx_buf;
181 *p++ = '(';
182 width = REAL8_W;
183 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL8(item), width, REAL8_D, REAL8_E,
184 1, __REAL8, plus_flag, TRUE, dc_flag, round, FALSE);
185 p = strip_blnk(p, conv_bufp);
186 if (dc_flag == TRUE)
187 *p++ = ';';
188 else
189 *p++ = ',';
190 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL8(item + 8), width, REAL8_D,
191 REAL8_E, 1, __REAL8, plus_flag, TRUE, dc_flag, round,
192 FALSE);
193 p = strip_blnk(p, conv_bufp);
194 *p++ = ')';
195 *p++ = '\0';
196 *lenp = strlen(cmplx_buf);
197 return cmplx_buf;
198 case __CPLX32:
199 p = cmplx_buf;
200 *p++ = '(';
201 width = G_REAL16_W;
202 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL16(item), width, G_REAL16_D,
203 REAL16_E, 1, __REAL16, plus_flag, TRUE, dc_flag, round,
204 TRUE);
205 p = strip_blnk(p, conv_bufp);
206 if (dc_flag == TRUE)
207 *p++ = ';';
208 else
209 *p++ = ',';
210 (void)__fortio_fmt_g((__BIGREAL_T)PP_REAL16(item + 16), width, G_REAL16_D,
211 REAL16_E, 1, __REAL16, plus_flag, TRUE, dc_flag, round,
212 TRUE);
213 p = strip_blnk(p, conv_bufp);
214 *p++ = ')';
215 *p++ = '\0';
216 *lenp = strlen(cmplx_buf);
217 return cmplx_buf;
218 case __LOG1:
219 width = 2;
220 if (PP_LOG1(item) & GET_FIO_CNFG_TRUE_MASK)
221 put_buf(width, "T", 1, 0);
222 else
223 put_buf(width, "F", 1, 0);
224 break;
225 case __LOG2:
226 width = 2;
227 if (PP_LOG2(item) & GET_FIO_CNFG_TRUE_MASK)
228 put_buf(width, "T", 1, 0);
229 else
230 put_buf(width, "F", 1, 0);
231 break;
232 case __LOG4:
233 width = 2;
234 if (PP_LOG4(item) & GET_FIO_CNFG_TRUE_MASK)
235 put_buf(width, "T", 1, 0);
236 else
237 put_buf(width, "F", 1, 0);
238 break;
239 case __LOG8:
240 width = 2;
241 i8val[0] = PP_LOG4(item);
242 i8val[1] = PP_LOG4(item + 4);
243 if (I64_LSH(i8val) & GET_FIO_CNFG_TRUE_MASK)
244 put_buf(width, "T", 1, 0);
245 else
246 put_buf(width, "F", 1, 0);
247 break;
248 case __STR:
249 if (item_length >= conv_bufsize) {
250 conv_bufsize = item_length + 128;
251 if (conv_bufp != __f90io_conv_buf)
252 free(conv_bufp);
253 conv_bufp = (char *)malloc(conv_bufsize);
254 }
255 (void) memcpy(conv_bufp, item, item_length);
256 conv_bufp[item_length] = '\0';
257 width = item_length;
258 break;
259 } /* end switch */
260
261 *lenp = width;
262 return conv_bufp;
263}

265char *
266__fortio_fmt_i(__BIGINT_T val, int width,
267 int mn, /* minimum # of digits (Iw.m) */
268 bool plus_flag)
269{
270 char *p;
271 int len;
272 int neg; /* flag word set by conv_int, becomes sign character */
273 int olen; /* output length of integer value */
274
275 field_overflow = FALSE;
276 p = conv_int(val, &len, &neg);
277
278 if (neg)
279 neg = '-';
280 else if (plus_flag)
281 neg = '+';
282 olen = len >= mn ? len : mn;
283 if (neg)
284 olen++;
285 if (olen > width) {
286 field_overflow = TRUE;
287 put_buf(width, p, len, neg);
288 } else {
289 if (mn == 0 && val == 0) /* Iw.0 gen's blanks if value is 0 */
290 neg = 0;
291 put_buf(width, p, len, neg);
292 if (mn > len) {
293 int i;
294 i = mn - len;
295 p = conv_bufp + (width - len);
296 while (i-- > 0)
297 *--p = '0';
298 if (neg)
299 *--p = neg;
300 }
301 }
302
303 return conv_bufp;
304}

306static char *
307conv_int(__BIGINT_T val, int *lenp, int *negp)
308{
309#define MAX_CONV_INT 32
310#define MAX_HX 0x80000000
311#define MAX_STR "2147483648"
312
313 static char tmp[MAX_CONV_INT];
314 char *p;
315 int len;
316 int neg;
317 unsigned int n;
318
319 if (val < 0) {
320 if (val == MAX_HX) {
321 *lenp = 10;
322 *negp = 1;
323 strcpy(tmp, MAX_STR);
324 return tmp;
325 }
326 neg = 1;
327 val = -val;
328 } else
329 neg = 0;
330
331 p = tmp + MAX_CONV_INT;
332 len = 0;
333 while (val > 0) {
334 n = val / 10;
335 *--p = (val - (n * 10)) + '0';
336 val = n;
337 len++;
338 }
339
340 *lenp = len;
341 *negp = neg;
342 return p;
343}

345char *
346__fortio_fmt_i8(DBLINT64 val,
347 int width,
348 int mn, /* minimum # of digits (Iw.m) */
349 bool plus_flag)
350{
351 char *p;
352 int len;
353 int neg; /* flag word set by conv_int, becomes sign character */
354 int olen; /* output length of integer value */
355
356 field_overflow = FALSE;
357 p = conv_int8(val, &len, &neg);
358
359 if (neg)
360 neg = '-';
361 else if (plus_flag)
362 neg = '+';
363 olen = len >= mn ? len : mn;
364 if (neg)
365 olen++;
366 if (olen > width) {
367 field_overflow = TRUE;
368 put_buf(width, p, len, neg);
369 } else {
370 /* Iw.0 gen's blanks if value is 0 */
371 if (mn == 0 && val[0] == 0 && val[1] == 0)
372 neg = 0;
373 put_buf(width, p, len, neg);
374 if (mn > len) {
375 int i;
376 i = mn - len;
377 p = conv_bufp + (width - len);
378 while (i--)
379 *--p = '0';
380 if (neg)
381 *--p = neg;
382 }
383 }
384
385 return conv_bufp;
386}

388static char *
389conv_int8(DBLINT64 val, int *lenp, int *negp)
390{
391#define MAX_CONV_INT8 32
392
393 static char tmp[MAX_CONV_INT8];
394 char *p;
395 int len;
396 DBLINT64 value;
397
398 *negp = 0;
399 value[0] = val[0];
400 value[1] = val[1];
401 if (__ftn_32in64_) {
402 if (I64_LSH(value) & 0x80000000)
403 I64_MSH(value) = -1;
404 else
405 I64_MSH(value) = 0;
406 } else if (I64_MSH(value) < 0) {
407 if (I64_MSH(value) == 0x80000000 && I64_LSH(value) == 0) {
408 *lenp = 19;
409 *negp = 1;
410 strcpy(tmp, "9223372036854775808");
411 return tmp;
412 }
413 *negp = 1;
414 /* now negate the value */
415 I64_MSH(value) = ~I64_MSH(value);
416 I64_LSH(value) = (~I64_LSH(value)) + 1;
417 if ((((unsigned)I64_LSH(val) >> 31) == 0) && I64_LSH(value) >= 0)
418 I64_MSH(value)++;
419 }
420
421 p = tmp;
422 len = 24;
423 __fort_i64toax(value, p, len, 0, 10);
424
425 *lenp = strlen(p);
426 return p;
427}

429static char fpbuf[64];
430static struct {
431 int exp; /* initially set by ecvt/fcvt. adjusted by the
432 * scale factor. WARNING: may be set to zero if
433 * value to be printed represents 0.
434 */
435 int sign; /* non-zero if value is negative (initially set by
436 * ecvt/fcvt). WARNING: conv_e/conv_f may set to
437 * zero if value to be printed represents zero.
438 */
439 int ndigits;
440 int decimal_char;
441 bool iszero; /* TRUE if after conversions with respect to the
442 * edit descriptor, the string to be printed represents
443 * zero. conv_e/conv_f initializes to TRUE; cvtp_cp
444 * sets FALSE.
445 */
446 char *cvtp;
447 char *curp;
448 char *buf;
449 int bufsize;
450 __BIGREAL_T zero; /* hide 0.0 from the optimizer here */
451} fpdat = {0, 0, 0, '.', 0, 0, 0, fpbuf, sizeof(fpbuf), 0.0};
452
453static void put_buf(int width, /* where width (# bytes) */
454 const char *valp, /* value in string form */
455 int len, /* length of value */
456 int sign_char) /* '-', '+', or 0 (nothing to prepend) */
457{
458 char *bufp;
459 int cnt;
460 int sign_cnt;
461
462 if (DBGBIT(0x1))
463 __io_printf("put_buf: width=%d, len=%d, val=%.*s#, sign_char=%d\n", width,
464 len, len, valp, sign_char);
465 if (width >= conv_bufsize) {
466 conv_bufsize = width + 128;
467 if (conv_bufp != __f90io_conv_buf)
468 free(conv_bufp);
469 conv_bufp = (char *)malloc(conv_bufsize);
470 }
471 bufp = conv_bufp;
472 if (width == 0) {
473 /* short circuit for if no transfer - do we need error message */
474 *bufp = '\0';
475 return;
476 }
477 if (field_overflow)
478 goto fill_asterisks;
479 sign_cnt = sign_char != 0;
480 if (len + sign_cnt > width) {
481 /*
482 * special floating point case: if value begins with "0.",
483 * try ignoring the '0'.
484 */
485 if (*valp != '0' || *(valp + 1) != fpdat.decimal_char ||
486 (len + sign_cnt - 1) > width)
487 goto fill_asterisks;
488 len--;
489 valp++;
490 }
491 cnt = width - len - sign_cnt;
492 while (cnt-- > 0)
493 *bufp++ = ' ';
494 if (sign_char)
495 *bufp++ = sign_char;
496 cnt = len;
497 while (cnt-- > 0)
498 *bufp++ = *valp++;
499 *bufp = '\0';
500 return;
501
502fill_asterisks:
503 (void) memset(bufp, '*', width);
504 bufp += width;
505 *bufp = '\0';
506 field_overflow = FALSE;
507}

511extern char *
512__fortio_fmt_d(__BIGREAL_T val, int w, int d, int sf, int type, bool plus_flag,
513 int round)
514{
515 int sign_char;
516 int newd;
517
518 exp_letter = 'D';
519 field_overflow = FALSE;
520 /*
521 fp_canon(val, type, round);
522 */
523 if ((sf < 0) && (-d) >= sf) {
524 field_overflow = TRUE;
525 put_buf(w, (char *)0, 0, 0);
526 exp_letter = 'E'; /* Make sure to change this back! */
527 return conv_bufp;
528 }
529 newd = d + ((sf > 0) ? 1 : sf);
530 fpdat.cvtp = __io_ecvt(val, w, newd, &fpdat.exp, &fpdat.sign, round, FALSE);
531 fpdat.ndigits = strlen(fpdat.cvtp);
532 fpdat.curp = fpdat.buf;
533
534 if (DBGBIT(0x4)) {
535 __io_printf("fio_fmt_d ");
536 __fortio_printbigreal(val);
537 __io_printf(" = #%s#, exp=%d, sign=%d, sf=%d, pf=%d\n", fpdat.cvtp,
538 fpdat.exp, fpdat.sign, sf, plus_flag);
539 __io_printf(" w=%d, d=%d\n", w, d);
540 }
541 if (*fpdat.cvtp < '0' || *fpdat.cvtp > '9') {
542 if (fpdat.sign)
543 sign_char = '-';
544 else if (plus_flag)
545 sign_char = '+';
546 else
547 sign_char = 0;
548 put_buf(w, fpdat.cvtp, fpdat.ndigits, sign_char);
549 } else {
550 conv_e(d, 2, sf, FALSE, FALSE);
551 if (fpdat.sign) /* must check after conv_e */
552 sign_char = '-';
553 else if (plus_flag)
554 sign_char = '+';
555 else
556 sign_char = 0;
557 put_buf(w, fpdat.buf, (int)(fpdat.curp - fpdat.buf), sign_char);
558 }
559 exp_letter = 'E';
560 return conv_bufp;
561}

563extern char *
564__fortio_fmt_g(__BIGREAL_T val, int w, int d, int e, int sf, int type,
565 bool plus_flag, bool e_flag, bool dc_flag, int round,
566 int is_quad) /* TRUE, if the value is quad precision. */
567{
568 int sign_char;
569 int newd;
570#if defined(TARGET_X8664)
571 /*
572 * the following guarded IF may look like a no-op, but is
573 * needed when val is a denorm and DAZ is enabled. In this case, the
574 * comparison will say val is identical to 0, but the bits of val will
575 * indicate otherwise and the ensuing code may go down the wrong path.
576 */
577 if (val == fpdat.zero && !is_quad) {
578 union {
579 __BIGREAL_T vv;
580 int ii[2];
581 } u;
582 u.vv = val;
583 val = fpdat.zero;
584 if (u.ii[1] < 0) {
585 ((int *)&val)[1] |= 0x80000000;
586 }
587 }
588#endif
589 field_overflow = FALSE;
590 /*
591 fp_canon(val, type, round);
592 */
593 if ((sf < 0) && (-d) >= sf) {
594 field_overflow = TRUE;
595 put_buf(w, (char *)0, 0, 0);
596 return conv_bufp;
597 }
598 if (!is_quad)
599 newd = d + ((sf > 0) ? 1 : sf);
600 else
601 newd = d + ((sf > 0) ? 0 : sf - 1);
602 fpdat.cvtp = __io_ecvt(val, w, newd, &fpdat.exp, &fpdat.sign, round, is_quad);
603 fpdat.ndigits = strlen(fpdat.cvtp);
604 fpdat.curp = fpdat.buf;
605
606 if (DBGBIT(0x4)) {
607 __io_printf("fio_fmt_g ");
608 __fortio_printbigreal(val);
609 __io_printf(" = #%s#, exp=%d, sign=%d, sf=%d, pf=%d, ef=%d\n", fpdat.cvtp,
610 fpdat.exp, fpdat.sign, sf, plus_flag, e_flag);
611 __io_printf(" w=%d, d=%d\n", w, d);
612 }
613 if (dc_flag == TRUE)
614 fpdat.decimal_char = ',';
615 else
616 fpdat.decimal_char = '.';
617 if (*fpdat.cvtp < '0' || *fpdat.cvtp > '9') {
618 if (fpdat.sign)
619 sign_char = '-';
620 else if (plus_flag)
621 sign_char = '+';
622 else
623 sign_char = 0;
624 put_buf(w, fpdat.cvtp, fpdat.ndigits, sign_char);
625 } else if ((val != (__BIGREAL_T)0.0) &&
626 (*fpdat.cvtp == '0' || fpdat.exp < 0 || fpdat.exp >= d + 1)) {
627 /* m .lt. 0.1 or m .ge. 10**d */
628 /* Ew.dEe */
629 conv_e(d, e, sf, e_flag, is_quad);
630 if (fpdat.sign) /* must check after conv_e */
631 sign_char = '-';
632 else if (plus_flag)
633 sign_char = '+';
634 else
635 sign_char = 0;
636 put_buf(w, fpdat.buf, (int)(fpdat.curp - fpdat.buf), sign_char);
637 } else {
638 char *p;
639 int m, n;
640 int texp;
641 int ww, dd;
642 /* Fm.<d-exp><n blanks>, where n = e + 2, m = w - n */
643 n = e + 2;
644 m = w - n;
645 if (*fpdat.cvtp == '0') {
646 fpdat.exp = 1;
647 }
648 /* switching to F, scale factor is suppressed */
649 /*
650 * use fcvt to convert value correctly rounded to a certain number of
651 * digits right or left of the decimal point. The last digit rounded
652 * is the sum of the value of 'd'.
653 * If this value is positive, then the digit is to the right of '.';
654 * negative => to the left of '.'.
655 */
656 fpdat.cvtp =
657 __io_fcvt(val, w, d - fpdat.exp, 0, &texp, &fpdat.sign, round, is_quad);
658 if (val == (__BIGREAL_T)0.0) {
659 texp = fpdat.exp;
660 } else if (texp != fpdat.exp) {
661 fpdat.exp = texp;
662 fpdat.cvtp =
663 __io_fcvt(val, w, d - texp, 0, &texp, &fpdat.sign, round, is_quad);
664 }
665 fpdat.ndigits = strlen(fpdat.cvtp);
666 ww = m;
667 dd = d - texp;
668 if (DBGBIT(0x4)) {

831 int is_quad) /* TRUE, if the value is quad */
832{
833 char *p;

836 fpdat.iszero = TRUE;

838 /* Maximum space required for fpdat.buf:
839 * "O." + <d>digits + "E+/-" <e>digits + "\0"

843 if (sf == 0) {
844 /* 0 . <d digits> */
845 /* cvtp_round(d); */
846 *fpdat.curp++ = '0';
847 *fpdat.curp++ = fpdat.decimal_char;

868 /* __io_printf("conv_e: illegal scale factor\n"); */

877 fpdat.sign = 0;
