69static void INLINE
70__reduction_slowpath(vdouble const a,
71S(vdouble *vrs, vmask *vhs) SINCOS_COMMA C(vdouble *vrc, vmask *vhc))
72{
73 int ivl = sizeof a / sizeof(a[0]);
74 union {
75 vdouble vd;
76 vmask vm;
77 double sd[sizeof(vdouble) / sizeof(double)];
78 uint64_t sm[sizeof(vmask) / sizeof(uint64_t)];
79 } S(rs, hs) SINCOS_COMMA C(rc, hc);
80
81 for (int i = 0; i < ivl; ++i) {
82 reduction_slowpath(a[i], S(&rs.sd[i], &hs.sm[i]) SINCOS_COMMA
83 C(&rc.sd[i], &hc.sm[i]));
84 }
85 S(*vrs = rs.vd;)
86 S(*vhs = hs.vm;)
87 C(*vrc = rc.vd;)
88 C(*vhc = hc.vm;)
89}
