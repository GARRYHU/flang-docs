327typedef struct {
328 int level; /* level number of loop */

333 int loop; /* loop for the sorted list */

336 * innermost
337 */

339 * the loops immediately enclosed by a loop

341 * child, and linked using the sibling field.

353 * variable with its & taken

356 * store via a pointer
357 */

359 * while or for to if-do-while
360 */
361 unsigned ptr_load : 1; /* loop or any enclosed loop contains a
362 * load via a pointer
363 */
364 unsigned nobla : 1; /* loop or any enclosed loop contains a
365 * block with its BIH_NOBLA flag set.
366 */
367 unsigned qjsr : 1; /* loop or any enclosed loop contains a
368 * block with BIH_QJSR set
369 */
370 unsigned mexits : 1; /* loop contains multiple exits */
371 unsigned jmp_tbl : 1; /* loop or any enclosed loop contains a
372 * jump using a jump tbl
373 */
374 unsigned invarif : 1; /* invarif loop optz. performed for loop or
375 * a descendant
376 */
377 unsigned cs : 1; /* region contains a block with BIH_CS set;
378 * not propagated to its parent's LP_CS.
379 */
380 unsigned csect : 1; /* loop or any enclosed loop contains a block
381 * with BIH_CS set.
382 */

409 UINT all;

531/* * * * * Scalar Definitions * * * * */
532/* NOTES:
533 * A called function implicitly defines those variables which can be affected
534 * by a call (e.g., variables, file static variables, &, etc.). In
535 * order to track these implicit definitions, definition 1 is reserved as
536 * the def implied by a call. If a block contains a call, definition 1 will
537 * be in its GEN set. We use just 1 definition instead of adding definitions
538 * for each of the call-affected variables to the block containing the call.
539 * The IN and OUT sets of a block will contain this def if any call can reach
540 * the block.
541 * A store via pointer implicitly defines those variables whose address
542 * has been taken. Definition 2 is reserved as the def implied by a store
543 * via a pointer. If a block contains a store via a pointer, def 2 will be
544 * in its GEN set.
545 *
546 * Definition 3 is the first user def.
547 */
548typedef struct {/* def table entry */
549 int fg;
550 int std;
551 int nm;
552 int lhs; /* lefthand side of the def - A_ID ast of the
553 * destination */
554 int addr; /* address ast of the destination */
555 int rhs; /* righthand side of the def - source ast */
556 int next; /* next def for this names */
557 int lnext; /* next definition in fg; 0 terminates list*/
558 union {
559 UINT all;
560 struct {
561 unsigned gen : 1; /* def reaches end of the block */
562 unsigned cnst : 1; /* def's value is a constant. */
563 unsigned delete : 1; /* def has been deleted */
564 unsigned self : 1; /* sym defined appears in def expr */
565 unsigned confl : 1; /* def's msize conflicts with a use's msize */
566 unsigned doinit : 1; /* initial def for a do/forall */
567 unsigned doend : 1; /* increment def for a enddo/endforall */
568 unsigned arg : 1; /* sym appears as an actual argument*/
569 unsigned other : 1; /* other implicit def: allocate status, etc. */
570 unsigned precise : 1; /* address is precise/function-invariant */
571 unsigned mark1 : 1;
572 unsigned mark2 : 1;
573 unsigned loop : 1; /* added at loop entry */
574 unsigned aggr : 1; /* def is of an aggregate */
575 } bits;
576 } flags;
577 DU *du; /* uses reached by this def */
578 DU *csel; /* cse uses not reached by this def; occurs
579 * for postfix expressions: i = j = k++;
580 * k's csel includes the cse uses of k stored
581 * into i and j.
582 */
583} DEF;
