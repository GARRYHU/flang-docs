20void ENT3F(GETARG, getarg)(int *n, DCHAR(arg) DCLEN(arg))
21{

27

30 q = arg;
31 } else {

33 p = v[i];
34 /* copy characters into arg: */

37 }
38 /* blank fill if necessary: */
39 for (; i < len; q++, i++)
40 *q = ' ';
41}
