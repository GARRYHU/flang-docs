13/*
14 * These arbitrary integer values are equivalent to the code numbers
15 * used in the PGI Fortran run-time library so that its rounding
16 * mode flags can be passed through without translation.

18enum decimal_rounding {
19 DECIMAL_ROUND_DEFAULT, /* same as COMPATIBLE */
20 DECIMAL_ROUND_IN = 90, /* RZ: toward zero (truncation) */
21 DECIMAL_ROUND_UP = 69, /* RU: toward +Inf */
22 DECIMAL_ROUND_DOWN = 70, /* RD: toward -Inf */
23 DECIMAL_ROUND_NEAREST = 71, /* RN: the usual IEEE "round to nearest", ties make even */
24 DECIMAL_ROUND_COMPATIBLE = 72, /* RC: Fortran's round to nearest, ties diverge from zero */
25 DECIMAL_ROUND_PROCESSOR_DEFINED = 73 /* RP: obey FPCR */
26};
