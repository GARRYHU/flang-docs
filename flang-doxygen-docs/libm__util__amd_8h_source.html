109/* Alternatives to the above functions which don't have
110 problems when using high optimization levels on gcc */
111#define GET_BITS_SP32(x, ux) \
112 { \
113 volatile union { \
114 float f; \
115 unsigned int i; \
116 } _bitsy; \
117 _bitsy.f = (x); \
118 ux = _bitsy.i; \
119 }
120#define PUT_BITS_SP32(ux, x) \
121 { \
122 volatile union { \
123 float f; \
124 unsigned int i; \
125 } _bitsy; \
126 _bitsy.i = (ux); \
127 x = _bitsy.f; \
128 }
