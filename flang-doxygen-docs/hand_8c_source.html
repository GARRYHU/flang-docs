59/* print signal message */
60
61void __fort_psignal(int lcpu, int s)
62{
63 char buf[256];
64 int n;
65
66 n = 0;
67 while ((sigs[n].sig != 0) && (sigs[n].sig != s)) {
68 n++;
69 }
70 if (sigs[n].sig == 0) {
71 sprintf(buf, "%d: killed by unknown signal %d\n", lcpu, s);
72 write(2, buf, strlen(buf));
73 } else if (sigs[n].str != NULL) {
74 sprintf(buf, "%d: %s\n", lcpu, sigs[n].str);
75 write(2, buf, strlen(buf));
76 }
77}

79/* handler for signals */

81static void sighand(int s)
82{
83 int lcpu;
84
85 lcpu = __fort_myprocnum();
86 __fort_psignal(lcpu, s); /* print message */
87#if !defined(_WIN64)
88 sleep(1); /* wait for message to clear */
89#endif
90 __fort_abort(NULL); /* abort */
91}

95void
96__fort_sethand()
97{
98 const char *p;
99 char *q;
100 int n;
101
102 p = __fort_getopt("-sigmsg");
103 if (p == NULL) {
104 return;
105 }
106 if ((*p == 'y') || (*p == 'Y') || (*p == 'a') || (*p == 'A') ||
107 (*p == '\0')) {
108 n = 0;
109 while (sigs[n].sig != 0) {
110 signal(sigs[n].sig, sighand);
111 n++;
112 }
113 } else if (*p != '\0') {
114 do {
115 n = __fort_strtol(p, &q, 0);
116 signal(n, sighand);
117 p = (*q == ',' ? q + 1 : q);
118 } while (*q != '\0');
119 }
120}
