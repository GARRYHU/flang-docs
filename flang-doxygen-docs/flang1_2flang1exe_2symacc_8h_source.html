29template <typename T, typename Index> class IndexBy
30{
31 T *rep;
32
33public:
34 T &operator[](Index index)
35 {
36 return rep[index];
37 }
38 /* Rest of operators are required to make macro NEED work.
39 Their functionality is deliberately minimized with intent
40 to minimize accidental use outside macro NEED. It would
41 be better that "operator void*" have the explicit keyword, but
42 Microsoft 10.0 Open Tools does not support that C++11 feature. */
43 operator char*() const
44 {
45 return reinterpret_cast<char*>(rep);
46 }
47 void operator=(T *ptr)
48 {
49 rep = ptr;
50 }
51 bool operator!() const
52 {
53 return !rep;
54 }
55 void *operator+(int offset) const
56 {
57 return reinterpret_cast<void *>(rep + offset);
58 }
59};
