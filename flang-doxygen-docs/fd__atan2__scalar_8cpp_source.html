35// We use the relationship between atan2(y,x) and atan(x),
36// as described in:
37// https://en.wikipedia.org/wiki/Atan2
38// to create an atan2(y, x) implementation based on our atan(x) implementation
39// Namely:
40// atan2(y, x) = atan(y/x) for x > 0
41// atan2(y, x) = atan(y/x) + pi for x < 0 and y >= 0
42// atan2(y, x) = atan(y/x) - pi for x < 0 and y < 0
43// atan2(y, x) = pi/2 for x = 0 and y > 0
44// atan2(y, x) = -pi/2 for x = 0 and y < 0
45
46// Also, from the C99 standards we need:
47// atan2(+-0.0, +0.0) = +-0.0
48// atan2(+-0.0, -0.0) = +-PI
49
50// Special care need to be taken when both x and y are +-INFINITY, where
51// the ieee definitions are equivalent to letting x and y tend to infinity at
52// the same rate.
53
54double atan2_d_scalar(double const y, double const x) {
55
56 // Special return values when both inputs are infinity, or 0's
57 // (or any absolute equal numbers, the results are the same):
58 if (__builtin_expect(fabs(y) == fabs(x), 1)) {
59
60 // Using (as_ulong(x) > 0x7FFFFFFFFFFFFFFF) rather than (x < 0.0) here
61 // seems to give a performance boost when looping over this function
62 // many times
63 double ans =
64 FMA((as_ulong(x) > 0x7FFFFFFFFFFFFFFF), PI_OVER_2, PI_OVER_4);
65
66 // Special return values for (y, x) = (+-0.0, +-0.0)
67 if (x == 0.0) {
68 ans = (as_ulong(x) == 0x0) ? 0.0 : PI;
69 }
70
71 return copysign(ans, y);
72 }
73
74 double xReduced;
75
76 // xReduced = ((fabs(y) > fabs(x)) ? x : y) / ((fabs(y) > fabs(x)) ? y : x);
77 // Seems to be the fastest way of getting x/y or y/x:
78 if (fabs(y) > fabs(x)) {
79 xReduced = x / y;
80 } else {
81 xReduced = y / x;
82 }
83
84 // The same Estrin scheme as is used in atan(x):
85 double x2 = xReduced * xReduced;
86 double x4 = x2 * x2;
87 double x8 = x4 * x4;
88 double x16 = x8 * x8;
89
90 double L1 = FMA(x2, C6, C5);
91 double L2 = FMA(x2, C8, C7);
92 double L3 = FMA(x2, C10, C9);
93 double L4 = FMA(x2, C12, C11);
94 double L5 = FMA(x2, C14, C13);
95 double L6 = FMA(x2, C16, C15);
96 double L7 = FMA(x2, C18, C17);
97 double L8 = FMA(x2, C20, C19);
98
99 // L1 + x4*L2 + x8*L3 + x12*L4 + x16*L5 + x20*L6 + x24*L7 + x28*L8
100 double M1 = FMA(x4, L2, L1);
101 double M2 = FMA(x4, L4, L3);
102 double M3 = FMA(x4, L6, L5);
103 double M4 = FMA(x4, L8, L7);
104
105 // M1 + x8*M2 + x16*M3 + x24*M4
106 // (M1 + x8*M2) + x16*(M3 + x8*M4)
107 // (M1 + x8*M2) + x16*(M3 + x8*M4)
108 double N1 = FMA(x8, M2, M1);
109 double N2 = FMA(x8, M4, M3);
110
111 // c2 + x2*c3 + x4*c4 + x6*(N1 + x16*N2):
112 double poly = FMA(x16, N2, N1);
113
114 poly = FMA(x4, FMA(x2, poly, C4), FMA(x2, C3, C2));
115
116 double result_d = poly;
117
118 double pi_factor = 0.0;
119
120 if (fabs(y) > fabs(x)) {
121 // pi/2 with the sign of xReduced:
122 const double signedPi_2 = as_double(
123 as_ulong(PI_OVER_2) | (as_ulong(xReduced) & 0x8000000000000000));
124
125 xReduced = -xReduced;
126 pi_factor = signedPi_2;
127 }
128
129 result_d = FMA(x2 * xReduced, poly, xReduced);
130
131 // Faster than the if statement:
132 // pi with the sign of y:
133 const double signedPi =
134 as_double(as_ulong(PI) | (as_ulong(y) & 0x8000000000000000));
135
136 // Again, faster:
137 pi_factor = FMA((as_ulong(x) > 0x7FFFFFFFFFFFFFFF), signedPi, pi_factor);
138 // if ((as_ulong(x) > 0x7FFFFFFFFFFFFFFF)) {
139 // pi_factor += signedPi;
140 //}
141
142 result_d += pi_factor;
143
144 // Unfortunately we have to do a copysign here to return the correctly
145 // signed 0.0 for atan2(+-0.0, x > 0.0) inputs:
146 result_d = copysign(result_d, y);
147
148#ifdef IACA
149#pragma message("IACA END")
150 IACA_END
151#endif
152
153 return result_d;
154}
