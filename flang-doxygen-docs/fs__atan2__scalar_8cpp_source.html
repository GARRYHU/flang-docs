31// We use the relationship between atan2(y,x) and atan(x),
32// as described in:
33// https://en.wikipedia.org/wiki/Atan2
34// to create an atan2(y, x) implementation based on our atan(x) implementation
35// Namely:
36// atan2(y, x) = atan(y/x) for x > 0
37// atan2(y, x) = atan(y/x) + pi for x < 0 and y >= 0
38// atan2(y, x) = atan(y/x) - pi for x < 0 and y < 0
39// atan2(y, x) = pi/2 for x = 0 and y > 0
40// atan2(y, x) = -pi/2 for x = 0 and y < 0
41
42// Also, from the C99 standards we need:
43// atan2(+-0.0, +0.0) = +-0.0
44// atan2(+-0.0, -0.0) = +-PI
45
46// Special care need to be taken when both x and y are +-INFINITY, where
47// the ieee definitions are equivalent to letting x and y tend to infinity at
48// the same rate.
49
50float atan2_scalar(float const y, float const x) {
51
52 // Special return values when both inputs are infinity, or 0's
53 // (or any absolute equal numbers, the results are the same):
54 if (__builtin_expect(fabs(y) == fabs(x), 1)) {
55
56 // Using (as_uint(x) > 0x7FFFFFFF) rather than (x < 0.0) here
57 // seems to give a performance boost when looping over this function
58 // many times
59 float ans = FMAF((as_uint(x) > 0x7FFFFFFF), PI_OVER_2, PI_OVER_4);
60
61 // Special return values for (y, x) = (+-0.0, +-0.0)
62 if (x == 0.0f) {
63 ans = (as_uint(x) == 0x0) ? 0.0f : PI;
64 }
65
66 return copysign(ans, y);
67 }
68
69 float xReduced;
70
71 // xReduced = (fabs(y) > fabs(x) ? x : y) / (fabs(y) > fabs(x) ? y : x);
72 // Seems to be the fastest way of getting x/y or y/x
73 // Comparing these as unsinged int's is a bit faster:
74 if (as_uint(fabs(y)) > as_uint(fabs(x))) {
75 xReduced = x / y;
76 } else {
77 xReduced = y / x;
78 }
79
80 // The same Estrin scheme as is used in atan(x):
81 float x2 = xReduced * xReduced;
82 float x4 = x2 * x2;
83 float x8 = x4 * x4;
84
85 // First layer of Estrin:
86 float L1 = FMAF(x2, C2, C1);
87 float L2 = FMAF(x2, C4, C3);
88 float L3 = FMAF(x2, C6, C5);
89 float L4 = FMAF(x2, C8, C7);
90
91 // Second layer of estrin
92 float M1 = FMAF(x4, L2, L1);
93 float M2 = FMAF(x4, L4, L3);
94
95 float poly = FMAF(x8, M2, M1);
96
97 float result_f = poly;
98
99 float pi_factor = 0.0f;
100
101 // Comparing these as unsinged int's is a bit faster:
102 if (as_uint(fabs(y)) > as_uint(fabs(x))) {
103 // pi/2 with the sign of xReduced:
104 // Manually doing the copysign here seems to be faster:
105 // const float signedPi_2 = copysignf(PI_OVER_2, xReduced);
106 const float signedPi_2 =
107 as_float(as_uint(PI_OVER_2) | (as_uint(xReduced) & 0x80000000));
108
109 xReduced = -xReduced;
110 pi_factor = signedPi_2;
111 }
112
113 result_f = FMAF(x2 * xReduced, poly, xReduced);
114
115 // pi with the sign of y:
116 // const float signedPi = as_float(as_uint(PI) | (as_uint(y) & 0x80000000));
117 const float signedPi = copysignf(PI, y);
118
119 // We need to check for -0.0 here as well as x < 0.0, so we cast to uint:
120 // Again, faster:
121 pi_factor = FMAF(as_uint(x) >= 0x80000000, signedPi, pi_factor);
122 // if (as_uint(x) > 0x7FFFFFFF) {
123 // pi_factor += signedPi;
124 //}
125
126 result_f += pi_factor;
127
128 // Unfortunately we need to do a copysign here because of the potential of y
129 // to be -0.0, and we return an incorrectly signed 0.0
130 result_f = copysignf(result_f, y);
131
132 return result_f;
133}
