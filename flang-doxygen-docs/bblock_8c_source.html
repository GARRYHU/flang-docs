58/** \brief Return nonzero if there are any CUDA Fortran kernels
59 */
60int
61bblock()
62{
63 int std;
64 int ast;
65 int sptr;
66 int label;
67 int ent;
68 int next_std;
69 int new_label;
70 int new_ast;
71 int new_std;
72 int penul_std, last_std;
73 int ret_label;
74 int aret_id;
75 int ret_cnt;
76 int tmp;
77 INT ent_cnt;
78 int ent_select_id;
79 int has_kernel = 0;
80 ITEM *itemp;
81
82 if (STD_NEXT(0) == STD_PREV(0)) { /* end only ? */
83 /* add something for entry -- lfm */
84 new_ast = mk_stmt(A_CONTINUE, 0);
85 add_stmt_after(new_ast, 0);
86 }
87
88 erly_bnds_depd.base = NULL;
89 erly_bnds_depd.sz = 10;
90 erly_bnds_depd.avl = 0;
91 NEW(erly_bnds_depd.base, int, erly_bnds_depd.sz);
92
93 sem.temps_reset = TRUE;
94 entry_point = 0;
95 last_std = STD_LAST;
96
97 if (gbl.arets) {
98 /* for alternate returns, will use a compiler-created local symbol
99 * which will contain the alternate return value
100 */
101 sptr = getsymbol("z__ret");
102 STYPEP(sptr, ST_VAR);
103 DTYPEP(sptr, DT_INT);
104 DCLDP(sptr, 1);
105 SCP(sptr, SC_LOCAL);
106 aret_id = mk_id(sptr);
107 }
108
109 penul_std = STD_PREV(last_std);
110 ret_label = 0;
111 if ((label = STD_LABEL(last_std))) {
112 /* if end statement is labeled, insert a continue before the end
113 * and transfer the label to the continue. This continue becomes
114 * the common exit point for the subprogram.
115 */
116 ret_label = label;
117 new_ast = mk_stmt(A_CONTINUE, 0);
118 gbl.exitstd = penul_std = add_stmt_after(new_ast, penul_std);
119 STD_LABEL(penul_std) = ret_label;
120 STD_LABEL(last_std) = 0;
121 STD_LINENO(penul_std) = STD_LINENO(last_std);
122 } else /* if (gbl.arets) */ {
123 /* if there are alternate returns, always create a new exit point
124 * since aret_id contains the alternate return value. Will generate
125 * return when A_END is seen.
126 */
127 /* LFM -- always create a new last statement, since we may need to
128 * add stuff around the last statement.
129 */
130 ret_label = getlab();
131 new_ast = mk_stmt(A_CONTINUE, 0);
132 gbl.exitstd = penul_std = add_stmt_after(new_ast, penul_std);
133 STD_LABEL(penul_std) = ret_label;
134 STD_LINENO(penul_std) = STD_LINENO(last_std);
135 }
136
137 ret_cnt = 0;
138
139 /* Use the ast visit routines to keep track of the id asts representing
140 * the left-hand sides of any assignments added to the prologue which
141 * assign values to adjustable bounds temporaries. If the id ast has
142 * been visited, then its assignment has been written.
143 */
144 ast_visit(1, 1);
145
146 if (gbl.rutype != RU_PROG) {
147 init_newargs(gbl.currsub);
148 if (ASSUMSHPG(gbl.currsub)) /* must process before adjustable args */
149 assumshp_args(gbl.currsub);
150 }
151
152 adj_based_arrays();
153 gen_early_str_len();
154
155 FREE(erly_bnds_depd.base);
156 erly_bnds_depd.sz = 0;
157 erly_bnds_depd.avl = 0;
158 ast_unvisit(); /* clean up visit fields of the id asts of bound temps*/
159
160 ent_cnt = ent_select_id = 0;
161 ENTNUMP(gbl.currsub, 0);
162 if (SYMLKG(gbl.currsub) > NOSYM) {
163 gbl.ent_select = getsymbol("z__ent");
164 STYPEP(gbl.ent_select, ST_VAR);
165 DTYPEP(gbl.ent_select, DT_INT);
166 DCLDP(gbl.ent_select, 1);
167 SCP(gbl.ent_select, SC_LOCAL);
168 HCCSYMP(gbl.ent_select, TRUE);
169 ent_select_id = mk_id(gbl.ent_select);
170 ast = mk_assn_stmt(ent_select_id, mk_cval(ent_cnt, DT_INT), DT_INT);
171 entry_point = add_stmt_after(ast, entry_point);
172 }
173 ENTSTDP(gbl.currsub, entry_point);
174
175 par = cs = parsect = atomic = 0;
176 kernel = 0;
177
178 for (std = STD_NEXT(entry_point); std; std = next_std) {
179 next_std = STD_NEXT(std); /* 'cause insertions may alter STD_NEXT */
180 gbl.lineno = STD_LINENO(std);
181 label = STD_LABEL(std);
182 ast = STD_AST(std);
183 switch (A_TYPEG(ast)) {
184 case A_ENTRY:
185 new_label = 0;
186 if (A_TYPEG(STD_AST(STD_PREV(std))) != A_GOTO) {
187 int astlab;
188 new_label = getlab();
189 DEFDP(new_label, 1);
190 RFCNTI(new_label);
191 new_ast = mk_stmt(A_GOTO, 0);
192 astlab = mk_label(new_label);
193 A_L1P(new_ast, astlab);
194 (void)add_stmt_before(new_ast, std);
195 }
196
197 ent = A_SPTRG(ast);
198 entry_point = std;
199
200 ast_visit(1, 1);
201 init_newargs(ent);
202 if (ASSUMSHPG(ent)) /* must process before adjustable args */
203 assumshp_args(ent);
204 adj_based_arrays();
205
206 ast_unvisit();
207
208 ent_cnt++;
209 ast = mk_assn_stmt(ent_select_id, mk_cval(ent_cnt, DT_INT), DT_INT);
210 entry_point = add_stmt_after(ast, entry_point);
211 ENTSTDP(ent, entry_point);
212 ENTNUMP(ent, ent_cnt);
213
214 new_ast = mk_stmt(A_CONTINUE, 0);
215 new_std = add_stmt_after(new_ast, entry_point);
216 if (new_label)
217 STD_LABEL(new_std) = new_label;
218
219 break;
220
221 case A_RETURN:
222 if (ret_cnt == 0 && penul_std == std) {
223 /* the subprogram's only exit point is from the return which
224 * is the penultimate statement. The subprogram doesn't not
225 * contain alternate returns.
226 */
227 if (label) {
228 /* since the return is labeled, insert a continue before
229 * the return and transfer the label to the continue.
230 * The continue becomes the subprogram's exit point.
231 */
232 new_ast = mk_stmt(A_CONTINUE, 0);
233 new_std = add_stmt_before(new_ast, std);
234 STD_LABEL(new_std) = label;
235 STD_LABEL(std) = 0;
236 gbl.exitstd = new_std;
237 } else
238 /* since the return isn't labeled, just use the statement
239 * before the return as the subprogram's exit point.
240 */
241 gbl.exitstd = STD_PREV(std);
242 } else {
243 int astlab;
244 /* change this return to a 'goto ret_label' */
245 ret_cnt++;
246 if (ret_label == 0)
247 ret_label = getlab();
248 tmp = A_LOPG(ast); /* just in case it's an alternate ret.*/
249 A_LOPP(ast, 0);
250 A_TYPEP(ast, A_GOTO);
251 RFCNTI(ret_label);
252 astlab = mk_label(ret_label);
253 A_L1P(ast, astlab);
254 if (gbl.arets) {
255 if (tmp == 0)
256 tmp = astb.i0;
257 new_ast = mk_assn_stmt(aret_id, tmp, DT_INT);
258 new_std = add_stmt_before(new_ast, std);
259 if (label) {
260 STD_LABEL(new_std) = label;
261 STD_LABEL(std) = 0;
262 }
263 }
264 }
265 break;
266
267 case A_END:
268 if (gbl.arets) {
269 /* gbl.exitstd already set */
270 new_ast = mk_stmt(A_RETURN, 0);
271 A_LOPP(new_ast, aret_id);
272 (void)add_stmt_before(new_ast, std);
273 } else if (gbl.exitstd == 0) {
274 if (ret_label) {
275 new_ast = mk_stmt(A_CONTINUE, 0);
276 gbl.exitstd = add_stmt_before(new_ast, std);
277 STD_LABEL(gbl.exitstd) = ret_label;
278 } else
279 gbl.exitstd = STD_PREV(std);
280 }
281 break;
282
283 case A_CONTINUE:
284 if (penul_std != std && (label == 0 || RFCNTG(label) == 0)) {
285 int s;
286 s = STD_PREV(std);
287 STD_PREV(next_std) = s;
288 STD_NEXT(s) = next_std;
289 }
290 break;
291
292 case A_MP_PARALLEL:
293 par++;
294 break;
295 case A_MP_ENDPARALLEL:
296 STD_PAR(std) = 1;
297 par--;
298 break;
299 case A_MP_CRITICAL:
300 case A_MP_ATOMIC:
301 cs++;
302 break;
303 case A_MP_ENDATOMIC:
304 case A_MP_ENDCRITICAL:
305 STD_CS(std) = 1;
306 cs--;
307 break;
308 case A_ATOMIC:
309 atomic++;
310 break;
311 case A_ENDATOMIC:
312 STD_ATOMIC(std) = 1;
313 atomic--;
314 break;
315 case A_MP_TASK:
316 case A_MP_TASKLOOP:
317 task++;
318 break;
319 case A_MP_ENDTASK:
320 case A_MP_ETASKLOOP:
321 STD_TASK(std) = 1;
322 task--;
323 break;
324 case A_MP_MASTER:
325 case A_MP_SINGLE:
326 case A_MP_SECTIONS:
327 parsect++;
328 break;
329 case A_MP_ENDMASTER:
330 case A_MP_ENDSINGLE:
331 case A_MP_ENDSECTIONS:
332 STD_PARSECT(std) = 1;
333 parsect--;
334 break;
335
336 default:
337 if (label) {
338 if (RFCNTG(label)) {
339 new_ast = mk_stmt(A_CONTINUE, 0);
340 new_std = add_stmt_before(new_ast, std);
341 STD_LABEL(new_std) = label;
342 set_std_parflags(new_std);
343 }
344 STD_LABEL(std) = 0;
345 }
346 break;
347 }
348 set_std_parflags(std);
349 }
350
351 if (gbl.arets)
352 /* since the alternate return id must be initialized to 0, add
353 * the assignment to the 'prologue' of each entry.
354 */
355 for (ent = gbl.currsub; ent != NOSYM; ent = SYMLKG(ent)) {
356 entry_point = ENTSTDG(ent);
357 new_ast = mk_assn_stmt(aret_id, astb.i0, DT_INT);
358 entry_point = add_stmt_after(new_ast, entry_point);
359 ENTSTDP(ent, entry_point);
360 }
361
362 if (sem.type_initialize) {
363 int std2;
364 for (std2 = ENTSTDG(gbl.currsub); STD_LINENO(std2) == 0;
365 std2 = STD_NEXT(std2))
366 ;
367 std2 = STD_PREV(std2);
368 for (itemp = sem.type_initialize; itemp; itemp = itemp->next) {
369 int stdx = CONSTRUCTSYMG(itemp->t.sptr) ?
370 BLOCK_ENTRY_STD(itemp->t.sptr) : std2;
371 gen_type_initialize_for_sym(itemp->t.sptr, stdx, 0, 0);
372 }
373 }
374
375 for (itemp = sem.alloc_mem_initialize; itemp; itemp = itemp->next) {
376 int stdx = CONSTRUCTSYMG(itemp->t.sptr) ?
377 BLOCK_ENTRY_STD(itemp->t.sptr) : ENTSTDG(gbl.currsub);
378 gen_alloc_mem_initialize_for_sym(itemp->t.sptr, stdx);
379 }
380
381 for (itemp = sem.auto_dealloc; itemp; itemp = itemp->next) {
382 int stdx = CONSTRUCTSYMG(itemp->t.sptr) ?
383 STD_PREV(BLOCK_EXIT_STD(itemp->t.sptr)) : gbl.exitstd;
384 gen_conditional_dealloc_for_sym(itemp->t.sptr, stdx);
385 }
386
387 for (itemp = sem.auto_finalize; itemp; itemp = itemp->next) {
388 int stdx = CONSTRUCTSYMG(itemp->t.sptr) ?
389 STD_PREV(BLOCK_EXIT_STD(itemp->t.sptr)) : gbl.exitstd;
390 gen_finalization_for_sym(itemp->t.sptr, stdx, 0);
391 }
392
393#if DEBUG
394 if (DBGBIT(10, 2))
395 dump_std();
396#endif
397 return has_kernel;
398}

400/* This routine is used to initialize the dummies to newarg and newdsc
401 * For example, call sub(a) will be later call sub(a$b, a$s).
402 * this routine creates a$b and a$s and store them to the first and
403 * second halves of address field of a
404 */
405static void
406init_newargs(int ent)
407{
408 int dscptr;
409 int arg, narg;
410 int i;
411 int newarg, newdsc;
412
413 narg = PARAMCTG(ent);
414 dscptr = DPDSCG(ent);
415 for (i = 0; i < narg; i++) {
416 arg = aux.dpdsc_base[dscptr + i];
417 if (STYPEG(arg) == ST_ENTRY)
418 error(480, 4, gbl.lineno, SYMNAME(arg), CNULL);
419 if (STYPEG(arg) == ST_PROC)
420 continue;
421 if (POINTERG(arg) && STYPEG(arg) != ST_ARRAY &&
422 (DDTG(DTYPEG(arg)) == DT_DEFERCHAR ||
423 DDTG(DTYPEG(arg)) == DT_DEFERNCHAR)) {
424 } else if (!ALLOCATTRG(arg) && STYPEG(arg) != ST_ARRAY &&
425 DTY(DTYPEG(arg)) != TY_ARRAY) {
426 continue;
427 }
428 if (is_bad_dtype(DTYPEG(arg)))
429 continue;
430 newarg = NEWARGG(arg);
431 newdsc = NEWDSCG(arg);
432 if (newarg != 0 && newdsc != 0)
433 continue;
434 if ((SEQG(arg) && !POINTERG(arg)) || F90POINTERG(arg)) {
435 newarg = arg;
436 } else if (XBIT(57, 0x80) || (XBIT(57, 0x10000) && ASSUMSHPG(arg)) ||
437 XBIT(57, 0x80000)) {
438 newarg = arg;
439 } else {
440 newarg = sym_get_formal(arg);
441 }
442 if (XBIT(57, 0x80000) && SDSCG(arg)) {
443 newdsc = SDSCG(arg);
444 } else {
445 newdsc = sym_get_arg_sec(arg);
446 }
447 NEWARGP(arg, newarg);
448 NEWDSCP(arg, newdsc);
449 }
450}

456static struct {
457 int *abase;
458 int acnt;
459 int sacnt;
460} assumshp;
461
462/*
463 * process assumed-shape array arguments before any adjustable arrays.
464 * the function result could be an adjustable array whose size depends
465 * on the size of one of the dummy arguments.
466 * Need to check their lower bounds for any dependencies on other assumed-shape
467 * arguments; need to first process those arguments on which other arrays
468 * depend.
469 */
470static void
471assumshp_args(int ent)
472{
473 int cnt;
474 int dscptr;
475 int arg;
476 int i;
477 int any;
478
479 cnt = PARAMCTG(ent);
480 if (cnt == 0)
481 return;
482 /*
483 * Create a table large enough to be hold two copies of the assumed-shape
484 * arguments. First, collect the assumed-shape arguments into the first
485 * part of the table.
486 */
487 assumshp.acnt = 0;
488 NEW(assumshp.abase, int, 2 * cnt + 1);
489 dscptr = DPDSCG(ent);
490 while (TRUE) {
491 arg = aux.dpdsc_base[dscptr];
492 if (arg && STYPEG(arg) != ST_PROC && DTY(DTYPEG(arg)) == TY_ARRAY &&
493 ASSUMSHPG(arg)) {
494 assumshp.abase[assumshp.acnt] = arg;
495 assumshp.acnt++;
496 VISITP(arg, 1); /* is an assumed-shape arg to the function */
497 VISIT2P(arg, 0); /* have not yet checked for dependencies */
498 }
499 if (--cnt == 0)
500 break;
501 dscptr++;
502 }
503
504 /*
505 * Recursively check the lower bounds for any dependencies on other
506 * assumed-shape arguments; the second part of the table will contain
507 * the arguments in an order such that they will be processed before
508 * any dependents.
509 */
510 any = 0;
511 assumshp.sacnt = assumshp.acnt;
512 if (assumshp.acnt > 1) {
513 for (i = 0; i < assumshp.acnt; i++) {
514 arg = assumshp.abase[i];
515 find_assumshp_dep(arg);
516 any = 1;
517 }
518 } else if (assumshp.acnt == 1)
519 add_assumshp(assumshp.abase[0]);
520
521 if (any) {
522 ast_unvisit(); /* clean up the visit list */
523 /* repeat visit setup for keeping track of assignments to the
524 * adjustable bounds temporaries.
525 */
526 ast_visit(1, 1);
527 }
528
529 ENTSTDP(ent, entry_point);
530 for (i = assumshp.acnt; i < assumshp.sacnt; i++) {
531 arg = assumshp.abase[i];
532 VISITP(arg, 0);
533 VISIT2P(arg, 0);
534 set_assumed_bounds(arg, ent, 0);
535 }
536 entry_point = ENTSTDG(ent);
537 FREE(assumshp.abase);
538}

540static void
541find_assumshp_dep(int arg)
542{
543 int ndim;
544 ADSC *ad;
545 int d;
546 int lb;
547
548 if (VISIT2G(arg)) /* already checked for dependencies */
549 return;
550 VISIT2P(arg, 1);
551 ad = AD_DPTR(DTYPEG(arg));
552 ndim = AD_NUMDIM(ad);
553 for (d = 0; d < ndim; d++) {
554 lb = AD_LWBD(ad, d);
555 if (!lb || A_ALIASG(lb)) {
556 continue;
557 }
558 ast_traverse(lb, _find_assumshp_dep, NULL, NULL);
559 }
560 add_assumshp(arg);
561}

563static LOGICAL
564_find_assumshp_dep(int ast, LOGICAL *rr)
565{
566 if (A_TYPEG(ast) == A_ID) {
567 int sym;
568
569 sym = A_SPTRG(ast);
570 switch (STYPEG(sym)) {
571 case ST_ARRAY:
572 case ST_VAR:
573 if (SCG(sym) == SC_DUMMY && DTY(DTYPEG(sym)) == TY_ARRAY &&
574 ASSUMSHPG(sym) && VISITG(sym)) {
575 find_assumshp_dep(sym);
576 }
577 break;
578 default:;
579 }
580 }
581 return FALSE;
582}

584static void
585add_assumshp(int arg)
586{
587 int i;
588
589 for (i = assumshp.acnt; i < assumshp.sacnt; i++) {
590 if (arg == assumshp.abase[i])
591 return;
592 }
593 assumshp.abase[assumshp.sacnt] = arg;
594 assumshp.sacnt++;
595}

597static int
598early_specification_stmt_needed(int ast)
599{
600 int argt;
601 int argcnt;
602 int i;
603 int sptr;
604
605 if (ast) {
606 switch (A_TYPEG(ast)) {
607 case A_BINOP:
608 return (early_specification_stmt_needed(A_LOPG(ast)) ||
609 early_specification_stmt_needed(A_ROPG(ast)));
610 case A_UNOP:
611 case A_CONV:
612 case A_PAREN:
613 return (early_specification_stmt_needed(A_LOPG(ast)));
614 case A_INTR:
615 sptr = A_SPTRG(A_LOPG(ast));
616 switch (STYPEG(sptr)) {
617 case ST_GENERIC:
618 case ST_INTRIN:
619 /* simple */
620 argt = A_ARGSG(ast);
621 argcnt = ARGT_CNT(argt);
622 for (i = 0; i < argcnt; i++) {
623 if (early_specification_stmt_needed(ARGT_ARG(argt, i))) {
624 return TRUE;
625 }
626 }
627 break;
628 default:
629 return TRUE;
630 }
631 break;
632 case A_FUNC:
633 /* Part of the fix for FS1551. Early specification stmt is
634 * needed if any function call is found.
635 */
636 return TRUE;
637 }
638 }
639 return FALSE;
640}

650static void
651gen_early_bnd_dependencies(int ast)
652{
653 int sptr;
654 int std;
655 ADSC *ad;
656 int ndims;
657 int i;
658 int bndsptr;
659 int argt;
660 int argcnt;
661 int dtype;
662 LOGICAL early_spec_gend = FALSE;
663
664 if (!ast)
665 return;
666
667 switch (A_TYPEG(ast)) {
668 case A_ID:
669 sptr = A_SPTRG(ast);
670 /* insert dependencies before dependent bnds exprs */
671 std = CONSTRUCTSYMG(sptr) ? BLOCK_ENTRY_STD(sptr) : ENTSTDG(gbl.currsub);
672 if (STYPEG(sptr) == ST_ARRAY && ADJARRG(sptr) && !EARLYSPECG(sptr)) {
673 ad = AD_DPTR(DTYPEG(sptr));
674 ndims = AD_NUMDIM(ad);
675 for (i = 0; i < ndims; i++) {
676 if (A_TYPEG(AD_LWAST(ad, i)) != A_CNST) {
677 bndsptr = A_SPTRG(AD_LWAST(ad, i));
678 if (!EARLYSPECG(bndsptr)) {
679 std = add_stmt_after(
680 mk_assn_stmt(AD_LWAST(ad, i), AD_LWBD(ad, i), astb.bnd.dtype),
681 std);
682 EARLYSPECP(bndsptr, 1);
683 gen_early_bnd_dependencies(AD_LWBD(ad, i));
684 }
685 early_spec_gend = TRUE;
686 }
687 if (A_TYPEG(AD_UPAST(ad, i)) != A_CNST) {
688 bndsptr = A_SPTRG(AD_UPAST(ad, i));
689 if (!EARLYSPECG(bndsptr)) {
690 std = add_stmt_after(
691 mk_assn_stmt(AD_UPAST(ad, i), AD_UPBD(ad, i), astb.bnd.dtype),
692 std);
693 EARLYSPECP(bndsptr, 1);
694 gen_early_bnd_dependencies(AD_UPBD(ad, i));
695 }
696 early_spec_gend = TRUE;
697 }
698 }
699 }
700 if (ADJLENG(sptr)) {
701 if (!EARLYSPECG(sptr)) {
702 int rhs, cvlen;
703 dtype = DDTG(DTYPEG(sptr));

957static void merge(int, int);
958static int donetwice1, donetwice2; /* to prevent message coming out twice */
959
960/** \brief Merge common blocks.
961
962 If two modules declare the same common block, both sets of symbols
963 appear in the program. If either uses the same variable names,
964 by Fortran 90 rules, the names may not be used, though they could be
965 hidden by USE, ONLY clauses. To generate correct output, we must
966 merge the two definitions. This routine checks that the distributed
967 variables in the duplicate commons have the same type, distribution,
968 offsets, and sizes. It replaces all references to one of the commons
969 by references to the other. For nondistributed data, this routine
970 generates EQUIVALENCE data.
971 */
972void
973merge_commons()
974{
975 int cmn1, pcmn1, biggest1, cmn2, pcmn2;
976 donetwice1 = 0;
977 donetwice2 = 0;
978 pcmn1 = 0;
979 for (cmn1 = gbl.cmblks; cmn1 > NOSYM; cmn1 = SYMLKG(cmn1)) {
980 int removed1, any;
981 biggest1 = cmn1;
982 any = 0;
983 for (cmn2 = SYMLKG(cmn1); cmn2 > NOSYM; cmn2 = SYMLKG(cmn2)) {
984 if (NMPTRG(cmn1) == NMPTRG(cmn2)) {
985 any = 1;
986 if (SIZEG(cmn2) > SIZEG(biggest1)) {
987 biggest1 = cmn2;
988 }
989 }
990 }
991 removed1 = 0;
992 if (any) {
993 pcmn2 = pcmn1;
994 for (cmn2 = cmn1; cmn2 > NOSYM; cmn2 = SYMLKG(cmn2)) {
995 if (NMPTRG(biggest1) == NMPTRG(cmn2) && cmn2 != biggest1) {
996 merge(biggest1, cmn2);
997 /* remove cmn2 */
998 if (pcmn2) {
999 SYMLKP(pcmn2, SYMLKG(cmn2));
1000 } else {
1001 gbl.cmblks = SYMLKG(cmn2);
1002 }
1003 if (cmn2 == cmn1)
1004 removed1 = 0;
1005 } else {
1006 pcmn2 = cmn2;
1007 }
1008 }
1009 }
1010 if (!removed1)
1011 pcmn1 = cmn1;
1012 }
1013} /* merge_commons */

1169static void
1170add_equivalence(int cmn1, int cmn2, int mem1, int mem2)
1171{
1172 /* if the symbols are mutually aligned, insert an equivalence */
1173 ISZ_T off1, off2, off, size1, size2, size;
1174 int dty1, dtype1, dty2, dtype2, CASE;
1175 if (merge_member == mem2)
1176 return;
1177 off1 = ADDRESSG(mem1);
1178 off2 = ADDRESSG(mem2);
1179 dty1 = dtype1 = DTYPEG(mem1);
1180 dty2 = dtype2 = DTYPEG(mem2);
1181 if (DTY(dtype1) == TY_ARRAY)
1182 dty1 = DTY(dtype1 + 1);
1183 if (DTY(dtype2) == TY_ARRAY)
1184 dty2 = DTY(dtype2 + 1);
1185 size1 = size_of(dty1);
1186 size2 = size_of(dty2);
1187 if (size1 <= size2) {
1188 size = size1;
1189 off = off2 - off1;
1190 CASE = 1;
1191 } else {
1192 size = size2;
1193 off = off1 - off2;
1194 CASE = 2;
1195 }
1196 /* check alignment */
1197 if ((off % size) == 0 && (size1 % size) == 0 && (size2 % size) == 0) {
1198 ISZ_T o, s, m2, m1;
1199 int evp, ss;
1200 /* we want to declare EQUIVALENCE( mem1(lb1+m1), mem2(lb2+m2) )
1201 * but in general we don't know the offsets m1 and m2.
1202 * In the general case, we have a stupid diophantine equation:
1203 * off1 + m1*size1 = off2 + m2*size2
1204 * m1*size1 - m2*size2 = off2-off1 = off [if case 1, size1==size]
1205 * m2*size2 - m1*size1 = off1-off2 = off [if case 2, size2==size]
1206 * we know 'size' divides size1 and size2, so divide by 'size'
1207 * m1 - m2*(size2/size) = (off/size) [if case 1, size1==size]
1208 * m1 = (off/size) + m2*(size2/size)
1209 * m2 - m1*(size1/size) = (off/size) [if case 2, size2==size]
1210 * m2 = (off/size) + m1*(size1/size)
1211 * in either case, we choose a positive value of the independent
1212 * variable that makes the dependent variable also positive.
1213 * In case 1, we have
1214 * (off/size) + m2*(size2/size) > 0
1215 * m2 > -(off/size)/(size2/size)
1216 * m2 > -(off*size)/(size2*size)
1217 * m2 > -off/size2
1218 * In case 2, we have
1219 * (off/size) + m1*(size1/size) > 0
1220 * m1 > -off/size1
1221 */
1222 if (CASE == 1) {
1223 s = size2 / size;
1224 o = off / size;
1225 if (o >= 0) {
1226 m2 = 0;
1227 } else {
1228 m2 = -off / size2;
1229 }
1230 m1 = o + m2 * s;
1231 } else {
1232 s = size1 / size;
1233 o = off / size;
1234 if (o >= 0) {
1235 m1 = 0;
1236 } else {
1237 m1 = -o / s;
1238 }
1239 m2 = o + m1 * s;
1240 }
1241#if DEBUG
1242 if (DBGBIT(10, 8))
1243 fprintf(gbl.dbgfil, "common/%s/ %s(%d) and /%s/ %s(%d)\n", SYMNAME(cmn1),
1244 SYMNAME(mem1), (int)m1, SYMNAME(cmn2), SYMNAME(mem2), (int)m2);
1245#endif
1246 if (m1 >= 0 && m2 >= 0 && (DTY(dtype1) == TY_ARRAY || m1 == 0) &&
1247 (DTY(dtype2) == TY_ARRAY || m2 == 0)) {
1248 /* Convert 'm1' into subscripts */
1249 if (DTY(dtype1) != TY_ARRAY) {
1250 ss = 0;
1251 } else {
1252 ss = make_equiv_ss(m1, dtype1);
1253 }
1254 evp = sem.eqv_avail;
1255 ++sem.eqv_avail;
1256 NEED(sem.eqv_avail + 1, sem.eqv_base, EQVV, sem.eqv_size,
1257 sem.eqv_size + 20);
1258 EQV(evp).sptr = mem1;
1259 EQV(evp).is_first = 1;
1260 EQV(evp).lineno = 0;
1261 EQV(evp).subscripts = ss;
1262 EQV(evp).substring = 0;
1263 EQV(evp).byte_offset = m1;
1264 EQV(evp).next = evp + 1;
1265
1266 /* Convert 'm2' into subscripts */
1267 if (DTY(dtype2) != TY_ARRAY) {
1268 ss = 0;
1269 } else {
1270 ss = make_equiv_ss(m2, dtype2);
1271 }
1272 evp = sem.eqv_avail;
1273 ++sem.eqv_avail;
1274 EQV(evp).sptr = mem2;
1275 EQV(evp).is_first = 0;
1276 EQV(evp).lineno = 0;
1277 EQV(evp).subscripts = ss;
1278 EQV(evp).substring = 0;
1279 EQV(evp).byte_offset = m2;
1280 EQV(evp).next = sem.eqvlist;
1281 sem.eqvlist = evp - 1;
1282 merge_member = mem2;
1283 return;
1284 }
1285 }
1286 /* unaligned, need to add something */
1287 if (merge_common_symbol == 0) {
1288 int d1, d2, dty, size, sizeast;
1289 /* find smallest item in cmn1, cmn2 */
1290 d1 = smallest_dtype(cmn1);
1291 d2 = smallest_dtype(cmn2);
1292 if (size_of(d2) < size_of(d1)) {
1293 d1 = d2;
1294 }

1378 * replace one by the other. This works even for distributed

1388 mem2 = nmem2;
1389 continue;
1390 }
1391 }

1393 * same size, insert an equivalence. This works only for
1394 * nondistributed variables */

1397 /* does mem1 completely come before mem2? */
1398 if (off1 + size1 <= off2) {
1399 mem1 = nmem1;
1400 /* else, does mem2 completely come before mem1? */
1401 } else if (off2 + size2 <= off1) {
1402 mem2 = nmem2;
1403 } else {

1405 /* add an equivalence statement */
1406 add_equivalence(cmn1, cmn2, mem1, mem2);
1407 /* add SOC record */
1408 add_soc(mem1, mem2);
1409

1411 mem1 = nmem1;
1412 } else if (off1 + size1 > off2 + size2) {
1413 mem2 = nmem2;
1414 } else {
1415 mem1 = nmem1;
1416 mem2 = nmem2;
1417 }
1418 }
1419 }
1420 rewrite_all_asts();
1421 ast_unvisit();

1424 for (mem2 = CMEMFG(cmn2); mem2 > NOSYM; mem2 = nmem2) {

1429 SYMLKP(mem1, mem2);
1430 EQVP(mem2, 1);
1431 mem1 = mem2;
1432 }
1433 }
1434 CMEMLP(cmn1, mem1);
1435 IGNOREP(cmn2, 1);
