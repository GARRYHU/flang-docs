78 * SHOULD have a matching close_pragma().

81open_pragma(int line)
82{

169/** \brief Find the set of loop pragmas associated with a line number (the
170 * set's beginning and ending line numbers enclose the requested line number).
171 *
172 * Since the loop sets occur in lexical order, the last loop set found is the
173 * set with beginning line number is the floor of the requested line number.
174 */
175static int
176find_lpprg(int line)
177{
178 int i;
179 LPPRG *lpprg;
180 int match;
181
182 if (first == 0 || line == 0)
183 return 0;
184
185 match = 0;
186 for (i = first; i < direct.lpg.avail; i++) {
187 lpprg = direct.lpg.stgb + i;
188 if (lpprg->beg_line < 0 || lpprg->beg_line > line)
189 break;
190 if (lpprg->beg_line <= line && line <= lpprg->end_line)
191 match = i;
192 }
193 TR2("find_lpprg:for line %d, match = %d\n", line, match);
194
195 return match;
196}

204/** \brief Find and open and stack the set of loop pragmas associated with a
205 * line number.
206 *
207 * MUST have a matching pop_pragma(). If a match isn't found, push the
208 * routine's set.
209 */
210void
211push_pragma(int line)
212{
213 LPPRG *lpprg;
214 int match;
215
216 top++;
217 if (top >= STK_SZ) {
218#if DEBUG
219 interr("push_pragma:stkovflw", line, ERR_Severe);
220#endif
221 top = STK_SZ - 1;
222 }
223 match = find_lpprg(line);
224 stk[top] = match;
225 if (match) {
226 lpprg = direct.lpg.stgb + match;
227 TR2(" push: begline %d, endline %d\n", lpprg->beg_line, lpprg->end_line);
228 load_dirset(&lpprg->dirset);
229 } else {
230 TR0(" push: using routine\n");
231 load_dirset(&direct.rou_begin);
232 }
233
234}

236/** \brief Pop the set from the stack and open/restore the set which is on the
237 * top of stack.
238 */
239void
240pop_pragma(void)
241{
242 LPPRG *lpprg;
243 int match;
244
245#if DEBUG
246 if (top <= 0) {
247 interr("pop_pragma:stkuflw", gbl.lineno, ERR_Severe);
248 load_dirset(&direct.rou_begin);
249 return;
250 }
251#endif
252 top--;
253 match = stk[top];
254 if (match) {
255 lpprg = direct.lpg.stgb + match;
256 TR1(" pop: using %d\n", match);
257 load_dirset(&lpprg->dirset);
258 } else {
259 TR0(" pop: using routine\n");
260 load_dirset(&direct.rou_begin);
261 }
