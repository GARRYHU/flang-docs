174/* macros for handling the red zone of a stack, i.e., the 128-byte area
175 * below a leaf function's stack pointer as defined in the linux abi.
176 * For other enviroments, the red zone must be allocated.
177 * Macros:
178 * RZ_PUSH -- create the 128-byte redzone
179 * RZ_POP -- relinquish the 128-byte redzone
180 * RZ_OFF(n) -- produce the offset representing n (positive) bytes below the
181 * stackpointer wrt the linux abi; e.g., for the linux abi
182 * RZ_OFF(24)(%rsp) => -24(%rsp); for other systems, the
183 * offset would be computed as 128-24.
184 */
185
186#if defined(TARGET_WIN_X8664)
187#define RZ_PUSH subq $128, % rsp
188#define RZ_POP addq $128, % rsp
189#define RZ_OFF(n) (128## - n)
190#else
191#define RZ_PUSH
192#define RZ_POP
193#define RZ_OFF(n) -##n
194#endif

216/*
217 * Define a set of generic vex prefixed floating point instructions
218 * (macros) that can be used for both Intel 3 operand and AMD's 4
219 * operand instruction sets.
220 *
221 * Conform to Intel's instruction notation.
222 *
223 * Instructions (macros) use the C preprocessor and not the ".macro"
224 * facility builtin with most assemblers because of the inconsistent
225 * syntax and availability of pseudo ops across various platforms.
226 *
227 * Selection between FMA3 and FMA4(AMD only) instructions is controlled
228 * by macro "VFMA_IS_FMA3_OR_FMA4" which must be defined to be either
229 * "FMA3" or "FMA4".
230 *
231 * Map:
232 * Instruction "Pseudo OP" (CPP macro)
233 * vfmadd{132,213,231}{ss,sd,ps,pd} VFMA_{132,213,231}{SS,SD,PS,PD}
234 * vfnmadd{132,213,231}{ss,sd,ps,pd} VNFMA_{132,213,231}{SS,SD,PS,PD}
235 * vfmsub{132,213,231}{ss,sd,ps,pd} VFMS_{132,213,231}{SS,SD,PS,PD}
236 * vfnmsub{132,213,231}{ss,sd,ps,pd} VNFMS_{132,213,231}{SS,SD,PS,PD}
237 */
238
239#define VFMA_FMA4_ORDER_132(src_rm,src_r,dst) src_r,src_rm,dst,dst
240#define VFMA_FMA4_ORDER_213(src_rm,src_r,dst) src_rm,src_r,dst,dst
241#define VFMA_FMA4_ORDER_231(src_rm,src_r,dst) dst,src_rm,src_r,dst
242
243#define IF_VFMA_IS_FMA3(op,order,size,src_rm,src_r,dst) \
244 ASM_CONCAT3(op,order,size) src_rm,src_r,dst
245
246#define IF_VFMA_IS_FMA4(op,order,size,src_rm,src_r,dst) \
247 ASM_CONCAT(op,size) ASM_CONCAT(VFMA_FMA4_ORDER_,order)(src_rm,src_r,dst)
248
249#define VFMA_FMA3_OR_FMA4(op,order,size,src_rm,src_r,dst) \
250 ASM_CONCAT(IF_VFMA_IS_,VFMA_IS_FMA3_OR_FMA4)(op,order,size,src_rm,src_r,dst)

253 * A couple of true/false (1/0) macros to use to determine whether FMA3 or
254 * FMA4 is being targetted.
255 *
256 * Note: Macro "VFMA_IS_FMA3_OR_FMA4" must be defined.

259#define VFMA_IS_FMA3_(l,r) l
260#define VFMA_IS_FMA4_(l,r) r
261#define VFMA_IS_FMA3 ASM_CONCAT3(VFMA_IS_,VFMA_IS_FMA3_OR_FMA4,_)(1,0)
262#define VFMA_IS_FMA4 ASM_CONCAT3(VFMA_IS_,VFMA_IS_FMA3_OR_FMA4,_)(0,1)
