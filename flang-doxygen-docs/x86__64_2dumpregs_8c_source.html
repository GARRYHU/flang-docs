48/*
49 * greg2u64: convert value of typedef greg_t to uint64_t.
50 * Avoids recasting through pointers.
51 */
52static inline uint64_t greg2u64(greg_t r)
53{
54 struct {
55 union {
56 greg_t g;
57 uint64_t u64;
58 };
59 } g2u;
60 g2u.g = r;
61 return g2u.u64;
