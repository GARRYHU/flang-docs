23/* Payne-Hanek style argument reduction. */
24static float
25reduction_slowpath(float const a, int32_t *h)
26{
27 uint2 m;
28 uint32_t ia = float_as_int(a);
29 uint32_t s = ia & 0x80000000;
30 uint32_t result[7];
31 uint32_t hi, lo;
32 uint32_t e;
33 int32_t idx;
34 int32_t q;
35 e = ((ia >> 23) & 0xff) - 127;
36 ia = (ia << 8) | 0x80000000;
37
38 /* compute x * 2/pi */
39 idx = 4 - ((e >> 5) & 3);
40
41 hi = 0;
42 for (q = 0; q < 6; q++) {
43 m = umad32wide(i2opi_f[q], ia, hi);
44 lo = m.x;
45 hi = m.y;
46 result[q] = lo;
47 }
48 result[q] = hi;
49
50 e = e & 31;
51 /* shift result such that hi:lo<63:63> is the least significant
52 integer bit, and hi:lo<62:0> are the fractional bits of the result
53 */
54 uint64_t p = ((uint64_t)result[idx + 2] << 32) | result[idx + 1];
55
56 if (e) {
57 q = 32 - e;
58 p = (p << e) | (result[idx] >> q);
59 }
60
61 /* fraction */
62 q = (result[idx + 2] << e) & 0x80000000;
63 p &= 0x7fffffffffffffffULL;
64
65 if (p & 0x4000000000000000ULL) {
66 p |= 0x8000000000000000ULL;
67 q ^= 0x80000000;
68 }
69 *h = q;
70
71 double d = (double)(int64_t)p;
72 d *= PI_2_M64;
73 float r = (float)d;
74
75 return int_as_float(float_as_int(r) ^ s);
76}

78float __attribute__ ((noinline))
79__fs_tan_1_avx2(float x)
80{
81
82 float p, k, r, s, t;
83 int h = 0;
84
85 p = int_as_float(float_as_int(x) & 0x7fffffff);
86 if (float_as_int(p) > float_as_int(THRESHOLD_F)) {
87 x = float_as_int(p) >= 0x7f800000 ? x * 0.0f : reduction_slowpath(x, &h);
88 } else {
89 k = FMAF(x, _2_OVER_PI_F, 12582912.0f);
90 h = float_as_int(k) << 31;
91 k -= 12582912.0f;
92 x = FMAF(k, -PI_2_HI_F, x);
93 x = FMAF(k, -PI_2_MI_F, x);
94 x = FMAF(k, -PI_2_LO_F, x);
95 }
96 s = x * x;
97 r = A_F;
98 r = FMAF(r, s, B_F);
99 r = FMAF(r, s, C_F);
100 r = FMAF(r, s, D_F);
101 r = FMAF(r, s, E_F);
102 r = FMAF(r, s, F_F);
103 t = s * x;
104 r = FMAF(r, t, x);
105
106 if (h) r = -1.0f / r;
107
108 return r;
109}
