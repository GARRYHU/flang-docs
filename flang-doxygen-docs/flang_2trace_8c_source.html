14/** \brief Initialization */
15int
16__fort_trac_init(void)
17{
18 const char *p;
19 char *q;
20 int n;
21
22 p = __fort_getopt("-trace");
23 if (p) {
24 n = (int)__fort_strtol(p, &q, 0);
25 if (q == p)
26 tracing = 1;
27 else if (*q != '\0' || n < 0 || n >= GET_DIST_TCPUS)
28 __fort_abort("invalid -trace processor");
29 else
30 tracing = (n == GET_DIST_LCPU);
31 }
32 return 0;
33}
