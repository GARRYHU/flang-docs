64/* write two lines of info for symbol elem to listing. */
65static void
66putentry(int elem)
67{
68 char buf[200];
69 const char *ptr;
70 const char *sc_p;
71 int stype, dtype;
72 char hyp;
73 static const char *scs[] = {"n/a", "local", "static", "dummy",
74 "common", "extern", "based", "seven"};
75
76 stype = STYPEG(elem);
77
78 ptr = SYMNAME(elem);
79 if (stype == ST_LABEL)
80 ptr += 2; /* delete ".L" from front of label name */
81
82 dtype = 0;
83 if (stype != ST_LABEL && stype != ST_CMBLK) {
84 dtype = DTYPEG(elem);
85 }
86
87 sprintf(buf, "%-16.16s ", ptr);
88 getdtype(dtype, &buf[17]);
89 strcat(buf, " ");
90 strcat(buf, stb.stypes[stype]);
91 list_line(buf);
92
93 if ((int)strlen(ptr) > 16) {
94 ptr += 16;
95 hyp = '-';
96 } else {
97 ptr = "";
98 hyp = ' ';
99 }
100
101 switch (stype) {
102 case ST_LABEL:
103 sprintf(buf, "%c%-15.15s addr: %" ISZ_PF "x", hyp, ptr, ADDRESSG(elem));
104 break;
105 case ST_PARAM:
106 /* Changed to display ALL PARAMETER data types LDE */
107 /* if (dtype <= DT_INT)
108 sprintf(buf, "%c%-15.15s value: %d", hyp, ptr, CONVAL1G(elem));
109 else
110 sprintf(buf, "%c%-15.15s", hyp, ptr);
111 */
112 sprintf(buf, "%c%-15.15s value: %s", hyp, ptr, parmprint(elem));
113 break;
114 case ST_ENTRY:
115 sprintf(buf, "%c%-15.15s addr: %" ISZ_PF "x", hyp, ptr, ADDRESSG(elem));
116 break;
117 case ST_CMBLK:
118 sprintf(buf, "%c%-15.15s size: %" ISZ_PF "d bytes", hyp, ptr,
119 SIZEG(elem));
120 break;
121
122 case ST_VAR:
123 case ST_IDENT:
124 case ST_ARRAY:
125 case ST_DESCRIPTOR:
126 case ST_STRUCT:
127 case ST_UNION:
128 if (SCG(elem) == SC_CMBLK)
129 sc_p = SYMNAME(CMBLKG(elem));
130 else
131 sc_p = scs[SCG(elem)];
132 sprintf(buf, "%c%-15.15s sc: %s addr: %" ISZ_PF "x", hyp, ptr, sc_p,
133 ADDRESSG(elem));
134 break;
135
136 case ST_PROC:
137 case ST_STFUNC:
138 case ST_USERGENERIC:
139 case ST_GENERIC:
140 case ST_INTRIN:
141 default:
142 if (hyp == '-')
143 sprintf(buf, "%c%-15.15s", hyp, ptr);
144 else
145 hyp = 0;
146 }
147
148 if (hyp != 0)
149 list_line(buf);
150
151}

155static void
156printxref(int liststart)
157{
158 int sptr;
159 INT refptr;
160 int i;
161 char buf[200];
162
163 /* print header */
164 list_line("Cross Reference Listing:");
165 list_line("");
166
167 /* for each ref'd element of symbol table with a name */
168 for (sptr = liststart; sptr != 0; sptr = HASHLKG(sptr)) {
169 assert((NMPTRG(sptr) != 0), "printxref bad record name", sptr, 3);
170 assert((XREFLKG(sptr) != 0), "printxref bad record link", sptr, 3);
171
172 /* print stuff about symbol table entry */
173 putentry(sptr);
174
175 refptr = XREFLKG(sptr);
176
177#define ENDOFLINE 80
178 /* put references info */
179 i = 999; /* i is char position in line being printed */
180 do {
181 if (i > (ENDOFLINE - 5)) {
182 if (i != 999)
183 list_line(buf);
184 sprintf(buf, " ");
185 i = 16;
186 }
187 sprintf(&buf[i], "%5d%c", (baseptr + refptr)->lineno,
188 (baseptr + refptr)->use);
189 refptr = (baseptr + refptr)->link;
190 i += 6;
191 } while (refptr != 0);
192 list_line(buf);
193 }
194
195 /* print trailer */
196 list_line("---------------------------------");
197 list_page();
198}

202static int
203insertsrt(int q, int r)
204{
205 int i, last, next;
206 char *ptr;
207
208 HASHLKP(0, 0);
209
210 for (i = q; i <= r; i++) {
211 /* skip over those records without ref lists */
212 if (XREFLKG(i) == 0)
213 continue;
214 ptr = SYMNAME(i);
215 last = 0;
216 next = HASHLKG(0);
217 while ((next != 0) && (strcmp(ptr, SYMNAME(next)) >= 0)) {
218 last = next;
219 next = HASHLKG(next);
220 }
221 HASHLKP(i, next);
222 HASHLKP(last, i);
223 }
224 return (HASHLKG(0));
225}

229static int
230merge(int q, int r)
231{
232 int i, j, k;
233
234 k = 0;
235 i = q;
236 j = r;
237 while ((i != 0) && (j != 0)) {
238 if (strcmp(SYMNAME(i), SYMNAME(j)) <= 0) {
239 HASHLKP(k, i);
240 k = i;
241 i = HASHLKG(i);
242 } else {
243 HASHLKP(k, j);
244 k = j;
245 j = HASHLKG(j);
246 }
247 }
248 if (i == 0)
249 HASHLKP(k, j);
250 else
251 HASHLKP(k, i);
252
253 return (HASHLKG(0));
254}

258static int
259mergesrt(int low, int high)
260{
261 int mid;
262 int p, q, r;
263#define SORTLIM 16
264
265 if ((high - low) < SORTLIM)
266 p = insertsrt(low, high);
267 else {
268 mid = (low + high) / 2;
269 q = mergesrt(low, mid);
270 r = mergesrt(mid + 1, high);
271 p = merge(q, r);
272 }
273 return (p);
274}

278void
279xref(void)
280{
281 struct memitem *iptr;
282 int sptr;
283 INT index;
284 int i;
285 int liststart, nsyms;
286
287 if (!xrefcnt)
288 return;
289
290 /* rewind file */
291 if (fseek(fd, 0L, 0) == EOF)
292 interr("xref's fseek failed", 0, 4);
293
294 /* allocate space for ref lists */
295 NEW(baseptr, struct memitem, xrefcnt + 1);
296 if (baseptr == NULL)
297 interr("xref couldn't alloc enough space", 0, 4);
298
299 /* fill symbol table's hash field with NULLs (sort links) */
300 /* fill w7 field in symbol table with NULLs (ref list head) */
301 nsyms = stb.stg_avail;
302 for (i = 1; i <= nsyms; i++) {
303 HASHLKP(i, 0);
304 XREFLKP(i, 0);
305 }
306
307 /* read ref info into memory - 0 pos is wasted */
308 if ((fread((char *)(baseptr + 1), sizeof(struct memitem), xrefcnt, fd)) == 0)
309 interr("xref fread failed", xrefcnt, 4);
310
311 /* process in reverse order so lists are in correct order */
312 iptr = baseptr + xrefcnt;
313 index = xrefcnt;
314 while (iptr > baseptr) {
315 sptr = iptr->link;
316 iptr->link = XREFLKG(sptr);
317 XREFLKP(sptr, index--);
318 iptr--;
319 }
320
321 /* sort symbol table on SYMNAMES */
322 liststart = mergesrt(1, nsyms);
323
324 printxref(liststart);
325 fclose(fd);
326
327}
