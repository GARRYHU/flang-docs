260/* Machine Register Information -
261 *
262 * This information is in two pieces:
263 * 1. a structure exists for each machine's register class which defines
264 * the attributes of registers.
265 * These attributes define a register set with the following properties:
266 * 1). a set is just an increasing set of numbers,
267 * 2). scratch registers are allocated in increasing order (towards
268 * the globals),
269 * 3). globals are allocated in decreasing order (towards the
270 * scratch registers).
271 * 4). the scratch set that can be changed by a procedure
272 * [min .. first_global-1]
273 * 5). the scratch set that can be changed by an intrinisic
274 * [min .. intrinsic]
275 *
276 * 2. a structure exists for all of the generic register classes which will
277 * map a register type (macros in registers.h) to the appropriate
278 * machine register class.
279 */
280
281/***** Machine Register Table *****/
282
283typedef struct {
284 char min; /* minimum register # */
285 char max; /* maximum register # */
286 char intrinsic; /* last scratch that can be changed by an intrinsic */
287 const char first_global; /* first register # that can be global
288 * Note that the globals are allocated in increasing
289 * order (first_global down to last_global).
290 */
291 const char end_global; /* absolute last register # that can be global. */
292 /* the following two really define the working set
293 of registers that can be assigned. */
294 char next_global; /* next global register # */
295 char last_global; /* last register # that can be global. */
296 char nused; /* number of global registers assigned */
297 const char mapbase; /* offset in register bit vector where
298 this class of MACH_REGS begins. */
299 const char Class; /* class or type of register. code generator needs
300 to know what kind of registers these represent.
301 'i' (integer), 'f' (float stk), 'x' (float xmm) */
302} MACH_REG;
303
304/***** Register Mapping Table *****/
305
306typedef struct {
307 char max; /* maximum number of registers */
308 char nused; /* number of registers assigned */
309 char joined; /* non-zero value if registers are formed from
310 * multiple machine registers. 1==>next machine
311 * register is used; other values TBD.
312 */
313 int rcand; /* register candidate list */
314 MACH_REG *mach_reg; /* pointer to struct of the actual registers */
315 INT const_flag; /* flag controlling assignment of consts */
316} REG; /* [rtype] */
