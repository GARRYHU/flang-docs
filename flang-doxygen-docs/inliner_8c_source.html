147/*
148 * Return TRUE if the current function can be extracted.
149 */
150LOGICAL
151extractor_possible(void)
152{
153 char *sCurrFunc;
154 FI *pfi;
155 int sptr, std;
156 int dtyp;
157
158 /* Search for a named program unit. */
159 if (pfiStart) {
160 sCurrFunc = SYMNAME(gbl.currsub);
161 for (pfi = pfiStart; pfi; pfi = pfi->pfiNext)
162 if (!strcmp(pfi->sFunc, sCurrFunc))
163 break;
164 if (!pfi) {
165 TRACE1("can't extract: program unit %s not named\n", sCurrFunc);
166 return FALSE;
167 }
168 }
169 if (gbl.rutype != RU_SUBR && gbl.rutype != RU_FUNC) {
170 TRACE0("can't extract: program unit not subroutine or function\n");
171 return FALSE;
172 }
173 if (SYMLKG(gbl.entries) != NOSYM) {
174 TRACE0("can't extract: multiple entry points\n");
175 ccff_info(MSGNEGINLINER, "INL024", gbl.findex, gbl.funcline,
176 "%module%separator%function is not HL inlineable: multiple entry "
177 "points",
178 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
179 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
180 SYMNAME(gbl.currsub), NULL);
181 return FALSE;
182 }
183 if (gbl.arets) {
184 TRACE0("can't extract: multiple returns\n");
185 ccff_info(
186 MSGNEGINLINER, "INL025", gbl.findex, gbl.funcline,
187 "%module%separator%function is not HL inlineable: alternate returns",
188 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "", "separator=%s",
189 gbl.currmod ? "::" : "", "function=%s", SYMNAME(gbl.currsub), NULL);
190 return FALSE;
191 }
192 for (sptr = stb.firstusym; sptr < stb.stg_avail; sptr++) {
193 if (ST_ISVAR(STYPEG(sptr)) && SCOPEG(sptr) != stb.curr_scope)
194 continue;
195 if (STYPEG(sptr) == ST_NML) {
196 TRACE0("can't extract: namelist present\n");
197 ccff_info(MSGNEGINLINER, "INL026", gbl.findex, gbl.funcline,
198 "%module%separator%function is not HL inlineable: namelist "
199 "statements",
200 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
201 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
202 SYMNAME(gbl.currsub), NULL);
203 return FALSE;
204 }
205 if (STYPEG(sptr) == ST_LABEL && FMTPTG(sptr)) {
206 TRACE0("can't extract: format list present\n");
207 ccff_info(
208 MSGNEGINLINER, "INL027", gbl.findex, gbl.funcline,
209 "%module%separator%function is not HL inlineable: format statements",
210 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "", "separator=%s",
211 gbl.currmod ? "::" : "", "function=%s", SYMNAME(gbl.currsub), NULL);
212 return FALSE;
213 }
214 if (ST_ISVAR(STYPEG(sptr)) && SCG(sptr) == SC_STATIC && SAVEG(sptr)) {
215 TRACE0("can't extract: SAVEd local present\n");
216 ccff_info(MSGNEGINLINER, "INL028", gbl.findex, gbl.funcline,
217 "%module%separator%function is not HL inlineable: SAVEd local "
218 "variables",
219 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
220 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
221 SYMNAME(gbl.currsub), NULL);
222 return FALSE;
223 }
224 if (ST_ISVAR(STYPEG(sptr)) && DINITG(sptr)) {
225 TRACE0("can't extract: data initialization present\n");
226 ccff_info(MSGNEGINLINER, "INL029", gbl.findex, gbl.funcline,
227 "%module%separator%function is not HL inlineable: DATA "
228 "initialization",
229 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
230 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
231 SYMNAME(gbl.currsub), NULL);
232 return FALSE;
233 }
234 if (ST_ISVAR(STYPEG(sptr)) && SCG(sptr) == SC_DUMMY && !IGNOREG(sptr)) {
235 dtyp = DTYPEG(sptr);
236 if (DTY(dtyp) == TY_CHAR || DTY(dtyp) == TY_NCHAR)
237 if (dtyp == DT_ASSCHAR || dtyp == DT_ASSNCHAR || dtyp == DT_DEFERCHAR ||
238 dtyp == DT_DEFERNCHAR) {
239 TRACE0("can't extract: assumed/deferred length dummy present\n");
240 ccff_info(MSGNEGINLINER, "INL030", gbl.findex, gbl.funcline,
241 "%module%separator%function is not HL inlineable: "
242 "assumed-length character dummy",
243 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
244 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
245 SYMNAME(gbl.currsub), NULL);
246 return FALSE;
247 }
248 }
249 }
250 if (!XBIT(13, 0x200) && !INMODULEG(gbl.currsub)) {
251 /* only extract module subprograms */
252 return FALSE;
253 }
254 for (std = STD_NEXT(0); std; std = STD_NEXT(std)) {
255 /* whether to allow loops or conditionals */
256 switch (A_TYPEG(STD_AST(std))) {
257 case A_DO:
258 if (!XBIT(13, 0x100)) {
259 ccff_info(MSGNEGINLINER, "INL031", gbl.findex, gbl.funcline,
260 "%module%separator%function is not HL inlineable: DO "
261 "statements disallowed",
262 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
263 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
264 SYMNAME(gbl.currsub), NULL);
265 return FALSE;
266 }
267 break;
268 case A_DOWHILE:
269 ccff_info(MSGNEGINLINER, "INL032", gbl.findex, gbl.funcline,
270 "%module%separator%function is not HL inlineable: DOWHILE "
271 "statements disallowed",
272 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
273 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
274 SYMNAME(gbl.currsub), NULL);
275 return FALSE;
276 case A_IF:
277 case A_IFTHEN:
278 if (!XBIT(13, 0x80)) {
279 ccff_info(MSGNEGINLINER, "INL033", gbl.findex, gbl.funcline,
280 "%module%separator%function is not HL inlineable: IF "
281 "statements disallowed",
282 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
283 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
284 SYMNAME(gbl.currsub), NULL);
285 return FALSE;
286 }
287 break;
288 case A_AIF:
289 ccff_info(MSGNEGINLINER, "INL034", gbl.findex, gbl.funcline,
290 "%module%separator%function is not HL inlineable: arithmetic "
291 "IF statements disallowed",
292 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
293 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
294 SYMNAME(gbl.currsub), NULL);
295 return FALSE;
296 case A_GOTO:
297 ccff_info(MSGNEGINLINER, "INL035", gbl.findex, gbl.funcline,
298 "%module%separator%function is not HL inlineable: GOTO "
299 "statements disallowed",
300 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
301 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
302 SYMNAME(gbl.currsub), NULL);
303 return FALSE;
304 case A_CGOTO:
305 ccff_info(MSGNEGINLINER, "INL036", gbl.findex, gbl.funcline,
306 "%module%separator%function is not HL inlineable: computed "
307 "GOTO statements disallowed",
308 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
309 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
310 SYMNAME(gbl.currsub), NULL);
311 return FALSE;
312 case A_AGOTO:
313 ccff_info(MSGNEGINLINER, "INL037", gbl.findex, gbl.funcline,
314 "%module%separator%function is not HL inlineable: assigned "
315 "GOTO statements disallowed",
316 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
317 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
318 SYMNAME(gbl.currsub), NULL);
319 return FALSE;
320 case A_ASNGOTO:
321 ccff_info(MSGNEGINLINER, "INL038", gbl.findex, gbl.funcline,
322 "%module%separator%function is not HL inlineable: ASSIGN "
323 "statements disallowed",
324 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
325 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
326 SYMNAME(gbl.currsub), NULL);
327 return FALSE;
328 case A_STOP:
329 ccff_info(MSGNEGINLINER, "INL039", gbl.findex, gbl.funcline,
330 "%module%separator%function is not HL inlineable: STOP "
331 "statements disallowed",
332 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
333 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
334 SYMNAME(gbl.currsub), NULL);
335 return FALSE;
336 case A_PAUSE:
337 ccff_info(MSGNEGINLINER, "INL040", gbl.findex, gbl.funcline,
338 "%module%separator%function is not HL inlineable: PAUSE "
339 "statements disallowed",
340 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
341 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
342 SYMNAME(gbl.currsub), NULL);
343 return FALSE;
344 case A_MP_PARALLEL:
345 ccff_info(MSGNEGINLINER, "INL041", gbl.findex, gbl.funcline,
346 "%module%separator%function is not HL inlineable: OpenMP "
347 "parallel section disallowed",
348 "module=%s", gbl.currmod ? SYMNAME(gbl.currmod) : "",
349 "separator=%s", gbl.currmod ? "::" : "", "function=%s",
350 SYMNAME(gbl.currsub), NULL);
351 return FALSE;
352 }
353 }
354
355 return TRUE;
356} /* extractor_possible */

358/*
359 * Extract the current subroutine.
360 */
361void
362extractor(void)
363{
364 FILE *fd;
365 char *sCurrFunc;
366 const char *sCurrHost, *sCurrMod;
367 int iFile, nFile;
368 LE *ple;
369 int iStat;
370 char sExtFile[MAX_FNAME_LEN];
371
372 TRACE0("----------\n");
373 TRACE0("Extractor:\n");
374 if (!sExtDir) {
375 errsev(270);
376 return;
377 }
378 if (!extractor_possible())
379 return;
380 sCurrFunc = SYMNAME(gbl.currsub);
381 if (gbl.internal <= 1 || gbl.outersub == 0) {
382 sCurrHost = ".";
383 } else {
384 sCurrHost = SYMNAME(gbl.outersub);
385 }
386 if (gbl.currmod == 0) {
387 sCurrMod = ".";
388 } else {
389 sCurrMod = SYMNAME(gbl.currmod);
390 }
391
392 /* Get the highest extract file #. */
393 nFile = 0;
394 for (ple = pleStart; ple; ple = ple->pleNext) {
395 iStat = sscanf(ple->sModFile, MODNAME_FMT, &iFile);
396 if (iStat != 1)
397 return;
398 if (iFile > nFile)
399 nFile = iFile;
400 }
401
402 /* Search for an existing library entry. */
403 ple = find_libentry(sCurrMod, sCurrHost, sCurrFunc);
404
405 if (!ple) {
406 /* Create a new library entry. */
407 ple = (LE *)getitem(PERM_AREA, sizeof(LE));
408 if (!ple)
409 ERROR("extractor: can't allocate memory", 3);
410 ple->sFunc = STASH(sCurrFunc, PERM_AREA);
411 ple->sHost = STASH(sCurrHost, PERM_AREA);
412 ple->sMod = STASH(sCurrMod, PERM_AREA);
413 sprintf(sExtFile, MODNAME_FMT, nFile + 1);
414 ple->sModFile = STASH(sExtFile, PERM_AREA);
415 ple->pleNext = pleStart;
416 pleStart = ple;
417 }
418
419 sprintf(sExtFile, "%s/%s", sExtDir, ple->sModFile);
420 fd = fopen(sExtFile, "w");
421 if (!fd) {
422 error(275, 2, gbl.lineno, sExtFile, NULL);
423 return;
424 }
425 export_inline(fd, sCurrFunc, sExtFile);
426}

428void
429extractor_end(void)
430{
431 FILE *fd;
432 char sTOCFile[MAX_FNAME_LEN];
433 LE *ple;
434
435 if (!sExtDir)
436 return;
437
438 /* Create the TOC file. */
439 fndpath("TOC", sTOCFile, MAX_FNAME_LEN, sExtDir);
440
441 fd = fopen(sTOCFile, "w");
442 if (!fd) {
443 error(274, 2, gbl.lineno, sExtDir, NULL);
444 return;
445 }
446
447 /* Write the header. */
448 fprintf(fd, TOC_HEADER_FMT, INLINER_VERSION);
449 fprintf(fd, "\n");
450
451 /* Write out the library entries. */
452 for (ple = pleStart; ple; ple = ple->pleNext) {
453 fprintf(fd, TOC_ENTRY_FMT, ple->sMod, ple->sHost, ple->sFunc,
454 ple->sModFile);
455 fprintf(fd, "\n");
456 }
457 fclose(fd);
458
459 freearea(PERM_AREA);
460}

506/*
507 * Perform inlining on all STDs between stdStart & stdLast repeating
508 * recursively for iLevels.
509 */
510static void
511inline_stds(int stdStart, int stdLast, int iLevels, int level)
512{
513 int std, stdNext;
514 int ast, astNew;
515
516 if (!iLevels)
517 return;
518
519 if (stdLast)
520 stdLast = STD_NEXT(stdLast);
521 for (std = stdStart; std != stdLast; std = stdNext) {
522 stdNext = STD_NEXT(std);
523 if (STD_LINENO(std))
524 gbl.lineno = STD_LINENO(std);
525 ast = STD_AST(std);
526 astNew = inline_ast(std, ast, iLevels, level);
527 if (astNew == ast)
528 continue;
529 if (!astNew) {
530 delete_stmt(std);
531 continue;
532 }
533 STD_AST(std) = astNew;
534 A_STDP(astNew, std);
535 }
536}

538/*
539 * Inline any function calls within ast, and repeat on the inlined statements
540 * iLevels times. Add inlined statements prior to
541 * std. Replace any inlined call with a new temp. After replacement, return
542 * the new AST. Return 0 if std will be deleted after inlining.
543 */
544static int
545inline_ast(int std, int ast, int iLevels, int level)
546{
547 int astNewl, astNewr, astNew, vastSubs[7], astNewu, astNews, astNewc;
548 int asd, sptrEntry;
549 int nsubs, sub;
550 int sptrRet;
551 INT argt, argtNew;
552 int arg, nargs;
553 LOGICAL bChanged;
554
555 if (ast == 0)
556 return 0;
557 /* Explicit recursion is used here instead of a call to ast_traverse()
558 * because ast_traverse() is called by rewrite_inlined_args() to replace
559 * dummy parameters with actual. */
560 switch (A_TYPEG(ast)) {
561 case A_NULL:
562 case A_ID:
563 case A_CNST:
564 case A_LABEL:
565 case A_CMPLXC:
566 case A_MEM:
567 return ast;
568 case A_BINOP:
569 astNewl = inline_ast(std, A_LOPG(ast), iLevels, level);
570 astNewr = inline_ast(std, A_ROPG(ast), iLevels, level);
571 if (astNewl == A_LOPG(ast) && astNewr == A_ROPG(ast))
572 return ast;
573 astNew = mk_binop(A_OPTYPEG(ast), astNewl, astNewr, A_DTYPEG(ast));
574 return astNew;
575 case A_UNOP:
576 astNewl = inline_ast(std, A_LOPG(ast), iLevels, level);
577 if (astNewl == A_LOPG(ast))
578 return ast;
579 astNew = mk_unop(A_OPTYPEG(ast), astNewl, A_DTYPEG(ast));
580 return astNew;
581 case A_CONV:
582 astNewl = inline_ast(std, A_LOPG(ast), iLevels, level);
583 if (astNewl == A_LOPG(ast))
584 return ast;
585 astNew = mk_convert(astNewl, A_DTYPEG(ast));
586 return astNew;
587 case A_PAREN:
588 astNewl = inline_ast(std, A_LOPG(ast), iLevels, level);
589 if (astNewl == A_LOPG(ast))
590 return ast;
591 astNew = mk_paren(astNewl, A_DTYPEG(ast));
592 return astNew;
593 case A_SUBSCR:
594 asd = A_ASDG(ast);
595 nsubs = ASD_NDIM(asd);
596 bChanged = FALSE;
597 for (sub = 0; sub < nsubs; sub++) {
598 vastSubs[sub] = inline_ast(std, ASD_SUBS(asd, sub), iLevels, level);
599 bChanged |= (vastSubs[sub] != ASD_SUBS(asd, sub));
600 }
601 if (!bChanged)
602 return ast;
603 astNew = mk_subscr(A_LOPG(ast), vastSubs, nsubs, A_DTYPEG(ast));
604 return astNew;
605 case A_SUBSTR:
606 astNewl = inline_ast(std, A_LEFTG(ast), iLevels, level);
607 astNewr = inline_ast(std, A_RIGHTG(ast), iLevels, level);
608 if (astNewl == A_LEFTG(ast) && astNewr == A_RIGHTG(ast))
609 return ast;
610 astNew = mk_substr(A_LOPG(ast), astNewl, astNewr, A_DTYPEG(ast));
611 return astNew;
612 case A_TRIPLE:
613 astNewl = inline_ast(std, A_LBDG(ast), iLevels, level);
614 astNewu = inline_ast(std, A_UPBDG(ast), iLevels, level);
615 astNews = inline_ast(std, A_STRIDEG(ast), iLevels, level);
616 if (astNewl == A_LBDG(ast) && astNewu == A_UPBDG(ast) &&
617 astNews == A_STRIDEG(ast))
618 return ast;
619 astNew = mk_triple(astNewl, astNewu, astNews);
620 return astNew;
621 case A_FUNC:
622 case A_INTR:
623 case A_CALL:
624 case A_ICALL:
625 argt = A_ARGSG(ast);
626 nargs = A_ARGCNTG(ast);
627 argtNew = mk_argt(nargs);
628 bChanged = FALSE;
629 for (arg = 0; arg < nargs; arg++) {
630 ARGT_ARG(argtNew, arg) =
631 inline_ast(std, ARGT_ARG(argt, arg), iLevels, level);
632 if (ARGT_ARG(argtNew, arg) != ARGT_ARG(argt, arg))
633 bChanged = TRUE;
634 }
635 if (bChanged == FALSE) {
636 unmk_argt(nargs);
637 astNew = ast;
638 } else {
639 astNew = mk_func_node(A_TYPEG(ast), A_LOPG(ast), nargs, argtNew);
640 A_SHAPEP(astNew, A_SHAPEG(ast));
641 A_DTYPEP(astNew, A_DTYPEG(ast));
642 if (A_TYPEG(ast) == A_INTR || A_TYPEG(ast) == A_ICALL) {
643 A_OPTYPEP(astNew, A_OPTYPEG(ast));
644 }
645 }
646 if (A_TYPEG(ast) == A_INTR || A_TYPEG(ast) == A_ICALL) {
647 return astNew;
648 }
649#if DEBUG
650 if (flg.x[17] != 0 && ninlined >= flg.x[17])
651 return astNew;
652#endif
653 if (A_TYPEG(ast) == A_CALL) {
654 bChanged = inline_func(std, astNew, iLevels, level, &sptrEntry);
655 if (bChanged)
656 return 0;
657 else
658 return astNew;
659 }
660 bChanged = inline_func(std, astNew, iLevels, level, &sptrEntry);
661 if (!bChanged)
662 return astNew;
663 sptrRet = FVALG(sptrEntry);
664 assert(sptrRet, "inline_ast: function value not found", std, 4);
665 SCP(sptrRet, SC_LOCAL); /* ensures declaration's generation */
666 astNew = mk_id(sptrRet);
667 return astNew;
668 case A_ASN:
669 astNewr = inline_ast(std, A_SRCG(ast), iLevels, level);
670 astNewl = inline_ast(std, A_DESTG(ast), iLevels, level);
671 if (astNewr == A_SRCG(ast) && astNewl == A_DESTG(ast))
672 return ast;
673 astNew = mk_assn_stmt(astNewl, astNewr, A_DTYPEG(ast));
674 return astNew;
675 case A_IF:
676 case A_IFTHEN:
677 astNewl = inline_ast(std, A_IFEXPRG(ast), iLevels, level);
678 if (astNewl == A_IFEXPRG(ast))
679 return ast;
680 astNew = mk_stmt(A_TYPEG(ast), 0);
681 A_IFEXPRP(astNew, astNewl);
682 A_IFSTMTP(astNew, A_IFSTMTG(ast));
683 return astNew;
684 case A_DO:
685 astNewl = inline_ast(std, A_M1G(ast), iLevels, level);
686 astNewr = inline_ast(std, A_M2G(ast), iLevels, level);
687 astNews = inline_ast(std, A_M3G(ast), iLevels, level);
688 astNewc = inline_ast(std, A_M4G(ast), iLevels, level);
689 if (astNewl == A_M1G(ast) && astNewr == A_M2G(ast) &&
690 astNews == A_M3G(ast) && astNewc == A_M4G(ast))
691 return ast;
692 astNew = mk_stmt(A_TYPEG(ast), 0);
693 A_DOLABP(astNew, A_DOLABG(ast));
694 A_DOVARP(astNew, A_DOVARG(ast));
695 A_M1P(astNew, astNewl);
696 A_M2P(astNew, astNewr);
697 A_M3P(astNew, astNews);
698 A_M4P(astNew, astNewc);
699 return astNew;
700 default:
701 return ast;
702 }
703}

706 * if we inline foo into bar, and foo calls ugh, and bar has another
707 * symbol for ugh, reuse bar's symbol for ugh instead
708 */
709static int use_old_subprogram_limits[2] = {0, 0};
710static void
711use_old_subprograms(int oldsymavl)
712{
713 int sptr, tptr;
714 use_old_subprogram_limits[0] = oldsymavl;
715 use_old_subprogram_limits[1] = stb.stg_avail - 1;
716 for (sptr = oldsymavl; sptr < stb.stg_avail; ++sptr) {
717 switch (STYPEG(sptr)) {
718 case ST_PROC:
719 /* see if there is an identical function in the caller */
720 for (tptr = HASHLKG(sptr); tptr > NOSYM; tptr = HASHLKG(tptr)) {
721 if (NMPTRG(tptr) != NMPTRG(sptr))
722 continue;
723 if (STYPEG(tptr) != ST_PROC)
724 continue;
725 if (tptr >= sptrHigh)
726 continue;
727 if (INMODULEG(tptr) != INMODULEG(sptr))
728 continue;
729 if (INMODULEG(tptr)) {
730 if (NMPTRG(ENCLFUNCG(tptr)) != NMPTRG(ENCLFUNCG(sptr)))
731 continue;
732 if (SCOPEG(sptr) && STYPEG(SCOPEG(sptr)) == ST_ALIAS &&
733 SCOPEG(SCOPEG(sptr)) &&
734 STYPEG(SCOPEG(SCOPEG(sptr))) == ST_MODULE) {
735 if (!SCOPEG(tptr) || STYPEG(SCOPEG(tptr)) != ST_ALIAS ||
736 !SCOPEG(SCOPEG(sptr)) ||
737 STYPEG(SCOPEG(SCOPEG(sptr))) != ST_MODULE) {
738 continue;
739 }
740 if (NMPTRG(SCOPEG(SCOPEG(tptr))) != NMPTRG(SCOPEG(SCOPEG(sptr))))
741 continue;
742 }
743 }
744 if (HCCSYMG(tptr))
745 continue;
746 /* tptr and sptr are two functions with the same name */
747 FUNCLINEP(sptr, tptr);
748 IGNOREP(sptr, 1);
749 DPDSCP(sptr, 0); /* don't generate interface block */
750 break;
751 }
752 break;
753 default:;
754 }
755 }

799/*
800 * If CALL/FUNC ast residing in std has been extracted, inline the called
801 * program unit, and continue inlining iLevels times.
802 * Return TRUE if inlining performed.
803 */
804static LOGICAL
805inline_func(int std, int ast, int iLevels, int level, int *psptrEntry)
806{
807 int sptrCall, sptrEntry;
808 int sptrMod, sptrHost;
809 char *sFunc;
810 const char *sMod, *sHost, *currMod;
811 FI *pfi;
812 LE *ple;
813 char sExtFile[MAX_FNAME_LEN];
814 FILE *fd;
815 int stdstart, stdfinal, sptrHighCopy;
816
817#if DEBUG
818 if (flg.x[22] && gbl.func_count != flg.x[22])
819 return FALSE;
820#endif
821 assert(A_TYPEG(ast) == A_FUNC || A_TYPEG(ast) == A_CALL,
822 "inline_func: bad type", ast, 4);
823 assert(A_TYPEG(A_LOPG(ast)) == A_ID, "inline_func: bad call type", ast, 4);
824 remove_alias(std, ast);
825 sptrCall = A_SPTRG(A_LOPG(ast));
826 sFunc = SYMNAME(sptrCall);
827 if (!INTERNALG(sptrCall)) {
828 sptrHost = sptrCall;
829 sHost = ".";
830 } else {
831 sptrHost = SCOPEG(sptrCall);
832 sHost = SYMNAME(sptrHost);
833 }
834 /* see if this is a module subprogram */
835 currMod = NULL;
836 sMod = ".";
837 if (INMODULEG(sptrHost)) {
838 for (sptrMod = sptrHost; sptrMod > NOSYM; sptrMod = SCOPEG(sptrMod)) {
839 if (STYPEG(sptrMod) == ST_MODULE) {
840 sMod = SYMNAME(sptrMod);
841 currMod = sMod;
842 break;
843 }
844 if (SCOPEG(sptrMod) == sptrMod) {
845 break;
846 }
847 }
848 }
849
850 if (pfiStart) {
851 /* Determine if sptrCall is a function specified for inlining. */
852 for (pfi = pfiStart; pfi; pfi = pfi->pfiNext)
853 if (!strcmp(pfi->sFunc, sFunc))
854 break;
855 if (!pfi)
856 return FALSE;
857 }
858
859 ple = find_libentry(sMod, sHost, sFunc);
860 if (!ple) {
861 return FALSE;
862 }
863
864 if (!XBIT(13, 0x400) && missing_optionals(sptrCall, ast)) {
865 ccff_info(
866 MSGNEGINLINER, "INL044", gbl.findex, gbl.funcline,
867 "%module%separator%function not inlined: missing OPTIONAL arguments",
868 "module=%s", currMod ? SYMNAME(sptrMod) : "", "separator=%s",
869 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
870 return FALSE;
871 }
872
873 sprintf(sExtFile, "%s/%s", sExtDir, ple->sModFile);
874 fd = fopen(sExtFile, "r");
875 if (!fd)
876 error(275, 4, gbl.lineno, sExtFile, NULL);
877 stdEnd = STD_PREV(0);
878 sptrHighCopy = sptrHigh = stb.stg_avail;
879 if (import_inline(fd, sExtFile)) {
880 ccff_info(
881 MSGNEGINLINER, "INL045", gbl.findex, gbl.funcline,
882 "%module%separator%function not inlined: conflicting declarations",
883 "module=%s", currMod ? SYMNAME(sptrMod) : "", "separator=%s",
884 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
885 return FALSE;
886 }
887
888 sptrEntry = find_entry(A_SPTRG(A_LOPG(ast)));
889
890 if (STD_PREV(0) == stdEnd) {
891 ccff_info(MSGNEGINLINER, "INL047", gbl.findex, gbl.funcline,
892 "%module%separator%function not inlined: empty subprogram",
893 "module=%s", currMod ? SYMNAME(sptrMod) : "", "separator=%s",
894 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
895 remove_inlined_symbols(sptrHigh);
896 return FALSE;
897 }
898 if (aliased_args(sptrEntry, ast)) {
899 ccff_info(MSGNEGINLINER, "INL048", gbl.findex, gbl.funcline,
900 "%module%separator%function not inlined: argument aliasing",
901 "module=%s", currMod ? SYMNAME(sptrMod) : "", "separator=%s",
902 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
903 remove_inlined_stds();
904 remove_inlined_symbols(sptrHigh);
905 return FALSE;
906 }
907
908 use_old_subprograms(sptrHigh);
909
910 modify_inlined_stds(sptrEntry, ast, STD_NEXT(stdEnd), STD_PREV(0));
911 if (STD_PREV(0) == stdEnd) {
912 ccff_info(MSGNEGINLINER, "INL049", gbl.findex, gbl.funcline,
913 "%module%separator%function not inlined: no statements",
914 "module=%s", currMod ? SYMNAME(sptrMod) : "", "separator=%s",
915 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
916 remove_inlined_symbols(sptrHigh);
917 return FALSE;
918 }
919
920#if DEBUG
921 if (flg.x[18] != 0 && nskip < flg.x[18]) {
922 ++nskip;
923 ccff_info(MSGNEGINLINER, "INL050", gbl.findex, gbl.funcline,
924 "%module%separator%function not inlined: skipped", "module=%s",
925 currMod ? SYMNAME(sptrMod) : "", "separator=%s",
926 currMod ? "::" : "", "function=%s", SYMNAME(sptrCall), NULL);
927 remove_inlined_stds();
928 remove_inlined_symbols(sptrHigh);
929 return FALSE;
930 }
931#endif
932
933 /* Insert the inlined STDs before std. */
934 stdstart = STD_NEXT(stdEnd);
935 stdfinal = STD_PREV(0);
936 if (XBIT(49, 0x4000)) {
937 /* line-level profiling */
938 int std;
939 for (std = stdstart; std; std = STD_NEXT(std)) {
940 STD_LINENO(std) = 0;
941 if (std == stdfinal)
942 break;
943 }
944 }
945 STD_NEXT(stdEnd) = 0;
946 STD_PREV(0) = stdEnd;
947 STD_PREV(stdstart) = STD_PREV(std);
948 STD_NEXT(STD_PREV(std)) = stdstart;
949 STD_NEXT(stdfinal) = std;
950 STD_PREV(std) = stdfinal;
951
952 /* change inlined ST_ENTRY */
953 erase_entry(sptrHigh);
954
955 if (XBIT(0, 8)) {
956 ccff_info(
957 MSGNEGINLINER, "INL051", gbl.findex, gbl.funcline,
958 "%module%separator%function inlined", "module=%s",
959 sptrHost == sptrCall && INMODULEG(sptrHost) ? SYMNAME(sptrMod) : "",
960 "separator=%s", sptrHost == sptrCall && INMODULEG(sptrHost) ? "::" : "",
961 "function=%s", SYMNAME(sptrCall), NULL);
962 }
963 ++ninlined;
964 inline_stds(stdstart, stdfinal, iLevels - 1, level + 1);
965
966 if (psptrEntry)
967 *psptrEntry = sptrEntry;
968 return TRUE;
969}

972 * Modify inlined STDs between stdstart & stdlast for call AST ast.
973 * Remove the END and RETURN statements.
974 * Move labels on branching statements to CONTINUE statements.
975 * If an argument is INTENT IN, or the actual argument is not an
976 * identifier or subscript, copy the actual to the dummy parameter
977 * at the beginning of the inlined code.
978 * If an argument is INTENT INOUT/OUT and the actual argument is an identifer
979 * or subscript, replace all occurrences of the dummy parameter with the
980 * actual.
981 */
982static void
983modify_inlined_stds(int sptrEntry, int ast, int stdstart, int stdlast)

986 case A_CALL:
987 case A_FUNC:
988 break;

990 interr("modify_inlined_stds: unrecognized CALL", stdstart, 4);

993 /* Copy expressions and arguments with INTENT IN to dummy parameters. */
994 ast = copy_inargs(sptrEntry, ast, stdstart, stdlast);
995
996 /* Remove RETURN & END statements. */
997 remove_returns(stdstart, stdlast);
998
999 /* Move labels on branching constructs. */
1000 move_labels(stdstart, stdlast);
1001
1002 /* Replace dummy parameters with INTENT OUT or INOUT with identifier
1003 * arguments. */
1004 rewrite_inlined_args(ast, sptrEntry, stdstart, stdlast);
1005
1006 /* Allocate adjustable arrays. */
1007 allocate_adjarrs(stdstart, stdlast);
1008}

1028/*
1029 * Copy arguments within astCall that have INTENT IN, or are expressions,
1030 * to new variables, which are allocated
1031 * prior to stdstart, and deallocated after stdlast. Replace the original
1032 * variables within call/func astCall with the new variables, and return
1033 * the new call AST.
1034 */
1035static int
1036copy_inargs(int sptrEntry, int astCall, int stdstart, int stdlast)
1037{
1038 int dscend = DPDSCG(sptrEntry) + PARAMCTG(sptrEntry);
1039 int dsc;
1040 int sptrCall, sptrDummy, sptrCopy, sptrBnd;
1041 int nargs, arg;
1042 int argtNew;
1043 int astArg, astCopy, ast, astl, astu, asts, vastSubs[7], astAlloc;
1044 int shd;
1045 int dtyp;
1046 int ndims, dim;
1047 ADSC *adCopy;
1048
1049 assert(A_TYPEG(astCall) == A_CALL || A_TYPEG(astCall) == A_FUNC,
1050 "copy_inargs: call not found", astCall, 4);
1051 sptrCall = A_SPTRG(A_LOPG(astCall));
1052 if (PARAMCTG(sptrEntry) != A_ARGCNTG(astCall)) {
1053 error(271, 2, gbl.lineno, SYMNAME(sptrCall), NULL);
1054 remove_inlined_stds();
1055 return astCall;
1056 }
1057
1058 /* Find an argument that needs to be copied. */
1059 nargs = A_ARGCNTG(astCall);
1060 for (arg = 0; arg < nargs; arg++)
1061 if (make_arg_copy(sptrEntry, astCall, arg))
1062 break;
1063 if (arg == nargs)
1064 return astCall; /* No copies needed. */
1065
1066 argtNew = mk_argt(nargs); /* Create a new argument descriptor. */
1067 arg = 0;
1068 for (dsc = DPDSCG(sptrEntry); dsc < dscend; dsc++, arg++) {
1069 astArg = ARGT_ARG(A_ARGSG(astCall), arg);
1070 ARGT_ARG(argtNew, arg) = astArg;
1071 if (!make_arg_copy(sptrEntry, astCall, arg))
1072 continue;
1073 sptrDummy = aux.dpdsc_base[dsc];
1074 if (A_TYPEG(astArg) == A_SUBSCR && !A_SHAPEG(astArg) &&
1075 DTY(DTYPEG(sptrDummy)) == TY_ARRAY)
1076 astArg = promote_assumsz_arg(sptrDummy, astArg);
1077 shd = A_SHAPEG(astArg);
1078 if (shd) {
1079 dtyp = DDTG(A_DTYPEG(astArg));
1080 ndims = SHD_NDIM(shd);
1081 sptrCopy = sym_get_array(SYMNAME(sptrDummy), "inl", dtyp, ndims);
1082 adCopy = AD_DPTR(DTYPEG(sptrCopy));
1083 for (dim = 0; dim < ndims; dim++) {
1084 /* Assign 1 to a new lower bound temp. */
1085 sptrBnd = sym_get_scalar("l", "inl", DT_INT);
1086 astl = mk_id(sptrBnd);
1087 ast = mk_assn_stmt(astl, astb.i1, DT_INT);
1088 add_stmt_before(ast, stdstart);
1089 AD_LWBD(adCopy, dim) = AD_LWAST(adCopy, dim) = astl;
1090
1091 /* Compute the size of dimension dim. */
1092 ast = mk_binop(OP_SUB, SHD_UPB(shd, dim), SHD_LWB(shd, dim), DT_INT);
1093 asts = SHD_STRIDE(shd, dim);
1094 if (!asts)
1095 asts = astb.i1;
1096 ast = mk_binop(OP_ADD, asts, ast, DT_INT);
1097 if (asts != astb.i1)
1098 ast = mk_binop(OP_DIV, ast, asts, DT_INT);
1099
1100 /* Assign the computed size to a new upper bound temp. */
1101 sptrBnd = sym_get_scalar("u", "inl", DT_INT);
1102 astu = mk_id(sptrBnd);
1103 ast = mk_assn_stmt(astu, ast, DT_INT);
1104 add_stmt_before(ast, stdstart);
1105 AD_UPBD(adCopy, dim) = AD_UPAST(adCopy, dim) = astu;
1106
1107 vastSubs[dim] = mk_triple(astl, astu, 0);
1108 }
1109 /* Create ALLOCATE & DEALLOCATE statements for the new array. */
1110 astCopy = mk_id(sptrCopy);
1111 ast = mk_subscr(astCopy, vastSubs, ndims, DTYPEG(sptrCopy));
1112 astAlloc = mk_stmt(A_ALLOC, 0);
1113 A_TKNP(astAlloc, TK_ALLOCATE);
1114 A_LOPP(astAlloc, 0);
1115 A_SRCP(astAlloc, ast);
1116 add_stmt_before(astAlloc, stdstart);
1117 astAlloc = mk_stmt(A_ALLOC, 0);
1118 A_TKNP(astAlloc, TK_DEALLOCATE);
1119 A_LOPP(astAlloc, 0);
1120 A_SRCP(astAlloc, astCopy);
1121 add_stmt_after(astAlloc, stdlast);
1122 } else {
1123 sptrCopy = sym_get_scalar(SYMNAME(sptrDummy), "inl", A_DTYPEG(astArg));
1124 astCopy = mk_id(sptrCopy);
1125 }
1126 /* Create a statement for copying the actual argument to the temp. */
1127 ast = mk_assn_stmt(astCopy, astArg, A_DTYPEG(astArg));
1128 add_stmt_before(ast, stdstart);
1129 ARGT_ARG(argtNew, arg) = astCopy;
1130 }
1131 /* Create a new CALL/FUNC AST. */
1132 ast = mk_func_node(A_TYPEG(astCall), A_LOPG(astCall), nargs, argtNew);
1133 A_SHAPEP(ast, A_SHAPEG(astCall));
1134 A_DTYPEP(ast, A_DTYPEG(astCall));
1135 return ast;
1136}

1166static void
1167remove_inlined_symbols(int oldsymavl)
1168{
1169 int sptr, hashval, len;
1170 char *np;
1171 INT V[4];
1172 for (sptr = stb.stg_avail - 1; sptr >= oldsymavl; --sptr) {
1173 hashval = -1;
1174 switch (STYPEG(sptr)) {
1175 case ST_CONST:
1176 switch (DTY(DTYPEG(sptr))) {
1177 case TY_BINT:
1178 case TY_SINT:
1179 case TY_INT:
1180 case TY_BLOG:
1181 case TY_SLOG:
1182 case TY_LOG:
1183 case TY_WORD:
1184 case TY_REAL:
1185 case TY_DWORD:
1186 case TY_DBLE:
1187 case TY_CMPLX:
1188 case TY_INT8:
1189 case TY_LOG8:
1190 V[0] = CONVAL1G(sptr);
1191 V[1] = CONVAL2G(sptr);
1192 hashval = HASH_CON(V);
1193 if (hashval < 0)
1194 hashval = -hashval;
1195 break;
1196 case TY_QUAD:
1197 case TY_DCMPLX:
1198 case TY_QCMPLX:
1199 case TY_HOLL:
1200 case TY_NCHAR:
1201 V[0] = CONVAL1G(0);

1289/*
1290 * Promote an actual assumed-size array argument to an array section, and
1291 * return the new AST.
1292 * Expand 'n' dimensions, where 'n' is the number of dimensions of
1293 * sptrDummy. Also, if the expanded dimension is the last assumed-size
1294 * dimension, check that sptrDummy is also assumed-size
1295 */
1296static int
1297promote_assumsz_arg(int sptrDummy, int astArg)
1298{
1299 int asd;
1300 int nsubs, sub;
1301 int astl, astu, vastSubs[7], ast;
1302 int dtyp;
1303 int dtypeDummy, nsubsDummy, sptr, dtype;
1304
1305 if (A_TYPEG(astArg) != A_SUBSCR)
1306 return astArg;
1307 if (A_SHAPEG(astArg))
1308 return astArg;
1309 dtypeDummy = DTYPEG(sptrDummy);
1310 if (DTY(dtypeDummy) != TY_ARRAY)
1311 return astArg;
1312 nsubsDummy = ADD_NUMDIM(dtypeDummy);
1313 asd = A_ASDG(astArg);
1314 nsubs = ASD_NDIM(asd);
1315 if (nsubsDummy > nsubs)
1316 return astArg;
1317 for (sub = 0; sub < nsubs; sub++)
1318 vastSubs[sub] = ASD_SUBS(asd, sub);
1319 sptr = memsym_of_ast(astArg);
1320 dtype = DTYPEG(sptr);
1321 for (sub = 0; sub < nsubsDummy; ++sub) {
1322 astl = ASD_SUBS(asd, sub);
1323 /* subscripts must be equal to lower bound, except last one */
1324 if (astl != ADD_LWAST(dtype, sub) && sub != nsubsDummy - 1)
1325 return astArg;
1326 astu = ADD_UPAST(dtype, sub);
1327 if (astu != 0) {
1328 vastSubs[sub] = mk_triple(astl, astu, 0);
1329 } else {
1330 /* had better be last dimension of actual */
1331 if (sub != nsubs - 1)
1332 return astArg;
1333 /* actual had better be assumed-size */
1334 if (!ADD_ASSUMSZ(dtype))
1335 return astArg;
1336 /* dummy had better be assumed-size */
1337 if (!ADD_ASSUMSZ(dtypeDummy))
1338 return astArg;
1339 vastSubs[sub] = mk_triple(astl, astu, 0);
1340 }
1341 }
1342 dtyp = get_array_dtype(nsubsDummy, A_DTYPEG(astArg));
1343 ast = mk_subscr(A_LOPG(astArg), vastSubs, nsubs, dtyp);
1344 return ast;
1345}

1362/*
1363 * Create an ALLOCATE statement before stdstart for array sptr, and a
1364 * DEALLOCATE statement for the array after stdlast.
1365 */
1366static void
1367allocate_array(int sptr, int stdstart, int stdlast)
1368{
1369 ADSC *ad;
1370 int sptrBnd;
1371 int vastSubs[7], astArr, ast, astAlloc, astl, astu;
1372 int std1 = STD_PREV(stdstart), std2 = stdlast;
1373 int ndims, dim;
1374
1375 if (STYPEG(sptr) != ST_ARRAY)
1376 return;
1377
1378 ad = AD_DPTR(DTYPEG(sptr));
1379 ndims = AD_NUMDIM(ad);
1380 for (dim = 0; dim < ndims; dim++) {
1381 if (AD_LWBD(ad, dim) && A_TYPEG(AD_LWAST(ad, dim)) == A_ID) {
1382 /* Explicitly assign bound expression to the lower bound. */
1383 ast = mk_assn_stmt(AD_LWAST(ad, dim), AD_LWBD(ad, dim), DT_INT);
1384 std1 = add_stmt_after(ast, std1);
1385 }
1386
1387 /* Create a new array bound. */
1388 sptrBnd = sym_get_scalar("l", "inl", DT_INT);
1389 astl = mk_id(sptrBnd);
1390 ast = mk_assn_stmt(mk_id(sptrBnd), AD_LWAST(ad, dim), DT_INT);
1391 std1 = add_stmt_after(ast, std1);
1392 AD_LWAST(ad, dim) = astl;
1393 AD_LWBD(ad, dim) = astl;
1394 if (AD_UPBD(ad, dim) && A_TYPEG(AD_UPAST(ad, dim)) == A_ID) {
1395 /* Explicitly assign bound expression to the upper bound. */
1396 ast = mk_assn_stmt(AD_UPAST(ad, dim), AD_UPBD(ad, dim), DT_INT);
1397 std1 = add_stmt_after(ast, std1);
1398 }
1399
1400 /* Create a new array bound. */
1401 sptrBnd = sym_get_scalar("u", "inl", DT_INT);
1402 astu = mk_id(sptrBnd);
1403 ast = mk_assn_stmt(mk_id(sptrBnd), AD_UPAST(ad, dim), DT_INT);
1404 std1 = add_stmt_after(ast, std1);
1405 AD_UPAST(ad, dim) = astu;
1406 AD_UPBD(ad, dim) = astu;
1407 AD_EXTNTAST(ad, dim) = mk_extent(astl, astu, dim);
1408 vastSubs[dim] = mk_triple(astl, astu, 0);
1409 }
1410 astArr = mk_id(sptr);
1411 ast = mk_subscr(astArr, vastSubs, ndims, DTYPEG(sptr));
1412 astAlloc = mk_stmt(A_ALLOC, 0);
1413 A_TKNP(astAlloc, TK_ALLOCATE);
1414 A_LOPP(astAlloc, 0);
1415 A_SRCP(astAlloc, ast);
1416 std1 = add_stmt_after(astAlloc, std1);
1417 astAlloc = mk_stmt(A_ALLOC, 0);
1418 A_TKNP(astAlloc, TK_DEALLOCATE);
1419 A_LOPP(astAlloc, 0);
1420 A_SRCP(astAlloc, astArr);
1421 std2 = add_stmt_after(astAlloc, std2);
1422 ALLOCP(sptr, TRUE);
1423 ADJARRP(sptr, FALSE);
1424 AD_DEFER(ad) = TRUE;
1425 AD_ADJARR(ad) = FALSE;
1426}

1428/*
1429 * Remove RETURN & END statements within STDs between stdstart & stdlast.
1430 * Set line #s on all STDs.
1431 */
1432static void
1433remove_returns(int stdstart, int stdlast)
1434{

1437 int sptrEnd;
1438 LOGICAL bEndFound = FALSE;
1439
1440 /* Create a new label for the END statement. */
1441 sptrEnd = getlab();
1442 astLbl = mk_label(sptrEnd);

1444 for (std = stdstart; std != STD_NEXT(stdlast); std = STD_NEXT(std)) {
1445 STD_LINENO(std) = gbl.lineno;

1447 switch (A_TYPEG(ast)) {
1448 case A_RETURN:
1449 /* Create a GOTO statement to the END-statement's new label. */

1460 A_STDP(astStmt, std);

1463 break;
1464 }
1465 }
1466 assert(!RFCNTG(sptrEnd) || bEndFound, "remove_returns: no END found",
1467 stdstart, 4);

1523/*
1524 * Within ast, replace every occurrence of a sub-expression containing
1525 * a dummy parameter with an expression derived from the corresponding
1526 * actual parameter.
1527 */
1528static int
1529replace_parms(int ast, void *extra_arg)
1530{
1531 int iar;
1532 int astLop, astNew, astSS, astSub, vastSubs[7], ast1;
1533 int sptr, dtypeOld, dtypeNew, memOld, memNew, astMemNew;
1534 int asd, asdSS;
1535 int ndims, dim;
1536 LOGICAL bChanged;
1537
1538 switch (A_TYPEG(ast)) {
1539 case A_ID:
1540 sptr = A_SPTRG(ast);
1541 for (iar = 1; iar < iaravail; iar++)
1542 if (sptr == AR_SPTRDUMMY(iar))
1543 break;
1544 if (iar < iaravail) {
1545 astNew = replace_arg(ast, AR_ASTACT(iar));
1546 ast_replace(ast, astNew);
1547 } else {
1548 if (STYPEG(sptr) == ST_PROC) {
1549 if (!HCCSYMG(sptr) && IGNOREG(sptr) &&
1550 sptr >= use_old_subprogram_limits[0] &&
1551 sptr <= use_old_subprogram_limits[1]) {
1552 int tptr;
1553 tptr = FUNCLINEG(sptr);
1554 astNew = mk_id(tptr);
1555 ast_replace(ast, astNew);
1556 } else if (IGNOREG(sptr)) {
1557 /* reset IGNORE flag, we have inlined the call */
1558 IGNOREP(sptr, 0);
1559 }
1560 }
1561 }
1562 return TRUE;
1563 case A_MEM:
1564 astLop = A_PARENTG(ast);
1565 ast_traverse(astLop, replace_parms, NULL, NULL);
1566 astNew = ast_rewrite(astLop);
1567 if (astNew == astLop) {
1568 ast_replace(ast, ast);
1569 } else {
1570 /* we have to find the corresponding member name in any new datatype */
1571 dtypeOld = A_DTYPEG(astLop);
1572 dtypeNew = A_DTYPEG(astNew);
1573 if (DTY(dtypeOld) == TY_ARRAY)
1574 dtypeOld = DTY(dtypeOld + 1);
1575 if (DTY(dtypeNew) == TY_ARRAY)
1576 dtypeNew = DTY(dtypeNew + 1);
1577 if (dtypeOld == dtypeNew) {
1578 astMemNew = A_MEMG(ast);
1579 memNew = A_SPTRG(astMemNew);
1580 } else {
1581 sptr = A_SPTRG(A_MEMG(ast));
1582 if (DTY(dtypeOld) != DTY(dtypeNew) || DTY(dtypeOld) != TY_DERIVED) {
1583 interr("replace_parms: unmatched derived types", ast, 4);
1584 }
1585 for (memOld = DTY(dtypeOld + 1), memNew = DTY(dtypeNew + 1);
1586 memOld > NOSYM && memNew > NOSYM;
1587 memOld = SYMLKG(memOld), memNew = SYMLKG(memNew)) {
1588 if (memOld == sptr)
1589 break;
1590 }
1591 if (memOld <= NOSYM || memNew <= NOSYM ||
1592 ADDRESSG(memOld) != ADDRESSG(memNew)) {
1593 interr("replace_parms: unmatched derived type members", ast, 4);
1594 }
1595 astMemNew = mk_id(memNew);
1596 }
1597 astNew = mk_member(astNew, astMemNew, DTYPEG(memNew));
1598 ast_replace(ast, astNew);
1599 }
1600 return TRUE;
1601
1602 case A_SUBSCR:
1603 asd = A_ASDG(ast);
1604 bChanged = FALSE;
1605 ndims = ASD_NDIM(asd);
1606 for (dim = 0; dim < ndims; dim++) {
1607 astSub = ASD_SUBS(asd, dim);
1608 ast_traverse(astSub, replace_parms, NULL, NULL);
1609 astNew = ast_rewrite(astSub);
1610 vastSubs[dim] = astNew;
1611 bChanged |= astNew != astSub;
1612 }
1613 astLop = A_LOPG(ast);
1614 ast_traverse(astLop, replace_parms, NULL, NULL);
1615 astNew = ast_rewrite(astLop);
1616 astSS = 0;
1617 if (A_TYPEG(astNew) == A_SUBSCR) {
1618 astSS = astNew;
1619 asdSS = A_ASDG(astSS);
1620 astNew = A_LOPG(astNew);
1621 }
1622 bChanged |= astNew != astLop;
1623 ast1 = ast;
1624 if (astSS && ASD_NDIM(asdSS) != ndims) {
1625 int nSubs[7], j;
1626 bChanged = 1;
1627 j = 0;
1628 ndims = ASD_NDIM(asdSS);
1629 for (dim = 0; dim < ndims; ++dim) {
1630 astSub = ASD_SUBS(asdSS, dim);
1631 if (A_SHAPEG(astSub) || A_TYPEG(astSub) == A_TRIPLE) {
1632 nSubs[dim] = vastSubs[j++];
1633 } else {
1634 nSubs[dim] = astSub;
1635 }
1636 }
1637 for (dim = 0; dim < ndims; ++dim)
1638 vastSubs[dim] = nSubs[dim];
1639 }
1640 if (bChanged)
1641 ast1 = mk_subscr(astNew, vastSubs, ndims, A_DTYPEG(ast));
1642 ast_replace(ast, ast1);
1643 return TRUE;
1644 case A_INTR:
1645 /* look for PRESENT() call */
1646 if (A_OPTYPEG(ast) == I_PRESENT) {
1647 /* 1st argument just be A_ID */
1648 int argt, arg;
1649 INT v[4];
1650 if (A_ARGCNTG(ast) != 1) {
1651 interr("replace_parms: PRESENT with wrong argument count",
1652 A_ARGCNTG(ast), 4);
1653 }
1654 argt = A_ARGSG(ast);
1655 arg = ARGT_ARG(argt, 0);
1656 if (A_TYPEG(arg) != A_ID) {
1657 interr("replace_parms: PRESENT with wrong argument type", A_TYPEG(ast),
1658 4);
1659 }
1660 sptr = A_SPTRG(arg);
1661 for (iar = 1; iar < iaravail; ++iar)
1662 if (sptr == AR_SPTRDUMMY(iar))
1663 break;
1664 if (iar >= iaravail) {
1665 interr("replace_parms: PRESENT with bad argument symbol", sptr, 4);
1666 }
1667 ast1 = AR_ASTACT(iar);
1668 if (ast1 == astb.ptr0) {
1669 /* missing optional, replace by .false. */
1670 v[0] = 0;
1671 v[1] = 0;
1672 v[2] = 0;
1673 v[3] = 0;
1674 sptr = getcon(v, DT_LOG4);
1675 astNew = mk_cnst(sptr);
1676 ast_replace(ast, astNew);
1677 return TRUE;
1678 } else if (A_TYPEG(ast1) == A_ID) {
1679 /* get the symbol; if it is a dummy, replace it */
1680 sptr = sym_of_ast(ast1);
1681 if (SCG(sptr) == SC_DUMMY) {
1682 /* use the dummy in the PRESENT call */
1683 ast_replace(arg, ast1);
1684 } else {
1685 /* present optional, replace by .true. */
1686 v[0] = -1;
1687 v[1] = -1;
1688 v[2] = 0;
1689 v[3] = 0;
1690 sptr = getcon(v, DT_LOG4);
1691 astNew = mk_cnst(sptr);
1692 ast_replace(ast, astNew);
1693 return TRUE;
1694 }
1695 } else {
1696 /* present optional, replace by .true. */
1697 v[0] = -1;
1698 v[1] = -1;
1699 v[2] = 0;
1700 v[3] = 0;
1701 sptr = getcon(v, DT_LOG4);
1702 astNew = mk_cnst(sptr);
1703 ast_replace(ast, astNew);
1704 return TRUE;
1705 }
1706 }
1707 break;
1708 }
1709 return FALSE;
1710}

1780 NEED(iaravail, parbase, AR, iarsize, iarsize + 100);
1781 BZERO(&parbase[iar], AR, 1);
1782 AR_SPTRDUMMY(iar) = sptrDummy;
1783 AR_ASTACT(iar) = astArg;
1784 if (SDSCG(sptrDummy)) {

1788 iar = iaravail++;
