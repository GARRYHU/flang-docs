33void
34dinit_put(int dtype, INT conval)
35{
36 int n;
37
38 if (mode != 'w') { /* create a new file */
39 if ((df = tmpfile()) == NULL)
40 errfatal(5);
41 mode = 'w';
42#if DEBUG
43 if (DBGBIT(6, 1))
44 fprintf(gbl.dbgfil, "create dinit file\n");
45#endif
46 }
47
48 t.dtype = dtype;
49 t.conval = conval;
50#if DEBUG
51 if (DBGBIT(6, 1))
52 dump_buff("put");
53#endif
54
55 n = fwrite((char *)&t, sizeof(t), 1, df);
56 if (n != 1)
57 error(10, 4, 0, "(data init file)", CNULL);
58} /* dinit_put */

60DREC *
61dinit_read(void)
62{
63 int n;
64
65 if (mode == 'n' || mode == 'e')
66 return NULL;
67 if (mode == 'w') { /* rewind before reading */
68 n = fseek(df, 0L, 0);
69 if (n == -1)
70 perror("dinit_read - fseek error");
71 assert(n == 0, "dinit_read:bad rewind", n, 4);
72 mode = 'r';
73#if DEBUG
74 if (DBGBIT(6, 1))
75 fprintf(gbl.dbgfil, "rewind dinit file to read\n");
76#endif
77 }
78
79 n = fread((char *)&t, sizeof(t), 1, df);
80 if (n == 0) { /* end of file */
81 mode = 'e';
82 return NULL;
83 }
84#if DEBUG
85 if (DBGBIT(6, 1))
86 dump_buff("get");
87#endif
88
89 return &t;
90}
