180/* Fortran I/O file control block struct */
181
182typedef struct fcb {
183 struct fcb *next; /* pointer to next fcb in avail or allocd
184 * list.
185 */
186 FILE *fp; /* UNIX file pointer from fopen(). Note that a
187 * non-NULL value for this field is what
188 * indicates that a particular FCB is in use.
189 */
190 char *name; /* file name */
191 int unit; /* unit number */
192 __INT8_T reclen; /* access record length in bytes or words for
193 * direct access files
194 */
195 __INT8_T
196 partial; /* Flag/count of bytes in last record when the last record is
197 * shorter than reclen. Set and used only during a direct,
198 * unformated read of last record . */
199 int wordlen; /* length of words in bytes */
200 __INT8_T nextrec; /* record number of next record */
201 __INT8_T maxrec; /* maximum record number (direct access only) */
202 __INT8_T skip; /* After a nonadvancing write statement, this
203 * field is the number of characters remaining
204 * in the buffer, i.e., it's possible that not
205 * all of data in the buffer is transferred to
206 * file. For example, the descriptors, T & TL
207 * could effect a record position before data
208 * which was already present in the buffer.
209 */
210 char *skip_buff; /* If skip is nonzero, this field is a pointer
211 * to an allocated temporary which contains the
212 * characters remaining in the buffer and not
213 * transferred to file. Upon an ensuing write
214 * of the same file, the characters in the
215 * temporary buffer will be copied to the buffer
216 * used by fmtwrite.c.
217 */
218 short status; /* FIO_OLD or FIO_SCRATCH */
219 short dispose; /* KEEP, DELETE or SAVE */
220 short acc; /* FIO_DIRECT or FIO_SEQUENTIAL (never APPEND)*/
221 short action; /* READ, WRITE, or READWRITE */
222 short blank; /* FIO_NULL or ZERO */
223 short form; /* FIO_FORMATTED or FIO_UNFORMATTED */
224 short pad; /* YES or NO */
225 short pos; /* ASIS, REWIND, or APPEND */
226 short delim; /* APOSTROPHE, QUOTE, or NONE */
227 short coherent; /* coherency check for read & write (e.g. write
228 * followed by read needs a seek):
229 * 0 = no seek necessary for read/write
230 * 1 = coherent only if write.
231 * 2 = coherent only if read.
232 */
233 short share; /* bit vector of file sharing values TBD */
234 short decimal; /* COMMA, POINT, */
235 short encoding; /* UTF-8, UNKNOWN */
236 short round; /* UP, DOWN, ZERO, NEAREST, COMPATIBLE,
237 * PROCESSOR_DEFINED
238 */
239 short sign; /* PLUS, SUPPRESS, PROCESSOR_DEFINED */
240 sbool eof_flag; /* indicates that (imaginary) eof record has
241 * been read. Initially FALSE, set by ENDFILE
242 * or read past endoffile; cleared by REWIND
243 * and BACKSPACE
244 */
245 sbool named; /* whether file is named or not */
246 sbool stdunit; /* FCB connected to stdin/stderr/stdout */
247 sbool truncflag; /* for sequential files only. If write
248 * stmt occurs, file must be truncated if
249 * necessary
250 */
251 sbool binary; /* for unformatted files only, binary mode.
252 * if set, record length words are not present.
253 */
254 sbool ispipe; /* FCB connected to a tty or named pipe */
255 sbool nonadvance; /* last fmt write had advance=no */
256 sbool eor_flag; /* nonadvancing unit is at the end-of-record;
257 * detected when the unit is a stdunit
258 */
259 /*
260 * byte_swap, native: two flags set when the CONVERT open specifier
261 * is present. The default value (the CONVERT specifer is absent)
262 * for both flags is false.
263 */
264 sbool byte_swap; /* unformatted data needs to be byte swapped */
265 sbool native; /* unformatted data is in native format */
266 sbool asy_rw; /* async read/write stmt active */
267 struct asy *asyptr; /* pointer to asynch information,set by open */
268 char *pread; /* points to buffer of already read line
269 * this is currently used in namelist only
270 * record is read per line, we must point back
271 * to a position after '=' for a child io.
272 */
273 char *pback; /* need to keep track of the last line read
274 * used in nmlread too.
275 */
276} FIO_FCB;
