39void
40__fort_print_scalar(void *adr, dtype kind)
41{
42 if (adr == NULL) {
43 fprintf(__io_stderr(), "nil");
44 return;
45 }
46 if (!ISPRESENT(adr) || (char *)adr == ABSENTC) {
47 fprintf(__io_stderr(), "absent");
48 return;
49 }
50 switch (kind) {
51 case __INT1:
52 fprintf(__io_stderr(), "%d", *(__INT1_T *)adr);
53 break;
54 case __INT2:
55 fprintf(__io_stderr(), "%d", *(__INT2_T *)adr);
56 break;
57 case __INT4:
58 fprintf(__io_stderr(), "%d", *(__INT4_T *)adr);
59 break;
60 case __INT8:
61 fprintf(__io_stderr(), "%ld", *(__INT8_T *)adr);
62 break;
63 case __REAL4:
64 fprintf(__io_stderr(), "%g", *(__REAL4_T *)adr);
65 break;
66 case __REAL8:
67 fprintf(__io_stderr(), "%g", *(__REAL8_T *)adr);
68 break;
69 case __REAL16:
70 fprintf(__io_stderr(), "%lg", *(__REAL16_T *)adr);
71 break;
72 case __CPLX8:
73 fprintf(__io_stderr(), "(%g,%g)", ((__CPLX8_T *)adr)->r,
74 ((__CPLX8_T *)adr)->i);
75 break;
76 case __CPLX16:
77 fprintf(__io_stderr(), "(%g,%g)", ((__CPLX16_T *)adr)->r,
78 ((__CPLX16_T *)adr)->i);
79 break;
80 case __CPLX32:
81 fprintf(__io_stderr(), "(%lg,%lg)", ((__CPLX32_T *)adr)->r,
82 ((__CPLX32_T *)adr)->i);
83 break;
84 case __LOG1:
85 fprintf(__io_stderr(),
86 *(__LOG1_T *)adr & GET_DIST_MASK_LOG1 ? ".TRUE." : ".FALSE.");
87 break;
88 case __LOG2:
89 fprintf(__io_stderr(),
90 *(__LOG2_T *)adr & GET_DIST_MASK_LOG2 ? ".TRUE." : ".FALSE.");
91 break;
92 case __LOG4:
93 fprintf(__io_stderr(),
94 *(__LOG4_T *)adr & GET_DIST_MASK_LOG4 ? ".TRUE." : ".FALSE.");
95 break;
96 case __LOG8:
97 fprintf(__io_stderr(),
98 *(__LOG8_T *)adr & GET_DIST_MASK_LOG8 ? ".TRUE." : ".FALSE.");
99 break;
100 case __SHORT:
101 fprintf(__io_stderr(), "%d", *(__SHORT_T *)adr);
102 break;
103 case __CINT:
104 fprintf(__io_stderr(), "%d", *(__CINT_T *)adr);
105 break;
106 case __FLOAT:
107 fprintf(__io_stderr(), "%g", *(__FLOAT_T *)adr);
108 break;
109 case __DOUBLE:
110 fprintf(__io_stderr(), "%g", *(__DOUBLE_T *)adr);
111 break;
112 case __STR:
113 fprintf(__io_stderr(), "'%c'", *(char *)adr);
114 break;
115 default:
116 fprintf(__io_stderr(), "%x", *(int *)adr);
117 }
118}

144void I8(__fort_show_section)(F90_Desc *d)
145{
146 DECL_DIM_PTRS(dd);
147 __INT_T dx;
148
149 if (ISSEQUENCE(d)) {
150 fprintf(__io_stderr(), "SEQUENCE");
151 return;
152 } else if (ISSCALAR(d)) {
153 fprintf(__io_stderr(), "SCALAR");
154 return;
155 } else if (F90_TAG_G(d) != __DESC) {
156 fprintf(__io_stderr(), "not a descriptor\n");
157 return;
158 }
159
160 fprintf(__io_stderr(), "(");
161 for (dx = 0; dx < F90_RANK_G(d); ++dx) {
162 if (dx > 0)
163 fprintf(__io_stderr(), ",");
164 SET_DIM_PTRS(dd, d, dx);
165 if (F90_DPTR_LBOUND_G(dd) != 1)
166 fprintf(__io_stderr(), "%d:", F90_DPTR_LBOUND_G(dd));
167 fprintf(__io_stderr(), "%d", DPTR_UBOUND_G(dd));
168 }
169 fprintf(__io_stderr(), ")[%d]", F90_GSIZE_G(d));
170}

205 dist_target_spec =

207 dist_format_spec =
208 (_io_spec)(flags >> __DIST_FORMAT_SHIFT & __DIST_FORMAT_MASK);

216 if (flags & __INHERIT)
217 fprintf(__io_stderr(), ", INHERIT");
218 if (dist_target_spec | dist_format_spec) {
219 fprintf(__io_stderr(), ", %s DIST-FORMAT, %s DIST-TARGET",
220 specnames[dist_format_spec], specnames[dist_target_spec]);
221 }
222 if (flags & __DIST_TARGET_AXIS)
223 fprintf(__io_stderr(), ", DIST-TARGET AXIS");
224 if (flags & __ASSUMED_OVERLAPS)
225 fprintf(__io_stderr(), ", ASSUMED_OVERLAPS");
226 if (flags & __SECTZBASE)

232 if (flags & __TEMPLATE)
233 fprintf(__io_stderr(), ", TEMPLATE");
234 if (flags & __LOCAL)
235 fprintf(__io_stderr(), ", LOCAL");
236 if (flags & __OFF_TEMPLATE)
237 fprintf(__io_stderr(), ", OFF TEMPLATE");
238 if (flags & __NOT_COPIED)
239 fprintf(__io_stderr(), ", NOT COPIED");
240 if (flags & __SEQUENTIAL_SECTION)
241 fprintf(__io_stderr(), ", SEQUENTIAL_SECTION");

283 SET_DIM_PTRS(dd, d, dx - 1);

285 DIST_DPTR_TAXIS_G(dd), DIST_DPTR_TSTRIDE_G(dd),
286 DIST_DPTR_TOFFSET_G(dd), DIST_DPTR_COFSTR_G(dd),
287 (DIST_MAPPED_G(d) >> (dx - 1)) & 1,

290 }

292 fprintf(__io_stderr(),
293 "dim tlb tub dfmt blck cycl clb cno px pcrd pshp pstr\n");
294 for (dx = 1; dx <= F90_RANK_G(d); ++dx) {

307 SET_DIM_PTRS(dd, d, dx - 1);

318 fprintf(__io_stderr(), "align-target@%x rank=%d pbase=%d\n", t->tag,

320#if !defined(DESC_I8)

342 fprintf(__io_stderr(), "dist-target@%x rank=%d size=%d base=%d\n", p->tag,

354 pd->coord, (DIST_REPLICATED_G(d) >> (px - 1)) & 1);
355 }
356 }
357}

408void I8(__fort_describe)(char *b, F90_Desc *d)
409{
410 __INT_T dx;
411
412 if (ISSEQUENCE(d)) {
413 fprintf(__io_stderr(), "sequence %s at %p = ",
414 GET_DIST_TYPENAMES(TYPEKIND(d)), b);
415#if !defined(DESC_I8)
416 __fort_print_scalar(b, (int)TYPEKIND(d));
417#endif
418 fprintf(__io_stderr(), "\n");
419 return;
420 } else if (ISSCALAR(d)) {
421 fprintf(__io_stderr(), "scalar %s at %p = ",
422 GET_DIST_TYPENAMES(TYPEKIND(d)), b);
423#if !defined(DESC_I8)
424 __fort_print_scalar(b, (int)TYPEKIND(d));
425#endif
426 fprintf(__io_stderr(), "\n");
427 return;
428 } else if (F90_TAG_G(d) != __DESC) {
429 fprintf(__io_stderr(), "not a descriptor\n");
430 return;
431 }
432
433 if (~F90_FLAGS_G(d) & __TEMPLATE) {
434 fprintf(__io_stderr(), "%s a_%x(", GET_DIST_TYPENAMES(F90_KIND_G(d)), d->tag);
435 for (dx = 0; dx < F90_RANK_G(d); ++dx) {
436 if (dx > 0)
437 fprintf(__io_stderr(), ",");
438 if (F90_DIM_LBOUND_G(d, dx) != 1)
439 fprintf(__io_stderr(), "%d:", F90_DIM_LBOUND_G(d, dx));
440 fprintf(__io_stderr(), "%d", DIM_UBOUND_G(d, dx));
441 }
442 fprintf(__io_stderr(), ") at %p\n", b);
443 fprintf(__io_stderr(), "!hpf$ shadow a_%x(", d->tag);
444 for (dx = 0; dx < F90_RANK_G(d); ++dx) {
445 if (dx > 0)
446 fprintf(__io_stderr(), ",");
447 fprintf(__io_stderr(), "%d:%d", DIST_DIM_NO_G(d, dx),
448 DIST_DIM_PO_G(d, dx));
449 }
450 fprintf(__io_stderr(), ")\n");
451 fprintf(__io_stderr(), "local shape (");
452 for (dx = 0; dx < F90_RANK_G(d); ++dx) {
453 if (dx > 0)
454 fprintf(__io_stderr(), ",");
455 if (DIST_DIM_LAB_G(d, dx) != 1)
456 fprintf(__io_stderr(), "%d:", DIST_DIM_LAB_G(d, dx));
457 fprintf(__io_stderr(), "%d", DIST_DIM_UAB_G(d, dx));
458 }
459 fprintf(__io_stderr(), ")[%d] map (", F90_LSIZE_G(d));
460 for (dx = 0; dx < F90_RANK_G(d); ++dx) {
461 if (dx > 0)
462 fprintf(__io_stderr(), ")+(");
463 if (F90_DIM_LSTRIDE_G(d, dx) != 1)
464 fprintf(__io_stderr(), "%d*", F90_DIM_LSTRIDE_G(d, dx));
465 fprintf(__io_stderr(), "%c", 'i' + dx);
466 if (DIST_DIM_LOFFSET_G(d, dx) != 0)
467 fprintf(__io_stderr(), "%+d", DIST_DIM_LOFFSET_G(d, dx));
468 }
469 fprintf(__io_stderr(), ") lbase=%d scoff=%d\n", F90_LBASE_G(d),
470 DIST_SCOFF_G(d));
471#if !defined(DESC_I8)
472 __fort_show_flags(F90_FLAGS_G(d));
473#endif
474 fprintf(__io_stderr(), "\n");
475 }
476}

485static void I8(print_row)(void *ab, __INT_T str, __INT_T cnt, dtype kind)
486{
487 __INT_T i, *ci;
488 __INT1_T *i1;
489 __INT2_T *i2;
490 __INT4_T *i4;
491 __INT8_T *i8;
492 __LOG1_T *l1;
493 __LOG2_T *l2;
494 __LOG4_T *l4;
495 __LOG8_T *l8;
496 __REAL4_T *r4;
497 __REAL8_T *r8;
498 __REAL16_T *r16;
499 __CPLX8_T *c8;
500 __CPLX16_T *c16;
501 __CPLX32_T *c32;
502
503 switch (kind) {
504 case __CINT:
505 ci = ab;
506 for (i = 0; i < cnt; ++i) {
507 if (i > 0 && (i & 15) == 0)
508 fprintf(__io_stderr(), "\n");
509 fprintf(__io_stderr(), " %4d", ci[i * str]);
510 }
511 break;
512 case __INT1:
513 i1 = ab;
514 for (i = 0; i < cnt; ++i) {
515 if (i > 0 && (i & 15) == 0)
516 fprintf(__io_stderr(), "\n");
517 fprintf(__io_stderr(), " %4d", i1[i * str]);
518 }
519 break;
520 case __INT2:
521 i2 = ab;
522 for (i = 0; i < cnt; ++i) {
523 if (i > 0 && (i & 15) == 0)
524 fprintf(__io_stderr(), "\n");
525 fprintf(__io_stderr(), " %4d", i2[i * str]);
526 }
527 break;
528 case __INT4:
529 i4 = ab;
530 for (i = 0; i < cnt; ++i) {
531 if (i > 0 && (i & 15) == 0)
532 fprintf(__io_stderr(), "\n");
533 fprintf(__io_stderr(), " %4d", i4[i * str]);
534 }
535 break;
536 case __INT8:
537 i8 = ab;
538 for (i = 0; i < cnt; ++i) {
539 if (i > 0 && (i & 15) == 0)
540 fprintf(__io_stderr(), "\n");
541 fprintf(__io_stderr(), " %ld", i8[i * str]);
542 }
543 break;
544 case __LOG1:
545 l1 = ab;
546 for (i = 0; i < cnt; ++i) {
547 if (i > 0 && (i & 31) == 0)
548 fprintf(__io_stderr(), "\n");
549 fprintf(__io_stderr(), l1[i * str] & GET_DIST_MASK_LOG1 ? " T" : " F");
550 }
551 break;
552 case __LOG2:
553 l2 = ab;
554 for (i = 0; i < cnt; ++i) {
555 if (i > 0 && (i & 31) == 0)
556 fprintf(__io_stderr(), "\n");
557 fprintf(__io_stderr(), l2[i * str] & GET_DIST_MASK_LOG2 ? " T" : " F");
558 }
559 break;
560 case __LOG4:
561 l4 = ab;
562 for (i = 0; i < cnt; ++i) {
563 if (i > 0 && (i & 31) == 0)
564 fprintf(__io_stderr(), "\n");
565 fprintf(__io_stderr(), l4[i * str] & GET_DIST_MASK_LOG4 ? " T" : " F");
566 }
567 break;
568 case __LOG8:
569 l8 = ab;
570 for (i = 0; i < cnt; ++i) {
