389void
390fpp(void)
391{
392 int toktyp;
393 char *p;
394 PPSYM *sp;
395 char **cp;
396 int done;
397 int i;
398 INT mon;
399 char **dirp;
400 static char adate[] = "\377\"Mmm dd yyyy\"";
401 static const char *months[12] = {"Jan", "Feb", "Mar", "Apr", "May", "Jun",
402 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
403 static char atime[] = "\377\"hh:mm:ss\"";
404 static char a99[] = "\37799";
405
406 /* By default always preprocess using ANSI-C99 semantics (accpp)
407 * 0x20: Preprocessor does not add whitespace
408 * 0x800: Preprocessor does not collapse whitespace
409 */
410 if (XBIT(123, 0x4000000) == 0) { /* If legacy (fpp) mode is off... */
411 flg.x[123] |= 0x20 | 0x800;
412 accpp();
413 return;
414 }
415
416 list_flag = flg.list;
417
418 if (strlen(gbl.file_name) < MAX_FNAME_LEN) {
419 strcpy(inclstack[0].fname, gbl.file_name);
420 } else {
421 strncpy(inclstack[0].fname, gbl.file_name, MAX_FNAME_LEN);
422 inclstack[0].fname[MAX_FNAME_LEN - 1] = 0;
423 }
424 dirnam(inclstack[0].fname, inclstack[0].dirname);
425 gbl.curr_file = gbl.file_name;
426 inclstack[0].ifp = gbl.srcfil;
427 inclstack[0].lineno = 0;
428 ifp = inclstack[0].ifp;
429
430/* process def command line options */
431
432 for (cp = flg.def; cp && *cp; ++cp) {
433 /* enter it */
434 for (p = *cp; *p && *p != '='; ++p)
435 ;
436 if (!*p) {
437 sp = lookup(*cp, 1);
438 sp->value = strstore("\3771"); /* define to be 1 */
439 } else {
440 int savec;
441 savec = *p;
442 *p = 0;
443 sp = lookup(*cp, 1);
444 *p = '\377';
445 sp->value = strstore(p);
446 *p = savec;
447 }
448 }
449 strcpy(cur_fname, inclstack[0].fname);
450 cur_line = 1;
451 pr_line(cur_fname, cur_line);
452
453 iftop = -1;
454 ifpush();
455 ifstack(truth) = 1;
456
457/* add std predefined macros */
458#define chkdef(name, val) \
459 { \
460 sp = lookup(name, 1); \
461 if (sp->value != 0) \
462 pperror(220, name, 1); \
463 else \
464 sp->value = strstore(val); \
465 }
466
467#define PGIF "__PGIF90__"
468#define PGIF_MINOR "__PGIF90_MINOR__"
469#define PGIF_PATCHLEVEL "__PGIF90_PATCHLEVEL__"
470
471#ifdef TARGET_SUPPORTS_QUADFP
472 chkdef("__flang_quadfp__", "1");
473#endif
474
475 if (XBIT(124, 0x200000)) {
476 chkdef("pgi", "\3771"); /* define to be 1 */
477 }
478 chkdef("__PGI", "\3771"); /* define to be 1 */
479
480 chkdef("__LINE__", "0");
481 lineloc = sp - hashrec;
482 chkdef("__FILE__", "1");
483 fileloc = sp - hashrec;
484
485 strncpy(&adate[3], gbl.datetime, 10);
486 adate[5] = adate[8] = ' '; /* delete '/' */
487 (void)atoxi(&adate[3], &mon, 2, 10);
488 strncpy(&adate[2], months[mon - 1], 3);
489 strncpy(&atime[2], &gbl.datetime[12], 8);
490
491 chkdef("__DATE__", adate);
492 chkdef("__TIME__", atime);
493
494 chkdef("__STDC__", "\3771"); /* define to be 1 */
495
496 p = version.vsn;
497 tokval[0] = '\377';
498 for (i = 0; i < TOKMAX - 1 && isdig(*p); i++)
499 tokval[i + 1] = *p++;
500 tokval[i + 1] = 0;
501 if (i == 0) { /* no digits */
502 chkdef("__PGIC__", a99) chkdef("__PGIC_MINOR__", a99) chkdef(PGIF, a99)
503 chkdef(PGIF_MINOR, a99)
504 } else {
505 chkdef("__PGIC__",
506 tokval) while (*p != 0 && !isdig(*p)) /* skip non-digits */
507 p++;
508 for (i = 0; i < TOKMAX - 1 && isdig(*p); i++)
509 tokval[i + 1] = *p++;
510 tokval[i + 1] = 0;
511 if (i == 0) { /* no digits */
512 chkdef("__PGIC_MINOR__", a99) chkdef(PGIF_MINOR, a99)
513 } else {
514 chkdef("__PGIC_MINOR__", tokval) chkdef(PGIF_MINOR, tokval)
515 }
516 }
517 p = version.bld;
518 tokval[0] = '\377';
519 while (*p != 0 && !isdig(*p)) /* skip non-digits */
520 p++;
521 for (i = 0; i < TOKMAX - 1 && isdig(*p); i++)
522 tokval[i + 1] = *p++;
523 tokval[i + 1] = 0;
524 if (i == 0) { /* no digits */
525 chkdef("__PGIC_PATCHLEVEL__", a99) chkdef(PGIF_PATCHLEVEL, a99)
526 } else {
527 chkdef("__PGIC_PATCHLEVEL__", tokval) chkdef(PGIF_PATCHLEVEL, tokval)
528 }
529
530 /* value of _OPENMP is 201307, July, 2013 - version 4.0 */
531 if (flg.smp && !XBIT(69, 1))
532 chkdef("_OPENMP", "\377201307") /* NO SEMI */
533
534 argbuf = sccalloc(ARGMAX + 1);
535 if (argbuf == NULL)
536 error(7, 4, 0, CNULL, CNULL);
537
538#undef chkdef
539
540/* process undef command line options */
541
542 for (cp = flg.undef; cp && *cp; ++cp) {
543 /* undef it */
544 delete (*cp); /* should check if IDENT */
545 }
546
547 if (flg.list_macros) {
548 fpp_display_macros();
549
550 FREE(argbuf);
551 FREE(deftab);
552 FREE(hashrec);
553 if (incllist != NULL) {
554 FREE(incllist);
555 }
556
557 flg.es = TRUE;
558 finish();
559 }
560
561 idir.b_avl = 0;
562 idir.b_sz = 2048;
563 NEW(idir.b, char, idir.b_sz);
564
565 idir.cnt = 0;
566 idir.sz = 64;
567 NEW(idir.path, int, idir.sz);
568
569 for (dirp = flg.idir; dirp && *dirp; dirp++)
570 stash_paths(*dirp);
571 if (flg.stdinc == 0)
572 stash_paths(DIRSINCS);
573 else if (flg.stdinc != (char *)1)
574 stash_paths(flg.stdinc);
575
576 inclstack[0].path_idx = idir.last = -1;
577
578 for (;;) {
579 toktyp = findtok(tokval, ifstack(truth));
580 if (toktyp == EOF)
581 break;
582 done = 0;
583 toktyp = skipbl(tokval, 1);
584 if (toktyp == '\n') { /* do nothing, ANSI 'NULL' directive */
585 done = 1; /* no clreol */
586 } else if (toktyp != T_IDENT) {
587 /* check for # <lineno> <file>; just ignore it */
588 if (!isdig(*tokval))
589 pperr(234, 3);
590 } else {
591 switch (dlookup(tokval)) {
592 case D_IF:
593 ifpush();
594 if (_ifs[iftop - 1].truth) {
595 done = 1;
596 if (doparse()) {
597 ifstack(truth) = 1;
598 ifstack(true_seen) = 1;
599 } else
600 ifstack(truth) = 0;
601 } else
602 ifstack(truth) = 0;
603 break;
604 case D_ELIF:
605 if (iftop == 0)
606 pperr(258, 3);
607 else if (ifstack(else_seen))
608 pperr(221, 3);
609 else if (ifstack(true_seen))
610 ifstack(truth) = 0;
611 else if (!ifstack(truth) && !ifstack(true_seen) &&
612 _ifs[iftop - 1].truth) {
613 done = 1;
614 if (doparse()) {
615 ifstack(true_seen) = 1;
616 ifstack(truth) = 1;
617 } else
618 ifstack(truth) = 0;
619 }
620 break;
621 case D_ELSE:
622 if (iftop == 0)
623 pperr(259, 3);
624 else if (ifstack(else_seen))
625 pperr(222, 3);
626 else if (ifstack(true_seen))
627 ifstack(truth) = 0;
628 else if (!ifstack(truth) && !ifstack(true_seen) &&
629 _ifs[iftop - 1].truth) {
630 ifstack(true_seen) = 1;
631 ifstack(truth) = 1;
632 }
633 if (iftop > 0)
634 ifstack(else_seen) = 1;
635 break;
636 case D_ENDIF:
637 if (iftop == 0)
638 pperr(260, 3);
639 else
640 --iftop;
641 break;
642 case D_IFDEF:
643 ifpush(); /* sets truth value to old truth value */
644 toktyp = skipbl(tokval, 0);
645 if (toktyp != T_IDENT) {
646 pperr(251, 2);
647 ifstack(truth) = 0;
648 if (toktyp == '\n')
649 done = 1;
650 } else if (lookup(tokval, 0) == 0)
651 ifstack(truth) = 0;
652 else {
653 ifstack(truth) &= 1;
654 ifstack(true_seen) = 1;
655 }
656 break;
657 case D_IFNDEF:
658 ifpush(); /* sets truth value to old truth value */
659 toktyp = skipbl(tokval, 0);
660 if (toktyp != T_IDENT) {
661 pperr(252, 2);
662 ifstack(truth) = 0;
663 if (toktyp == '\n')
664 done = 1;
665 } else if (lookup(tokval, 0) != 0)
666 ifstack(truth) = 0;
667 else {
668 ifstack(truth) &= 1;
669 ifstack(true_seen) = 1;
670 }
671 break;
672 case D_INCLUDE:
673 if (ifstack(truth)) {
674 doincl(FALSE);
675 done = 1;
676 }
677 break;
678 case D_INCLUDE_NEXT:
679 done = 0;
680 if (ifstack(truth)) {
681 doincl(TRUE);
682 done = 1;
683 }
684 break;
685 case D_DEFINE:
686 if (ifstack(truth)) {
687 dodef();
688 done = 1;
689 }
690 break;
691 case D_UNDEF:
692 if (ifstack(truth)) {
693 doundef();
694 done = 1;
695 }
696 break;
697 case D_LINE:
698 if (ifstack(truth)) {
699 doline();
700 done = 1;
701 }
702 break;
703 case D_PRAGMA:
704 if (ifstack(truth)) {
705 dopragma();
706 done = 1;
707 }
708 break;
709 case D_MODULE:
710 if (ifstack(truth)) {
711 domodule();
712 done = 1;
713 }
714 break;
715 case D_LIST:
716 list_flag = flg.list;
717 break;
718 case D_NOLIST:
719 list_flag = 0;
720 break;
721 case D_IDENT:
722 if (ifstack(truth)) {
723 doident();
724 done = 1;
725 }
726 break;
727 case D_ERROR:
728 done = 0;
729 if (ifstack(truth)) {
730 doerror();
731 done = 1;
732 }
733 break;
734 case D_WARNING:
735 done = 0;
736 if (ifstack(truth)) {
737 dowarning();
738 done = 1;
739 }
740 break;
741 default:
742 if (ifstack(truth))
743 pperror(256, tokval, 3);
744 break;
745 } /* switch */
746 } /* else */
747 if (!done)
748 clreol();
749 }
750
751 if (iftop != 0)
752 pperr(238, 3);
753
754 if (!flg.es)
755 (void)fseek(gbl.cppfil, 0L, 0);
756
757 FREE(argbuf);
758 FREE(deftab);
759 FREE(hashrec);
760
761 /* -M option: Print list of include files to stdout */
762 /* -MD option: Print list of include files to file <program>.d */
763 if (XBIT(123, 2) || XBIT(123, 8)) {
764 int count;
765 if (gbl.dependfil == NULL) {
766 if ((gbl.dependfil = tmpfile()) == NULL)
767 errfatal(5);
768 }
769 count = strlen(gbl.module) + strlen(gbl.src_file) + 6;
770 for (i = 0; i < incfiles; i++) {
771 if ((count += (int)strlen(incllist[i])) >= 80) {
772 fprintf(gbl.dependfil, "\\\n ");
773 count = strlen(incllist[i]) + 3;
774 }
775 if (!XBIT(123, 0x40000))
776 fprintf(gbl.dependfil, "%s ", incllist[i]);
777 else
778 fprintf(gbl.dependfil, "\"%s\" ", incllist[i]);
779 }
780 }
781 if (incllist != 0) {
782 FREE(incllist);
783 }
784}

952dodef(void)
953{
954 int toktyp;
955 PPSYM *sp;
956 int nformals;
957 char macbuf[MACMAX + 1];
958 char *formals[FORMALMAX];
959 char *p;
960 char *defp;
961 int i;
962 char *q, *q1;
963 char savec;
964
965 /* fetch symbol after #define */
966 p = macbuf;
967 toktyp = skipbl(tokval, 0);
968 if (toktyp != T_IDENT) {
969 pperr(235, 2); /* illegal macro name is warning in cpp */
970 clreol();
971 return;
972 }
973 /* look up the name */
974 sp = lookup(tokval, 1);
975 /* here we need to read the definition */
976 nformals = -1;
977 toktyp = gtok(tokval, 0);
978 if (toktyp == '(') {
979 nformals = 0;
980 toktyp = skipbl(tokval, 0);
981 for (;;) {
982 if (toktyp == ')')
983 break;
984 if (toktyp != T_IDENT) {
985 pperror(244, tokval, 3);
986 clreol();
987 goto nulldef;
988 }
989 /* save formal */
990 if (nformals + 1 > FORMALMAX)
991 pperror(254, &deftab[sp->name], 3);
992 else if (p - macbuf + (int)strlen(tokval) + 1 > MACMAX)
993 pperror(231, &deftab[sp->name], 3);
994 else {
995 strcpy(p, tokval);
996 formals[nformals++] = p;
997 p += strlen(tokval) + 1;
998 }
999 toktyp = skipbl(tokval, 0);
1000 if (toktyp == ',') {
1001 toktyp = skipbl(tokval, 0);
1002 continue;
1003 }
1004 if (toktyp == '\n') {
1005 pperror(262, &deftab[sp->name], 3);
1006 goto nulldef;
1007 }
1008 }
1009 toktyp = gtok(tokval, 0);
1010 }
1011 /*
1012 * now nformals has # formals, formals has formals, and token is first
1013 * after macro def
1014 */
1015
1016 if (toktyp == '\n') {
1017 nulldef:
1018 if (sp->value && (((int)deftab[sp->value] & 0xFF) != (nformals & 0xFF) ||
1019 deftab[sp->value + 1])) {
1020 pperror(241, &deftab[sp->name], 2); /* redef is warning in cpp */
1021 } else if (sp->value)
1022 pperror(242, &deftab[sp->name], 1);
1023 sp->value = strstore("\377");
1024 deftab[sp->value] = nformals;
1025 return;
1026 }
1027 if (nformals == -1 && !isblank(toktyp))
1028 pperr(245, 1); /* this maybe should be removed or made inform */
1029 if (isblank(toktyp)) {
1030 /* never add a blank for macros - the blank is a problem if the
1031 * macro expands to a directive.
1032 */
1033 toktyp = skipbl(tokval, 0);
1034 }
1035 /* collect definition */
1036 defp = p;
1037 *p++ = nformals;
1038 for (;;) {
1039 if (toktyp == T_IDENT) {
1040 if ((i = inlist(formals, nformals, tokval)) != 0) {
1041 /* we found a formal */
1042 if (p - macbuf + 2 > MACMAX) {
1043 goto lenerr;
1044 }
1045 *p++ = i;
1046 *(unsigned char *)p =
1047 ARGST; /* comes last since subst scans backwards */
1048 p++;
1049 } else
1050 goto literal;
1051 } else if (toktyp == T_STRING) {
1052 /* must scan the string looking for an ident in formal */
1053 q = tokval;
1054 for (;;) {
1055 while (*q && !iscsym(*q)) {
1056 if (p - macbuf + 1 > MACMAX)
1057 goto lenerr;
1058 *p++ = *q++;
1059 }
1060 if (*q == 0)
1061 break;
1062 q1 = q;
1063 while (iscsym(*q))
1064 ++q;
1065 savec = *q;
1066 *q = 0;
1067 if ((i = inlist(formals, nformals, q1)) != 0) {
1068 /* we found a formal */
1069 if (p - macbuf + 2 > MACMAX) {
1070 lenerr:
1071 pperror(227, &deftab[sp->name], 3);
1072 goto nulldef;
1073 }
1074 *p++ = i;
1075 *(unsigned char *)p = ARGST;
1076 p++;
1077 } else {
1078 if (p - macbuf + (int)strlen(q1) > MACMAX)
1079 goto lenerr;
1080 strcpy(p, q1);
1081 p += strlen(q1);
1082 }
1083 *q = savec;
1084 } /* for(;;) */
1085 } else {
1086 literal:
1087 if (p - macbuf + (int)strlen(tokval) > MACMAX)
1088 goto lenerr;
1089 /* if only a space and a comment after macro name arrive here,
1090 define macro but it has no body */
1091 if (toktyp == '\n')
1092 goto nulldef;
1093 strcpy(p, tokval);
1094 p += strlen(tokval);
1095 }
1096 toktyp = gtok(tokval, 0);
1097 if (toktyp == '\n')
1098 break;
1099 }
1100 *p = 0;
1101 /* install definition */
1102 if (nformals == 0)
1103 *defp = 1;
1104 /* compare definitions */
1105 if (sp->value && (strcmp(defp + 1, &deftab[sp->value + 1]) ||
1106 ((int)deftab[sp->value] & 0xFF) != (nformals & 0xFF))) {
1107 pperror(241, &deftab[sp->name], 2); /* warning in cpp */
1108 } else if (sp->value)
1109 pperror(242, &deftab[sp->name], 1); /* maybe should be removed */
1110 sp->value = strstore(defp);
1111 if (nformals == 0)
1112 deftab[sp->value] = nformals;
1113}

1696/* parse "current" expression and return value computed
1697 by constant folding it. */
1698static INT
1699parse(int rbp)
1700{
1701 INT right, tmp, left;
1702 int op;
1703 LOGICAL valid_left = FALSE;
1704 do {
1705 op = token;
1706 tmp = tokenval;
1707 token = gettoken();
1708
1709 if (!valid_left) {
1710 valid_left = TRUE;
1711 op = PT[op].nud;
1712 if (op == 0)
1713 goto syntax_error;
1714 } else {
1715 if (PT[op].led == 0)
1716 goto syntax_error;
1717 }
1718
1719 if (op != Number && op != Defined)
1720 right = parse(PT[op].rbp);
1721
1722 switch (op) {
1723 case Number:
1724 left = tmp;
1725 break;
1726 case Lparen:
1727 left = right;
1728 CHECK(Rparen);
1729 break;
1730 case Defined:
1731 tmp = token;
1732 if (token == Lparen)
1733 token = gettoken();
1734 left = tokenval;
1735 CHECK(Number);
1736 if (tmp == Lparen)
1737 CHECK(Rparen);
1738 break;
1739 case Mult:
1740 left *= right;
1741 break;
1742 case Divide:
1743 left /= right;
1744 break;
1745 case Mod:
1746 left %= right;
1747 break;
1748 case Plus:
1749 left += right;
1750 break;
1751 case Minus:
1752 left -= right;
1753 break;
1754 case Lshift:
1755 case Rshift:
1756#ifdef TM_REVERSE_SHIFT
1757 if (right < 0) {
1758 right = -right;
1759 op = (Lshift + Rshift) - op;
1760 }
1761#endif
1762 if (op == Lshift)
1763 left = LSHIFT(left, right);
1764 else /* for now, can't tell if it's signed */
1765 left = URSHIFT(left, right);
1766 break;
1767 case Lt:
1768 left = left < right;
1769 break;
1770 case Gt:
1771 left = left > right;
1772 break;
1773 case Le:
1774 left = left <= right;
1775 break;
1776 case Ge:
1777 left = left >= right;
1778 break;
1779 case Eq:
1780 left = left == right;
1781 break;
1782 case Ne:
1783 left = left != right;
1784 break;
1785 case And:
1786 left &= right;
1787 break;
1788 case Xor:
1789 left ^= right;
1790 break;
1791 case Or:
1792 left |= right;
1793 break;
1794 case Andand:
1795 left = (left && right);
1796 break;
1797 case Oror:
1798 left = (left || right);
1799 break;
1800 case Question:
1801 CHECK(Colon);
1802 tmp = parse(PT[Question].rbp);
1803 left = left ? right : tmp;
1804 break;
1805 case Comma:
1806 left = right;
1807 break;
1808 case Minusx:
1809 left = -right;
1810 break;
1811 case Not:
1812 left = !right;
1813 break;
1814 case Compl:
1815 left = ~right;
1816 break;
1817 default:
1818 goto syntax_error;
1819 }
1820 } while (PT[token].lbp > rbp);
1821
1822 return left;
1823
1824syntax_error:
1825 pperr(246, 3);
1826 syntaxerr = 1;
1827 token = Eoln;
1828 return 1;
