41/*
42 * Windows does not recognize the "_Complex" keyword for complex types but does
43 * know about the "_Fcomplex" (float) and "_Dcomplex" (double) types.
44 *
45 * To minimize support for both Windows and non-Windows systems, define two
46 * typedefs, float_complex_t and double_complex_t, that are used throughout
47 * libpgmath that define a system agnostic complex type.
48 *
49 * CPP function macro PGMATH_CMPLX_CONST should only be used to initialize
50 * complex variables with constant values (not variables).
51 *
52 * Assignment to complex variables from float/double real/imaginary values
53 * contained in variables should use one of the following:
54 *
55 * float_complex_t pgmath_cmplxf(float r, float i);
56 *
57 * double_complex_t pgmath_cmplx(double r, double i);
58 *
59 * quad_complex_t pgmath_cmplxl(float128_t r, float128_t i);
60 */
61
62#if defined(TARGET_WIN) && defined(__clang__)
63typedef _Fcomplex float_complex_t;
64typedef _Dcomplex double_complex_t;
65#define PGMATH_CMPLX_CONST(r,i) {r, i}
66#else
67typedef float _Complex float_complex_t;
68typedef double _Complex double_complex_t;
69#define PGMATH_CMPLX_CONST(r,i) r + I*i
70#endif

111/*
112 * \brief pgmath_cmplxf - return type float_complex_t from two float arguments.
113 *
114 * Common method across all platforms. Does not use "real + I*imag".
115 */
116
117static inline __attribute__((always_inline))
118float_complex_t pgmath_cmplxf(float r, float i)
119{
120 struct {
121 union {
122 float_complex_t _c;
123 float _f[2];
124 };
125 } _cf;
126 _cf._f[0] = r;
127 _cf._f[1] = i;
128 return _cf._c;
129}

131/*
132 * \brief pgmath_cmplx - return type double_complex_t from double arguments.
133 *
134 * Common method across all platforms. Does not use "real + I*imag".
135 */
136
137static inline __attribute__((always_inline))
138double_complex_t pgmath_cmplx(double r, double i)
139{
140 struct {
141 union {
142 double_complex_t _z;
143 double _d[2];
144 };
145 } _zd;
146 _zd._d[0] = r;
147 _zd._d[1] = i;
148 return _zd._z;
149}

151/*
152 * \brief pgmath_cmplxl - return type quad_complex_t from quad precision arguments.
153 *
154 * Common method across all platforms. Does not use "real + I*imag".

156

168 _lzq._q[1] = i;

173/*
174 * Complex ABI conventions.
175 *
176 * The macros that define the function signature (example CMPLXFUNC_C)
177 * and the macro that defines local arguments (example CMPLXFUNC_C)
178 * work in tandem so that regardless of whether the native original
179 * complex ABI or the C99 ABI is used, there are always the following
180 * variables defined:
181 *
182 * carg: the argument defined as either float _Complex or double _Complex
183 * real: creal(carg), crealf(carg) for float
184 * imag: cimag(carg), cimagf(carg) for float
185 *
186 * When there are two or more complex arguments, the local variables are
187 * suffixed with the digits (ie 1, 2, ...) representing their order as
188 * specified in the argument list.
189 */

750 * The following intrinsics are defined for platforms that do not have
751 * architecture specific versions.
752 * It is an attempt to standardize the math library source code across
753 * architectures.
754 *
755 * For example, cexp.c was coded as:
756 *
757 * #include "mthdecls.h"
758 *
759 * void
760 * __mth_i_cexp(cmplx_t *cmplx, float real, float imag)
761 * {
762 * float x, y, z;
763 * x = EXPF(real);
764 * #ifndef LINUX8664
765 * y = COSF(imag);
766 * z = SINF(imag);
767 * #else
768 * __mth_sincos(imag, &z, &y);
769 * #endif
770 * y *= x;
771 * z *= x;
772 * r_dummy(y, z);
773 * }
774 *
775 * The special casing of whether __mth_sincos() is available for
776 * individual source files is not scalable. A better alternative is to
777 * have a version of __mth_sincos, even if it is not external available
778 * during the build process.
779 */
