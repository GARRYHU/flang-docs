346/* Return ili (icon/kcon, or a loaded value) for use with mk_kmpc_api_call
347 * arguments.
348 */
349static int
350ld_sptr(SPTR sptr)
351{
352 ISZ_T sz = size_of(DTYPEG(sptr));
353
354 if (STYPEG(sptr) == ST_CONST) {
355 if (sz == 8)
356 return ad_kcon(CONVAL1G(sptr), CONVAL2G(sptr));
357 return ad_icon(CONVAL2G(sptr));
358 } else {
359 int nme = addnme(NT_VAR, sptr, 0, 0);
360 int ili = mk_address(sptr);
361 if (ILI_OPC(ili) == IL_LDA)
362 nme = ILI_OPND(ili, 2);
363 if (sz == 8)
364 return ad3ili(IL_LDKR, ili, nme, MSZ_I8);
365 return ad3ili(IL_LD, ili, nme, mem_size(DTY(DTYPEG(sptr))));
366 }
367
368 assert(0, "Invalid sptr for mk_kmpc_api_call arguments", sptr, ERR_Fatal);
369}

384DTYPE
385ll_make_kmpc_struct_type(int count, const char *name, KMPC_ST_TYPE *meminfo, ISZ_T sz)
386{
387 DTYPE dtype;
388 int i;
389 SPTR mem, tag, prev_mem, first_mem;
390 char sname[MXIDLEN];
391
392 tag = SPTR_NULL;
393 dtype = cg_get_type(6, TY_STRUCT, NOSYM);
394 if (name) {
395 sprintf(sname, "struct%s", name);
396 tag = getsymbol(sname);
397 DTYPEP(tag, dtype);
398 }
399
400 prev_mem = first_mem = SPTR_NULL;
401 mem = NOSYM;
402 for (i = 0; i < count; ++i) {
403 mem = addnewsym(meminfo[i].name);
404 STYPEP(mem, ST_MEMBER);
405 PAROFFSETP(mem, meminfo[i].psptr);
406 DTYPEP(mem, meminfo[i].dtype);
407 if (prev_mem > 0)
408 SYMLKP(prev_mem, mem);
409 SYMLKP(mem, NOSYM);
410 PSMEMP(mem, mem);
411 VARIANTP(mem, prev_mem);
412 CCSYMP(mem, 1);
413 ADDRESSP(mem, sz);
414 SCP(mem, SC_NONE);
415 if (first_mem == 0)
416 first_mem = mem;
417 sz += size_of(meminfo[i].dtype);
418 prev_mem = mem;
419 }
420
421 DTySetAlgTy(dtype, first_mem, sz, tag, 0, 0);
422 return dtype;
423}

522/* The return value is allocated and maintained locally, please do not call
523 * 'free' on this, bad things will probably happen.
524 *
525 * Caller is responsible for calling va_end()
526 *
527 * This function will maintain one allocation for unique function name.
528 */
529static const char *
530build_kmpc_api_name(int kmpc_api, va_list va)
531{
532 static hashmap_t names; /* Maintained in this routine */
533
534 if (!names)
535 names = hashmap_alloc(hash_functions_strings);
536
537 if (KMPC_FLAGS(kmpc_api) & KMPC_FLAG_STR_FMT) {
538 char *nm, *res;
539
540 /* Construct the name */
541 vasprintf(&nm, KMPC_NAME(kmpc_api), va);
542 // FIXME: Add check for win32
543 #ifndef _WIN64
544 assert(NULL != nm, "build_kmpc_api_name: Incorrect return value", 0, ERR_Fatal);
545 #endif
546#pragma GCC diagnostic push
547#pragma GCC diagnostic warning "-Wcast-qual"
548 /* If the name has already been allocated, use that to save memory */
549 if (hashmap_lookup(names, (hash_key_t)nm, (hash_data_t *)&res)) {
550 free(nm);
551 return res;
552 } else {
553 hashmap_insert(names, (hash_key_t)nm, (hash_data_t)nm);
554 return nm;
555 }
556#pragma GCC diagnostic pop
557 } else
558 return KMPC_NAME(kmpc_api);
559
560 assert(false, "build_kmpc_api_name: Incorrect return value", 0, ERR_Fatal);
561}

563/* Returns the function prototype sptr.
564 * 'reg_opc' ILI op code for return value, e.g., IL_DFRIR or 0 if void
565 * 'ret_dtype' dtype representing return value
566 */
567static int
568mk_kmpc_api_call(int kmpc_api, int n_args, DTYPE *arg_dtypes, int *arg_ilis,
569 ...)
570{
571 int i, ilix, altilix, gargs;
572 SPTR fn_sptr;
573 int *garg_ilis = ALLOCA(int, n_args);
574 const char *nm;
575 const ILI_OP ret_opc = KMPC_RET_ILIOPC(kmpc_api);
576 va_list va;
577
578 /* Some calls will make use of this (see: KMPC_FLAG_STR_FMT) */
579 va_start(va, arg_ilis);
580
581 /* Create the prototype for the API call */
582 nm = build_kmpc_api_name(kmpc_api, va);
583 fn_sptr = ll_make_kmpc_proto(nm, kmpc_api, n_args, arg_dtypes);
584 sym_is_refd(fn_sptr);
585
586 /* Update ACC routine tables and then create the JSR */
587 update_acc_with_fn(fn_sptr);
588 ilix = ll_ad_outlined_func2(ret_opc, IL_JSR, fn_sptr, n_args, arg_ilis);
589
590 /* Create the GJSR */
591 for (i = n_args - 1; i >= 0; --i) /* Reverse the order */
592 garg_ilis[i] = arg_ilis[n_args - 1 - i];
593 gargs = ll_make_outlined_garg(n_args, garg_ilis, arg_dtypes);
594 altilix = ad3ili(IL_GJSR, fn_sptr, gargs, 0);
595
596 /* Add gjsr as an alt to the jsr */
597 if (ret_opc)
598 ILI_ALT(ILI_OPND(ilix, 1)) = altilix;
599 else
600 ILI_ALT(ilix) = altilix;
601
602 va_end(va);
603 return ilix;
604}

1008/* Given an mp (schedule enumeration) return a KMPC equivalent enumerated
1009 * value.
1010 */
1011#define SCHED_PREFIX(_enum_val) (MP_SCH_##_enum_val)
1012kmpc_sched_e
1013mp_sched_to_kmpc_sched(int sched)
1014{
1015 if(sched & MP_SCH_ATTR_DEVICEDIST)
1016 return KMP_DISTRIBUTE_STATIC_CHUNKED_CHUNKONE;
1017 switch (sched) {
1018 case SCHED_PREFIX(AUTO):
1019 return KMP_SCH_AUTO;
1020 case SCHED_PREFIX(DYNAMIC):
1021 return KMP_SCH_DYNAMIC_CHUNKED;
1022 case SCHED_PREFIX(GUIDED):
1023 return KMP_SCH_GUIDED_CHUNKED;
1024 case SCHED_PREFIX(RUNTIME):
1025 return KMP_SCH_RUNTIME_CHUNKED;
1026 case SCHED_PREFIX(STATIC):
1027 return KMP_SCH_STATIC;
1028 case SCHED_PREFIX(DIST_STATIC):
1029 return KMP_DISTRIBUTE_STATIC;
1030
1031 /* Ordered */
1032 case SCHED_PREFIX(AUTO) | MP_SCH_ATTR_ORDERED:
1033 return KMP_ORD_AUTO;
1034 case SCHED_PREFIX(RUNTIME) | MP_SCH_ATTR_ORDERED:
1035 return KMP_ORD_RUNTIME;
1036 case SCHED_PREFIX(STATIC) | MP_SCH_ATTR_ORDERED:
1037 return KMP_ORD_STATIC;
1038 case SCHED_PREFIX(STATIC) | MP_SCH_ATTR_ORDERED | MP_SCH_ATTR_CHUNKED:
1039 return KMP_ORD_STATIC_CHUNKED;
1040 case SCHED_PREFIX(DYNAMIC) | MP_SCH_ATTR_ORDERED:
1041 case SCHED_PREFIX(DYNAMIC) | MP_SCH_ATTR_ORDERED | MP_SCH_ATTR_CHUNKED:
1042 return KMP_ORD_DYNAMIC_CHUNKED;
1043
1044 /* Special cases of static */
1045 case SCHED_PREFIX(STATIC) | MP_SCH_ATTR_CHUNKED:
1046 case SCHED_PREFIX(STATIC) | MP_SCH_ATTR_CHUNKED | MP_SCH_CHUNK_1:
1047 case SCHED_PREFIX(STATIC) | MP_SCH_CHUNK_1:
1048 case SCHED_PREFIX(STATIC) | MP_SCH_ATTR_CHUNKED | MP_SCH_BLK_CYC:
1049 case SCHED_PREFIX(STATIC) | MP_SCH_BLK_CYC:
1050 case SCHED_PREFIX(STATIC) | MP_SCH_BLK_ALN:
1051 return KMP_SCH_STATIC_CHUNKED;
1052
1053 /* distribute scheduling */
1054 case SCHED_PREFIX(DIST_STATIC) | MP_SCH_ATTR_CHUNKED:
1055 return KMP_DISTRIBUTE_STATIC_CHUNKED;
1056
1057 case SCHED_PREFIX(DIST_STATIC) | MP_SCH_ATTR_CHUNKED | MP_SCH_BLK_CYC:
1058 case SCHED_PREFIX(DIST_STATIC) | MP_SCH_ATTR_CHUNKED | MP_SCH_CHUNK_1:
1059 return KMP_DISTRIBUTE_STATIC_CHUNKED;
1060
1061 default:
1062 error(S_0155_OP1_OP2, ERR_Warning, gbl.lineno,
1063 "Unsupported OpenMP schedule type.", NULL);
1064 }
1065 return KMP_SCH_DEFAULT;
1066}
