14/**
15 \brief Look up symbol having a specific symbol type.
16
17 If a symbol is found in the same overloading class and has the same
18 symbol type, it is returned to the caller. If a symbol is found in
19 the same overloading class, the action of declref depends on the
20 stype of the existing symbol and value of the argument def:
21 1. if symbol is an unfrozen intrinsic and def is 'd' (define), its
22 intrinsic property is removed and a new symbol is declared,
23 2. if def is 'd', a multiple declaration error occurs, or
24 3. if def is not 'd', an 'illegal use' error occurs.
25
26 If an error occurs or a matching symbol is not found, one is
27 created and its symbol type is initialized.
28 */
29SPTR declref(SPTR sptr, SYMTYPE stype, int def);

31/**
32 \brief Declare a new symbol.
33
34 An error can occur if the symbol is already in the symbol table:
35 - if the symbol types match treat as an error if 'errflg' is true
36 otherwise its okay and return symbol to caller
37 - else if symbol is an intrinsic attempt to remove symbol's
38 intrinsic property otherwise it is an error
39 */
40SPTR declsym(SPTR sptr, SYMTYPE stype, bool errflg);

42/**
43 \brief Look up symbol matching overloading class of given symbol
44 type.
45
46 The sptr is returned for the symbol whose overloading class matches
47 the overloading class of the symbol type given. If no symbol is
48 found in the given overloading class one is created. (If scoping
49 becomes a Fortran feature, this routine will not use it)
50 */
51SPTR getocsym(SPTR sptr, int oclass);

53/**
54 \brief Reset fields of intrinsic or generic symbol, sptr, to zero
55 in preparation for changing its symbol type by the Semantic
56 Analyzer. If the symbol type of the symbol has been 'frozen', issue
57 an error message and notify the caller by returning a zero symbol
58 pointer.
59 */
60SPTR newsym(SPTR sptr);

62/**
63 \brief Reference a symbol when it's known the context requires an
64 identifier. If an error occurs (e.g., symbol which is frozen as an
65 intrinsic), a new symbol is created so that processing can
66 continue. If the symbol found is ST_UNKNOWN, its stype is changed
67 to ST_IDENT.
68 */
69SPTR ref_ident(SPTR sptr);

71/**

76 a new sptr is returned. (If scoping becomes a Fortran feature,
77 this routine will implement it)
78 */
79SPTR refsym(SPTR sptr, int oclass);
