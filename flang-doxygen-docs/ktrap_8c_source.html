26__ktrap(void)
27{
28 if (&__ktrapval != 0) {
29 int bv = __ktrapval;
30 if (bv != 0) {
31 /*
32 * -Ktrap bv
33 * fp 0x001 => inv | divz | ovf
34 * inv 0x008
35 * divz 0x020
36 * ovf 0x040
37 * unf 0x080
38 * inexact 0x100
39 */
40 int excepts = 0;
41 if (bv & 0x001)
42 bv = 0x008 | 0x020 | 0x040;
43 if (bv & 0x008)
44 excepts |= FE_INVALID;
45 if (bv & 0x020)
46 excepts |= FE_DIVBYZERO;
47 if (bv & 0x040)
48 excepts |= FE_OVERFLOW;
49 if (bv & 0x080)
50 excepts |= FE_UNDERFLOW;
51 if (bv & 0x100)
52 excepts |= FE_INEXACT;
53#ifdef TARGET_OSX
54 __fenv_feenableexcept(excepts);
55#else
56 feenableexcept(excepts); /* glibc 2.2 extension to fenv.h */
57#endif
58 }
59 }
60}
