114static void minlocs_common(red_parm *z, __INT_T *rb, char *ab, char *mb,
115 F90_Desc *rs, F90_Desc *as, F90_Desc *ms)
116{
117 double vb[4];
118 char *strvb;
119
120 __fort_red_what = "MINLOC";
121
122 z->kind = F90_KIND_G(as);
123 z->len = F90_LEN_G(as);
124 z->mask_present = (F90_TAG_G(ms) == __DESC && F90_RANK_G(ms) > 0);
125 if (!z->mask_present) {
126 z->lk_shift = GET_DIST_SHIFTS(__LOG);
127 } else {
128 z->lk_shift = GET_DIST_SHIFTS(F90_KIND_G(ms));
129 }
130 z->l_fn_b = l_minloc_b[z->lk_shift][z->kind];
131 z->g_fn = g_minloc[z->kind];
132 z->zb = GET_DIST_MAXS(z->kind);
133
134 if (z->kind == __STR) {
135 strvb = (char *)__fort_gmalloc(z->len);
136 memset(strvb, *((char *)z->zb), z->len);
137 I8(__fort_red_scalarlk)(z, strvb, ab, mb, rs, as, ms, rb, __MINLOC);
138 __fort_gfree(strvb);
139 } else {
140 I8(__fort_red_scalarlk)(z, (char *)vb, ab, mb, rs, as, ms, rb, __MINLOC);
141 }
142}
