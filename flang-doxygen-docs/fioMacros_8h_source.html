80/* __gen_block implementation__
81 * The following mask is used to check for a gen_block dimension against
82 * the isstar argument in ENTFTN(template) and ENTFTN(qopy_in) located
83 * in rdst.c
84 *
85 * We use 3 bit sets per dimension in the range of bits 7..27 of the
86 * isstar argument in ENTFTN(template) and ENTFTN(qopy_in) ...
87 *
88 * isstar == 0 => block, block(k), cyclic, or cyclic(k)
89 * isstar == 1 => gen_block
90 * isstar == 2 => indirect (not yet supported)
91 * isstar == 3 => reserved for future expansion
92 * isstar == 4 => reserved for future expansion
93 * isstar == 5 => reserved for future expansion
94 * isstar == 6 => reserved for future expansion
95 * isstar == 7 => reserved for future expansion
96 */

798 * It is used in the mapping of an array section dimension to its original
799 * array. See the __fort_finish_descriptor() runtime routine in dist.c for an
800 * example of how to compute this field. See the print_loop() runtime routine
801 * in dbug.c for an example of how to use this field.
802 *
803 * The \ref ubound field is the upperbound of the dimension.
804 */

818/** \brief Fortran descriptor header
819 *
820 * The fields minus F90_DescDim below should remain consistent
821 * with the object_desc and proc_desc structures of type.h in terms of length
822 * and type. These fields are also mirrored in the Fortran Front-end's
823 * rte.h header file.
824 *
825 * The \ref tag field is used to identify the type of the descriptor. This is
826 * typically \ref __DESC, which identifies the descriptor as a regular array
827 * section descriptor. If the tag is a basic type, such as an \ref __INT4,
828 * \ref__REAL4, etc. then it is a 1 word pseudo descriptor. The pseudo
829 * descriptor is used as a place holder when we want to pass a scalar into a
830 * runtime routine that also requires a descriptor argument. When tag is
831 * \ref __POLY, then we have an object_desc (see its definition in type.h).
832 * When tag is \ref __PROCPTR, we have a proc_desc (see its definition in
833 * type.h).
834 *
835 * The \ref rank field equals the total number of dimensions of the associated
836 * array. If rank is 0, then this descriptor may be associated with a
837 * derived type object, a pointer to a derived type object, or an
838 * allocatable scalar.
839 *
840 * The \ref kind field holds the base type of the associated array. It is one
841 * of the basic types defined in \ref _DIST_TYPE.
842 *
843 * The \ref flags field holds various descriptor flags defined above. Most of
844 * the flags defined above, denoted as reserved, are not used for Fortran.
845 * The flags that are typically used for Fortran are \ref __ASSUMED_SIZE and
846 * \ref __ASSUMED_SHAPE.
847 *
848 * The \ref len field holds the byte length of the associated array's base type
849 * (see also kind field).
850 *
851 * The \ref lsize field holds the total number of elements in the associated
852 * array section.
853 *
854 * In distributed memory languages, such as HPF, \ref gsize represents the total
855 * number of elements that are distributed across multiple processors. In
856 * Fortran, the \ref lsize and \ref gsize fields are usually the same, however
857 * this is a case in the reshape intrinsic where \ref gsize != \ref lsize. There
858 * may just be an incidental difference during the execution of reshape. There
859 * may also be others in the Fortran runtime where \ref gsize != \ref lsize,
860 * however, they too may just be incidental differences. Therefore, use
861 * \ref lsize instead of \ref gsize when querying the total number of elements
862 * in a Fortran array.
863 *
864 * The \ref lbase field is the index offset section adjustment. It is used in
865 * the mapping of an array section to its original array.
866 * See the \ref __DIST_SET_SECTIONXX and \ref __DIST_SET_SECTIONX macros below
867 * for examples of how to compute this field. See the __fort_print_local()
868 * runtime routine in dbug.c for an example of how to use this field.
869 *
870 * The \ref gbase field historically was used in distributed memory languages
871 * like HPF. Therefore, \ref gbase is usually 0 and may always be 0 in the
872 * Fortran runtime (needs more investigation to confirm if it's always 0 in
873 * this case).
874 *
875 * When set, the \ref dist_desc field holds a pointer to the type descriptor of
876 * the associated object (see also the \ref TYPE_DESC definition in type.h).
877 *
878 * The \ref dim fields hold up to \ref number of F90_DescDim structures. It's
879 * also possible that \ref dim is empty when this descriptor is associated with
880 * a derived type, a pointer to a derived type, or an allocatable scalar.
881 *
882 * The number in paranthesis for each field below corresponds with the
883 * subscript index in the descriptor. The first 9 values are denoted.
884 * The first index is 1 because we assume Fortran style arrays when we
885 * reference these fields in the Fortran front-end. When generating assembly,
886 * the first index will be 0. After the gbase field,
887 * the subscript value depends on three conditions: Whether the target's
888 * pointers are 64-bit, whether the target's native integers are 64-bit,
889 * and whether large arrays are enabled. See the \ref DESC_HDR_LEN macro
890 * in the Fortran front-end's rte.h file for more information. The first
891 * 9 subscript values are also mirrored in the following macros in
892 * rte.h: \ref DESC_HDR_TAG, \ref DESC_HDR_RANK, \ref DESC_HDR_KIND,
893 * \ref DESC_HDR_BYTE_LEN, \ref DESC_HDR_FLAGS, \ref DESC_HDR_LSIZE,
894 * \ref DESC_HDR_GSIZE, \ref DESC_HDR_LBASE, and \ref DESC_HDR_GBASE.
895 *
896 */
897struct F90_Desc {
898
899 __INT_T tag; /**< (1) tag field; usually \ref __DESC
900 (see also _DIST_TYPE) */
901 __INT_T rank; /**< (2) array section rank */
902 __INT_T kind; /**< (3) array base type */
903 __INT_T len; /**< (4) byte length of base type */
904 __INT_T flags; /**< (5) descriptor flags */
905 __INT_T lsize; /**< (6) local array section size */
906 __INT_T gsize; /**< (7) global array section size

908 __INT_T lbase; /**< (8) index offset section adjustment */
909 POINT(__INT_T, gbase); /**< (9) global offset of first element of
910 section (usually 0) */

912 object's type descriptor */
913 F90_DescDim dim[MAXDIMS]; /**< F90 dimensions (Note: We append
914 \ref rank number of F90_DescDim
915 structures to an F90_Desc structure) */
916};

1362 * \brief This macro is used for creating an array section.
1363 *
1364 * Note: We no longer need section stride/section offset.
1365 *
1366 * This macro is the same as __DIST_SET_SECTIONX,
1367 * except it collects the strides into the LSTRIDE;
1368 * SSTRIDE and SOFFSET are just copied (left at one, zero)
1369 */
1370#define __DIST_SET_SECTIONXX(d, ddim, a, adim, l, u, s, noreindex, gsize) \
1371 { \
1372 DECL_DIM_PTRS(__dd); \
1373 DECL_DIM_PTRS(__ad); \
1374 __INT_T __extent, __myoffset; \
1375 SET_DIM_PTRS(__ad, a, adim - 1); \
1376 SET_DIM_PTRS(__dd, d, ddim - 1); \
1377 __extent = u - l + s; /* section extent */ \
1378 if (s != 1) { \
1379 if (s == -1) \
1380 __extent = -__extent; \
1381 else \
1382 __extent /= s; \
1383 } \
1384 if (__extent < 0) \
1385 __extent = 0; \
1386 if (noreindex && s == 1) { \
1387 F90_DPTR_LBOUND_P(__dd, l); /* lower bound */ \
1388 DPTR_UBOUND_P(__dd, __extent == 0 ? l - 1 : u); /* upper bound */ \
1389 __myoffset = 0; \
1390 } else { \
1391 F90_DPTR_LBOUND_P(__dd, 1); /* lower bound */ \
1392 DPTR_UBOUND_P(__dd, __extent); /* upper bound */ \
1393 __myoffset = l - s; \
1394 } \
1395 F90_DPTR_SSTRIDE_P(__dd, 1); \
1396 F90_DPTR_SOFFSET_P(__dd, 0); \
1397 F90_DPTR_LSTRIDE_P(__dd, F90_DPTR_LSTRIDE_G(__ad) * s); \
1398 F90_LBASE_P(d, F90_LBASE_G(d) + __myoffset * F90_DPTR_LSTRIDE_G(__ad)); \
1399 if (F90_DPTR_LSTRIDE_G(__dd) != gsize) \
1400 F90_FLAGS_P(d, (F90_FLAGS_G(d) & ~__SEQUENTIAL_SECTION)); \
1401 gsize *= __extent; \
1402 }
