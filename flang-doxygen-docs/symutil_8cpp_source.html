52 struct Field {
53 std::string name; // field name
54 std::string type; // type name (optional)
55 int size; // size in bytes
56 int offs; // offset in bytes starting from 0 ltor, ttob
57 bool shared; // true if shared
58 bool flag; // true if a flag
59 bool shareduse; // true if used in all symtypes
60
61 bool operator<(const Field &f) const
62 {
63 bool result = false;
64 bool equal = false;
65 if (flag && f.flag) {
66 result = offs < f.offs;
67 equal = offs == f.offs;
68 } else if (flag) {
69 result = true;
70 } else if (f.flag) {
71 result = false;
72 } else {
73 result = offs < f.offs;
74 equal = offs == f.offs;
75 }
76 if (equal) {
77 result = name < f.name;
78 }
79 return result;
80 }
81 };

138 Symutil(const std::vector<std::string> &args)
139 {
140 elt[".OC"] = LT_OC;
141 elt[".SF"] = LT_SF;
142 elt[".ST"] = LT_ST;
143 elt[".Sc"] = LT_Sc;
144 elt[".SM"] = LT_SM;
145 elt[".SE"] = LT_SE;
146 elt[".SI"] = LT_SI;
147 elt[".FL"] = LT_FL;
148 elt[".TY"] = LT_TY;
149 elt[".DT"] = LT_DT;

153 elt[".Ik"] = LT_Ik;

155 checkmode = OFF;
156
157 enum { INPUT, OUTPUT, NROFF, SPHINX } state = INPUT;
158 int filename_argument = 0;
159 // FIXME: MSVC fails to infer type correctly when auto used in
160 // this case. Replace with 'auto' when moved to a newer MSVC.
161 for (std::vector<std::string>::const_iterator arg = args.begin() + 1,
162 E = args.end();
163 arg != E; ++arg) {
164 if (*arg == "-check-known") {
165 checkmode = KNOWN;
166 } else if (*arg == "-check") {
167 checkmode = ALL;
168 } else if (*arg == "-o") {
169 filename_argument = 0;
170 state = OUTPUT;
171 } else if (*arg == "-n") {
172 if (state != OUTPUT) {
173 usage("-n option must be used only after -o option");
174 }
175 state = NROFF;
176 } else if (*arg == "-s") {
177 state = SPHINX;
178 } else {
179 std::ofstream *os = nullptr;
180 bool is_nroff = false;
181 switch (state) {
182 case INPUT:
183 switch (filename_argument) {
184 case 0:
185 symtab_n_filename = *arg;
186 break;
187 case 1:
188 symtab_in_h_filename = *arg;
189 break;
190 default:
191 usage("too many input files");
192 }
193 break;
194 case OUTPUT:
195 switch (filename_argument) {
196 case 0:
197 os = &out1;
198 break;
199 case 1:
200 os = &out2;
201 break;
202 case 2:
203 os = &out3;
204 break;
205 case 3:
206 os = &out4;
207 break;
208 case 4:
209 os = &out5;
210 break;
211 default:
212 usage("too many output files");
213 }
214 break;
215 case NROFF:
216 switch (filename_argument) {
217 case 0:
218 os = &out1;
219 break;
220 case 1:
221 os = &out2;
222 break;
223 case 2:
224 os = &out3;
225 break;
226 default:
227 usage("too many output files");
228 }
229 is_nroff = true;
230 state = OUTPUT;
231 break;
232 case SPHINX:
233 sphinx_filename = *arg;
234 break;
235 }
236 if (state == OUTPUT) {
237 os->open(arg->c_str());
238 if (!*os) {
239 usage("can't create output file");
240 }
241 outputNotModifyComment(os, *arg, args[0], is_nroff);
242 }
243 if (state == SPHINX) {
244 state = OUTPUT;
245 } else {
246 ++filename_argument;
247 }
248 }
249 }
250 if (symtab_n_filename == "") {
251 usage("no symtab.n file is given");
252 }
253 if (symtab_in_h_filename == "") {
254 usage("no symtab.in.h file is given");
255 }
256 if (!out1 || !out2 || !out3 || !out4) {
257 usage("output file is missing");
258 }
259 }

295 void
296 read_symtab_n()
297 {
298 // FIXME: get rid of c_str() when C++11 STL is available
299 ifs.open(symtab_n_filename.c_str());
300 if (!ifs) {
301 printError(FATAL, "Can't open ", symtab_n_filename.c_str());
302 }
303 if (!sphinx_filename.empty()) {
304 sphinx.setOutputFile(sphinx_filename);
305 }
306 std::vector<std::string> dtfields;
307 std::vector<int> cursyms;
308 std::ostream *os = &out1;
309 int pdoffs = 0;
310
311 auto lt = getLineAndTokenize(elt, os);
312 while (1) { // once for each line
313 auto tok = getToken();
314 switch (lt) {
315 case LT_OC:
316 if (tok.length() > 1) {
317 ocnames.push_back(tok);
318 }
319 break;
320 case LT_Sc:
321 if (tok.length() > 1) {
322 scnames.push_back(tok);
323 }
324 break;
325 case LT_Ik:
326 if (tok.length() > 1) {
327 iknames.push_back(tok);
328 }
329 break;
330 case LT_SF:
331 if (tok.length() > 1) {
332 (void)add_field(tok, true, false, true);
333 }
334 break;
335 case LT_ST: // symbol type
336 if (tok.length() > 1) {
337 (void)findsym(tok);
338 }
339 break;
340 case LT_SM:
341 flushsym(cursyms); // flush info for current sym
342 cursyms.clear();
343 if (tok.length() == 1) {
344 /* end of SM definitions */
345 os = &out1;
346 goto again;
347 }
348 while (!tok.empty()) {
349 cursyms.push_back(findsym(tok));
350 tok = getToken();
351 }
352 // redirect output up to the next SM
353 tmpss.str("");
354 tmpss.clear();
355 os = &tmpss;
356 flushline(&out1);
357 goto again;
358
359 case LT_SI:
360 addoclass(cursyms, tok);
361 tok = getToken();
362 for (std::vector<int>::size_type idx = 0; !tok.empty(); ++idx) {
363 addsname(cursyms, idx, tok);
364 tok = getToken();
365 }
366 break;
367
368 case LT_SE:
369 add_field_to_symbol(cursyms, add_field(tok, false, false,
370 checkmode && cursyms[0] == 0));
371 break;
372
373 case LT_FL:
374 add_field_to_symbol(
375 cursyms, add_field(tok, false, true, checkmode && cursyms[0] == 0));
376 break;
377
378 case LT_TY:
379 if (tok.length() > 1) {
380 types.push_back(Type());
381 auto it = types.end() - 1;
382 it->name = tok;
383 it->sname = getToken();
384 tok = getToken();
385 while (!tok.empty()) {
386 it->attribs.push_back(addattr(tok));
387 tok = getToken();
388 }
389 }
390 break;
391
392 case LT_DT:
393 // flush info for current dt
394 flushdt(cursyms, dtfields);
395 cursyms.clear();
396 if (tok.length() == 1) {
397 // end of DT definitions
398 os = &out1;
399 goto again;
400 }
401 while (!tok.empty()) {
402 cursyms.push_back(finddt(tok));
403 tok = getToken();
404 }
405 dtfields.clear();
406 // redirect output up to next DT
407 tmpss.str("");
408 tmpss.clear();
409 os = &tmpss;
410 flushline(&out1);
411 goto again;
412
413 case LT_DE:
414 dtfields.push_back(tok);
415 break;
416
417 case LT_PD:
418 if (tok.length() == 1) {
419 if (tok[0] == 'B') {
420 pdoffs = 0;
421 out3 << "int pd_dtype[DT_MAX+1] = {\n";
422 } else if (tok[0] == 'E') {
423 out3 << "};\n";
424 pd_dtypes.push_back(PDType(pdoffs));
425 }
426 break;
427 }
428 pd_dtypes.push_back(PDType(tok, pdoffs));
429 out3 << " /* " << std::left << std::setw(13) << tok << "*/";
430 getToken(); // skip sname
431 tok = getToken();
432 while (!tok.empty()) {
433 out3 << ' ' << tok << ',';
434 tok = getToken();
435 ++pdoffs;
436 }
437 out3 << '\n';
438 break;
439
440 case LT_EOF:
441 return;
442 default:
443 printError(FATAL, "Unknown LT: can't happen\n");
444 }
445 flushline(os);
446 again:
447 lt = getLineAndTokenize(elt, os);
448 }
449 }

451 void
452 flushdt(std::vector<int> &cursyms, std::vector<std::string> &dtfields)
453 {
454 if (cursyms.empty())
455 return;
456 for (std::vector<int>::const_iterator it = cursyms.begin(),
457 E = cursyms.end();
458 it != E; ++it) {
459 out1 << ".TS\n"
460 << "tab(%) allbox;\n";
461 for (std::vector<std::string>::size_type k = 0; k <= dtfields.size();
462 ++k) {
463 out1 << "cw(0.8i) ";
464 }
465 out1 << ".\n" << types[*it].name;
466 for (std::vector<std::string>::size_type k = 0; k != dtfields.size();
467 ++k) {
468 out1 << "%" << dtfields[k];
469 }
470 out1 << "\n.TE\n";
471 }
472 // append temp file contents to troff output
473 out1 << tmpss.str();
474 }

487 addattr(std::string &tok)
488 {
489 auto it = std::find(attrnames.begin(), attrnames.end(), tok);

491 return it - attrnames.begin();

497 /**
498 * Print the declaration of an enumeration, both as an enum
499 * and as reflexive "#define"s. The enum permits debuggers
500 * to show values nicely; the latter permits the preprocessor
501 * to query whether a value exists.
502 *
503 * \param enum_name: name of the enumeration and typedef.
504 * \param name_of_max: "...MAX" name of maximum value, or empty string
505 * to suppress printing it.
506 * \param n: number of defined values.
507 * \param name_of: functor mapping i to name of ith enumeration value.
508 * \param value_of: functor mapping i to value of ith enumeration value.
509 * \param allow_declare_as_int: if true, allow enumeration to be defined
510 * as int if DECLARE_enum_name_AS_INT is #defined.
511 * \param strut: optional "id = value" to be injected into enumeration.
512 * Used to force enum size.
513 * \param lwst optional "id = value" to be injected into the enumeration.
514 * Used to force enum to be signed (for comparison, etc.)
515 *
516 * FIXME - it's a waste of space/compilation time for this method
517 * to be a template. When we have C++11 std::function, declare
518 * name_of and value_of as std::function<std::string,size_t> would
519 * be more frugal.
520 */
521 template <typename Fname, typename Fvalue>
522 void
523 write_enum_with_defines(const std::string &enum_name,
524 const std::string &name_of_max, size_t n,
525 Fname name_of, Fvalue value_of,
526 bool allow_declare_as_int = false,
527 const char *strut = nullptr,
528 const char *lwst = nullptr)
529 {
530 // Print the enumeration values
531 if (allow_declare_as_int)
532 out2 << "#ifndef DECLARE_" << enum_name << "_AS_INT\n\n";
533 out2 << "typedef enum " << enum_name << " {\n";
534 if (lwst)
535 out2 << " " << lwst << ",\n";
536 for (size_t i = 0; i < n; ++i)
537 out2 << " " << name_of(i) << " = " << value_of(i)
538 << (i < n - 1 || strut ? "," : "") << "\n";
539 // Print the strut declaration if one was supplied
540 if (strut)
541 out2 << " " << strut << "\n";
542 out2 << "} " << enum_name << ";\n\n";
543 // Print reflexive #define for each value
544 for (size_t i = 0; i < n; ++i)
545 out2 << "#define " << name_of(i) << " " << name_of(i) << "\n";
546 out2 << "\n";
547 if (allow_declare_as_int) {
548 // Print deprecated alternative - enumeration type becomes an int,
549 // and values are #define'd so that preprocessor can evaluate them.
550 out2 << "#else\n\n"
551 "/* DEPRECATED TEMPORARY WORKAROUND FOR BACKWARDS COMPATIBILITY "
552 "*/\n"
553 "typedef int "
554 << enum_name << ";\n";
555 for (size_t i = 0; i < n; ++i)
556 out2 << "#define " << name_of(i) << " " << value_of(i) << "\n";
557 out2 << "\n"
558 "#endif\n\n";
559 }
560 if (!name_of_max.empty()) {
561 // Print the maximum value
562 out2 << "#define " << name_of_max << " " << value_of(n - 1) << "\n\n";
563 }
564
565 if (out5.is_open()) {
566 // Print array mapping enum values to names.
567 out5 << "const char *" << enum_name << "_names[] = {\n";
568 size_t index = 0; // index in array
569 for (size_t i = 0; i < n; ++i, ++index) {
570 size_t value = value_of(i);
571 for (; index < value; ++index) {
572 out5 << " /* " << index << " */ \"" << enum_name << ' ' << index
573 << "\",\n";
574 }
575 out5 << " /* " << value << " */ \"" << name_of(i) << "\",\n";
576 }
577 out5 << "};\n\n";
578 }
579 }

581 void
582 write_symtab()
583 {
584 // FIXME: get rid of c_str() when C++11 STL is available
585 std::ifstream ifs(symtab_in_h_filename.c_str());
586 if (!ifs) {
587 printError(FATAL, "Can't open ", symtab_in_h_filename.c_str());
588 }
589 std::sort(fields.begin(), fields.end());
590 // read the symtab.h boilerplate file and replace NROFF macros
591 for (std::string line; std::getline(ifs, line); ) {
592 auto lt = elt.match(line);
593 if (!lt) {
594 out2 << line << '\n'; // because getline strips EOL char.
595 continue;
596 }
597 switch (lt) {
598 case LT_OC: // print overloading classes
599 write_enum_with_defines("OVCLASS", "OC_MAX", ocnames.size(),
600 [&](size_t i) { return ocnames[i]; },
601 [](size_t i) { return i + 1; });
602 break;
603 case LT_Sc: // print storage classes
604 write_enum_with_defines("SC_KIND", "SC_MAX", scnames.size(),
605 [&](size_t i) { return scnames[i]; },
606 [](size_t i) { return i; });
607 break;
608 case LT_Ik: // print intrinsic kinds
609 for (std::vector<std::string>::size_type i = 0; i != iknames.size();
610 ++i)
611 out2 << "#define " << iknames[i] << " " << i << "\n";
612 out2 << "#define IK_MAX " << iknames.size() - 1 << "\n\n";
613 break;
614 case LT_ST: // print symbol types
615 write_enum_with_defines("SYMTYPE", "ST_MAX", symbols.size(),
616 [&](size_t i) { return symbols[i].stype; },
617 [](size_t i) { return i; });
618 break;
619 case LT_TY: // print type names
620 write_enum_with_defines("TY_KIND", "TY_MAX", types.size(),
621 [&](size_t i) { return types[i].name; },
622 [](size_t i) { return i; }, true);
623
624 // pd_dtypes contains one extra element, which holds the DT_MAX+1 value.
625 // We print that one specially after writing the regular enum values.
626 write_enum_with_defines("DTYPE", "", pd_dtypes.size() - 1,
627 [&](size_t i) { return pd_dtypes[i].name; },
628 [&](size_t i) { return pd_dtypes[i].pdnum; },
629 true, "DT_MAXIMUM_POSSIBLE_INDEX = 0x7FFFFFFF",
630 "DT_MINUMUM_PSEUDO_VALUE = -32768");
631 out2 << "#define DT_MAX " << pd_dtypes.back().pdnum - 1 << "\n";
632 break;
633 case LT_TA: // print type attributes
634 for (std::vector<std::string>::size_type i = 0; i < attrnames.size();
635 ++i)
636 out2 << "#define _TY_" << attrnames[i] << " " << (1 << i) << "\n";
637 out2 << "\n";
638 out3 << "\nshort dttypes[TY_MAX+1] = {\n";
639 for (std::vector<Type>::size_type i = 0; i != types.size(); ++i) {
640 out3 << " /* " << std::setw(13) << types[i].name << "*/ ";
641 auto k = types[i].attribs.size();
642 if (k == 0) {
643 out3 << "0,\n";
644 } else {
645 out3 << "_TY_" << attrnames[types[i].attribs[0]];
646 for (std::vector<int>::size_type j = 1; j < k; ++j)
647 out3 << "|_TY_" << attrnames[types[i].attribs[j]];
648 out3 << ",\n";
649 }
650 }
651 out3 << "};\n";
652 break;
653 case LT_SE: // print fields access macros
654 write_symtab_check_funcs();
655 for (std::vector<Field>::size_type i = 0; i != fields.size(); ++i) {
656 auto field = fields[i];
657 auto name = field.name;
658 if (field.flag) {
659 write_symtab_accessors(field, i, "f", field.offs);
660 } else if (field.shared) {
661 if ("flags" == name || "flags2" == name ||
662 "flags3" == name || "flags4" == name)
663 continue;
664 write_symtab_accessors(field, i, makeLower(name), 0);
665 } else {
666 std::string name;
667 int offset;
668 switch (field.size) {
669 case 1:
670 name = "b";
671 offset = field.offs + 1;
672 break;
673 case 2:
674 name = "hw";
675 offset = field.offs / 2 + 1;
676 break;
677 case 4:
678 name = "w";
679 offset = field.offs / 4 + 1;
680 break;
681 default:
682 // FIXME check this error when fields is filled.
683 printError(WARN, "Field not b,h,w in macro");
684 offset = 0;
685 break;
686 }
687 write_symtab_accessors(field, i, name, offset);
688 }
689 }
690 if (checkmode) {
691 out2 << "extern char sym_type_check[ST_MAX+1][" << fields.size()
692 << "];\n";
693 }
694 break;
695 default:
696 printError(WARN, "Unknown line type");
697 break;
698 }
699 }

701 // write dinit for STB
702 out3 << "\nextern STB stb;\n"; // declaration into .h file
703 out4 << "\n#include \"symacc.h\"\n"; // forward declaration of the STB type
704 out4 << "\n#pragma clang diagnostic ignored "; // work around Clang warning
705 out4 << "\"-Wmissing-field-initializers\"\n"; // (cont'd)
706 out4 << "\nSTB stb = {\n {"; // definition into .c file
707 int j = 6, k;
708 for (std::vector<Symbol>::size_type i = 0; i != symbols.size(); ++i) {
709 if ((j += (k = symbols[i].sname.length() + 3)) > 80) {
710 out4 << "\n ";
711 j = 6 + k;
712 }
713 out4 << "\"" << symbols[i].sname << "\",";
714 }
715 out4 << "},\n";
716
717 j = 6;
718 out4 << " {";
719 for (std::vector<Symbol>::size_type i = 0; i != symbols.size(); ++i) {
720 if ((j += (k = ocnames[symbols[i].oclass - 1].length() + 1)) > 80) {
721 out4 << "\n ";
722 j = 6 + k;
723 }
724 out4 << ocnames[symbols[i].oclass - 1] << ",";
725 }
726 out4 << "},\n";
727
728 out4 << " {\"???\",";
729 for (std::vector<std::string>::size_type i = 0; i != ocnames.size(); ++i)
730 out4 << "\"" << ocnames[i] << "\",";
731 out4 << "},\n";
732
733 out4 << " {";
734 for (std::vector<std::string>::size_type i = 0; i != scnames.size(); ++i)
735 out4 << "\"" << scnames[i] << "\",";
736 out4 << "},\n";
737

997 bool
998 chk_overlap(int f1, int f2, bool flag)
999 {
1000 int t1, t2, t3, t4;
1001
1002 if (fields[f1].flag && fields[f2].flag) {
1003 if (fields[f1].offs == fields[f2].offs) {
1004 if (flag) {
1005 std::ostringstream oss;
1006 oss << "name " << fields[f1].name << " overlaps name "
1007 << fields[f2].name << " at offset " << fields[f1].offs << "\n";
1008 printError(WARN, "Flag overlaps one already defined",
1009 oss.str().c_str());
1010 }
1011 return true;
1012 }
1013 } else if (fields[f1].flag || fields[f2].flag)
1014 return false;
1015 t1 = fields[f1].offs; // check for field overlap
1016 t2 = t1 + fields[f1].size - 1;
1017 t3 = fields[f2].offs;
1018 t4 = t3 + fields[f2].size - 1;
1019 if ((t1 >= t3 && t1 <= t4) || (t2 >= t3 && t2 <= t4) ||
1020 (t3 >= t1 && t3 <= t2) || (t4 >= t1 && t4 <= t2)) {
1021 if (flag) {
1022 std::ostringstream oss;
1023 oss << "name " << fields[f1].name << " overlaps name "
1024 << fields[f2].name << " at offset " << fields[f1].offs << "\n";
1025 printError(WARN, "Field overlaps one already defined",
1026 oss.str().c_str());
1027 }
1028 return true;
1029 }
1030 return false;

1058 int
1059 add_field(std::string &ptok, bool sharedflag, bool flagflag, bool shareduse)
1060 {
1061 int size = 0, offs = 0;
1062 const char *aftp; // position after w<d>
1063
1064 for (auto it = fields.begin(), E = fields.end(); it != E; ++it)
1065 if (ptok == it->name)
1066 return it - fields.begin();
1067
1068 fields.push_back(Field());
1069 auto it = fields.end() - 1;
1070 it->name = ptok;
1071 auto tok = getToken();
1072 if (tok.empty()) {
1073 printError(WARN, "Field location not specified");
1074 goto fixup;
1075 }
1076 /* parse location */
1077 if (flagflag) {
1078 if (tok[0] != 'f')
1079 goto badloc;
1080 sscanf(tok.substr(1).c_str(), "%d", &offs);
1081 if (offs == 0)
1082 goto badloc;
1083 goto done;
1084 }
1085 if (tok[0] != 'w' || tok[1] < '1')
1086 goto badloc;
1087 offs = 0;
1088 sscanf(tok.substr(1).c_str(), "%d", &offs);
1089 if (offs == 0 || offs > SYMLEN)
1090 goto badloc;
1091 aftp = tok.c_str() + 2;
1092 if (offs > 9)
1093 aftp++;
1094 offs = (offs - 1) * 4;
1095 if (aftp[0] == 0)
1096 size = 4;
1097 else if (aftp[0] != ':') {
1098 printError(WARN, ": must follow word spec");
1099 size = 4;
1100 } else if (aftp[1] != 'h' && aftp[1] != 'b') {
1101 printError(WARN, "Bad subfield spec");
1102 size = 4;
1103 } else if (aftp[1] == 'h') {
1104 size = 2;
1105 if (aftp[2] < '1' || aftp[2] > '2')
1106 printError(WARN, "Bad halfword spec");
1107 else
1108 offs += (aftp[2] - '1') * 2;
1109 } else if (aftp[1] == 'b') {
1110 size = 1;
1111 if (aftp[2] < '1' || aftp[2] > '4')
1112 printError(WARN, "Bad byte spec");
1113 else
1114 offs += aftp[2] - '1';
1115 if (aftp[3] == '-') {
1116 if (fields[fields.size() - 1].name != "flags" || aftp[4] < '1' ||
1117 aftp[4] > '4')
1118 printError(WARN, "Bad flags spec");
1119 else
1120 size = (aftp[4] - aftp[2]) + 1;
1121 }
1122 }
1123
1124 it->type = getToken(); // optional type: may be empty
1125 done:
1126 it->size = size;
1127 it->offs = offs;
1128 it->shared = sharedflag;
1129 it->shareduse = shareduse;
1130 it->flag = flagflag;
1131 return it - fields.begin();
1132
1133 badloc:
1134 printError(WARN, "Bad field location");
1135 // intentional fall through
1136 fixup:
1137 it->size = fields[fields.size() - 1].offs = 0;
1138 it->shared = false;
1139 it->flag = false;
1140 it->shareduse = shareduse;
1141 return it - fields.begin();
1142 }
