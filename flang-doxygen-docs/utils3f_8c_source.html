17char *
18__fstr2cstr(char *from, int from_len)
19{
20 char *str;
21 int len;
22
23 for (len = from_len; len > 0; len--)
24 if (from[len - 1] != ' ')
25 break;
26 str = _mp_malloc(len + 1);
27 memcpy(str, from, len);
28 str[len] = '\0';
29 return str;
30}

38void
39__fcp_cstr(char *to, int to_len, char *from)
40{
41 char ch;
42
43 if (from)
44 while ((ch = *from++)) {
45 if (to_len-- <= 0)
46 break;
47 *to++ = ch;
48 }
49 while (to_len-- > 0)
50 *to++ = ' ';
51}

59bool __isatty3f(int unit)
60{
61 void *p;
62 int fd;
63
64 p = __fio_find_unit(unit);
65 if (p == NULL || FIO_FCB_STDUNIT(p)) {
66 switch (unit) {
67 case 0:
68 fd = 2; /* stderr */
69 break;
70 case 5:
71 fd = 0; /* stdin */
72 break;
73 case 6:
74 fd = 1; /* stdout */
75 break;
76 default:
77 return FALSE;
78 }
79 return __io_isatty(fd);
80 }
81 return FALSE;
82}

86FILE *__getfile3f(int unit)
87{
88 void *p;
89
90 p = __fio_find_unit(unit);
91 if (p != NULL) {
92 if (FIO_FCB_ASY_RW(p)) {/* disable any async i/o */
93 FIO_FCB_SET_ASY_RW(p, 0);
94 if (Fio_asy_disable(FIO_FCB_ASYPTR(p)) == -1) {
95 __abort(1, "3F routine found asynchronous I/O error");
96 }
97 }
98 return FIO_FCB_FP(p);
99 }
100 switch (unit) {
101 case 0:
102 return __io_stderr();
103 case 5:
104 return __io_stdin();
105 case 6:
106 return __io_stdout();
107 default:
108 return NULL;
109 }
110}

184__GETFILEINFOQQ(char *ffiles_arg, char *buffer, int *handle, int ffiles_len)
185{
186 char *files;
187 int rslt = 0, i;
188 WIN32_FIND_DATA FindFileData;
189 HANDLE hFind;
190 ULARGE_INTEGER fileTime;
191 char *name;
192 struct FILE$INFO {
193 unsigned int creation;
194 unsigned int lastWrite;
195 unsigned int lastAccess;
196 int length;
197 int permit;
198 char name[256];
199 } fileInfo;
200
201 files = __fstr2cstr(ffiles_arg, ffiles_len);
202 if (!files || !buffer || !handle) {
203 __io_errno();
204 return 0;
205 }
206 if (*handle <= 0) {
207 hFind = FindFirstFile(files, &FindFileData);
208 *handle = (int)hFind;
209 } else {
210 hFind = (HANDLE)*handle;
211 if (FindNextFile(hFind, &FindFileData) == 0) {
212 *handle = FILE$LAST;
213 FindClose(hFind);
214 goto rtn;
215 }
216 }
217
218 if (hFind == INVALID_HANDLE_VALUE) {
219 *handle = FILE$ERROR;
220 goto rtn;
221 }
222
223 fileTime.u.LowPart = FindFileData.ftCreationTime.dwLowDateTime;
224 fileTime.u.HighPart = FindFileData.ftCreationTime.dwHighDateTime;
225 __GetTimeToSecondsSince1970(&fileTime, &(fileInfo.creation));
226
227 fileTime.u.LowPart = FindFileData.ftLastWriteTime.dwLowDateTime;
228 fileTime.u.HighPart = FindFileData.ftLastWriteTime.dwHighDateTime;
229 __GetTimeToSecondsSince1970(&fileTime, &(fileInfo.lastWrite));
230
231 fileTime.u.LowPart = FindFileData.ftLastAccessTime.dwLowDateTime;
232 fileTime.u.HighPart = FindFileData.ftLastAccessTime.dwHighDateTime;
233 __GetTimeToSecondsSince1970(&fileTime, &(fileInfo.lastAccess));
234
235 fileInfo.length = (int)((FindFileData.nFileSizeHigh * (MAXDWORD + 1)) +
236 FindFileData.nFileSizeLow);
237 fileInfo.permit = FindFileData.dwFileAttributes;
238 name = FindFileData.cFileName;
239 rslt = strlen(name);
240
241 __fcp_cstr(fileInfo.name, 255, name);
242 memcpy(buffer, &fileInfo, sizeof(fileInfo));
243rtn:
244 __cstr_free(files);
245 return rslt;
246}
