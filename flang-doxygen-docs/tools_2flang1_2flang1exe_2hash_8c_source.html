20/** \file
21 * \brief Open addressing, quadratically probed hash tables.
22 *
23 * All keys are kept in a single array with 2^n entries. A key is first looked
24 * up at index hash(k) mod 2^k, then following a quadratic sequence of offsets.
25 * The quadratic probing has decent locality of reference for the first few
26 * probes.
27 *
28 * The search terminates when the key is found or a NULL is encountered. A
29 * special ERASED marker is used to avoid severing probe sequences when erasing
30 * keys.
31 */
32
33#ifdef HOST_WIN
34#define ERASED (hash_key_t) ~0ULL
35#define LONG long long
36#else
37#define ERASED (hash_key_t) ~0UL
38#define LONG long
39#endif
40#define ISKEY(k) ((k) != NULL && (k) != ERASED)
41
42struct hashset_ {
43 hash_functions_t func;
44 hash_key_t *table;
45
46 /** Table size minus 1 is a bit mask. */
47 unsigned mask;
48
49 /** Number of ISKEY() entries in table. */
50 unsigned entries;
51
52 /** Number of ERASED entries in table. */
53 unsigned erased;
54};

61#define DTABLE(h) (((h)->table + 1) + (h)->mask)
62
63/*
64 Quadrating probing sequence.
65
66 Given an initial hash value h, we probe at table indices
67
68 h, h+2, h+1+2, h+1+2+3, ... ( mod 2^n ), or
69
70 (i+1)i/2 ( mod 2^n ), for i = 0 .. 2^n.
71
72 This sequence visits the entire table without duplicates. Suppose two
73 indexes in the sequence are the same entry:
74
75 (i+1)i/2 = (j+1)j/2 ( mod 2^n )
76 ==> i^2 + i = j^2 + j ( mod 2^(n+1) )
77 ==> (i+j)(i-j) + i-j = 0 ( mod 2^(n+1) )
78 ==> (i+j+1)(i-j) = 0 ( mod 2^(n+1) )
79
80 Look at the difference of the factors on the left-hand side:
81
82 (i+j+1) - (i-j) = 2j + 1
83
84 This is an odd number, so one of the factors must be odd and thus coprime
85 with 2^(n+1). Therefore, 2^(n+1) must divide the other factor. The range of
86 the factors can be determined from the ranges of i and j:
87
88 1 <= i+j+1 <= 2^(n+1)-1, and
89 -(2^n-1) <= i-j <= 2^n-1.
90
91 The only way 2^(n+1) can divide one of the two factors is if 1 = j.
92
93 q.e.d.
94*/
95
96/** \brief Search for key, return index that terminated the search. */
97static unsigned
98search(hashset_t h, hash_key_t key)
99{
100 unsigned p, s = 1;
101 hash_equality_t eq = h->func.equals;
102
103 assert(ISKEY(key), "Invalid key for hash", HKEY2INT(key), ERR_Fatal);
104
105 p = h->func.hash(key) & h->mask;
106 if (eq) {
107 /* General case where we have to call eq() to determine if keys are
108 * equivalent. */
109 while (h->table[p]) {
110 if (h->table[p] != ERASED && eq(key, h->table[p]))
111 break;
112 p = (p + s++) & h->mask;
113 }
114 } else {
115 /* Optimized case when eq is NULL and we simply test for pointer
116 * equality. */
117 while (h->table[p]) {
118 if (key == h->table[p])
119 break;
120 p = (p + s++) & h->mask;
121 }
122 }
123 return p;
124}

126/** \brief Find insertion point for key. */
127static unsigned
128insertion_point(hashset_t h, hash_key_t key)
129{
130 unsigned p, s = 1;
131
132 assert(ISKEY(key), "Invalid key for hash", HKEY2INT(key), ERR_Fatal);
133
134 p = h->func.hash(key) & h->mask;
135 while (ISKEY(h->table[p])) {
136 p = (p + s++) & h->mask;
137 }
138 return p;
139}
140
141/*
142 The number of NULL entries is kept above 1/8th of the whole table to avoid
143 too long probe sequences. The table is rehashed before the NULL entries drop
144 below that.
145
146 The ideal table size is the smallest power of two that keeps the load factor
147 below 2/3.
148*/
149
150#define NEED_REHASH(h) ((h)->entries + (h)->erased >= (h)->mask - (h)->mask / 8)
151#define MINSIZE 16
152
153static unsigned
154mask_for_entries(unsigned n)
155{
156 unsigned m = (n + n / 2) | (MINSIZE - 1);
157
158 /* Arithmetic overflow happens after we have billions of entries. */
159 assert(m > n, "Hash table full", n, ERR_Fatal);
160
161 /* Round up to the next power of two minus one. */
162 m |= m >> 1;
163 m |= m >> 2;
164 m |= m >> 4;
165 m |= m >> 8;
166 m |= m >> 16;
167
168 return m;
169}

171/* Compute h->mask from h->entries and allocate table and dtable. */
172static void
173alloc_tables(hashset_t h, int with_dtable)
174{
175 h->mask = mask_for_entries(h->entries);
176
177 /* Allocate table + dtable in one calloc(). */
178 if (with_dtable)
179 h->table = (const void**) calloc(
180 1 + (size_t)h->mask, sizeof(hash_key_t) + sizeof(hash_data_t));
181 else
182 h->table = (const void**) calloc(1 + (size_t)h->mask, sizeof(hash_key_t));
183}

185/* Resize and rehash table to get rid of ERASED entries.
186
187 If not NULL, dtable points to a table of data entries that should be
188 rearranged the same way as h->table.
189 */
190static void
191rehash(hashset_t h, int with_dtable)
192{
193 hash_key_t *old_table = h->table;
194 hash_data_t *old_dtable = DTABLE(h);
195 size_t n, old_size = 1 + (size_t)h->mask;
196 hash_data_t *new_dtable = NULL;
197
198 alloc_tables(h, with_dtable);
199 new_dtable = DTABLE(h);
200
201 /* There will be no ERASED markers after the rehash. */
202 h->erased = 0;
203
204 for (n = 0; n < old_size; n++) {
205 hash_key_t key = old_table[n];
206 if (ISKEY(key)) {
207 unsigned i = insertion_point(h, key);
208 h->table[i] = key;
209 if (with_dtable)
210 new_dtable[i] = old_dtable[n];
211 }
212 }
213
214 free((void*)old_table);
215}

295void
296hashset_insert(hashset_t h, hash_key_t key)
297{
298 unsigned i;
299
300#if DEBUG
301 assert(hashset_lookup(h, key) == NULL, "Duplicate hash key", 0, ERR_Fatal);
302#endif
303
304 if (NEED_REHASH(h))
305 rehash(h, 0);
306
307 i = insertion_point(h, key);
308 if (h->table[i] == ERASED)
309 h->erased--;
310 h->table[i] = key;
311 h->entries++;
312}

314void
315hashmap_insert(hashmap_t h, hash_key_t key, hash_data_t data)
316{
317 unsigned i;
318
319#if DEBUG
320 assert(hashmap_lookup(h, key, NULL) == NULL, "Duplicate hash key", 0, ERR_Fatal);
321#endif
322
323 if (NEED_REHASH(&h->set))
324 rehash(&h->set, 1);
325
326 i = insertion_point(&h->set, key);
327 if (h->set.table[i] == ERASED)
328 h->set.erased--;
329 h->set.table[i] = key;
330 DTABLE(&h->set)[i] = data;
331 h->set.entries++;
332}

367/* We never rehash while erasing elements. The rehash() at insertion can also
368 * shrink the table. */
369hash_key_t
370hashset_erase(hashset_t h, hash_key_t key)
371{
372 unsigned i = search(h, key);
373 hash_key_t old = h->table[i];
374
375 if (!old)
376 return NULL;
377
378 h->table[i] = ERASED;
379 h->erased++;
380 h->entries--;
381 return old;
382}

397 *data = DTABLE(&h->set)[i];

426void
427hashmap_sort(hashmap_t h, void (*f)(hash_key_t, hash_data_t, void *context),
428 void *context)
429{
430 typedef int (*compare_func_t)(const void*, const void*);
431 size_t n,i, size = 1 + (size_t)h->set.mask;
432 size_t count = 0;
433 hash_data_t *dtable = DTABLE(&h->set);
434 sortmap_t *mysortmap;
435
436 for (n = 0; n < size; n++) {
437 hash_key_t key = h->set.table[n];
438 if (ISKEY(key))
439 count++;
440 }
441
442 mysortmap = (sortmap_t*) malloc(count * sizeof(sortmap_t));
443
444 i = 0;
445 for (n = 0; n < size; n++) {
446 hash_key_t key = h->set.table[n];
447 if (ISKEY(key)) {
448 mysortmap[i].table =(hash_key_t*)h->set.table[n];
449 mysortmap[i].dtable = dtable[n];
450 i++;
451 }
452 }
453
454 qsort(mysortmap, count, sizeof(sortmap_t), (compare_func_t)string_cmp);
455 for (n = 0; n < count; n++) {
456 f(mysortmap[n].table, mysortmap[n].dtable, context);
457 }
458
459 free(mysortmap);
460}

477static hash_value_t
478string_hash(hash_key_t key)
479{
480 const unsigned char *p = (const unsigned char *)key;
481 hash_accu_t hacc = HASH_ACCU_INIT;
482 for (; *p; p++)
483 HASH_ACCU_ADD(hacc, *p);
484 HASH_ACCU_FINISH(hacc);
485 return HASH_ACCU_VALUE(hacc);
486}

496/* Direct hashing. */
497static hash_value_t
498direct_hash(hash_key_t key)
499{
500 unsigned LONG k = (unsigned LONG)key;
501 hash_accu_t hacc = HASH_ACCU_INIT;
502 HASH_ACCU_ADD(hacc, k);
503 /* On an LP64 system, this ignores the high 8 bits of k. That's OK if k
504 * represents a pointer, since current 64-bit systems don't use the high
505 * bits of addresses. A shift by 32 would be undefined when long is a
506 * 32-bit type. */
507 HASH_ACCU_ADD(hacc, k >> 24);
508 HASH_ACCU_FINISH(hacc);
509 return HASH_ACCU_VALUE(hacc);
510}

548static void
549basic_direct(void **state)
550{
551 unsigned LONG i;
552 hashset_t h = hashset_alloc(hash_functions_direct);
553 assert_true(h != NULL);
554 assert_int_equal(hashset_size(h), 0);
555 assert_int_equal(h->mask, MINSIZE - 1);
556
557 assert_int_equal(hashset_lookup(h, (hash_key_t)1), 0);
558 assert_int_equal(hashset_lookup(h, (hash_key_t)2), 0);
559
560 hashset_insert(h, (hash_key_t)1);
561 assert_int_equal(hashset_lookup(h, (hash_key_t)1), 1);
562 assert_int_equal(hashset_lookup(h, (hash_key_t)2), 0);
563
564 assert_int_equal(hashset_replace(h, (hash_key_t)1), 1);
565 assert_int_equal(hashset_replace(h, (hash_key_t)2), 0);
566
567 for (i = 3; i <= 14; i++)
568 hashset_insert(h, (hash_key_t)i);
569
570 /* Table has 16 entries, 2 are NULL for exactly 7/8 load factor. */
571 assert_int_equal(hashset_size(h), 14);
572 assert_int_equal(h->mask, MINSIZE - 1);
573
574 /* Trigger a rehash when we add the 15th element. */
575 hashset_insert(h, (hash_key_t)15);
576 assert_int_equal(hashset_size(h), 15);
577 assert_int_equal(h->mask, 2 * MINSIZE - 1);
578
579 /* Remove odd entries. */
580 for (i = 1; i <= 15; i += 2)
581 assert_int_equal(hashset_erase(h, (hash_key_t)i), i);
582 assert_int_equal(hashset_size(h), 7);
583 assert_int_equal(h->mask, 2 * MINSIZE - 1);
584 assert_int_equal(h->erased, 8);
585
586 /* Remove everything. */
587 for (i = 1; i <= 15; i++)
588 assert_int_equal(hashset_erase(h, (hash_key_t)i), i % 2 ? 0 : i);
589
590 /* Set is empty, but hasn't rehashed yet. */
591 assert_int_equal(hashset_size(h), 0);
592 assert_int_equal(h->mask, 2 * MINSIZE - 1);
593 assert_int_equal(h->erased, 15);
594
595 for (i = 1; i <= 20; i++)
596 assert_int_equal(hashset_erase(h, (hash_key_t)i), 0);
597 assert_int_equal(hashset_size(h), 0);
598 assert_int_equal(h->mask, 2 * MINSIZE - 1);
599
600 /* Insert triggers a table shrink. Eventually. The exact timing depends on
601 * the hash function behavior. */
602 for (i = 1; i <= 10; i++)
603 hashset_insert(h, (hash_key_t)(100 + i));
604 assert_int_equal(h->mask, 2 * MINSIZE - 1);
605 for (i = 1; i <= 10; i++)
606 hashset_erase(h, (hash_key_t)(100 + i));
607 assert_int_equal(h->mask, 2 * MINSIZE - 1);
608 for (i = 1; i <= 10; i++)
609 hashset_insert(h, (hash_key_t)(200 + i));
610 assert_int_equal(h->mask, 2 * MINSIZE - 1);
611 for (i = 1; i <= 10; i++)
612 hashset_erase(h, (hash_key_t)(200 + i));
613 assert_int_equal(h->mask, 2 * MINSIZE - 1);
614 for (i = 1; i <= 10; i++)
615 hashset_insert(h, (hash_key_t)(300 + i));
616 assert_int_equal(h->mask, 2 * MINSIZE - 1);
617 for (i = 1; i <= 10; i++)
618 hashset_erase(h, (hash_key_t)(300 + i));
619 assert_int_equal(h->mask, 2 * MINSIZE - 1);
620 for (i = 1; i <= 10; i++)
621 hashset_insert(h, (hash_key_t)(400 + i));
622 assert_int_equal(h->erased, 0);
623 assert_int_equal(h->mask, MINSIZE - 1);
624
625 hashset_free(h);
626}

701static void
702basic_map(void **state)
703{
704 hashmap_t h = hashmap_alloc(hash_functions_direct);
705 hash_data_t datum;
706 const char *s1 = "foo", *s2 = "bar";
707 unsigned LONG i;
708
709 assert_int_equal(hashmap_size(h), 0);
710
711 /* Lookup nonexistent data with NULL data pointer. */
712 assert_int_equal(hashmap_lookup(h, (hash_key_t)1, NULL), 0);
713
714 /* Lookup nonexistent data with non-NULL data pointer. */
715 datum = &h;
716 assert_int_equal(hashmap_lookup(h, (hash_key_t)1, &datum), 0);
717 assert_int_equal(datum, &h);
