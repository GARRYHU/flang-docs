19/**
20 \brief VERIFY_LEVEL specifies the depth to which to verify ILI.
21
22 Deeper levels require more time. A good strategy is to choose a depth for
23 which the verification time can be amortized against compiler work that have
24 happened since the last verification.
25 */
26typedef enum VERIFY_LEVEL {
27 VERIFY_BLOCK, /**< Verify down to blocks, but no deeper. */
28 VERIFY_ILT, /**< Verify down to ILTs. */
29 VERIFY_ILI_SHALLOW, /**< Verify down to first ILI encountered. */
30 VERIFY_ILI_DEEP /**< Verify down to ILI leaves. */
31} VERIFY_LEVEL;
