18{

30/* pack, optional vector arg present. pack masked elements of array
31 into result and fill remainder of result with corresponding
32 elements of vector */

34void ENTFTN(PACK, pack)(void *rb, /* result base */
35 void *ab, /* array base */
36 void *mb, /* mask base */
37 void *vb, /* vector base */
38 F90_Desc *result, /* result descriptor */
39 F90_Desc *array, /* array descriptor */
40 F90_Desc *mask, /* mask descriptor */
41 F90_Desc *vector) /* vector descriptor */
42{
43 char *la, *rf, *vf;
44 __INT_T rindex;
45 __INT_T vindex;
46 __INT_T aindex[MAXDIMS];
47 __INT_T mindex[MAXDIMS];
48 __BIGREAL_T tmp[4];
49 __INT_T mlen;
50 __INT_T i, mask_is_array, more_array, more_vector, mval;
51
52 if (result == NULL || F90_TAG_G(result) != __DESC)
53 __fort_abort("PACK: invalid result descriptor");
54
55 if (vector == NULL || F90_TAG_G(vector) != __DESC)
56 __fort_abort("PACK: invalid vector descriptor");
57
58 if (F90_GSIZE_G(result) == 0 || F90_GSIZE_G(vector) == 0)
59 return;
60
61 rf = (char *)rb + DIST_SCOFF_G(result) * F90_LEN_G(result);
62 vf = (char *)vb + DIST_SCOFF_G(vector) * F90_LEN_G(vector);
63
64 rindex = F90_DIM_LBOUND_G(result, 0);
65 vindex = F90_DIM_LBOUND_G(vector, 0);
66 for (i = F90_RANK_G(array); --i >= 0;) {
67 aindex[i] = F90_DIM_LBOUND_G(array, i);
68 }
69
70 if (ISSCALAR(mask)) {
71 mlen = GET_DIST_SIZE_OF(TYPEKIND(mask));
72 mval = I8(__fort_varying_log)(mb, &mlen);
73 mask_is_array = 0;
74 } else if (F90_TAG_G(mask) == __DESC) {
75 for (i = F90_RANK_G(mask); --i >= 0;)
76 mindex[i] = F90_DIM_LBOUND_G(mask, i);
77 mask_is_array = 1;
78 } else
79 __fort_abort("PACK: invalid mask descriptor");
80
81 more_array = more_vector = 1;
82 while (more_array & more_vector) {
83
84 /* get mask value */
85
86 if (mask_is_array) {
87 I8(__fort_get_scalar)(tmp, mb, mask, mindex);
88 switch (F90_KIND_G(mask)) {
89 case __LOG1:
90 mval = (*(__LOG1_T *)tmp & GET_DIST_MASK_LOG1) != 0;
91 break;
92 case __LOG2:
93 mval = (*(__LOG2_T *)tmp & GET_DIST_MASK_LOG2) != 0;
94 break;
95 case __LOG4:
96 mval = (*(__LOG4_T *)tmp & GET_DIST_MASK_LOG4) != 0;
97 break;
98 case __LOG8:
99 mval = (*(__LOG8_T *)tmp & GET_DIST_MASK_LOG8) != 0;
100 break;
101 case __INT1:
102 mval = (*(__INT1_T *)tmp & GET_DIST_MASK_INT1) != 0;
103 break;
104 case __INT2:
105 mval = (*(__INT2_T *)tmp & GET_DIST_MASK_INT2) != 0;
106 break;
107 case __INT4:
108 mval = (*(__INT4_T *)tmp & GET_DIST_MASK_INT4) != 0;
109 break;
110 case __INT8:
111 mval = (*(__INT8_T *)tmp & GET_DIST_MASK_INT8) != 0;
112 break;
113 default:
114 __fort_abort("PACK: unknown mask type");
115 }
116 more_array &= I8(next_index)(mindex, mask);
117 }
118
119 /* if mask is true, store the corresponding array element into
120 the next result element and also advance to the next vector
121 element. */
122
123 if (mval) {
124 la = I8(__fort_local_address)(rf, result, &rindex);
125 if (la == NULL)
126 la = (char *)tmp;
127 I8(__fort_get_scalar)(la, ab, array, aindex);
128 more_vector &= I8(next_index)(&rindex, result);
129 more_vector &= I8(next_index)(&vindex, vector);
130 }
131 more_array &= I8(next_index)(aindex, array);
132 }
133
134 /* if there are fewer masked elements than result elements, fill
135 the remainder of the result with the corresponding vector
136 elements. */
137
138 while (more_vector) {
139 la = I8(__fort_local_address)(rf, result, &rindex);
140 if (la == NULL)
141 la = (char *)tmp;
142 I8(__fort_get_scalar)(la, vf, vector, &vindex);
143 more_vector &= I8(next_index)(&rindex, result);
144 more_vector &= I8(next_index)(&vindex, vector);
145 }
146}

305void ENTFTN(UNPACK, unpack)(void *rb, /* result base */
306 void *vb, /* vector base */
307 void *mb, /* mask base */
308 void *fb, /* field base */
309 F90_Desc *result, /* result descriptor */
310 F90_Desc *vector, /* vector descriptor */
311 F90_Desc *mask, /* mask descriptor */
312 F90_Desc *field) /* field descriptor */
313{
314 char *la, *rf;
315 __INT_T rindex[MAXDIMS];
316 __INT_T vindex;
317 __INT_T findex[MAXDIMS];
318 __INT_T mindex[MAXDIMS];
319 __BIGREAL_T tmp[4];
320 __INT_T field_is_array, i, more, mval;
321
322 if (result == NULL || F90_TAG_G(result) != __DESC)
323 __fort_abort("UNPACK: invalid result descriptor");
324
325 if (F90_GSIZE_G(result) == 0 || F90_GSIZE_G(mask) == 0)
326 return;
327
328 rf = (char *)rb + DIST_SCOFF_G(result) * F90_LEN_G(result);
329
330 for (i = F90_RANK_G(result); --i >= 0;)
331 rindex[i] = F90_DIM_LBOUND_G(result, i);
332
333 if (mask == NULL || F90_TAG_G(mask) != __DESC)
334 __fort_abort("UNPACK: invalid mask descriptor");
335
336 for (i = F90_RANK_G(mask); --i >= 0;)
337 mindex[i] = F90_DIM_LBOUND_G(mask, i);
338
339 vindex = F90_DIM_LBOUND_G(vector, 0);
340
341 if (ISSCALAR(field)) {
342 field_is_array = 0;
343 } else if (F90_TAG_G(field) == __DESC) {
344 for (i = F90_RANK_G(field); --i >= 0;)
345 findex[i] = F90_DIM_LBOUND_G(field, i);
346 field_is_array = 1;
347 } else
348 __fort_abort("UNPACK: invalid field descriptor");
349
350 more = 1;
351 while (more) {
352
353 /* get mask value */
354
355 I8(__fort_get_scalar)(tmp, mb, mask, mindex);
356 switch (F90_KIND_G(mask)) {
357 case __LOG1:
358 mval = (*(__LOG1_T *)tmp & GET_DIST_MASK_LOG1) != 0;
359 break;
360 case __LOG2:
361 mval = (*(__LOG2_T *)tmp & GET_DIST_MASK_LOG2) != 0;
362 break;
363 case __LOG4:
364 mval = (*(__LOG4_T *)tmp & GET_DIST_MASK_LOG4) != 0;
365 break;
366 case __LOG8:
367 mval = (*(__LOG8_T *)tmp & GET_DIST_MASK_LOG8) != 0;
368 break;
369 case __INT1:
370 mval = (*(__INT1_T *)tmp & GET_DIST_MASK_INT1) != 0;
371 break;
372 case __INT2:
373 mval = (*(__INT2_T *)tmp & GET_DIST_MASK_INT2) != 0;
374 break;
375 case __INT4:
376 mval = (*(__INT4_T *)tmp & GET_DIST_MASK_INT4) != 0;
377 break;
378 case __INT8:
379 mval = (*(__INT8_T *)tmp & GET_DIST_MASK_INT8) != 0;
380 break;
381 default:
382 __fort_abort("UNPACK: unknown mask type");
383 }
384
385 /* if the mask is true, move the next vector element to the
386 result element corresponding to the mask. Otherwise, copy
387 the field element corresponding to the mask to the result
388 element. */
389
390 la = I8(__fort_local_address)(rf, result, rindex);
391 if (la == NULL)
392 la = (char *)tmp;
393 if (mval) {
394 I8(__fort_get_scalar)(la, vb, vector, &vindex);
395 I8(next_index)(&vindex, vector);
396 } else if (field_is_array)
397 I8(__fort_get_scalar)(la, fb, field, findex);
398 else
399 __fort_scalar_copy[F90_KIND_G(result)](la, fb, F90_LEN_G(result));
400
401 more &= I8(next_index)(rindex, result);
402 more &= I8(next_index)(mindex, mask);
403 if (field_is_array)
404 more &= I8(next_index)(findex, field);
405 }
406}
