43/** \brief Make an argument list
44 *
45 * Create a compiler generated array temporary for an argument list.
46 * its size is cnt and its dtype is dt. NOTE that the caller may modify
47 * the size of this array (a target may require alignment other than int).
48 * mkarglist guarantees reuse as long as expb.arglcnt.next is reset
49 * to expb.arglcnt.start.
50 */
51void
52mkarglist(int cnt, DTYPE dt)
53{
54
55 DTYPE dtype;
56 ADSC *ad;
57 static INT ival[2];
58
59 ival[1] = cnt;
60 expb.arglist = getccsym('a', expb.arglcnt.next++, ST_ARRAY);
61 dtype = DTYPEG(expb.arglist);
62 if (dtype) {
63 ad = AD_DPTR(dtype);
64 if (ival[1] > ad_val_of(AD_NUMELM(ad)))
65 AD_NUMELM(ad) = AD_UPBD(ad, 0) = getSptr_bnd_con(ival[1]);
66 } else {
67 SCP(expb.arglist, SC_AUTO);
68 STYPEP(expb.arglist, ST_ARRAY);
69 dtype = get_array_dtype(1, dt);
70 DTYPEP(expb.arglist, dtype);
71
72 ad = AD_DPTR(dtype);
73 AD_MLPYR(ad, 0) = stb.i1;
74 AD_LWBD(ad, 0) = stb.i1;
75 AD_UPBD(ad, 0) = getSptr_bnd_con(ival[1]);
76 AD_NUMDIM(ad) = 1;
77 AD_SCHECK(ad) = 0;
78 AD_ZBASE(ad) = stb.i1;
79 AD_NUMELM(ad) = AD_UPBD(ad, 0); /* numelm must be set ater numdim */
80 }
81
82 if (expb.arglcnt.max < expb.arglcnt.next)
83 expb.arglcnt.max = expb.arglcnt.next;
84}

86/** \brief Create a block during expand
87 */
88void
89cr_block(void)
90{
91 expb.curbih = exp_addbih(expb.curbih);
92 BIH_LINENO(expb.curbih) = expb.curlin ? expb.curlin : gbl.lineno;
93 BIH_GUARDEE(expb.curbih) = expb.isguarded >= 1 ? 1 : 0;
94 expb.flags.bits.noblock = 0;
95 ILT_NEXT(0) = 0;
96 ILT_PREV(0) = 0;
97 if (expb.curlin && expb.flags.bits.dbgline) {
98 expb.curilt =
99 addilt(0, ad2ili(IL_QJSR, mkfunc("dbg_i_line"), ad1ili(IL_NULL, 0)));
100 ILT_DBGLINE(expb.curilt) = 1;
101 } else
102 expb.curilt = 0;
103 expb.curlin = 0;
104 if (EXPDBG(8, 32))
105 fprintf(gbl.dbgfil, "---cr_block: bih %d, ILM line %d\n", expb.curbih,
106 gbl.lineno);
107}

109/** \brief Write a block during expand
110 */
111void
112wr_block(void)
113{
114 int iltx, opc = 0;
115
116 if (EXPDBG(8, 64))
117 fprintf(gbl.dbgfil, "%6d ilm words in block %6d\n", expb.ilm_words,
118 expb.curbih);
119 expb.ilm_words = 0;
120 iltx = ILT_PREV(0);
121 if (iltx != 0)
122 opc = ILI_OPC(ILT_ILIP(iltx));
123 if (iltx == 0 || !ILT_BR(iltx) || (opc != IL_JMP && opc != IL_QSWITCH &&
124 opc != IL_JMPMK &&
125 opc != IL_JMPM))
126 BIH_FT(expb.curbih) = 1;
127
128 propagate_bihflags();
129
130 expb.flags.bits.callfg |= bihb.callfg;
131 wrilts(expb.curbih); /* wrilts zeros bihb.callfg & qjsrfg */
132#ifdef BIH_ASM
133 bihb.gasm = 0;
134#endif
135 if (EXPDBG(8, 32))
136 fprintf(gbl.dbgfil, "---wr_block: bih %d, ILM line %d\n", expb.curbih,
137 gbl.lineno);
138 expb.flags.bits.noblock = 1;
139}

141static void
142propagate_bihflags(void)
143{
144 /* Set of flags for which it's necessary to propagate from iltb
145 * to bihb to BIH because the write of an ilt may be deferred.
146 */
147 BIH_EX(expb.curbih) = bihb.callfg;
148 BIH_LDVOL(expb.curbih) = bihb.ldvol;
149 BIH_STVOL(expb.curbih) = bihb.stvol;
150 BIH_QJSR(expb.curbih) = bihb.qjsrfg;
151#ifdef BIH_ASM
152 BIH_ASM(expb.curbih) = bihb.gasm;
153#endif
154
155 /*
156 * Set of flags where it's known that a block has been created
157 * because of the context implied by the flags.
158 */
159 BIH_PAR(expb.curbih) |= bihb.parfg;
160 BIH_CS(expb.curbih) |= bihb.csfg;
161 BIH_PARSECT(expb.curbih) |= bihb.parsectfg;
162 BIH_TASK(expb.curbih) |= bihb.taskfg;
163}

174/** \brief Put out expb.saveili if necesssary.
175 *
176 * May be an unadded ili if expb.flags.bits.waitlbl is set
177 */
178static void
179flsh_saveili(void)
180{
181 int savefg; /* save for the ilt call flag */
182 char ldvol; /* save for the ilt ldvol flag */
183 char stvol; /* save for the ilt stvol flag */
184 bool qjsrfg; /* save for the ilt qjsrfg flag */
185
186 if (expb.flags.bits.waitlbl) {
187
188 /*
189 * waiting for a label ilm; curilt will become the end of the current
190 * block if the opt level is not 1 -- i.e., the current block is
191 * written out and a new block is created. And, the ilt for saveili
192 * becomes the first ilt of this block. For opt level 1, an ilt for
193 * saveili is added to the current block
194 *
195 * NOTE: this does not happen at opt=0 -- this flag is not set (see the
196 * end of this routine)
197 */
198
199 expb.flags.bits.waitlbl = 0;
200 if (flg.opt != 1) {
201 if (EXPDBG(8, 32))
202 fprintf(gbl.dbgfil, "---flsh_saveili: wait end, curilt %d\n",
203 expb.curilt);
204
205 /* write out the block with curilt as its last ilt */
206
207 wr_block();
208
209 /*
210 * create a new block - saveili will be the first ilt of the new
211 * block
212 */
213 cr_block();
214 }
215 /* append the ilt to the current block */
216
217 if (EXPDBG(8, 32))
218 fprintf(gbl.dbgfil, "---flsh_saveili: wait add, curilt %d, saveili %d\n",
219 expb.curilt, expb.saveili);
220 savefg = iltb.callfg;
221 ldvol = iltb.ldvol;
222 stvol = iltb.stvol;
223 qjsrfg = iltb.qjsrfg;
224 iltb.callfg = 0;
225 iltb.ldvol = 0;
226 iltb.stvol = 0;
227 iltb.qjsrfg = false;
228 expb.curilt = addilt(expb.curilt, expb.saveili);
229 iltb.callfg = savefg;
230 iltb.ldvol = ldvol;
231 iltb.stvol = stvol;
232 iltb.qjsrfg = qjsrfg;
233 }
234}

236/** \brief Check for end of ILT block
237 *
238 * Check if the current block is at its end. newili locates an
239 * ili tree which represents an ili statement.
240 */
241void
242chk_block(int newili)
243{
244 int ili; /* ili of the current ilt */
245 ILI_OP opc; /* ili opcode of the current ilt */
246 int c_noprs; /* # of operands for conditional branch */
247 int c_lb; /* label of conditional branch */
248 int u_lb; /* label of unconditional branch */
249 int old_lastilt;
250
251 if (newili == 0)
252 /* seems odd that newili is 0, but this can happen if one is adding
253 * a branch ili -- iliutil could change the branch into a nop.
254 * The caller could check the ili first -- there are many cases
255 * where we do -- but it's something that can be easily overlooked.
256 */
257 return;
258
259 flsh_saveili(); /* write out saveili if waitlbl is set. (see
260 * below) */
261
262 if (!ILT_BR(expb.curilt) && (flg.opt < 2 || !ILT_CAN_THROW(expb.curilt))) {
263
264 /* The current end of the block is not a branch and [at -O2] cannot throw.
265 * Just create a new ilt and add it to the block. Note that if the block
266 * is null (expb.curilt is zero), the flags of ilt 0 are zero.
267 */
268 if (EXPDBG(8, 32))
269 fprintf(gbl.dbgfil, "---chk_block: curilt %d not br, add newili %d\n",
270 expb.curilt, newili);
271 old_lastilt = expb.curilt;
272 expb.curilt = addilt(expb.curilt, newili);
273 /* addilt does not update BIH_ILTLAST(). We need to do it here: */
274 if (expb.curbih && BIH_ILTLAST(expb.curbih) == old_lastilt) {
275 BIH_ILTLAST(expb.curbih) = expb.curilt;
276 }
277 return;
278 }
279 /* the current end of the block is some sort of branch,
280 call that can throw, or store of result of a call
281 that can throw. */
282
283 ili = ILT_ILIP(expb.curilt);
284 opc = ILI_OPC(ili);
285 if (opc == IL_JMP) {
286
287 /*
288 * the current end of the block is an unconditional branch. newili is
289 * unreachable, don't add it
290 */
291 opc = ILI_OPC(newili);
292 if (IL_TYPE(opc) == ILTY_BRANCH &&
293 IL_OPRFLAG(opc, ilis[opc].oprs) == ILIO_SYM)
294 RFCNTD(ILI_OPND(newili, ilis[opc].oprs));
295 iltb.callfg = 0;
296 if (EXPDBG(8, 32))
297 fprintf(gbl.dbgfil, "---chk_block: newili %d not reached\n", newili);
298 return;
299 }
300 /* the current end of the block is a conditional branch,
301 or something that can throw. */
302
303 if (ILI_OPC(newili) != IL_JMP || ILT_CAN_THROW(expb.curilt)) {
304
305 /* the new ili is not an unconditional branch, or the new ili is
306 an IL_JMP just past a potential throw point. */
307
308 if (flg.opt == 1 && !XBIT(137, 1) && !XBIT(163, 1))
309 /* create an extended basic block -- add a new ilt */
310
311 expb.curilt = addilt(expb.curilt, newili);
312 else {
313
314 /*
315 * the current block is at its end; write it out, create a new
316 * block with its first ilt locating newili
317 */
318 wr_block();
319 cr_block();
320 expb.curilt = addilt(expb.curilt, newili);
321 }
322 if (EXPDBG(8, 32))
323 fprintf(gbl.dbgfil, "---chk_block: add newili %d\n", newili);
324 return;
325 }
326 /*
327 * the current end of the block is a conditional branch and the new ili
328 * is an unconditional branch
329 */
330 c_noprs = ilis[opc].oprs;
331 c_lb = ILI_OPND(ili, c_noprs);
332 u_lb = ILI_OPND(newili, 1);
333 if (opc != IL_JMPM &&
334 opc != IL_JMPA &&
335 opc != IL_JMPMK &&
336 u_lb == c_lb) {
337
338 /*
339 * the labels are the same; search the current ilt and create new
340 * ilts for any procedures found. These ilts are added before the
341 * current ilt
342 */
343 (void)reduce_ilt((int)ILT_PREV(expb.curilt), ili);
344
345 /*
346 * the current ilt is changed to be the unconditional branch; the
347 * label's reference count is also decremented
348 */
349 if (EXPDBG(8, 32))
350 fprintf(gbl.dbgfil,
351 "---chk_block: uncond/cond, newili %d, to same label\n", newili);
352 ILT_ILIP(expb.curilt) = newili;
353 RFCNTD(ILI_OPND(newili, 1));
354 return;
355 }
356 /*
357 * the current end of the block is a conditional branch and the new ili
358 * is an unconditional branch whose labels are different. the checking of
359 * this block is delayed just in case a label ilm is processed before any
360 * other ilt producing ilms. This is done to catch cases of the form:
361 * if <cond> goto x;
362 * goto y;
363 * x: --- NOTE: this does not happen at opt 0;
364 * the current block is written out (with previlt as the last ilt) and
365 * previlt becomes the first ilt of the new block.
366 */
367 if (flg.opt != 0) {
368 if (EXPDBG(8, 32))
369 fprintf(gbl.dbgfil,
370 "---chk_block: uncond/cond, newili %d, to diff label\n", newili);
371 if (opc != IL_JMPM &&
372 opc != IL_JMPA &&
373 opc != IL_JMPMK &&
374 ILIBLKG(c_lb) == 0 && ILIBLKG(u_lb)) {
375 /* conditional branch is a forward branch; unconditional branch
376 * is a backward branch
377 */
378 ILT_ILIP(expb.curilt) = compl_br(ili, u_lb);
379 newili = ad1ili(IL_JMP, c_lb);
380 if (EXPDBG(8, 32))
381 fprintf(gbl.dbgfil, "---chk_block: swap lbls %d %d, c_br %d, u_br %d\n",
382 c_lb, u_lb, (int)ILT_ILIP(expb.curilt), newili);
383 }
384 expb.flags.bits.waitlbl = 1;
385 expb.saveili = newili;
386
387 } else {
388 wr_block(); /* the current block */
389 cr_block();
390 expb.curilt = addilt(expb.curilt, newili);
391 }
392}

407/** \brief Check an ILM which has been evaluated
408 *
409 * This routine checks an ILM which has already been evaluated
410 * (i.e., the ILM is referenced again). Depending on the type
411 * of the ili which defines this ILM, certain actions may occur
412 * to "redefine" the ILM (i.e., create a new ILI for the ILM).
413 *
414 * \param ilmx -- ILM index of the ILM evaluated
415 * \param ilix -- ILI index of the ILI for the ILM
416 */
417int
418check_ilm(int ilmx, int ilix)
419{
420 int cse; /* cse ILI */
421 SPTR sym; /* symbol table index */
422 int base, /* address ILI */
423 nme, /* names entry */
424 blk, /* bih index */
425 iltx; /* ilt index */
426 ILI_OP opc; /* opcode of ilix */
427
428 int saveilix = ilix;
429 switch (IL_TYPE(opc = ILI_OPC(ilix))) {
430
431 case ILTY_CONS:
432 /* a constant ILI is okay to re-use */
433 if (EXPDBG(8, 2))
434 fprintf(gbl.dbgfil, "check_ilm, ILM const: ilm %d, result ili %d\n", ilmx,
435 ilix);
436 return ilix;
437
438 case ILTY_ARTH:
439 case ILTY_LOAD:
440 case ILTY_DEFINE:
441 case ILTY_MOVE:
442/*
443 * these results represent an expression which may contain
444 * side-effect operations. The new result is a CSE ILI of ilix
445 */
446 /* assertion: for fortran the only side-effects which may occur are
447 * those due to function calls; however, there are cases where
448 * it's necessary that several ili statements "belong" to the
449 * statement. To represent this, the sequence will begin with
450 * a pseudo store, followed by statements which have cse uses.
451 * For this case, the assertion (fortran-only) is that this
452 * will only occur for pseudo stores and stores.
453 */
454 cse = ilix;
455 if (EXPDBG(8, 2))
456 fprintf(gbl.dbgfil,
457 "check_ilm, ILM expr: ilm %d, old ili %d, cse ili %d\n", ilmx,
458 ilix, cse);
459 /*
460 * no longer have:
461 * !BIH_QJSR(expb.curbih) &&
462 * leader block may not yet be created
463 */
464 if (!iltb.qjsrfg && qjsr_in(ilix)) {
465 iltb.qjsrfg = true;
466 if (EXPDBG(8, 2))
467 fprintf(gbl.dbgfil, "check_ilm - qjsr_in(%d)\n", ilix);
468 }
469 return cse;
470
471 case ILTY_OTHER:
472#ifdef ILTY_PSTORE
473 case ILTY_PSTORE:
474#endif
475#ifdef ILTY_PLOAD
476 case ILTY_PLOAD:
477#endif
478 /* handle FREEIR... ili with ILTY_STORE */
479 if (!is_freeili_opcode(opc)) {
480 /* not a FREE ili */
481 return ilix;
482 }
483 goto like_store;
484
485 case ILTY_STORE:
486/*
487 * assertion: for fortran the only side-effects which may occur are
488 * those due to function calls; however, there are cases where
489 * it's necessary that several ili statements "belong" to the
490 * statement. To represent this, the sequence will begin with
491 * a pseudo store, followed by statements which have cse uses.
492 * For this case, the assertion (fortran-only) is that this
493 * will only occur for pseudo stores and stores.
494 */
495
496 like_store:
497 /*** __OLDCSE ***/
498
499 cse = ad_cse((int)ILI_OPND(ilix, 1));
500
501 if ((blk = ILM_BLOCK(ilmx)) == expb.curbih) {
502 if (ILM_OPC((ILM *)(ilmb.ilm_base + ilmx)) == IM_PSEUDOST) {
503 /* We generate a pseudo store for postfix expressions.
504 * If the value prior to incrementing/decrementing is needed
505 * later by an arithmetic op, then we cannot safely CSE it.
506 * For example (from PH lang test): i++ != 2 ? 0 : i; In
507 * this case, a comparison followed by a branch needs the
508 * old value.
509 */
510 int len, opc2, i, found_use, found_arth;
511 int use = ILM_OPND((ILM *)(ilmb.ilm_base + ilmx), 2);
512 int ilmx2 = ilmx + ilms[IM_PSEUDOST].oprs + 1;
513 for (found_use = 0, found_arth = 0; ilmx2 < expb.nilms; ilmx2 += len) {
514 opc2 = ILM_OPC((ILM *)(ilmb.ilm_base + ilmx2));
515 len = ilms[opc2].oprs + 1;
516 if (IM_VAR(opc2))
517 len += ILM_OPND((ILM *)(ilmb.ilm_base + ilmx2), 1);
518 for (i = 1; i < len; ++i) {
519 if (IM_OPRFLAG(opc2, i) == OPR_LNK &&
520 ILM_OPND((ILM *)(ilmb.ilm_base + ilmx2), i) == use) {
521 found_use = 1;
522 } else if (found_use && IM_OPRFLAG(opc2, i) == OPR_LNK &&
523 ILM_OPND((ILM *)(ilmb.ilm_base + ilmx2), i) == ilmx) {
524 /* Convert the pseudo store to a real
525 * store on comparisons and other artimetics only.
526 */
527 if (IM_TYPE(opc2) != IMTY_ARTH) {
528 found_arth = 0;
529 goto break_out;
530 }
531 found_arth = 1;
532 }
533 }
534 }
535 break_out:
536 if (found_arth)
537 goto conv_pseudo_st;
538
539 } /* end if (ILM_OPC(...) == IM_PSEUDOST) */
540
541 /* reference is in the current block */
542
543 if (EXPDBG(8, 2))
544 fprintf(gbl.dbgfil,
545 "check_ilm, store re-used: ilm %d, old ili %d, cse ili %d\n",
546 ilmx, ilix, cse);
547 return cse;
548 } else {
549 /* The reference is in the current block if we jumped here
550 * from above (see the lines after "break_out:"), otherwise
551 * it's across block boundaries.
552 */
553 int save_iltb_callfg, save_iltb_ldvol, save_iltb_stvol;
554 bool save_iltb_qjsrfg;
555
556 conv_pseudo_st:
557 /* JHM (8 Dec 2011) bug-fix:
558 * This is quite complicated so I'll explain the logic in
559 * full detail. We're about to read in the ILTs of a
560 * block 'blk' (which may be 'expb.curbih' itself or a
561 * different block), then possibly add a new store ILT to
562 * it, then read in the ILTs of 'expb.curbih' again.
563 * Before doing this we must do the following:
564 *
565 * (1) Save the values of 'iltb.{x}', where {x} = {callfg,
566 * ldvol, stvol, qjsrfg}, to be restored later.
567 *
568 * (2) Set bihb.{x} |= iltb.{x}
569 *
570 * (3) Call 'propagate_bihflags()' to copy each bihb.{x}
571 * value to the corresponding BIH_{X}( expb.curbih ) field.
572 *
573 * Then at the end, after reading in 'expb.curbih's ILTs again:
574 *
575 * (4) Copy the saved values back to 'iltb.{x}'.
576 *
577 * Normally (2) would be performed by the next call to
578 * 'addilt()' and (3) by the next call to 'wr_block()'.
579 * However, because we call 'rdilts( blk )' and possibly
580 * add an ILT to 'blk', the above fields may be over-written
581 * before these actions can take place, as follows:
582 *
583 * -- 'rdilts( blk )' sets:
584 * bihb.{x} = BIH_{X}( blk )
585 * thus (potentially) over-writing the current values of bihb.{x}.
586 *
587 * -- Then 'addilt()' (if it's called) sets:
588 * bihb.{x} |= iltb.{x}
589 * iltb.{x} = 0
590 * thus over-writing the current values of iltb.{x}.
591 *
592 * -- Finally 'rdilts( expb.curbih )' sets:
593 * bihb.{x} = BIH_{X}( expb.curbih )
594 * which is only correct if BIH_{X}( expb.curbih ) contains
595 * the correct values, i.e. if actions (2) and (3) above
596 * have already been performed.
597 *
598 * By performing actions (2) and (3) we ensure that
599 * bihb.{x} and BIH_{X}( expb.curbih ) are correct at the
600 * end of this code, and (1) and (4) ensure that the
601 * current values are restored to 'iltb.{x}'.
602 */
603 bihb.callfg |= (save_iltb_callfg = iltb.callfg);
604 bihb.ldvol |= (save_iltb_ldvol = iltb.ldvol);
605 bihb.stvol |= (save_iltb_stvol = iltb.stvol);
606 bihb.qjsrfg |= (save_iltb_qjsrfg = iltb.qjsrfg);
607
608 iltb.callfg = 0; /* ...it's used by 'addilt()' to set ILT_EX() */
609 propagate_bihflags();
610 fihb.currfindex = BIH_FINDEX(expb.curbih);
611 fihb.currftag = BIH_FTAG(expb.curbih);
612 wrilts(expb.curbih); /* write out the current block and */
613 rdilts(blk); /* read in the block of the store */
614
615 sym = mkrtemp_sc(cse, expb.sc);
616 base = ad_acon(sym, (INT)0);
617 nme = addnme(NT_VAR, sym, 0, (INT)0);
618
619 for (iltx = ILT_PREV(0); iltx != 0; iltx = ILT_PREV(iltx)) {

622 case IL_ST:

624 break;
625 case IL_FREEIR:
626 ilix = ad4ili(IL_ST, cse, base, nme, MSZ_WORD);
627 break;
628 case IL_STKR:
629 ilix = ad4ili(IL_STKR, cse, base, nme, ILI_OPND(ilix, 4));
630 break;
631 case IL_FREEKR:
632 ilix = ad4ili(IL_STKR, cse, base, nme, MSZ_I8);
633 break;
634 case IL_STA:
635 case IL_FREEAR:
636 ilix = ad3ili(IL_STA, cse, base, nme);

638 break;
639 case IL_STSP:
640 case IL_FREESP:
641 ilix = ad4ili(IL_STSP, cse, base, nme, MSZ_F4);
642 break;
643 case IL_STDP:
644 case IL_FREEDP:
645 ilix = ad4ili(IL_STDP, cse, base, nme, MSZ_F8);
646 break;
647 case IL_STSCMPLX:
648 case IL_FREECS:
649 ilix = ad4ili(IL_STSCMPLX, cse, base, nme, MSZ_F8);
650 break;
651 case IL_STDCMPLX:
652 case IL_FREECD:
653 ilix = ad4ili(IL_STDCMPLX, cse, base, nme, MSZ_F16);
654 break;
655#ifdef LONG_DOUBLE_FLOAT128
656 case IL_FLOAT128ST:
657 case IL_FLOAT128FREE:
658 ilix = ad4ili(IL_FLOAT128ST, cse, base, nme, MSZ_F16);
659 break;
660#endif /* LONG_DOUBLE_FLOAT128 */
661 default:
662 interr("check_ilm: illegal store", ilix, ERR_Severe);
663 goto wr_out;
664 }
665 ADDRCAND(ilix, nme);
666 iltx = addilt(iltx, ilix);
667 ilix = ad_load(ilix);
668 if (ilix) {
669 if (EXPDBG(8, 2))
670 fprintf(gbl.dbgfil,
671 "check_ilm: store across block, ilm %d, ilt %d, ili %d\n",
672 ilmx, iltx, ilix);
673 ADDRCAND(ilix, nme);
674 } else {
675 ilix = ILT_ILIP(iltx);
676 interr("check_ilm: illegal store1", ilix, ERR_Severe);
677 }
678 goto wr_out;
679 }
680 }
681
682 /* no store found; just use cse as the result */
683
684 ilix = cse;
685 if (EXPDBG(8, 2))
686 fprintf(gbl.dbgfil,
687 "check_ilm: store not found, ilm %d, ilt %d, ili %d\n", ilmx,
688 iltx, ilix);
689
690 wr_out:
691 wrilts(blk); /* write out the modified block */
692 rdilts(expb.curbih); /* read back in the current block */
693
694 iltb.callfg = save_iltb_callfg;
695 iltb.ldvol = save_iltb_ldvol;
696 iltb.stvol = save_iltb_stvol;
697 iltb.qjsrfg = save_iltb_qjsrfg;
698 }

701 default:
702 if (EXPDBG(8, 2))
703 fprintf(gbl.dbgfil,
704 "check_ilm: bad reference, ilm %d(%s), ili %d, iliopc %d\n", ilmx,
705 ilms[ILM_OPC((ILM *)(ilmb.ilm_base + ilmx))].name, ilix, opc);
706 interr("check_ilm: bad reference", ilmx, ERR_Severe);
707 }
708

710}

765int
766access_swtab_base_label(int base_label, int sptr, int flag)
767{
768
769 /* Store a list of jump table sptr's and their branch label. Used
770 * on the ST100 for the enhanced method 3 scheme.
771 * if flag == -2 then we are just access the routine's mode (GP16/GP32).
772 * if flag == -1 then we are just accessing the routine sptr
773 * If flag == 1 then we are just accessing the value.
774 * If flag == 0 then we are adding a new value.
775 * If flag == 2 then we are also removing the record from the list.
776 * Returns 0 if not found, else branch_label_sptr.
777 */
778
779 typedef struct swtab_branch_label {
780 int branch_label;
781 int sptr_swtab;
782 int sptr_routine;
783 int mode;
784 struct swtab_branch_label *next;
785 } swtab_branch_label;
786
787 int branch_label_sptr = 0, mode = 0, routine_sptr = 0;
788
789 static swtab_branch_label swtab_info = {0, 0, 0, 0, 0};
790
791 swtab_branch_label *curr, *prev;
792
793 if (flag != 0) {
794
795 prev = &swtab_info;
796 for (curr = swtab_info.next; curr; curr = curr->next) {
797
798 if (curr->sptr_swtab == sptr) {
799
800 branch_label_sptr = curr->branch_label;
801 mode = curr->mode;
802 routine_sptr = curr->sptr_routine;
803 if (flag == 2) {
804 /* Remove record from list */
805 prev->next = curr->next;
806 FREE(curr);
807 }
808
809 if (flag == -1) {
810 return routine_sptr;
811 } else if (flag == -2) {
812 return mode;
813 } else {
814 return branch_label_sptr;
815 }
816 }
817 prev = curr;
818 }
819
820 } else {
821 NEW(curr, swtab_branch_label, sizeof(swtab_branch_label));
822 curr->branch_label = base_label;
823 curr->sptr_swtab = sptr;
824 curr->sptr_routine = gbl.currsub;
825 curr->mode = 0;
826 curr->next = swtab_info.next;
827 swtab_info.next = curr;
828 branch_label_sptr = base_label;
829 }
830
831 return branch_label_sptr;
832}

896int

898{

901 INT case_val;
902 SWEL *swel;

905 sym = getccsym('J', expb.swtcnt++, ST_PLIST); /* get switch array */
906 DTYPEP(sym, tabdtype);
907 SCP(sym, SC_STATIC);
908 PLLENP(sym, n);
909
910 /* initialize this array with the switch table */
911
912 SWELP(sym, swhdr - switch_base);
913 DEFLABP(sym, deflab);
914 if (doinit) {
915/*
916 * generate the entry for the default label.
917 */
918 DINITSWTAB_put(DINIT_LOC, sym);
919
920 if (deflab) {
921 /*
922 * If the default label is passed, then it will not appear in the
923 * switch list. Just make sure that swel locates the first case.
924 */
925 swel = swhdr;
926 } else {
927 /*
928 * Since the default label is not passed, then the switch list
929 * contains the default label. Extract it and make sure that it
930 * gets added to the initialization of the switch array.
931 */
932 deflab = swhdr->clabel;
933 DINITSWTAB_put(DINIT_LABEL, (INT)deflab);
934 swel = switch_base + swhdr->next;
935 }
936
937 case_val = swel->val; /* start with first case value */
938 do {
939 /*
940 * generate the remainder of the label table -- if there are
941 * holes, the default label is generated. case_val denotes
942 * the expected case value.
943 */
944 for (case_val = swel->val - case_val; case_val; case_val--) {
945 DINITSWTAB_put(DINIT_LABEL, deflab); /* default */
946 RFCNTI((int)swhdr->clabel);
947#if DEBUG
948 interr("mk_swtab: CGOTO has holes, swidx", swhdr - switch_base, ERR_Warning);
949#endif
950 }
951 DINITSWTAB_put(DINIT_LABEL, (INT)swel->clabel); /* case label */
952 case_val = swel->val + 1;
953 i = swel->next;
954 swel = switch_base + i;
955 } while (i != 0);
956 }
957
958 return sym;
959}
