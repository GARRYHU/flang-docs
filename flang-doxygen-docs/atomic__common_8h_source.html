12/**
13 * \brief Memory reference size/type codes.
14 *
15 * Legacy assumptions observed:
16 * - (code & 3) < 2 if and only if the type size is 1 or 2 bytes.
17 * - (code & 3) == log2(type size) if the type size is 1, 2, 4, or 8 bytes.
18 */
19typedef enum MSZ {
20 MSZ_SBYTE = 0x00, /* signed byte */
21 MSZ_SHWORD = 0x01, /* signed 16-bit short */
22 MSZ_UBYTE = 0x04, /* unsigned byte */
23 MSZ_UHWORD = 0x05, /* unsigned 16-bit short */
24 MSZ_FHALF = 0x09, /* 16-bit half precision float */
25
26 /* Codes for types larger than two bytes. These are all distinct values
27 * suitable for use as case labels in switches. The holes in this sequence
28 * of code values avoid violating the first legacy assumption described above.
29 */
30 MSZ_SWORD = 0x02, /* signed 32-bit int */
31 MSZ_SLWORD = 0x03, /* signed 64-bit long */
32 MSZ_UWORD = 0x06, /* unsigned 32-bit int */
33 MSZ_ULWORD = 0x07, /* unsigned 64-bit long */
34 MSZ_FWORD = 0x0a, /* 32-bit single precision float */
35 MSZ_FLWORD = 0x0b, /* 64-bit double precision float */
36 MSZ_I8 = 0x0f, /* distinct 64-bit integer type */
37 MSZ_PTR = 0x13, /* distinct 64-bit pointer type */
38 MSZ_F10 = 0x16, /* X87 FPU 80-bit extended precision */
39 MSZ_F16 = 0x17, /* 128-bit quad precision float */
40 MSZ_F32 = 0x1a, /* 256-bit float */
41 MSZ_F8x2 = 0x1b, /* 128-bit double-double float */
42
43 MSZ_UNDEF = 0xff, /* undefined MSZ code */
44} MSZ;

46/** Specifies memory order of an atomic operation.
47 Values corresponding to C11/C++11 memory orders are guaranteed
48 to match those of the target's C11/C++11 header. */
49typedef enum MEMORY_ORDER {

51 MO_CONSUME,
52 MO_ACQUIRE,
53 MO_RELEASE,
54 MO_ACQ_REL,
55 MO_SEQ_CST,

57 MO_MAX_DEF = MO_SEQ_CST, /**< maximum value with defined meaning */
58 MO_UNDEF = 0xFF /**< denotes "undefined" */
59} MEMORY_ORDER;
