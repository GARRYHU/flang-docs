51static void
52dotp_cplx8(__CPLX8_T *c, int nj, __CPLX8_T *a, int ai, int ais, __CPLX8_T *b,
53 int bk, int bks)
54{
55 register double cr, ci;
56 cr = c->r;
57 ci = c->i;
58 for (; --nj >= 0; ai += ais, bk += bks) {
59 cr += a[ai].r * b[bk].r + a[ai].i * b[bk].i;
60 ci += a[ai].r * b[bk].i - a[ai].i * b[bk].r;
61 }
62 c->r = cr;
63 c->i = ci;
64}

217void ENTFTN(DOTPR, dotpr)(char *cb, char *ab0, char *bb0, F90_Desc *cs,
218 F90_Desc *as0, F90_Desc *bs0)
219{
220 char *ab = 0, *bb = 0;
221 DECL_HDR_PTRS(as);
222 DECL_HDR_PTRS(bs);
223 DECL_HDR_VARS(as1);
224 DECL_HDR_VARS(bs1);
225 DECL_DIM_PTRS(ajd);
226 DECL_DIM_PTRS(bjd);
227 __INT_T flags, kind, len;
228 mmul_fn dotp;
229 __INT_T al, alof, an, aoff, astr, au, bl, blof, bn, boff, bstr, bu, acl, bcl,
230 cn;
231 int copy_required;
232
233#if defined(DEBUG)
234 if (__fort_test & DEBUG_MMUL) {
235 printf("%d dotpr a", GET_DIST_LCPU);
236 I8(__fort_show_section)(as0);
237 printf(" x b");
238 I8(__fort_show_section)(bs0);
239 printf("\n");
240 }
241 if (F90_TAG_G(cs) == __DESC && F90_RANK_G(cs) != 0)
242 __fort_abort("DOT_PRODUCT: result must be scalar");
243#endif
244
245 kind = F90_KIND_G(as0);
246 len = F90_LEN_G(as0);
247
248 copy_required = I8(is_nonsequential_section)(as0, F90_RANK_G(as0));
249 if (copy_required) {
250 as = as1;
251 flags = (__ASSUMED_SHAPE + __ASSUMED_OVERLAPS + __INTENT_IN + __INHERIT +
252 __TRANSCRIPTIVE_DIST_TARGET + __TRANSCRIPTIVE_DIST_FORMAT);
253 ENTFTN(QOPY_IN, qopy_in)
254 (&ab, (__POINT_T *)ABSENT, ab0, as, ab0, as0, &_1, &kind, &len, &flags,
255 &_1);
256 } else {
257 ab = ab0;
258 as = as0;
259 }
260
261 copy_required = I8(is_nonsequential_section)(bs0, F90_RANK_G(bs0));
262 if (copy_required)
263 {
264 bs = bs1;
265 flags = (__ASSUMED_SHAPE + __ASSUMED_OVERLAPS + __INTENT_IN +
266 __PRESCRIPTIVE_ALIGN_TARGET + __IDENTITY_MAP);
267 ENTFTN(QOPY_IN, qopy_in)
268 (&bb, (__POINT_T *)ABSENT, bb0, bs, bb0, bs0, &_1, &kind, &len, &flags, as,
269 &_1, &_1); /* conform, lb */
270 } else {
271 bb = bb0;
272 bs = bs0;
273 }
274
275 switch (kind) {
276 case __REAL4:
277 dotp = (mmul_fn)dotp_real4;
278 break;
279 case __REAL8:
280 dotp = (mmul_fn)dotp_real8;
281 break;
282 case __CPLX8:
283 dotp = (mmul_fn)dotp_cplx8;
284 break;
285 case __CPLX16:
286 dotp = (mmul_fn)dotp_cplx16;
287 break;
288 case __INT1:
289 dotp = (mmul_fn)dotp_int1;
290 break;
291 case __INT2:
292 dotp = (mmul_fn)dotp_int2;
293 break;
294 case __INT4:
295 dotp = (mmul_fn)dotp_int4;
296 break;
297 case __LOG1:
298 dotp = (mmul_fn)dotp_log1;
299 break;
300 case __LOG2:
301 dotp = (mmul_fn)dotp_log2;
302 break;
303 case __LOG4:
304 dotp = (mmul_fn)dotp_log4;
305 break;
306 case __INT8:
307 dotp = (mmul_fn)dotp_int8;
308 break;
309 case __LOG8:
310 dotp = (mmul_fn)dotp_log8;
311 break;
312 default:
313 __fort_abort("DOT_PRODUCT: unimplemented for data type");
314 }
315
316 __fort_scalar_copy[kind](cb, GET_DIST_ZED, len);
317
318#if defined(DEBUG)
319 if ((F90_FLAGS_G(as) ^ F90_FLAGS_G(bs)) & __OFF_TEMPLATE)
320 __fort_abort("DOT_PRODUCT: misaligned arguments");
321#endif
322
323 if (~(F90_FLAGS_G(as) | F90_FLAGS_G(bs)) & __OFF_TEMPLATE) {
324 I8(__fort_cycle_bounds)(as);
325 I8(__fort_cycle_bounds)(bs);
326
327 SET_DIM_PTRS(ajd, as, 0);
328 SET_DIM_PTRS(bjd, bs, 0);
329
330 cn = DIST_DPTR_CN_G(ajd);
331 acl = DIST_DPTR_CL_G(ajd);
332 bcl = DIST_DPTR_CL_G(ajd);
333 alof = DIST_DPTR_CLOF_G(ajd);
334 blof = DIST_DPTR_CLOF_G(bjd);
335 astr = F90_DPTR_SSTRIDE_G(ajd) * F90_DPTR_LSTRIDE_G(ajd);
336 bstr = F90_DPTR_SSTRIDE_G(bjd) * F90_DPTR_LSTRIDE_G(bjd);
337
338 /* compute dot product of local elements */
339
340 for (; cn > 0; --cn) {
341 an = I8(__fort_block_bounds)(as, 1, acl, &al, &au);
342 bn = I8(__fort_block_bounds)(bs, 1, bcl, &bl, &bu);
343
344 aoff = F90_LBASE_G(as) - 1 +
345 F90_DPTR_LSTRIDE_G(ajd) * (F90_DPTR_SSTRIDE_G(ajd) * al +
346 F90_DPTR_SOFFSET_G(ajd) - alof);
347
348 boff = F90_LBASE_G(bs) - 1 +
349 F90_DPTR_LSTRIDE_G(bjd) * (F90_DPTR_SSTRIDE_G(bjd) * bl +
350 F90_DPTR_SOFFSET_G(bjd) - blof);
351
352 dotp(cb, bn, ab, aoff, astr, bb, boff, bstr);
353
354 acl += DIST_DPTR_CS_G(ajd);
355 bcl += DIST_DPTR_CS_G(bjd);
356 alof += DIST_DPTR_CLOS_G(ajd);
357 blof += DIST_DPTR_CLOS_G(bjd);
358 }
359 }
360
361 /* do global reduction to complete dot product */
362
363 I8(__fort_reduce_section)(cb, kind, len, NULL, kind, len, 1,
364 I8(__fort_g_sum)[kind], 1, as);
365
366 I8(__fort_replicate_result)(cb, kind, len, NULL, kind, len, 1, as);
367
368 if (bs == bs1)
369 I8(__fort_copy_out)(bb0, bb, bs0, bs, __INTENT_IN);
370 if (as == as1)
371 I8(__fort_copy_out)(ab0, ab, as0, as, __INTENT_IN);
372
373#if defined(DEBUG)
374 if (__fort_test & DEBUG_MMUL) {
375 printf("%d dotpr ", GET_DIST_LCPU);
376 __fort_show_scalar(cb, kind);
377 printf("\n");
378 }
379#endif
380}

434 } else {

443 flags = (__ASSUMED_SHAPE + __ASSUMED_OVERLAPS + __INTENT_IN +

573 F90_DPTR_SOFFSET_G(aid) - ilof);
