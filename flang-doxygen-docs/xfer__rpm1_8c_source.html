30{

36#if defined(DEBUG)

41#if defined(DEBUG)
42 if (p->cnt != q->cnt)
43 __fort_abort("ebcopys: inconsistent counts");

47 __DIST_ENTRY_COPY(p->len);
48 __fort_bcopysl(q->adr, p->adr, q->cnt, q->str, p->str, q->ilen);

51 q++;
52 }
