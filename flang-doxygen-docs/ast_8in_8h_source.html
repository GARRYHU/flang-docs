131/* Shape Descriptor (SHD) */
132
133typedef struct {
134 /* A shape descscriptor is composed of 'n+1' elements to represent
135 * a shape of rank 'n'. The first element of a SHD is:
136 * lwb -> ndim, rank of the shape descriptor
137 * upb -> next, next SHD with the same number of subscripts
138 * stride -> not used
139 * The ensuing 'n' elements describe the lower bound, upper bound, and
140 * stride for each dimension.
141 */
142 int lwb; /* ast of lower bound */
143 int upb; /* ast of upper bound */
144 int stride; /* ast of stride */
145} SHD;
146
147#define SHD_NDIM(i) astb.shd.stg_base[i].lwb
148#define SHD_NEXT(i) astb.shd.stg_base[i].upb
149#define SHD_FILL(i) astb.shd.stg_base[i].stride
150#define SHD_LWB(i,j) astb.shd.stg_base[i+j+1].lwb
151#define SHD_UPB(i,j) astb.shd.stg_base[i+j+1].upb
152#define SHD_STRIDE(i,j) astb.shd.stg_base[i+j+1].stride
