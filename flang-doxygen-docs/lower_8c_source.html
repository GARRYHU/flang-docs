99/** \brief Lower the STDs and ASTs to something that the back end can use.
100
101 Essentially, lower to ILMs, save them to a file. In the mean time, save
102 the symbol table to the same file, as well as anything else needed to
103 compile the program.
104 */
105void
106lower(int staticinit)
107{
108 int std, nextstd, a;
109 int did_debug_label;
110 FILE *save_lowerfile;
111 int save_internal, save_docount, save_outersub, save_outerentries;
112
113 /* clear the VISIT flags for all symbols */
114 lower_clear_visit_fields();
115
116 /* set 'noconflict' bit, pointer types */
117 lower_set_symbols();
118
119#if DEBUG
120 if (DBGBIT(47, 4)) {
121 symdmp(gbl.dbgfil, 0);
122 dmp_dtype();
123 }
124#endif
125
126 if (staticinit) {
127 /* static init routine for host subprogram */
128 save_docount = lowersym.docount;
129 save_lowerfile = lowersym.lowerfile;
130 save_outersub = lowersym.outersub;
131 save_outerentries = lowersym.outerentries;
132 save_internal = gbl.internal;
133 lowersym.lowerfile = gbl.outfil;
134 lowersym.docount = 0;
135 lowersym.outersub = 0;
136 lowersym.outerentries = 0;
137 gbl.internal = 0;
138 } else {
139 switch (gbl.internal) {
140 case 0:
141 /* an outer subprogram that does not contain any others */
142 /* no need to copy to/from a temp file, copy directly to output file */
143 lowersym.lowerfile = gbl.outfil;
144 lowersym.docount = 0;
145 break;
146 case 1:
147 /* an outer subprogram that contains others */
148 /* create a temporary file */
149 lowersym.lowerfile = tmpfile();
150 if (lowersym.lowerfile == NULL) {
151 error(0, 4, 0, "could not open temporary ILM symbol file", "");
152 }
153 init_contained();
154 lowersym.docount = 0;
155 break;
156 default:
157 /* a contained subprogram; leave lowersym.lowerfile at the temp file */
158 /* save all the entry point names in a list of contained names */
159 if (lowersym.lowerfile == NULL || lowersym.lowerfile == gbl.outfil ||
160 contained == NULL) {
161 /* must have been an error in the containing routine; give up */
162 return;
163 }
164 save_contained();
165 break;
166 }
167 }
168
169 errors = 0;
170
171 lower_line = FUNCLINEG(gbl.currsub);
172
173 lower_init_sym();
174
175 lower_fill_member_parent();
176
177 if (gbl.rutype != RU_BDATA) {
178 lower_add_pghpf_commons();
179 }
180 /* mark all common blocks */
181 lower_linearized();
182 lower_common_sizes();
183 lower_namelist_plists();
184
185 /* clear the A_OPT1 and A_OPT2 fields for use here */
186 for (a = 0; a < astb.stg_avail; ++a) {
187#if DEBUG
188 A_NDTYPEP(a, -1);
189#endif
190 A_ILMP(a, 0);
191 A_BASEP(a, 0);
192 }
193
194 /* put out header lines */
195 lower_ilm_header();
196
197 /* should this be converted to a subroutine? */
198 if (gbl.rutype == RU_FUNC) {
199 switch (DTY(DTYPEG(gbl.currsub))) {
200 case TY_DERIVED:
201 case TY_STRUCT:
202 if (!CFUNCG(gbl.currsub)) {
203 gbl.rutype = RU_SUBR;
204 }
205 break;
206 }
207 }
208
209 lower_start_subroutine(gbl.rutype);
210
211 lower_data_stmts();
212
213 /* generate the code for each AST tree */
214 did_debug_label = 0;
215 for (std = STD_NEXT(0); std; std = nextstd) {
216 nextstd = STD_NEXT(std);
217 if (did_debug_label == 0 &&
218 (staticinit || STD_LINENO(std) || nextstd == 0 || STD_ORIG(std))) {
219 if (XBIT(52, 4) || XBIT(58, 0x22)) {
220 fill_entry_bounds(gbl.currsub, STD_LINENO(std));
221 }
222 lower_debug_label();
223 did_debug_label = 1;
224 }
225 lower_stmt(std, STD_AST(std), STD_LINENO(std), STD_LABEL(std));
226 } /* for std */
227
228 /* mark the current subroutine */
229 lower_mark_entries();
230
231 lower_set_craypointer();
232
233 lower_check_generics();
234
235 /* need to process iface here to lower correct data types and iface symbols */
236 stb_fixup_llvmiface();
237
238 lower_sym_header();
239
240 lower_fileinfo();
241
242 lower_program(gbl.rutype);
243
244 if (!staticinit && gbl.internal >= 1) {
245 /* put out all outer symbols */
246 lower_outer_symbols();
247 }
248
249 /* dump out the data types */
250 lower_data_types();
251
252 /* dump out the used symbols */
253 lower_symbols();
254 lower_end_subroutine();
255
256 lower_ilm_finish();
257 lower_finish_sym();
258 lower_exp_finish();
259 lower_directives();
260 if (errors) {
261 interr("Errors in Lowering", errors, 4);
262 }
263#if DEBUG
264 if (DBGBIT(47, 4)) {
265 dump_std();
266 symdmp(gbl.dbgfil, 0);
267 dmp_dtype();
268 }
269#endif
270 for (a = 0; a < astb.stg_avail; ++a) {
271 A_NDTYPEP(a, 0);
272 }
273 lower_unset_symbols();
274 if (staticinit) {
275 /* static init routine for host subprogram */
276 lowersym.lowerfile = save_lowerfile;
277 lowersym.docount = save_docount;
278 gbl.internal = save_internal;
279 lowersym.outersub = save_outersub;
280 lowersym.outerentries = save_outerentries;
281 } else if (gbl.internal == 0) {
282 lowersym.outersub = 0;
283 lowersym.outerentries = 0;
284 } else if (gbl.internal == 1) {
285 lowersym.outersub = gbl.currsub;
286 lowersym.outerentries = gbl.entries;
287 lowersym.last_outer_sym = stb.stg_avail;
288 }
289} /* lower */

309static void
310lower_start_subroutine(int rutype)
311{
312 /* int lab1, lab2; */
313 int syminit, ilm;
314
315 lowersym.labelcount = -1;
316 lowersym.last_lineno = -1;
317
318 plower("olnn", "BOS", gbl.funcline, gbl.findex, 0);
319 plower("o", "NOP");
320 plower("o", "--------------------");
321
322 switch (rutype) {
323 case RU_BDATA:
324 break;
325 case RU_SUBR:
326 case RU_FUNC:
327 /* generate code to fill bounds of adjustable arrays */
328 lower_start_stmt(gbl.funcline, 0, FALSE, 0);
329 lower_pointer_init();
330 lower_end_stmt(0);
331 break;
332 case RU_PROG:
333 /* call 'init' */
334 lower_start_stmt(gbl.funcline, 0, FALSE, 0);
335 if (1 /* flg.defaulthpf || XBIT(49,0x1000)*/) {
336 syminit = lower_makefunc(mkRteRtnNm(RTE_init), DT_NONE, TRUE);
337 ilm = plower("oS", "ICON", lowersym.intzero);
338 plower("onsiC", "CALL", 1, syminit, ilm, syminit);
339 }
340 lower_pointer_init();
341 lower_end_stmt(0);
342 break;
343 }
344} /* lower_start_subroutine */

378static void
379save_contained_name(char *name)
380{
381 int len;
382
383 NEED(last_contained + 1, contained, int, size_contained,
384 size_contained + 100);
385 contained[last_contained] = last_contained_char;
386 ++last_contained;
387
388 len = strlen(name);
389 NEED(last_contained_char + len + 1, contained_char, char, size_contained_char,
390 size_contained_char + 1000);
391 strcpy(contained_char + last_contained_char, name);
392 last_contained_char += len + 1;
393} /* save_contained_name */

395static void
396markid(int astx, int *unused)
397{
398 if (A_TYPEG(astx) == A_ID) {
399 int sptr, dtype;
400 sptr = A_SPTRG(astx);
401 if (sptr >= lowersym.first_outer_sym && sptr < lowersym.last_outer_sym_orig)
402 outerflags[sptr - lowersym.first_outer_sym] |= 1;
403 switch (STYPEG(sptr)) {
404 case ST_ARRAY:
405 case ST_VAR:
406 case ST_UNION:
407 case ST_STRUCT:
408 if (MIDNUMG(sptr) && MIDNUMG(sptr) >= lowersym.first_outer_sym &&
409 MIDNUMG(sptr) < lowersym.last_outer_sym_orig)
410 outerflags[MIDNUMG(sptr) - lowersym.first_outer_sym] |= 1;
411 if (SDSCG(sptr) && SDSCG(sptr) >= lowersym.first_outer_sym &&
412 SDSCG(sptr) < lowersym.last_outer_sym_orig)
413 outerflags[SDSCG(sptr) - lowersym.first_outer_sym] |= 1;
414 break;
415 default:
416 break;
417 }
418 dtype = DTYPEG(sptr);
419 if (dtype && DTY(dtype) == TY_ARRAY) {
420 int numdim, i;
421 numdim = ADD_NUMDIM(dtype);
422 for (i = 0; i < numdim; ++i) {
423 if (ADD_LWBD(dtype, i) != 0)
424 ast_traverse_more(ADD_LWBD(dtype, i), unused);
425 if (ADD_UPBD(dtype, i) != 0)
426 ast_traverse_more(ADD_UPBD(dtype, i), unused);
427 if (ADD_LWAST(dtype, i) != 0)
428 ast_traverse_more(ADD_LWAST(dtype, i), unused);
429 if (ADD_UPAST(dtype, i) != 0)
430 ast_traverse_more(ADD_UPAST(dtype, i), unused);
431 if (ADD_MLPYR(dtype, i) != 0)
432 ast_traverse_more(ADD_MLPYR(dtype, i), unused);
433 if (ADD_EXTNTAST(dtype, i) != 0)
434 ast_traverse_more(ADD_EXTNTAST(dtype, i), unused);
435 }
436 if (ADD_ZBASE(dtype) != 0)
437 ast_traverse_more(ADD_ZBASE(dtype), unused);
438 if (ADD_NUMELM(dtype) != 0)
439 ast_traverse_more(ADD_NUMELM(dtype), unused);
440 }
441 }
442} /* markid */

444static void
445save_contained(void)
446{
447 int sptr, stdx;
448 for (sptr = stb.firstusym; sptr < stb.stg_avail; ++sptr) {
449 if (STYPEG(sptr) == ST_ENTRY) {
450 save_contained_name(SYMNAME(sptr));
451 }
452 }
453 ast_visit(1, 1);
454 for (stdx = STD_NEXT(0); stdx; stdx = STD_NEXT(stdx)) {
455 ast_traverse(STD_AST(stdx), NULL, markid, NULL);
456 }
457 ast_unvisit_norepl();
458} /* save_contained */

460void
461lower_end_contains(void)
462{
463#define LOWERBUFSIZ 10000
464 char buffer[LOWERBUFSIZ];
465 int symbolslist;
466 int outer;
467 LOGICAL stbok = TRUE;
468
469 Trace(("end of containing routine"));
470 if (lowersym.lowerfile == NULL) {
471 Trace(("no lowersym.lowerfile"));
472 if (contained)
473 FREE(contained);
474 if (contained_char)
475 FREE(contained_char);
476 if (outerflags)
477 FREE(outerflags);
478 return;
479 }
480 if (lowersym.lowerfile == gbl.outfil) {
481 /* should not happen */
482 Trace(("lowersym.lower is gbl.outfil"));
483 if (contained)
484 FREE(contained);
485 if (contained_char)
486 FREE(contained_char);
487 if (outerflags)
488 FREE(outerflags);
489 return;
490 }
491
492 rewind(lowersym.lowerfile);
493 symbolslist = 1; /* 1 => reading symbols */
494 outer = 1;
495 while (fgets(buffer, LOWERBUFSIZ, lowersym.lowerfile) != NULL) {
496
497 if (buffer[0] == 'e') {
498 switch (symbolslist) {
499 int c;
500 case 1: /* symbols */
501 /* put out names of all the contained subroutines. */
502 for (c = 0; c < last_contained; ++c) {
503 char *ch;
504 ch = contained_char + contained[c];
505 fprintf(gbl.outfil, "contained %lu:%s\n", strlen(ch), ch);
506 if (gbl.stbfil)
507 fprintf(gbl.stbfil, "contained %lu:%s\n", strlen(ch), ch);
508
509 }
510 if (outer) {
511 /* put out the names of outer block symbols referenced
512 * in the internal routines */
513 int sptr;
514 for (sptr = lowersym.first_outer_sym; sptr < lowersym.last_outer_sym;
515 ++sptr) {
516 switch (STYPEG(sptr)) {
517 case ST_ARRAY:
518 case ST_DESCRIPTOR:
519 case ST_VAR:
520 case ST_UNION:
521 case ST_STRUCT:
522 case ST_PLIST:
523 if (sptr >= lowersym.last_outer_sym_orig ||
524 (outerflags[sptr - lowersym.first_outer_sym] &
525 1) /*REFINTG(sptr)*/) {
526#if DEBUG
527 if (DBGBIT(47, 8)) {
528 if (GSCOPEG(sptr) || XBIT(7, 0x200000))
529 {
530 fprintf(gbl.outfil, "global:%s\n", getprint(sptr));
531 if (gbl.stbfil)
532 fprintf(gbl.stbfil, "global:%s\n", getprint(sptr));
533 }
534 } else
535#endif
536 if (GSCOPEG(sptr) || XBIT(7, 0x200000))
537 {
538 fprintf(gbl.outfil, "global:%d\n", sptr);
539 if (gbl.stbfil)
540 fprintf(gbl.stbfil, "global:%d\n", sptr);
541 }
542 }
543 break;
544 case ST_PROC:
545 if (SCG(sptr) == SC_DUMMY &&
546 (sptr >= lowersym.last_outer_sym_orig ||
547 (outerflags[sptr - lowersym.first_outer_sym] & 1))) {
548#if DEBUG
549 if (DBGBIT(47, 8)) {
550 fprintf(gbl.outfil, "global:%s\n", getprint(sptr));
551 if (gbl.stbfil)
552 fprintf(gbl.stbfil, "global:%s\n", getprint(sptr));
553 } else
554#endif
555 {
556 fprintf(gbl.outfil, "global:%d\n", sptr);
557 if (gbl.stbfil)
558 fprintf(gbl.stbfil, "global:%d\n", sptr);
559 }
560 }
561 break;
562 default:
563 break;
564 }
565 }
566 outer = 0;
567 }
568 break;
569 case 2: /* ilms */
570 break;
571 case 3: /* directives */
572 symbolslist = 0;
573 break;
574 }
575 /* step theu symbolslist, ilmlist, directives list */
576 symbolslist++;
577 }
578 fputs(buffer, gbl.outfil);
579 if (gbl.stbfil) {
580 /* Skip ast */
581 if (buffer[0] == 'A') { /* AST2ILM */
582 stbok = FALSE;
583 } else if (!stbok && buffer[0] == 'T') { /* TOILM */
584 stbok = TRUE;
585 }
586 if (stbok)
587 fputs(buffer, gbl.stbfil);
588
589 if (buffer[0] == 'e') { /* end */
590 stbok = FALSE;
591 }
592 }
593 }
594 fclose(lowersym.lowerfile);
595 if (gbl.currmod)
596 lowersym.lowerfile = gbl.outfil;
597 else
598 lowersym.lowerfile = NULL;
599 FREE(contained);
600 FREE(contained_char);
601 FREE(outerflags);
602} /* lower_end_contains */

648/*
649 * Routines and data for determining how to return bind(C) function retvals
650 * according to the ABI
651 * These values must be kept insync with the values in the BE file exp_rte.c
652 */

654#if defined(TARGET_WIN_X8664)
655#define MAX_PASS_STRUCT_SIZE 8
656#else

660#define PACK(i, j) (((j) << 8) | ((i)&0xFF))
661#define UNPACKLOW(i) ((i)&0xFF)
662#define UNPACKHIGH(i) ((i) >> 8)
663#define REGSIZE 8
664
665#define CLASS_INT(t) (t >= CLASS_INT1 && t <= CLASS_INT8)
666#define CLASS_SSESP(t) (t == CLASS_SSESP4 || t == CLASS_SSESP8)
667#define CLASS_SSE(t) \
668 (t == CLASS_SSESP4 || t == CLASS_SSESP8 || t == CLASS_SSEDP || \
669 t == CLASS_SSEQ)

673/* return the sum of a and b using merge rules (integer over float) */
674static int
675addclasses(int a, int b)
676{
677 int class;
678
679 if (a == b) {
680 if (CLASS_INT(a)) {
681 class = a + b;
682 if (class > CLASS_INT8)
683 class = CLASS_MEM;
684 } else if (a == CLASS_SSESP4)
685 class = CLASS_SSESP8;
686 else
687 class = CLASS_MEM;
688 } else if (a == CLASS_NONE)
689 class = b;
690 else if (b == CLASS_NONE)
691 class = a;
692 else if (a == CLASS_MEM || b == CLASS_MEM)
693 class = CLASS_MEM;
694 else if (CLASS_INT(a) && CLASS_INT(b)) {
695 class = a + b;
696 if (class > CLASS_INT8)
697 class = CLASS_MEM;
698 } else if (CLASS_INT(a) && b == CLASS_SSESP4) {
699 class = a + CLASS_INT4;
700 if (class > CLASS_INT8)
701 class = CLASS_MEM;
702 } else if (CLASS_INT(b) && a == CLASS_SSESP4) {
703 class = b + CLASS_INT4;
704 if (class > CLASS_INT8)
705 class = CLASS_MEM;
706 } else if (a == CLASS_SSESP4 && b == CLASS_SSESP4)
707 class = CLASS_SSESP8;
708 else
709 class = CLASS_MEM;
710
711 return class;
712}

714static void
715trav_struct(int dtype, int off)
716{
717 int i;
718 int d;
719 int addr, mem;
720 int elems, elemsize;
721 int regi;
722 int tmpclass[2];
723 ADSC *ad;
724
725#if DEBUG
726 assert(off < REGSIZE * 2, "trav_struct - bad offset", off, 3);
727#endif
728
729 regi = off / REGSIZE; /* are we looking at 1st or 2nd reg. */
730 tmpclass[0] = CLASS_NONE;
731 tmpclass[1] = CLASS_NONE;
732
733 switch (DTY(dtype)) {
734 case TY_CHAR:
735 case TY_BINT:
736 case TY_BLOG:
737 regclass[regi] = CLASS_INT1;
738 return;
739 case TY_SINT:
740 case TY_SLOG:
741 regclass[regi] = CLASS_INT2;
742 return;
743 case TY_INT:
744 case TY_LOG:
745 regclass[regi] = CLASS_INT4;
746 return;
747 case TY_INT8:
748 case TY_PTR:
749 case TY_LOG8:
750 regclass[regi] = CLASS_INT8;
751 return;
752 case TY_128: /*m128*/
753 regclass[regi] = CLASS_SSEQ;
754 return;
755 case TY_QUAD:
756#if DEBUG
757 if (sizeof(DT_QUAD) == 16)
758 interr("trav_struct: update handling of long doubles", dtype, 3);
759#endif
760 /* We're treating this like DBLE for now. */
761 FLANG_FALLTHROUGH;
762 case TY_DBLE:
763 regclass[regi] = CLASS_SSEDP;
764 return;
765 case TY_FLOAT:
766 regclass[regi] = CLASS_SSESP4;
767 return;
768 case TY_CMPLX:
769 regclass[regi] = CLASS_SSESP8;
770 return;
771 case TY_DCMPLX:
772 assert(regi == 0, "trav_struct - bad offset for DCMPLX", off, 3);
773 regclass[0] = CLASS_SSEDP;
774 regclass[1] = CLASS_SSEDP;
775 return;
776 case TY_STRUCT:
777 case TY_DERIVED:
778 /* regclass will be the sum of the members in its eightbyte */
779 for (mem = DTY(dtype + 1); mem != NOSYM; mem = SYMLKG(mem)) {
780 if (CLASSG(mem) && VTABLEG(mem) && (TBPLNKG(mem) || FINALG(mem)))
781 continue;
782 addr = ADDRESSG(mem) + off;
783 trav_struct(DTYPEG(mem), addr);
784 regi = addr / REGSIZE;
785 regclass[regi] = addclasses(regclass[regi], tmpclass[regi]);
786 tmpclass[0] = regclass[0];
787 tmpclass[1] = regclass[1];
788 }
789 return;
790
791 case TY_ARRAY:
792 /* handle case of array in struct/union */
793 ad = AD_DPTR(dtype);
794 d = AD_NUMELM(ad);
795 if (A_TYPEG(d) != A_CNST)
796 return;
797 elems = ad_val_of(A_SPTRG(d));
798 elemsize = size_of(DTY(dtype + 1));
799 if (elems > 0 && elems <= 16) {
800 for (i = 0; i < elems; i++) {
801 addr = elemsize * i + off;
802 trav_struct(DTY(dtype + 1), addr);
803 regi = addr / REGSIZE;
804 regclass[regi] = addclasses(regclass[regi], tmpclass[regi]);
805 tmpclass[0] = regclass[0];
806 tmpclass[1] = regclass[1];
807 }
808 } else {
809#if DEBUG
810 assert(FALSE, "trav_struct - unexpected elems", elems, 3);
811#endif
812 ;
813 }
814 return;
815
816 default:
817 interr("invalid type for trav_struct", dtype, 3);
818 return;
819 }
820}
