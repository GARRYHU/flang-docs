290#define IL_COMM(i) ((ilis[i].attr >> 4) & 0x1) /* Yields ILIA_COMM or 0 */
291#define IL_RES(i) \
292 ((ILIA_RESULT)((ilis[i].attr >> 5) & 0x1f)) /* Yields ILIA_TRM..ILIA_AR */
293#define IL_LNK(i) ((ilis[i].attr >> 5) & 0x1f) /* Yields ILIA_LNK or 0 */
294#define IL_DOM(i) ((ilis[i].attr >> 10) & 0x3) /* Yields ILIA_DOM or 0 */
295#define IL_CSEG(i) ((ilis[i].attr >> 10) & 0x3) /* Yields ILIA_CSE or 0 */
296#define IL_IATYPE(i) ((ilis[i].attr >> 10) & 0x3) /* ILIA_DOM, ILIA_CSE or 0*/
297#define IL_SSENME(i) ((ilis[i].attr >> 12) & 0x1) /* Yields 1 or 0 */
298#define IL_VECT(i) ((ilis[i].attr >> 13) & 0x1) /* Yields 1 or 0 */
299/* Can this operation have a memory fence? */
300#define IL_HAS_FENCE(i) (((ilis[i].attr >> 14) & 3) != 0)
301/* Is this operation an IL_ATOMICRMWx? */
302#define IL_IS_ATOMICRMW(i) (((ilis[i].attr >> 14) & 0x3) == 2)
303/* Is this operation an IL_CMPXCHGx? */
304#define IL_IS_CMPXCHG(i) (((ilis[i].attr >> 14) & 0x3) == 3)
305/* Does this operation perform an atomic update?
306 IL_OPND(2) is the address of the operand to be updated. */
307#define IL_IS_ATOMIC_UPDATE(i) (((ilis[i].attr >> 14) & 0x3) >= 2)
308/* The attribute of the atomic opcodes are non-zero
309 * in the 14th and 15th bit. */
310#define IL_IS_ATOMIC_OPC(i) (((ilis[i].attr >> 14) & 0x3) != 0)
