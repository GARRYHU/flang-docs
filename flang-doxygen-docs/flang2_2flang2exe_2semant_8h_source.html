245#define NEED_LOOP(df, typ) \
246 { \
247 df = ++sem.doif_depth; \
248 NEED(df + 1, sem.doif_base, DOIF, sem.doif_size, sem.doif_size + 8); \
249 DI_EXIT_LAB(df) = DI_CYCLE_LABEL(df) = 0; \
250 DI_NAME(df) = 0; \
251 DI_LINENO(df) = gbl.lineno; \
252 DI_ID(df) = typ; \
253 DI_NOSCOPE_AVL(df) = 0; \
254 DI_NOSCOPE_SIZE(df) = 0; \
255 DI_NOSCOPE_BASE(df) = NULL; \
256 DI_NEST(df) = DI_NEST(df - 1) | DI_B(typ); \
257 }
