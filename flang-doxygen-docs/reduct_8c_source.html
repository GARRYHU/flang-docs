14/* translate a linearized index within a processor subgrid to the
15 corresponding processor number (the linearized index within the
16 whole grid). If the index is invalid, then -1 is returned. */
17
18static __INT_T I8(map_to_processor)(int c, /* linearized subgrid index */
19 __INT_T p0, /* base processor number */
20 __INT_T pr, /* subgrid rank */
21 __INT_T *pe, /* grid extents */
22 __INT_T *ps) /* grid strides */
23{
24 __INT_T i, k;
25
26 for (i = 0; c > 0 && i < pr; ++i) {
27 if (pe[i] > 1) {
28 k = c;
29 c /= pe[i];
30 k -= c * pe[i];
31 p0 += k * ps[i];
32 }
33 }
34 if (c != 0)
35 p0 = -1;
36 return p0;
37}

41static void
42 I8(broadcast)(__INT_T from, /* source cpu's linearized subgrid index */
43 __INT_T me, /* this cpu's linearized subgrid index */
44 __INT_T np, /* number of processors in subgrid */
45 __INT_T p0, /* base processor number */
46 __INT_T pr, /* subgrid rank */
47 __INT_T *pe, /* grid extents */
48 __INT_T *ps, /* grid strides */
49 void *adr, /* address of first data item */
50 __INT_T cnt, /* number of items */
51 __INT_T str, /* item stride */
52 dtype typ, /* item type */
53 __INT_T len) /* item length */
54{
55 __INT_T bit, cpu, dst, src;
56
57 /* translate grid indexes so the source processor is 0 */
58
59 me -= from;
60 if (me < 0)
61 me += np;
62
63 if (me != 0) {
64 bit = me & -me; /* least significant bit of (me) */
65 src = (me ^ bit) + from;
66 if (src >= np)
67 src -= np;
68 cpu = I8(map_to_processor)(src, p0, pr, pe, ps);
69 __fort_rrecvl(cpu, adr, cnt, str, typ, len);
70 } else {
71 bit = 1;
72 while (bit < np)
73 bit <<= 1;
74 }
75 bit >>= 1;
76
77 while (bit > 0) {
78 dst = me ^ bit;
79 if (dst < np) {
80 dst += from;
81 if (dst >= np)
82 dst -= np;
83 cpu = I8(map_to_processor)(dst, p0, pr, pe, ps);
84 __fort_rsendl(cpu, adr, cnt, str, typ, len);
85 }
86 bit >>= 1;
87 }
88}

102void I8(__fort_global_reduce)(char *rb, char *hb, int dims, F90_Desc *rd,
103 F90_Desc *hd, const char *what,
104 global_reduc_fn fn[__NTYPES])
105{
106 DECL_HDR_PTRS(ht); /* align-target */
107 DECL_DIM_PTRS(hdd);
108 DECL_DIM_PTRS(htd);
109 char *tmp; /* temp buffer pointer */
110 double buf[8]; /* small buffer */
111 dtype kind;
112 __INT_T bufsiz, it, bit;
113 __INT_T cnt, cpu, i, idx, len, mask, me, np, offgrid, pc, pl, pr;
114 __INT_T pe[MAXDIMS], ps[MAXDIMS];
115
116#if defined(DEBUG)
117 if (hd == NULL || F90_TAG_G(hd) != __DESC)
118 global_reduce_abort(what, "invalid home descriptor");
119#endif
120
121 if (LOCAL_MODE)
122 return;
123
124 if (F90_TAG_G(rd) == __DESC) {
125 kind = F90_KIND_G(rd);
126 len = F90_LEN_G(rd);
127 cnt = F90_LSIZE_G(rd);
128 } else if (ISSCALAR(rd) || ISSEQUENCE(rd)) {
129 kind = TYPEKIND(rd);
130 len = GET_DIST_SIZE_OF(kind);
131 cnt = 1;
132 } else
133 global_reduce_abort(what, "invalid result descriptor");
134
135 mask = DIST_MAPPED_G(hd) & dims;
136
137 if (mask != 0) {
138
139 /* all processors allocate same size temporary buffer */
140
141 bufsiz = ALIGNR(cnt * len); /* round up buffer size */
142 if (bufsiz > sizeof(buf))
143 tmp = (char *)__fort_gmalloc(bufsiz);
144 else
145 tmp = (char *)buf;
146
147 /* set up home template and processor grid */
148
149 np = 1;
150 pl = GET_DIST_LCPU;
151 pr = me = 0;
152 offgrid = 0;
153 for (i = 1; i <= F90_RANK_G(hd); ++i) {
154 if ((mask >> (i - 1)) & 1) {
155 SET_DIM_PTRS(hdd, hd, i - 1);
156
157 /* only processors in the grid participate in
158 reduction */
159
160 offgrid |= (DIST_DPTR_PCOORD_G(hdd) < 0);
161 if (!offgrid) {
162 pl -= DIST_DPTR_PCOORD_G(hdd) * DIST_DPTR_PSTRIDE_G(hdd);
163 me += DIST_DPTR_PCOORD_G(hdd) * np;
164 pe[pr] = DIST_DPTR_PSHAPE_G(hdd);
165 ps[pr] = DIST_DPTR_PSTRIDE_G(hdd);
166 pr++;
167 }
168 np *= DIST_DPTR_PSHAPE_G(hdd);
169 }
170 }
171
172 /* probably need to exclude processors because of scalar
173 subscripts or single alignments ? */
174
175 if (!offgrid) {
176
177#if defined(DEBUG)
178 if (__fort_test & DEBUG_REDU) {
179 printf("%d REDUCE_GLOBAL np=%d me=%d pr=%d pl=%d pe=", GET_DIST_LCPU,
180 np, me, pr, pl);
181 I8(__fort_show_index)(pr, pe);
182 printf(" ps=");
183 I8(__fort_show_index)(pr, ps);
184 printf("\n");
185 }
186#endif
187
188 /* this pattern guarantees identical results on all
189 processors. collect and reduce partial results from
190 tree below */
191
192 bit = 1;
193 while (bit < np) {
194 it = me ^ bit;
195 bit <<= 1;
196 if (it < me)
197 break;
198 if (it < np) {
199 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
200 __fort_rrecvl(cpu, tmp, cnt, 1, kind, len);
201 fn[kind](cnt, rb, tmp, /* optional args */ NULL, NULL, 0);
202 }
203 }
204 bit >>= 1;
205
206 /* send partial results up the tree, receive final results */
207
208 it = me ^ bit;
209 if (it < me) {
210 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
211 __fort_rsendl(cpu, rb, cnt, 1, kind, len);
212 __fort_rrecvl(cpu, rb, cnt, 1, kind, len);
213 bit >>= 1;
214 }
215
216 /* send final result to tree below */
217
218 while (bit > 0) {
219 it = me ^ bit;
220 bit >>= 1;
221 if (it > me && it < np) {
222 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
223 __fort_rsendl(cpu, rb, cnt, 1, kind, len);
224 }
225 }
226 }
227
228 /* free temporary buffer */
229
230 if (tmp != (char *)buf)
231 __fort_gfree(tmp);
232 }
233
234 /* replicate the result across the mapped dimensions of the
235 align-target which had scalar subscripts or single
236 alignments in the home array. result replication has
237 already occurred in the unmapped dimensions, so a similar
238 broadcast pattern is done within each replication group. */
239
240 ht = (SUBGROUP_MODE) ? hd : DIST_ALIGN_TARGET_G(hd);
241 mask = DIST_SINGLE_G(hd) & DIST_MAPPED_G(ht);
242 if (mask != 0) {
243
244 np = 1;
245 cpu = pl = GET_DIST_LCPU;
246 idx = me = 0;
247 offgrid = 0;
248 for (pr = i = 0; i < F90_RANK_G(ht); ++i) {
249 if (mask >> i & 1) {
250 SET_DIM_PTRS(htd, ht, i);
251 offgrid |= (DIST_DPTR_PCOORD_G(htd) < 0);
252 if (!offgrid) {
253 pc = DIST_INFO_G(hd, i) - DIST_DPTR_TLB_G(htd);
254 if (DFMT(ht, i + 1) == DFMT_GEN_BLOCK) {
255
256 /*
257 * need to find the processor that owns pc
258 * based on gen_block array
259 */
260
261 __INT_T *gb;
262 __INT_T idxUB;
263 __INT_T pshape;
264 __INT_T j;
265 pshape = DIST_DPTR_PSHAPE_G(htd);
266 gb = DIST_DPTR_GEN_BLOCK_G(htd);
267 ++pc; /* in terms of fortran indices (1 ... upperbound) */
268
269#if defined(DEBUG)
270 if (__fort_test & DEBUG_REDU)
271 printf("owner of single alignment %d (gen_block dim)", pc);
272#endif
273 idxUB = *gb;
274 for (j = 0; j < pshape; ++j) {
275 if (idxUB >= pc)
276 break;
277 ++gb;
278 idxUB += *gb;
279 }
280
281 if (j < pshape)
282 pc = j;
283 else
284 __fort_abort("global_reduce: bad gen_block (internal error)");
285#if defined(DEBUG)
286 if (__fort_test & DEBUG_REDU)
287 printf(" is processor %d\n", pc);
288#endif
289
290 } else {
291 if (DIST_DPTR_BLOCK_G(htd) != 1)
292 RECIP_DIV(&pc, pc, DIST_DPTR_BLOCK_G(htd));
293 if (pc >= DIST_DPTR_PSHAPE_G(htd)) /* cyclic */
294 RECIP_MOD(&pc, pc, DIST_DPTR_PSHAPE_G(htd));
295 }
296
297 cpu -= (DIST_DPTR_PCOORD_G(htd) - pc) * DIST_DPTR_PSTRIDE_G(htd);
298 idx += pc * np;
299 pl -= DIST_DPTR_PCOORD_G(htd) * DIST_DPTR_PSTRIDE_G(htd);
300 me += DIST_DPTR_PCOORD_G(htd) * np;
301 pe[pr] = DIST_DPTR_PSHAPE_G(htd);
302 ps[pr] = DIST_DPTR_PSTRIDE_G(htd);
303 ++pr;
304 }
305 np *= DIST_DPTR_PSHAPE_G(htd);
306 }
307 }
308 if (!offgrid) {
309#if defined(DEBUG)
310 if (__fort_test & DEBUG_REDU) {
311 printf("%d repl cpu=%d idx=%d me=%d np=%d pr=%d pl=%d pe=",
312 GET_DIST_LCPU, cpu, idx, me, np, pr, pl);
313 I8(__fort_show_index)(pr, pe);
314 printf(" ps=");
315 I8(__fort_show_index)(pr, ps);
316 printf("\n");
317 }
318#endif
319 I8(broadcast)(idx, me, np, pl, pr, pe, ps, rb, cnt, 1, kind, len);
320 }
321 }
322
323/* also replicate results to all processors outside the processor
324 grid. */
325
326 pl = 0;
327 np = GET_DIST_TCPUS - pl;
328 me = GET_DIST_LCPU - pl;
329
330 if (np > 1 && me >= 0)
331 I8(broadcast)(0, me, np, pl, 1, &np, &_1, rb, cnt, 1, kind, len);
332}

334/* global, parallel reduction across the mapped dimensions of the
335 source section. if dim <= 0, reduce in all dimensions, otherwise
336 reduce in dimension dim only. arguments vec1 and optional vec2 are
337 local intermediate results that are combined in-place with results
338 from all other processors to produce the final result. */
339
340void I8(__fort_reduce_section)(void *vec1, dtype typ1, int len1, void *vec2,
341 dtype typ2, int len2, int cnt,
342 global_reduc_fn fn_g, int dim, F90_Desc *a)
343{
344 DECL_DIM_PTRS(ad);
345 char *tmp1, *tmp2; /* temporary buffer pointers */
346 double buf1[4], buf2[4];
347 __INT_T bit, cpu, i, it, mask, n, offgrid, pr, np, pc, pl, pe[MAXDIMS],
348 ps[MAXDIMS];
349
350 mask = DIST_MAPPED_G(a);
351 if (dim > 0) {
352 mask &= (1 << (dim - 1)); /* reduce only in specified dim */
353 }
354
355 if (LOCAL_MODE || mask == 0)
356 return;
357
358 np = 1;
359 pl = GET_DIST_LCPU;
360 pr = pc = 0;
361 offgrid = 0;
362 for (i = 0; i < F90_RANK_G(a); ++i) {
363 if (mask >> i & 1) {
364 SET_DIM_PTRS(ad, a, i);
365 offgrid |= (DIST_DPTR_PCOORD_G(ad) < 0);
366 if (!offgrid) {
367 pl -= DIST_DPTR_PCOORD_G(ad) * DIST_DPTR_PSTRIDE_G(ad);
368 pc += DIST_DPTR_PCOORD_G(ad) * np;
369 pe[pr] = DIST_DPTR_PSHAPE_G(ad);
370 ps[pr] = DIST_DPTR_PSTRIDE_G(ad);
371 pr++;
372 }
373 np *= DIST_DPTR_PSHAPE_G(ad);
374 }
375 }
376 if (np == 1)
377 return;
378
379 /* allocate temporary buffers */
380
381 n = cnt * len1;
382 if (n > sizeof(buf1))
383 tmp1 = (char *)__fort_gmalloc(n);
384 else
385 tmp1 = (char *)buf1;
386
387 if (vec2 != NULL) {
388 n = cnt * len2;
389 if (n > sizeof(buf2))
390 tmp2 = (char *)__fort_gmalloc(n);
391 else
392 tmp2 = (char *)buf2;
393 } else
394 tmp2 = NULL;
395
396 /* only processors in the grid participate */
397
398 if (!offgrid) {
399
400#if defined(DEBUG)
401 if (__fort_test & DEBUG_REDU) {
402 printf("%d reduce np=%d pc=%d pr=%d pl=%d pe=", GET_DIST_LCPU, np, pc,
403 pr, pl);
404 I8(__fort_show_index)(pr, pe);
405 printf(" ps=");
406 I8(__fort_show_index)(pr, ps);
407 printf("\n");
408 }
409#endif
410
411 /* this pattern guarantees identical results on all processors.
412 collect and reduce partial results from tree below */
413
414 bit = 1;
415 while (bit < np) {
416 it = pc ^ bit;
417 bit <<= 1;
418 if (it < pc)
419 break;
420 if (it < np) {
421 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
422 __fort_rrecvl(cpu, tmp1, cnt, 1, typ1, len1);
423 if (tmp2 != NULL)
424 __fort_rrecvl(cpu, tmp2, cnt, 1, typ2, len2);
425 fn_g(cnt, vec1, tmp1, vec2, tmp2, /* optional arg */ 0);
426 }
427 }
428 bit >>= 1;
429
430 /* send partial result up the tree and receive final result */
431
432 it = pc ^ bit;
433 if (it < pc) {
434 bit >>= 1;
435 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
436 __fort_rsendl(cpu, vec1, cnt, 1, typ1, len1);
437 if (vec2 != NULL)
438 __fort_rsendl(cpu, vec2, cnt, 1, typ2, len2);
439 __fort_rrecvl(cpu, vec1, cnt, 1, typ1, len1);
440 if (vec2 != NULL)
441 __fort_rrecvl(cpu, vec2, cnt, 1, typ2, len2);
442 }
443
444 /* send final result to tree below */
445
446 while (bit > 0) {
447 it = pc ^ bit;
448 bit >>= 1;
449 if (it > pc && it < np) {
450 cpu = I8(map_to_processor)(it, pl, pr, pe, ps);
451 __fort_rsendl(cpu, vec1, cnt, 1, typ1, len1);
452 if (vec2 != NULL)
453 __fort_rsendl(cpu, vec2, cnt, 1, typ2, len2);
454 }
455 }
456 }
457
458 /* free temporary buffers */
459
460 if (tmp1 != (char *)buf1)
461 __fort_gfree(tmp1);
462 if (tmp2 != (char *)buf2)
463 __fort_gfree(tmp2);
464}

470void I8(__fort_replicate_result)(void *vec1, dtype typ1, int len1, void *vec2,
471 dtype typ2, int len2, int cnt, F90_Desc *a)
472{
473 DECL_HDR_PTRS(t);
474 DECL_DIM_PTRS(td);
475 __INT_T fromcpu, fromidx, i, mask, me, np, offgrid, p0, pc, pr, pe[MAXDIMS],
476 ps[MAXDIMS];
477
478 if (LOCAL_MODE)
479 return;
480
481 /* replicate the result across the partitioned dimensions of the
482 align-target which had scalar subscripts or single alignments
483 in the alignee. result replication has already occurred in the
484 non-partitioned dimensions, so a similar broadcast pattern is
485 done within each replication group. */
486
487 t = DIST_ALIGN_TARGET_G(a);
488 mask = DIST_SINGLE_G(a) & DIST_MAPPED_G(t);
489 if (mask != 0) {
490
491 np = 1;
492 fromcpu = p0 = GET_DIST_LCPU;
493 fromidx = me = 0;
494 offgrid = 0;
495 for (pr = i = 0; i < F90_RANK_G(t); ++i) {
496 if (mask >> i & 1) {
497 SET_DIM_PTRS(td, t, i);
498 offgrid |= (DIST_DPTR_PCOORD_G(td) < 0);
499 if (!offgrid) {
500 pc = DIST_INFO_G(a, i) - DIST_DPTR_TLB_G(td);
501 if (DIST_DPTR_BLOCK_G(td) != 1)
502 RECIP_DIV(&pc, pc, DIST_DPTR_BLOCK_G(td));
503 if (pc >= DIST_DPTR_PSHAPE_G(td)) /* cyclic */
504 RECIP_MOD(&pc, pc, DIST_DPTR_PSHAPE_G(td));
505 fromcpu -= (DIST_DPTR_PCOORD_G(td) - pc) * DIST_DPTR_PSTRIDE_G(td);
506 fromidx += pc * np;
507 p0 -= DIST_DPTR_PCOORD_G(td) * DIST_DPTR_PSTRIDE_G(td);
508 me += DIST_DPTR_PCOORD_G(td) * np;
509 pe[pr] = DIST_DPTR_PSHAPE_G(td);
510 ps[pr] = DIST_DPTR_PSTRIDE_G(td);
511 ++pr;
512 }
513 np *= DIST_DPTR_PSHAPE_G(td);
514 }
515 }
516 if (!offgrid) {
517#if defined(DEBUG)
518 if (__fort_test & DEBUG_REDU) {
519 printf("%d repl fromcpu=%d fromidx=%d me=%d np=%d"
520 " pr=%d p0=%d pe=",
521 GET_DIST_LCPU, fromcpu, fromidx, me, np, pr, p0);
522 I8(__fort_show_index)(pr, pe);
523 printf(" ps=");
524 I8(__fort_show_index)(pr, ps);
525 printf("\n");
526 }
527#endif
528 I8(broadcast)(fromidx, me, np, p0, pr, pe, ps, vec1, cnt, 1, typ1, len1);
529
530 if (vec2 != NULL)
531 I8(broadcast)(fromidx, me, np, p0, pr, pe, ps, vec2, cnt, 1,
532 typ2, len2);
533 }
534 }
