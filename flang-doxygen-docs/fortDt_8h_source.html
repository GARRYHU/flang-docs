18/** \typedef _DIST_TYPE
19 *
20 * \brief Data type codes.
21 *
22 * These must correspond to the compiler's codes.
23 *
24 * Any changes to the intrinsic types must get reflected in the
25 * F2003 type descriptors defined in type.c (see I8(__f03_ty_to_id)[] array).
26 * (see also \ref __NTYPES below).
27 */

32 __CINT = 3, /**< C signed int */

34 __LONG = 5, /**< C signed long int */
35 __ULONG = 6, /**< C unsigned long int */
36 __FLOAT = 7, /**< C float */
37 __DOUBLE = 8, /**< C double */
38 __CPLX8 = 9, /**< Fortran complex*8 (2x real*4) */
39 __CPLX16 = 10, /**< Fortran complex*16 (2x real*8) */
40 __CHAR = 11, /**< C signed char */
41 __UCHAR = 12, /**< C unsigned char */
42 __LONGDOUBLE = 13, /**< C long double */
43 __STR = 14, /**< Fortran character */
44 __LONGLONG = 15, /**< C long long */
45 __ULONGLONG = 16, /**< C unsigned long long */
46 __LOG1 = 17, /**< Fortran logical*1 */
47 __LOG2 = 18, /**< Fortran logical*2 */
48 __LOG4 = 19, /**< Fortran logical*4 */
49 __LOG8 = 20, /**< Fortran logical*8 */
