26struct pent {
27 const char *func; /* function name (no \0) */
28 __CLEN_T funcl; /* length of above */
29 const char *file; /* file name (no \0) */
30 __CLEN_T filel; /* length of above */

58/** \brief Initialization */
59void
60__fort_entry_init(void)
61{
62 pentb = (struct pent *)__fort_malloc(PENTS * sizeof(struct pent));
63 pentc = pentb;
64 pente = pentb + PENTS;
65 pentc->func = UNKNOWN;
66 pentc->funcl = sizeof(UNKNOWN);
67 pentc->file = UNKNOWN;
68 pentc->filel = sizeof(UNKNOWN);
69 __fort_stat_mflag = __fort_stat_init();
70 __fort_prof_mflag = __fort_prof_init();
71 __fort_trac_mflag = __fort_trac_init();
72 __fort_entry_mflag = __fort_stat_mflag | __fort_prof_mflag | __fort_trac_mflag;
73}

78void ENTFTN(FUNCTION_ENTRYA, function_entrya)(__INT_T *line, __INT_T *lines,
79 DCHAR(func),
80 DCHAR(file) DCLEN64(func) DCLEN64(file))
81{
82 int cline; /* calling line number */
83 int n;
84
85 cline = *LINENO;
86 pentc->cline = *LINENO; /* save current line number on stack */
87 pentc++; /* next frame */
88 if (pentc == pente) {
89 n = pente - pentb;
90 pentb =
91 (struct pent *)__fort_realloc(pentb, (n + PENTS) * sizeof(struct pent));
92 pentc = pentb + n;
93 pente = pentb + n + PENTS;
94 }
95 pentc->func = CADR(func);
96 pentc->funcl = CLEN(func);
97 pentc->file = CADR(file);
98 pentc->filel = CLEN(file);
99 *LINENO = *line;
100 pentc->line = *line;
101 pentc->lines = *lines;
102 __fort_stat_function_entry(*line, *lines, cline, CADR(func), CADR(file),
103 CLEN(func), CLEN(file));
104 __fort_prof_function_entry(*line, *lines, cline, CADR(func), CADR(file),
105 CLEN(func), CLEN(file));
106 __fort_trac_function_entry(*line, *lines, cline, CADR(func), CADR(file),
107 CLEN(func), CLEN(file));
108}

141/* print traceback */
142
143void
144__fort_traceback()
145{
146 struct pent *pe;
147 char buf[512];
148 char *p;
149 int lcpu;
150
151 if (pentb == (struct pent *)0) {
152 return;
153 }
154 if (pentc == pentb) {
155 return;
156 }
157 pentc->cline = *LINENO; /* set current line */
158 pe = pentc;
159 lcpu = GET_DIST_LCPU;
160 sprintf(buf, "%d: Traceback:\n", lcpu);
161 write(2, buf, strlen(buf));
162 while (pe > pentb) {
163 p = buf;
164 sprintf(p, "%d: ", lcpu);
165 p += strlen(p);
166 strncpy(p, pe->func, pe->funcl);
167 p += pe->funcl;
168 sprintf(p, " at line %d in file \"", pe->cline);
169 p += strlen(p);
170 strncpy(p, pe->file, pe->filel);
171 p += pe->filel;
172 strcpy(p, "\"\n");
173 write(2, buf, strlen(buf));
174 pe--;
175 }
176}

180/* print message with one level call traceback */
181
182void
183__fort_tracecall(const char *msg)
184{
185 struct pent *pe;
186 char buf[512];
187 char *p;
188
189 p = buf;
190 sprintf(p, "%d: %s", GET_DIST_LCPU, msg);
191 p += strlen(p);
192
193 pe = pentc;
194 if (pentb && pe > pentb) {
195 strcpy(p, " in ");
196 p += 4;
197 strncpy(p, pe->func, pe->funcl);
198 p += pe->funcl;
199 strcpy(p, " at \"");
200 p += 5;
201 strncpy(p, pe->file, pe->filel);
202 p += pe->filel;
203 sprintf(p, "\":%d", *LINENO);
204 p += strlen(p);
205 if (--pe > pentb) {
206 strcpy(p, " called from ");
207 p += 13;
208 strncpy(p, pe->func, pe->funcl);
209 p += pe->funcl;
210 strcpy(p, " at \"");
211 p += 5;
212 strncpy(p, pe->file, pe->filel);
213 p += pe->filel;
214 sprintf(p, "\":%d", pe->cline);
215 p += strlen(p);
216 }
217 }
218 *p++ = '\n';
219 *p++ = '\0';
220 write(2, buf, strlen(buf));
221}

223void ENTFTN(TRACECALLA, tracecalla)(DCHAR(msg) DCLEN64(msg))
224{
225 char buf[257];
226 __CLEN_T i;
227 __CLEN_T len = CLEN(msg);
228 char *p = CADR(msg);
229 if (len > 256)
230 len = 256;
231 for (i = 0; i < len; ++i)
232 buf[i] = p[i];
233 buf[len] = '\0';
234 __fort_tracecall(buf);
