114void ENTFTN(FINDLOCS, findlocs)(__INT_T *rb, char *ab, char *val, char *mb,
115 __INT_T *back, F90_Desc *rs, F90_Desc *as,
116 F90_Desc *vs, F90_Desc *ms, F90_Desc *bs)
117{
118 red_parm z;
119 double vb[4];
120 char *strvb;
121
122 INIT_RED_PARM(z);
123 __fort_red_what = "FINDLOC";
124
125 z.kind = F90_KIND_G(as);
126 z.len = F90_LEN_G(as);
127 z.mask_present = (F90_TAG_G(ms) == __DESC && F90_RANK_G(ms) > 0);
128 if (!z.mask_present) {
129 z.lk_shift = GET_DIST_SHIFTS(__LOG);
130 } else {
131 z.lk_shift = GET_DIST_SHIFTS(F90_KIND_G(ms));
132 }
133 z.l_fn_b = l_findloc_b[z.lk_shift][z.kind];
134 z.g_fn = g_findloc[z.kind];
135 z.zb = val;
136 z.back = *(__LOG_T *)back;
137
138 if (z.kind == __STR) {
139 strvb = (char *)__fort_gmalloc(z.len);
140 memcpy(strvb, ((char *)z.zb), z.len);
141 I8(__fort_red_scalarlk)(&z, strvb, ab, mb, rs, as, ms, rb, __FINDLOC);
142 __fort_gfree(strvb);
143 } else {
144 I8(__fort_red_scalarlk)(&z, (char *)vb, ab, mb, rs, as, ms, rb, __FINDLOC);
145 }
146}
