55static INLINE
56float _Complex set_cmplx(float x, float y)
57{
58 float _Complex result;
59#if defined __INTEL_COMPILER
60 result = (x + I*y); PRINT(result); // this causes GCC to emit multiply by zero, because I is read as complex and not just imaginary
61#else
62 *(0 + (float *)(&result)) = x;
63 *(1 + (float *)(&result)) = y;
64#endif
65 PRINT(result);
66 return result;

95static INLINE
96float my_fmaf(float a, float b, float c)
97{
98#if (defined __INTEL_COMPILER)
99 // ICC somehow doesn't recognize fmaf as builtin under fp-model source
100 // this hack improves performance as FMA function call is slow, but it
101 // also breaks vectorization because of the use of mm types.
102 #define F2MM(x) _mm_set_ss(x)
103 return _mm_cvtss_f32(_mm_fmadd_ss(F2MM(a), F2MM(b), F2MM(c)));
104#elif (defined __PGI)
105 // PGI also doesn't recognize the function as builtin, yet it doesn't
106 // support Intel intrinsics, so below is just to allow inlining of the
107 // function.
108 // FIXME: it breaks FP program
109 return a*b + c;
110#else
111 // same thing happened with older GCC/clang
112 return __builtin_fmaf(a, b, c);
113#endif
114}
