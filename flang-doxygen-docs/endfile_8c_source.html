17static int
18_f90io_endfile(__INT_T *unit, __INT_T *bitv, __INT_T *iostat)
19{
20 FIO_FCB *f;
21
22 __fortio_errinit03(*unit, *bitv, iostat, "ENDFILE");
23 if (ILLEGAL_UNIT(*unit))
24 return __fortio_error(FIO_EUNIT);
25
26 /* call rwinit to get FCB pointer, do error checking, and truncate
27 file if necessary: */
28
29 f = __fortio_rwinit(*unit, FIO_UNFORMATTED, 0L, 2 /*endfile*/);
30 if (f == NULL)
31 return ERR_FLAG;
32
33 f->eof_flag = TRUE;
34 return 0; /* no error occurred */
35}
