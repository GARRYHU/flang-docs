35/* Payne-Hanek style argument reduction. */
36static double
37__reduction_slowpath_pio2(double const a, int64_t *h, double *rlo)
38{
39 // Compute bits of n = a * 1/pi using integer arithmetic.
40 // No need to compute integer bits representing quotient 2
41 // and above as trig functions are 2*pi periodic anyway.
42 // Ultimately we'll be interested in 3 leading bits of n to
43 // determine the octant into which a falls while the lower
44 // bits will be used to compute a - n*pi that don't
45 // cancel out in the subtraction. We'd like to obtain
46 // the reduced argument to 10 bits more than available in
47 // double precision.
48 // We know thanks to Kahan that at most 60-61 bits may get
49 // cancelled, so we'd need at most 61 + 53 + 10 = 124 bits
50 // of the product, yet our choice of table storage leads
51 // to a maximum of another 63 bits shift, so we will
52 // end up computing 3*64 = 192 bits using 4*64 = 256 bits
53 // of 1/pi.
54
55 uint64_t result[4];
56 // get sign bit
57 uint64_t s = D2L(a) & DB_SIGN_BIT;
58 // get unbiased exponent
59 uint64_t e = ((D2L(a) >> 52) & 0x7ff) - DB_EXP_BIAS;
60 // normalized unsigned integer representation of the
61 // significand of a with implicit leading bit restored
62 uint64_t ia = ((D2L(a) << 11) | 0x8000000000000000ULL);
63 // index into the table of 1/pi bits based on exponent e
64 // lookup table is organized as 19 words of 64-bit width
65 int32_t idx = 15 - ((e >> 6) & 0xF);
66 int32_t q;
67
68 // multiply significand of a by bits of 1/pi
69 __uint128_t acc = 0;
70 for (q = 0; q < 4; q++) {
71 // no integer wraparound here: given 3 p-bit integers
72 // x*y+z doesn't exceed (2^p-1)*(2^p-1)+2^p-1 =
73 // (2^p-1)*2^p, which is strictly less than 2^2p
74 acc += (__uint128_t)ia * i1opi_f[idx + q];
75 result[q] = (uint64_t)acc;
76 acc >>= 64;
77 }
78
79 uint64_t p = result[3];
80 // we avoided shifting or bit-addressing the lookup table,
81 // by working with coarse 64-bit words, now need to fine-tune
82 // the result
83 e = e & 63;
84 if (e) {
85 p = (p << e) | (result[2] >> (64 - e));
86 result[2] = (result[2] << e) | (result[1] >> (64 - e));
87 }
88
89 // p * pi may approach 1.1111...b * pi < 2*pi,
90 // so its leading bit has cost of 2^(2) in the worst case.
91 // Prepare double precision sign and exponent of the result
92 uint64_t shi = s | (((uint64_t)(2 + DB_EXP_BIAS)) << 52);
93
94 // 3 selector bits: pi, pi/2, pi/4
95 *h = p & 0xE000000000000000ULL;
96 // eliminate r > pi/4 by subtracting r from pi/2
97 if (p & 0x2000000000000000ULL)
98 {
99 p = ~p;
100 result[2] = ~result[2];
101 }
102 // discard selector bits
103 p &= 0x1fffffffffffffffULL;
104
105 // normalize p, adjust exponent bits accordingly
106 int lz = __builtin_clzll(p);
107 p = p << lz | result[2] >> (64 - lz);
108 shi -= (uint64_t)lz << 52;
109
110 // normalized p times normalized pi
111 __uint128_t prod = p * (__uint128_t)0xc90fdaa22168c235ULL;
112 // take 53 leading bits of the product and
113 // convert to FP.
114 uint64_t lhi = prod >> (64 + 11);
115 // Leading bit in lhi may still be zero even though
116 // we've normalized both p and pi, so we won't OR the
117 // significant bits into FP mantissa, but will use
118 // type conversion + multiply which will also take care of
119 // this leading bit normalization.
120 // convert 53-bit integer into high part
121 double r = L2D(shi - (52ull << 52)) * (double)lhi;
122
123 uint64_t llo = prod >> 11;
124 // low part is at least 53 orders of magnitude
125 // smaller than high, plus adjust scaling for
126 // 64-bit integer conversion to FP.
127 uint64_t slo = shi - (53ull << 52) - (63ull << 52);
128 *rlo = L2D(slo) * (double)llo;
129
130 return r;
131}

133static double _Complex __attribute__ ((noinline))
134__cis_d_scalar(double x)
135{
136 // This function computes pair of sin/cos results per input and
137 // relies on octant reduction: x -> r such that |r| < pi/4.
138 // We then use minimax approximations for sin and cos.
139 // It is possible to use quadrant reduction to be able to compute
140 // only sin() and use it for both sin and cos reconstruction, but
141 // we do not go that way because the reduced argument is larger
142 // and requires more multiprecision in the polynomial computation
143 // to not exceed our accuracy target.
144
145 double p, w;
146 int64_t h, selector;
147 uint64_t ssign, csign;
148
149 // p = |x|
150 p = L2D(D2L(x) & DB_ABS_MASK); PRINT(p);
151
152#undef THRESHOLD
153#define THRESHOLD L2D(0x42dce2c35b04bada)
154#define DELTA 0.005
155 if (__builtin_expect(p > THRESHOLD, 0))
156 {
157 // huge arguments reduction
158 p = (D2L(p) >= DB_PINF) ? p * 0.0 : __reduction_slowpath_pio2(p, &h, &w); PRINT(p); PRINT(h); PRINT(w);
159 // h contains: 3 selector bits for [pi, pi/2, extra swap]
160 ssign = h & DB_SIGN_BIT; PRINT(ssign);
161 ssign = ssign ^ (D2L(x) & DB_SIGN_BIT); PRINT(ssign);
162 csign = ((h + (1ull<<62)) & DB_SIGN_BIT); PRINT(csign);
163 selector = ((h << 1) + (h << 2)) >> 63; PRINT(selector);
164 } else {
165 // Here we follow the reduction process presented in
166 // [1] S. Boldo, M. Daumas & R-C. Li,
167 // Formally Verified Argument Reduction with a Fused-Multiply-Add,
168 // IEEE Transactions on Computers (IF: 1.680), 58(8), pp. 1139-1145, 2009.
169 // i.e. we compute (x - N*pi/2) using 3-double representation of
170 // pi/2 and obtain result as an unevaluated sum of two doubles: p + w.
171
172 // NOTE: this process becomes inaccurate because of the first step
173 // where we compute x / (pi/2) via multiplication of x by R,
174 // rounded 2/pi. So when x grows that rounding error in the 2/pi
175 // may propagate into integer part of the quotient and the resulting
176 // reduced argument will fall out of desired [-pi/4, pi/4] range.
177 // Here's what happens:
178 // R = 2/pi + eps
179 // where eps is an absolute error in the rounded constant.
180 // We may always mathematically represent x as
181 // x = N*pi/2 + m*pi/2
182 // with N being nearest integer and m < 1/2.
183 // Now the multiplication x*R results in:
184 // x*R = x*2/pi + x*eps = N + m + x*eps
185 // In order for this to round correctly to nearest integer N
186 // x*eps shall be small enough, so that it doesn't bump m over 1/2.
187 // Suppose m = 1/2 - delta, delta is positive since m < 1/2.
188 // x*eps must be less than delta in order for the rounding to be
189 // correct and result in N.
190 // eps is known to us by definition of R. Acceptable range for the
191 // input |x| < T is given to us by the paper [1]. So the error will
192 // appear when T*eps > delta.
193 // For T = pi/2 * (2^(p-2) - 1) and eps ~= 2/pi*2^(-p-1) we get
194 // T*eps ~= 1/8
195 // Now it is easy to find inputs around integer
196 // multiples of pi/16 that will have delta <= 1/8:
197 // e.g. N*pi/2 + 1/2*pi/2 - 1/8 * pi/2
198 // So what happens when the rounding goes wrong by 1? The reduction
199 // r = x - (N+1)*pi/2 = m*pi/2 - pi/2 = pi/2*(m - 1)
200 // = -pi/2*(1/2 + delta) = -pi/4*(1 + 2*delta)
201 // produces reduced argument r, which is off from the target interval
202 // proportionally to 2*delta. Approximation error for the minimax
203 // polynomial oscillates near the interval bounds and
204 // grows significantly outside the interval, so we may end up with
205 // computed results that will be off by too much.
206 // We may also notice that small delta doesn't actually pose the same
207 // threat as the large delta in terms of being off from the
208 // approximation interval, so we are not interested in finding
209 // the lower limit on delta in the given precision.
210 // Instead we accept that the problem with reduction comes from the
211 // arguments 'not too close' to multiples of pi/4. And the larger
212 // |x| is, the larger x*eps becomes and thus we need to accomodate
213 // for larger deltas. We may reduce the effect of the interval error
214 // by allowing our approximation interval to be wider by some
215 // margin, perhaps grow it until we need to add an extra degree to
216 // the polynomial.
217 // But we also do not want delta to be too large since the
218 // reduced argument may end up in a different binade and that would
219 // affect the polynomial evaluation error: instead of reducing
220 // to pi/4 < 2^0 we may end up with r >= 2^0.
221 // With bad delta capped by DELTA from the choice of
222 // approximation interval bounds:
223 // pi/4*(1 + 2*DELTA)
224 // we shall work from the other end and ensure x*eps < DELTA
225 // by lowering the threshold for |x| < min(T, DELTA/eps).
226
227#undef _1_OVER_PI
228#define _1_OVER_PI L2D(0x3fd45f306dc9c883)
229 double k = fma(p, _1_OVER_PI*2.0, 0x1.8p52); PRINT(k);
230
231 ssign = (D2L(k) & 0x2ull) << 62; PRINT(ssign);
232 ssign = ssign ^ (D2L(x) & DB_SIGN_BIT); PRINT(ssign);
233 csign = ((D2L(k) + 1) & 0x2ull) << 62; PRINT(csign);
234 //sign extend to obtain bitmask
235 selector = 0 - (signed long long int)(D2L(k) & 0x1ull); PRINT(selector);
236 k -= 0x1.8p52; PRINT(k);
237#undef PI_HI
238#undef PI_MI
239#undef PI_LO
240#define PI_HI L2D(0x400921fb54442d18)
241#define PI_MI L2D(0x3ca1a62633145c00)
242#define PI_LO L2D(0x398b839a252049c1)
243
244 // first reduction step
245 double u = fma(k, -PI_HI/2.0, p); PRINT(u);
246 // second reduction step
247 double v1 = fma(k, -PI_MI/2.0, u); PRINT(v1);
248 double p1, p2;
249 fast2mul_dp(k, -PI_MI/2.0, &p1, &p2); PRINT(p1); PRINT(p2);
250 double t1, t2;
251 fast2sum_dp(u, p1, &t1, &t2); PRINT(t1); PRINT(t2);
252 double v2 = (t1 - v1);
253 v2 = v2 + t2;
254 v2 = v2 + p2; PRINT(v1); PRINT(v2);
255 // third reduction step
256 w = fma(k, -PI_LO/2.0, v2); PRINT(w);
257 // now v1 + w give us the reduced argument in the double-double form
258 // redistribute bits between high and low parts
259 fast2sum_dp(v1, w, &p, &w); PRINT(p); PRINT(w);
260 }
261
262 // assume that the range reduction produced argument
263 // suitable for further minimax evaluation or NaN
264 assert( (p != p) || ((fabs(p) / (PI_HI/4.0) - 1.0) < 2.0*DELTA) );
265
266 double ps, pc;
267
268 double twoWp = (2.0*w*p);
269 double s = fma(p,p,twoWp); PRINT(s);
270
271 // 2^(-55), interval: [+-pi/4*(1 + 2*delta)], delta=0.005
272 #define S_1 L2D(0x3ff0000000000000)
273 #define S_3 L2D(0xbfc5555555555554)
274 #define S_5 L2D(0x3f81111111110a18)
275 #define S_7 L2D(0xbf2a01a019e3c0a6)
276 #define S_9 L2D(0x3ec71de37484189f)
277 #define S_11 L2D(0xbe5ae5fc23541697)
278 #define S_13 L2D(0x3de5df2f7b851b81)
279
280 // 2^(-61), interval: [+-pi/4*(1 + 2*delta)], delta=0.005
281 #define C_0 L2D(0x3ff0000000000000)
282 #define C_2 L2D(0xbfe0000000000000)
283 #define C_4 L2D(0x3fa5555555555552)
284 #define C_6 L2D(0xbf56c16c16c15ee2)
285 #define C_8 L2D(0x3efa01a019df3bcb)
286 #define C_10 L2D(0xbe927e4f8e81f5e4)
287 #define C_12 L2D(0x3e21eea7b7f9f23e)
288 #define C_14 L2D(0xbda8ff5144c2ae4b)
289
290 pc = C_14;
291 pc = fma(pc, s, C_12);
292 pc = fma(pc, s, C_10);
293 pc = fma(pc, s, C_8);
294 pc = fma(pc, s, C_6);
295 pc = fma(pc, s, C_4);
296 pc = fma(pc, s, C_2);
297 pc = fma(pc, s, C_0);
298
299 double p3 = s*p;
300 ps = S_13;
301 ps = fma(ps, s, S_11);
302 ps = fma(ps, s, S_9);
303 ps = fma(ps, s, S_7);
304 ps = fma(ps, s, S_5);
305 ps = fma(ps, s, S_3);
306 ps = fma(ps, p3, w);
307 ps += p;
308
309 double rsin, rcos;
310 rsin = L2D((selector & D2L(pc)) | ((~selector) & D2L(ps))); PRINT(rsin);
311 rsin = L2D(D2L(rsin) ^ ssign); PRINT(rsin);
312 rcos = L2D((selector & D2L(ps)) | ((~selector) & D2L(pc))); PRINT(rcos);
313 rcos = L2D(D2L(rcos) ^ csign); PRINT(rcos);
314
315 return set_cmplxd(rcos, rsin);
316}

318static vdouble __attribute__((noinline)) __vcis_d_slowpath(vdouble x, vopmask m)
319{
320 vdouble in = x;
321 vdouble res = vSETd(0.0);
322 int i;
323 const int vlen = sizeof(vdouble) / sizeof(double _Complex);
324 for (i = 0; i < vlen; i++)
325 {
326 // only work on even elements as odd ones are duplicates
327 double a = *(2*i + (double *)(&in));
328 // FIXME: here it might be more robust if we read the mask and decide,
329 // instead of duplicating the real condition
330 if (__builtin_expect(!(L2D(D2L(a) & DB_ABS_MASK) > THRESHOLD), 0)) continue;
331 *(0 + i + (double _Complex *)(&res)) = __cis_d_scalar(a);
332 }
333 return res;
334}

337static vdouble INLINE
338__vcis_d_kernel(vdouble va)
339{
340 // This function will compute cos and sin in parallel
341 // over the SIMD register filled with duplicate input
342 // values: x x y y z z, etc
343 // Will use the same range reduction for every value.
344 // If we need to take huge reduction slow path, then
345 // we will recall about duplicates.
346
347 const vdouble vRS = vSETd(0x1.8p52);
348 vdouble vabsa = vL2D(vand_vi2_vi2_vi2(vD2L(va), vSETll(DB_ABS_MASK))); PRINT(vabsa);
349 vdouble k = vfma_vd_vd_vd_vd(vabsa, vSETd(_1_OVER_PI * 2.0), vRS); PRINT(k);
350 // Need k in sin positions and k+1 in cos positions
351 const vint2 _0_1 = vSETLLL(0, 1);
352 // cssign merged sign
353 vint2 cssign = vand_vi2_vi2_vi2(vadd_vi2_vi2_vi2(vD2L(k), _0_1), vSETll(2));
354 cssign = vsll64_vi2_vi2_i(cssign, 62);
355 cssign = vxor_vi2_vi2_vi2(cssign, vand_vi2_vi2_vi2(vD2L(va), vSETLLL(DB_SIGN_BIT, 0)));
356 // parity bit of k is the selector
357 vopmask selector = veq64_vo_vm_vm(vand_vi2_vi2_vi2(vD2L(k), vSETll(1)), vSETll(1));
358 // k rounded to integer, now in normalized FP format
359 k = vsub_vd_vd_vd(k, vRS); PRINT(k);
360 // first reduction step
361 vdouble u = vfma_vd_vd_vd_vd(k, vSETd(-PI_HI/2.0), vabsa); PRINT(u);
362 // second reduction step
363 vdouble v1 = vfma_vd_vd_vd_vd(k, vSETd(-PI_MI/2.0), u); PRINT(v1);
364 vdouble p1, p2;
365 vfast2mul_dp(k, vSETd(-PI_MI/2.0), &p1, &p2); PRINT(p1); PRINT(p2);
366 vdouble t1, t2;
367 vfast2sum_dp(u, p1, &t1, &t2);
368 vdouble v2 = vsub_vd_vd_vd(t1, v1);
369 v2 = vadd_vd_vd_vd(v2, t2);
370 v2 = vadd_vd_vd_vd(v2, p2);
371 // third reduction step
372 vdouble w = vfma_vd_vd_vd_vd(k, vSETd(-PI_LO/2.0), v2);
373 vdouble p;
374 // now v1 + w give us the reduced argument in the double-double form
375 // redistribute bits between high and low parts
376 vfast2sum_dp(v1, w, &p, &w);
377 vdouble x2 = vfma_vd_vd_vd_vd(p, p, vmul_vd_vd_vd(vmul_vd_vd_vd(p, w), vSETd(2.0)));
378
379 // compute polynomials: minimax on +-pi/4.0*1.01
380 //sin coef, cos coef
381 vdouble c14s13 = vL2D(vSETLLL(D2L(S_13), D2L(C_14)));
382 vdouble c12s11 = vL2D(vSETLLL(D2L(S_11), D2L(C_12)));
383 vdouble c10s9 = vL2D(vSETLLL(D2L( S_9), D2L(C_10)));
384 vdouble c8s7 = vL2D(vSETLLL(D2L( S_7), D2L( C_8)));
385 vdouble c6s5 = vL2D(vSETLLL(D2L( S_5), D2L( C_6)));
386 vdouble c4s3 = vL2D(vSETLLL(D2L( S_3), D2L( C_4)));
387
388 vdouble x2_0 = vL2D(vand_vi2_vi2_vi2(vD2L(x2), vSETLLL(0, -1LL)));
389 vdouble x2_1 = vL2D(vor_vi2_vi2_vi2(vD2L(x2_0), vSETLLL(DB_ONE, 0)));
390 vdouble _0p = vL2D(vand_vi2_vi2_vi2(vD2L(p), vSETLLL(-1LL, 0)));
391 vdouble _0w = vL2D(vand_vi2_vi2_vi2(vD2L(w), vSETLLL(-1LL, 0)));
392 vdouble _1p = vL2D(vor_vi2_vi2_vi2(vD2L(_0p),vSETLLL(0, DB_ONE)));
393 vdouble c2w = vL2D(vor_vi2_vi2_vi2(vD2L(_0w),vSETLLL(0, D2L(C_2))));
394 vdouble x2x3 = vmul_vd_vd_vd(x2, _1p);
395
396 vdouble pcps = c14s13;
397 pcps = vfma_vd_vd_vd_vd(pcps, x2, c12s11);
398 pcps = vfma_vd_vd_vd_vd(pcps, x2, c10s9);
399 pcps = vfma_vd_vd_vd_vd(pcps, x2, c8s7);
400 pcps = vfma_vd_vd_vd_vd(pcps, x2, c6s5);
401 pcps = vfma_vd_vd_vd_vd(pcps, x2, c4s3);
402 pcps = vfma_vd_vd_vd_vd(pcps, x2x3, c2w);
403 pcps = vfma_vd_vd_vd_vd(pcps, x2_1, _1p);
404
405 // swap cos and sin in SIMD register
406 vdouble pspc = vrev21_vd_vd(pcps);
407 // select cos or sin
408 vdouble vres = vsel_vd_vo_vd_vd(selector, pspc, pcps);
409 // fixup sign
410 vres = vL2D(vxor_vi2_vi2_vi2(vD2L(vres), cssign)); PRINT(vres);
411
412 vopmask m = vgt_vo_vd_vd(vabsa, vSETd(THRESHOLD)); PRINT(m);
413 if (__builtin_expect(!vtestz_i_vo(m), 0))
414 {
415 // blend slow and fast path results
416 vdouble vslowfres = __vcis_d_slowpath(va, m);
417 vres = vsel_vd_vo_vd_vd(m, vslowfres, vres);
418 }
419
420 return vres;
421}
