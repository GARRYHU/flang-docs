15/** \brief
16 * Compare rank and shape of objects s and t. Return true if s and t
17 * are conformable under the axis mappings smap and tmap. Scalars are
18 * conformable with arrays. conform(A,B) implies conform(B,A).
19*/
20
21int I8(__fort_conform)(F90_Desc *s, __INT_T *smap, F90_Desc *t, __INT_T *tmap)
22{
23 DECL_DIM_PTRS(sd);
24 DECL_DIM_PTRS(td);
25 __INT_T dim;
26
27 if (s == NULL || t == NULL)
28 return 0;
29 if (s == t || F90_TAG_G(s) != __DESC || F90_TAG_G(t) != __DESC)
30 return 1;
31 if (F90_GSIZE_G(s) == 0 && F90_GSIZE_G(t) == 0)
32 return 1;
33 if (F90_RANK_G(s) != F90_RANK_G(t))
34 return 0;
35 for (dim = F90_RANK_G(s); --dim >= 0;) {
36 SET_DIM_PTRS(sd, s, smap[dim] - 1);
37 SET_DIM_PTRS(td, t, tmap[dim] - 1);
38 if (F90_DPTR_EXTENT_G(sd) != F90_DPTR_EXTENT_G(sd))
39 return 0;
40 }
41 return 1;
42}
