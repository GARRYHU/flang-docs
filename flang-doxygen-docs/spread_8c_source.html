13/* spread intrinsic -- copy sections for ncopies into appropriate
14 dimensions */
15
16void ENTFTN(SPREAD, spread)(void *rb, /* result base */
17 void *sb, /* source base */
18 void *dimb, /* dimension base */
19 void *ncopiesb, /* ncopies base */
20 F90_Desc *rd, /* result descriptor */
21 F90_Desc *sd, /* source descriptor */
22 F90_Desc *dimd, /* dimension descriptor */
23 F90_Desc *ncopiesd) /* ncopies descriptor */
24{
25 char *rp, *sp;
26 chdr *c;
27 DECL_DIM_PTRS(rdd);
28 DECL_HDR_VARS(td);
29 int k, dim, ncopies;
30 __INT_T i, rank, rx, tx;
31 __INT_T olb[MAXDIMS], oub[MAXDIMS];
32 __INT_T flags, lbase, pbase, repli, scoff;
33
34 dim = I8(__fort_fetch_int)(dimb, dimd);
35 ncopies = I8(__fort_fetch_int)(ncopiesb, ncopiesd);
36
37 /* form temporary descriptor with a scalar subscript in the spread
38 dimension */
39
40 rank = F90_RANK_G(rd) - 1;
41 __DIST_INIT_SECTION(td, rank, rd);
42
43 /* set up non-spread dimensions */
44
45 for (tx = 0, rx = 1; rx <= F90_RANK_G(rd); ++rx) {
46 if (rx != dim) {
47 SET_DIM_PTRS(rdd, rd, rx - 1);
48 I8(__fort_set_section)(td, ++tx, rd, rx, F90_DPTR_LBOUND_G(rdd),
49 DPTR_UBOUND_G(rdd), 1);
50 }
51 }
52
53 flags = F90_FLAGS_G(td); /* save descriptor fields */
54 lbase = F90_LBASE_G(td);
55 pbase = DIST_PBASE_G(td);
56 repli = DIST_REPLICATED_G(td);
57 scoff = DIST_SCOFF_G(td);
58 for (i = rank; --i >= 0;) {
59 olb[i] = DIST_DIM_OLB_G(td, i);
60 oub[i] = DIST_DIM_OUB_G(td, i);
61 }
62
63 sp = (char *)sb + DIST_SCOFF_G(sd) * F90_LEN_G(sd);
64 SET_DIM_PTRS(rdd, rd, dim - 1);
65 for (k = 0; k < ncopies; ++k) {
66
67 /* set scalar subscript in spread dimension */
68
69 I8(__fort_set_single)((td), rd, dim, F90_DPTR_LBOUND_G(rdd) + k, __SCALAR);
70 I8(__fort_finish_section)((td));
71
72 rp = (char *)rb + DIST_SCOFF_G(td) * F90_LEN_G(td);
73 c = I8(__fort_copy)(rp, sp, td, sd, NULL);
74 __fort_doit(c);
75 __fort_frechn(c);
76
77 F90_FLAGS_P(td, flags); /* restore descriptor fields */
78 F90_LBASE_P(td, lbase);
79 DIST_PBASE_P(td, pbase);
80 DIST_REPLICATED_P(td, repli);
81 DIST_SCOFF_P(td, scoff);
82 for (i = rank; --i >= 0;) {
83 DIST_DIM_OLB_P(td, i, olb[i]);
84 DIST_DIM_OUB_P(td, i, oub[i]);
85 }
86 DIST_CACHED_P(td, 0);
87 }
88}

91void ENTFTN(SPREADSA, spreadsa)(void *rb, /* result base */
92 void *sb, /* source base */
93 void *dimb, /* dimension base */
94 void *ncopiesb, /* ncopies base */
95 __CLEN_T *szb, /* sizeof source base */
96 F90_Desc *rd, /* result descriptor */
97 F90_Desc *sd, /* source descriptor */
98 F90_Desc *dimd, /* dimension descriptor */
99 F90_Desc *ncopiesd, /* ncopies descriptor */
100 F90_Desc *szd) /* sizeof source descriptor */
101{
102 char *rp;
103 int ncopies;
104 __CLEN_T size;
105
106 /* we assume that result is replicated and contiguous */
107
108 ncopies = I8(__fort_fetch_int)(ncopiesb, ncopiesd);
109 size = *szb;
110 rp = (char *)rb;
111 while (ncopies-- > 0) {
112 __fort_bcopy(rp, sb, size);
113 rp = rp + size;
114 }
115}

203/* set up result descriptor for spread intrinsic -- used when the dim
204 arg is variable. the added spread dimension is given a collapsed
205 distribution and the remaining dimensions are aligned with the
206 corresponding source dimensions. lbounds are set to 1 and overlap
207 allowances are set to 0. */
208
209void ENTFTN(SPREAD_DESCRIPTOR,
210 spread_descriptor)(F90_Desc *rd, /* result descriptor */
211 F90_Desc *sd, /* source descriptor */
212 __INT_T *dimb, /* dimension */
213 __INT_T *ncopiesb) /* ncopies base */
214{
215 DECL_DIM_PTRS(sdd);
216 DECL_HDR_PTRS(td);
217 __INT_T dim, extent, m, ncopies, offset, rx, sx, tx;
218
219#if defined(DEBUG)
220 if (F90_TAG_G(sd) != __DESC)
221 __fort_abort("SPREAD: invalid source arg");
222#endif
223
224 dim = *dimb;
225 if (dim < 1 || dim > F90_RANK_G(sd) + 1)
226 __fort_abort("SPREAD: invalid dim");
227
228 ncopies = *ncopiesb;
229 if (ncopies < 0)
230 ncopies = 0;
231
232 td = DIST_ALIGN_TARGET_G(sd);
233 __DIST_INIT_DESCRIPTOR(rd, F90_RANK_G(sd) + 1, F90_KIND_G(sd), F90_LEN_G(sd),
234 F90_FLAGS_G(sd), td);
235 for (rx = sx = 1; sx <= F90_RANK_G(sd); ++rx, ++sx) {
236 if (sx == dim)
237 ++rx;
238 SET_DIM_PTRS(sdd, sd, sx - 1);
239 extent = F90_DPTR_EXTENT_G(sdd);
240 offset = DIST_DPTR_TSTRIDE_G(sdd) * (F90_DPTR_LBOUND_G(sdd) - 1) +
241 DIST_DPTR_TOFFSET_G(sdd);
242
243 /*
244 * added &DIST_DIM_GEN_BLOCK_G(td,(DIST_DPTR_TAXIS_G(sdd))-1) arg
245 */
246
247 I8(__fort_set_alignment)(rd, rx, 1, extent, DIST_DPTR_TAXIS_G(sdd),
248 DIST_DPTR_TSTRIDE_G(sdd), offset,
249 &DIST_DIM_GEN_BLOCK_G(td,(DIST_DPTR_TAXIS_G(sdd))-1));
250 __DIST_SET_ALLOCATION(rd, rx, 0, 0);
251 }
252 I8(__fort_set_alignment)(rd, dim, 1, ncopies, 0, 1, 0);
253 __DIST_SET_ALLOCATION(rd, dim, 0, 0);
254 m = DIST_SINGLE_G(sd);
255 for (tx = 1; m > 0; ++tx, m >>= 1) {
256 if (m & 1)
257 I8(__fort_set_single)(rd, td, tx, DIST_INFO_G(sd, tx - 1), __SINGLE);
258 }
259 I8(__fort_finish_descriptor)(rd);
260}
