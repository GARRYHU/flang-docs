30static void __exp_d_scalar_kernel(double a, double *poly, long long int *scale)
31{
32 if ( a > EXP_D_OVF ) a = EXP_D_OVF;
33 if ( a < -EXP_D_OVF ) a = -EXP_D_OVF;
34
35 // calculating exponent; stored in the LO of each 64-bit block
36 unsigned long long int i = D2L(fma(a, L2E, DBL2INT_CVT));
37
38 // calculate mantissa
39 // fast mul rint
40 double t = fma(a, L2E, DBL2INT_CVT) - DBL2INT_CVT;
41 double m = fma(t, -LN2_HI, a);
42 m = fma(t, -LN2_LO, m);
43
44 // evaluate highest 8 terms of polynomial with estrin, then switch to horner
45 double z10 = fma(EXP_POLY_11, m, EXP_POLY_10);
46 double z8 = fma(EXP_POLY_9, m, EXP_POLY_8);
47 double z6 = fma(EXP_POLY_7, m, EXP_POLY_6);
48 double z4 = fma(EXP_POLY_5, m, EXP_POLY_4);
49
50 double m2 = m * m;
51 z8 = fma(z10, m2, z8);
52 z4 = fma(z6, m2, z4);
53
54 double m4 = m2 * m2;
55 z4 = fma(z8, m4, z4);
56
57 t = fma(z4, m, EXP_POLY_3);
58 t = fma(t, m, EXP_POLY_2);
59 t = fma(t, m, EXP_POLY_1);
60 t = fma(t, m, EXP_POLY_0);
61
62 *poly = t * 0x1.p512;
63 *scale = ((signed long long int)(i << 13) >> 13) - 512;
64}

101static void INLINE
102__vexp_d_kernel(vdouble vx, vdouble * vpoly, vdouble * vscale)
103{
104 // This algorithm computes exp(vx) in a form of
105 // 2^(vscale) * vpoly, unevaluated. vscale is an integer.
106 // The intended use of this form is for subsequent
107 // multiplication of vpoly by sin/cos, which can be small.
108 // We don't know the values of sin/cos apriori, so need
109 // to compute exp() with the extended range, thus the need
110 // to hold the scale bits in a separate integer, wider than
111 // 8 bits provided by the IEEE binary32 format.
112 // To avoid potential loss of accuracy in denormals we
113 // make sure that vpoly * sin() is a normal number - for
114 // that we shift some of the scaling from vscale to vpoly.
115 // Later scaling by 2^(vscale) may still result in a denormal
116 // and the loss of accuracy, but in this case it will be
117 // bound by ~1 ulp, which is tolerable for the implementation.
118
119 // exp algorithm outline: we reduce argument to +-ln2/2 interval
120 // by representing x = N*ln2 + z, in this case exp(x) = 2^N * exp(z).
121 // We want N to be integer and thus it is obtained as:
122 // a) N = round_to_nearest_int(x * 1/ln2)
123 // And reduced argument z is:
124 // b) z = x - N*ln2
125 // We need to guarantee that |z| < ln2/2 and we need to estimate
126 // the error introduced by reduction too.
127 // exp(x) can quickly over/underflow, so given the bounds on argument
128 // x in which we want to compute exp(), we can infer the bounds on
129 // N and decide on the precision needs in finite approximations
130 // of 1/ln2 and ln2 constants.
131 //
132 // Once the reduced argument z is known, we compute the exp(z) as
133 // a polynomial approximation. We would like exp(0) to
134 // be exactly 1, so we chose the polynomial coefficients accordingly.
135
136 // FIXME: compute overflow threshold more accurately
137
138 // clamp range of x to over/underflow bounds to avoid errors in
139 // range reduction procedure leading to unbounded polynomial.
140 vopmask mover = vgt_vo_vd_vd(vx, vSETd(EXP_D_OVF));
141 vx = vsel_vd_vo_vd_vd(mover, vSETd(EXP_D_OVF), vx);
142 // exp underflows for x <= EXP_LO, it cannot be recovered with later
143 // multiply by sincos, which is <= 1
144 vopmask munder = vle_vo_vd_vd(vx, vSETd(-EXP_D_OVF));
145 vx = vsel_vd_vo_vd_vd(munder, vSETd(-EXP_D_OVF), vx);
146 // tt = [x / ln2] = [x * log2(e)], convert to integer using right-shifter
147 const vdouble vRS = vSETd(DBL2INT_CVT);
148 // least significant bits of t now contain an integer rounded
149 // according to current rounding mode, default: to nearest
150 // TODO: this algorithm will fail in directed rounding mode because
151 // of the over/underestimate in t and thus tt.
152 vdouble t = vfma_vd_vd_vd_vd(vx, vSETd(L2E), vRS); PRINT(t);
153 // subtract right-shifter to obtain the integer as a normalized FP number
154 vdouble tt= vsub_vd_vd_vd(t, vRS); PRINT(tt);
155
156 // FMA is essential here. If no FMA, need to provide exact multiplication by
157 // LN2_HI, and this constant shall be changed to have e.g. 10 trailing zeros
158 // so that the product can absorb the 10 bits of tt.
159 vdouble z = vfma_vd_vd_vd_vd(tt, vSETd(-LN2_HI), vx); PRINT(z);
160 z = vfma_vd_vd_vd_vd(tt, vSETd(-LN2_LO), z); PRINT(z);
161
162 vint2 exponent = vD2L(t); PRINT(exponent);
163 // sign-extend integer exp:
164 // wipe dummy FP sign, FP exponent field and two leading FP mantissa
165 // bits (1 bit implicit), which are leftovers from right-shifter.
166 // NOTE: this 64-bit code works even though we use 32-bit SIMD shifts
167 exponent = vsll_vi2_vi2_i(exponent, 13); PRINT(exponent);
168 exponent = vsra_vi2_vi2_i(exponent, 13); PRINT(exponent);
169
170 // compute polynomial approximation, it shall be on the order
171 // of exp() in the reduced range [-ln2/2, ln2/2], so zz is
172 // in [1/sqrt(2), sqrt(2)] or in (1/2, 2)
173 vdouble zz = __vexp_d_poly(z); PRINT(zz);
174
175 // exponent scaling factor is now somewhere in [-1024 - 53; 1024 + 1024 + 53].
176 // Subtract 512 from scaling and add it back to polynomial so that later
177 // polynomial * sin() always results in normalized numbers.
178 // Plus it also makes exponent range symmetric:
179 // [-1024 - 53 - 512; 1024 + 53 + 512], only 12 bits of
180 // storage together with the sign.
181 exponent = vsub64_vi2_vi2_vi2(exponent, vSETll(512)); PRINT(exponent);
182 // new poly is in (2^511, 2^513)
183 zz = vmul_vd_vd_vd(zz, vL2D(vSETll((512ULL+DB_EXP_BIAS) << (DB_PREC_BITS-1)))); PRINT(zz);
184
185 *vpoly = zz;
186 *vscale = vL2D(exponent);
187 return;
188}
