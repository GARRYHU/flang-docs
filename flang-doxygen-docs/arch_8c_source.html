36const char *get_arch() {
37 char *ret = (char *) malloc(sizeof(char) * 10);
38 strcpy(ret, "unknown");
39
40 if (CPUIDX8664(is_avx512vl)() == 1) {
41 strcpy(ret, "avx512");
42 } else if (CPUIDX8664(is_avx512f)() == 1) {
43 strcpy(ret, "avx512knl");
44 } else if (CPUIDX8664(is_avx2)() == 1) {
45 strcpy(ret, "avx2");
46 } else if (CPUIDX8664(is_avx)() == 1) {
47 if (CPUIDX8664(is_intel)() == 1) {
48 strcpy(ret, "avx");
49 }
50 if (CPUIDX8664(is_amd)() == 1) {
51 if (CPUIDX8664(is_fma4)() == 1) {
52 strcpy(ret, "avxfma4");
53 } else {
54 strcpy(ret, "sse4");
55 }
56 }
57 } else {
58 if ((CPUIDX8664(is_sse4a)() == 1) || (CPUIDX8664(is_sse41)() == 1)) {
59 strcpy(ret, "sse4");
60 } else {
61 strcpy(ret, "em64t");
62 }
63 }
64
65 return ret;
66}
