23/*---------------------------------------------------------------------
24 * Usage:
25 * symini in1 in2 -o out1 out2
26 *
27 * The following files are opened:
28 *
29 * in1 - Intrinsics, generics, and predeclared definitions (symini*.n)
30 * in2 - ILMs definitions (ilmtp.n)
31 *
32 * out1 - generated initializations of symbol tables (syminidf.h)
33 * out2 - generated macros which define the predeclared numbers (pd.h)

64 SyminiF90(const std::vector<std::string> &args)
65 {
66 argtype["W"] = DT_WORD;
67 argtype["I"] = DT_INT;
68 argtype["R"] = DT_REAL;
69 argtype["D"] = DT_DBLE;
70 argtype["C"] = DT_CMPLX;
71 argtype["CD"] = DT_DCMPLX;
72 argtype["CQ"] = DT_QCMPLX;
73 argtype["SI"] = DT_SINT;
74 argtype["H"] = DT_CHAR;
75 argtype["N"] = DT_NUMERIC;
76 argtype["A"] = DT_ANY;
77 argtype["L"] = DT_LOG;
78 argtype["SL"] = DT_SLOG;
79 argtype["I8"] = DT_INT8;
80 argtype["L8"] = DT_LOG8;
81 argtype["K"] = DT_NCHAR;
82 argtype["Q"] = DT_QUAD;
83 argtype["PI"] = __POINT_T;
84
85 elt[".IN"] = LT_IN;
86 elt[".GN"] = LT_GN;
87 elt[".PD"] = LT_PD;
88 elt[".sh"] = LT_sh;
89
90 enum { INPUT, OUTPUT, SPHINX } state = INPUT;
91 int filename_argument = 0;
92 for (std::vector<std::string>::const_iterator arg = args.begin() + 1,
93 E = args.end();
94 arg != E; ++arg) {
95 if (*arg == "-o") {
96 filename_argument = 0;
97 state = OUTPUT;
98 } else if (*arg == "-s") {
99 state = SPHINX;
100 } else {
101 switch (state) {
102 case INPUT:
103 switch (filename_argument) {
104 case 0:
105 symini_filename = *arg;
106 break;
107 case 1:
108 ilmtp_filename = *arg;
109 break;
110 default:
111 usage("too many input files");
112 }
113 break;
114 case OUTPUT: {
115 FILE **out = nullptr;
116 switch (filename_argument) {
117 case 0:
118 out = &out1;
119 break;
120 case 1:
121 out = &out2;
122 break;
123 default:
124 usage("too many output files");
125 }
126 *out = fopen(arg->c_str(), "w");
127 if (!*out) {
128 usage("can't create output file");
129 }
130 outputNotModifyComment(*out, arg->c_str(), args[0], false);
131 break;
132 }
133 case SPHINX:
134 sphinx_filename = *arg;
135 break;
136 }
137 if (state != SPHINX) {
138 ++filename_argument;
139 } else {
140 state = OUTPUT;
141 }
142 }
143 }
144 if (symini_filename == "") {
145 usage("no symini.n file is given");
146 }
147 if (ilmtp_filename == "") {
148 usage("no ilmtp.n file is given");
149 }
150 if (filename_argument != 2) {
151 usage("need two output file names");
152 }
153
154 sym_init_first();
155 init_ilm(); // read ILMs
156 }

159 run()
160 {
161 ifs.open(symini_filename.c_str());
162 if (!ifs) {
163 printError(FATAL, "Can't open ", symini_filename.c_str());
164 }
165 if (!sphinx_filename.empty()) {

172 write_symfile();

193 \brief find a symbol in the symbol table if any.

197 \return pointer if there is an entry for name in symbol table,

200 SPTR
201 find_symbol(const std::string &name)
202 {
203 auto length = name.length();
204 if (length > MAXIDLEN) {
205 length = MAXIDLEN;
206 }
207 INT hashval; /* index into hashtb. */
208 HASH_ID(hashval, name.c_str(), length);
209 for (SPTR sptr = stb.hashtb[hashval]; sptr; sptr = HASHLKG(sptr)) {
210 if (name == SYMNAME(sptr))
211 return sptr;
212 }
213 return SPTR_NULL;
214 }

242 int
243 get_ilm(const std::string &ilmn)
244 {
245 auto it = std::find(ilms.begin(), ilms.end(), ilmn);
246 if (it != ilms.end()) {
247 return it - ilms.begin() + 1;
248 }
249 std::string msg("illegal ilm, ");
250 msg += ilmn + ", in template";
251 printError(WARN, msg.c_str());
252 return 0;

255 void
256 process_intrinsics()
257 {
258 auto lt = getLineAndTokenize(elt);
259 if (lt != LT_sh) {
260 printError(FATAL, "missing .sh before intrinsics");
261 }
262 /* .IN name pcnt atyp dtype ILM pname arrayf */
263 for (lt = getLineAndTokenize(elt); lt == LT_IN;
264 lt = getLineAndTokenize(elt)) {
265 auto tok = makeLower(getToken()); /* get new intrinsic name */
266 const char *tokp = tok.c_str();
267 int len = (int)tok.length();
268 if (tok[0] == '=') { /* marks I8 specific */
269#ifdef TM_I8
270 tokp++;
271 len--; /* allow I8 specific; eat '=' */
272#else
273 continue; /* ignore it */
274#endif
275 }
276 sptr = installsym(tokp, len); /* get symbol pointer */
277 if (STYPEG(sptr) != ST_UNKNOWN)
278 printError(SEVERE, "Redefinition of intrinsic");
279 STYPEP(sptr, ST_INTRIN);
280
281 /* pcnt */
282 tok = getToken();
283 if (!isdigit(tok[0])) {
284 printError(SEVERE, "param count missing, assumed to be 1");
285 PARAMCTP(sptr, 1);
286 } else {
287 PARAMCTP(sptr, atoi(tok.c_str()));
288 }
289 /* atyp */
290 tok = getToken();
291 auto i = search_atyp(tok);
292 if (i == -1) {
293 printError(SEVERE, "bad atyp, assumed to be DT_INT");
294 i = DT_INT;
295 }
296 ARGTYPP(sptr, i);
297
298 /* dtype */
299 tok = getToken();
300 if ((i = search_atyp(tok)) == -1) {
301 printError(SEVERE, "bad dtype, assumed to be DT_INT");
302 i = DT_INT;
303 }
304 INTTYPP(sptr, i);
305
306 /* ILM */
307 tok = getToken();
308 ILMP(sptr, (tok == "tc" || tok[0] == '-') ? 0 : get_ilm(tok));
309
310 /* pname */
311 tok = getToken();
312 if (tok.empty()) {
313 PNMPTRP(sptr, 0);
314 ARRAYFP(sptr, 0);
315 continue;
316 }
317 PNMPTRP(sptr, tok[0] == '-' ? 0 : putsname(tok.c_str(), tok.length()));
318
319 /* aflag */
320 tok = getToken();
321 ARRAYFP(sptr, (tok.length() == 0 || tok[0] == '-') ? 0 : get_ilm(tok));
322 }
323 if (lt != LT_sh) {
324 printError(FATAL, "missing .sh after intrinsics");
325 }
326 }

328 void
329 process_generics()
330 {
331 /* .GN name siname iname rname dname cname cdname i8name */
332 auto lt = getLineAndTokenize(elt);
333 for (; lt == LT_GN; lt = getLineAndTokenize(elt)) {
334 auto tok = makeLower(getToken()); /* get new generic name */
335 const char *tokp = tok.c_str();
336 int len = (int)tok.length();
337 if (tok[0] == '=') { /* marks I8 generic */
338#ifdef TM_I8
339 tokp++;
340 len--; /* allow I8 generic; eat '=' */
341#else
342 continue; /* ignore it */
343#endif
344 }
345 sptr = installsym(tokp, len); /* get symbol pointer */
346 if (STYPEG(sptr) != ST_UNKNOWN)
347 printError(SEVERE, "Redefinition of generic");
348 STYPEP(sptr, ST_GENERIC);
349
350 /* should make sure types are correct here, but ... */
351 /* siname */
352 tok = makeLower(getToken());
353 if (tok.length() == 0 || tok[0] == '-')
354 GSINTP(sptr, 0);
355 else {
356 auto sptr1 = installsym(tok.c_str(), tok.length());
357 if (STYPEG(sptr1) != ST_INTRIN)
358 printError(SEVERE, "Non-existent SI intrinsic");
359 GSINTP(sptr, sptr1);
360 }
361 /* iname */
362 tok = makeLower(getToken());
363 if (tok.length() == 0 || tok[0] == '-')
364 GINTP(sptr, 0);
365 else {
366 auto sptr1 = installsym(tok.c_str(), tok.length());
367 if (STYPEG(sptr1) != ST_INTRIN)
368 printError(SEVERE, "Non-existent I intrinsic");
369 GINTP(sptr, sptr1);
370 }
371 /* rname */
372 tok = makeLower(getToken());
373 if (tok.length() == 0 || tok[0] == '-')
374 GREALP(sptr, 0);
375 else {
376 auto sptr1 = installsym(tok.c_str(), tok.length());
377 if (STYPEG(sptr1) != ST_INTRIN)
378 printError(SEVERE, "Non-existent R intrinsic");
379 GREALP(sptr, sptr1);
380 }
381 /* dname */
382 tok = makeLower(getToken());
383 if (tok.length() == 0 || tok[0] == '-')
384 GDBLEP(sptr, 0);
385 else {
386 auto sptr1 = installsym(tok.c_str(), tok.length());
387 if (STYPEG(sptr1) != ST_INTRIN)
388 printError(SEVERE, "Non-existent D intrinsic");
389 GDBLEP(sptr, sptr1);
390 }
391 /* cname */
392 tok = makeLower(getToken());
393 if (tok.length() == 0 || tok[0] == '-')
394 GCMPLXP(sptr, 0);
395 else {
396 auto sptr1 = installsym(tok.c_str(), tok.length());
397 if (STYPEG(sptr1) != ST_INTRIN)
398 printError(SEVERE, "Non-existent C intrinsic");
399 GCMPLXP(sptr, sptr1);
400 }
401 /* cdname */
402 tok = makeLower(getToken());
403 if (tok.length() == 0 || tok[0] == '-')
404 GDCMPLXP(sptr, 0);
405 else {
406 auto sptr1 = installsym(tok.c_str(), tok.length());
407 if (STYPEG(sptr1) != ST_INTRIN)
408 printError(SEVERE, "Non-existent CD intrinsic");
409 GDCMPLXP(sptr, sptr1);
410 }
411 /* i8name */
412 tok = makeLower(getToken());
413#ifdef TM_I8
414 if (tok.length() == 0 || tok[0] == '-')
415 GINT8P(sptr, 0);
416 else {
417 auto sptr1 = installsym(tok.c_str(), tok.length());
418 if (STYPEG(sptr1) != ST_INTRIN)
419 printError(SEVERE, "Non-existent I8 intrinsic");
420 GINT8P(sptr, sptr1);
421 }
422#else
423 GINT8P(sptr, 0);
424#endif
425 /* gsame */
426 auto sptr1 = find_symbol(std::string(".") + SYMNAME(sptr));
427 GSAMEP(sptr, sptr1 != 0 && STYPEG(sptr1) == ST_INTRIN ? sptr1 : 0);
428 }
429 if (lt != LT_sh) {
430 printError(FATAL, "missing .sh after generics");
431 }
432 }

434 void
435 process_predeclared()
436 {
437 /* .PD name call type */
438 npd = 0;
439 for (auto lt = getLineAndTokenize(elt); lt == LT_PD;
440 lt = getLineAndTokenize(elt)) {
441 ++npd;
442 auto tok = makeLower(getToken()); /* get new predeclared name */
443 sptr = installsym(tok.c_str(), tok.length()); /* get symbol pointer */
444 if (STYPEG(sptr) != ST_UNKNOWN)
445 printError(SEVERE, "Redefinition of predeclared");
446 STYPEP(sptr, ST_PD);
447 /* output predeclared line */
448 auto s = std::string("PD_") + tok;
449 fprintf(out2, "#define %-20s%6d\n", s.c_str(), npd);
450
451 /* init PD sym */
452 DTYPEP(sptr, DT_NONE);
453 PDNUMP(sptr, npd);
454 }
