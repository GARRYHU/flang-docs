228/*
229 * Data dependence framework:
230 * mr1 and mr2 are enclosed in n common loops, with n1 loops enclosing mr1
231 * but not mr2, and n2 loops enclosing mr2 but not mr1. common is the loop
232 * index of the innermost common loop; the nesting depth of common is n;
233 * the nesting depth of mr1 is n1+n; the nesting depth of mr2 is n2+n.
234 *
235 * L[1] DO ddinfo.outer_loop
236 * ... (ddinfo.n loops)
237 * L[n] DO ddinfo.common
238 * L1[1] DO
239 * ... (ddinfo.n1 loops)
240 * L1[n1] DO MR_LOOP(ddinfo.mr1)
241 * mr1
242 * ENDDO
243 * ...
244 * ENDDO
245 * L2[1] DO
246 * ... (ddinfo.n2 loops)
247 * L2[n2] DO MR_LOOP(ddinfo.mr2)
248 * mr2
249 * ENDDO
250 * ...
251 * ENDDO
252 * ENDDO
253 * ...
254 * ENDDO
255 *
256 * The loops are unpacked into lps as follows:
257 *
258 * L1[n1],...,L1[1], L2[n2],...,L2[1], L[n],...,L[1]
259 */

304static void
305dd_succ(int loop)
306{
307 /*
308 * We need to know, for two blocks b1 and b2, if b1 can execute before b2
309 * in the same iteration of the loop; if b1 cannot execute before b2,
310 * then there can be no dependence with an equals direction from a memory
311 * reference in b1 to a memory reference in b2. To do this, we create,
312 * for each flowgraph node, a bit vector giving all the successors of
313 * that node, except that successors of the loop tail are not included.
314 * Then, a transitive closure is performed on those bit vectors.
315 */
316 int i;
317 int nbits;
318 int nnodes;
319 int ntotal;
320 int offs;
321 int change;
322 PSI_P p;
323 BV *temp_set;
324
325 /*
326 * number of bits needed -- count the number of flowgraph nodes in the
327 * loop; for each of those, need a bit for each flowgraph node in the
328 * entire flowgraph.
329 */
330 hlv.fgn = 0;
331 hlv.natural_loop = 0;
332 build_nat_loop(loop);
333
334 nnodes = hlv.fgn;
335 nbits = opt.num_nodes + 1;
336 nbits = (nbits + BV_BITS - 1) / BV_BITS;
337 ntotal = nbits * (nnodes + 1);
338 NEW(fgsucc, BVP, opt.num_nodes + 1);
339 NEW(fgbase, BV, ntotal);
340
341 /* go through and initialize sets */
342 offs = 0;
343 for (i = hlv.natural_loop; i != 0; i = FG_NATNXT(i)) {
344 if (i == LP_TAIL(loop))
345 continue;
346 fgsucc[i] = fgbase + offs;
347 assert(offs < ntotal - nbits, "dd_succ: bad offs", loop, 4);
348 offs += nbits;
349 BZERO(fgsucc[i], BV, nbits);
350 /* initialize successors */
351 for (p = FG_SUCC(i); p != 0; p = PSI_NEXT(p))
352 bv_set(fgsucc[i], PSI_NODE(p));
353 }
354 /* tail has no successors */
355 fgsucc[LP_TAIL(loop)] = fgbase + offs;
356 assert(offs < ntotal - nbits, "dd_succ: bad offs", loop, 4);
357 offs += nbits;
358 BZERO(fgsucc[LP_TAIL(loop)], BV, nbits);
359
360 /* temporary set */
361 temp_set = fgbase + offs;
362 assert(offs <= ntotal - nbits, "dd_succ: bad offs", loop, 4);
363 offs += nbits;
364
365 /* now perform transitive closure */
366 do {
367 change = 0;
368 for (i = hlv.natural_loop; i != 0; i = FG_NATNXT(i)) {
369 if (i == LP_TAIL(loop))
370 continue;
371 bv_copy(temp_set, fgsucc[i], nbits);
372 /*
373 * union successor set of this node with successor sets of this
374 * node's successor
375 */
376 for (p = FG_SUCC(i); p != 0; p = PSI_NEXT(p)) {
377 bv_union(temp_set, fgsucc[PSI_NODE(p)], nbits);
378 }
379 if (bv_notequal(temp_set, fgsucc[i], nbits)) {
380 change = TRUE;
381 bv_copy(fgsucc[i], temp_set, nbits);
382 }
383 }
384 } while (change);
385#if DEBUG
386 if (DBGBIT(36, 32)) {
387 fprintf(gbl.dbgfil, "Successor bit sets\n");
388 for (i = hlv.natural_loop; i != 0; i = FG_NATNXT(i)) {
389 fprintf(gbl.dbgfil, "%3d:", i);
390 bv_print(fgsucc[i], opt.num_nodes + 1);
391 }
392 }
393#endif
394}

396static void
397build_loop_dd(int loop)
398{
399 int i, j;
400 int l1, l2;
401 int end, end1;
402
403 ++ddinfo.n;
404 TRACE2("build_loop_dd: loop %d lev %d\n", loop, ddinfo.n);
405 /* do inner loops first */
406 for (i = VL_CHILD(loop); i != 0; i = VL_SIBLING(i)) {
407 build_loop_dd(i);
408 }
409 ddinfo.common = loop;
410
411 /* build successor bit vector */
412 dd_succ(loop);
413
414 /* go through all memory references */
415
416 /* use pragmas for the common loop */
417 ddinfo.pragmas = VL_PRAGMAS(loop);
418
419 /* first, this loop */
420 end = VL_MRSTART(loop) + VL_MRCNT(loop);
421 for (i = VL_MRSTART(loop); i < end; ++i)
422 for (j = VL_MRSTART(loop); j <= i; ++j)
423 chkref(i, j);
424
425 /* next, this loop against inner loops */
426 end1 = VL_MRSTART(loop) + VL_MRECNT(loop);
427 for (i = VL_MRSTART(loop); i < end; ++i)
428 for (j = end; j < end1; ++j)
429 chkref(i, j);
430
431 /* finally, inner loops against other inner loops */
432 for (l1 = VL_CHILD(loop); l1 != 0; l1 = VL_SIBLING(l1))
433 for (l2 = VL_CHILD(loop); l2 != l1; l2 = VL_SIBLING(l2)) {
434 end = VL_MRSTART(l1) + VL_MRECNT(l1);
435 end1 = VL_MRSTART(l2) + VL_MRECNT(l2);
436 for (i = VL_MRSTART(l1); i < end; ++i)
437 for (j = VL_MRSTART(l2); j < end1; ++j)
438 chkref(i, j);
439 }
440
441 FREE(fgsucc);
442 FREE(fgbase);
443 --ddinfo.n;
444#if DEBUG
445 if (DBGBIT(36, 8)) {
446 fprintf(gbl.dbgfil, "----Mem ref after build_dd loop %d----\n", loop);
447 dump_memrefs(VL_MRSTART(loop), VL_MRECNT(loop));
448 }
449#endif
450}

452/* assume mr1, mr2 in same fg node; return TRUE if mr1 comes before mr2 */
453static LOGICAL
454mr_precedes(int mr1, int mr2)
455{
456 int ilt1, ilt2;
457
458 if (mr1 == mr2)
459 return FALSE;
460 ilt1 = MR_ILT(mr1);
461 ilt2 = MR_ILT(mr2);
462 if (ilt1 == ilt2) {
463 /* if this is the same ilt, don't allow the store to precede the
464 * load */
465 if (MR_TYPE(mr1) != 'l') {
466 assert(MR_TYPE(mr2) == 'l', "mr_precedes: too may st/br", mr2, 4);
467 return FALSE;
468 }
469 return TRUE;
470 }
471 while (ilt1 != 0) {
472 if (ilt1 == ilt2)
473 return TRUE;
474 ilt1 = ILT_NEXT(ilt1);
475 }
476 return FALSE;
477}

479static void
480chkref(int i, int j)
481{
482 DIRVEC exo_dirvec_ji;
483 DIRVEC exo_dirvec_ij;
484 int fg1, fg2;
485 DIRVEC vec;
486 DV *p;
487
488 if ((MR_TYPE(i) != 'l' && MR_TYPE(i) != 's') ||
489 (MR_TYPE(j) != 'l' && MR_TYPE(j) != 's'))
490 return;
491 if (MR_TYPE(i) != 's' && MR_TYPE(j) != 's')
492 return;
493
494 STRACE2("mem ref j %d i %d\n", j, i);
495
496 /*
497 * now compute data dependence vector indicating when a
498 * dependence can exist between memory references i and j
499 */
500 /* compute all possible dependence vectors */
501 ddinfo.dvlist = NULL;
502 ddinfo.mr1 = j;
503 ddinfo.mr2 = i;
504 dd_compute();
505
506 /** compute execution orders from j to i */
507 fg1 = MR_FG(j);
508 fg2 = MR_FG(i);
509 if (fg1 == fg2) {
510 if (mr_precedes(j, i))
511 exo_dirvec_ji = dirv_exo(ddinfo.n, TRUE);
512 else
513 exo_dirvec_ji = dirv_exo(ddinfo.n, FALSE);
514 } else if (bv_mem(fgsucc[fg1], fg2))
515 exo_dirvec_ji = dirv_exo(ddinfo.n, TRUE);
516 else
517 exo_dirvec_ji = dirv_exo(ddinfo.n, FALSE);
518
519 /** compute execution orders from i to j */
520 if (fg1 == fg2) {
521 if (mr_precedes(i, j))
522 exo_dirvec_ij = dirv_exo(ddinfo.n, TRUE);
523 else
524 exo_dirvec_ij = dirv_exo(ddinfo.n, FALSE);
525 } else if (bv_mem(fgsucc[fg2], fg1))
526 exo_dirvec_ij = dirv_exo(ddinfo.n, TRUE);
527 else
528 exo_dirvec_ij = dirv_exo(ddinfo.n, FALSE);
529
530 /*
531 * intersect the execution order direction vector with the
532 * dependence direction vectors to determine true dependence
533 * direction vector
534 */
535
536 for (p = ddinfo.dvlist; p != 0; p = p->next) {
537 /* j precedes i */
538 vec = p->vec & exo_dirvec_ji;
539 if (!dirv_chkzero(vec, ddinfo.n))
540 dd_edge(j, i, vec);
541
542 /* i precedes j; must invert dependence vector */
543 vec = dirv_inverse(p->vec) & exo_dirvec_ij;
544 if (!dirv_chkzero(vec, ddinfo.n))
545 dd_edge(i, j, vec);
546 }
547 freearea(HLV_AREA1);
548}

556void
557dd_edge(int src, int sink, DIRVEC vec)
558{
559 int type; /* type of dependence */
560 DDEDGE *p;
561
562 /* figure out type of dependence */
563 /* dependence goes from src to sink */
564 ddinfo.vec = 0;
565 dirv_gen(vec, (int *)0, 0, DIRV_BIGPOS, edge_func);
566 vec = ddinfo.vec;
567 if (vec == 0)
568 return;
569 TRACE3("dd_edge: src %d sink %d vec %s \n", src, sink, dirv_print(vec));
570 if (MR_TYPE(src) == 's' && MR_TYPE(sink) == 's')
571 type = DIRV_FOUT;
572 else if (MR_TYPE(src) == 'l' && MR_TYPE(sink) == 's') {
573 type = DIRV_FANTI;
574 /* under certain circumstances we can ignore anti dependences.
575 * If they are from the load of a scalar to the store of the same
576 * scalar, and the store has a loop-invariant RHS. There must
577 * also be only one assignment of the variable in the loop, and
578 * the memory refs must be in the same ilt.
579 * The theory is that it doesn't matter if we do the load before the
580 * store, since the same value is always stored.
581 */
582 if (MR_NME(src) == MR_NME(sink) && MR_INVAL(sink) &&
583 MR_ILT(src) == MR_ILT(sink)) {
584 int def;
585 int lp;
586 int inloop;
587
588 inloop = 0;
589 for (def = NME_DEF(MR_NME(src)); def != 0; def = DEF_NEXT(def)) {
590 /* is it in the loop? */
591 for (lp = FG_LOOP(DEF_FG(def)); lp != 0; lp = LP_PARENT(lp))
592 if (lp == ddinfo.common)
593 goto in_loop;
594 continue; /* not in loop */
595 in_loop:
596 if (inloop++)
597 goto skip;
598 }
599 TRACE0(" Ignoring edge because inv scalar\n");
600 return;
601 }
602 skip:;
603 } else if (MR_TYPE(src) == 's' && MR_TYPE(sink) == 'l')
604 type = DIRV_FFLOW;
605 else {
606 interr("unknown load/load dep in dd_edge", 0, 3);
607 type = 3;
608 }
609 /* add to list for this mem ref */
610 for (p = MR_SUCC(src); p != 0; p = DD_NEXT(p)) {
611 if (DD_SINK(p) == sink && DD_TYPE(p) == type) {
612 DD_DIRVEC(p) |= vec;
613 return;
614 }
615 }
616 /* add a new one */
617 p = (DDEDGE *)getitem(HLV_AREA, sizeof(DDEDGE));
618 DD_TYPE(p) = type;
619 DD_DIRVEC(p) = vec;
620 DD_SINK(p) = sink;
621 DD_NEXT(p) = MR_SUCC(src);
622 MR_SUCC(src) = p;
623}

638{
639 int nm1, nm2;
640 int ui, loop, mr;
641
642 STRACE4("dd_compute: mr1 %d mr2 %d common %d n %d\n", ddinfo.mr1, ddinfo.mr2,
643 ddinfo.common, ddinfo.n);
644
645 /* I believe that 'yuck' is the appropriate word here. All this
646 * deals with is initial values that the vectorizer added for induction
647 * vars with non-invariant initial values. Since the uses haven't
648 * yet been replaced, artificial checks must be done for a dependence
649 * that will be added. While ugly, the code is probably correct and
650 * reasonably efficient.
651 */
652 ui = FALSE;
653 if (MR_IVUSE(ddinfo.mr1) && MR_INIT(ddinfo.mr2)) {
654 loop = MR_LOOP(ddinfo.mr1);
655 mr = ddinfo.mr2;
656 if (VL_PREBIH(loop) == FG_TO_BIH(MR_FG(ddinfo.mr2)))
657 ui = TRUE;
658 } else if (MR_IVUSE(ddinfo.mr2) && MR_INIT(ddinfo.mr1)) {
659 loop = MR_LOOP(ddinfo.mr2);
660 mr = ddinfo.mr1;
661 if (VL_PREBIH(loop) == FG_TO_BIH(MR_FG(ddinfo.mr1)))
662 ui = TRUE;
663 }
664 if (ui) {
665 /* initial value-use situation */
666 int i;
667
668 for (i = VL_IVLIST(loop); i != 0; i = MR_NEXT(i)) {
669 if (MR_IV(i) == MR_IV(mr)) {
670 add_dep(dirv_fulldep(ddinfo.n));
671 return;
672 }
673 }
674 }
675
676 nm1 = MR_NME(ddinfo.mr1);
677 while (NME_TYPE(nm1) == NT_ARR || NME_TYPE(nm1) == NT_MEM)
678 nm1 = NME_NM(nm1);
679 ddinfo.basenm1 = nm1;
680
681 nm2 = MR_NME(ddinfo.mr2);
682 while (NME_TYPE(nm2) == NT_ARR || NME_TYPE(nm2) == NT_MEM)
683 nm2 = NME_NM(nm2);
684 ddinfo.basenm2 = nm2;
685
686 if (NME_TYPE(nm1) == NT_VAR && NME_TYPE(nm2) == NT_VAR) {
687 if (nm1 != nm2) {
688 int i;
689 /*
690 * for FORTRAN equivalence stmts... return TRUE if one symbol is
691 * in storage overlap chain of the other. Eventually we need to
692 * be more clever here.
693 */
694 int s1 = basesym_of(nm1);
695 int s2 = basesym_of(nm2);
696 if (!VP_DEPCHK(ddinfo.pragmas) || !VP_EQVCHK(ddinfo.pragmas))
697 return;
698 if (SOCPTRG(s1)) {
699 for (i = SOCPTRG(s1); i; i = SOC_NEXT(i))
700 if (SOC_SPTR(i) == s2) {
701 add_dep(dirv_fulldep(ddinfo.n));
702 return;
703 }
704 }
705 if ((POINTERG(s1) || POINTERG(s2)) &&
706 expr_dependent(MR_ILI(ddinfo.mr2), MR_ILI(ddinfo.mr1),
707 MR_ILT(ddinfo.mr2), MR_ILT(ddinfo.mr1))) {
708 add_dep(dirv_fulldep(ddinfo.n));
709 return;
710 }
711 return;
712 }
713 resolve_vv();
714 } else if ((NME_TYPE(nm1) == NT_IND && NME_TYPE(nm2) == NT_VAR) ||
715 (NME_TYPE(nm1) == NT_VAR && NME_TYPE(nm2) == NT_IND))
716 resolve_pv();
717 else if (NME_TYPE(nm1) == NT_IND && NME_TYPE(nm2) == NT_IND)
718 resolve_pp();
719 else if ((NME_TYPE(nm1) == NT_UNK && NME_TYPE(nm2) == NT_IND) ||
720 (NME_TYPE(nm1) == NT_IND && NME_TYPE(nm2) == NT_UNK) ||
721 (NME_TYPE(nm1) == NT_UNK && NME_TYPE(nm2) == NT_UNK)) {
722 if (!VP_DEPCHK(ddinfo.pragmas))
723 return;
724 add_dep(dirv_fulldep(ddinfo.n));
725 } else if ((NME_TYPE(nm1) == NT_UNK && NME_TYPE(nm2) == NT_VAR) ||
726 (NME_TYPE(nm1) == NT_VAR && NME_TYPE(nm2) == NT_UNK))
727 resolve_uv();
728 else
729 assert(0, "dd_compute: can't happen", 0, 4);
730}

783static void
784unpack_loops(void)
785{
786 int i, j;
787 /* unpack the loops */
788 i = 0;
789 ddinfo.n1 = 0;
790 for (j = MR_LOOP(ddinfo.mr1); j != ddinfo.common; j = LP_PARENT(j)) {
791 ddinfo.lps[i++] = j;
792 ++ddinfo.n1;
793 }
794 ddinfo.n2 = 0;
795 for (j = MR_LOOP(ddinfo.mr2); j != ddinfo.common; j = LP_PARENT(j)) {
796 ddinfo.lps[i++] = j;
797 ++ddinfo.n2;
798 }
799 j = ddinfo.common;
800 for (;;) {
801 ddinfo.lps[i++] = j;
802 if (j == ddinfo.outer_loop)
803 break;
804 j = LP_PARENT(j);
805 }
806 assert(i <= MAX_DD, "resolve_vv: bad i", ddinfo.common, 4);
807 assert(i == ddinfo.n1 + ddinfo.n2 + ddinfo.n, "resolve_vv: bad iA",
808 ddinfo.common, 4);
809 ddinfo.subs1 = MR_SUBST(ddinfo.mr1) + MR_SUBCNT(ddinfo.mr1);
810 ddinfo.subs2 = MR_SUBST(ddinfo.mr2) + MR_SUBCNT(ddinfo.mr2);
811 DTRACE2("resolve_vv: subs1 %d subs2 %d\n", ddinfo.subs1, ddinfo.subs2);

814static void
815resolve_vv(void)
816{
817 int i1, i2;
818 int nm1, nm2;
819 DIRVEC dir;
820 int buf1[MAXNMESIZ], buf2[MAXNMESIZ];
821 int i;
822 int j;
823
824 /* check for both scalar */
825 if (MR_SCALR(ddinfo.mr1) && MR_SCALR(ddinfo.mr2)) {
826 /* complex have different nmes */
827 assert(ddinfo.basenm1 == ddinfo.basenm2, "resolve_vv: mr1 != mr2",
828 ddinfo.mr1, 4);
829 /* expanded scalars are different */
830 for (i = VL_SCLIST(ddinfo.common); i != 0; i = SCLR_NEXT(i))
831 if (MR_NME(ddinfo.mr1) == SCLR_NME(i))
832 goto found;
833 add_dep(dirv_fulldep(ddinfo.n));
834 return;
835 found:
836 /* equivalent to equals direction in last place */
837 dir = DIRV_ALLEQ;
838 for (i = 1; i < ddinfo.n; ++i)
839 DIRV_ENTRYP(dir, i, DIRV_STAR);
840 DIRV_ENTRYP(dir, 0, DIRV_EQ);
841 add_dep(dir);
842 return;
843 }
844 /* unpack names info into buf1, buf2 */
845 i1 = i2 = MAXNMESIZ;
846 nm1 = MR_NME(ddinfo.mr1);
847 while (i1 > 0) {
848 if (NME_TYPE(nm1) == NT_VAR)
849 break;
850 buf1[--i1] = nm1;
851 nm1 = NME_NM(nm1);
852 }
853 /* too big? */
854 if (NME_TYPE(nm1) != NT_VAR) {
855 add_dep(dirv_fulldep(ddinfo.n));
856 return;
857 }
858
859 nm2 = MR_NME(ddinfo.mr2);
860 while (i2 > 0) {
861 if (NME_TYPE(nm2) == NT_VAR)
862 break;
863 buf2[--i2] = nm2;
864 nm2 = NME_NM(nm2);
865 }
866 /* too big? */
867 if (NME_TYPE(nm2) != NT_VAR) {
868 add_dep(dirv_fulldep(ddinfo.n));
869 return;
870 }
871
872 unpack_loops();
873
874 /* see if we can avoid any checking */
875 i = 0;
876 for (;;) {
877 /* invariant: we have a common prefix of buf1 and buf2 */
878 /* for example: a.b.c[...] */
879 if (i1 + i >= MAXNMESIZ || i2 + i >= MAXNMESIZ)
880 break;
881 if (NME_TYPE(buf1[i1 + i]) == NT_MEM && NME_TYPE(buf2[i2 + i]) == NT_MEM) {
882 if (NME_SYM(buf1[i1 + i]) != NME_SYM(buf2[i2 + i])) {
883 ddinfo.dvlist = NULL;
884 return;
885 }
886 }
887 ++i;
888 }
889 i = 0;
890 j = 0; /* count subscripts */
891 for (;;) {
892 /* all member references match */
893 if (i1 + i >= MAXNMESIZ || i2 + i >= MAXNMESIZ)
894 break;
895 if (NME_TYPE(buf1[i1 + i]) == NT_MEM && NME_TYPE(buf2[i2 + i]) == NT_MEM) {
896 assert(NME_SYM(buf1[i1 + i]) == NME_SYM(buf2[i2 + i]),
897 "resolve_vv: diff MEM symbols", 0, 4);
898 ++i;
899 } else if (NME_TYPE(buf1[i1 + i]) == NT_ARR &&
900 NME_TYPE(buf2[i2 + i]) == NT_ARR) {
901 while (i1 + i < MAXNMESIZ && NME_TYPE(buf1[i1 + i]) == NT_ARR) {
902 assert(NME_TYPE(buf2[i2 + i]) == NT_ARR, "resolve_vv: not both arrays",
903 0, 4);
904 assert(ASD_NDIM(NME_SUB(buf1[i1 + i])) ==
905 ASD_NDIM(NME_SUB(buf1[i1 + i])),
906 "resolve_vv: ndims not equal", 0, 4);
907 j += ASD_NDIM(NME_SUB(buf1[i1 + i]));
908 ++i;
909 }
910 } else
911 assert(0, "resolve_vv: can't happen", 0, 4);
912 }
913 do_subscript(j);
914 ddinfo.subs1 -= j;
915 ddinfo.subs2 -= j;
916}

951gcd(INT a, INT b)
952{
953 INT u, v;
954 INT r;
955
956 u = a > 0 ? a : -a;
957 v = b > 0 ? b : -b;
958 while (v != 0) {
959 r = u % v;
960 u = v;
961 v = r;
962 }
963 return u;
964}

967ceilg(int t, int g)
968{
969 int q;
970 int a;
971
972 a = t;
973 if (t < 0)
974 a = -t;
975 /* return ceil(t/g), g is positive */
976 q = a / g;

983 ddinfo.unknown = 1;
984 if (t > 0)

1012static LOGICAL
1013compare_one_bound(BOUND *low, BOUND *up, int var)
1014{
1015 int i, j;
1016 int tmp[MAX_N + 1];
1017 LOGICAL btype;
1018 int t, g, t1;
1019 BOUND b, *p;
1020 int icon0 = ad_icon(0);
1021
1022#if DEBUG
1023 if (DBGBIT(36, 64)) {
1024 fprintf(gbl.dbgfil, "compare_one_bound: ");
1025 dump_two_bound(low, up, var, FALSE);
1026 }
1027#endif
1028 /* compute difference of bounds */
1029 for (j = 1; j < var; ++j) {
1030 tmp[j] = ili_symbolic(
1031 ad2ili(IL_ISUB, ad2ili(IL_IMUL, up->bnd[j], ad_icon(low->mplyr)),
1032 ad2ili(IL_IMUL, low->bnd[j], ad_icon(up->mplyr))));
1033 }
1034 /* compute difference of lower bounds */
1035 tmp[0] = ili_symbolic(ad2ili(
1036 IL_ISUB, ad2ili(IL_IMUL, up->bnd[0], ad_icon(low->mplyr * low->gcd)),
1037 ad2ili(IL_IMUL, low->bnd[0], ad_icon(up->mplyr * up->gcd))));
1038#if DEBUG
1039 if (DBGBIT(36, 64)) {
1040 fprintf(gbl.dbgfil, "compare_one_bound: up-low: ");
1041 for (j = 0; j < var; ++j) {
1042 prilitree(tmp[j]);
1043 fprintf(gbl.dbgfil, ", ");
1044 }
1045 fprintf(gbl.dbgfil, "\n");
1046 }
1047#endif
1048 /* tmp[0] is divided by product of gcds */
1049 /* find highest non-zero coefficient */
1050 for (j = var - 1; j > 0; --j)
1051 if (tmp[j] != icon0)
1052 break;
1053 if (j == 0) {
1054 /* check this one immediately */
1055 if (IS_CNST(tmp[0])) {
1056 if (CNSTG(tmp[0]) < 0) {
1057 DTRACE0(
1058 "compare_one_bound finds constant inconsistency, returns nodep\n");
1059 return TRUE;
1060 }
1061 return FALSE; /* constant consistent */
1062 }
1063 /* variable inconsistent, unknown data dep */
1064 ddinfo.unknown = 1;
1065 return FALSE;
1066 }
1067 /* This coefficient must be constant or we'll ignore this bound */
1068 /* since we don't know if it is upper or lower */
1069 if (!IS_CNST(tmp[j])) {
1070 DTRACE1("compare_one_bound ignores bound for %d\n", var);
1071 ddinfo.unknown = 1;
1072 return FALSE;
1073 }
1074 /* decide whether this is a lower or upper bound */
1075 t = CNSTG(tmp[j]);
1076 if (t > 0) {
1077 btype = FALSE; /* lower */
1078 for (i = 0; i < j; ++i)
1079 tmp[i] = ili_symbolic(ad1ili(IL_INEG, tmp[i]));
1080 } else {
1081 btype = TRUE;
1082 tmp[j] = ad1ili(IL_INEG, tmp[j]);
1083 t = -t;
1084 }
1085 /* compute gcd if possible */
1086 g = 1;
1087 if (t != 1) {
1088 g = 0;
1089 for (i = 1; i <= j; ++i) {
1090 if (!IS_CNST(tmp[i]))
1091 break;
1092 g = gcd(g, CNSTG(tmp[i]));
1093 }
1094 if (g != 0) {
1095 /* divide thru */
1096 t /= g;
1097 for (i = 1; i <= j; ++i) {
1098 tmp[i] = ad_icon(CNSTG(tmp[i]) / g);
1099 }
1100 /* try to handle 0th element */
1101 if (IS_CNST(tmp[0])) {
1102 t1 = CNSTG(tmp[0]);
1103 /* compute floor/ceil ( t1 / g) */
1104 if (btype)
1105 t1 = floorg(t1, g);
1106 else
1107 t1 = ceilg(t1, g);
1108 tmp[0] = ad_icon(t1);
1109 g = 1;
1110 }
1111 } else
1112 g = 1;
1113 }
1114 b.gcd = up->gcd * low->gcd * g;
1115 b.mplyr = t;
1116 for (i = 0; i < j; ++i)
1117 b.bnd[i] = tmp[i];
1118
1119#if DEBUG
1120 if (DBGBIT(36, 64)) {
1121 fprintf(gbl.dbgfil, "compare_one_bound: new bound: ");
1122 dump_one_bound(&b, j, btype);
1123 }
1124#endif
1125 /* add this one */
1126 for (p = Bound[j][btype]; p != 0; p = p->next) {
1127 if (b.mplyr == p->mplyr && b.gcd == p->gcd) {
1128 for (i = 0; i < j; ++i)
1129 if (p->bnd[i] != tmp[i])
1130 goto cont;
1131 goto found;
1132 }
1133 cont:;
1134 }
1135 p = (BOUND *)getitem(HLV_AREA1, sizeof(BOUND));
1136 *p = b;
1137 p->next = Bound[j][btype];
1138 Bound[j][btype] = p;
1139#if DEBUG
1140 if (DBGBIT(36, 16)) {
1141 fprintf(gbl.dbgfil, "Add bound---:");
1142 dump_one_bound(p, j, btype);
1143 }
1144#endif
1145found:
1146 return FALSE;
1147}

1149/* ibound = bound in terms of index variables */
1150/* upflag = true if upper bound */
1151/* var = index var for which this is a bound */
1152static int
1153bound_add(int *ibound, LOGICAL upflag, int var)
1154{
1155 BOUND b1;
1156 BOUND b2;
1157 int j, k;
1158 int icon0 = ad_icon(0);
1159
1160 /* express the bound in terms of the free variables */
1161 /* create a linear combination of bounds on free variables */
1162 BTRACE4("bound_add: nvars: %d, nfree: %d, upflag: %d, var: %d\n", nvars,
1163 nfree, upflag, var);
1164 for (j = 0; j <= nfree; ++j) {
1165 b1.bnd[j] = icon0;
1166 }
1167 b1.bnd[0] = ibound[0];
1168 for (j = 1; j <= nvars; ++j) {
1169 /* b1.bnd += bound[j] * TU(*,j) */
1170 for (k = 0; k <= nfree; ++k) {
1171 b1.bnd[k] = ili_symbolic(
1172 ad2ili(IL_IADD, b1.bnd[k], ad2ili(IL_IMUL, ibound[j], TU[k][j - 1])));
1173 }
1174 }
1175 /* construct b2 */
1176 for (k = 0; k <= nfree; ++k)
1177 b2.bnd[k] = TU[k][var];
1178 b1.gcd = b2.gcd = 1;
1179 b1.mplyr = b2.mplyr = 1;
1180 if (upflag) {
1181 /* TU(*,var) <= tmp1 */
1182 return compare_one_bound(&b2, &b1, nfree + 1);
1183 } else
1184 return compare_one_bound(&b1, &b2, nfree + 1);
1185}

1187static LOGICAL
1188check_bounds(void)
1189{
1190 int k;
1191 BOUND *p, *q;
1192
1193 /* compare bounds from nfree down */
1194 for (k = nfree; k >= 1; --k) {
1195 for (p = Bound[k][1]; p != 0; p = p->next)
1196 for (q = Bound[k][0]; q != 0; q = q->next) {
1197 /* compare p and q , q <= p */
1198 if (compare_one_bound(q, p, k))
1199 return TRUE;
1200 }
1201 }
1202 return FALSE;

1206bnd_le(int k, LOGICAL eqflag)

1239/* check for dependence with direction vector dir */
1240static LOGICAL
1241check_new_bound(DIRVEC dir, int lev)
1242{
1243 int k;
1244 DIRVEC d;
1245
1246 BCOPY(Bound, SaveBound, BOUND *, BOUND_LEN);
1247 /* create equations describing this direction vector */
1248 for (k = 0; k < lev; ++k) {
1249 d = DIRV_ENTRYG(dir, lev - k - 1);
1250 switch (d) {
1251 case DIRV_STAR:
1252 break;
1253 case DIRV_LT:
1254 /* ik < jk --> ik+1 <= jk */
1255 if (bnd_le(k, FALSE))
1256 return TRUE;
1257 break;
1258 case DIRV_EQ:
1259 /* simulate with <=, >= */
1260 /* ik >= jk and ik <= jk */
1261 if (bnd_le(k, TRUE))
1262 return TRUE;
1263 if (bnd_ge(k, TRUE))
1264 return TRUE;
1265 break;
1266 case DIRV_GT:
1267 /* ik > jk --> ik >= jk+1 */
1268 if (bnd_ge(k, FALSE))
1269 return TRUE;
1270 break;
1271 }
1272 }
1273 return check_bounds();
1274}

1276static void
1277do_subscript(int nsubs)
1278{
1279 /*
1280 * this function is the heart of the data dependence analysis. subs1 and
1281 * subs2 point to subscript entries for the same array. do_subscript
1282 * determines a direction vector under which those subscripts will
1283 * intersect.
1284 */
1285 int i, j, k, k1, bnd, n1;
1286 int q, d1, d2, sgn, sc, ili;
1287 int d;
1288 int t;
1289 DIRVEC vec, vec1;
1290 int icon0, icon1, iconneg1;
1291 int sub;
1292 int invar1, invar2;
1293 int lp;
1294 int tmp[MAX_N + 1];
1295
1296 if (nsubs <= 0)
1297 goto give_up;
1298 icon0 = ad_icon(0L);
1299 icon1 = ad_icon(1L);
1300 iconneg1 = ad_icon(-1L);
1301 /* quick check for non-intersecting loop-invariant subscripts */
1302 /* also find out how many outer loops to remove */
1303 /* Set d to number of outer loops to consider based on
1304 * how far we were able to analyze both subscripts. */
1305 d = 0x7FFFFFFF;
1306 for (sub = 1; sub <= nsubs; ++sub) {
1307 k = ddinfo.n1 + ddinfo.n;
1308 for (i = 0; i < k; ++i) {
1309 if (SB_STRIDE(ddinfo.subs1 - sub)[i] == 0) {
1310 break;
1311 }
1312 if (SB_STRIDE(ddinfo.subs1 - sub)[i] < 0 ||
1313 SB_STRIDE(ddinfo.subs1 - sub)[i] >= astb.stg_avail) {
1314 fprintf(stderr, "sub=%d,ddinfo.subs1=%d,i=%d,SB_STRICE(%d)[%d]=%d\n",
1315 sub, ddinfo.subs1, i, ddinfo.subs1 - sub, i,
1316 SB_STRIDE(ddinfo.subs1 - sub)[i]);
1317 }
1318 }
1319 if (d > i - ddinfo.n1)
1320 d = i - ddinfo.n1;
1321 k = ddinfo.n2 + ddinfo.n;
1322 for (i = 0; i < k; ++i) {
1323 if (SB_STRIDE(ddinfo.subs2 - sub)[i] == 0) {
1324 break;
1325 }
1326 if (SB_STRIDE(ddinfo.subs2 - sub)[i] < 0 ||
1327 SB_STRIDE(ddinfo.subs2 - sub)[i] >= astb.stg_avail) {
1328 fprintf(stderr, "sub=%d,ddinfo.subs2=%d,i=%d,SB_STRICE(%d)[%d]=%d\n",
1329 sub, ddinfo.subs2, i, ddinfo.subs2 - sub, i,
1330 SB_STRIDE(ddinfo.subs2 - sub)[i]);
1331 }
1332 }
1333 if (d > i - ddinfo.n2)
1334 d = i - ddinfo.n2;
1335 }
1336 if (d <= 0)
1337 goto give_up;
1338 for (sub = 1; sub <= nsubs; ++sub) {
1339 invar1 = TRUE;
1340 k = ddinfo.n1 + d;
1341 for (i = 0; i < k; ++i)
1342 if (SB_STRIDE(ddinfo.subs1 - sub)[i] != icon0) {
1343 invar1 = FALSE;
1344 break;
1345 }
1346 invar2 = TRUE;
1347 k = ddinfo.n2 + d;
1348 for (i = 0; i < k; ++i)
1349 if (SB_STRIDE(ddinfo.subs2 - sub)[i] != icon0) {
1350 invar2 = FALSE;
1351 break;
1352 }
1353 if (invar1 && invar2) {
1354 d1 = SB_BASES(ddinfo.subs1 - sub)[ddinfo.n1 + d];
1355 d2 = SB_BASES(ddinfo.subs2 - sub)[ddinfo.n2 + d];
1356 if (A_TYPEG(d1) == A_TRIPLE || A_DTYPEG(d2) == A_TRIPLE) {
1357 /* get lower bound from d1, d2 */
1358 int d1low, d1high, d2low, d2high;
1359 if (A_TYPEG(d1) != A_TRIPLE) {
1360 d1low = d1high = d1;
1361 } else {
1362 d1low = A_LBDG(d1);
1363 d1high = A_UPBDG(d1);
1364 if (A_STRIDEG(d1)) {
1365 int d1str;
1366 d1str = ili_symbolic(A_STRIDEG(d1));
1367 if (!IS_CNST(d1str)) {
1368 goto give_up;
1369 } else if (CNSTG(d1str) < 0) {
1370 int t;
1371 t = d1low;
1372 d1low = d1high;
1373 d1high = t;
1374 }
1375 }
1376 }
1377 if (A_TYPEG(d2) != A_TRIPLE) {
1378 d2low = d2high = d2;
1379 } else {
1380 d2low = A_LBDG(d2);
1381 d2high = A_UPBDG(d2);
1382 if (A_STRIDEG(d2)) {
1383 int d2str;
1384 d2str = ili_symbolic(A_STRIDEG(d2));
1385 if (!IS_CNST(d2str)) {
1386 goto give_up;
1387 } else if (CNSTG(d2str) < 0) {
1388 int t;
1389 t = d2low;
1390 d2low = d2high;
1391 d2high = t;
1392 }
1393 }
1394 }
1395 if (IS_CNST(d1low) && IS_CNST(d2high) && CNSTG(d1low) > CNSTG(d2high)) {
1396 goto no_dep;
1397 }
1398 if (IS_CNST(d2low) && IS_CNST(d1high) && CNSTG(d2low) > CNSTG(d1high)) {
1399 goto no_dep;
1400 }
1401 continue;
1402 }
1403 if (RESG(d1) != RESG(d2))
1404 goto no_dep;
1405 if (IS_IRES(d1))
1406 t = ad2ili(IL_ISUB, d1, d2);
1407 else if (IS_ARES(d1))
1408 t = ad2ili(IL_ASUB, d1, d2);
1409 else
1410 interr("do_subscript: unknown base type for strides", ddinfo.common, 4);
1411 t = ili_symbolic(t);
1412 if (IS_CNST(t) && CNSTG(t) != 0L) {
1413 goto no_dep;
1414 }
1415 }
1416 }
1417
1418 DTRACE2("do_subscript: %d subs, # common loops = %d\n", nsubs, d);
1419 /* setup the equations */
1420 /* there are 2*d + ddinfo.n1 + ddinfo.n2 variables */
1421 nvars = 2 * d + ddinfo.n1 + ddinfo.n2;
1422 neqns = nsubs;
1423 for (sub = 1; sub <= nsubs; ++sub) {
1424 /* first set up coefficients for common loops */
1425 j = 0;
1426 for (i = 0; i < d; ++i) {
1427 /* subscript for A */
1428 UD[j][sub + nvars - 1] =
1429 SB_STRIDE(ddinfo.subs1 - sub)[ddinfo.n1 + d - i - 1];
1430 ++j;
1431 UD[j][sub + nvars - 1] =
1432 ad1ili(IL_INEG, SB_STRIDE(ddinfo.subs2 - sub)[ddinfo.n2 + d - i - 1]);
1433 ++j;
1434 }
1435 /* now set up coefficients for first loop */
1436 for (i = 0; i < ddinfo.n1; ++i) {
1437 UD[j][sub + nvars - 1] = SB_STRIDE(ddinfo.subs1 - sub)[ddinfo.n1 - i - 1];
1438 ++j;
1439 }
1440 /* now set up coefficients for second loop */
1441 for (i = 0; i < ddinfo.n2; ++i) {
1442 UD[j][sub + nvars - 1] =
1443 ad1ili(IL_INEG, SB_STRIDE(ddinfo.subs2 - sub)[ddinfo.n2 - i - 1]);
1444 ++j;
1445 }
1446 assert(j == nvars, "do_subscript: not enough vars", 0, 4);
1447 }
1448 /* set up identity matrix */
1449 for (i = 0; i < nvars; ++i) {
1450 for (j = 0; j < nvars; ++j) {
1451 UD[i][j] = icon0;
1452 }
1453 UD[i][i] = icon1;
1454 }
1455 /* set up rhs */
1456 for (sub = 1; sub <= nsubs; ++sub) {
1457 d1 = SB_BASES(ddinfo.subs1 - sub)[ddinfo.n1 + d];
1458 d2 = SB_BASES(ddinfo.subs2 - sub)[ddinfo.n2 + d];
1459 if (RESG(d1) != RESG(d2))
1460 goto no_dep;
1461 if (A_TYPEG(d1) == A_TRIPLE || A_DTYPEG(d2) == A_TRIPLE) {
1462 continue;
1463 }
1464 if (IS_IRES(d1))
1465 t = ad2ili(IL_ISUB, d2, d1);
1466 else if (IS_ARES(d1))
1467 t = ad2ili(IL_ASUB, d2, d1);
1468 else
1469 interr("do_subscript: unknown base type for constants", ddinfo.common, 4);
1470 C[sub - 1] = ili_symbolic(t);
1471 }
1472#if DEBUG
1473 if (DBGBIT(36, 64)) {
1474 prmat(0, 0);
1475 fprintf(gbl.dbgfil, "RHS: ");
1476 for (i = 0; i < neqns; ++i) {
1477 fprintf(gbl.dbgfil, "\t%d: ", i);
1478 prilitree(C[i]);
1479 fprintf(gbl.dbgfil, "\n");
1480 }
1481 }
1482#endif
1483
1484 /* Gaussian elimination on the matrix */
1485 bnd = neqns;
1486 if (bnd > nvars)
1487 bnd = nvars;
1488 n1 = neqns;
1489 j = 0;
1490 k1 = 0;
1491 while (k1 < bnd) {
1492 i = nvars - 1;
1493 while (i > k1) {
1494 if (UD[i][j + nvars] == icon0) {
1495 --i;
1496 continue;
1497 }
1498 /* work on rows i, i-1 */
1499 d1 = UD[i][j + nvars];
1500 d2 = UD[i - 1][j + nvars];
1501 /* check if d1 == d2 or d1 == -d2 */
1502 t = ad2ili(IL_ISUB, d1, d2);
1503 t = ili_symbolic(t);
1504 if (t == icon0) {
1505 /* d1 == d2, so sgn == 1, so q == 1 */
1506 q = icon1;
1507 } else {
1508 t = ad2ili(IL_IADD, d1, d2);
1509 t = ili_symbolic(t);
1510 if (t == icon0) {
1511 /* d1 == -d2, so sgn == -1, so q = -1 */
1512 q = ad_icon((INT)-1);
1513 } else {
1514 /* they had better both be constants */
1515 if (!IS_CNST(d1) || !IS_CNST(d2)) {
1516 if (!symbolic_divide(d2, d1, &q))
1517 goto give_up;
1518 } else {
1519 d1 = CNSTG(d1);
1520 d2 = CNSTG(d2);
1521 sgn = d1 * d2;
1522 if (d1 < 0)
1523 d1 = -d1;
1524 if (d2 < 0)
1525 d2 = -d2;
1526 q = d2 / d1;
1527 if (sgn < 0)
1528 q = -q;
1529 q = ad_icon(q);
1530 }
1531 }
1532 }
1533
1534 if (q != icon0) {
1535 /* saxpy */
1536 for (k = 0; k < nvars + neqns; ++k)
1537 UD[i - 1][k] = ili_symbolic(
1538 ad2ili(IL_ISUB, UD[i - 1][k], symbolic_mul(q, UD[i][k])));
1539 }
1540 /* interchange */
1541 for (k = 0; k < neqns + nvars; ++k) {
1542 q = UD[i - 1][k];
1543 UD[i - 1][k] = UD[i][k];
1544 UD[i][k] = q;
1545 }
1546 }
1547 if (UD[k1][j + nvars] == icon0) {
1548 /* This column is linear combination of prev. cols */
1549 ++j;
1550 --n1;
1551 if (bnd > n1)
1552 bnd = n1;
1553 continue;
1554 }
1555 ++j;
1556 ++k1;
1557 }
1558#if DEBUG
1559 if (DBGBIT(36, 64))
1560 prmat(j, i);
1561#endif
1562
1563 /* eliminate linear dependent columns (0 on the diagonal) from D */
1564 bnd = neqns;
1565 j = 0;
1566 while (j < bnd) {
1567 if (UD[j][nvars + j] != icon0) {
1568 j++;
1569 continue;
1570 }
1571 /* 0 found in D[j][j] */
1572 bnd--; /* decrement # of columns of D */
1573 for (i = j; i < bnd; i++) {
1574 for (k = 0; k < nvars; k++)
1575 /* copy D[k][(i+1)..(bnd+1)] over D[k][i..bnd] */
1576 UD[k][nvars + i] = UD[k][nvars + i + 1];
1577 C[i] = C[i + 1];
1578 }
1579 }
1580 neqns = bnd;
1581
1582 /* solve tD=C and get distances */
1583 bnd = nvars;
1584 if (bnd > neqns)
1585 bnd = neqns;
1586 /* solve TD = C */
1587 for (j = 0; j < bnd; j++) {
1588 int djj;
1589 sc = icon0;
1590 /* sc = T[0..(j-1)] * D[0..(j-1)][j] */
1591 for (i = 0; i < j; i++) {
1592 ili = ad2ili(IL_IMUL, T[i], UD[i][nvars + j]);
1593 ili = ad2ili(IL_IADD, sc, ili);
1594 sc = ili_symbolic(ili);
1595 }
1596 /* t = C[j] - sc */
1597 ili = ad2ili(IL_ISUB, C[j], sc);
1598 t = ili_symbolic(ili);
1599 djj = UD[j][j + nvars];
1600 /* T[j] = t / D[j][j] */
1601 if (djj == icon1)
1602 T[j] = t;
1603 else if (djj == iconneg1)
1604 T[j] = ad1ili(IL_INEG, t);
1605 else if (t == icon0 && (XBIT(2, 0x200) || IS_CNST(djj)))
1606 /* 0 / x == 0 */
1607 T[j] = icon0;
1608 else if (t == djj && (XBIT(2, 0x200) || IS_CNST(djj)))
1609 /* x / x == 1 */
1610 T[j] = icon1;
1611 else if (IS_CNST(djj) && IS_CNST(t)) {
1612 assert(djj != icon0, "do_subscript: linear combo not removed", 0, 4);
1613 d1 = CNSTG(djj);
1614 d2 = CNSTG(t);
1615 if (d2 % d1 != 0) {
1616 DTRACE0("do_subscript: inconsistent eqns, no dep\n");
1617 goto no_dep;
1618 }
1619 T[j] = ad_icon(d2 / d1);
1620 } else
1621 goto give_up;
1622 }
1623 /* if system is overdetermined, check for consistency */
1624 for (j = nvars; j < neqns; j++) {
1625 sc = icon0;
1626 /* sc = T[0..(nvars-1)] * D[0..(nvars-1)][j] */
1627 for (i = 0; i < nvars; i++) {
1628 ili = ad2ili(IL_IMUL, T[i], UD[i][nvars + j]);
1629 ili = ad2ili(IL_IADD, sc, ili);
1630 sc = ili_symbolic(ili);
1631 }
1632 /* t = sc - C[j] */
1633 ili = ad2ili(IL_ISUB, sc, C[j]);
1634 t = ili_symbolic(ili);
1635 if (IS_CNST(t) && t != icon0) {
1636 DTRACE0("do_subscript: inconsistent overdetermined eqns, no dep\n");
1637 goto no_dep;
1638 }
1639 }
1640 if (neqns > nvars)
1641 /* system is consistent; ignore the last neqns - nvars columns of UD */
1642 neqns = nvars;
1643
1644#if DEBUG
1645 if (DBGBIT(36, 64)) {
1646 prmat(0, 0);
1647 fprintf(gbl.dbgfil, "Solution:");
1648 for (j = 0; j < bnd; ++j) {
1649 fprintf(gbl.dbgfil, "\t%d: ", j);
1650 prilitree(T[j]);
1651 fprintf(gbl.dbgfil, "\n");
1652 }
1653 }
1654#endif
1655
1656 /* check for constant dependence distances */
1657 /* actually check only in common loops */
1658 for (j = 0; j < ddinfo.n; ++j)
1659 Dist[j] = 0;
1660 /* actually, only check in the common loops
1661 * that we are counting, that is, up to nest 'd' */
1662 for (j = 0; j < d; ++j) {
1663 for (k = neqns; k < nvars; ++k) {
1664 if (UD[k][2 * j + 1] != UD[k][2 * j])
1665 goto skip;
1666 }
1667 /* distance is constant for this loop */
1668 sc = icon0;
1669 for (k = 0; k < neqns; ++k) {
1670 sc = ili_symbolic(ad2ili(
1671 IL_IADD, sc, ad2ili(IL_IMUL, T[k], ad2ili(IL_ISUB, UD[k][2 * j + 1],
1672 UD[k][2 * j]))));
1673 }
1674 Dist[j] = sc;
1675 skip:;
1676 }
1677#if DEBUG
1678 if (DBGBIT(36, 16)) {
1679 fprintf(gbl.dbgfil, "Distance:\n");
1680 for (j = 0; j < ddinfo.n; ++j) {
1681 fprintf(gbl.dbgfil, "%d: ", j);
1682 if (Dist[j] == 0)
1683 fprintf(gbl.dbgfil, "<undef>\n");
1684 else {
1685 prilitree(Dist[j]);
1686 fprintf(gbl.dbgfil, "\n");
1687 }
1688 }
1689 }
1690#endif
1691 nfree = nvars - neqns;
1692 assert(nfree >= 0, "do_subscript: negative nfree", 0, 4);
1693
1694 /* Extended GCD test is done. Now: */
1695 /* 1. Express index variables in terms of free variables by
1696 * multiplying tU. This involves creating a matrix with one
1697 * column for each index variable and one row for each free
1698 * variable, plus one extra row to hold the constant coefficient
1699 * derived from the already solved variables in T
1700 */
1701 for (i = 0; i < nvars; ++i) {
1702 /* do const part */
1703 sc = icon0;
1704 for (j = 0; j < neqns; ++j)
1705 sc = ili_symbolic(ad2ili(IL_IADD, sc, ad2ili(IL_IMUL, T[j], UD[j][i])));
1706 TU[0][i] = sc;
1707 /* do var part */
1708 for (j = neqns + 1; j <= nvars; ++j)
1709 TU[j - neqns][i] = UD[j - 1][i];
1710 }
1711#if DEBUG
1712 if (DBGBIT(36, 64)) {
1713 fprintf(gbl.dbgfil, "Solution of index vars:\n");
1714 for (i = 0; i < nvars; ++i) {
1715 fprintf(gbl.dbgfil, "I%d= ", i);
1716 for (j = 0; j <= nfree; ++j) {
1717 fprintf(gbl.dbgfil, "\t");
1718 prilitree(TU[j][i]);
1719 if (j) {
1720 fprintf(gbl.dbgfil, "*h%d", j);
1721 }
1722 }
1723 fprintf(gbl.dbgfil, "\n");
1724 }
1725 }
1726#endif
1727 /* 2. Derive a list of upper and lower bounds from the loop limits.
1728 * Express the loop bounds in terms of the index variables, then
1729 * impose the bounds on the appropriate column, then simplify.
1730 */
1731 /* A bound is a linear combination of index variables */
1732 for (i = 0; i <= nfree; ++i) {
1733 Bound[i][0] = Bound[i][1] = 0;
1734 }
1735/* common loops */
1736#define BOUND_ADD(t, l, i) \
1737 if (bound_add(t, l, i)) \
1738 goto no_dep; \
1739 else
1740 for (i = 0; i < d; i++) {
1741 /* get the loop */
1742 lp = ddinfo.lps[ddinfo.n1 + ddinfo.n2 + d - i - 1];
1743 /* this loop derives bounds for variable 2*i and 2*i+1 */
1744 /* same as column 2*i and column 2*i+1 */
1745 /* see if we have to ignore some bounds */
1746 for (j = 0; j < i; ++j)
1747 if (SB_STRIDE(VL_LBND(lp))[j] == 0) {
1748 BTRACE1("Ignoring lower bound for loop %d\n", lp);
1749 goto skipl1;
1750 }
1751
1752 /*----- get lower bound for I */
1753 for (j = 1; j <= nvars; ++j)
1754 tmp[j] = icon0;
1755 tmp[0] = SB_BASES(VL_LBND(lp))[i]; /* const part */
1756 for (j = 0; j < i; ++j) {
1757 /* express the bound in terms of the index variables */
1758 tmp[2 * j + 1] = SB_STRIDE(VL_LBND(lp))[i - j - 1];
1759 }
1760 BOUND_ADD(tmp, FALSE, 2 * i);
1761
1762 /*----- get lower bound for J */
1763 for (j = 1; j <= nvars; ++j)
1764 tmp[j] = icon0;
1765 tmp[0] = SB_BASES(VL_LBND(lp))[i]; /* const part */
1766 for (j = 0; j < i; ++j) {
1767 /* express the bound in terms of the index variables */
1768 tmp[2 * j + 2] = SB_STRIDE(VL_LBND(lp))[i - j - 1];
1769 }
1770 BOUND_ADD(tmp, FALSE, 2 * i + 1);
1771
1772 skipl1:
1773 for (j = 0; j < i; ++j)
1774 if (SB_STRIDE(VL_UBND(lp))[j] == 0) {
1775 BTRACE1("Ignoring upper bound for loop %d\n", lp);
1776 goto skipu1;
1777 }
1778 /*----- get upper bound for I */
1779 for (j = 1; j <= nvars; ++j)
1780 tmp[j] = icon0;
1781 tmp[0] = SB_BASES(VL_UBND(lp))[i]; /* const part */
1782 for (j = 0; j < i; ++j) {
1783 /* express the bound in terms of the index variables */
1784 tmp[2 * j + 1] = SB_STRIDE(VL_UBND(lp))[i - j - 1];
1785 }
1786 BOUND_ADD(tmp, TRUE, 2 * i);
1787
1788 /*----- get upper bound for J */
1789 for (j = 1; j <= nvars; ++j)
1790 tmp[j] = icon0;
1791 tmp[0] = SB_BASES(VL_UBND(lp))[i]; /* const part */
1792 for (j = 0; j < i; ++j) {
1793 /* express the bound in terms of the index variables */
1794 tmp[2 * j + 2] = SB_STRIDE(VL_UBND(lp))[i - j - 1];
1795 }
1796 BOUND_ADD(tmp, TRUE, 2 * i + 1);
1797 skipu1:;
1798 }
1799
1800 for (i = 0; i < ddinfo.n1; ++i) {
1801 /* get the loop */
1802 lp = ddinfo.lps[ddinfo.n1 - i - 1];
1803 /* this loop derives bounds for variable 2*d+i */
1804 for (j = 0; j < i + d; ++j)
1805 if (SB_STRIDE(VL_LBND(lp))[j] == 0) {
1806 BTRACE1("Ignoring lower bound for loop %d\n", lp);
1807 goto skipl2;
1808 }
1809
1810 /*----- get lower bound for I */
1811 for (j = 1; j <= nvars; ++j)
1812 tmp[j] = icon0;
1813 tmp[0] = SB_BASES(VL_LBND(lp))[d + i];
1814 for (j = 0; j < d; ++j)
1815 tmp[2 * j + 1] = SB_STRIDE(VL_LBND(lp))[d + i - j - 1];
1816 for (j = d; j < d + i; ++j)
1817 tmp[d + j + 1] = SB_STRIDE(VL_LBND(lp))[d + i - j - 1];
1818 BOUND_ADD(tmp, FALSE, 2 * d + i);
1819
1820 skipl2:
1821 for (j = 0; j < i + d; ++j)
1822 if (SB_STRIDE(VL_UBND(lp))[j] == 0) {
1823 BTRACE1("Ignoring upper bound for loop %d\n", lp);
1824 goto skipu2;
1825 }
1826 /*----- get upper bound for I */
1827 for (j = 1; j <= nvars; ++j)
1828 tmp[j] = icon0;
1829 tmp[0] = SB_BASES(VL_UBND(lp))[d + i];
1830 for (j = 0; j < d; ++j)
1831 tmp[2 * j + 1] = SB_STRIDE(VL_UBND(lp))[d + i - j - 1];
1832 for (j = d; j < d + i; ++j)
1833 tmp[d + j + 1] = SB_STRIDE(VL_UBND(lp))[d + i - j - 1];
1834 BOUND_ADD(tmp, TRUE, 2 * d + i);
1835
1836 skipu2:;
1837 }
1838 for (i = 0; i < ddinfo.n2; ++i) {
1839 /* get the loop */
1840 lp = ddinfo.lps[ddinfo.n1 + ddinfo.n2 - i - 1];
1841 /* this loop derives bounds for variable 2*d+ddinfo.n1+i */
1842
1843 for (j = 0; j < i + d; ++j)
1844 if (SB_STRIDE(VL_LBND(lp))[j] == 0) {
1845 BTRACE1("Ignoring lower bound for loop %d\n", lp);
1846 goto skipl3;
1847 }
1848 /*----- get lower bound for J */
1849 for (j = 1; j <= nvars; ++j)
1850 tmp[j] = icon0;
1851 tmp[0] = SB_BASES(VL_LBND(lp))[d + i];
1852 for (j = 0; j < d; ++j)
1853 tmp[2 * j + 2] = SB_STRIDE(VL_LBND(lp))[d + i - j - 1];
1854 for (j = d; j < d + i; ++j)
1855 tmp[d + ddinfo.n1 + j + 1] = SB_STRIDE(VL_LBND(lp))[d + i - j - 1];
1856 BOUND_ADD(tmp, FALSE, 2 * d + ddinfo.n1 + i);
1857
1858 skipl3:
1859 for (j = 0; j < i + d; ++j)
1860 if (SB_STRIDE(VL_UBND(lp))[j] == 0) {
1861 BTRACE1("Ignoring upper bound for loop %d\n", lp);
1862 goto skipu3;
1863 }
1864 /*----- get upper bound for J */
1865 for (j = 1; j <= nvars; ++j)
1866 tmp[j] = icon0;
1867 tmp[0] = SB_BASES(VL_UBND(lp))[d + i];
1868 for (j = 0; j < d; ++j)
1869 tmp[2 * j + 1] = SB_STRIDE(VL_UBND(lp))[d + i - j - 1];
1870 for (j = d; j < d + i; ++j)
1871 tmp[d + ddinfo.n1 + j + 1] = SB_STRIDE(VL_UBND(lp))[d + i - j - 1];
1872 BOUND_ADD(tmp, TRUE, 2 * d + ddinfo.n1 + i);
1873
1874 skipu3:;
1875 }
1876
1877 /* 4. Direction vector hierarchy if dependent */
1878 vec = dirv_fulldep(ddinfo.n) & ~DIRV_ALLEQ;
1879 vec1 = 0;
1880 for (i = d; i < ddinfo.n; ++i)
1881 DIRV_ENTRYP(vec1, i, DIRV_STAR);
1882 hierarchy(vec, d, vec1);
1883 return;
1884
1885give_up:
1886 if (VP_DEPCHK(ddinfo.pragmas))
1887 add_dep(dirv_fulldep(ddinfo.n));
1888no_dep:
1889 return;
1890}

1892static void
1893hierarchy(DIRVEC dir, int lev, DIRVEC veco)
1894{
1895 int i;
1896 int dir2;
1897 LOGICAL top;
1898
1899 DTRACE2("hierarchy: dir 0x%lx -- %s\n", dir, dirv_print(dir));
1900
1901 for (i = 0; i < lev; ++i)
1902 if (DIRV_ENTRYG(dir, i) != DIRV_STAR)
1903 break;
1904 if (i == lev)
1905 /* all * */
1906 top = TRUE;
1907 else
1908 top = FALSE;
1909 for (i = 0; i < lev; ++i)
1910 if (DIRV_ENTRYG(dir, i) == DIRV_STAR)
1911 goto test;
1912 goto bottom;
1913
1914test:
1915 /*
1916 * test dependence at this level; if no dependence, don't need to go any
1917 * further: if (no dependence with direction vector dir) return 0;
1918 */
1919 if (top) {
1920 if (check_bounds() == TRUE)
1921 return;
1922 BCOPY(SaveBound, Bound, BOUND *, BOUND_LEN);
1923 } else {
1924 if (check_new_bound(dir, lev) == TRUE)
1925 return;
1926 }
1927
1928 /* '*' entry, refine it further */
1929 dir2 = dir;
1930 DIRV_ENTRYC(dir2, i);
1931 DIRV_ENTRYP(dir2, i, DIRV_LT);
1932 hierarchy(dir2, lev, veco);
1933
1934 dir2 = dir;
1935 DIRV_ENTRYC(dir2, i);
1936 DIRV_ENTRYP(dir2, i, DIRV_EQ);
1937 hierarchy(dir2, lev, veco);
1938
1939 dir2 = dir;
1940 DIRV_ENTRYC(dir2, i);
1941 DIRV_ENTRYP(dir2, i, DIRV_GT);
1942 hierarchy(dir2, lev, veco);
1943
1944 return;
1945
1946bottom:
1947 /*
1948 * we're at the bottom level. Test dependence with this direction vector
1949 * & return 0 or this direction vector. In addition, need to set the
1950 * all-equals bit if this direction vector is all equals
1951 */
1952 ddinfo.unknown = 0;
1953 if (check_new_bound(dir, lev) == TRUE)
1954 return;
1955 if (ddinfo.unknown && !VP_DEPCHK(ddinfo.pragmas))
1956 return;
1957 for (i = 0; i < lev; ++i)
1958 if (DIRV_ENTRYG(dir, i) != DIRV_EQ) {
1959 add_dep(veco | dir);
1960 return;
1961 }
1962 add_dep(veco | dir | DIRV_ALLEQ);
1963}

2037/*
2038 * Return the direction vector that corresponds to the input
2039 * direction vector dir under the mapping m. If m is NULL, the
2040 * trivial mapping is used. A mapping is a permutation of the integers
2041 * 0..nest-1 representing the order of the loops as permuted from
2042 * their original order, from inner to outer.
2043 * The trivial mapping is thus nest-1, ..., 0
2044 */
2045static DIRVEC
2046dirv_permute(DIRVEC dir, int *m, int nest)
2047{
2048 int i, j, k;
2049 DIRVEC e;
2050 int seen_lt;
2051 DIRVEC rdir;
2052 int base;
2053
2054 for (i = 0; DIRV_ENTRYG(dir, i) != 0 && i < MAX_LOOPS; ++i)
2055 ;
2056 base = nest - i;
2057#if DEBUG
2058 assert(m == 0 || base >= 0, "dirv_permute: nest is wrong", nest, 4);
2059#endif
2060 seen_lt = 0;
2061 rdir = 0;
2062 for (j = 0; j < i; ++j) {
2063 k = m ? m[j + base] : i - j - 1;
2064 e = DIRV_ENTRYG(dir, k);
2065 rdir <<= DIRV_ENTSIZ;
2066 if (!seen_lt)
2067 rdir |= (e & ~DIRV_GT);
2068 else
2069 rdir |= e;
2070 if (e & DIRV_LT)
2071 seen_lt = 1;
2072 }
2073 if (rdir == 0)
2074 return 0;
2075 /* check alleq case */
2076 for (j = 0; j < i; ++j) {
2077 if (DIRV_ENTRYG(rdir, j) != DIRV_EQ)
2078 goto skip;
2079 }
2080 /* all are equal */
2081 if (!(dir & DIRV_ALLEQ))
2082 return 0;
2083
2084skip:
2085 rdir |= DIRV_INFOPART(dir);
2086 return rdir;
2087}

2089static void
2090dovec(DIRVEC dir, DIRVEC dir1, int pos, int ig, void (*f)(DIRVEC), int alleq,
2091 int seenlt)
2092{
2093 DIRVEC dir2;
2094 DIRVEC e;
2095
2096 if (pos < 0) {
2097 if (dir1 != 0 && !alleq)
2098 (*f)(dir1);
2099 else if (dir1 != 0 && alleq && (dir & DIRV_ALLEQ))
2100 (*f)(dir1 | DIRV_ALLEQ);
2101 return;
2102 }
2103 e = DIRV_ENTRYG(dir, pos);
2104 if (e & DIRV_LT) {
2105 dir2 = dir1 | (DIRV_LT << DIRV_ENTSIZ * pos);
2106 if (pos <= ig)
2107 dovec(dir, dir2, pos - 1, ig, f, 0, 1);
2108 }
2109 if (e & DIRV_EQ) {
2110 dir2 = dir1 | (DIRV_EQ << DIRV_ENTSIZ * pos);
2111 dovec(dir, dir2, pos - 1, ig, f, alleq, seenlt);
2112 }
2113 if (e & DIRV_RD) {
2114 dir2 = dir1 | (DIRV_RD << DIRV_ENTSIZ * pos);
2115 dovec(dir, dir2, pos - 1, ig, f, 0, seenlt);
2116 }
2117 if (e & DIRV_GT) {
2118 dir2 = dir1 | (DIRV_GT << DIRV_ENTSIZ * pos);
2119 if (seenlt)
2120 dovec(dir, dir2, pos - 1, ig, f, 0, 1);
2121 }
2122}

2142/* Initialize subscript struct sub with information from expression
2143 * ast under multiplier astmpyr. The loop index of the ith outer loop
2144 * is the FORALL-index in the ith outer triplet within astliTriples.
2145 * astmpyr is invariant with respect to the FORALL-indices.
2146 * Return TRUE if ast is a linear expression. */
2147static LOGICAL
2148mkSub(int astliTriples, int sub, int astmpyr, int ast)
2149{
2150 int i;
2151 int astli;
2152 int aststride, astbase;
2153 LOGICAL bLinear;
2154
2155 switch (A_TYPEG(ast)) {
2156 case A_CONV:
2157 /*
2158 * when the convert case did not exist, mkSub() returned false
2159 * indicating the subscript expression is non-linear. The convert
2160 * could appear when mixing integer types or the use of -Mlarge_arrays
2161 * and default integer*4. For specaccel palm -Mlarge_arrays, a false
2162 * dependency was returned for some array assignment in an openacc
2163 * kernel; the assignment was not parallelized, but the acc CG
2164 * generated incorrect scalar code.
2165 * We could try pushing the convert into its operand; but all I'm
2166 * going to do is check if its operand is linear, and if so, I will
2167 * treat it as an ID (a single term)
2168 */
2169 if (!is_linear(A_LOPG(ast)))
2170 return FALSE;
2171 FLANG_FALLTHROUGH;
2172 case A_ID:
2173 i = 0;
2174 for (astli = astliTriples; astli; astli = ASTLI_NEXT(astli), i++)
2175 if (A_SPTRG(ast) == ASTLI_SPTR(astli)) {
2176 aststride = A_STRIDEG(ASTLI_TRIPLE(astli));
2177 if (!aststride)
2178 aststride = astb.bnd.one;
2179 aststride = mk_binop(OP_MUL, aststride, astmpyr, astb.bnd.dtype);
2180 aststride =
2181 mk_binop(OP_ADD, aststride, SB_STRIDE(sub)[i], astb.bnd.dtype);
2182 SB_STRIDE(sub)[i] = aststride;
2183 astbase = A_LBDG(ASTLI_TRIPLE(astli));
2184 if (A_DTYPEG(astbase) != astb.bnd.dtype) {
2185 astbase = mk_convert(astbase, astb.bnd.dtype);
2186 }
2187 astbase = mk_binop(OP_MUL, astbase, astmpyr, astb.bnd.dtype);
2188 astbase = mk_binop(OP_ADD, astbase, SB_BASE(sub), astb.bnd.dtype);
2189 SB_BASE(sub) = astbase;
2190 return TRUE;
2191 }
2192 astbase = mk_binop(OP_MUL, astmpyr, ast, astb.bnd.dtype);
2193 SB_BASE(sub) = mk_binop(OP_ADD, astbase, SB_BASE(sub), astb.bnd.dtype);
2194 return TRUE;
2195 case A_CNST:
2196 astbase = mk_binop(OP_MUL, astmpyr, ast, astb.bnd.dtype);
2197 SB_BASE(sub) = mk_binop(OP_ADD, astbase, SB_BASE(sub), astb.bnd.dtype);
2198 return TRUE;
2199 case A_UNOP:
2200 if (A_OPTYPEG(ast) != OP_SUB)
2201 return FALSE;
2202 astmpyr = mk_unop(OP_SUB, astmpyr, astb.bnd.dtype);
2203 bLinear = mkSub(astliTriples, sub, astmpyr, A_LOPG(ast));
2204 return bLinear;
2205 case A_BINOP:
2206 switch (A_OPTYPEG(ast)) {
2207 case OP_ADD:
2208 bLinear = (mkSub(astliTriples, sub, astmpyr, A_LOPG(ast)) &&
2209 mkSub(astliTriples, sub, astmpyr, A_ROPG(ast)));
2210 return bLinear;
2211 case OP_SUB:
2212 bLinear = mkSub(astliTriples, sub, astmpyr, A_LOPG(ast));
2213 if (!bLinear)
2214 return FALSE;
2215 astmpyr = mk_unop(OP_SUB, astmpyr, astb.bnd.dtype);
2216 bLinear = mkSub(astliTriples, sub, astmpyr, A_ROPG(ast));
2217 return bLinear;
2218 case OP_MUL:
2219 astmpyr = mk_binop(OP_MUL, A_LOPG(ast), astmpyr, astb.bnd.dtype);
2220 bLinear = mkSub(astliTriples, sub, astmpyr, A_ROPG(ast));
2221 if (!bLinear)
2222 return FALSE;
2223 astmpyr = mk_binop(OP_MUL, A_ROPG(ast), astmpyr, astb.bnd.dtype);
2224 bLinear = mkSub(astliTriples, sub, astmpyr, A_LOPG(ast));
2225 return bLinear;
2226 default:
2227 return FALSE;
2228 }
2229 default:
2230 return FALSE;
2231 }

2459 BZERO(&hlv.mrbase[mrSink], MEMREF, 1);
2460 MR_ILI(mrSink) = astArrSink;
2461 MR_TYPE(mrSink) = 'l';
2462
2463 mrSrc = hlv.mravail++;
2464 NEED(hlv.mravail, hlv.mrbase, MEMREF, hlv.mrsize, hlv.mrsize + 100);
2465 BZERO(&hlv.mrbase[mrSrc], MEMREF, 1);
2466 MR_ILI(mrSrc) = astArrSrc;
2467 MR_TYPE(mrSrc) = 's';
2468
2469 /* Create subscript structures for mrSink. */
2470 subStart = hlv.subavail;
2471 MR_SUBST(mrSink) = subStart;
2472 MR_SUBCNT(mrSink) = 0;
2473 nSrc = fill_subscripts(astArrSink, mrSink, subStart, ntriples, astliTriples);
2474 if (nSrc < 0) {
2475 fwd.bFwd = TRUE;
2476 goto nonlinear;
2477 }
2478
2479 /* Create subscript structures for mrSrc. */
2480 subStart = hlv.subavail;
2481 MR_SUBST(mrSrc) = subStart;
2482 MR_SUBCNT(mrSrc) = 0;
2483 nSink = fill_subscripts(astArrSrc, mrSrc, subStart, ntriples, astliTriples);
2484 if (nSink < 0) {
2485 fwd.bFwd = TRUE;
2486 goto nonlinear;
2487 }
2488
2489 /* Fill in ddinfo for dependence analysis. */
2490 ddinfo.mr1 = mrSrc;
2491 ddinfo.mr2 = mrSink;
2492 ddinfo.subs1 = MR_SUBST(mrSrc) + MR_SUBCNT(mrSrc);
2493 ddinfo.subs2 = MR_SUBST(mrSink) + MR_SUBCNT(mrSink);
2494
2495 ddinfo.n = ntriples; /* the array references are in same loop nest. */
2496 ddinfo.n1 = ddinfo.n2 = 0;
2497
2498 for (i = 0; i < ntriples; i++)
2499 ddinfo.lps[i] = (lpOuter + ntriples - 1) - i;
2500
2501 ddinfo.common = ddinfo.lps[0];
2502 ddinfo.outer_loop = lpOuter;
2503
2504 VP_DEPCHK(ddinfo.pragmas) = TRUE;
2505
2506 ddinfo.dvlist = NULL;
2507
2508 /* Do dependence analysis. */
2509 nsubs = nsubsSrc < nsubsSink ? nsubsSrc : nsubsSink;
2510 do_subscript(nsubs);
2511
2512 if (bSinkAfterSrc < 0) {
2513 DV *pdv;
2514 /* ANY loop-carried dependence is a forward dependence */
2515 fwd.bFwd = FALSE;
2516 for (pdv = ddinfo.dvlist; pdv; pdv = pdv->next) {
2517 DIRVEC dv;
2518 int i;
2519 dv = pdv->vec;
2520 for (i = 0; i < MAX_LOOPS; ++i) {
2521 int t;
2522 t = DIRV_ENTRYG(dv, i);
2523 if (t == 0)
2524 break; /* done */
2525 if (t & (DIRV_GT | DIRV_LT)) {
2526 /* found a loop-carried dependence */
2527 fwd.bFwd = TRUE;
2528 break;
2529 }
2530 }
2531 }
2532 } else {
2533 /* Generate direction vectors. */
2534 DIRVEC dvexo;
2535 DDEDGE *dd;
2536 DV *pdv;
2537 dvexo = dirv_exo(ntriples, bSinkAfterSrc);
2538 for (pdv = ddinfo.dvlist; pdv; pdv = pdv->next) {
2539 DIRVEC dv;
2540 dv = pdv->vec & dvexo;
2541 if (!dirv_chkzero(dv, ntriples))
2542 dd_edge(mrSrc, mrSink, dv);
2543 }
2544
2545 /* Determine if there is a loop-carried dependence. */
2546 fwd.bFwd = FALSE;
2547 fwd.nloops = ntriples;
2548 for (dd = MR_SUCC(mrSrc); dd; dd = DD_NEXT(dd)) {
2549 dirv_gen(dd->dirvec, 0, ntriples, ntriples, fwd_func);
2550 if (fwd.bFwd)
2551 break;
2552 }
2553 }
2554
2555nonlinear:
2556 /* Clean up allocated memory. */
2557 cln_visit();
2558 FREE(hlv.lpbase);
2559 FREE(hlv.mrbase);
2560 FREE(hlv.subbase);
2561 freearea(HLV_AREA1);
2562
2563 return fwd.bFwd;
2564}

2686/* Extract constant factors from the varfact member of arithmetic term atp.
2687 * Rules are:
2688 * r(<c,d>) = <c*d,1>, where d is a constant.
2689 * r(<c,x*y) = <c*d*e,x'*y'>, where:
2690 * <d,x'> = r(<1,x>)
2691 * <e,y'> = r(<1,y>).
2692 * r(<c,-x>) = <-c*d,y>, where
2693 * <d,y> = r(<1,x>).
2694 * r(<c,x>) = <c,x>, otherwise.
2695 */
2696static void
2697refactor(ARITH_LIST atp)
2698{
2699 ARITH_TERM at1, at2;
2700 int opc;
2701 int icon1 = ad_icon(1L);
2702
2703 at1.next = at2.next = NULL;
2704
2705 if (IS_CNST(atp->varfact)) {
2706 atp->confact *= CNSTG(atp->varfact);
2707 atp->varfact = icon1;
2708 return;
2709 }
2710
2711 opc = ILI_OPC(atp->varfact);
2712 switch (opc) {
2713 case IL_IMUL:
2714 at1.confact = at2.confact = 1L;
2715 at1.varfact = ILI_OPND(atp->varfact, 1);
2716 at2.varfact = ILI_OPND(atp->varfact, 2);
2717 refactor(&at1);
2718 refactor(&at2);
2719 atp->confact *= at1.confact * at2.confact;
2720 if (at1.varfact == icon1)
2721 atp->varfact = at2.varfact;
2722 else if (at2.varfact == icon1)
2723 atp->varfact = at1.varfact;
2724 else
2725 atp->varfact = ad2ili(IL_IMUL, at1.varfact, at2.varfact);
2726 break;
2727 case IL_INEG:
2728 at1.confact = 1L;
2729 at1.varfact = ILI_OPND(atp->varfact, 1);
2730 refactor(&at1);
2731 atp->confact *= -at1.confact;
2732 atp->varfact = at1.varfact;
2733 break;
2734 default:
2735 break;
2736 }
2737}

2765/* Return an arithmetic term list whose sum is equivalent to mpyr * ili,
2766 * where mpyr is distributed across terms in the resulting list. All confact
2767 * fields in the term list are set to 1.
2768 * Rules are:
2769 * distrib(x*y, m) = [distrib(x, yi) | yi <- distrib(y, m)].
2770 * distrib(x+y, m) = distrib(x, m) apnd distrib(y, m).
2771 * distrib(x-y, m) = distrib(x, m) apnd distrib(y, -m).
2772 * distrib(-x, m) = distrib(x, -m).
2773 * distrib(x, m + n) = distrib(m, x) apnd distrib(n, x); if x is a load.
2774 * distrib(x, m) = [x * m]; otherwise.
2775 */
2776static ARITH_LIST
2777distrib(int ili, int mpyr)
2778{
2779 int ili2, ilitmp;
2780 int opc, opc1;
2781 ARITH_LIST l1, l2, l;
2782
2783 assert(mpyr, "distrib: invalid multiplier", ili, 4);
2784 assert(ili, "distrib: invalid ili", ili, 4);
2785
2786 opc = ILI_OPC(ili);
2787 switch (opc) {
2788 case IL_INEG:
2789 ilitmp = ad1ili(IL_INEG, mpyr);
2790 l = distrib(ILI_OPND(ili, 1), ilitmp);
2791 return l;
2792 case IL_IADD:
2793 l1 = distrib(ILI_OPND(ili, 1), mpyr);
2794 l2 = distrib(ILI_OPND(ili, 2), mpyr);
2795 l = apnd(l1, l2);
2796 return l;
2797 case IL_ISUB:
2798 l1 = distrib(ILI_OPND(ili, 1), mpyr);
2799 ilitmp = ad1ili(IL_INEG, mpyr);
2800 l2 = distrib(ILI_OPND(ili, 2), ilitmp);
2801 l = apnd(l1, l2);
2802 return l;
2803 case IL_IMUL:
2804 l2 = distrib(ILI_OPND(ili, 2), mpyr);
2805 l = NULL;
2806 for (; l2; l2 = l2->next) {
2807 assert(l2->confact == 1L, "distrib: non-unit constant", ili, 4);
2808 l1 = distrib(ILI_OPND(ili, 1), l2->varfact);
2809 l = apnd(l1, l);
2810 }
2811 return l;
2812 case IL_IDIV:
2813 case IL_MOD:
2814 ilitmp = ili_symbolic(ILI_OPND(ili, 1));
2815 if (ilitmp != ili) {
2816 ilitmp = ad2ili(opc, ilitmp, ILI_OPND(ili, 2));
2817 ilitmp = ad2ili(IL_IMUL, ilitmp, mpyr);
2818 l = (ARITH_LIST)getitem(HLV_AREA1, sizeof(ARITH_TERM));
2819 l->confact = 1;
2820 l->varfact = ilitmp;
2821 l->next = NULL;
2822 return l;
2823 }
2824 FLANG_FALLTHROUGH;
2825 default:
2826 opc1 = ILI_OPC(mpyr);
2827 if (opc1 == IL_IADD) {
2828 l1 = distrib(ILI_OPND(mpyr, 1), ili);
2829 l2 = distrib(ILI_OPND(mpyr, 2), ili);
2830 l = apnd(l1, l2);
2831 return l;
2832 }
2833 if (opc1 == IL_ISUB) {
2834 l1 = distrib(ILI_OPND(mpyr, 1), ili);
2835 ili2 = ad1ili(IL_INEG, ILI_OPND(mpyr, 2));
2836 l2 = distrib(ili2, ili);
2837 l = apnd(l1, l2);
2838 return l;
2839 }
2840 if (IS_IRES(ili))
2841 ilitmp = ad2ili(IL_IMUL, ili, mpyr);
2842 else
2843 ilitmp = ili;
2844 l = (ARITH_LIST)getitem(HLV_AREA1, sizeof(ARITH_TERM));
2845 l->confact = 1;
2846 l->varfact = ilitmp;
2847 l->next = NULL;
2848 return l;
2849 }
2850}

2852/* Return TRUE if arithmetic term list lst contains a term whose varfact
2853 * member is equivalent to trm.
2854 * Rules are:
2855 * c_c_f([], x) = FALSE.
2856 * c_c_f([<c,x>|l], x) = TRUE.
2857 * c_c_f([x|l], y) = c_c_f(l, y), otherwise.

2871/* Sum the confact members of all terms in arithmetic term list lst
2872 * whose varfact members are all equal to trm. Return the sum.
2873 * Rules are:
2874 * c_t([], x) = 0.
2875 * c_t([<c,x>| l], x) = c + c_t(l, x).
2876 * c_t([x|l], y) = c_t(l, y); otherwise.
2877 */
2878static long
2879combine_terms(ARITH_LIST lst, int trm)
2880{
2881 ARITH_LIST l;
2882 long sum = 0L;
2883
2884 for (l = lst; l; l = l->next)
2885 if (l->varfact == trm)
2886 sum += l->confact;
2887 return sum;
2888}

2890/* Remove all arithmetic terms from arithmetic term list lst whose varfact
2891 * members are equivalent to trm.
2892 * Rules are:
2893 * r_t([], x) = [].
2894 * r_t([<c,x>|l], x) = l.
2895 * r_t([x|l], y) = [x | r_t(l,y)], otherwise
2896 */
2897static ARITH_LIST
2898remove_terms(ARITH_LIST lst, int trm)
2899{
2900 ARITH_LIST lcur, lprev, lhead;
2901
2902 assert(trm, "remove_terms: invalid term", trm, 4);
2903
2904 lhead = lst;
2905 lprev = NULL;
2906 for (lcur = lst; lcur; lcur = lcur->next)
2907 if (lcur->varfact == trm)
2908 /* remove the current term from the list */
2909 if (lprev == NULL)
2910 lhead = lcur->next;
2911 else
2912 lprev->next = lcur->next;
2913 else
2914 lprev = lcur;
2915 return lhead;
2916}
2917
2918/* Combine terms from arithmetic term list lst that have common factors, and
2919 * return the result.
2920 * Rules are:
2921 * elim([]) = [].
2922 * elim([<c,x>|l]) = if c_c_f(l, x) then
2923 * [<c_t(l, x)+c,x> | elim(r_t(l,x))]
2924 * else
2925 * [<c,x> | elim(l)].
2926 * When a reduction takes place, set global rdc to TRUE.
2927 */
2928static ARITH_LIST
2929elim(ARITH_LIST lst)
2930{
2931 ARITH_LIST lprev, lcur, lhead;
2932 long sum;
2933
2934 lhead = lst;
2935 lprev = NULL;
2936 for (lcur = lst; lcur; lcur = lcur->next) {
2937 if (contains_common_factor(lcur->next, lcur->varfact)) {
2938 sum = combine_terms(lcur, lcur->varfact);
2939 lcur->next = remove_terms(lcur->next, lcur->varfact);
2940 if (sum == 0L)
2941 if (lprev)
2942 lprev->next = lcur->next;
2943 else
2944 lhead = lcur->next;
2945 else {
2946 lcur->confact = sum;
2947 lprev = lcur;
2948 }
2949 rdc = TRUE;
2950 } else
2951 lprev = lcur;
2952 }
2953 return lhead;
2954}
