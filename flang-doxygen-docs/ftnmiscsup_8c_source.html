46

119/* function: Ftn_secnds:
120 *
121 * Returns the number of real time seconds since midnight minus the supplied
122 * value
123 */
124/* ***********************************************************************/
125
126float
127Ftn_secnds(float x)
128{
129 static int called = 0;
130 static int diffs;
131 int i;
132 time_t ltime;
133 struct tm *lt;
134 float f;
135
136 ltime = time(0);
137 if (called == 0) {
138 called = 1; /* first time called */
139 /*
140 * compute value to subtract from time(0) to give seconds since
141 * midnight
142 */
143 lt = localtime(&ltime);
144 i = lt->tm_sec + (60 * lt->tm_min) + (3600 * lt->tm_hour);
145 diffs = ltime - i;
146 }
147 f = (float)(ltime - diffs);
148 return (f - x);
149}

226/* function: Ftn_ran (Ftn_dran - double precision version of ran):
227 *
228 * Ftn_ran returns VMS-compatible random number sequence
229 */
230/* ***********************************************************************/
231float
232Ftn_ran(unsigned *seed)
233{
234 static unsigned tm24 = 0x33800000; /* 2 ** -24 */
235 unsigned u;
236
237 u = *seed = *seed * 69069 + 1;
238 /*
239 * extract higher-order 24 bits of new seed, convert to float, and
240 * scale converted value so that 0.0 <= value < 1.0.
241 */
242 return (((float)(u >> 8)) * *(float *)&tm24);
243}
