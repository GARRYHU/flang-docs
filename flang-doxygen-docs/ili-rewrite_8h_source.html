11/*
12 * A general postorder (bottom-up) ILI rewriting callback framework.
13 *
14 * Every statement (ILT) in the program is visited once in some arbitrary
15 * block order. For each statement, the ILIs are visited in postorder
16 * (leaves before interior nodes). Each ILI, as it is visited, is passed
17 * to a callback, which can either return it unchanged or replace it
18 * with a new ILI.
19 *
20 * If the callback replaces an ILI, the outer expression that linked to it
21 * will have that ILI link operand provisionally replaced before it is
22 * passed to the callback in its turn. In other words, the callback routine
23 * may safely assume that every ILI that is presented to the callback
24 * already reflects any changes that it may have made to its subexpressions.
25 *
26 * This rewriting framework automatically detects when this rewriting
27 * of operands prior to the invocation of the callback leads to improvement
28 * of the program. After the operands of an ILI are processed, if any
29 * have changed, the ILI is reconstructed by calling addili(), and the
30 * index that addili() returns is examined to see whether addili() itself
31 * rewrote the expression.
32 *
33 * The callback is supplied with a pointer to a context description
34 * structure that includes the ILI index, its statement context, its original
35 * form, and other information.
36 *
37 * Be advised: this framework knows about the subtle semantics of JSR
38 * and CSE operations, and will "do the right thing" to their operands
39 * in order to propagate any other ILI replacements returned by the
40 * visitation callback. However, the framework is unable to prevent
41 * the callback from messing up the CSE semantics.
42 *
43 * Be advised: the framework does not automatically adjust reference
44 * counts on labels. It originall did so, but it proved to be impossible
45 * to distinguish iliutil.c's changes from callback changes, and
46 * iliutil manages RFCNT fields explicitly.
47 */

53/* Callback function type for ILI rewriting.
54 * Declare callbacks for visit_ilis() like this:
55 *
56 * int my_callback(const ILI_coordinates *at) {
57 * struct my_context *context = at->context;
58 * if (my_transformation)
59 * return new_ili_index;
60 * return at->ili; // no change
61 * }
62 */
63typedef int (*ILI_visitor)(const struct ILI_coordinates *at);
64
65typedef bool (*ILI_tree_scan_visitor)(void *visitor_context, int ili);

67/**
68 \brief Visitation context structure passed to callbacks.
69 */
70typedef struct ILI_coordinates {
71 ILI_visitor visitor;
72 void *context; /**< callback's own state */
73 int original_ili; /**< preorder value, before recursion into operands */
74 int ili; /**< after recursion into operands (might be != original_ili) */
75 int bih;
76 int ilt; /**< position of statement containing this ILI instance */
77 /** null if ILI is root expression of ILT */
78 const struct ILI_coordinates *parent;
79 int parent_opnd; /**< ILI_OPND(parent->ili,parent_opnd) links here */
80 /** this ILI improved when operands updated */
81 bool this_ili_improved;
82 bool has_cse; /**< operand trees contain JSR and/or CSE */
83} ILI_coordinates;

93/**
94 \brief ILI rewriting driver. Returns TRUE if any change occurs.
95
96 The traversal is in block sequence order, with the ILTs being visited in
97 forward order within each block.
98
99 The void *visitor_context is passed to the visitation callback as
100 at->context.
101
102 Set the "context_insensitive" flag to TRUE for faster processing when the
103 rewriting visitor is known to not be sensitive to the context (i.e., it
104 always maps ILI 'x' to the same 'y' in any statement).
105 */
106bool visit_ilis(ILI_visitor visitor, void *visitor_context,
107 bool context_insensitive);
