118/** \brief Fortran front-end main entry
119 \param argc number of command-line arguments
120 \pram argv array of command-line argument strings
121*/
122int
123main(int argc, char *argv[])
124{
125 int savescope, savecurrmod = 0;
126 get_rutime();
127 init(argc, argv); /* initialize */
128 if (gbl.fn == NULL)
129 gbl.fn = gbl.src_file;
130
131#if DEBUG
132 if (debugfunconly > 0)
133 dodebug = 0;
134#endif
135
136 saveoptflag = flg.opt;
137 savevectflag = flg.vect;
138 savex8flag = flg.x[8];
139 saverecursive = flg.recursive;
140
141 if (IPA_INHERIT_ENABLED && (flg.opt >= 2 || IPA_COLLECTION_ENABLED)) {
142 ipa_init();
143 }
144
145 gbl.findex = addfile(gbl.fn, NULL, 0, 0, 0, 1, 0);
146
147 if (ipa_export_file && ipa_import_mode) {
148 ipa_import_open(ipa_export_file, ipa_import_offset);
149 }
150 if (IPA_ENABLED && ipa_export_file && !ipa_import_mode) {
151 /* export the program unit for IPA recompilation */
152 ipa_export_open(ipa_export_file);
153 }
154
155 if (gbl.srcfil == NULL) {
156 if (!ipa_import_mode) {
157 finish();
158 }
159 }
160 do { /* loop once for each user program unit */
161#if DEBUG
162 if (debugfunconly > 0) {
163 if (debugfunconly == gbl.func_count + 1)
164 dodebug = 1;
165 else
166 dodebug = 0;
167 }
168#endif
169 reinit();
170 errini();
171 if (ipa_export_file && ipa_import_mode && gbl.func_count == 0) {
172 ipa_import_highpoint();
173 }
174 if (IPA_ENABLED && ipa_export_file && !ipa_import_mode &&
175 gbl.func_count == 0) {
176 ipa_export_highpoint();
177 }
178 xtimes[0] += get_rutime();
179 if (ipa_export_file && ipa_import_mode) {
180 ipa_import();
181 if (gbl.eof_flag & 2)
182 break;
183 } else {
184 TR(DNAME " PARSER begins\n")
185 parser(); /* parse and do semantic analysis */
186 set_tag();
187 }
188 gbl.func_count++;
189 ccff_open_unit_f90();
190 if (gbl.internal <= 1) {
191 gbl.outersub = 0;
192 gbl.outerentries = 0;
193 }
194 savescope = stb.curr_scope;
195 if (gbl.currsub) {
196 if (SCOPEG(gbl.currsub)) {
197 stb.curr_scope = SCOPEG(gbl.currsub);
198 if (STYPEG(stb.curr_scope) != ST_ALIAS ||
199 SYMLKG(stb.curr_scope) != gbl.currsub) {
200 stb.curr_scope = gbl.currsub;
201 }
202 } else {
203 stb.curr_scope = gbl.currsub;
204 }
205 }
206 TR1("- after semant");
207 xtimes[1] += get_rutime();
208 DUMP("parser");
209 if (gbl.rutype == RU_BDATA) {
210 /* a module? */
211 if (has_cuda_data())
212 has_accel_code = TRUE;
213 }
214 if (gbl.currsub == 0) {
215 if (IPA_ENABLED && ipa_export_file && !ipa_import_mode) {
216 /* export the program unit for IPA recompilation */
217 ipa_export_endmodule();
218 }
219 continue; /* end of a module */
220 }
221 if (CUDAG(gbl.currsub) & (CUDA_GLOBAL | CUDA_DEVICE)) {
222 /* remember that this routine needs a constructor */
223 has_accel_code = TRUE;
224 }
225 savecurrmod = gbl.currmod;
226#if DEBUG
227 if (DBGBIT(5, 1))
228 symdmp(gbl.dbgfil, DBGBIT(5, 8));
229 if (DBGBIT(5, 2))
230 dump_std();
231 if (DBGBIT(5, 16))
232 dmp_dtype();
233#endif
234 if (IPA_ENABLED && ipa_export_file && !ipa_import_mode) {
235 /* export the program unit for IPA recompilation */
236 ipa_export();
237 }
238
239#if DEBUG
240 if (DBGBIT(4, 256))
241 dump_ast();
242#endif
243 if (IPA_INHERIT_ENABLED && gbl.rutype != RU_BDATA) {
244 ipa_startfunc(gbl.currsub);
245 ipa_header1(gbl.currsub);
246 ipa_header2(gbl.currsub);
247 }
248 postprocessing = FALSE;
249 if (gbl.maxsev < 3 && !DBGBIT(2, 4)) {
250 postprocessing = TRUE;
251
252 flg.ipa |= 0x20;
253 if (XBIT(57, 0x2000) && !flg.inliner) {
254 /* try to eliminate unused common blocks here */
255 eliminate_unused_variables(1);
256 DUMP("staticunused");
257 }
258 /* by default, generate data initialization inline. */
259 if (gbl.rutype != RU_BDATA) {
260 direct_rou_load(gbl.currsub);
261 if (flg.opt > 1 && !XBIT(47, 0x40000000)) {
262 if (sem.stats.allocs > 800 && sem.stats.nodes > 1000) {
263 direct_rou_setopt(gbl.currsub, 1);
264 /*
265 * Also, inhibit sectfloat() which is enabled with
266 * -fast or -O2 or greater.
267 */
268 flg.x[70] &= (~0x400);
269 }
270 }
271 ili_lpprg_init();
272
273 TR(DNAME " BBLOCK begins\n");
274 has_accel_code |= bblock();
275 TR1("- after bblock");
276 DUMP("bblock");
277 if (flg.inliner) {
278 TR(DNAME " INLINER begins\n");
279#if DEBUG
280 if (flg.x[29] == 0 || flg.x[29] == gbl.func_count)
281#endif
282 inliner();
283 DUMP("inliner");
284 TR1("- after inliner");
285 }
286
287 if (flg.opt >= 2 && XBIT(50, 0x40)) {
288 unconditional_branches();
289 DUMP("unconditional");
290 }
291 if (flg.opt >= 2 && !XBIT(47, 0x20)) {
292 TR(DNAME " OPTIMIZE_ALLOC begins\n");
293 optimize_alloc();
294 DUMP("optalloc");
295 TR1("- after optimize_alloc");
296 }
297
298 if (IPA_ENABLED) {
299 ipasave();
300 if (IPA_NO_ASM) {
301 ipasave_endfunc();
302 direct_rou_end();
303 continue;
304 }
305 }
306 if (IPA_INHERIT_ENABLED && !IPA_FUTURE_INHERIT_DISABLED) {
307 if (!IPA_ENABLED) {
308 fill_ipasym();
309 }
310 ipa();
311 DUMP("ipa");
312 if (IPA_Vestigial) {
313 ipasave_endfunc();
314 if (gbl.internal == 1) {
315 ipa_set_vestigial_host(); /* interf.c */
316 save_host_state(0x2 + (ipa_import_mode ? 0x20 : 0));
317 gbl.outersub = gbl.currsub;
318 gbl.outerentries = gbl.entries;
319 }
320 (void)summary(FALSE, FALSE);
321 continue;
322 }
323 }
324
325 /* infer array alignments */
326 TR(DNAME " PROCESS_ALIGN begins\n");
327 trans_process_align();
328 TR1("- after process_align");
329 DUMP("process-align");
330
331 if (flg.opt >= 2) {
332 if (XBIT(53, 2)) {
333 points_to_anal();
334 DUMP("pointsto");
335 }
336 pstride_analysis();
337 DUMP("pstride");
338 }
339
340 if (!XBIT(49, 1)) {
341 TR(DNAME " TRANSFORMER begins\n");
342 transform();
343 DUMP("transform");
344 TR1("- after transform");
345
346 forall_init();
347
348 if (!XBIT(49, 0x20)) {
349 if (flg.opt >= 2 && !XBIT(47, 0x02)) {
350 TR(DNAME " COMMUNICATIONS pre-OPTIMIZER begins\n");
351 comm_optimize_pre();
352 DUMP("comm-analyze-pre");
353 TR1("- after comm pre-optimizer");
354 }
355 TR(DNAME " COMMUNICATIONS ANALYZER begins\n");
356 comm_analyze();
357 DUMP("comm-analyze");
358 TR1("- after comm_analyze");
359
360 TR(DNAME " CALL ANALYZER begins\n");
361 call_analyze();
362 DUMP("call-analyze");
363 TR1("- after call_analyze");
364 if (flg.opt >= 2 && !XBIT(47, 0x01)) {
365 TR(DNAME " COMMUNICATIONS post-OPTIMIZER begins\n");
366 comm_optimize_post();
367 DUMP("comm-optimize-post");
368 TR1("- after comm post-optimizer");
369 }
370 if (flg.opt >= 2 && !XBIT(47, 0x08)) {
371 TR(DNAME " COMMUNICATIONS hoisting begins\n");
372 comm_invar();
373 DUMP("comm-invar");
374 TR1("- after comm_invar");
375 }
376 TR(DNAME " COMMUNICATIONS GENERATOR begins\n");
377 comm_generator();
378 DUMP("comm-generator");
379 TR1("- after comm_generator");
380 }
381 TR(DNAME " CONVERT_FORALL begins\n");
382 convert_forall();
383 DUMP("convert-forall");
384 TR1("- after convert_forall");
385
386 TR(DNAME " CONVERT_OUTPUT begins\n");
387 convert_output();
388 TR1("- after convert_output");
389 DUMP("convert-output");
390 }
391 if (XBIT(70, 0x400) || XBIT(47, 0x400000)
392 ) {
393 optimize(1);
394 DUMP("optimize0");
395 TR1("- after optimize0");
396 }
397 if (XBIT(70, 0x400)) {
398 sectfloat();
399 DUMP("sectfloat");
400 }
401 if (XBIT(47, 0x400000) || flg.opt >= 2 || XBIT(163, 1)
402 ) {
403 sectinline();
404 DUMP("sectinline");
405 }
406 if (XBIT(70, 0x18)) {
407 linearize_arrays();
408 DUMP("linearize");
409 }
410 if (!XBIT(70, 0x40)) {
411 DUMP("bredundss");
412 redundss();
413 DUMP("redundss");
414 }
415 if (flg.opt >= 2 && !XBIT(47, 0x1000)) {
416 TR(DNAME " OPTIMIZER begins\n");
417 optimize(0);
418 DUMP("optimize");
419 TR1("- after optimize");
420 }
421 if (IPA_ENABLED) {
422 ipasave_endfunc();
423 }
424 if (IPA_REPORT_ENABLED) {
425 ipa_report();
426 }
427
428 direct_rou_end();
429 if (flg.opt >= 2 && XBIT(53, 2)) {
430 fini_points_to_all();
431 }
432 } else { /* gbl.rutype == RU_BDATA */
433 direct_rou_load(gbl.currsub);
434 if (IPA_ENABLED) {
435 ipasave();
436 }
437 merge_commons();
438 if (XBIT(55, 2)) {
439 cleanup();
440 goto skip_compile;
441 }
442 /* block data must be transformed so that common blocks
443 * get handled -- lfm
444 */
445 /* infer array alignments */
446 TR("Blkdata -- " DNAME " PROCESS_ALIGN begins\n");
447 trans_process_align();
448 DUMP("process-align");
449 TR1("- after process_align");
450 if (!XBIT(49, 1)) {
451 TR("Blkdata -- " DNAME " TRANSFORMER begins\n");
452 transform();
453 DUMP("transform");
454 TR1("- after transform");
455 }
456 }
457#if DEBUG
458 if (XBIT(57, 0x100)) {
459 renumber_lines();
460 }
461#endif
462 if (XBIT(57, 0x2000)) {
463 DUMP("bunused");
464 eliminate_unused_variables(2);
465 DUMP("unused");
466 }
467 DUMP("before-output");
468 lower(0);
469 if (gbl.internal == 1) {
470 save_host_state(0x2 + (ipa_import_mode ? 0x20 : 0));
471 }
472 DUMP("output");
473 if (gbl.rutype != RU_BDATA && flg.opt >= 2 && XBIT(53, 2)) {
474 fini_pstride_analysis();
475 }
476#if DEBUG
477 if (DBGBIT(5, 4))
478 symdmp(gbl.dbgfil, DBGBIT(5, 8));
479 if (DBGBIT(5, 16))
480 dmp_dtype();
481#endif
482 } else { /* if( gbl.maxsev < 3 && !DBGBIT(2, 4) ) */
483 if (gbl.internal == 1) {
484 save_host_state(0x2);
485 }
486 } /* if( gbl.maxsev < 3 && !DBGBIT(2, 4) ) */
487
488 if (flg.xref) {
489 xref(); /* write cross reference map */
490 xtimes[7] += get_rutime();
491 }
492 skip_compile:
493 (void)summary(FALSE, FALSE);
494 errini();
495
496 if (gbl.internal == 1) {
497 gbl.outersub = gbl.currsub;
498 gbl.outerentries = gbl.entries;
499 }
500 stb.curr_scope = savescope;
501 ccff_close_unit_f90();
502 } while (!gbl.eof_flag);
503 finish(); /* finish does not return */
504 return 0; /* never reached */
505}

756 register_boolean_arg(arg_parser, "quad", (bool *)&(flg.quad), true);

773 parse_arguments(arg_parser, argc, argv);

775 /* Direct debug output */
776 if (was_value_set(arg_parser, &(flg.dbg)) ||
777 was_value_set(arg_parser, phase_dump_map)) {
778#if DEBUG
779 dodebug = 1;
780#endif

782 gbl.dbgfil = fopen(dbgfile, "w");
783 if (gbl.dbgfil == NULL)
784 errfatal(5);
785 } else if ((flg.dbg[0] & 1) || gbl.src_file == NULL) {
786 gbl.dbgfil = stderr;
787 } else {

789 tempfile = mkfname(gbl.src_file, file_suffix, ".qdbh");
790 } else {
791 tempfile = mkfname(gbl.src_file, file_suffix, ".qdbf");
792 if ((gbl.dbgfil = fopen(tempfile, "w")) == NULL)
793 errfatal(5);
794 }
795 }
796 }

798 /* Set preporocessor and Fortran source form
799 * ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
800 * FIXME this logic needs to be moved to where those values are consumed
801 */
802 if (was_value_set(arg_parser, &arg_preproc)) {
803 /* If the argument was present on command line set the value, otherwise
804 * keep "undefined" -1 */
805 preproc = arg_preproc;
806 }
807 if (was_value_set(arg_parser, &arg_freeform)) {
808 /* If the argument was present on command line set the value, otherwise
809 * keep "undefined" -1 */
810 flg.freeform = arg_freeform;

814 if (was_value_set(arg_parser, &arg_reentrant)) {
815 if (arg_reentrant) {
816 flg.x[7] |= 0x2; /* inhibit terminal func optz. */
817 flg.recursive = TRUE; /* no static locals */
818 } else {
819 flg.x[7] &= ~(0x2);
820 flg.recursive = FALSE;
821 }

825 destroy_arg_parser(&arg_parser);
826 destroy_action_map(&dump_map);

829 * ^^^^^^^^^^^^^^^^^^^^^^^^^^
830 */

832 /* Check optimization level */
833 if (flg.opt > 4) {
834 fprintf(stderr, "%s-W-Opt levels greater than 4 are not supported\n", version.lang);
835 }

837 if (!flg.doprelink)
838 flg.ipa |= 0x50; /* don't defer initialization, issue errors */
839
840 /* Postprocess target architecture */
841 do_set_tp(tp);
842#ifdef OMP_OFFLOAD_LLVM
843 if(omptp != NULL)
844 flg.omptarget = TRUE;
845#endif
846 /* Vectorizer settings */
847 flg.vect |= vect_val;
848 if (flg.vect & 0x10)
849 flg.x[19] &= ~0x10;
850 if (flg.vect & 0x20)
851 flg.x[19] &= ~8;
852 set_yflag(34, 0x30);

854 /* modexport file name */
855 mod_combined_name(modexport_val);
856 /* modindex file name */
857 mod_combined_index(modindex_val);
858
859 /* Extend source file lines */
860 if (arg_extend)
861 flg.extend_source = 132;
862
863 /* Set module directory linked list
864 * ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
865 * FIXME this is bad, have different implementations for different string
866 * lists, that needs to stop.
867 */
868 char **module_dir = module_dirs;
869 while (module_dir && *module_dir) {
870 moddir_list *mdl;
871 NEW(mdl, moddir_list, 1);
872 mdl->module_directory = *module_dir;
873 mdl->next = NULL;
874 if (module_directory_list == NULL) {
875 module_directory_list = mdl;
876 } else {
877 moddir_list *link;
878 for (link = module_directory_list; link->next; link = link->next)
879 ;
880 link->next = mdl;
881 }
882 ++module_dir;
883 }
884
885 flg.genilm = TRUE;
886 /* set -x 49 0x400000, F90-style output */
887 set_xflag(49, 0x400000);
888 /* set -x 58 0x10000, handle all pointers */
889 set_xflag(58, 0x10000);
890 gbl.is_f90 = TRUE;
891 /* set -x 58 0x40, and reset hpf flag, no static-init */
892 set_xflag(58, 0x40);
893 flg.defaulthpf = flg.hpf = FALSE;
894 flg.defaultsequence = flg.sequence = TRUE;
895 /* set -x 58 0x20000, allocate temps only as big as needed */
896 set_xflag(58, 0x20000);
897
898#ifdef TARGET_SUPPORTS_QUADFP
899 if (flg.qp) {
900 /* set -y 57 0x4, enable quad-precision real */
901 set_yflag(57, 0x4);
902 /* set -y 57 0x8, enable quad-precision complex */
903 set_yflag(57, 0x8);
904 }
905#endif
906
907 if (XBIT(25, 0xf0)) {
908 fprintf(stderr, "%s-I-Beta Release Optimizations Activated\n", version.lang);
909 }
910
911 if (flg.x[27] == -1)
912 flg.x[27] = 4; /* default/max overlap of 4 */
913
914 if (flg.es && !flg.p)
915 flg.x[123] |= 0x100;
916
917 empty_cl:
918 if (gbl.src_file == NULL) {
919 if (flg.ipa & 0x0a) {
920 /* for IPA propagation or when generating static$init, no sourcefile */
921 gbl.src_file = "pghpf.prelink.f";
922 sourcefile = strdup(gbl.src_file);
923 gbl.srcfil = NULL;
924 copy_curr_file = 0;
925 } else {
926 gbl.src_file = "STDIN.f";
927 sourcefile= strdup(gbl.src_file);
928 gbl.srcfil = stdin;
929 }
930 goto do_curr_file;
931 }
932
933 if (errflg)
934 finish();
935
936 if (ipa_import_mode) {
937 if (gbl.src_file) {
938 sourcefile = strdup(gbl.src_file);
939 basenam(gbl.src_file, "", sourcefile);
940 } else {
941 gbl.src_file = "STDIN.f";
942 sourcefile = strdup(gbl.src_file);
943 }
944 file_suffix = "";
945 for (char *s = sourcefile; *s; ++s) {
946 if (*s == '.')
947 file_suffix = s;
948 }
949 } else {
950 if (!nosuffixcheck) {
951 /* open sourcefile */
952 for (i = 0; suffixes[i].nm; ++i) {
953 int lsuf, lsrc;
954 lsuf = strlen(suffixes[i].nm);
955 lsrc = strlen(gbl.src_file);
956 if (lsuf >= lsrc)
957 continue;
958 if (strcmp(gbl.src_file + (lsrc - lsuf), suffixes[i].nm))
959 continue;
960 if ((gbl.srcfil = fopen(gbl.src_file, "r")) != NULL) {
961 /* fill in flg.freeform, file_suffix, fpp_, gbl.src_file */
962 if (flg.freeform == -1)
963 flg.freeform = suffixes[i].form;
964 file_suffix = suffixes[i].nm;
965 if (suffixes[i].fpp) {
966 if (preproc == -1 || preproc == 1)
967 fpp_ = TRUE;
968 /* -nopreproc overrides use of .F extension */
969 }
970 /* strip pathname, if any */
971 sourcefile = (char *)malloc(strlen(gbl.src_file) + 1);
972 /* base name strips the pathname */
973 basenam(gbl.src_file, "", sourcefile);
974 goto is_open;
975 }
976 /* ** else will be reported_as_an error(2...) below ** */
977 }
978 }
979 if ((gbl.srcfil = fopen(gbl.src_file, "r")) != NULL) {
980 /* fill in sourcefile, file_suffix */
981 sourcefile = (char *)malloc(strlen(gbl.src_file) + 1);
982 basenam(gbl.src_file, "", sourcefile);
983 file_suffix = "";
984 for (char *s = sourcefile; *s; ++s) {
985 if (*s == '.')
986 file_suffix = s;
987 }
988 goto is_open;
989 }
990 /* not found */
991 error(2, 4, 0, gbl.src_file, CNULL);
992 is_open:
993 if (preproc == 1)
994 fpp_ = TRUE; /* -preproc forces preprocessing */
995 }
996
997 do_curr_file:
998
999 if (gbl.file_name == NULL)
1000 gbl.file_name = gbl.src_file;
1001 if (sourcefile != NULL)
1002 gbl.module = mkfname(sourcefile, file_suffix, "");
1003 if (copy_curr_file)
1004 gbl.curr_file = gbl.src_file;
1005
1006 /* process object file: */
1007
1008 gbl.objfil = NULL;
1009
1010 if (sourcefile == NULL) {
1011 if (!flg.output || outfile_name == NULL) {
1012 gbl.outfil = stdout;
1013 } else {
1014 if ((gbl.outfil = fopen(outfile_name, "w")) == NULL)
1015 error(4, 0, 0, "Unable to open output file", outfile_name);
1016 }
1017 if (OUTPUT_DWARF && (dbg_file == NULL) && outfile_name != NULL) {
1018 /* make dwarf debug info file from the source file */
1019 mkDwfInfoFilename();
1020 if ((dbg_file = fopen(dbg_file_name, "wb")) == NULL)
1021 errfatal(9);
1022 dwarf_set_fn();
1023 } else {
1024 flg.debug = 0;
1025 }
1026 if (stboutfile != NULL) {
1027 if ((gbl.stbfil = fopen(stboutfile, "w")) == NULL)
1028 errfatal(9);
1029 } else {
1030 gbl.stbfil = 0;
1031 }
1032 } else {
1033 /* process listing file */
1034 if (flg.code || flg.list || flg.xref) {
1035 if (listfile == NULL) {
1036 /* make listing filename from sourcefile name */
1037 listfile = mkfname(sourcefile, file_suffix, LISTFILE);
1038 }
1039 if ((fd = fopen(listfile, "w")) == NULL)
1040 errfatal(3);
1041 list_init(fd);
1042 }
1043 if (OUTPUT_DWARF && (dbg_file == NULL)) {
1044 /* make dwarf debug info file from the source file */
1045 if (outfile_name != NULL) {
1046 mkDwfInfoFilename();
1047 } else {
1048 dbg_file_name = mkfname(sourcefile, file_suffix, ".dbg");
1049 }
1050 if ((dbg_file = fopen(dbg_file_name, "wb")) == NULL)
1051 errfatal(9);
1052 }
1053 if (stboutfile) {
1054 if ((gbl.stbfil = fopen(stboutfile, "w")) == NULL)
1055 errfatal(9);
1056 } else {
1057 gbl.stbfil = NULL;
1058 }
1059
1060 /* process assembly output file */
1061 if (flg.asmcode) {
1062 if (asmfile == NULL) {
1063 /* make assembly filename from sourcefile name */
1064 asmfile = mkfname(sourcefile, file_suffix, ASMFILE);
1065 }
1066 if ((gbl.asmfil = fopen(asmfile, "w")) == NULL)
1067 errfatal(9);
1068 } else /* do this for compilers which write asm code to stdout */
1069 gbl.asmfil = stdout;
1070 /* process source output file */
1071 if (flg.output && !flg.es) {
1072 /* make output filename from sourcefile name */
1073 if (outfile_name == NULL) {
1074 outfile_name = mkfname(sourcefile, file_suffix, ".ilm");
1075 }
1076 if ((gbl.outfil = fopen(outfile_name, "w")) == NULL)
1077 error(4, 0, 0, "Unable to open output file", outfile_name);
1078 } else
1079 gbl.outfil = stdout;
1080
1081
1082 if (flg.doprelink && (flg.ipa & 0x03) == 0 && gbl.ipaname == NULL) {
1083 gbl.ipaname = mkfname(sourcefile, file_suffix, ".d");
1084 gbl.gblfil = NULL;
1085 unlink(gbl.ipaname);
1086 }
1087
1088 /* create temporary file for preprocessor output & preprocess */
1089 if (!ipa_import_mode) {
1090 if (fpp_) {
1091 if (flg.es) {
1092 if (cppfile == NULL)
1093 gbl.cppfil = stdout;
1094 else if ((gbl.cppfil = fopen(cppfile, "w")) == NULL)
1095 errfatal(5);
1096 } else {
1097 if ((gbl.cppfil = tmpfile()) == NULL)
1098 errfatal(5);
1099 }
1100 fpp();
1101 if (flg.es || gbl.maxsev >= 3)
1102 finish();
1103 if (flg.list)
1104 list_page();
1105 scan_init(gbl.cppfil);
1106 } else
1107 scan_init(gbl.srcfil);
1108 }
1109#if DEBUG
1110 assert(flg.es == 0, "init:flg.esA", 0, 0);
1111#endif
1112 assemble_init();
1113 if (OUTPUT_DWARF && dbg_file != NULL) {
1114 dwarf_set_fn();
1115 }
1116 }
1117 free(sourcefile);
1118 gbl.func_count = 0;
1119
1120 if (XBIT(125, 0x8))
1121 gbl.ftn_true = 1;
1122 else
1123 gbl.ftn_true = -1;
1124
1125 /*
1126 * direct_init() must be called at a point where we are sure that
1127 * the values of flg members, especially xflags, can be propagated
1128 * to the global, routine, etc. directive data structures. For example,
1129 * direct_init() can only be called after the code above which can
1130 * disable the cuda/accel features in the code by clearing their
1131 * respective xflags.
1132 */
1133 direct_init();
1134
1135 /* set mach, currently need for -mp=align optimization on sandybridge */
1136 set_mach(&mach, flg.tpvalue[0]);
1137
1138 return;
1139}

1177{

1179 int i;
1180 INT total;
1181 int prct;
1182 int tmp;
1183
1184 total = 0;
1185 for (i = 0; i < _N_WHO; i++)
1186 total += xtimes[i];

1189 goto xbitcheck;
1190

1199 prct = tmp / total;

1201 prct);

1203 list_line(buf);

1205 fprintf(gbl.dbgfil, "%s\n", buf);

1208
1209 sprintf(buf, " Total time %15d millisecs", total);
1210 if (flg.code || flg.list || flg.xref) {
1211 list_line(buf);
1212 } else if (gbl.dbgfil)
1213 fprintf(gbl.dbgfil, "%s\n", buf);
1214

1227 }
1228 }
1229 sprintf(buf, " Total time %15d millisecs", total);
1230 fprintf(stderr, "%s\n", buf);

1318*/

1320finish(void)
1321{

1323 static int called = 0;
1324
1325 if (!ipa_import_mode)
1326 scan_fini();
1327 if (IPA_INHERIT_ENABLED && (flg.opt >= 2 || IPA_COLLECTION_ENABLED)) {
1328 ipa_fini();

1334 ast_fini();

1338 if (XBIT(123, 0x30000)) {

1340 }

1375 while (1) {
1376 int c;
1377 c = fgetc(gbl.dependfil);
1378 if (c == EOF)
1379 break;
1380 fputc(c, fp);
1381 }
1382 }
1383 fputc('\n', fp);
