64
65 linbuf[0] = '\0';
66 header();
67 readln();
68 while (1) {
69 tok = scan();
70 if (tok == TOK_ID) {
71 i = curchr - fstchr;
72 strncpy(name, &linbuf[fstchr], i);
73 name[i] = '\0';
74 tok = scan();
75 if (tok != TOK_EQUAL) {
76 error("equal expected");
77 goto close;
78 }
79 init();
80 while (1) {
81 tok = scan();
82 if (tok == TOK_OR) {
83 output();
84 } else if (tok == TOK_EOLN) {
85 output();
86 break;
87 } else if (tok == TOK_EOF) {
88 error("unexpected eof");
89 goto close;
90 }
91 }
92 } else if (tok == TOK_EOF)
93 break;
94 else if (tok == TOK_BREAK) {
95 /* output(); */
96 fprintf(outfile, " }\n\n");
97 current = 0;
98 fprintf(outfile, "/***** SEMANTIC BREAK - SEM%d *****/\n\n", bcnt);
99 fprintf(outfile, " switch(rednum) {\n");
100 ++bcnt;
101 } else if (tok != TOK_EOLN) {
102 printf("illegal token %s, line %d\n", tok_name[tok], lineno);
103 break;
104 }
105 }
106 footer();
107
108close:
109 fclose(infile);
110 fclose(outfile);
111 fclose(prodfile);

131static int
132scan()
133{
134 int token = -1;
135 char c;
136
137 while (1) {
138 c = linbuf[curchr];
139 fstchr = curchr;
140 curchr++;
141 if (c == ' ' || c == '\t')
142 continue;
143 if (c == '<') {
144 while (linbuf[curchr] != '>' && linbuf[curchr] != '\n')
145 curchr++;
146 if (linbuf[curchr] == '\n')
147 error("unmatched left angle bracket");
148 curchr++;
149 token = TOK_ID;
150 } else if (c == ':' && linbuf[curchr] == ':' && linbuf[curchr + 1] == '=') {
151 curchr = curchr + 2;
152 token = TOK_EQUAL;
153 } else if (c == '|') {
154 token = TOK_OR;
155 if (linbuf[curchr] != '\n') {
156 error("warning: | does not end line");
157 }
158 readln();
159 } else if (c == '\'') {
160 while (linbuf[curchr] != '\'' && linbuf[curchr] != '\n')
161 curchr++;
162 if (linbuf[curchr] == '\n')
163 error("unmatched single quote");
164 curchr++;
165 continue;
166 } else if (c == '\n') {
167 readln();
168 token = TOK_EOLN;
169 } else if (c == '\0') {
170 token = TOK_EOF;
171 } else if (c == '_') {
172 readln();
173 continue;
174 } else if (c == '>') {
175 error("isolated right angle bracket");
176 } else if (fstchr == 0 && linbuf[0] == '.' && linbuf[1] == 'B') {
177 readln();
178 token = TOK_BREAK;
179 } else if (linbuf[0] == '#') {
180 readln();
181 continue;
182 } else {
183 while (linbuf[curchr] != ' ' && linbuf[curchr] != '\t' &&
184 linbuf[curchr] != '\n')
185 curchr++;
186 token = TOK_OTHER;
187 }
188 break;
189 }
190 return token;
191}
