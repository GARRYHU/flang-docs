31double __fsd_acos_fma3(double const a)
32{
33 __m128d const AH0 = _mm_setr_pd(A0_D, H0_D);
34 __m128d const BI0 = _mm_setr_pd(B0_D, I0_D);
35 __m128d const CJ0 = _mm_setr_pd(C0_D, J0_D);
36 __m128d const DK0 = _mm_setr_pd(D0_D, K0_D);
37 __m128d const EL0 = _mm_setr_pd(E0_D, L0_D);
38 __m128d const FM0 = _mm_setr_pd(F0_D, M0_D);
39 __m128d const GN0 = _mm_setr_pd(G0_D, N0_D - a);
40
41 __m128d const AH1 = _mm_setr_pd(A1_D, H1_D);
42 __m128d const BI1 = _mm_setr_pd(B1_D, I1_D);
43 __m128d const CJ1 = _mm_setr_pd(C1_D, J1_D);
44 __m128d const DK1 = _mm_setr_pd(D1_D, K1_D);
45 __m128d const EL1 = _mm_setr_pd(E1_D, L1_D);
46 __m128d const FM1 = _mm_setr_pd(F1_D, M1_D);
47
48 double res;
49 double x = __ll_as_double(ABS_MASK_LL & __double_as_ll(a));
50 double x2 = a * a;
51 double a3 = x2 * a;
52 double x6 = a3 * a3;
53
54 if (__double_as_ll(x) >= __double_as_ll(THRESHOLD_D))
55 {
56 double sq = 1.0 - x;
57 /*
58 * There seems to be a concensus that setting errno is important
59 * for fastmath intrinsics.
60 * Disable using Intel hardware instruction sqrt.
61 */
62 sq = sqrt(sq);
63
64 double pi_hi = a < 0.0 ? PI_HI_D : 0.0;
65 long long fix = (long long)(a > 1.0) << 63;
66 long long sign = SGN_MASK_LL & __double_as_ll(a);
67 fix = fix ^ sign;
68
69 __m128d _x = _mm_set1_pd(x);
70 __m128d _p1;
71 _p1 = _mm_fmadd_pd(AH1, _x, BI1);
72 _p1 = _mm_fmadd_pd(_p1, _x, CJ1);
73 _p1 = _mm_fmadd_pd(_p1, _x, DK1);
74 _p1 = _mm_fmadd_pd(_p1, _x, EL1);
75 _p1 = _mm_fmadd_pd(_p1, _x, FM1);
76
77 double p1hi = _mm_cvtsd_f64(_p1);
78 _p1 = _mm_shuffle_pd(_p1, _p1, 3);
79 double p1lo = _mm_cvtsd_f64(_p1);
80
81 p1hi = FMA(p1hi, x, G1_D);
82 double p1 = FMA(p1hi, x6, p1lo);
83 p1 = FMA(sq, p1, -pi_hi);
84
85 res = __ll_as_double(fix ^ __double_as_ll(p1));
86 }
87 else
88 {
89 __m128d _x2 = _mm_set1_pd(x2);
90 __m128d _x2_a3 = _mm_setr_pd(x2, a3);
91 __m128d _p0;
92
93 _p0 = _mm_fmadd_pd(AH0, _x2, BI0);
94 _p0 = _mm_fmadd_pd(_p0, _x2, CJ0);
95 _p0 = _mm_fmadd_pd(_p0, _x2, DK0);
96 _p0 = _mm_fmadd_pd(_p0, _x2, EL0);
97 _p0 = _mm_fmadd_pd(_p0, _x2, FM0);
98 _p0 = _mm_fmadd_pd(_p0, _x2_a3, GN0);
99
100 double p0hi = _mm_cvtsd_f64(_p0);
101 _p0 = _mm_shuffle_pd(_p0, _p0, 3);
102 double p0lo = _mm_cvtsd_f64(_p0);
103
104 double x12 = x6 * x6;
105 double a15 = x12 * a3;
106 res = FMA(p0hi, a15, p0lo);
107 }
108
109 return res;
110}
