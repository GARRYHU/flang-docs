11/* mth.h - 'parameterize' the names of the __mth_i/__fmth_i ... functions.
12 * Macros are not used for the new naming conventions, i.e.,
13 * _f[sv][sd]_BASE, _g[sv][sdcz]_BASE[L]
14 *
15 * New ...
16 * Create the function representing a math routine:
17 * Name is
18 * __<ftype><data type>_<name>_<vectlen><mask>
19 *
20 * <ftype> : f - fastmath (default)
21 * r - relaxed math (-Mfprelaxed ...)
22 * p - precise math (-Kieee)
23 * <data type> : s - single precision
24 * d - double precision
25 * c - single precision complex
26 * z - double precision complex
27 * <name> : exp, log, log10, pow, powi, powk, powi1, powk1, sin, cos,
28 * tan, asin, acos, atan, sinh, cosh, tanh, atan2,
29 * <vectlen> : 1 (scalar), 2, 4, 8, 16
30 * <mask> : m or null
31 *
32exp
33log
34log10
35pow
36powi
37powk
38powi1
39powk1
40sin
41cos
42tan
43asin
44acos
45atan
46sinh
47cosh
48tanh
49atan2
50div
51mod
52floor
53ceil
54aint
55 */
