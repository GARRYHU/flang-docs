21int vasprintf(char **strp, const char *format, va_list ap)
22{
23 int len = _vscprintf(format, ap);
24 if (len == -1)
25 return -1;
26 char *str = (char*)malloc((size_t) len + 1);
27 if (!str)
28 return -1;
29 int retval = vsnprintf(str, len + 1, format, ap);
30 if (retval == -1) {
31 free(str);
32 return -1;
33 }
34 *strp = str;
35 return retval;
36}

38int asprintf(char **strp, const char *format, ...)
39{
40 va_list ap;
41 va_start(ap, format);
42 int retval = vasprintf(strp, format, ap);
43 va_end(ap);
44 return retval;
