39static void
40add_successor(INSTR_LIST *instr, INSTR_LIST *succ)
41{
42 int i, j;
43
44 /* avoid creating circular dependency */
45 if (instr == succ)
46 return;
47 if (succ->flags & ROOTDG)
48 return;
49 i = instr->rank - srank_dg;
50 j = succ->rank - srank_dg;
51 matrix_dg[size_dg * i + j] = succ;
52}

54static bool
55build_same_base_nme(int nme1, int nme2, int *res_nme)
56{
57 int nme;
58 if (nme1 && nme2) {
59 if (NME_TYPE(nme1) != NME_TYPE(nme2))
60 return false;
61
62 switch (NME_TYPE(nme1)) {
63 default:
64 break;
65 case NT_VAR:
66 if (NME_SYM(nme1) != NME_SYM(nme2) &&
67 STYPEG(NME_SYM(nme1)) == STYPEG(NME_SYM(nme2)) &&
68 DTY(DTYPEG(NME_SYM(nme1))) == DTY(DTYPEG(NME_SYM(nme2)))) {
69 *res_nme = addnme(NT_VAR, NME_SYM(nme1), 0, 0);
70 return true;
71 }
72 break;
73 case NT_ARR:
74 case NT_MEM:
75 case NT_IND:
76 if (build_same_base_nme(NME_NM(nme1), NME_NM(nme2), &nme)) {
77 *res_nme = add_arrnme(NME_TYPE(nme2), NME_SYM(nme2), nme,
78 NME_CNST(nme2), NME_SUB(nme2), NME_INLARR(nme2));
79 return true;
80 }
81 break;
82 }
83 }
84 return false;
85}

87int
88enhanced_conflict(int nme1, int nme2)
89{
90 int c;
91 int nme3;
92
93 c = conflict(nme1, nme2);
94 if (XBIT(183, 0x80) && c != NOCONFLICT && c != SAME) {
95 if (build_same_base_nme(nme1, nme2, &nme3))
96 c = conflict(nme1, nme3);
97 }
98 return c;

101static void
102build_idep_graph(INSTR_LIST *iroot, INSTR_LIST *cur_instr)
103{
104 INSTR_LIST *instr;
105 int instcount;
106 int ilix, c;
107 bool first_load, has_pred;
108 OPERAND *operand;
109
110 has_pred = false;
111 switch (cur_instr->i_name) {
112 default:
113 break;
114 case I_LOAD:
115 assert(cur_instr->tmps, "build_dep_graph():missing tmps for load instr ", 0,
116 ERR_Fatal);
117 instr = cur_instr->prev;
118 ilix = cur_instr->ilix;
119 first_load = true;
120 instcount = 0;
121 while (instr && instr != iroot) {
122 instcount++;
123 switch (instr->i_name) {
124 default:
125 break;
126 case I_STORE:
127 /* if conflicting store then add this load to its successor's list */
128 if ((ilix == 0) || (instr->ilix == 0) ||
129 (IL_TYPE(ILI_OPC(instr->ilix)) != ILTY_STORE) ||
130 (ILI_OPND(ilix, 1) == ILI_OPND(instr->ilix, 2))) {
131 has_pred = true;
132 add_successor(instr, cur_instr);
133 } else {
134 c = enhanced_conflict(ILI_OPND(ilix, 2), ILI_OPND(instr->ilix, 3));
135 if (c == SAME || (flg.depchk && c != NOCONFLICT)) {
136 has_pred = true;
137 add_successor(instr, cur_instr);
138 }
139 }
140 break;
141 }
142 instr = instr->prev;
143 }
144 break;
145 case I_STORE:
146 instr = cur_instr->prev;
147 ilix = cur_instr->ilix;
148 first_load = true;
149 while (instr && instr != iroot) {
150 switch (instr->i_name) {
151 default:
152 break;
153 case I_STORE:
154 /* if conflicting store then add this store to its successor's list */
155 if ((ilix == 0) || (instr->ilix == 0) ||
156 (IL_TYPE(ILI_OPC(instr->ilix)) != ILTY_STORE) ||
157 (ILI_OPND(ilix, 2) == ILI_OPND(instr->ilix, 2))) {
158 has_pred = true;
159 add_successor(instr, cur_instr);
160 } else {
161 c = enhanced_conflict(ILI_OPND(ilix, 3), ILI_OPND(instr->ilix, 3));
162 if (c == SAME || (flg.depchk && c != NOCONFLICT)) {
163 has_pred = true;
164 add_successor(instr, cur_instr);
165 }
166 }
167 break;
168 case I_LOAD:
169 /* if conflicting store then add this store to its successor's list */
170 if ((ilix == 0) || (instr->ilix == 0) ||
171 (IL_TYPE(ILI_OPC(instr->ilix)) != ILTY_STORE) ||
172 (ILI_OPND(ilix, 2) == ILI_OPND(instr->ilix, 1))) {
173 has_pred = true;
174 add_successor(instr, cur_instr);
175 } else {
176 c = enhanced_conflict(ILI_OPND(ilix, 3), ILI_OPND(instr->ilix, 2));
177 if (c == SAME || (flg.depchk && c != NOCONFLICT)) {
178 has_pred = true;
179 add_successor(instr, cur_instr);
180 }
181 }
182 break;
183 }
184 instr = instr->prev;
185 }
186 break;
187 }
188 operand = cur_instr->operands;
189 while (operand) {
190 if (operand->ot_type == OT_TMP) {
191 instr = cur_instr->prev;
192 while (instr) {
193 if (operand->tmps == instr->tmps) {
194 has_pred = true;
195 add_successor(instr, cur_instr);
196 break;
197 }
198 if (instr == iroot)
199 break;
200 instr = instr->prev;
201 }
202 }
203 operand = operand->next;
204 }
205 if (!has_pred)
206 add_successor(iroot, cur_instr);
207}

225static INSTR_LIST *
226build_block_idep_graph(INSTR_LIST *istart, bool *success)
227{
228 INSTR_LIST *instr, *bbinstr;
229 int inst_count = 0;
230 int interesting_stores = 0;
231 int interesting_loads = 0;
232 int interesting_instrs = 0;
233
234 instr = istart->next;
235 *success = false;
236 while (instr) {
237 instr->rank = irank++;
238 inst_count++;
239 switch (instr->i_name) {
240 case I_LOAD:
241 if (interesting_stores && is_lltype_interesting(instr->ll_type))
242 interesting_loads++;
243 break;
244 case I_STORE:
245 if (is_lltype_interesting(instr->operands->ll_type))
246 interesting_stores++;
247 break;
248 case I_CALL:
249 case I_SW:
250 case I_RET:
251 case I_BR:
252 case I_NONE:
253 case I_FDIV:
254 case I_FPTRUNC:
255 case I_FPEXT:
256 if (XBIT(183, 0x100) ||
257 ((inst_count > 50) && interesting_instrs &&
258 (interesting_loads > ((interesting_stores * 3) / 2)))) {
259 bbinstr = istart->next;
260 *success = init_sched_graph(inst_count, istart->rank);
261 if (*success) {
262 instr->flags |= ROOTDG;
263 while (bbinstr && (bbinstr != instr->next)) {
264 build_idep_graph(istart, bbinstr);
265 bbinstr = bbinstr->next;
266 }
267 }
268 }
269 return instr;
270 default:
271 if (BINOP(instr->i_name) || BITOP(instr->i_name) ||
272 CONVERT(instr->i_name) || PICALL(instr->i_name))
273 if (is_lltype_interesting(instr->ll_type)) {
274 interesting_instrs++;
275 }
276 }
277 instr = instr->next;
278 }
279 return instr;
280}

282void
283sched_block_breadth_first(INSTR_LIST *istart, int level)
284{
285 int i, j, k;
286 INSTR_LIST *succ, *pred, *entry;
287
288 i = istart->rank - srank_dg;
289
290 entry = last_instr;
291 /* Schedule GEP/LOAD/BITCAST first */
292 for (j = 0; j < size_dg; j++) {
293 succ = matrix_dg[i * size_dg + j];
294 if (succ) {
295 switch (succ->i_name) {
296 default:
297 break;
298 case I_GEP:
299 case I_BITCAST:
300 case I_LOAD:
301 matrix_dg[i * size_dg + j] = NULL;
302 pred = NULL;
303 for (k = 0; k < size_dg; k++) {
304 pred = matrix_dg[k * size_dg + j];
305 if (pred)
306 break;
307 }
308 if (pred == NULL) {
309 last_instr->next = succ;
310 succ->prev = last_instr;
311 last_instr = succ;
312 last_instr->next = NULL;
313 }
314 break;
315 }
316 }
317 }
318
319 /* Schedule any instructions but stores */
320 for (j = 0; j < size_dg; j++) {
321 succ = matrix_dg[i * size_dg + j];
322 if (succ) {
323 if (succ->i_name != I_STORE) {
324 matrix_dg[i * size_dg + j] = NULL;
325 pred = NULL;
326 for (k = 0; k < size_dg; k++) {
327 pred = matrix_dg[k * size_dg + j];
328 if (pred)
329 break;
330 }
331 if (pred == NULL) {
332 last_instr->next = succ;
333 succ->prev = last_instr;
334 last_instr = succ;
335 last_instr->next = NULL;
336 }
337 }
338 }
339 }
340
341 /* Schedule stores */
342 for (j = 0; j < size_dg; j++) {
343 succ = matrix_dg[i * size_dg + j];
344 if (succ) {
345 if (succ->i_name == I_STORE) {
346 matrix_dg[i * size_dg + j] = NULL;
347 pred = NULL;
348 for (k = 0; k < size_dg; k++) {
349 pred = matrix_dg[k * size_dg + j];
350 if (pred)
351 break;
352 }
353 if (pred == NULL) {
354 last_instr->next = succ;
355 succ->prev = last_instr;
356 last_instr = succ;
357 last_instr->next = NULL;
358 }
359 }
360 }
361 }
362
363 succ = entry->next;
364 while (succ) {
365 sched_block_breadth_first(succ, level + 1);
366 succ = succ->next;
367 }
368}

370void
371check_circular_dep(INSTR_LIST *istart)
372{
373 int i, j;
374 INSTR_LIST *succ;
375
376 i = istart->rank - srank_dg;
377 if (istart->flags & INST_VISITED) {
378 printf(" CIRCULAR DEPENDENCY !\n");
379 return;
380 }
381 istart->flags |= INST_VISITED;
382 for (j = 0; j < size_dg; j++) {
383 succ = matrix_dg[i * size_dg + j];
384 if (succ) {
385 printf("i%d -> ", istart->rank);
386 check_circular_dep(succ);
387 }
388 }
389 printf("\n");
390 istart->flags &= ~INST_VISITED;
391}

393void
394sched_block(INSTR_LIST *istart, INSTR_LIST *iend)
395{
396 if (istart && istart->next != iend) {
397 last_instr = istart;
398 last_instr->next = NULL;
399 sched_block_breadth_first(istart, 0);
400 last_instr->next = iend;
401 if (iend != NULL)
402 iend->prev = last_instr;
403 }
404}

406void
407sched_instructions(INSTR_LIST *istart)
408{
409 INSTR_LIST *instr;
410 bool dep_graph_created = false;
411 irank = 1;
412 istart->flags |= ROOTDG;
413
414 while (istart) {
415 instr = build_block_idep_graph(istart, &dep_graph_created);
416 if (dep_graph_created)
417 sched_block(istart, instr);
418 istart = instr;
