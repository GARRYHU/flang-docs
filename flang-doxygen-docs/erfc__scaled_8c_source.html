63double
64__mth_i_derfc_scaled(double arg)
65{
66 double x, y, ysq, xnum, xden, del;
67 int i;
68 double result;
69
70 x = arg;
71 y = fabs(x);
72
73 if (y <= thresh) {
74 /* evaluate erf for |x| <= 0.46875 */
75 ysq = zero;
76 if (y > xsmallr8)
77 ysq = y * y;
78 xnum = a[4] * ysq;
79 xden = ysq;
80 for (i = 0; i < 3; i++) {
81 xnum = (xnum + a[i]) * ysq;
82 xden = (xden + b[i]) * ysq;
83 }
84 result = x * (xnum + a[3]) / (xden + b[3]);
85 result = one - result;
86 result = exp(ysq) * result;
87 goto ret;
88 } else if (y <= four) {
89 /* evaluate erfc for 0.46875 <= |x| <= 4.0 */
90 xnum = c[8] * y;
91 xden = y;
92 for (i = 0; i < 7; i++) {
93 xnum = (xnum + c[i]) * y;
94 xden = (xden + d[i]) * y;
95 }
96 result = (xnum + c[7]) / (xden + d[7]);
97 } else {
98 /* evaluate erfc for |x| > 4.0 */
99 result = zero;
100 if (y >= xbig) {
101 if (y > xmaxr8)
102 goto negval;
103 if (y >= xhuge) {
104 result = sqrpi / y;
105 goto negval;
106 }
107 }
108 ysq = one / (y * y);
109 xnum = p[5] * ysq;
110 xden = ysq;
111 for (i = 0; i < 4; i++) {
112 xnum = (xnum + p[i]) * ysq;
113 xden = (xden + q[i]) * ysq;
114 }
115 result = ysq * (xnum + p[4]) / (xden + q[4]);
116 result = (sqrpi - result) / y;
117 }
118negval:
119 /* fix up for negative argument, erf, etc. */
120 if (x < zero) {
121 if (x < xneg) {
122 result = xinfr8;
123 } else {
124#if defined(TARGET_WIN)
125 double tmp = x * sixten;
126 long l = tmp;
127 tmp = l;
128 ysq = tmp / sixten;
129#else
130 ysq = trunc(x * sixten) / sixten;
131#endif
132 del = (x - ysq) * (x + ysq);
133 y = exp(ysq * ysq) * exp(del);
134 result = (y + y) - result;
135 }
136 }
137ret:
138 return result;
139}
