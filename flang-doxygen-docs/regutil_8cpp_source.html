8/** \file
9 * \brief Register allocation module used by the expander and the optimizer
10 *
11 * Contains:
12 * - void reg_init(int) - initializes the register information for a function
13 * - void addrcand(int) - adds an ILI to the appropriate candidate list

84 \brief Add a potential candidate, an ILI, to the appropriate candidate list.
85 */
86void
87addrcand(int ilix)
88{
89 int atype, /* type of the candidate */
90 rtype, /* register type of the ili */
91 msize, /* memory size of the ili */
92 nme, /* names entry of the ili */
93 rcand; /* RCAND for the ili */
94
95 switch (ILI_OPC(ilix)) { /* case according to the ILI opcode */
96
97 case IL_ACEXT:
98 /* these are never seen by optimizer in c , for now, just return */
99 assert(STYPEG(CONVAL1G(ILI_OPND(ilix, 1))) == ST_LABEL,
100 "unexpected acext in addrcand", ilix, ERR_Severe);
101 return;
102 case IL_LD: /* load data register */
103 rtype = RATA_IR;
104 msize = ILI_OPND(ilix, 3);
105 goto ac_load;
106
107 case IL_LDKR:
108 rtype = RATA_KR;
109 msize = MSZ_I8;
110 goto ac_load;
111
112 case IL_VCON:
113 case IL_VLD:
114 case IL_VLDU:
115 case IL_VST:
116 case IL_VSTU:
117 /* ignore these */
118 return;
119
120 case IL_LDA: /* load address register */
121 rtype = RATA_AR;
122 msize = MSZ_PTR;
123 goto ac_load;
124 case IL_LDQ: /* load m128*/
125 rtype = RATA_DP;
126 msize = MSZ_F16;
127 goto ac_load;
128 case IL_LD256: /* load m256*/
129 rtype = RATA_DP;
130 msize = MSZ_F32;
131 goto ac_load;
132 case IL_LD256A: /* load m256*/
133 rtype = RATA_DP;
134 msize = MSZ_F32;
135 goto ac_load;
136 case IL_LDSCMPLX:
137 rtype = RATA_CSP;
138 msize = MSZ_F8;
139 goto ac_load;
140 case IL_LDDCMPLX:
141 rtype = RATA_CDP;
142 msize = MSZ_F16;
143 goto ac_load;
144
145 case IL_LDSP: /* load single precision */
146 rtype = RATA_SP;
147 msize = MSZ_F4;
148 goto ac_load;
149
150 case IL_LDDP: /* load double precision */
151 rtype = RATA_DP;
152 msize = MSZ_F8;
153
154 ac_load: /* common entry for the loads */
155
156 /*
157 * check if the ili is already a candidate; if so, just increment its
158 * (the candidate entry for the ili) count
159 */
160 if ((rcand = ILI_RAT(ilix)) != 0) {
161 RCAND_COUNT(rcand) += rcandb.weight;
162 if (RCAND_ATYPE(rcand) == RATA_ILI && RCAND_VAL(rcand) != ilix) {
163 RCAND_OLOAD(rcand) = ilix;
164 }
165 } else if ((rcand = NME_RAT(nme = ILI_OPND(ilix, 2))) != 0) {
166 if (RCAND_MSIZE(rcand) != msize) {
167 /*
168 * the nme was entered by a store and then another type of
169 * load ILI followed, or just another type of load ILI
170 * occurred. This occurs when type casting the & of a
171 * variable and then using the lvalue in a context which is
172 * inconsistent with its data type.
173 */
174 RCAND_CONFL(rcand) = 1;
175 } else {
176 /*
177 * the store for this load has already been processed. enter
178 * the load ili into the candidate list and back fill the ili
179 */
180 ILI_RAT(ilix) = rcand;
181 if (RCAND_ATYPE(rcand) == RATA_ILI && RCAND_VAL(rcand) != ilix) {
182 RCAND_OLOAD(rcand) = ilix;
183 } else {
184 RCAND_VAL(rcand) = ilix;
185 RCAND_ATYPE(rcand) = RATA_ILI;
186 }
187 RCAND_COUNT(rcand) += rcandb.weight;
188 }
189 } else {
190
191 /*
192 * the ili is a new entry; enter it into the appropriate
193 * candidate list and back fill the ili and its names entry
194 */
195 atype = RATA_ILI;
196 goto add_entry;
197 }
198 break;
199
200 add_entry: /* add an entry to the candidate table; there
201 * are three variables -- rtype, atype, and
202 * ilix
203 */
204 GET_RCAND(rcand);
205 RCAND_NEXT(rcand) = reg[rtype].rcand;
206 reg[rtype].rcand = rcand;
207 RCAND_RTYPE(rcand) = rtype;
208 RCAND_ATYPE(rcand) = atype;
209 RCAND_MSIZE(rcand) = msize;
210 RCAND_OLOAD(rcand) = 0;
211
212 if (atype != RATA_NME) { /* for loads, back fill the ili */
213 ILI_RAT(ilix) = rcand;
214 RCAND_STORE(rcand) = 0;
215 } else /* for RATA_NME (stores) don't back fill the
216 * store ILI */
217 RCAND_STORE(rcand) = 1;
218
219 RCAND_VAL(rcand) = ilix;
220 NME_RAT(nme) = rcand;
221 RCAND_COUNT(rcand) = rcandb.weight;
222 break;
223
224 case IL_ST: /* store data register */
225 rtype = RATA_IR;
226 msize = ILI_OPND(ilix, 4);
227 goto ac_store;
228
229 case IL_STKR:
230 rtype = RATA_KR;
231 msize = MSZ_I8;
232 goto ac_store;
233
234 case IL_STA: /* store address register */
235 rtype = RATA_AR;
236 msize = MSZ_PTR;
237 goto ac_store;
238 case IL_STQ: /* store m128 */
239 rtype = RATA_DP;
240 msize = MSZ_F16;
241 goto ac_store;
242 case IL_ST256: /* store m256 */
243 rtype = RATA_DP;
244 msize = MSZ_F32;
245 goto ac_store;
246
247 case IL_STSP: /* store single precision */
248 case IL_SSTS_SCALAR:
249 rtype = RATA_SP;
250 msize = MSZ_F4;
251 goto ac_store;
252 case IL_STSCMPLX:
253 rtype = RATA_CSP;
254 msize = MSZ_F8;
255 goto ac_store;
256 case IL_STDCMPLX:
257 rtype = RATA_CDP;
258 msize = MSZ_F16;
259 goto ac_store;
260 case IL_STDP: /* store double precision */
261 case IL_DSTS_SCALAR:
262 rtype = RATA_DP;
263 msize = MSZ_F8;
264 ac_store: /* common entry for the stores */
265 if ((rcand = NME_RAT(nme = ILI_OPND(ilix, 3))) != 0)
266 if (RCAND_MSIZE(rcand) != msize) {
267 /* this store conflicts with the existing candidate */
268 RCAND_CONFL(rcand) = 1;
269 } else {
270 /*
271 * the store has already been added to the candidate list
272 * (denoted by the rat field of its names entry)
273 */
274 RCAND_COUNT(rcand) += rcandb.weight;
275 RCAND_STORE(rcand) = 1;
276 }
277 else {
278
279 /*
280 * the store is a new entry; enter the names entry into the
281 * candidate list (the candidate type is names) and back fill the
282 * name entry
283 */
284 atype = RATA_NME;
285 goto add_entry;
286 }
287 break;
288 case IL_ACON:
289 rtype = RATA_AR;
290 msize = MSZ_PTR;
291 goto add_constant;
292 case IL_ICON:
293 rtype = RATA_IR;
294 msize = MSZ_WORD;
295 goto add_constant;
296 case IL_KCON:
297 rtype = RATA_KR;
298 msize = MSZ_I8;
299 goto add_constant;
300 case IL_FCON:
301 rtype = RATA_SP;
302 msize = MSZ_F4;
303 goto add_constant;
304 case IL_SCMPLXCON:
305 rtype = RATA_CSP;
306 msize = MSZ_F8;
307 goto add_constant;
308 case IL_DCMPLXCON:
309 rtype = RATA_CDP;
310 msize = MSZ_F16;
311 goto add_constant;
312 case IL_DCON:
313 rtype = RATA_DP;
314 msize = MSZ_F8;
315 add_constant:
316 rcand = ILI_RAT(ilix);
317 if (rcand) {
318 RCAND_COUNT(rcand) += rcandb.weight;
319 } else {
320 GET_RCAND(rcand); /* RCAND_OK = 0; prove it's ok to assn reg */
321 RCAND_RTYPE(rcand) = rtype;
322 RCAND_ATYPE(rcand) = RATA_CONST;
323 RCAND_MSIZE(rcand) = msize;
324 RCAND_VAL(rcand) = ilix;
325 RCAND_COUNT(rcand) = rcandb.weight;
326 ILI_RAT(ilix) = rcand;
327 RCAND_NEXT(rcand) = reg[rtype].rcand;
328 reg[rtype].rcand = rcand;
329 }
330 break;
331
332#ifdef LONG_DOUBLE_FLOAT128
333 case IL_FLOAT128CON:
334 case IL_FLOAT128LD:
335 case IL_FLOAT128ST:
336 /* float128 values are not register candidates. */
337 return;
338#endif /* LONG_DOUBLE_FLOAT128 */
339
340 default:
341 if (ILI_RAT(ilix) == 0) {
342 assert(ILI_RAT(ilix) != 0, "addrcand: no cand for ili", ilix, ERR_Severe);
343 return;
344 }
345 RCAND_COUNT(ILI_RAT(ilix)) += rcandb.weight;
346 break;
347 }
348}

350static void
351dump_cand(int candl)
352{
353 int temp, val, sym, i;
354 fprintf(gbl.dbgfil, "Type Value Count\n");
355 do {
356 val = RCAND_VAL(candl);
357 fprintf(gbl.dbgfil, "%4s ", atype_names[temp = RCAND_ATYPE(candl)]);
358 fprintf(gbl.dbgfil, "%-5u %4d", val, (int)RCAND_COUNT(candl));
359 sym = 0;
360 switch (temp) {
361 case RATA_NME:
362 fprintf(gbl.dbgfil, " - %-9s", ilis[temp = ILI_OPC(val)].name);
363 if (IL_TYPE(temp) == ILTY_STORE) {
364 fprintf(gbl.dbgfil, "\"");
365 sym = print_nme((int)ILI_OPND(val, 3));
366 fprintf(gbl.dbgfil, "\"");
367 }
368 break;
369 case RATA_IND:
370 fprintf(gbl.dbgfil, " - \"");
371 sym = print_nme((int)ILI_OPND(val, 2));
372 fprintf(gbl.dbgfil, "\"");
373 for (i = RCAND_TEMP(candl); i; i = RCAND_NEXT(i))
374 fprintf(gbl.dbgfil, ", %u^", RCAND_VAL(i));
375 break;
376 case RATA_ARR:
377 fprintf(gbl.dbgfil, " - \"");
378 sym = print_nme((int)ILI_OPND(val, 2));
379 fprintf(gbl.dbgfil, "\"");
380 i = RCAND_TEMP(candl);
381 fprintf(gbl.dbgfil, ", %u^ %u^", RCAND_VAL(i), RCAND_TEMP(i));
382 break;
383 case RATA_ILI:
384 fprintf(gbl.dbgfil, " - %-9s", ilis[temp = ILI_OPC(val)].name);
385 if (IL_TYPE(temp) == ILTY_LOAD) {
386 fprintf(gbl.dbgfil, "\"");
387 sym = print_nme((int)ILI_OPND(val, 2));
388 fprintf(gbl.dbgfil, "\"");
389 }
390 break;
391 case RATA_TEMP:
392 fprintf(gbl.dbgfil, " - %-9s", ilis[ILI_OPC(val)].name);
393 fprintf(gbl.dbgfil, " \"%s\"", getprint((int)RCAND_TEMP(candl)));
394 break;
395 case RATA_RPL:
396 fprintf(gbl.dbgfil, " - %-9s", ilis[ILI_OPC(val)].name);
397 fprintf(gbl.dbgfil, " %u^", RCAND_TEMP(candl));
398 break;
399 case RATA_CONST:
400 fprintf(gbl.dbgfil, " - %-9s", ilis[ILI_OPC(val)].name);
401 temp = ILI_OPND(val, 1);
402 if (ILI_OPC(val) != IL_ACON)
403 fprintf(gbl.dbgfil, " \"%s\"", getprint(temp));
404 else {
405 if (CONVAL1G(temp))
406 fprintf(gbl.dbgfil, " \"%s", getprint((int)CONVAL1G(temp)));
407 else
408 fprintf(gbl.dbgfil, " \"%d", CONVAL1G(temp));
409 fprintf(gbl.dbgfil, ",%" ISZ_PF "d\"", ACONOFFG(temp));
410 }
411 break;
412 }
413 if (RCAND_RTYPE(candl) != RATA_VECT)
414 fprintf(gbl.dbgfil, " <%s>", get_msize(RCAND_MSIZE(candl)));
415 else
416 fprintf(gbl.dbgfil, " <DT %d>", RCAND_MSIZE(candl));
417 if (RCAND_CONFL(candl))
418 fprintf(gbl.dbgfil, "<confl>");
419 if (sym) {
420 if (ADDRTKNG(sym))
421 fprintf(gbl.dbgfil, "<&>");
422 if (RCAND_STORE(candl))
423 fprintf(gbl.dbgfil, "<stored>");
424 }
425 if (RCAND_OK(candl))
426 fprintf(gbl.dbgfil, "<cnst ok>");
427 if (RCAND_NOREG(candl))
428 fprintf(gbl.dbgfil, "<noreg>");
429 if (RCAND_CSE(candl))
430 fprintf(gbl.dbgfil, "<cse>");
431 if (RCAND_IGNORE(candl))
432 fprintf(gbl.dbgfil, "<ignore>");
433 if (RCAND_INV(candl))
434 fprintf(gbl.dbgfil, "<inv>");
435 fprintf(gbl.dbgfil, "\n");
436 candl = RCAND_NEXT(candl);
437 } while (candl != 0);
438 fflush(gbl.dbgfil);
439}

624/**
625 \brief Get a candidate from the candidate list, candl.
626 */
627int
628getrcand(int candl)
629{
630 int cand, count;
631
632 cand = 0; /* value of getrcand if a candidate is not
633 * found
634 */
635 count = GR_THRESHOLD; /* a candidate must have a count greater than
636 * this value (defined in machreg.h)
637 */
638
639 /*
640 * scan through the candidate list to find a the candidate with the
641 * maximum count
642 */
643 for (; candl; candl = RCAND_NEXT(candl))
644 if (RCAND_COUNT(candl) > count) {
645
646 /*
647 * if the candidate variable was used in contexts which conflicts
648 * with its true register type, get the next candidate
649 */
650 if (RCAND_CONFL(candl)) {
651 RCAND_COUNT(candl) = 0;
652 continue;
653 }
654 cand = candl; /* potential candidate */
655 count = RCAND_COUNT(cand); /* new maximum count */
656 }
657 rcandb.count = RCAND_COUNT(cand);
658 RCAND_COUNT(cand) = 0; /* clear the candidate's count. this
659 * "removes" the candidate from the list
660 */
661 return (cand);
662}

664/**
665 \brief Go through all of the candidate lists to reinitialize the back
666 pointers of the names entries and ILI.
667
668 The regs' rcand fields are set to null, and the register candidate list is
669 reset.
670 */
671void
672endrcand(void)
673{
674 int rtype, cand, val, i;

676 for (rtype = 0; rtype <= RATA_ALL; rtype++) {

820/* Register Temporary Section -
821 These routines provide a mechanism to create temporaries during
822 the expansion of ILM blocks which can be re-used for different
823 ILM blocks and are unique across functions. The scenario (at
824 opt 0 or 1) is:
825 1. At the beginning of an ILM block, initialize (mkrtemp_init).
826 2. Create temporaries during expansion by calling mkrtemp.
827 3. At the end of a function, ensure that the temporaries created
828 for the function are unique from those created in the next
829 function (mkrtemp_end).
830
831 For opt 2, the scenario for the loops in a function is:
832 0. At the beginning of a function, mkrtemp_copy
833 1. For innermost loops, mkrtemp_init. Otherwise, use
834 mkrtemp_reinit (this gets the values created from calls to
835 mkrtemp_update of all contained loops).
836 2. During optimizations, use mkrtemp.
837 3. At the end of optimizing a loop, mkrtemp_update for the
838 parent of the loop. This will keep track of the rtemps used
839 in contained loops.
840 4. At the end of a function, mkrtemp_end();
841
842 NOTE: any change in the number of elements in the rtemp array
843 must also be applied to the macro RTEMPS which is defined in
844 regutil.h.
845*/

861 {'k', "ka", DT_DCMPLX, 0, 0, -1}, /* 6: double complex temps */

863 {'v', "va", DT_NONE, 0, 0, -1}, /* 8: vector temps */
