15/* advance index n elements and return the remaining extent in the
16 first dimension from that point. When the end of the array is
17 reached, index is reset to the first element and 0 is returned. */
18
19static int I8(advance)(F90_Desc *d, __INT_T *index, __INT_T n)
20{
21 __INT_T i, r;
22
23 index[0] += n;
24 if (index[0] <= DIM_UBOUND_G(d, 0))
25 return DIM_UBOUND_G(d, 0) - index[0] + 1;
26 else if (index[0] > DIM_UBOUND_G(d, 0) + 1)
27 __fort_abort("RESHAPE: internal error, advance past ubound");
28 else
29 index[0] = F90_DIM_LBOUND_G(d, 0);
30 r = F90_RANK_G(d);
31 for (i = 1; i < r; ++i) {
32 index[i]++;
33 if (index[i] <= DIM_UBOUND_G(d, i))
34 return DIM_UBOUND_G(d, 0) - index[0] + 1;
35 else
36 index[i] = F90_DIM_LBOUND_G(d, i);
37 }
38 return 0;
39}

41/* note: dimensions in order vector are zero-based. */
42
43static int I8(advance_permuted)(F90_Desc *d, __INT_T *index, int *order,
44 __INT_T n)
45{
46 int j, k;
47 __INT_T i, r;
48
49 k = order[0];
50 index[k] += n;
51 if (index[k] <= DIM_UBOUND_G(d, k))
52 return DIM_UBOUND_G(d, k) - index[k] + 1;
53 else if (index[k] > DIM_UBOUND_G(d, k) + 1)
54 __fort_abort("RESHAPE: internal error, advance past ubound");
55 else
56 index[k] = F90_DIM_LBOUND_G(d, k);
57 r = F90_RANK_G(d);
58 for (i = 1; i < r; i++) {
59 j = order[i];
60 index[j]++;
61 if (index[j] <= DIM_UBOUND_G(d, j))
62 return DIM_UBOUND_G(d, k) - index[k] + 1;
63 else
64 index[j] = F90_DIM_LBOUND_G(d, j);
65 }
66 return 0;
67}

71void ENTFTN(RESHAPE, reshape)(char *resb, /* result base */
72 char *srcb, /* source base */
73 char *shpb, /* shape base */
74 char *padb, /* pad base */
75 char *ordb, /* order base */
76 F90_Desc *resd, /* result descriptor */
77 F90_Desc *srcd, /* source descriptor */
78 F90_Desc *shpd, /* shape descriptor */
79 F90_Desc *padd, /* pad descriptor */
80 F90_Desc *ordd) /* order descriptor */
81{
82 __INT_T resx[MAXDIMS];
83 __INT_T srcx[MAXDIMS];
84 __INT_T padx[MAXDIMS];
85 int shape[MAXDIMS];
86 int order[MAXDIMS];
87 DECL_HDR_VARS(fromd);
88 DECL_HDR_VARS(tod);
89 char *fromb, *tob;
90 chdr *ch;
91 __INT_T more_res, more_src, more_pad, n;
92 int i, j, k, m, r;
93
94#if defined(DEBUG)
95 if (resd == NULL || F90_TAG_G(resd) != __DESC)
96 __fort_abort("RESHAPE: invalid result descriptor");
97 if (srcd == NULL || F90_TAG_G(srcd) != __DESC)
98 __fort_abort("RESHAPE: invalid SOURCE descriptor");
99 if (shpd == NULL || F90_TAG_G(shpd) != __DESC)
100 __fort_abort("RESHAPE: invalid SHAPE descriptor");
101 if (padd == NULL || F90_TAG_G(padd) != __DESC && F90_TAG_G(padd) != __NONE)
102 __fort_abort("RESHAPE: invalid PAD descriptor");
103 if (ordd == NULL || F90_TAG_G(ordd) != __DESC && F90_TAG_G(ordd) != __NONE)
104 __fort_abort("RESHAPE: invalid ORDER descriptor");
105#endif
106
107 if (F90_KIND_G(resd) != F90_KIND_G(srcd) ||
108 F90_LEN_G(resd) != F90_LEN_G(srcd))
109 __fort_abort("RESHAPE: result type != SOURCE type");
110 if (F90_TAG_G(padd) == __DESC && (F90_KIND_G(padd) != F90_KIND_G(srcd) ||
111 F90_LEN_G(padd) != F90_LEN_G(srcd)))
112 __fort_abort("RESHAPE: PAD type != SOURCE type");
113
114 /* don't really need the shape vector because the shape is already
115 set in the result descriptor, but check its validity anyway */
116
117 if (F90_RANK_G(shpd) <= 0)
118 __fort_abort("RESHAPE: invalid SHAPE argument");
119
120 r = DIM_UBOUND_G(shpd, 0) - F90_DIM_LBOUND_G(shpd, 0) + 1;
121 if (r < 0 || r > MAXDIMS || r != F90_RANK_G(resd))
122 __fort_abort("RESHAPE: invalid SHAPE argument");
123
124 I8(__fort_fetch_int_vector)(shpb, shpd, shape, r);
125 for (i = r; --i >= 0;) {
126 if (shape[i] < 0)
127 __fort_abort("RESHAPE: invalid SHAPE argument");
128 }
129
130 /* get the order vector */
131
132 if (F90_TAG_G(ordd) == __DESC) {
133 I8(__fort_fetch_int_vector)(ordb, ordd, order, r);
134 m = 0;
135 for (i = r; --i >= 0;) {
136 if (order[i] < 1 || order[i] > r)
137 __fort_abort("RESHAPE: invalid ORDER argument");
138 --order[i]; /* zero-based dimension number */
139 m |= 1 << order[i];
140 }
141 if (m != ~(-1 << r))
142 __fort_abort("RESHAPE: invalid ORDER argument");
143 } else { /* absent */
144 for (i = r; --i >= 0;)
145 order[i] = i;
146 }
147
148 /* initialize indices and first column extents */
149
150 if (F90_GSIZE_G(resd) <= 0)
151 return;
152 for (i = r; --i >= 0;)
153 resx[i] = F90_DIM_LBOUND_G(resd, i);
154 k = order[0];
155 more_res = DIM_UBOUND_G(resd, k) - F90_DIM_LBOUND_G(resd, k) + 1;
156
157 if (F90_GSIZE_G(srcd) > 0) {
158 for (i = F90_RANK_G(srcd); --i >= 0;)
159 srcx[i] = F90_DIM_LBOUND_G(srcd, i);
160 more_src = DIM_UBOUND_G(srcd, 0) - F90_DIM_LBOUND_G(srcd, 0) + 1;
161 } else
162 more_src = 0;
163
164 if (F90_TAG_G(padd) == __DESC && F90_GSIZE_G(padd) > 0) {
165 for (i = F90_RANK_G(padd); --i >= 0;)
166 padx[i] = F90_DIM_LBOUND_G(padd, i);
167 more_pad = DIM_UBOUND_G(padd, 0) - F90_DIM_LBOUND_G(padd, 0) + 1;
168 } else
169 more_pad = 0;
170
171 /* loop -- transfer matching column vector sections and advance
172 indices until result array is filled */
173
174 while (more_res) {
175
176 if (more_src) {
177 n = Min(more_src, more_res);
178
179 __DIST_INIT_SECTION(fromd, 1, srcd);
180 I8(__fort_set_section)(fromd, 1, srcd, 1, srcx[0], srcx[0] + n - 1, 1);
181 for (i = 1; i < F90_RANK_G(srcd); ++i)
182 I8(__fort_set_single)(fromd, srcd, i + 1, srcx[i], __SCALAR);
183 I8(__fort_finish_section)(fromd);
184 fromb = srcb;
185
186 more_src = I8(advance)(srcd, srcx, n);
187 } else if (more_pad) {
188 n = Min(more_pad, more_res);
189
190 fromb = padb;
191 __DIST_INIT_SECTION(fromd, 1, padd);
192 I8(__fort_set_section)(fromd, 1, padd, 1, padx[0], padx[0] + n - 1, 1);
193 for (i = 1; i < F90_RANK_G(padd); ++i)
194 I8(__fort_set_single)(fromd, padd, i + 1, padx[i], __SCALAR);
195 I8(__fort_finish_section)(fromd);
196
197 more_pad = I8(advance)(padd, padx, n);
198 if (!more_pad) /* start over if end reached */
199 more_pad = DIM_UBOUND_G(padd, 0) - F90_DIM_LBOUND_G(padd, 0) + 1;
200 } else
201 __fort_abort("RESHAPE: not enough elements in SOURCE array");
202
203 __DIST_INIT_SECTION(tod, 1, resd);
204 I8(__fort_set_section)(tod, 1, resd, k + 1, resx[k], resx[k] + n - 1, 1);
205 for (i = 1; i < F90_RANK_G(resd); ++i) {
206 j = order[i];
207 I8(__fort_set_single)(tod, resd, j + 1, resx[j], __SCALAR);
208 }
209 I8(__fort_finish_section)(tod);
210
211 fromb += DIST_SCOFF_G(fromd) * F90_LEN_G(fromd);
212 tob = resb + DIST_SCOFF_G(tod) * F90_LEN_G(tod);
213
214 ch = I8(__fort_copy)(tob, fromb, tod, fromd, NULL);
215 __fort_doit(ch);
216 __fort_frechn(ch);
217
218 more_res = I8(advance_permuted)(resd, resx, order, n);
219 }
220}
