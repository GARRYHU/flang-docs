33/** \brief
34 * __fortio_assign - store a value found during list-directed or
35 * namelist read. return 0 if successful, otherwise an
36 * error code (typically, a conversion error).
37 */
38int
39__fortio_assign(char *item, /* where to store */
40 int type, /* data type of item (as in pghpft.h) */
41 __CLEN_T item_length, /* number of chars if type == __STR */
42 AVAL *valp /* value to store */
43)
44{
45 __CLEN_T len;
46 AVAL *lp;
47
48 switch (valp->dtype) {
49 case __BIGCPLX:
50 lp = valp->val.cmplx;
51 goto assn_shared;
52 case __INT8:
53 if (__ftn_32in64_)
54 I64_MSH(valp->val.i8) = 0;
55 FLANG_FALLTHROUGH;
56 case __BIGINT:
57 case __BIGREAL:
58 lp = valp;
59 assn_shared:
60 switch (type) {
61 default:
62 assert(0);
63 break;
64 case __INT1:
65 PP_INT1(item) = to_bigint(lp);
66 break;
67 case __INT2:
68 PP_INT2(item) = to_bigint(lp);
69 break;
70 case __INT4:
71 PP_INT4(item) = to_bigint(lp);
72 break;
73 case __INT8:
74 PP_INT8(item) = to_int8(lp);
75 break;
76 case __REAL4:
77 PP_REAL4(item) = (__REAL4_T)to_bigreal(lp);
78 break;
79 case __REAL8:
80 PP_REAL8(item) = (__REAL8_T)to_bigreal(lp);
81 break;
82 case __REAL16:
83 PP_REAL16(item) = (__REAL16_T)to_bigreal(lp);
84 break;
85 case __WORD16:
86 return FIO_EQUAD;
87 case __CPLX8:
88 PP_REAL4(item) = (__REAL4_T)to_bigreal(lp);
89 if (valp->dtype == __BIGCPLX)
90 PP_REAL4(item + (sizeof(__CPLX8_T) >> 1)) =
91 (__REAL4_T)to_bigreal(lp + 1);
92 else
93 PP_REAL4(item + (sizeof(__CPLX8_T) >> 1)) = 0;
94 break;
95 case __CPLX16:
96 PP_REAL8(item) = (__REAL8_T)to_bigreal(lp);
97 if (valp->dtype == __BIGCPLX)
98 PP_REAL8(item + (sizeof(__CPLX16_T) >> 1)) =
99 (__REAL8_T)to_bigreal(lp + 1);
100 else
101 PP_REAL8(item + (sizeof(__CPLX16_T) >> 1)) = 0;
102 break;
103 case __CPLX32:
104 PP_REAL16(item) = (__REAL16_T)to_bigreal(lp);
105 if (valp->dtype == __BIGCPLX)
106 PP_REAL16(item + (sizeof(__CPLX32_T) >> 1)) =
107 (__REAL16_T)to_bigreal(lp + 1);
108 else
109 PP_REAL16(item + (sizeof(__CPLX32_T) >> 1)) = 0;
110 break;
111 case __LOG1:
112 PP_LOG1(item) = to_bigint(lp);
113 break;
114 case __LOG2:
115 PP_LOG2(item) = to_bigint(lp);
116 break;
117 case __LOG4:
118 PP_LOG4(item) = to_bigint(lp);
119 break;
120 case __LOG8:
121 PP_LOG8(item) = to_log8(lp);
122 break;
123 case __STR:
124 case __NCHAR:
125 goto assn_err;
126 }
127 break;
128
129 case __BIGLOG:
130 switch (type) {
131 default:
132 assert(0);
133 break;
134 case __INT1:
135 PP_INT1(item) = valp->val.i;
136 break;
137 case __INT2:
138 PP_INT2(item) = valp->val.i;
139 break;
140 case __INT4:
141 PP_INT4(item) = valp->val.i;
142 break;
143 case __INT8:
144 PP_INT8(item) = valp->val.i;
145 break;
146 case __REAL4:
147 PP_REAL4(item) = (__REAL4_T)valp->val.i;
148 break;
149 case __REAL8:
150 PP_REAL8(item) = (__REAL8_T)valp->val.i;
151 break;
152 case __REAL16:
153 PP_REAL16(item) = (__REAL16_T)valp->val.i;
154 break;
155 case __WORD16:
156 return FIO_EQUAD;
157 case __CPLX8:
158 PP_REAL4(item) = (__REAL4_T)valp->val.i;
159 PP_REAL4(item + (sizeof(__CPLX8_T) >> 1)) = 0;
160 break;
161 case __CPLX16:
162 PP_REAL8(item) = valp->val.i;
163 PP_REAL8(item + (sizeof(__CPLX16_T) >> 1)) = 0;
164 break;
165 case __CPLX32:
166 PP_REAL16(item) = valp->val.i;
167 PP_REAL16(item + (sizeof(__CPLX32_T) >> 1)) = 0;
168 break;
169 case __LOG1:
170 PP_LOG1(item) = valp->val.i;
171 break;
172 case __LOG2:
173 PP_LOG2(item) = valp->val.i;
174 break;
175 case __LOG4:
176 PP_LOG4(item) = valp->val.i;
177 break;
178 case __LOG8:
179 PP_LOG8(item) = to_log8(valp);
180 break;
181 case __STR:
182 case __NCHAR:
183 goto assn_err;
184 }
185 break;
186
187 case __LOG8:
188 switch (type) {
189 default:
190 assert(0);
191 break;
192 case __INT1:
193 PP_INT1(item) = I64_LSH(valp->val.i8);
194 break;
195 case __INT2:
196 PP_INT2(item) = I64_LSH(valp->val.i8);
197 break;
198 case __INT4:
199 PP_INT4(item) = I64_LSH(valp->val.i8);
200 break;
201 case __INT8:
202 if (__ftn_32in64_)
203 I64_MSH(valp->val.i8) = 0;
204 PP_INT4(item) = valp->val.i8[0];
205 PP_INT4(item + 4) = valp->val.i8[1];
206 break;
207 case __REAL4:
208 PP_REAL4(item) = (float)I64_LSH(valp->val.i8);
209 break;
210 case __REAL8:
211 PP_REAL8(item) = I64_LSH(valp->val.i8);
212 break;
213 case __REAL16:
214 case __WORD16:
215 return FIO_EQUAD;
216 case __CPLX8:
217 PP_REAL4(item) = (float)I64_LSH(valp->val.i8);
218 PP_REAL4(item + 4) = 0;
219 break;
220 case __CPLX16:
221 PP_REAL8(item) = I64_LSH(valp->val.i8);
222 PP_REAL8(item + 8) = 0;
223 break;
224 case __CPLX32:
225 return FIO_EQUAD;
226 case __LOG1:
227 PP_LOG1(item) = I64_LSH(valp->val.i8);
228 break;
229 case __LOG2:
230 PP_LOG2(item) = I64_LSH(valp->val.i8);
231 break;
232 case __LOG4:
233 PP_LOG4(item) = I64_LSH(valp->val.i8);
234 break;
235 case __LOG8:
236 PP_LOG8(item) = to_log8(valp);
237 break;
238 case __STR:
239 case __NCHAR:
240 goto assn_err;
241 }
242 break;
243
244 case __STR:
245 if (type != __STR)
246 goto assn_err;
247 len = valp->val.c.len;
248 if (len > item_length)
249 len = item_length;
250 (void)memcpy(item, valp->val.c.str, len);
251 if (len < item_length)
252 (void)memset(item + len, ' ', item_length - len);
253 break;
254
255 default:
256 assert(0);
257 break;
258 }
259
260 return 0;
261assn_err:
262 return FIO_EERR_DATA_CONVERSION;
263}

267static __BIGINT_T
268to_bigint(AVAL *valp)
269{
270 if (valp->dtype == __BIGREAL)
271 return (__BIGINT_T)valp->val.d;
272 assert(valp->dtype == __BIGINT || valp->dtype == __BIGLOG);
273 return (__BIGINT_T)valp->val.i;
274}
275
276static __BIGREAL_T
277to_bigreal(AVAL *valp)
278{
279 if (valp->dtype == __BIGREAL)
280 return valp->val.d;
281 if (valp->dtype == __INT8 || valp->dtype == __LOG8) {
282 return (__BIGREAL_T)valp->val.i8v;
283 }
284 assert(valp->dtype == __BIGINT || valp->dtype == __BIGLOG);
285 return (__BIGREAL_T)valp->val.i;
286}
287
288static __INT8_T
289to_int8(AVAL *valp)
290{
291 if (valp->dtype == __INT4 || valp->dtype == __LOG4) {
292 I64_LSH(valp->val.i8) = valp->val.i;
293 if (valp->val.i < 0)
294 I64_MSH(valp->val.i8) = 0xFFFFFFFF;
295 else
296 I64_MSH(valp->val.i8) = 0;
297 } else if (valp->dtype == __BIGREAL) {
298 return (__INT8_T)valp->val.d;
299 }
300 return valp->val.i8v;
301}
302
303static __LOG8_T
304to_log8(AVAL *valp)
305{
306 if (valp->dtype == __INT4 || valp->dtype == __LOG4) {
307 I64_LSH(valp->val.ui8) = valp->val.i;
308 I64_MSH(valp->val.ui8) = 0;
309 }
310 return (__LOG8_T)valp->val.ui8v;
311}
