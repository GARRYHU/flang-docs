46/* For saving the directives for the backends of the native compilers */

48typedef struct {
49 int beg_line; /* beginning line # of loop */
50 int end_line; /* ending line # of loop */
51 DIRSET change; /* what's changed */
52 DIRSET newset; /* dirset state for the routine & loops */
53 /* How an entry in newset is propagated depends on the
54 * type (value vs bit vector) of the entry:
55 * - if value:
56 * change == nonzero => propagate the value
57 * - if a bit vector:
58 * change == nonzero => for each bit which is
59 * set, the new value of the corresponding bit
60 * position is propagated. All other bits are
61 * left untouched.
62 * The change value is computed as
63 * change = new ^ old
64 * The new value of the bit vector is
65 * computed as
66 * val = (new & change) | (current & !change)
67 */
68} SVDIR;
69
70static struct {
71 SVDIR *stgb; /* [0] is reserved for the routine's directives.
72 * [>=1] used for the loops' directives.
73 */
74 int size;
75 int avail; /* avail is actually tracked as direct.lpg.avail;
76 * svdir.avail is needed for direct_export()
77 */
78} svdir = {NULL, 0, 0};

87/** \brief Initialize directive structure
88 *
89 * Initialize directive structure which is global for the
90 * source file and the structure which is used per routine.
91 * must be called just once for the compilation and after all of the
92 * command line options have been processed.
93 *
94 * The initial values of the global structure are extracted from
95 * the command line options. The routine structure is initialized
96 * with values from the global structure.
97 */
98void
99direct_init(void)
100{
101
102/* Set/clear any xbits for which the command line processing has no
103 * effect
104 */
105 flg.x[8] |= 0x8; /* disable global reg assignment */
106
107 store_dirset(&direct.gbl);
108
109#ifdef FE90
110 inigbl = direct.gbl;
111#endif
112 direct.rou = direct.gbl;
113 direct.loop = direct.gbl;
114 direct.rou_begin = direct.gbl;
115
116 direct.loop_flag = false; /* seen pragma with loop scope */
117 direct.in_loop = false; /* in loop with pragmas */
118 direct.carry_fwd = false;
119
120 direct.avail = 0;
121 NEW(direct.stgb, DIRSET, (direct.size = 16));
122
123 direct.lpg.avail = 1;
124
125 NEW(direct.lpg.stgb, LPPRG, (direct.lpg.size = 16));
126 BZERO(direct.lpg.stgb, LPPRG, 1);
127 direct.lpg_stk.top = 0;
128 NEW(direct.lpg_stk.stgb, LPG_STK, (direct.lpg_stk.size = 8));
129
130#ifdef FE90
131 direct.dynlpg.avail = 1;
132 NEW(direct.dynlpg.stgb, LPPRG, (direct.dynlpg.size = 16));
133 if (flg.genilm) {
134 /* [0] is reserved for the 'routine' directives */
135 NEW(svdir.stgb, SVDIR, (svdir.size = 16));
136 BZERO(svdir.stgb, SVDIR, 1); /* init [0] to zero */
137 }
138 direct.indep = NULL;
139 direct.index_reuse_list = NULL;
140#endif
141
142}

171/** \brief Re-initialize the routine structure
172 *
173 * Must be called after the end of a function is processed by semant and
174 * before the next function is parsed. For C, this is when the END ilm is
175 * written for a function. For Fortran, this is after a subprogram has been
176 * processed by all phases of the compiler.
177 *
178 * For C, process any symbol-/variable- related pragmas which may have
179 * occurred. Also, save the index into the lpprg table which is the beginning
180 * of the function's lpprg segment; mark the end of the segment with an entry
181 * whose beg_line is -1.
182 */
183void
184direct_rou_end(void)
185{
186/* CPLUS also needs to save routine's structure: */
187#ifdef FE90
188 if (flg.genilm) {
189 update_rou_begin();
190 }
191#endif
192 direct.lpg.avail = 1;
193
194 direct.rou = direct.gbl;
195 direct.loop = direct.gbl;
196 direct.rou_begin = direct.gbl;
197 direct.carry_fwd = false;
198#ifdef FE90
199 direct.dynlpg.avail = 1;
200#endif
201
202}

204void
205direct_loop_enter(void)
206{
207 if (direct.loop_flag || (direct.carry_fwd && !direct.in_loop)) {
208 push_lpprg(gbl.lineno);
209#ifdef FE90
210 if (flg.genilm) {
211 NEEDB(direct.lpg.avail, svdir.stgb, SVDIR, svdir.size,
212 direct.lpg.avail + 8);
213 diff_dir(&svdir.stgb[direct.lpg.avail - 1],
214 &direct.lpg.stgb[direct.lpg.avail - 1].dirset, &direct.loop);
215 }
216#endif
217 }

221/** \brief Re-initialize the loop structure
222 *
223 * Must be called after the end of a loop is processed by semant for which
224 * loop-scoped pragmas/directives apply.
225 */
226void
227direct_loop_end(int beg_line, int end_line)
228{
229 int i;
230 LPPRG *lpprg;
231
232 if (!direct.in_loop)
233 return;
234 i = direct.lpg_stk.stgb[direct.lpg_stk.top].dirx;
235 lpprg = direct.lpg.stgb + i;
236 if (lpprg->beg_line != beg_line)
237 return;
238
239/***** pop_lpprg *****/
240
241 TR1("---pop_lpprg: top %d,", direct.lpg_stk.top);
242 direct.lpg_stk.top--;
243 TR1(" lpprg %d,", i);
244 lpprg = direct.lpg.stgb + i;
245
246 lpprg->end_line = end_line;
247 TR2(" beg %d, end %d\n", lpprg->beg_line, lpprg->end_line);
248
249 direct.loop = direct.rou;
250
251#ifdef FE90
252 if (flg.genilm) {
253 svdir.stgb[i].beg_line = lpprg->beg_line;
254 svdir.stgb[i].end_line = lpprg->end_line;
255 }
256#endif
257
258 if (direct.lpg_stk.top == 0) {
259 direct.loop_flag = false;
260 direct.in_loop = false;
261 } else if (XBIT(59, 1)) {
262 direct.loop =
263 direct.lpg.stgb[direct.lpg_stk.stgb[direct.lpg_stk.top].dirx].dirset;
264 } else {
265 /*
266 * propagate selected directives/pragmas to all nested
267 */
268 i = direct.lpg_stk.stgb[direct.lpg_stk.top].dirx;
269 direct.loop.depchk = direct.lpg.stgb[i].dirset.depchk;
270 }
271#if DEBUG
272 if (DBGBIT(1, 512))
273 dmp_lpprg(direct.lpg_stk.stgb[direct.lpg_stk.top + 1].dirx);
274#endif
275
276}

331/** \brief Load direct.rou for the current function
332 *
333 * Called after the parse phase is complete; called once per function. For C
334 * this means the call occurs during expand when it sees the ENTRY ilm; for
335 * Fortran, this is at the beginning of expand (in main).
336 *
337 * DIRSET direct.rou_begin represents the state of the directives/pragmas at
338 * the beginning of the function.
339 *
340 * \param func - symbol of current function
341 */
342void
343direct_rou_load(int func)
344{
345 DIRSET *currdir;
346 XF *xfp;
347 char *fnp;
348
349 currdir = &direct.rou_begin;
350
351 load_dirset(currdir);
352
353 fnp = SYMNAME(gbl.currsub);
354 for (xfp = xf_p; xfp != NULL; xfp = xfp->next) {
355 if (strcmp(xfp->fn, fnp) == 0) {
356 /*printf("-xf %s %d 0x%x\n", xfp->fn, xfp->x, xfp->v);*/
357 set_xflag(xfp->x, xfp->v);
358 currdir->x[xfp->x] = flg.x[xfp->x];
359 }
360 }
361 for (xfp = yf_p; xfp != NULL; xfp = xfp->next) {
362 if (strcmp(xfp->fn, fnp) == 0) {
363 /*printf("-yf %s %d 0x%x\n", xfp->fn, xfp->x, xfp->v);*/
364 set_yflag(xfp->x, xfp->v);
365 currdir->x[xfp->x] = flg.x[xfp->x];
366 }
367 }
368
369#ifndef FE90
370 /*
371 * the optimizer doesn't handle assigned goto's correctly.
372 * (Doesn't know where to put loop exit code if you assign
373 * goto out of loop)
374 */
375 if ((gbl.asgnlbls == -1) && (flg.opt >= 2)) {
376 error(I_0127_Optimization_level_for_OP1_changed_to_opt_1_OP2, ERR_Informational, 0, SYMNAME(gbl.currsub), "due to assigned goto");
377 currdir->opt = flg.opt = 1;
378 currdir->vect = flg.vect = 0;
379 }
380 if (gbl.vfrets) {
381 /*
382 * temporarily disable optimizations not correctly
383 * handle if variable functions occur.
384 */
385 if (flg.opt >= 2) {
386 error(I_0127_Optimization_level_for_OP1_changed_to_opt_1_OP2, ERR_Informational, 0, SYMNAME(gbl.currsub), "due to < > in FORMAT");
387 currdir->opt = flg.opt = 1;
388 currdir->vect = flg.vect = 0;
389 }
390 flg.x[8] |= 0x8; /* no globalregs at opt 1 */
391 }
392#endif
393
394#if DEBUG
395 if (DBGBIT(1, 256)) {
396 fprintf(gbl.dbgfil, "---dirset for func ");
397 fprintf(gbl.dbgfil, "%s\n", SYMNAME(func));
398 dmp_dirset(currdir);
399 }
400#endif
401
402#if (defined(TARGET_X86) || defined(TARGET_LLVM)) && !defined(FE90)
403 set_mach(&mach, direct.rou_begin.tpvalue[0]);
404#endif
405
406}

576static void
577update_rou_begin(void)
578{
579 SVDIR *df;
580 DIRSET *new, *old, *older;
581 int i;
582
583 df = &svdir.stgb[0];
584 new = &direct.rou_begin;
585 old = &direct.gbl;
586 older = &inigbl;
587
588 diff_dir(df, new, old);
589
590 if (old->opt != older->opt)
591 df->change.opt = 1;
592 if (old->vect ^ older->vect)
593 df->change.vect = 1;
594 if (old->depchk != older->depchk)
595 df->change.depchk = 1;
596 for (i = 0; i < sizeof(flg.x) / sizeof(int); i++) {
597 if (is_xflag_bit(i)) {
598 if (old->x[i] ^ older->x[i])
599 df->change.x[i] = 1;
600 } else {
601 if (old->x[i] != older->x[i])
602 df->change.x[i] = 1;
603 }
604 }
605 svdir.avail = direct.lpg.avail;
606}

608static void
609diff_dir(SVDIR *df, DIRSET *new, DIRSET *old)
610{
611 int i;
612
613 df->newset = *new;
614
615 df->change.opt = new->opt != old->opt;
616 df->change.vect = new->vect ^ old->vect;
617 df->change.depchk = new->depchk != old->depchk;
618 for (i = 0; i < sizeof(flg.x) / sizeof(int); i++) {
619 if (is_xflag_bit(i))
620 df->change.x[i] = new->x[i] ^ old->x[i];
621 else
622 df->change.x[i] = new->x[i] != old->x[i];
623 }
624}

724static int
725rd_dir(DIRSET *dd)
726{
727 int ret;
728 int v;
729 int change;
730 int idx;
731
732#undef ST_VAL
733#undef ST_BV
734#undef UADDR
735#define ST_VAL(m) \
736 if (change) \
737 dd->m = v
738#define ST_BV(m) \
739 if (change) \
740 dd->m = (v & change) | (dd->m & ~change)
741#define UADDR(x) (unsigned int *) & x
742
743 while (true) {
744 /* read input line */
745 if (read_line())
746 return 1;
747 switch (line[0]) {
748 case 'z':
749 return 0;
750 case 'o': /* read opt line */
751 ret = sscanf(line, "o:%x %x", UADDR(change), UADDR(v));
752 if (ret != 2)
753 return 1;
754 {
755 ST_VAL(opt);
756 }
757 break;
758 case 'v': /* read vect line */
759 ret = sscanf(line, "v:%x %x", UADDR(change), UADDR(v));
760 if (ret != 2)
761 return 1;
762 if (dd != &direct.rou_begin) {
763 ST_BV(vect);
764 } else {
765 ST_VAL(vect);
766 }
767 break;
768 case 'd': /* read depchk line */
769 ret = sscanf(line, "d:%x %x", UADDR(change), UADDR(v));
770 if (ret != 2)
771 return 1;
772 {
773 ST_VAL(depchk);
774 }
775 break;
776 case 'x':
777 /* read x flag. The line is of the form:
778 * x<n>:change new [idx:change new ]...
779 * <n> is in decimal; change & new are in hex.
780 */
781 ret = sscanf(line, "x%d:%x %x", &idx, UADDR(change), UADDR(v));
782 if (ret != 3)
783 return 1;
784 if (dd == &direct.rou_begin) {
785 ST_VAL(x[idx]);
786 } else if (is_xflag_bit(idx)) {
787 ST_BV(x[idx]);
788 } else {
789 ST_VAL(x[idx]);
790 }
791 break;
792 default:
793 return 1;
794 }
795 }
796 return 0;
797}
