40/* add a number to the temp name, first char is base 32, the rest is base 64 */
41
42static char *
43addn(char *p, int val, int n)
44{
45 char bufn[9] = {0}; /* avoid some valgrind uninitialized errors */
46 int d;
47 int i;
48
49 i = 0;
50 while (i < n) {
51 d = val & 0x3f;
52 val >>= 6;
53 bufn[i] = chars[d];
54 ++i;
55 if (val == 0)
56 break;
57 }
58 p = add(p, bufn, i);
59 return (p);
60}

64static char *
65gentmp(char *dir, char *pfx)
66{
67 extern long getpid(void);
68 extern long time(void *);
69 char *buf;
70 char *p, *q;
71 const char *tmp;
72 int n;
73
74 tmp = getenv("TMPDIR");
75 if (tmp == NULL || tmp[0] == '\0') {
76 tmp = getenv("TMP");
77 if (tmp == NULL || tmp[0] == '\0') {
78 tmp = dir;
79 if (tmp == NULL || tmp[0] == '\0') {
80 tmp = P_tmpdir;
81 }
82 }
83 }
84 buf = p = malloc(strlen(tmp) + 32);
85 if (p == NULL) {
86 return (NULL);
87 }
88 p = add(p, tmp, -1);
89 p = add(p, "/", -1);
90 if (pfx != NULL) {
91 p = add(p, pfx, 5);
92 }
93
94 if (rand == 0) { /* first time, create seed */
95 rand = 0;
96 n = 0;
97 q = getenv("USER");
98 q = (q == NULL ? getenv("USERNAME") : q);
99 if (q != NULL) {
100 while (*q != '\0') {
101 rand ^= (*q++) << n++;
102 }
103 }
104 n = 0;
105 q = getenv("HOSTNAME");
106 if (q != NULL) {
107 while (*q != '\0') {
108 rand ^= (*q++) << n++;
109 }
110 }
111 rand ^= getpid();
112 rand ^= (int)((long)buf) >> 4;
113 rand ^= time((long *)0);
114 }
115
116 *p++ = chars[rand & 0x1f];
117 rand = (rand << 16) + rand * 3;
118 p = addn(p, rand, 4);
119 rand = (rand << 16) + rand * 3;
120 p = addn(p, rand, 4);
121 rand = (rand << 16) + rand * 3;
122 p = addn(p, rand, 4);
123 return (buf);
124}
