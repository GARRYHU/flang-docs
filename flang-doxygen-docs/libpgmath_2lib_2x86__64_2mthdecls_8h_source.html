33/*
34 * Windows does not recognize the "_Complex" keyword for complex types but does
35 * know about the "_Fcomplex" (float) and "_Dcomplex" (double) types.
36 *
37 * To minimize support for both Windows and non-Windows systems, define two
38 * typedefs, float_complex_t and double_complex_t, that are used throughout
39 * libpgmath that define a system agnostic complex type.
40 *
41 * CPP function macro PGMATH_CMPLX_CONST should only be used to initialize
42 * complex variables with constant values (not variables).
43 *
44 * Assignment to complex variables from float/double real/imaginary values
45 * contained in variables should use either:
46 * float_complex_t pgmath_cmplxf(float r, float i);
47 * or
48 * double_complex_t pgmath_cmplx(double r, double i);
49 *
50 */
51
52#if defined(TARGET_WIN_X8664) && defined(__clang__)
53typedef _Fcomplex float_complex_t;
54typedef _Dcomplex double_complex_t;
55#define PGMATH_CMPLX_CONST(r,i) {r, i}
56#else
57typedef float _Complex float_complex_t;
58typedef double _Complex double_complex_t;
59#define PGMATH_CMPLX_CONST(r,i) r + I*i
60#endif

96/*
97 * \brief pgmath_cmplxf - return type float_complex_t from two float arguments.
98 *
99 * Common method across all platforms. Does not use "real + I*imag".
100 */
101
102static inline __attribute__((always_inline)) float_complex_t pgmath_cmplxf(float r, float i)
103{
104 struct {
105 union {
106 float_complex_t _c;
107 float _f[2];
108 };
109 } _cf ;
110 _cf._f[0] = r;
111 _cf._f[1] = i;
112 return _cf._c;
113}
114
115/*
116 * \brief pgmath_cmplx - return type double_complex_t from double arguments.
117 *
118 * Common method across all platforms. Does not use "real + I*imag".
119 */
120
121static inline __attribute__((always_inline)) double_complex_t pgmath_cmplx(double r, double i)
122{
123 struct {
124 union {
125 double_complex_t _z;
126 double _d[2];
127 };
128 } _zd ;
129 _zd._d[0] = r;
130 _zd._d[1] = i;
131 return _zd._z;
132}

134/*
135 * Complex ABI conventions.
136 *
137 * The macros that define the function signature (example CMPLXFUNC_C)
138 * and the macro that defines local arguments (example CMPLXFUNC_C)
139 * work in tandem so that regardless of whether the native original
140 * complex ABI or the C99 ABI is used, there are always the following
141 * variables defined:
142 *
143 * carg: the argument defined as either float _Complex or double _Complex
144 * real: creal(carg), crealf(carg) for float
145 * imag: cimag(carg), cimagf(carg) for float
146 *
147 * When there are two or more complex arguments, the local variables are
148 * suffixed with the digits (ie 1, 2, ...) representing their order as
149 * specified in the argument list.
150 */
151

585/*
586 * The following intrinsics are defined for platforms that do not have
587 * architecture specific versions.
588 * It is an attempt to standardize the math library source code across
589 * architectures.
590 *
591 * For example: cexp.c was coded as:
592 * #include "mthdecls.h"
593 *
594 * void
595 * __mth_i_cexp(cmplx_t *cmplx, float real, float imag)
596 * {
597 * float x, y, z;
598 * x = EXPF(real);
599 * #ifndef LINUX8664
600 * y = COSF(imag);
601 * z = SINF(imag);
602 * #else
603 * __mth_sincos(imag, &z, &y);
604 * #endif
605 * y *= x;
606 * z *= x;
607 * r_dummy(y, z);
608 * }
609 *
610 * The special casing of whether __mth_sincos() is available for
611 * individual source files is not scalable. A better alternative is to
612 * have a version of __mth_sincos, even if it is not external available
613 * during the build process.
614 */
