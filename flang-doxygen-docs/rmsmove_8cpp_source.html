35static int
36fixup_nme(int nmex, int msize, int offset, int iter)
37{
38 SPTR new_sym;
39 int new_nme;
40 DTYPE new_dtype;
41 static char buf[100];
42 int buf_len = 100;
43 int sym;
44 char *name = NULL;
45 int name_len;
46 bool is_malloced = false;
47
48 if (nmex <= 0 || NME_TYPE(nmex) != NT_VAR || NME_SYM(nmex) <= 0)
49 return nmex;
50
51 switch (msize) {
52 case 8:
53 new_dtype = DT_UINT8;
54 break;
55 case 4:
56 new_dtype = DT_INT;
57 break;
58 case 2:
59 new_dtype = DT_USINT;
60 break;
61 case 1:
62 new_dtype = DT_BINT;
63 break;
64 }
65
66 sym = NME_SYM(nmex);
67 name_len = strlen(SYMNAME(sym)) + 15;
68 if (name_len <= buf_len) {
69 name = &buf[0];
70 } else {
71 name = (char *)malloc(name_len);
72 assert(name != NULL, "Fail to malloc a buffer", nmex, ERR_Fatal);
73 is_malloced = true;
74 }
75
76 sprintf(name, "..__smove__%s__%d", SYMNAME(sym), iter);
77 new_sym = getsymbol(name);
78 DTYPEP(new_sym, new_dtype);
79 STYPEP(new_sym, ST_MEMBER);
80 CCSYMP(new_sym, 1);
81 ADDRESSP(new_sym, offset);
82 new_nme = addnme(NT_MEM, SPTR_NULL, nmex, 0);
83 NME_SYM(new_nme) = new_sym;
84
85 if (is_malloced)
86 free(name);
87 return new_nme;
88} /* fixup_nme */

91void
92rm_smove(void)
93{
94 int bihx, iltx, ilix, new_acon;
95 /*
96 * First implementation of GSMOVE will be under XBIT(2,0x800000). When this
97 * is the only method, presumably, the code in exp_remove_gsmove() will be
98 * moved to the ensuing loop.
99 */
100 if (USE_GSMOVE)
101 exp_remove_gsmove();
102 for (bihx = gbl.entbih; bihx; bihx = BIH_NEXT(bihx)) {
103 bool have_smove = false;
104 rdilts(bihx);
105 for (iltx = BIH_ILTFIRST(bihx); iltx; iltx = ILT_NEXT(iltx)) {
106 ilix = ILT_ILIP(iltx);
107 if (ILI_OPC(ilix) == IL_SMOVEJ) {
108 int srcx, src_nme, destx, dest_nme, len;
109 int i, n, offset = 0, any = 0;
110 /* target-dependent optimizations */
111 srcx = ILI_OPND(ilix, 1);
112 src_nme = ILI_OPND(ilix, 3);
113 destx = ILI_OPND(ilix, 2);
114 dest_nme = ILI_OPND(ilix, 4);
115 len = ILI_OPND(ilix, 5);
116 offset = 0;
117 if (len > SMOVE_MIN) {
118 /* turn the SMOVEI into SMOVE, change the len to IL_ACON */
119 ILI_OPCP(ilix, IL_SMOVE);
120 ILI_OPND(ilix, 1) = srcx;
121
122/* For LLVM we use memcpy and do not need to chunk up the
123 * copies.
124 * XXX: Fortran has a special case in make_stmt (cgmain.c)
125 * for the 'case STMT_SMOVE'. Fortran will multiply the
126 * length by 8 or 4 depending on the architecture.
127 * C just takes the length as given, and we do not want to chunk the
128 * length since we will be using memcpy and let llvm's memcpy handle
129 * how it performs the copy.
130 */
131 n = len / SMOVE_CHUNK;
132 offset = n * SMOVE_CHUNK;
133 new_acon = ad_aconi(n);
134 ILI_OPND(ilix, 3) = new_acon;
135 len -= offset;
136 ++any;
137 have_smove = true;
138 }
139 if (XBIT(2, 0x4000)) {
140 src_nme = NME_UNK;
141 dest_nme = NME_UNK;
142 }
143 for (i = INFO1; i < 4; ++i) {
144 int msz = info[i].msz;
145 int msize = info[i].msize;
146 while (len >= msize) {
147 int ilioffset, ilix2;
148 int ndest_nme = dest_nme, nsrc_nme = src_nme;
149 /* add the load, store */
150 if (any == 1) {
151 srcx = ad1ili(IL_CSEAR, srcx);
152 destx = ad1ili(IL_CSEAR, destx);
153 }
154 if (!XBIT(2, 0x4000)) {
155 nsrc_nme = fixup_nme(src_nme, msize, offset, i);
156 ndest_nme = fixup_nme(dest_nme, msize, offset, i);
157 }
158 ilioffset = ad_aconi(offset);
159 ilix = ad3ili(IL_AADD, srcx, ilioffset, 0);
160 ilix = ad3ili(info[i].ld, ilix, nsrc_nme, msz);
161 ilix2 = ad3ili(IL_AADD, destx, ilioffset, 0);
162 ilix = ad4ili(info[i].st, ilix, ilix2, ndest_nme, msz);
163 if (!any) {
164 /* reuse this ILT */
165 ILT_ILIP(iltx) = ilix;
166 /* flag this as a store operation */
167 ILT_ST(iltx) = 1;
168 } else {
169 iltx = addilt(iltx, ilix);
170 /* ILT_ST gets set by addilt here */
171 }
172 ++any;
173 offset += msize;
174 len -= msize;
175 }
176 }
177 }
178 }
179 wrilts(bihx);
180 if (have_smove)
181 BIH_SMOVE(bihx) = 1;
182 }
183} /* rm_smove */
