16{

20 a->avl = a->beg + (b->avl - b->beg);

28/* merge channel structures */
29
30chdr *
31__fort_comm_merge(chdr *b, chdr *c)
32{
33 chdr *a; /* new channel */
34 int acpi, bcpi, ccpi; /* channel cpu entry index */
35 int aspi, arpi; /* next avail ents index */
36 int nmax; /* next higher power of 2 cpus */
37 int cpu; /* next cpu to check */
38 int lcpu;
39 int r; /* cpu index */
40 int tcpus;
41
42 /* check for channels that shouldn't be merged */
43
44 if ((b->next != (chdr *)0) || (c->next != (chdr *)0)) {
45 __fort_abort("__fort_comm_merge: cannot merge linked channels");
46 }
47 if ((b->typ != c->typ) || (b->typ != c->typ)) {
48 __fort_abort("__fort_comm_merge: differing base types");
49 }
50 for (cpu = 0; cpu < b->cn; cpu++) {
51 if (b->cp[cpu].opt != (char *)0) { /* free opt area */
52 __fort_free(b->cp[cpu].opt);
53 }
54 }
55 for (cpu = 0; cpu < c->cn; cpu++) {
56 if (c->cp[cpu].opt != (char *)0) { /* free opt area */
57 __fort_free(c->cp[cpu].opt);
58 }
59 }
60
61 /* allocate new channel, ents, and cpu structures */
62
63 a = __fort_allchn((chdr *)0, b->rn + c->rn, b->sn + c->sn, b->cn + c->cn);
64 if ((b->bases != c->bases) || (b->baser != c->baser) || (b->typ != c->typ)) {
65 a->bases = (char *)0;
66 a->baser = (char *)0;
67 a->typ = 0;
68 a->ilen = 0;
69 a->flags = 0;
70 } else {
71 a->bases = b->bases;
72 a->baser = b->baser;
73 a->typ = b->typ;
74 a->ilen = b->ilen;
75 a->flags = b->flags & CHDR_BASE;
76 }
77
78 /* find next higher power than number of cpus */
79
80 nmax = 1;
81 tcpus = GET_DIST_TCPUS;
82 while (tcpus > nmax) {
83 nmax = nmax << 1;
84 }
85
86 /* merge structures */
87
88 acpi = 0;
89 aspi = 0;
90 arpi = 0;
91 lcpu = GET_DIST_LCPU;
92 for (r = 0; r < nmax; r++) {
93
94 cpu = r ^ lcpu; /* next cpu to check */
95 bcpi = 0; /* channel b cpu index */
96 ccpi = 0; /* channel c cpu index */
97
98 while ((bcpi < b->cn) || (ccpi < c->cn)) {
99 while (bcpi < b->cn) { /* check channel b */
100 if (b->cp[bcpi].cpu == cpu) {
101 break;
102 }
103 bcpi++;
104 }
105 while (ccpi < c->cn) { /* check channel c */
106 if (c->cp[ccpi].cpu == cpu) {
107 break;
108 }
109 ccpi++;
110 }
111
112 /* both channels access this cpu? */
113
114 if (((bcpi < b->cn) && (b->cp[bcpi].cpu == cpu)) &&
115 ((ccpi < c->cn) && (c->cp[ccpi].cpu == cpu))) {
116
117 a->cp[acpi].cpu = b->cp[bcpi].cpu;
118
119 /* same op? */
120
121 if (b->cp[bcpi].op == c->cp[ccpi].op) {
122 a->cp[acpi].op = b->cp[bcpi].op;
123 if (b->cp[bcpi].op == CPU_COPY) {
124 a->cp[acpi].rp = &(a->rp[arpi++]);
125 a->cp[acpi].sp = &(a->sp[aspi++]);
126 merge_ent(a->sp, b->sp, c->sp);
127 merge_ent(a->rp, b->rp, c->rp);
128 } else if (b->cp[bcpi].op == CPU_RECV) {
129 a->cp[acpi].rp = &(a->rp[arpi++]);
130 merge_ent(a->rp, b->rp, c->rp);
131 } else if (b->cp[bcpi].op == CPU_SEND) {
132 a->cp[acpi].sp = &(a->sp[aspi++]);
133 merge_ent(a->sp, b->sp, c->sp);
134 }
135 acpi++;
136 bcpi++;
137 ccpi++;
138
139 /* channel b copy? */
140
141 } else if (b->cp[bcpi].op == CPU_COPY) {
142 a->cp[acpi].op = b->cp[bcpi].op;
143 a->cp[acpi].rp = &(a->rp[arpi++]);
144 a->cp[acpi].sp = &(a->sp[aspi++]);
145 *(a->cp[acpi].sp) = *(b->cp[bcpi].sp);
146 *(a->cp[acpi].rp) = *(b->cp[bcpi].rp);
147 acpi++;
148 bcpi++;
149
150 /* channel c copy? */
151
152 } else if (c->cp[ccpi].op == CPU_COPY) {
153 a->cp[acpi].op = c->cp[ccpi].op;
154 a->cp[acpi].rp = &(a->rp[arpi++]);
155 a->cp[acpi].sp = &(a->sp[aspi++]);
156 *(a->cp[acpi].sp) = *(c->cp[ccpi].sp);
157 *(a->cp[acpi].rp) = *(c->cp[ccpi].rp);
158 acpi++;
159 ccpi++;
160
161 /* do send first */
162
163 } else if (cpu > lcpu) {
164 if (b->cp[bcpi].op == CPU_SEND) {
165 a->cp[acpi].op = b->cp[bcpi].op;
166 a->cp[acpi].sp = &(a->sp[aspi++]);
167 *(a->cp[acpi].sp) = *(b->cp[bcpi].sp);
168 acpi++;
169 bcpi++;
170 } else if (c->cp[ccpi].op == CPU_SEND) {
171 a->cp[acpi].op = c->cp[ccpi].op;
172 a->cp[acpi].sp = &(a->sp[aspi++]);
173 *(a->cp[acpi].sp) = *(c->cp[ccpi].sp);
174 acpi++;
175 ccpi++;
176 }
177
178 /* do recv first */
179
180 } else {
181 a->cp[acpi].rp = &(a->rp[arpi++]);
182 if (b->cp[bcpi].op == CPU_RECV) {
183 a->cp[acpi].op = b->cp[bcpi].op;
184 a->cp[acpi].rp = &(a->rp[arpi++]);
185 *(a->cp[acpi].rp) = *(b->cp[bcpi].rp);
186 acpi++;
187 bcpi++;
188 } else if (c->cp[ccpi].op == CPU_RECV) {
189 a->cp[acpi].op = c->cp[ccpi].op;
190 a->cp[acpi].rp = &(a->rp[arpi++]);
191 *(a->cp[acpi].rp) = *(c->cp[ccpi].rp);
192 acpi++;
193 ccpi++;
194 }
195 }
196
197 /* only channel b op */
198
199 } else if (((bcpi < b->cn) && (b->cp[bcpi].cpu == cpu))) {
200 a->cp[acpi].cpu = b->cp[bcpi].cpu;
201 a->cp[acpi].op = b->cp[bcpi].op;
202 if (a->cp[acpi].op == CPU_RECV) {
203 a->cp[acpi].rp = &(a->rp[arpi++]);
204 *(a->cp[acpi].rp) = *(b->cp[bcpi].rp);
205 } else if (a->cp[acpi].op == CPU_SEND) {
206 a->cp[acpi].sp = &(a->sp[aspi++]);
207 *(a->cp[acpi].sp) = *(b->cp[bcpi].sp);
208 } else if (a->cp[acpi].op == CPU_COPY) {
209 a->cp[acpi].rp = &(a->rp[arpi++]);
210 *(a->cp[acpi].rp) = *(b->cp[bcpi].rp);
211 a->cp[acpi].sp = &(a->sp[aspi++]);
212 *(a->cp[acpi].sp) = *(b->cp[bcpi].sp);
213 }
214 acpi++;
215 bcpi++;
216
217 /* only channel c op */
218
219 } else if (((ccpi < c->cn) && (c->cp[ccpi].cpu == cpu))) {
220 a->cp[acpi].cpu = c->cp[ccpi].cpu;
221 a->cp[acpi].op = c->cp[ccpi].op;
222 if (a->cp[acpi].op == CPU_RECV) {
223 a->cp[acpi].rp = &(a->rp[arpi++]);
224 *(a->cp[acpi].rp) = *(c->cp[ccpi].rp);
225 } else if (a->cp[acpi].op == CPU_SEND) {
226 a->cp[acpi].sp = &(a->sp[aspi++]);
227 *(a->cp[acpi].sp) = *(c->cp[ccpi].sp);
228 } else if (a->cp[acpi].op == CPU_COPY) {
229 a->cp[acpi].rp = &(a->rp[arpi++]);
230 *(a->cp[acpi].rp) = *(c->cp[ccpi].rp);
231 a->cp[acpi].sp = &(a->sp[aspi++]);
232 *(a->cp[acpi].sp) = *(c->cp[ccpi].sp);
233 }
234 acpi++;
235 ccpi++;
236 }
237 }
238 }
239
240 a->cn = acpi;
241
242 /* free original channels */
243
244 __fort_free((char *)b);
245 __fort_free((char *)c);
246
247 return (a);
