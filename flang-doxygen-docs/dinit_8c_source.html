46/**
47 Instead of creating dinit records during the processing of data
48 initializations, we need to save information so the records are written
49 at the end of semantic analysis (during semfin). This is necessary for
50 at least a couple of reasons: 1). a record dcl with inits in its STRUCTURE
51 could occur before resolution of its storage class (problematic is
52 SC_CMBLK) 2). with VMS ftn, an array may be initialized (not by implied
53 DO) before resolution of its stype (i.e., its DIMENSION).
54
55 The information we need to save is the pointers to the var list and
56 constant tree. This also implies that the getitem areas
57 (4, 5) need to stay around until dinit output.
58 */
59void
60dinit(VAR *ivl, ACL *ict)
61{
62 int nw;
63 char *ptr;
64
65 if (astb.df == NULL) {
66 if ((astb.df = tmpfile()) == NULL)
67 errfatal(5);
68 sem.dinit_nbr_inits = 0;
69 }
70 nw = fwrite(&gbl.lineno, sizeof(gbl.lineno), 1, astb.df);
71 if (nw != 1)
72 error(10, 40, 0, "(data init file)", CNULL);
73 nw = fwrite(&gbl.findex, sizeof(gbl.findex), 1, astb.df);
74 if (nw != 1)
75 error(10, 40, 0, "(data init file)", CNULL);
76 ptr = (char *)ivl;
77 nw = fwrite(&ptr, sizeof(ivl), 1, astb.df);
78 if (nw != 1)
79 error(10, 40, 0, "(data init file)", CNULL);
80 ptr = (char *)ict;
81 nw = fwrite(&ptr, sizeof(ict), 1, astb.df);
82 if (nw != 1)
83 error(10, 40, 0, "(data init file)", CNULL);
84
85 if (ivl && ivl->u.varref.id == S_IDENT &&
86 (STYPEG(A_SPTRG(ivl->u.varref.ptr)) == ST_PARAM ||
87 PARAMG(A_SPTRG(ivl->u.varref.ptr)))) {
88 sem.dinit_nbr_inits++;
89 }
90 mark_dinit(ivl, ict);
91}

93/** \brief Read in the information a "record" (1 word, 2 pointers) at a time
94 saved by dinit(), and write dinit records for each record.
95 */
96void
97do_dinit(void)
98{
99 VAR *ivl;
100 ACL *ict;
101 char *ptr;
102 int nw;
103 int fileno;
104
105 if (astb.df == NULL)
106 return;
107 nw = fseek(astb.df, 0L, 0);
108#if DEBUG
109 assert(nw == 0, "do_dinit:bad rewind", nw, 4);
110#endif
111
112 while (TRUE) {
113 nw = fread(&gbl.lineno, sizeof(gbl.lineno), 1, astb.df);
114 if (nw == 0)
115 break;
116#if DEBUG
117 assert(nw == 1, "do_dinit: lineno error", nw, 4);
118#endif
119 /* Don't use gbl.findex here because we don't want its value to change */
120 nw = fread(&fileno, sizeof(fileno), 1, astb.df);
121 if (nw == 0)
122 break;
123#if DEBUG
124 assert(nw == 1, "do_dinit: fileno error", nw, 4);
125#endif
126
127 nw = fread(&ptr, sizeof(ivl), 1, astb.df);
128 if (nw == 0)
129 break;
130#if DEBUG
131 assert(nw == 1, "do_dinit: ict error", nw, 4);
132#endif
133 ivl = (VAR *)ptr;
134 nw = fread(&ptr, sizeof(ict), 1, astb.df);
135#if DEBUG
136 assert(nw == 1, "do_dinit: ivl error", nw, 4);
137#endif
138 ict = (ACL *)ptr;
139#if DEBUG
140 if (DBGBIT(6, 32)) {
141 fprintf(gbl.dbgfil, "---- deferred dinit read: ivl %p, ict %p\n",
142 (void *)ivl, (void *)ict);
143 }
144#endif
145 if (ict && ict->no_dinitp)
146 no_dinitp = TRUE;
147 df_dinit(ivl, ict);
148 no_dinitp = FALSE;
149 }
150
151 if (gbl.maxsev >= 3) {
152 /* since errors occur during semant, print_stmts() will not
153 * be called; need to clean up the ast dinit file stuff.
154 */
155 fclose(astb.df);
156 astb.df = NULL;
157 /* freearea(15); */ /* saved dinit records & equivalence lists */
158 }
159
160}

183void

185{
186 if (DBGBIT(6, 3)) {

188 if (DBGBIT(6, 2)) {
189 if (ivl) {

200 }
201
202 if (ivl) {

208 } /* init'ed structures */
209
210 if (DBGBIT(6, 3))
211 fprintf(gbl.dbgfil, "\nDINIT RETURNING ----------------\n\n");
212}

214static void
215dinit_data(VAR *ivl, ACL *ict, int dtype)
216{
217 /* ivl : pointer to initializer variable list */
218 /* ict : pointer to initializer constant tree */
219 /* dtype : if this is a structure initialization, the ptr to dtype */
220
221 int sptr, memptr;
222 INT num_elem = 0;
223 INT ict_rc = 0;
224 LOGICAL is_array;
225 int member = 0;
226 int substr_dtype;
227
228 if (ivl == NULL) {
229 assert(dtype, "dinit_data: no object to initialize", 0, 2);
230 member = DTY(dtype + 1);
231 /* for derived type extension */
232 if (PARENTG(DTY(dtype + 3)) && get_seen_contains()
233 && (DTY(DTYPEG(member)) == TY_DERIVED)
234 && (DTY(ict->dtype) != TY_DERIVED)) {
235 member = SYMLKG(member);
236 }
237 }
238
239 do {
240 substr_dtype = 0;
241 if (member && (is_empty_typedef(DTYPEG(member)) ||
242 is_tbp_or_final(member))) {
243 memptr = SYMLKG(member);
244 member = memptr == NOSYM ? 0 : memptr;
245 continue;
246 }
247 if ((ivl && ivl->id == Varref) || member) {
248 is_array = FALSE;
249 num_elem = 1;
250 if (member) {
251 memptr = sptr = member;
252 if (!POINTERG(sptr) && !ALLOCATTRG(sptr) &&
253 DTY(DTYPEG(sptr)) == TY_ARRAY) {
254 /* A whole array; determine number of elements to init */
255 if (size_of_array(DTYPEG(sptr)))
256 num_elem = get_int_cval(sym_of_ast(AD_NUMELM(AD_PTR(sptr))));
257 else
258 num_elem = 0;
259 is_array = TRUE;
260 }
261 } else {
262 int ast = ivl->u.varref.ptr;
263
264 find_base(ast, &sptr, &memptr);
265 if (sem.dinit_error)
266 goto error_exit;
267 if (A_TYPEG(ast) == A_ID || A_TYPEG(ast) == A_MEM) {
268 /* We're initialising a scalar or whole array,
269 * which may or may not be a derived type component.
270 * (N.B. The derived type case may be an A_ID or
271 * A_MEM node, depending on the value of DTF90OUTPUT.
272 * In the former case, memptr == sptr.)
273 */
274 if (!POINTERG(sptr) && DTY(DTYPEG(memptr)) == TY_ARRAY) {
275 /* A whole array */
276 if (size_of_array(DTYPEG(memptr)))
277 num_elem = get_int_cval(sym_of_ast(AD_NUMELM(AD_PTR(memptr))));
278 else
279 num_elem = 0;
280 is_array = TRUE;
281 }
282 } else if (A_TYPEG(ast) == A_SUBSTR) {
283 ISZ_T len;
284 int s;
285 s = A_SPTRG(A_ALIASG(A_RIGHTG(ast)));
286 if (s)
287 len = get_isz_cval(s);
288 else
289 len = string_length(DDTG(DTYPEG(memptr)));
290 s = A_SPTRG(A_ALIASG(A_LEFTG(ast)));
291 if (s)
292 len = len - get_isz_cval(s) + 1;
293 if (len < 0)
294 len = 1;
295 substr_dtype = get_type(2, DTY(A_DTYPEG(ast)), mk_cval(len, DT_INT4));
296 } else {
297 /* We're initialising an array element or section,
298 */
299 if (ivl->u.varref.shape != 0)
300 uf("- initializing an array section");
301 }
302 }
303
304 sym_is_dinitd(sptr);
305
306 /* now process enough dinit constant list items to
307 * take care of the current varref item. For a Cray target,
308 * a Hollerith constant may initialize more than one array
309 * element.
310 */
311 do {
312 if (ict_rc == 0) {
313 if (ict == NULL) {
314 if (is_array && XBIT(49, 0x1040000)) {
315 /* T3D/T3E or C90 target: the number of initializers
316 * may be less than the number of elements
317 */
318 if (flg.standard)
319 ERR170("The number of initializers is less than number of "
320 "elements of",
321 SYMNAME(memptr));
322 break;
323 }
324 errsev(66);
325 goto error_exit;
326 }
327 ict_rc = dinit_eval(ict->repeatc);
328 }
329 if (ict_rc > 0) {
330 /* Note: repeat factor ict_rc == 0 is allowed! */
331 int ni; /* number of elements consumed by a constant */
332
333 ni = 1;
334 if (ivl && DTY(ivl->u.varref.dtype) == TY_DERIVED &&
335 !is_zero_size_typedef(ivl->u.varref.dtype) && !POINTERG(sptr))
336 dinit_data(ivl->u.varref.subt, ict->subc, ict->dtype);
337 else if (member && DTY(ict->dtype) == TY_DERIVED &&
338 !is_zero_size_typedef(ict->dtype) && !POINTERG(sptr))
339 if (ict->subc) {
340 /* derived type member-by-member initialization */
341 dinit_data(NULL, ict->subc, ict->dtype);
342 } else {
343 /* derived type initialized by a named constant */
344 dinit_acl_val(member, ict->dtype, ict);
345 return;
346 }
347 else if (is_array && ict->dtype == DT_HOLL)
348 ni = dinit_hollerith(sptr, DDTG(DTYPEG(memptr)),
349 A_SPTRG(A_ALIASG(ict->u1.ast)));
350 else if (is_array) {
351 if (ict->id == AC_IEXPR) {
352 if (DTY(ict->dtype) == TY_ARRAY) {
353 if (ict->u1.expr->op == AC_ARRAYREF) {
354 ni = arrayref_size(ict->u1.expr->rop);
355 } else {
356 ni = dinit_eval(ADD_NUMELM(ict->dtype));
357 }
358 dinit_acl_val(sptr, DDTG(DTYPEG(memptr)), ict);
359 } else {
360 dinit_acl_val(sptr, DTYPEG(memptr), ict);
361 }
362 } else if (ict->id == AC_ACONST) {
363 ACL *subict = ict->subc;
364 /* MORE most of these calls to dinit_eval should be calls to
365 * get_int_cval, dinit_eval is
366 * for evaluating implied so expressions only */
367 if (subict != 0 && subict->id == AC_IDO) {
368 ni = num_elem; // Use the previously calculated size.
369 } else {
370 ni = dinit_eval(ADD_NUMELM(ict->dtype));
371 }
372 dinit_acl_val(sptr, DTYPEG(sptr), ict);
373 } else {
374 /* AC_AST, either a constant or a named constant */
375 if (DTY(ict->dtype) == TY_ARRAY) {
376 ni = dinit_eval(ADD_NUMELM(ict->dtype));
377 }
378 (void)dinit_val(sptr, DDTG(DTYPEG(memptr)), ict->dtype,
379 ict->u1.ast, 0);
380 }
381 } else if (substr_dtype) {
382 dinit_acl_val(sptr, substr_dtype, ict);
383 } else {
384 /* Could be Superfluous ict if POINTER is set:
385 * dinit_acl_val() will catch!
386 */
387 dinit_acl_val(sptr, DDTG(DTYPEG(memptr)), ict);
388 }
389
390 switch (ni) {
391 case 1:
392 ni = (ict_rc < num_elem) ? ict_rc : num_elem;
393 num_elem -= ni;
394 ict_rc -= ni;
395 break;
396 default:
397 num_elem -= ni;
398 ict_rc--;
399 break;
400 }
401 }
402 if (ict_rc == 0)
403 ict = ict->next;
404 } while (num_elem > 0);
405 if (num_elem < 0)
406 errsev(67);
407 } else if (ivl->id == Dostart) {
408 if (sem.top == &sem.dostack[MAX_DOSTACK]) {
409 /* nesting maximum exceeded. */
410 errsev(34);
411 return;
412 }
413 sem.top->sptr = chk_doindex(ivl->u.dostart.indvar);
414 if (sem.top->sptr == 1)
415 return;
416 sem.top->currval = dinit_eval(ivl->u.dostart.lowbd);
417 sem.top->upbd = dinit_eval(ivl->u.dostart.upbd);
418 sem.top->step = dinit_eval(ivl->u.dostart.step);
419 if ((sem.top->step > 0 && sem.top->currval <= sem.top->upbd) ||
420 (sem.top->step <= 0 && sem.top->currval >= sem.top->upbd))
421 ++sem.top;
422 else {
423 /* A 'zero trip' implied DO loop. Go directly to the
424 corresponding 'Doend' node */
425 int depth;
426
427 depth = 1;
428 do {
429 ivl = ivl->next;
430 if (!ivl)
431 break;
432 switch (ivl->id) {
433 case Dostart:
434 ++depth;
435 break;
436 case Doend:
437 --depth;
438 break;
439 }
440 } while (depth);
441 }
442 } else {
443 assert(ivl->id == Doend, "dinit:badid", 0, 3);
444 --sem.top;
445 sem.top->currval += sem.top->step;
446 if ((sem.top->step > 0 && sem.top->currval <= sem.top->upbd) ||
447 (sem.top->step <= 0 && sem.top->currval >= sem.top->upbd)) {
448 /* go back to start of this do loop */
449 ++sem.top;
450 ivl = ivl->u.doend.dostart;
451 }
452 }
453 if (sem.dinit_error)
454 goto error_exit;
455 if (ivl)
456 ivl = ivl->next;
457 if (member) {
458 if (POINTERG(member) || ALLOCATTRG(member))
459 member = SYMLKG(member); // skip <ptr>$p
460 memptr = SYMLKG(member);
461 member = memptr == NOSYM ? 0 : memptr;
462 }
463 } while (ivl || member);
464
465 while (ict && num_elem > 0) {
466 /* Some dinit constants remain. That's OK if they have 0
467 * repeat factor, otherwise it's an error. */
468 if (ict_rc == 0)
469 ict_rc = dinit_eval(ict->repeatc);
470 if (ict_rc == 0)
471 ict = ict->next;
472 else {
473 errsev(67);
474 goto error_exit;
475 }
476 }
477error_exit:
478 return;
479}

481/*
482 * compute the size of an arrayref (a section). The section information is
483 * created by semant and consists of a list of ACL items of the form:
484 * subscr_ast <-(lop) [array_ref] (rop)-> lb -> ub -> stride, ...
485 * repeated for the remaining dimensions.
486 * Each ACL contains an const AST representing the value of the bounds/stride.
487 */
488static ISZ_T
489arrayref_size(ACL *sect)
490{
491 ACL *c, *t;
492 ISZ_T size, nelem;
493 ISZ_T lowb, upb, stride;
494
495 size = 1;
496 t = sect;
497 do {
498 if (t == 0) {
499 interr("dinit: arrayref: missing subscript array \n", 0, 3);
500 return 1;
501 }
502
503 c = t->u1.expr->rop;
504
505 if (c == 0) {
506 interr("dinit: arrayref: missing array section lb\n", 0, 3);
507 return 1;
508 }
509 lowb = get_isz_cval(A_SPTRG(c->u1.ast));
510
511 if ((c = c->next) == 0) {
512 interr("dinit: arrayref: missing array section ub\n", 0, 3);
513 return 1;
514 }
515 upb = get_isz_cval(A_SPTRG(c->u1.ast));
516
517 if ((c = c->next) == 0) {
518 interr("dinit: arrayref: missing array section stride\n", 0, 3);
519 return 1;
520 }
521 stride = get_isz_cval(A_SPTRG(c->u1.ast));
522
523 if (stride < 0)
524 nelem = (lowb - upb + (-stride)) / (-stride);
525 else if (stride != 0)
526 nelem = (upb - lowb + stride) / stride;
527 else
528 interr("dinit: arrayref: array section stride is 0\n", 0, 3);
529 if (nelem > 0)
530 size *= nelem;
531 else
532 size = 0;
533 t = t->next;
534 } while (t);
535 return size;
536}

540/* pointer to initializer constant tree */
541static void
542dinit_subs(ACL *ict)
543{
544 int sptr; /* symbol ptr to identifier to get initialized */
545 int i;
546
547 /*
548 * We come into this routine to follow the ict links for a substructure.
549 */
550 while (ict) {
551 switch (ict->id) {
552 case AC_TYPEINIT:
553 dinit_subs(ict->subc);
554 break;
555 case AC_IDENT:
556 case AC_CONST:
557 case AC_IEXPR:
558 case AC_AST:
559 case AC_IDO:
560 case AC_ACONST:
561 case AC_SCONST:
562 case AC_EXPR:
563 case AC_REPEAT:
564 dinit_acl_val(ict->sptr, DDTG(ict->dtype), ict);
565 break;
566 default:
567 if (ict->subc) {
568 /* Follow substructure down before continuing at this level */
569 for (i = dinit_eval(ict->repeatc); i != 0; i--)
570 dinit_subs(ict->subc);
571 } else {
572 /* Handle basic type declaration init statement */
573 /* If new member or member has a repeat start a new block */
574 if (ict->sptr) {
575 /* A new member to initialize */
576 sptr = ict->sptr;
577 }
578 (void)dinit_val(sptr, DDTG(DTYPEG(sptr)), ict->dtype, ict->u1.ast, 0);
579 }
580 }
581 ict = ict->next;
582 } /* End of while */
583}

585static void
586setConval(int sptr, int conval, int op)
587{
588 if (conval && !PARMFING(sptr)) {
589 int val = PARMINITG(sptr);
590 switch (op) {
591 case AC_ADD:
592 val += conval;
593 break;
594 case AC_SUB:
595 val -= conval;
596 break;
597 case AC_MUL:
598 val *= conval;
599 break;
600 case AC_DIV:
601 val /= conval;
602 break;
603 case AC_NEG:
604 val = -conval;
605 break;
606 case AC_LNOT:
607 val = ~conval;
608 break;
609 case AC_LOR:
610 val |= conval;
611 break;
612 case AC_LAND:
613 val &= conval;
614 break;
615 case AC_EQ:
616 val = (val == conval) ? -1 : 0;
617 break;
618 case AC_GE:
619 val = (val >= conval) ? -1 : 0;
620 break;
621 case AC_GT:
622 val = (val > conval) ? -1 : 0;
623 break;
624 case AC_LE:
625 val = (val <= conval) ? -1 : 0;
626 break;
627 case AC_LT:
628 val = (val < conval) ? -1 : 0;
629 break;
630 case AC_NE:
631 val = (val != conval) ? -1 : 0;
632 break;
633 case 0:
634 val = conval;
635 break;
636 default:
637 val = conval;
638 error(155, 3, gbl.lineno, "Invalid operator for kind type parameter "
639 "initialization",
640 NULL);
641 }
642 PARMINITP(sptr, val);
643 }

646static void
647process_real_kind(int sptr, ACL *ict, int op)
648{
649 int ast, con1, conval;
650
651 ast = ict->u1.ast;
652 conval = 0;
653 if (A_TYPEG(ast) == A_CNST) {
654
655 con1 = A_SPTRG(ast);
656 con1 = CONVAL2G(con1);
657 if (con1 <= 6)
658 conval = 4;
659 else if (con1 <= 15)
660 conval = 8;
661 else if (con1 <= 31 && !XBIT(57, 4))
662 conval = 16;
663 else
664 conval = -1;
665 }
666
667 ict = ict->next;
668 if (ict) {
669 ast = ict->u1.ast;
670
671 if (A_TYPEG(ast) == A_CNST) {
672 con1 = A_SPTRG(ast);
673 con1 = CONVAL2G(con1);
674 if (XBIT(49, 0x40000)) {
675 /* Cray C90 */
676 if (con1 <= 37) {
677 if (conval > 0 && conval < 4)
678 conval = 4;
679 } else if (con1 <= 2465) {
680 if (conval > 0 && conval < 8)
681 conval = 8;
682 } else {
683 if (conval > 0)
684 conval = 0;
685 conval -= 2;
686 }
687 } else {
688 /* ANSI */
689 if (con1 <= 37) {
690 if (conval > 0 && conval < 4)
691 conval = 4;
692 } else if (con1 <= 307) {
693 if (conval > 0 && conval < 8)
694 conval = 8;
695 } else if (con1 <= 4931 && !XBIT(57, 4)) {
696 if (conval > 0 && conval < 16)
697 conval = 16;
698 } else {
699 if (conval > 0)
700 conval = 0;
701 conval -= 2;
702 }
703 }
704 }
705 }
706 if (conval) {
707 setConval(sptr, conval, op);
708 }
709}

711static void
712dinit_acl_val2(int sptr, int dtype, ACL *ict, int op)
713{
714 int dvl_val = 0;
715
716 if (ict->id == AC_IEXPR) {
717 switch (ict->u1.expr->op) {
718 case AC_LNOT:
719 case AC_NEG:
720 case AC_CONV:
721 dinit_acl_val2(sptr, dtype, ict->u1.expr->lop, 0);
722 break;
723 case AC_ADD:
724 case AC_SUB:
725 case AC_MUL:
726 case AC_DIV:
727 case AC_EXP:
728 case AC_EXPK:
729 case AC_EXPX:
730 case AC_LOR:
731 case AC_LAND:
732 case AC_LEQV:
733 case AC_LNEQV:
734 case AC_EQ:
735 case AC_GE:
736 case AC_GT:
737 case AC_LE:
738 case AC_LT:
739 case AC_NE:
740 dinit_acl_val2(sptr, dtype, ict->u1.expr->lop, ict->u1.expr->op);
741 dinit_acl_val2(sptr, dtype, ict->u1.expr->rop, ict->u1.expr->op);
742 break;
743 case AC_ARRAYREF:
744 if (!cmpat_dtype_with_size(dtype, ict->dtype)) {
745 errsev(91);
746 }
747 break;
748 case AC_MEMBR_SEL:
749 if (!cmpat_dtype_with_size(dtype, ict->dtype)) {
750 errsev(91);
751 }
752 break;
753 case AC_INTR_CALL:
754 if (ict->id == AC_IEXPR) {
755 ACL *subict = ict->u1.expr->rop;
756 int intr = ict->u1.expr->lop->u1.i;
757 int conval, con1, ast;
758 if (subict && subict->id == AC_AST) {
759 conval = 0;
760 switch (intr) {
761 case AC_I_selected_int_kind:
762 ast = subict->u1.ast;
763 if (A_TYPEG(ast) == A_CNST) {
764 con1 = CONVAL2G(A_SPTRG(ast));
765 conval = 4;
766 if (con1 > 18 || (con1 > 9 && XBIT(57, 2)))
767 conval = -1;
768 else if (con1 > 9)
769 conval = 8;
770 else if (con1 > 4)
771 conval = 4;
772 else if (con1 > 2)
773 conval = 2;
774 else
775 conval = 1;
776 }
777 setConval(sptr, conval, op);
778 break;
779 case AC_I_selected_real_kind:
780 process_real_kind(sptr, subict, op);
781 break;
782 case AC_I_selected_char_kind:
783 ast = subict->u1.ast;
784 if (A_TYPEG(ast) == A_CNST) {
785 int dty;
786 con1 = A_SPTRG(ast);
787 dty = DTY(DTYPEG(con1));
788 if (dty == TY_CHAR || dty == TY_NCHAR) {
789 conval = _selected_char_kind(con1);
790 } else {
791 interr("dinit: selected_char_kind: unexpected arg type", 0, 3);
792 break;
793 }
794 }
795 setConval(sptr, conval, op);
796 break;
797 default:
798 /* Other intrinsics are handled by backend dinit. */
799 break;
800 }
801 }
802 }
803 dinit_intr_call(sptr, dtype, ict);
804 break;
805 }
806 } else if (ict->id == AC_ACONST) {
807 if ((ict->subc != 0) && (ict->subc->id == AC_IDO)
808 && (ict->subc->subc != 0) && (ict->subc->subc->id == AC_IDO)) {
809 /* Perform a more relaxed check for a nested implied-do loop. */
810 if (!cmpat_dtype_array_cast(dtype, ict->dtype)) {
811 errsev(91);
812 }
813 } else {
814 if (!cmpat_dtype_with_size(dtype, ict->dtype)) {
815 errsev(91);
816 }
817 if (!cmpat_dtype_with_size(DDTG(dtype), DDTG(ict->dtype))) {
818 errsev(91);
819 }
820 }
821 } else if (ict->id == AC_IDO) {
822 dinit_acl_val2(sptr, dtype, ict->subc, 0);
823 } else if (ict->id == AC_AST) {
824 /*
825 * Superfluous ict if POINTER is set; would be better if we
826 * didn't generate the entry, but ss a hack, just ignore it.
827 */
828 if (!POINTERG(sptr))
829 dvl_val =
830 dinit_val(sptr, dtype, DDTG(A_DTYPEG(ict->u1.ast)), ict->u1.ast, op);
831
832 if (STYPEG(sptr) == ST_MEMBER && KINDG(sptr) && !USEKINDG(sptr) &&
833 A_TYPEG(ict->u1.ast) == A_CNST) {
834 int val = CONVAL2G(A_SPTRG(ict->u1.ast));
835 setConval(sptr, val, op);
836 }
837
838 } else if (ict->id == AC_IDENT || ict->id == AC_CONST) {
839 dvl_val =
840 dinit_val(sptr, dtype, DDTG(A_DTYPEG(ict->u1.ast)), ict->u1.ast, op);
841 }
842 if (!XBIT(7, 0x100000)) {
843 if (flg.opt >= 2 && sptr && STYPEG(sptr) == ST_VAR &&
844 SCG(sptr) == SC_LOCAL && !ARGG(sptr) && !ASSNG(sptr) && dvl_val) {
845 if (DTY(dtype) == TY_CHAR) {
846 if (sptr && DTYPEG(sptr) != dtype)
847 return;
848 if (string_length(dtype) != string_length(DDTG(A_DTYPEG(ict->u1.ast))))
849 return;
850 } else if (DTY(dtype) == TY_NCHAR) {
851 if (sptr && DTYPEG(sptr) != dtype)
852 return;
853 if (string_length(dtype) != string_length(DDTG(A_DTYPEG(ict->u1.ast))))
854 return;
855 }
856 NEED(aux.dvl_avl + 1, aux.dvl_base, DVL, aux.dvl_size, aux.dvl_size + 32);
857 DVL_SPTR(aux.dvl_avl) = sptr;
858 DVL_CONVAL(aux.dvl_avl) = dvl_val;
859 aux.dvl_avl++;
860 }
861 }

873static void
874dinit_intr_call(int sptr, int dtype, ACL *ict)
875{
876 ACL *aclp, *next_save;
877
878 assert(ict->u1.expr->lop->id == AC_ICONST,
879 "dinit_intr_call: incorrect ACL type for intrinsic selector\n", 0, 4);
880
881 if (ict->u1.expr->lop->u1.i == AC_I_null) {
882 /* Currently handles only NULL() */
883 if (!POINTERG(sptr) && !PTRVG(sptr) && !ALLOCATTRG(sptr)) {
884 errsev(459);
885 }
886 /* HACK: this is the only place before the backend where there is any
887 * linkage between the VAR list and the ACL list (?). Therefore it is
888 * the only place where initialization of a <ptr> object with a NULL()
889 * call can be modified. First change the intrinsic call to a constant
890 * zero initialization. Then, if <ptr> is a derived type member, add
891 * constant zeros to the ACL list to initialize any associated <ptr>$o,
892 * <ptr>$sd, and/or <ptr>$td values that have been added as hidden
893 * members of the type, but skip <ptr>$p values. Processing for
894 * non-derived type pointers is done in lower_pointer_init.
895 */
896
897 ict->id = AC_AST;
898 ict->dtype = DT_PTR; /* may have problems with XBIT(125,0x2000) */
899 if (!ict->ptrdtype) {
900 /* build pointer type for backend upper/dinit */
901 ict->ptrdtype = get_type(2, TY_PTR, DDTG(DTYPEG(sptr)));
902 }
903 if (ict->sptr && (POINTERG(ict->sptr) || ALLOCATTRG(sptr))) {
904 /* use <ptr>$p */
905 ict->sptr = MIDNUMG(sptr);
906 }
907
908 /* If astb.i0 will be changed to something else, it must change in
909 * chk_struct_constructor as well.
910 */
911 ict->u1.ast = astb.i0;
912 ict->is_const = 1;
913
914 if (STYPEG(sptr) != ST_MEMBER)
915 return;
916
917 aclp = ict;
918 next_save = ict->next;
919
920 for (sptr = SYMLKG(SYMLKG(sptr)); HCCSYMG(sptr); sptr = SYMLKG(sptr)) {
921 aclp = aclp->next = GET_ACL(15);
922 aclp->id = AC_AST;
923 aclp->is_const = 1;
924 aclp->dtype = DDTG(DTYPEG(sptr));
925 aclp->u1.ast = aclp->dtype == DT_INT8 ? astb.k0 : astb.i0;
926 if (ict->sptr)
927 aclp->sptr = sptr;
928 if (DTY(DTYPEG(sptr)) == TY_ARRAY)
929 aclp->repeatc = ADD_NUMELM(DTYPEG(sptr));
930 }
931
932 aclp->next = next_save;
933 }
934}

938/* dinit_val - make sure constant value is correct data type to initialize
939 * symbol (sptr) to. Then call dinit_put to generate dinit record.
940 */
941static int
942dinit_val(int sptr, int dtype, int dtypev, int astval, int op)
943{
944 INT val;
945 INT newval[2];
946 int newast;
947 char buf[2];
948 int do_dvl = 0;
949
950 if (DTY(dtypev) == TY_DERIVED) {
951 if (!eq_dtype(dtype, dtypev))
952 errsev(91);
953 return 0;
954 }
955 if (A_ALIASG(astval))
956 astval = A_ALIASG(astval);
957
958 if (is_procedure_ptr(sptr)) {
959 /* TBD: Eventually do this for regular pointers? */
960 return astval;
961 }
962 if (POINTERG(sptr)) {
963 error(457, 3, gbl.lineno, SYMNAME(sptr), CNULL);
964 return 0;
965 }
966 if (!XBIT(7, 0x100000)) {
967 if (flg.opt >= 2 && sptr && STYPEG(sptr) == ST_VAR &&
968 SCG(sptr) == SC_LOCAL && !ARGG(sptr) && !ASSNG(sptr) &&
969 DTY(DTYPEG(sptr)) != TY_DERIVED && op == 0) {
970 do_dvl = 1;
971 }
972 }
973
974 switch (DTY(A_DTYPEG(astval))) {
975 case TY_DWORD:
976 case TY_DBLE:
977 case TY_QUAD:
978 case TY_CMPLX:
979 case TY_DCMPLX:
980 case TY_QCMPLX:
981 case TY_CHAR:
982 case TY_NCHAR:
983 case TY_INT8:
984 case TY_LOG8:
985 val = A_SPTRG(astval);
986 break;
987 case TY_ARRAY:
988 /*
989 * an array value does not require any special processing and
990 * do not want to let it fall into the CHAR case; the CONVAL1
991 * field of its sptr isn't defined.
992 */
993 return 0;
994 case TY_HOLL:
995 val = CONVAL1G(A_SPTRG(astval));
996 break;
997 default:
998 val = CONVAL2G(A_SPTRG(astval));
999 }
1000
1001 if (DTYG(dtypev) == TY_HOLL) {
1002 /* convert hollerith character string to one of proper length */
1003 val = cngcon(val, (int)DTYPEG(val), dtype);
1004 if (do_dvl == 1) {
1005 switch (dtype) {
1006 case TY_DBLE:
1007 case TY_QUAD:
1008 case TY_INT8:
1009 case TY_LOG8:
1010 case TY_CMPLX:
1011 case TY_DCMPLX:
1012 case TY_QCMPLX:
1013 newast = mk_cnst(val);
1014 break;
1015 default:
1016 newval[0] = 0;
1017 newval[1] = val;
1018 val = getcon(newval, dtype);
1019 newast = mk_cnst(val);
1020 }
1021 }
1022 } else if (DTYG(dtypev) == TY_CHAR || DTYG(dtypev) == TY_NCHAR ||
1023 DTYG(dtypev) != DTY(dtype)) {
1024 /* check for special case of initing character*1 to numeric. */
1025 if (DTY(dtype) == TY_CHAR && DTY(dtype + 1) == astb.i1) {
1026 if (DT_ISINT(dtypev) && !DT_ISLOG(dtypev)) {
1027 if (flg.standard)
1028 error(172, 2, gbl.lineno, SYMNAME(sptr), CNULL);
1029 if (val < 0 || val > 255) {
1030 error(68, 3, gbl.lineno, SYMNAME(sptr), CNULL);
1031 val = getstring(" ", 1);
1032 } else {
1033 buf[0] = (char)val;
1034 buf[1] = 0;
1035 val = getstring(buf, 1);
1036 }
1037 dtypev = DT_CHAR;
1038 }
1039 }
1040 /* Convert character string to one of proper length or,
1041 * convert constant to type of identifier.
1042 */
1043 val = cngcon(val, dtypev, dtype);
1044 if (do_dvl == 1) {
1045 if (DTYG(dtypev) != DTY(dtype)) {
1046 switch (DTY(dtype)) {
1047 case TY_HOLL:
1048 val = getcon(&val, dtype);
1049 break;
1050 case TY_CHAR:
1051 case TY_NCHAR:
1052 break;
1053 case TY_LOG:
1054 case TY_SLOG:
1055 case TY_BLOG:
1056 case TY_INT:
1057 case TY_SINT:
1058 case TY_BINT:
1059 case TY_WORD:
1060 case TY_FLOAT:
1061 newval[0] = 0;
1062 newval[1] = val;
1063 val = getcon(newval, dtype);
1064 break;
1065 }
1066 newast = mk_cnst(val);
1067 } else {
1068 newast = mk_cnst(val);
1069 }
1070 }
1071 } else if (do_dvl == 1) {
1072 if (DTYG(dtypev) != DTY(dtype))
1073 newast = mk_cnst(val);
1074 else
1075 newast = astval;
1076 }
1077
1078 if (do_dvl == 1 && op == 0) {
1079 return newast;
1080 }
1081 return 0;
1082}

1363 * of an array reference (i.e. base sptr).

1366find_base(int ast, int *psptr, int *pmemptr)
