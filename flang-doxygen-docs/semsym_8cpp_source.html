21SPTR
22declref(SPTR sptr, SYMTYPE stype, int def)
23{
24 SYMTYPE st;
25 SPTR sptr1;
26 SPTR first;
27
28 first = sptr;
29 do {
30 st = STYPEG(sptr);
31 if (st == ST_UNKNOWN)
32 goto return1; /* stype not set yet, set it */
33 if (stb.ovclass[st] == stb.ovclass[stype]) {
34 if (stype != st) {
35 if (def == 'd') {
36 /* Redeclare of intrinsic symbol is okay unless frozen */
37 if (IS_INTRINSIC(st)) {
38 sptr1 = newsym(sptr);
39 if (sptr1 != SPTR_NULL)
40 sptr = sptr1;
41 goto return1;
42 }
43 /* multiple declaration */
44 error(S_0044_Multiple_declaration_for_symbol_OP1, ERR_Severe,
45 gbl.lineno, SYMNAME(first), CNULL);
46 } else
47 /* illegal use of symbol */
48 error(S_0084_Illegal_use_of_symbol_OP1_OP2, ERR_Severe, gbl.lineno,
49 SYMNAME(first), CNULL);
50 break;
51 }
52 goto return2; /* found, return it */
53 }
54 sptr = HASHLKG(sptr);
55 } while (sptr && NMPTRG(sptr) == NMPTRG(first));
56
57 /* create new one if def or illegal use */
58 sptr = insert_sym(first);
59return1:
60 STYPEP(sptr, stype);
61return2:
62 if (flg.xref)
63 xrefput(sptr, def);
64 return sptr;
65}

67SPTR
68declsym(SPTR sptr, SYMTYPE stype, bool errflg)
69{
70 SYMTYPE st;
71 SPTR sptr1, first;
72
73 first = sptr;
74 do {
75 st = STYPEG(sptr);
76 if (st == ST_UNKNOWN)
77 goto return1; /* Brand new symbol, return it. */
78 if (st == ST_IDENT && stb.ovclass[st] == stb.ovclass[stype])
79 goto return1; /* Found sym in same overloading class */
80 if (stb.ovclass[st] == stb.ovclass[stype]) {
81 if (stype == st) {
82 /* Possible attempt to multiply define symbol */
83 if (errflg) {
84 error(S_0044_Multiple_declaration_for_symbol_OP1, ERR_Severe,
85 gbl.lineno, SYMNAME(first), CNULL);
86 break;
87 } else
88 goto return2;
89 } else {
90 /* Redeclare of intrinsic symbol is okay unless frozen */
91 if (IS_INTRINSIC(st)) {
92 if ((sptr1 = newsym(sptr)) != 0)
93 sptr = sptr1;
94 goto return1;
95 } else {
96 error(S_0043_Illegal_attempt_to_redefine_OP1_OP2, ERR_Severe,
97 gbl.lineno, "symbol", SYMNAME(first));
98 break;
99 }
100 }
101 }
102 sptr = HASHLKG(sptr);
103 } while (sptr && NMPTRG(sptr) == NMPTRG(first));
104
105 /* create new one if def or illegal use */
106 sptr = insert_sym(first);
107return1:
108 STYPEP(sptr, stype);
109return2:
110 if (flg.xref)
111 xrefput(sptr, 'd');
112 return sptr;
113}

116refsym(SPTR sptr, int oclass)
117{
118 int st;
119 SPTR first;
120
121 first = sptr;
122 do {
123 st = STYPEG(sptr);
124 if (st == ST_UNKNOWN)
125 goto return1;
126 if (stb.ovclass[st] == oclass)
127 goto returnit;
128 sptr = HASHLKG(sptr);
129 } while (sptr && NMPTRG(sptr) == NMPTRG(first));
130
131 /* Symbol in given overloading class not found, create new one */
132 sptr = insert_sym(first);
133return1:
134 if (flg.xref)
135 xrefput(sptr, 'd');
136returnit:
137 return sptr;

165SPTR
166newsym(SPTR sptr)
167{
168 SPTR sp2;
169
170 if (EXPSTG(sptr)) {
171 /* Symbol previously frozen as an intrinsic */
172 error(S_0043_Illegal_attempt_to_redefine_OP1_OP2, ERR_Severe, gbl.lineno,
173 "intrinsic", SYMNAME(sptr));
174 return SPTR_NULL;
175 }
176 /*
177 * try to find another sym in the same overloading class; we need to
178 * try this first since there could be multiple occurrences of an
179 * intrinsic and therefore the sptr appears more than once in the
180 * semantic stack. E.g.,
181 * call sub (sin, sin)
182 * NOTE that in order for this to work we need to perform another getsym
183 * to start at the beginning of the hash links for symbols whose names
184 * are the same.
185 */
186 sp2 = getsym(LOCAL_SYMNAME(sptr), strlen(SYMNAME(sptr)));
187 sp2 = getocsym(sp2, OC_OTHER);
188 if (sp2 != sptr)
189 return sp2;
190 /*
191 * create a new symbol with the same name:
192 */
193 error(I_0035_Predefined_intrinsic_OP1_loses_intrinsic_property,
194 ERR_Informational, gbl.lineno, SYMNAME(sptr), CNULL);
195 sp2 = insert_sym(sptr);
196
197 /* transfer dtype if it was explicitly declared for sptr: */
198
199 if (DCLDG(sptr)) {
200 DTYPEP(sp2, DTYPEG(sptr));
201 DCLDP(sp2, 1);
202 }
203
204 return sp2;
205}

207SPTR
208ref_ident(SPTR sptr)
209{
210 SPTR sym;
211
212 sym = refsym(sptr, OC_OTHER);
213 if (IS_INTRINSIC(STYPEG(sym))) {
214 sym = newsym(sym);
215 if (sym == 0)
216 sym = insert_sym(sptr);
217 }
218 if (STYPEG(sym) == ST_UNKNOWN)
219 STYPEP(sym, ST_IDENT);
220
221 return sym;
222}
