61void
62comm_generator(void)
63{
64 int std, stdnext;
65 int ast;
66 int type;
67
68 rt_outvalue();
69 eliminate_redundant();
70
71 for (std = STD_NEXT(0); std; std = stdnext) {
72 stdnext = STD_NEXT(std);
73 gbl.lineno = STD_LINENO(std);
74 ast = STD_AST(std);
75 type = A_TYPEG(ast);
76 if (type == A_ASN)
77 type = A_TYPEG(A_SRCG(ast));
78 switch (type) {
79 case A_HALLOBNDS:
80 generate_hallobnds(ast);
81 break;
82 case A_HSECT:
83 generate_sect(ast);
84 break;
85 case A_HCOPYSECT:
86 generate_copy(ast);
87 break;
88 case A_HGATHER:
89 generate_gather(ast);
90 break;
91 default:
92 break;
93 }
94 }
95 generate_get_scalar();
96 if (!XBIT(58, 0x8000000)) {
97 pointer_changer();
98 }
99}

101static void
102generate_get_scalar(void)
103{
104
105 int l, lsym;
106 int astnew;
107 int i, j, nargs, argt;
108 int asd;
109 int ndim;
110 int same;
111 int dest, src, dest1;
112 int commstd, rt;
113 int std, ast;
114 int nd, a;
115 int asn, lop;
116
117 init_gstbl();
118 find_get_scalar();
119
120 for (i = 0; i < gstbl.avl; i++) {
121 commstd = gstbl.base[i].f1;
122 rt = STD_AST(commstd);
123 nd = A_OPT1G(rt);
124 assert(A_TYPEG(rt) == A_HGETSCLR, "generate_get_scalar: wrong ast type", 2,
125 rt);
126 assert(nd, "generate_get_scalar: something is wrong", 2, rt);
127 std = FT_STD(nd); /* this is where getscalar is used */
128 ast = STD_AST(std);
129 if (STD_DELETE(commstd)) {
130 same = FT_GETSCLR_SAME(nd);
131 src = A_SRCG(same);
132 dest = A_DESTG(same);
133 dest1 = A_DESTG(rt);
134 asn = mk_stmt(A_ASN, 0);
135 A_DESTP(asn, dest1);
136 A_SRCP(asn, dest);
137 add_stmt_after(asn, commstd);
138 assert(src && dest, "generate_get_scalar: something is wrong", 2, rt);
139 delete_stmt(commstd);
140 continue;
141 }
142
143 a = A_SRCG(rt);
144 asd = A_ASDG(a);
145 ndim = ASD_NDIM(asd);
146 l = A_LOPG(a);
147 lsym = memsym_of_ast(l);
148 dest = A_DESTG(rt);
149
150 /* put out a call to fetch the value */
151 /* call pghpf_get_scalar(temp, array_base, array, subscripts) */
152 nargs = ndim + 3;
153 argt = mk_argt(nargs);
154 ARGT_ARG(argt, 0) = dest;
155 ARGT_ARG(argt, 1) = l;
156 if (A_LOPG(rt)) {
157 ARGT_ARG(argt, 2) = A_LOPG(rt);
158 } else {
159 ARGT_ARG(argt, 2) = check_member(l, mk_id(DESCRG(lsym)));
160 }
161 DESCUSEDP(lsym, 1);
162 for (j = 0; j < ndim; ++j) {
163 a = mk_default_int(ASD_SUBS(asd, j));
164 if (normalize_bounds(lsym))
165 a = sub_lbnd(A_DTYPEG(l), j, a, 0);
166 ARGT_ARG(argt, j + 3) = a;
167 }
168 astnew = mk_stmt(A_CALL, 0);
169 lop = mk_id(sym_mkfunc(mkRteRtnNm(RTE_get_scalar), DT_NONE));
170 A_LOPP(astnew, lop);
171 A_ARGCNTP(astnew, nargs);
172 A_ARGSP(astnew, argt);
173 add_stmt_before(astnew, commstd);
174 report_comm(commstd, GETSCALAR_CAUSE);
175 delete_stmt(commstd);
176 continue;
177 }
178 free_gstbl();
179}

182generate_hallobnds(int ast)
183{

185 int i;
186 int asd, ndim;
187 int subs[7];
188 int arr;
189 int std;

191 int same;

197 std = A_STDG(ast);
198 if (STD_DELETE(std)) {
199 same = FT_ALLOC_SAME(nd);
200 delete_stmt(std);
201 return;
202 /*A_OTRIPLEP(ast, A_OTRIPLEG(same)); */
203 }
204 STD_DELETE(std) = 1;
205 deallocstd = FT_ALLOC_FREE(nd);
206 arr = A_LOPG(ast);
207 asd = A_ASDG(arr);
208 ndim = ASD_NDIM(asd);
209 for (i = 0; i < ndim; i++)
210 subs[i] = ASD_SUBS(asd, i);
211 newalloc = mk_mem_allocate(A_LOPG(arr), subs, std, 0);
212 STD_RESCOPE(newalloc) = 1;
213 if (deallocstd) {
214 newdealloc = mk_mem_deallocate(A_LOPG(arr), deallocstd);
215 STD_RESCOPE(newdealloc) = 1;
216 }
217 delete_stmt(std);

268static void
269generate_copy(int ast)
270{
271 int freestd;
272 int sptr;
273 int std;
274 int nd, nd1;
275 int same;
276 int cp;
277 int copy;
278 int alloc_std, alloc;
279 int nargs, argt, astnew;
280 int sectl_std, sectl;
281 int sectr_std, sectr;
282 int src, dest;
283 int asn, desc;
284 int atp;
285
286 copy = A_SRCG(ast);
287 assert(A_TYPEG(copy) == A_HCOPYSECT, "generate_copy: wrong ast type", 2, ast);
288 nd = A_OPT1G(copy);
289 assert(nd, "generate_sect: some thing is wrong", 2, ast);
290 std = A_STDG(ast);
291 if (STD_DELETE(std)) {
292 same = FT_CCOPY_SAME(nd);
293 delete_stmt(std);
294 return;
295 }
296
297 sectl_std = FT_CCOPY_SECTL(nd);
298 sectl = A_SRCG(STD_AST(sectl_std));
299 nd1 = A_OPT1G(sectl);
300 sptr = FT_SECT_OUT(nd1);
301 desc = check_member(FT_SECT_ARR(nd1), mk_id(sptr));
302 A_DDESCP(copy, desc);
303
304 sectr_std = FT_CCOPY_SECTR(nd);
305 sectr = A_SRCG(STD_AST(sectr_std));
306 nd1 = A_OPT1G(sectr);
307 sptr = FT_SECT_OUT(nd1);
308 desc = check_member(FT_SECT_ARR(nd1), mk_id(sptr));
309 A_SDESCP(copy, desc);
310
311 if (FT_CCOPY_USELHS(nd)) {
312 dest = A_LOPG(FT_CCOPY_LHS(nd));
313 A_DESTP(copy, dest);
314 } else {
315 alloc_std = FT_CCOPY_ALLOC(nd);
316 alloc = STD_AST(alloc_std);
317 nd1 = A_OPT1G(alloc);
318 sptr = FT_ALLOC_OUT(nd1);
319 dest = mk_id(sptr);
320 A_DESTP(copy, dest);
321 }
322
323 src = A_SRCG(copy);
324 A_SRCP(copy, A_LOPG(src));
325
326 freestd = FT_CCOPY_FREE(nd);
327 cp = FT_CCOPY_OUT(nd);
328
329 /* cp =pghpf_comm_copy_(void *db, void *sb, section *sd, section *ss); */
330 nargs = 4;
331 argt = mk_argt(nargs);
332 ARGT_ARG(argt, 0) = A_DESTG(copy);
333 ARGT_ARG(argt, 1) = A_SRCG(copy);
334 ARGT_ARG(argt, 2) = A_DDESCG(copy);
335 ARGT_ARG(argt, 3) = A_SDESCG(copy);
336
337 astnew = mk_func_node(A_FUNC,
338 mk_id(sym_mkfunc(mkRteRtnNm(RTE_comm_copy), DT_ADDR)),
339 nargs, argt);
340 A_DTYPEP(astnew, DT_INT);
341 NODESCP(A_SPTRG(A_LOPG(astnew)), 1);
342
343 asn = mk_stmt(A_ASN, 0);
344 dest = mk_id(cp);
345 A_DESTP(asn, dest);
346 A_SRCP(asn, astnew);
347 add_stmt_before(asn, std);
348
349 /* pghpf_comm_free_(cp) */
350 /* free communication schedules */
351
352 argt = mk_argt(2);
353 ARGT_ARG(argt, 0) = astb.i1;
354 ARGT_ARG(argt, 1) = mk_id(cp);
355 ast = mk_stmt(A_CALL, 0);
356 atp = mk_id(sym_mkfunc(mkRteRtnNm(RTE_comm_free), DT_NONE));
357 A_LOPP(ast, atp);
358 NODESCP(A_SPTRG(A_LOPG(ast)), 1);
359 A_ARGCNTP(ast, 2);
360 A_ARGSP(ast, argt);
361 add_stmt_after(ast, freestd);
362 delete_stmt(std);
363}

365static void
366generate_gather(int ast)
367{
368 int freestd;
369 int sptr;
370 int i;
371 int std;
372 int nd, nd1;
373 int same;
374 int cp;
375 int gather;
376 int alloc_std, alloc;
377 int nargs, argt, astnew;
378 int sectm_std, sectm;
379 int sectv_std, sectv;
380 int dest;
381 int asn;
382 int v;
383 int j;
384 int mask, mask_sec_ast;
385 int ndim1, asd1;
386 int npermute;
387 int vsub, nvsub, vsub_ast;
388 int vsubstd, nvsubstd, sectvsub, sectnvsub;
389 int nvsub_ast, vsub_sec_ast, nvsub_sec_ast;
390 int vec[7], vecsec[7], permute[7];
391 int array_sec_ast, result_sec_ast, result_ast, array_ast;
392 int vflag, pflag, nvec;
393 int vdim, pdim;
394 int func;
395 int commstd;
396 int comm_type;
397 int atp;
398
399 gather = A_SRCG(ast);
400 assert(A_TYPEG(gather) == A_HGATHER, "generate_gather: wrong ast type", 2,
401 ast);
402 nd = A_OPT1G(gather);
403 assert(nd, "generate_sect: some thing is wrong", 2, ast);
404 std = A_STDG(ast);
405 if (STD_DELETE(std)) {
406 same = FT_CGATHER_SAME(nd);
407 delete_stmt(std);
408 return;
409 }
410
411 vsub = FT_CGATHER_VSUB(nd);
412 vsub_ast = A_LOPG(vsub);
413 asd1 = A_ASDG(vsub);
414 ndim1 = ASD_NDIM(asd1);
415
416 vsubstd = FT_CGATHER_SECTVSUB(nd);
417 if (vsubstd) {
418 sectvsub = A_SRCG(STD_AST(vsubstd));
419 nd1 = A_OPT1G(sectvsub);
420 vsub_sec_ast = check_member(vsub_ast, mk_id(FT_SECT_OUT(nd1)));
421 } else {
422 sptr = memsym_of_ast(vsub);
423 vsub_sec_ast = check_member(vsub_ast, DESCRG(sptr));
424 }
425
426 nvsub = FT_CGATHER_NVSUB(nd);
427 nvsubstd = FT_CGATHER_SECTNVSUB(nd);
428
429 if (nvsubstd) {
430 sectnvsub = A_SRCG(STD_AST(nvsubstd));
431 nd1 = A_OPT1G(sectnvsub);
432 nvsub_sec_ast = check_member(A_LOPG(sectnvsub), mk_id(FT_SECT_OUT(nd1)));
433 } else {
434 sptr = memsym_of_ast(A_LOPG(nvsub));
435 nvsub_sec_ast = check_member(nvsub, DESCRG(sptr));
436 }
437
438 comm_type = FT_CGATHER_TYPE(nd);
439
440 if (comm_type == A_HGATHER) {
441 if (FT_CGATHER_USELHS(nd)) {
442 nvsub_ast = A_LOPG(FT_CGATHER_LHS(nd));
443 } else {
444 alloc_std = FT_CGATHER_ALLOC(nd);
445 alloc = STD_AST(alloc_std);
446 nd1 = A_OPT1G(alloc);
447 nvsub_ast = mk_id(FT_ALLOC_OUT(nd1));
448 }
449 } else if (comm_type == A_HSCATTER) {
450 nvsub_ast = A_LOPG(nvsub);
451 }
452
453 mask = A_MASKG(gather);
454 if (mask) {
455 sectm_std = FT_CGATHER_SECTM(nd);
456 sectm = A_SRCG(STD_AST(sectm_std));
457 nd1 = A_OPT1G(sectm);
458 mask_sec_ast = mk_id(FT_SECT_OUT(nd1));
459 } else {
460 mask = mk_cval(1, DT_LOG);
461 mask_sec_ast = mk_cval(dtype_to_arg(A_DTYPEG(mask)), DT_INT);
462 }
463
464 if (comm_type == A_HGATHER) {
465 func = mk_id(sym_mkfunc(mkRteRtnNm(RTE_comm_gatherx), DT_ADDR));
466 array_sec_ast = vsub_sec_ast;
467 result_sec_ast = nvsub_sec_ast;
468 result_ast = nvsub_ast;
469 array_ast = vsub_ast;
470 } else if (comm_type == A_HSCATTER) {
471 func = mk_id(sym_mkfunc(mkRteRtnNm(RTE_comm_scatterx), DT_ADDR));
472 result_sec_ast = vsub_sec_ast;
473 array_sec_ast = nvsub_sec_ast;
474 result_ast = vsub_ast;
475 array_ast = nvsub_ast;
476 }
477
478 vflag = FT_CGATHER_VFLAG(nd);
479 vdim = FT_CGATHER_VDIM(nd);
480 pflag = FT_CGATHER_PFLAG(nd);
481 pdim = FT_CGATHER_PDIM(nd);
482 nvec = FT_CGATHER_NVEC(nd);
483 npermute = FT_CGATHER_NPER(nd);
484
485 asd1 = A_ASDG(vsub);
486 ndim1 = ASD_NDIM(asd1);
487 for (i = 0; i < ndim1; i++) {
488 if (getbit(vdim, i)) {
489 int vsptr;
490 v = FT_CGATHER_V(nd, i);
491 vsptr = memsym_of_ast(v);
492 sectv_std = FT_CGATHER_SECTV(nd, i);
493 if (sectv_std) {
494 sectv_std = FT_CGATHER_SECTV(nd, i);
495 sectv = A_SRCG(STD_AST(sectv_std));
496 nd1 = A_OPT1G(sectv);
497 sptr = FT_SECT_OUT(nd1);
498 } else
499 sptr = DESCRG(vsptr);
500
501 vec[i] = v;
502 vecsec[i] = check_member(v, mk_id(sptr));
503 }
504 if (getbit(pdim, i)) {
505 permute[i] = FT_CGATHER_PERMUTE(nd, i);
506 }
507 }
508
509 nargs = 2 * 3 + 1 + 1 + 2 * nvec + npermute;
510 argt = mk_argt(nargs);
511
512 ARGT_ARG(argt, 0) = result_ast;
513 ARGT_ARG(argt, 1) = array_ast;
514 ARGT_ARG(argt, 2) = mask;
515
516 /* sections */
517 ARGT_ARG(argt, 3) = result_sec_ast;
518 ARGT_ARG(argt, 4) = array_sec_ast;
519 ARGT_ARG(argt, 5) = mask_sec_ast;
520
521 ARGT_ARG(argt, 6) = mk_cval(vflag, DT_INT);
522 ARGT_ARG(argt, 7) = mk_cval(pflag, DT_INT);
523 j = 8;
524 asd1 = A_ASDG(vsub);
525 ndim1 = ASD_NDIM(asd1);
526 for (i = 0; i < ndim1; i++) {
527 if (getbit(vdim, i)) {
528 ARGT_ARG(argt, j) = vec[i];
529 j++;
530 ARGT_ARG(argt, j) = vecsec[i];
531 j++;
532 }
533 if (getbit(pdim, i)) {
534 ARGT_ARG(argt, j) = permute[i];
535 j++;
536 }
537 }
538
539 astnew = mk_func_node(A_FUNC, func, nargs, argt);
540 A_DTYPEP(astnew, DT_INT);
541
542 cp = FT_CGATHER_OUT(nd);
543
544 asn = mk_stmt(A_ASN, 0);
545 dest = mk_id(cp);
546 A_DESTP(asn, dest);
547 A_SRCP(asn, astnew);
548 commstd = add_stmt_before(asn, std);
549 NODESCP(A_SPTRG(A_LOPG(astnew)), 1);
550 delete_stmt(std);
551
552 if (!FT_CGATHER_INDEXREUSE(nd)) {
553 /*
554 * Free communication schedules:
555 * call pghpf_comm_free_(cp)
556 */
557 argt = mk_argt(2);
558 ARGT_ARG(argt, 0) = astb.i1;
559 ARGT_ARG(argt, 1) = mk_id(cp);
560 ast = mk_stmt(A_CALL, 0);
561 atp = mk_id(sym_mkfunc(mkRteRtnNm(RTE_comm_free), DT_NONE));
562 A_LOPP(ast, atp);
563 NODESCP(A_SPTRG(A_LOPG(ast)), 1);
564 A_ARGCNTP(ast, 2);
565 A_ARGSP(ast, argt);
566
567 freestd = FT_CGATHER_FREE(nd);
568 add_stmt_after(ast, freestd);
569 }
570}

573static void
574generate_shift(int ast)
575{
576 int freestd;
577 int sptr;
578 int i, j;
579 int asd, ndim;
580 int std;
581 int nd;
582 int same;
583 int cp;
584 int shift;
585 int nargs, argt, astnew;
586 int src, dest;
587 int asn;
588 int align;
589 int type, boundary;
590 int atp;
591
592 shift = A_SRCG(ast);
593 assert(A_TYPEG(shift) == A_HOVLPSHIFT, "generate_shift: wrong ast type", 2,
594 ast);
595 nd = A_OPT1G(shift);
596 std = A_STDG(ast);
597 if (STD_DELETE(std)) {
598 assert(nd, "generate_shift: can't delete stmt", 2, ast);
599 same = FT_SHIFT_SAME(nd);
600 delete_stmt(std);
601 return;
602 }
603
604 src = A_SRCG(shift);
605 sptr = memsym_of_ast(src);
606 align = ALIGNG(sptr);
607 asd = A_ASDG(src);
608 ndim = ASD_NDIM(asd);
609 DESCUSEDP(sptr, 1);
610
611 if (nd) {
612 freestd = FT_SHIFT_FREE(nd);
613 cp = FT_SHIFT_OUT(nd);
614 } else {
615 freestd = 0;
616 cp = A_SPTRG(A_DESTG(ast));
617 }
618
619 /* create overlap_shift schedule
620 * cp = pghpf_comm_shift_(void *b, section *g, ...);
621 * ... = -shift, +shift for each dim
622 */
623 type = (nd ? FT_SHIFT_TYPE(nd) : 0);
624 if (type == I_EOSHIFT)
625 nargs = ndim * 2 + 2 + 1;
626 else
627 nargs = ndim * 2 + 2;
628
629 argt = mk_argt(nargs);
630 j = 0;
631 ARGT_ARG(argt, j) = A_LOPG(src);
632 j++;
633 ARGT_ARG(argt, j) = check_member(A_LOPG(src), mk_id(DESCRG(sptr)));
634 j++;
635 DESCUSEDP(sptr, 1);
636 if (type == I_EOSHIFT) {
637 boundary = FT_SHIFT_BOUNDARY(nd);
638 if (!boundary)
639 boundary = astb.ptr0;
640 ARGT_ARG(argt, j) = boundary;
641 j++;
642 }
643
644 for (i = 0; i < ndim; ++i) {
645 ARGT_ARG(argt, j) = mk_default_int(A_LBDG(ASD_SUBS(asd, i)));
646 j++;
647 ARGT_ARG(argt, j) = mk_default_int(A_UPBDG(ASD_SUBS(asd, i)));
648 j++;
649 }
650
651 if (type == I_EOSHIFT)
652 astnew = mk_func_node(
653 A_FUNC, mk_id(sym_mkfunc(mkRteRtnNm(RTE_olap_eoshift), DT_ADDR)), nargs,
654 argt);
655 else if (type == I_CSHIFT)
656 astnew = mk_func_node(
657 A_FUNC, mk_id(sym_mkfunc(mkRteRtnNm(RTE_olap_cshift), DT_ADDR)), nargs,
658 argt);
659 else
660 astnew = mk_func_node(
661 A_FUNC, mk_id(sym_mkfunc(mkRteRtnNm(RTE_olap_shift), DT_ADDR)), nargs,
662 argt);
663
664 A_DTYPEP(astnew, DT_INT);
665 NODESCP(A_SPTRG(A_LOPG(astnew)), 1);
666
667 asn = mk_stmt(A_ASN, 0);

871is_use_lhs_final(int a, int forall, LOGICAL same_nidx, LOGICAL independent,

918 return FALSE;

1228/* Algorithm:
1229 * cyclic(1) distribution with perfect alignment.
1230 *

1238 * lof = lof + los
1239 * end do
1240 *
1241 * Variable names has the same meaning in the comment and code.
1242 */

1245fill_cyclic_1(int ast)
1246{

1255 assert(A_TYPEG(ast) == A_HCYCLICLP, "fill_cyclic_1: wrong ast type", 2, ast);
1256
1257 nd = A_OPT1G(ast);
1258 forall = STD_AST(FT_STD(nd));
1259 assert(A_TYPEG(forall) == A_FORALL, "fill_cyclic_1: some thing is wrong", 2,
1260 ast);
1261 nd1 = A_OPT1G(forall);
1262 ct = FT_CYCLIC(nd1);
1263
1264 itriple = A_ITRIPLEG(ast);
1265 otriple = A_OTRIPLEG(ast);
1266 otriple1 = A_OTRIPLE1G(ast);
1267 lof0 = A_LBDG(otriple1);
1268 los = A_UPBDG(otriple1);
1269 lof = A_STRIDEG(otriple1);

1276 A_DESTP(astnew, lof);
1277 A_SRCP(astnew, lof0);

1281 A_DESTP(astnew, lof);

1283 A_SRCP(astnew, rhs);
1284 ct->c_inc[cdim] = astnew;

1287/* Algorithm:
1288 * call pghpf_cyclic_loop(d_a, dim, l, u, s, cl, cu, cs, lof0, los)
1289 * do ci = cl, cu, cs
1290 * lof = lof0
1291 * call pghpf_block_loop(d_a, dim, l, u, s, ci, bl, bu, bs)
1292 * do i = bl, bu, bs
1293 * a(i-lof) = ... rhs(i) ...
1294 * end do
1295 * lof = lof + los
1296 * end do
1297 *
1298 * Variable names has the same meaning in the comment and code.
1299 */
1300
1301static int
1302fill_cyclic_k(int ast)
1303{
1304 int lb, ub;
1305 int l, u, s;
1306 int argt;
1307 int cl, cu, cs, ci;
1308 int lof, los, lof0;
1309 int astnew;
1310 int nargs;
1311 CTYPE *ct;
1312 int nd, nd1;
1313 int itriple, otriple, otriple1;
1314 int dim, cdim;
1315 int forall, lop, src;
1316
1317 assert(A_TYPEG(ast) == A_HCYCLICLP, "fill_cyclic_k: wrong ast type", 2, ast);
1318
1319 nd = A_OPT1G(ast);
1320 forall = STD_AST(FT_STD(nd));
1321 assert(A_TYPEG(forall) == A_FORALL, "fill_cyclic_k: some thing is wrong", 2,
1322 ast);
1323 nd1 = A_OPT1G(forall);
1324 ct = FT_CYCLIC(nd1);
1325
1326 itriple = A_ITRIPLEG(ast);
1327 l = A_LBDG(itriple);
1328 u = A_UPBDG(itriple);
1329 s = A_STRIDEG(itriple);
1330 if (s == 0)
1331 s = mk_cval(1, DT_INT);
1332 otriple = A_OTRIPLEG(ast);
1333 cl = A_LBDG(otriple);
1334 cu = A_UPBDG(otriple);
1335 cs = A_STRIDEG(otriple);
1336
1337 otriple1 = A_OTRIPLE1G(ast);
1338 lof0 = A_LBDG(otriple1);
1339 los = A_UPBDG(otriple1);
1340 lof = A_STRIDEG(otriple1);
1341 dim = A_DIMG(ast);
1342 cdim = get_int_cval(A_SPTRG(A_ALIASG(dim))) - 1;
1343
1344 ct->c_lof[cdim] = lof;
1345
1346 astnew = mk_stmt(A_ASN, DT_INT);
1347 A_DESTP(astnew, lof);
1348 A_SRCP(astnew, lof0);
1349 ct->cb_init[cdim] = astnew;
1350
1351 astnew = mk_stmt(A_DO, 0);
1352 ci = mk_id(trans_getbound(0, 8));
1353 A_DOVARP(astnew, ci);
1354 A_M1P(astnew, cl);
1355 A_M2P(astnew, cu);
1356 A_M3P(astnew, cs);
1357 A_M4P(astnew, 0);
1358 ct->cb_do[cdim] = astnew;
1359
1360 lb = mk_id(trans_getbound(0, 0));
1361 ub = mk_id(trans_getbound(0, 1));
1362
1363 nargs = 8;
1364 argt = mk_argt(nargs);
1365 ARGT_ARG(argt, 0) = A_LOPG(ast);
1366 /* DESCUSEDP(lhs, 1); */
1367
1368 ARGT_ARG(argt, 1) = mk_cval(cdim + 1, DT_INT);
1369 ARGT_ARG(argt, 2) = mk_default_int(l);
1370 ARGT_ARG(argt, 3) = mk_default_int(u);
1371 ARGT_ARG(argt, 4) = mk_default_int(s);
1372 ARGT_ARG(argt, 5) = mk_default_int(ci);
1373 ARGT_ARG(argt, 6) = mk_default_int(lb);
1374 ARGT_ARG(argt, 7) = ub;
1375 astnew = mk_stmt(A_CALL, 0);
1376 lop = mk_id(sym_mkfunc(mkRteRtnNm(RTE_block_loop), DT_NONE));
1377 A_LOPP(astnew, lop);
1378 A_ARGCNTP(astnew, nargs);
1379 A_ARGSP(astnew, argt);
1380 ct->cb_block[cdim] = astnew;
1381
1382 astnew = mk_stmt(A_ASN, DT_INT);
1383 A_DESTP(astnew, lof);
1384 src = mk_binop(OP_ADD, lof, los, DT_INT);
1385 A_SRCP(astnew, src);
1386 ct->cb_inc[cdim] = astnew;
1387
1388 astnew = mk_stmt(A_ENDDO, 0);
1389 ct->cb_enddo[cdim] = astnew;
1390 return mk_triple(lb, ub, 0);
1391}

1746static int
1747pointer_squeezer(int ast)
1748{
1749 int atype;
1750 int astnew;
1751 int opnd;
1752 int l, r, m1, m2, m3, m4, v;
1753 int dtype;
1754 int asd;
1755 int numdim;
1756 int subs[7];
1757 int argt;
1758 int argcnt;
1759 int argtnew;
1760 int i, any;
1761 int astli, astlinew;
1762 int sptr;
1763 int align;
1764
1765 if (ast == 0)
1766 return 0; /* watch for a 'null' argument */
1767 dtype = A_DTYPEG(ast);
1768 switch (atype = A_TYPEG(ast)) {
1769 case A_CMPLXC:
1770 case A_CNST:
1771 case A_ID:
1772 case A_LABEL:
1773 astnew = ast;
1774 break;
1775 case A_MEM:
1776 astnew = pointer_squeezer((int)A_PARENTG(ast));
1777 if (astnew == A_PARENTG(ast)) {
1778 astnew = ast;
1779 } else {
1780 astnew = mk_member(astnew, A_MEMG(ast), A_DTYPEG(ast));
1781 }
1782 break;
1783 case A_SUBSTR:
1784 opnd = pointer_squeezer((int)A_LOPG(ast));
1785 l = pointer_squeezer((int)A_LEFTG(ast));
1786 r = pointer_squeezer((int)A_RIGHTG(ast));
1787 if (opnd == A_LOPG(ast) && l == A_LEFTG(ast) && r == A_RIGHTG(ast)) {
1788 astnew = ast;
1789 } else {
1790 astnew = mk_substr(opnd, l, r, dtype);
1791 }
1792 break;
1793 case A_BINOP:
1794 l = pointer_squeezer((int)A_LOPG(ast));
1795 r = pointer_squeezer((int)A_ROPG(ast));
1796 if (l == A_LOPG(ast) && r == A_ROPG(ast)) {
1797 astnew = ast;
1798 } else {
1799 astnew = mk_binop((int)A_OPTYPEG(ast), l, r, dtype);
1800 }
1801 break;
1802 case A_UNOP:
1803 l = pointer_squeezer((int)A_LOPG(ast));
1804 if (l == A_LOPG(ast)) {
1805 astnew = ast;
1806 } else {
1807 astnew = mk_unop((int)A_OPTYPEG(ast), l, dtype);
1808 }
1809 break;
1810 case A_PAREN:
1811 l = pointer_squeezer((int)A_LOPG(ast));
1812 if (l == A_LOPG(ast)) {
1813 astnew = ast;
1814 } else {
1815 astnew = mk_paren(l, dtype);
1816 }
1817 break;
1818 case A_CONV:
1819 l = pointer_squeezer((int)A_LOPG(ast));
1820 if (l == A_LOPG(ast)) {
1821 astnew = ast;
1822 } else {
1823 astnew = mk_convert(l, dtype);
1824 }
1825 break;
1826 case A_SUBSCR:
1827 any = 0;
1828 opnd = pointer_squeezer((int)A_LOPG(ast));
1829 if (opnd != A_LOPG(ast))
1830 ++any;
1831 asd = A_ASDG(ast);
1832 numdim = ASD_NDIM(asd);
1833 sptr = sptr_of_subscript(ast);
1834 align = ALIGNG(sptr);
1835 if (!POINTERG(sptr) || A_SHAPEG(ast)) {
1836 for (i = 0; i < numdim; ++i) {
1837 l = pointer_squeezer((int)ASD_SUBS(asd, i));
1838 if (l != ASD_SUBS(asd, i))
1839 ++any;
1840 subs[i] = l;
1841 }
1842 if (any == 0) {
1843 astnew = ast;
1844 } else {
1845 astnew = mk_subscr(opnd, subs, numdim, dtype);
1846 }
1847 break;
1848 }
1849 assert(numdim > 0 && numdim <= 7, "pointer_squeezer: bad numdim", ast, 4);
1850 for (i = 0; i < numdim; ++i) {
1851 l = pointer_squeezer((int)ASD_SUBS(asd, i));
