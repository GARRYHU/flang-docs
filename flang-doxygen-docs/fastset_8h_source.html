8/** \file
9 * \brief Fast sets of small nonnegative integers.
10 *
11 * Element membership testing and removal are constant-time operations, as is
12 * emptying a set. Adding a member to a set is constant-time unless memory
13 * allocation is required. Element traversal has time complexity linear in the
14 * number elements. Set union is O(elements of second set); set difference and
15 * set intersection are O(elements of smallest set).
16 *
17 * Always use fastset_init() after declaring or allocating a new set, and
18 * always use fastset_free() when the set is destroyed.
19 *
20 * The data structure is an example of a space/time trade-off being
21 * made entirely in favor of time. These sets should not be used for values
22 * larger than a few thousand. (For sets with really small maximum values,
23 * consider using bitsets; for sets with larger maximum values, try a 3-2 tree.)
24 *
25 * The idea for this fast set data structure is attributed by Preston Briggs
26 * in his register assignment thesis to himself and Linda Torczon.
27 */
28

74/**
75 * \brief Implementation
76 *
77 * Sets are represented by a pair of arrays. The first, set->member[],
78 * contains the distinct elements of the set in arbitrary order in its
79 * leading set->members positions, and garbage afterwards. For each
80 * element x in a set, *both* of these predicates must be true:
81 * 1) 0 <= set->index[x] < set->members, and
82 * 2) set->member[set->index[x]] == x
83 *
84 * For all values x that are *not* in a set, either x >= set->limit, or
85 * set->index[x] >= set->members, or set->member[set->index[x]] != x.
86 *
87 * Unsigned integers are used internally to reduce needless comparisons
88 * to zero.
89 *
90 * The predicate fastset_contains(), q.v. below, demonstrates all of the
91 * conditions that hold when a value is an element of a fastset. The
92 * value must be small enough to be used as an offset into set->index[],
93 * and the index value at that position must be small enough to be used
94 * as an offset into the list of distinct set values stored contiguously
95 * in the leading elements of the set->member[] array.
96 *
97 * Note that neither of the set->member[] or set->index[] array contents
98 * need be initialized, and that a set can be emptied in constant time by
99 * setting set->members = 0 as in fastset_vacate(). It is okay if the values
100 * in set->index[] are garbage or if multiple values point to the same
101 * offset in set->member[].
102 *
103 */
104struct fastset {
105 unsigned members, limit, *member, *index;
106};

158/** Predicate: is a particular value a current member of a set?
159 * Any integer value "x" may be tested.
160 */
161INLINE static int
162fastset_contains(const fastset *set, int x)
163{
164 unsigned idx;
165 return (unsigned)x < set->limit && (idx = set->index[x]) < set->members &&
166 set->member[idx] == (unsigned)x;
167}

180/** Adds a value to a set, if it is distinct from current members. */
181INLINE static void
182fastset_add(fastset *set, unsigned x)
183{
184 if (x >= set->limit)
185 fastset_resize(set, x + 1);
186 else {
187 unsigned idx = set->index[x];
188 if (idx < set->members && set->member[idx] == x)
189 return;
190 }
191 set->member[set->index[x] = set->members++] = x;
192}
193
194/** Removes a value from a set, if present. "x" can be any integer, since
195 * out-of-range values can't be in the set anyway.
196 */
197INLINE static void
198fastset_remove(fastset *set, int x)
199{
200 if ((unsigned)x < set->limit) {
201 unsigned idx = set->index[x];
202 if (idx < set->members && set->member[idx] == (unsigned)x) {
203 /* x is a member of the set; remove it and fill the vacated
204 * position with the last set member
205 */
206 unsigned last = set->member[--set->members];
207 set->member[set->index[last] = idx] = last;
208 }
209 }
210}
