9 * CPP and some helper C functions to convert SSE/AVX Intel/GNU intrinsics to
10 * IBM Altivec equivalents.
11 *
12 * Not all intrinsics translate as single Altivec instruction, but most
13 * do. For an example, see "FMA Instructions" below.

17 * XXX XXX XXX
18 * WARNING WARNING WARNING WARNING WARNING WARNING WARNING WARNING
19 *
20 * Some of the CPP marcos used for scalar operations are not correct with
21 * regards to elements other than 0 (first in the vector).
22 *
23 * The macros below work with the FMA versions of the math intrinsics

41/*
42 * No corresponding Altivec intrinsic to generate a scalar mask
43 * from corresponding vector elements.

45static inline unsigned int
46_mm_movemask_epi8(__m128i a)
47{
48 unsigned char t[16] __attribute__((aligned(16)));
49 unsigned int r;
50 int i;
51
52 vec_st((vector unsigned char)a, 0, t);
53 r = 0;
54 for (i = 0; i < 16; i++) {
55 r = (r << 1) | (t[i] >> 7);
56 }
57 return r;
58}

60static inline unsigned int
61_mm_movemask_epi32(__m128i a)
62{
63 unsigned int t[4] __attribute__((aligned(16)));
64 unsigned int r;
65 int i;
66
67 vec_st((vector unsigned int)a, 0, t);
68 r = 0;
69 for (i = 0; i < 4; i++) {
70 r = (r << 1) | (t[i] >> 31);
71 }
72 return r;
73}

75static inline __m128i
76_mm_blend_epi32 (__m128i a, __m128i b, int imm8)
77{
78 unsigned int t[4] __attribute__((aligned(16)));
79 int i;
80
81 vec_st((vector unsigned int) a, 0, t);
82 for (i = 0; i < 3; i++) {
83 if (imm8 & 0x1)
84 t[i] = vec_extract((vector unsigned int)b, i);
85 imm8 >>= 1;
86 }
87 return vec_ld(0, (__m128i *)t);
88}

104static inline __m128d
105_mm_shuffle_pd(__m128d a, __m128d b, int imm8)
106{
107 double r[2];
108 r[0] = imm8 & 0x1 ? vec_extract(a, 1) : vec_extract(a, 0);
109 r[1] = imm8 & 0x2 ? vec_extract(b, 1) : vec_extract(b, 0);
110
111 return vec_ld(0, (__m128d *)r);
112}

136static inline __m128d
137_mm_min_sd (__m128d a, __m128d b)
138{
139 double aa = vec_extract(a, 0);
140 double bb = vec_extract(b, 0);
141 aa < bb ? aa : bb;
142 return vec_insert(aa, a, 0);
143}

145static inline __m128d
146_mm_max_sd (__m128d a, __m128d b)
147{
148 double aa = vec_extract(a, 0);
149 double bb = vec_extract(b, 0);
150 aa > bb ? aa : bb;
151 return vec_insert(aa, a, 0);
