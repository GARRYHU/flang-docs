79static void
80add_ctor(const char *constructor)
81{
82 LL_Type *ret = ll_create_basic_type(cpu_llvm_module, LL_VOID, 0);
83 LL_Function *fn;
84 char buff[128];
85 snprintf(buff, sizeof(buff), "declare void @%s()", constructor);
86 fn = ll_create_function(cpu_llvm_module, buff + 13, ret, 0, 0, 0, "",
87 LL_NO_LINKAGE);
88 llvm_ctor_add(constructor);
89 ll_proto_add(fn->name, NULL);
90 ll_proto_set_intrinsic(fn->name, buff);
91}

93char *
94put_next_member(char *ptr)
95{
96 if (!ptr)
97 return NULL;
98 if (*ptr == ',')
99 ptr++;
100 while (*ptr != ',' && *ptr != '\0') {
101 fprintf(ASMFIL, "%c", *ptr);
102 ptr++;
103 }
104 fprintf(ASMFIL, " ");
105 return ptr;
106}

108ISZ_T
109put_skip(ISZ_T old, ISZ_T New, bool is_char)
110{
111 ISZ_T amt;
112 const char *s = "i8 0";
113 const char *str = "i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,"
114 "i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,"
115 "i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0,i8 0";
116
117 if (is_char) {
118 s = "i8 32";
119 str = "i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,"
120 "i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,"
121 "i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32,i8 32";
122 }
123 if ((amt = New - old) > 0) {
124 INT i;
125 i = amt;
126 while (i > 32) {
127 fprintf(ASMFIL, str);
128 i -= 32;
129 if (i)
130 fprintf(ASMFIL, ",");
131 }
132 if (i) {
133 while (1) {
134 fprintf(ASMFIL, s);
135 i--;
136 if (i == 0)
137 break;
138 fprintf(ASMFIL, ",");
139 }
140 }
141 } else {
142 assert(amt == 0, "assem.c-put_skip old,new not in sync", New, ERR_Severe);
143 }
144 return amt;
145}

156void
157emit_init(DTYPE tdtype, ISZ_T tconval, ISZ_T *addr, ISZ_T *repeat_cnt,
158 ISZ_T loc_base, ISZ_T *i8cnt, int *ptrcnt, char **cptr, bool is_char)
159{
160 ISZ_T al;
161 int area;
162 int size_of_item;
163 int putval;
164 INT skip_size;
165 char str[32];
166 area = LLVM_LONGTERM_AREA;
167 const ISZ_T orig_tconval = tconval;
168 char *oldptr;
169
170 switch ((int)tdtype) {
171 case 0: /* alignment record */
172 /* word or halfword alignment required: */
173#if DEBUG
174 assert(tconval == 7 || tconval == 3 || tconval == 1 || tconval == 0,
175 "emit_init:bad align", (int)tconval, ERR_Severe);
176#endif
177 skip_size = ALIGN(*addr, tconval) - *addr;
178 if (skip_size == 0) {
179 *addr = ALIGN(*addr, tconval);
180 break;
181 }
182 if (DBGBIT(5, 32)) {
183 fprintf(gbl.dbgfil, "emit_init:0 first_data:%d i8cnt:%ld ptrcnt:%d\n",
184 first_data, *i8cnt, *ptrcnt);
185 }
186 if (*ptrcnt) {
187 if (!first_data)
188 fprintf(ASMFIL, ", ");
189 *cptr = put_next_member(*cptr);
190 fprintf(ASMFIL, "[" /*]*/);
191 *ptrcnt = 0;
192 } else if (!(*i8cnt)) {
193 if (!first_data)
194 fprintf(ASMFIL, ", ");
195 *cptr = put_next_member(*cptr);
196 fprintf(ASMFIL, "[" /*]*/);
197 } else if (*i8cnt) {
198 if (!first_data)
199 fprintf(ASMFIL, ", ");
200 }
201 *i8cnt = *i8cnt + put_skip(*addr, ALIGN(*addr, tconval), is_char);
202 *addr = ALIGN(*addr, tconval);
203 first_data = 0;
204 break;
205 case DINIT_ZEROES:
206 if (!tconval) {
207 *addr += tconval;
208 break;
209 }
210 if (DBGBIT(5, 32)) {
211 fprintf(gbl.dbgfil,
212 "emit_init:DINIT_ZEROES first_data:%d i8cnt:%ld ptrcnt:%d\n",
213 first_data, *i8cnt, *ptrcnt);
214 }
215 if (*ptrcnt) {
216 if (!first_data)
217 fprintf(ASMFIL, ", ");
218 *cptr = put_next_member(*cptr);
219 fprintf(ASMFIL, "[" /*]*/);
220 *ptrcnt = 0;
221 } else if (!(*i8cnt)) {
222 if (!first_data)
223 fprintf(ASMFIL, ", ");
224 *cptr = put_next_member(*cptr);
225 fprintf(ASMFIL, "[" /*]*/);
226 } else if (*i8cnt) {
227 if (!first_data)
228 fprintf(ASMFIL, ", ");
229 }
230 put_zeroes((int)tconval);
231 *i8cnt = *i8cnt + ((int)tconval);
232 *addr += tconval;
233 first_data = 0;
234 break;
235#ifdef DINIT_PROC
236 case DINIT_PROC:
237 if (*i8cnt) {
238 fprintf(ASMFIL, /*[*/ "] ");
239 *i8cnt = 0;
240 }
241 if (!first_data) {
242 fprintf(ASMFIL, ",");
243 } else {
244 first_data = 0;
245 }
246 write_proc_pointer((SPTR)tconval);
247 (*ptrcnt)++;
248 *addr += size_of(DT_CPTR);
249 break;
250#endif
251 case DINIT_LABEL:
252 /* word to be init'ed with address of label 'tconval' */
253 al = alignment(DT_CPTR);
254 skip_size = ALIGN(*addr, al) - *addr;
255 if (DBGBIT(5, 32)) {
256 fprintf(gbl.dbgfil,
257 "emit_init:DINIT_LABEL first_data:%d i8cnt:%ld ptrcnt:%d\n",
258 first_data, *i8cnt, *ptrcnt);
259 }
260
261 if (skip_size) { /* if *i8cnt - just add to the end */
262 if (!first_data)
263 fprintf(ASMFIL, ", ");
264 if (*i8cnt) {
265 *i8cnt = put_skip(*addr, ALIGN(*addr, al), is_char);
266 *i8cnt = 0;
267 fprintf(ASMFIL, /*[*/ "], ");
268 } else if (*ptrcnt || !(*i8cnt)) {
269#ifdef OMP_OFFLOAD_LLVM
270 // TODO ompaccel. Hackery for TGT structs. It must be fixed later.
271 if (flg.omptarget)
272 fprintf(ASMFIL, " ptr ");
273 else
274#endif
275 *cptr = put_next_member(*cptr);
276 fprintf(ASMFIL, "[" /*]*/);
277 *i8cnt = put_skip(*addr, ALIGN(*addr, al), is_char);
278 fprintf(ASMFIL, /*[*/ "], ");
279 }
280 } else if (*i8cnt) {
281 fprintf(ASMFIL, /*[*/ "], ");
282 *i8cnt = 0;
283 } else if (!first_data)
284 fprintf(ASMFIL, ", ");
285#ifdef OMP_OFFLOAD_LLVM
286 // TODO ompaccel. Hackery for TGT structs. It must be fixed later.
287 if (flg.omptarget)
288 fprintf(ASMFIL, " ptr ");
289 else
290#endif
291 *cptr = put_next_member(*cptr);
292 *addr = ALIGN(*addr, al);
293 if (DBGBIT(5, 32)) {
294 fprintf(gbl.dbgfil,
295 "emit_init:DINIT_LABEL calling put_addr "
296 "first_data:%d i8cnt:%ld ptrcnt:%d\n",
297 first_data, *i8cnt, *ptrcnt);
298 }
299 put_addr((SPTR)tconval, 0, DT_NONE); // ???
300 (*ptrcnt)++;
301 *addr += size_of(DT_CPTR);
302 first_data = 0;
303 break;
304#ifdef DINIT_FUNCCOUNT
305 case DINIT_FUNCCOUNT:
306 gbl.func_count = tconval;
307 break;
308#endif
309 case DINIT_OFFSET:
310 skip_size = (tconval + loc_base) - *addr;
311 if (skip_size == 0) {
312 *addr = tconval + loc_base;
313 break;
314 }
315 if (DBGBIT(5, 32)) {
316 fprintf(gbl.dbgfil,
317 "emit_init:DINIT_OFFSET first_data:%d i8cnt:%ld ptrcnt:%d\n",
318 first_data, *i8cnt, *ptrcnt);
319 }
320 if (*ptrcnt) {
321 if (!first_data)
322 fprintf(ASMFIL, ", ");
323 *cptr = put_next_member(*cptr);
324 fprintf(ASMFIL, "[" /*]*/);
325 *ptrcnt = 0;
326 } else if (!(*i8cnt)) {
327 if (!first_data)
328 fprintf(ASMFIL, ", ");
329 *cptr = put_next_member(*cptr);
330 fprintf(ASMFIL, "[" /*]*/);
331 } else if (*i8cnt) {
332 if (!first_data)
333 fprintf(ASMFIL, ", ");
334 }
335 *i8cnt = *i8cnt + put_skip(*addr, tconval + loc_base, is_char);
336 *addr = tconval + loc_base;
337 first_data = 0;
338 break;
339 case DINIT_REPEAT:
340 *repeat_cnt = tconval;
341 break;
342 case DINIT_SECT:
343 break;
344 case DINIT_DATASECT:
345 break;
346 case DINIT_STRING:
347 /* read the string from the dinit file until the length is exhausted */
348 *addr += tconval;
349 if (tconval == 0) {
350 break;
351 }
352 if (DBGBIT(5, 32)) {
353 fprintf(gbl.dbgfil,
354 "emit_init:DINIT_STRING first_data:%d i8cnt:%ld ptrcnt:%d\n",
355 first_data, *i8cnt, *ptrcnt);
356 }
357 if (*ptrcnt) {
358 if (!first_data)
359 fprintf(ASMFIL, ", ");
360 *cptr = put_next_member(*cptr);
361 fprintf(ASMFIL, "[" /*]*/);
362 *ptrcnt = 0;
363 } else if (!(*i8cnt)) {
364 if (!first_data)
365 fprintf(ASMFIL, ", ");
366 *cptr = put_next_member(*cptr);
367 fprintf(ASMFIL, "[" /*]*/);
368 } else if (*i8cnt) {
369 if (!first_data)
370 fprintf(ASMFIL, ", ");
371 }
372
373 /* Output the data */
374 *i8cnt += tconval;
375 while (tconval > 0) {
376 if (tconval != orig_tconval)
377 fprintf(ASMFIL, ", ");
378 if (tconval > 32) {
379 dinit_read_string(32, str);
380 put_string_n(str, 32, 0);
381 tconval -= 32;
382 } else {
383 dinit_read_string(tconval, str);
384 put_string_n(str, tconval, 0);
385 tconval = 0;
386 }
387 }
388
389 /* We have printed out an entire string, close it */
390 first_data = 0;
391 break;
392 default:
393 assert(tdtype > 0, "emit_init:bad dinit rec", tdtype, ERR_Severe);
394 size_of_item = size_of(tdtype);
395
396 if (*repeat_cnt > 1) {
397 /* TO DO: We may be able to optimize this with zeroinitializer
398 * if all i8 before and after this are all zero.
399 *
400 */
401 switch (DTY(tdtype)) {
402 case TY_INT8:
403 case TY_LOG8:
404 if (CONVAL2G(tconval) == 0 &&
405 (!XBIT(124, 0x400) || CONVAL1G(tconval) == 0))
406 goto do_zeroes;
407 break;
408 case TY_INT:
409 case TY_LOG:
410 case TY_SINT:
411 case TY_SLOG:
412 case TY_BINT:
413 case TY_BLOG:
414 case TY_FLOAT:
415 if (tconval == 0)
416 goto do_zeroes;
417 break;
418 case TY_DBLE:
419 if (tconval == stb.dbl0)
420 goto do_zeroes;
421 break;
422#ifdef TARGET_SUPPORTS_QUADFP
423 case TY_QUAD:
424 if (tconval == stb.quad0)
425 goto do_zeroes;
426 break;
427#endif
428 case TY_CMPLX:
429 if (CONVAL1G(tconval) == 0 && CONVAL2G(tconval) == 0)
430 goto do_zeroes;
431 break;
432 case TY_DCMPLX:
433 if (CONVAL1G(tconval) == stb.dbl0 && CONVAL2G(tconval) == stb.dbl0)
434 goto do_zeroes;
435 break;
436#ifdef TARGET_SUPPORTS_QUADFP
437 case TY_QCMPLX:
438 if (CONVAL1G(tconval) == stb.quad0 && CONVAL2G(tconval) == stb.quad0)
439 goto do_zeroes;
440 break;
441#endif
442#ifdef LONG_DOUBLE_FLOAT128
443 case TY_FLOAT128:
444 if (tconval == stb.float128_0)
445 goto do_zeroes;
446 break;
447 case TY_CMPLX128:
448 if (CONVAL1G(tconval) == stb.float128_0 &&
449 CONVAL2G(tconval) == stb.float128_0)
450 goto do_zeroes;
451 break;
452#endif /* LONG_DOUBLE_FLOAT128 */
453 default:
454 break;
455 }
456 }
457 /* emit data value, loop if repeat count present */
458 putval = 1;
459 if (size_of_item == 0) {
460 putval = 0;
461 *repeat_cnt = 1;
462 break;
463 }
464 do {
465 if (DTY(tdtype) != TY_PTR && DTY(tdtype) != TY_STRUCT) {
466 if (*ptrcnt) {
467 if (!first_data)
468 fprintf(ASMFIL, ", ");
469 *cptr = put_next_member(*cptr);
470 fprintf(ASMFIL, " [" /*]*/);
471 *ptrcnt = 0;
472 } else if (!(*i8cnt)) {
473 if (!first_data)
474 fprintf(ASMFIL, ", ");
475 *cptr = put_next_member(*cptr);
476 fprintf(ASMFIL, " [" /*]*/);
477 } else if (*i8cnt) {
478 if (!first_data)
479 fprintf(ASMFIL, ", ");
480 }
481 }
482 switch (DTY(tdtype)) {
483 case TY_INT8:
484 case TY_LOG8:
485 case TY_DWORD:
486 if (DBGBIT(5, 32)) {
487 fprintf(gbl.dbgfil,
488 "emit_init:put_i32 first_data:%d i8cnt:%ld ptrcnt:%d\n",
489 first_data, *i8cnt, *ptrcnt);
490 }
491 put_i32(CONVAL2G(tconval));
492 fprintf(ASMFIL, ", ");
493 if (DBGBIT(5, 32)) {
494 fprintf(gbl.dbgfil,
495 "emit_init:put_i32 first_data:%d i8cnt:%ld ptrcnt:%d\n",
496 first_data, *i8cnt, *ptrcnt);
497 }
498 if (XBIT(124, 0x400)) {
499 put_i32(CONVAL1G(tconval));
500 } else {
501 put_i32(0);
502 }
503 break;
504
505 case TY_INT:
506 case TY_LOG:
507 case TY_WORD:
508 if (DBGBIT(5, 32)) {
509 fprintf(gbl.dbgfil,
510 "emit_init:put_i32 first_data:%d i8cnt:%ld ptrcnt:%d\n",
511 first_data, *i8cnt, *ptrcnt);
512 }
513 put_i32(tconval);
514 break;
515 case TY_SINT:
516 case TY_SLOG:
517 if (DBGBIT(5, 32)) {
518 fprintf(gbl.dbgfil,
519 "emit_init:put_i16 first_data:%d i8cnt:%ld ptrcnt:%d\n",
520 first_data, *i8cnt, *ptrcnt);
521 }
522 put_i16((int)tconval);
523 break;
524
525 case TY_BINT:
526 case TY_BLOG:
527 if (DBGBIT(5, 32)) {
528 fprintf(gbl.dbgfil,
529 "emit_init:put_i8 first_data:%d i8cnt:%ld ptrcnt:%d\n",
530 first_data, *i8cnt, *ptrcnt);
531 }
532 put_i8((int)tconval);
533 break;
534
535 case TY_FLOAT:
536 if (DBGBIT(5, 32)) {
537 fprintf(gbl.dbgfil,
538 "emit_init:put_r4 first_data:%d i8cnt:%ld ptrcnt:%d\n",
539 first_data, *i8cnt, *ptrcnt);
540 }
541 put_r4(tconval);
542 break;
543
544 case TY_DBLE:
545 if (DBGBIT(5, 32)) {
546 fprintf(gbl.dbgfil,
547 "emit_init:put_r8 first_data:%d i8cnt:%ld ptrcnt:%d\n",
548 first_data, *i8cnt, *ptrcnt);
549 }
550 put_r8((int)tconval, putval);
551 break;
552
553#ifdef TARGET_SUPPORTS_QUADFP
554 case TY_QUAD:
555 if (DBGBIT(5, 32)) {
556 fprintf(gbl.dbgfil,
557 "emit_init:put_r18 first_data:%d i8cnt:%ld ptrcnt:%d\n",
558 first_data, *i8cnt, *ptrcnt);
559 }
560 put_r16((int)tconval, putval);
561 break;
562#endif
563
564 case TY_CMPLX:
565 if (DBGBIT(5, 32)) {
566 fprintf(gbl.dbgfil,
567 "emit_init:put_cmplx_n first_data:%d i8cnt:%ld ptrcnt:%d\n",
568 first_data, *i8cnt, *ptrcnt);
569 }
570 put_cmplx_n((int)tconval, putval);
571 break;
572
573 case TY_DCMPLX:
574 if (DBGBIT(5, 32)) {
575 fprintf(gbl.dbgfil,
576 "emit_init:put_dcmplx_n first_data:%d i8cnt:%ld ptrcnt:%d\n",
577 first_data, *i8cnt, *ptrcnt);
578 }
579 put_dcmplx_n((int)tconval, putval);
580 break;
581
582#ifdef TARGET_SUPPORTS_QUADFP
583 case TY_QCMPLX:
584 if (DBGBIT(5, 32)) {
585 fprintf(gbl.dbgfil,
586 "emit_init:put_qcmplx_n first_data:%d i8cnt:%ld ptrcnt:%d\n",
587 first_data, *i8cnt, *ptrcnt);
588 }
589 put_qcmplx_n((int)tconval, putval);
590 break;
591#endif
592
593 case TY_PTR:
594 if (*i8cnt) {
595 fprintf(ASMFIL, /*[*/ "], ");
596 } else if (!first_data)
597 fprintf(ASMFIL, ", ");
598 *ptrcnt = *ptrcnt + 1;
599 *i8cnt = 0;
600 oldptr = *cptr;
601 *cptr = put_next_member(*cptr);
602
603 if (DBGBIT(5, 32)) {
604 fprintf(gbl.dbgfil,
605 "emit_init:put_addr first_data:%d i8cnt:%ld ptrcnt:%d\n",
606 first_data, *i8cnt, *ptrcnt);
607 }
608 if (STYPEG(tconval) != ST_CONST) {
609 put_addr(SPTR_NULL, tconval, DT_NONE);
610 } else {
611 SPTR conval1 = SymConval1((SPTR)tconval);
612 ISZ_T conval2 = CONVAL2G(tconval);
613
614 if ((!conval1) && (conval2)) {
615 fprintf(ASMFIL, "inttoptr (i64 %ld to ", (long)conval2);
616 oldptr = put_next_member(oldptr);
617 assert(oldptr == *cptr, "emit_init:unexpected behaviour", (int)conval2, ERR_Severe);
618 fprintf(ASMFIL, ")");
619 } else
620 put_addr(conval1, conval2, DT_NONE);
621 }
622 break;
623
624 case TY_CHAR:
625 size_of_item = DTY(DTYPEG(tconval) + 1);
626 if (DBGBIT(5, 32)) {
627 fprintf(gbl.dbgfil,
628 "emit_init:put_str_n first_data:%d i8cnt:%ld ptrcnt:%d\n",
629 first_data, *i8cnt, *ptrcnt);
630 }
631 put_string_n(stb.n_base + CONVAL1G((int)tconval),
632 DTY(DTYPEG((int)tconval) + 1), 0);
633 break;
634
635 case TY_NCHAR:
636 /* need to write nchar in 2 bytes because we make everything as i8 */
637 put_ncharstring_n(stb.n_base + CONVAL1G((int)tconval),
638 DTY(DTYPEG((int)tconval) + 1), 16);
639 break;
640
641 case TY_STRUCT:
642 if (is_empty_typedef(tdtype)) {
643 break;
644 }
645 // TODO: Print the struct?
646 break;
647
648#ifdef LONG_DOUBLE_FLOAT128
649 case TY_X87:
650 put_r8(CONVAL1G(tconval), putval);
651 fputc(',', ASMFIL);
652 put_r8(CONVAL2G(tconval), putval);
653 fputc(',', ASMFIL);
654 put_r8(CONVAL3G(tconval), putval);
655 fputc(',', ASMFIL);
656 put_r8(0, putval);
657 put_r8(CONVAL4G(tconval), putval);
658 fputc(',', ASMFIL);
659 break;
660 case TY_X87CMPLX:
661 put_r8(CONVAL1G(CONVAL1G(tconval)), putval);
662 fputc(',', ASMFIL);
663 put_r8(CONVAL2G(CONVAL1G(tconval)), putval);
664 fputc(',', ASMFIL);
665 put_r8(CONVAL3G(CONVAL1G(tconval)), putval);
666 fputc(',', ASMFIL);
667 put_r8(CONVAL4G(CONVAL1G(tconval)), putval);
668 fputc(',', ASMFIL);
669 put_r8(0, putval);
670 put_r8(CONVAL1G(CONVAL2G(tconval)), putval);
671 fputc(',', ASMFIL);
672 put_r8(CONVAL2G(CONVAL2G(tconval)), putval);
673 fputc(',', ASMFIL);
674 put_r8(CONVAL3G(CONVAL2G(tconval)), putval);
675 fputc(',', ASMFIL);
676 put_r8(CONVAL4G(CONVAL2G(tconval)), putval);
677 fputc(',', ASMFIL);
678 put_r8(0, putval);
679 break;
680#endif /* LONG_DOUBLE_FLOAT128 */

688 putval = 0;
689 first_data = 0;
690
691 } while (--(*repeat_cnt));
692 *repeat_cnt = 1;
693 break;
694 do_zeroes:
695 if (*ptrcnt) {
696 if (!first_data)
697 fprintf(ASMFIL, ", ");
698 *cptr = put_next_member(*cptr);
699 fprintf(ASMFIL, " [" /*]*/);
700 *ptrcnt = 0;
701 } else if (!(*i8cnt)) {
702 if (!first_data)
703 fprintf(ASMFIL, ", ");

705 fprintf(ASMFIL, " [" /*]*/);

707 if (!first_data)

712 "emit_init:put_zeroes at end first_data:%d i8cnt:%ld ptrcnt:%d\n",

714 }
715 put_zeroes((*repeat_cnt) * size_of_item);
716 *i8cnt = *i8cnt + (*repeat_cnt) * size_of_item;
717 *addr += (*repeat_cnt) * size_of_item;
718 *repeat_cnt = 1;
719 first_data = 0;
720 break;
721 }

931/**
932 \brief Generate an expression to add an offset to a ptr
933 \param offset the addend
934 \param ret_type the type of the pointer (LL_Type)
935 \param ptr_nm the identifier for the pointer
936
937 For example,
938 <pre>
939 getelementptr(i8* bitcast(<ret_type> <ptr_nm> to i8*), i32 <offset>)
940 </pre>
941
942 The caller expects a string that is an i8*.
943 */
944LL_Value *
945gen_ptr_offset_val(int offset, LL_Type *ret_type, const char *ptr_nm)
946{
947 /* LL_Type for i8* (used as bitcast target for GEP to get byte offsets) */
948 LL_Type *ll_type_i8_ptr =
949 ll_get_pointer_type(ll_create_int_type(cpu_llvm_module, 8));
950
951 /* Create an LL_Value from LL_Type ... */
952 LL_Value *llv = ll_create_value_from_type(cpu_llvm_module, ret_type, ptr_nm);
953 /*... and use it to generate a bitcast instruction to i8* */
954 llv = ll_get_const_bitcast(cpu_llvm_module, llv, ll_type_i8_ptr);
955 /*... then use it to generate a GEP instruction to get element at a byte
956 * offset */
957 llv = ll_get_const_gep(cpu_llvm_module, llv, 1, offset);
958 return llv;
959}

961/**
962 \brief Produce a getelementptr that would fetch a value out of one of the
963 global structs.
964
965 Print something along the lines of (LLVM version dependent):
966 \verbatim
967 getelementptr (i8* bitcast (%struct$name* @getsname(sptr)
968 to i8*), i32 0) to i8*)
969 \endverbatim
970 */
971void
972put_addr(SPTR sptr, ISZ_T off, DTYPE dtype)
973{
974 const char *name, *elem_type;
975 bool is_static_or_common_block_var, in_fortran;
976
977 in_fortran = false;
978 in_fortran = true;
979
980 /* Static and common block variables require special handling for now */
981 is_static_or_common_block_var = (SCG(sptr) == SC_STATIC);
982#ifdef SC_CMBLK
983 is_static_or_common_block_var =
984 is_static_or_common_block_var || (SCG(sptr) == SC_CMBLK);
985#endif
986
987 elem_type = "";
988 /* Decide whether we need to provide element type to GEP */
989 if (ll_feature_explicit_gep_load_type(&cpu_llvm_module->ir))
990 elem_type = "i8, ";
991
992 if (sptr) {
993 if ((name = getsname(sptr))) {
994 if (is_static_or_common_block_var && in_fortran) {
995 /* Statics and common block initializations in Fortran are
996 * strings composed using fprintf;
997 * FIXME need to generate them using LL_Value, like below */
998
999 /* Statics and common blocks in fortran as stored as structs, we need to
1000 * add offset inside of struct */
1001 off += ADDRESSG(sptr);
1002
1003 /* Text type for contansts is produced via char_type */
1004 if (STYPEG(sptr) == ST_CONST)
1005 fprintf(ASMFIL,
1006 "getelementptr(%sptr @%s, i32 %ld)",
1007 elem_type, name, off);
1008 /* Structures have type name mirroring variable name */
1009 else
1010 fprintf(
1011 ASMFIL,
1012 "getelementptr(%sptr @%s, i32 %ld)",
1013 elem_type, name, off);
1014 } else {
1015 /* Convert to LLVM-compatible structures */
1016 if (!LLTYPE(sptr)) {
1017 process_sptr(sptr);
1018 }
1019
1020 LL_Type *ll_type = LLTYPE(sptr);
1021
1022 /* Convert to pointer type if needed
1023 * TODO implications of this are unclear, it works for now
1024 * because this is only used in data initialization */
1025 if (ll_type->data_type != LL_PTR && need_ptr(sptr, SCG(sptr), dtype))
1026 ll_type = ll_get_pointer_type(ll_type);
1027
1028 /* Produce pointer offset code */
1029 LL_Value *ll_offset = gen_ptr_offset_val(off, ll_type, SNAME(sptr));
1030 fprintf(ASMFIL, "%s", ll_offset->data);
1031 }
1032 } else
1033 fprintf(ASMFIL, "null");
1034 } else if (off == 0)
1035 fprintf(ASMFIL, "null");
1036 else
1037 fprintf(ASMFIL, "%ld", (long)off);
1038}
