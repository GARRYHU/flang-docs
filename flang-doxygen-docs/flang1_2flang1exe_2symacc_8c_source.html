36void
37sym_init_first(void)
38{
39 int i;
40
41 int sizeof_SYM = sizeof(SYM) / sizeof(INT);
42/* Disable checking SYM size on Windows
43 * https://github.com/flang-compiler/flang/issues/1043
44 */
45#ifndef _WIN64
46 assert(sizeof_SYM == 44, "bad SYM size", sizeof_SYM, ERR_Fatal);
47#endif

49 if (stb.stg_base == NULL) {
50 STG_ALLOC(stb, 1000);
51 assert(stb.stg_base, "sym_init: no room for symtab", stb.stg_size,

59 ERR_Fatal);
60 /* basically, this is sidecar of dt_base */
61
62 stb.w_size = 32;
63 NEW(stb.w_base, INT, stb.w_size);
64 assert(stb.w_base, "sym_init: no room for wtab", stb.w_size, ERR_Fatal);
65 }
66 /* allocate deepcopy info */
67
68 stb.namavl = 1;
69 stb.wrdavl = 0;
70 for (i = 0; i <= HASHSIZE; i++)
71 stb.hashtb[i] = SPTR_NULL;
72
73 DT_INT = DT_INT4;
74 DT_REAL = DT_REAL4;
75 DT_CMPLX = DT_CMPLX8;
76 DT_LOG = DT_LOG4;
77 DT_DBLE = DT_REAL8;
78 DT_DCMPLX = DT_CMPLX16;
79 DT_PTR = DT_INT4;

82/** \brief Expand symbol storage area when NEWSYM runs out of area.
83
84 It is assumed that stb.stg_avail is 1 more than the index of the current
85 symbol being created. */
86void
87realloc_sym_storage()
88{
89 DEBUG_ASSERT(stb.stg_avail > stb.stg_size,
90 "realloc_sym_storage: call only if necessary");
91 if (stb.stg_avail > SPTR_MAX + 1 || stb.stg_base == NULL)
92 symini_errfatal(7);
93 /* Use unsigned arithmetic to avoid risk of overflow. */
94 DEBUG_ASSERT(stb.stg_size > 0,
95 "realloc_sym_storage: symbol storage not initialized?");
96 STG_NEED(stb);
97 DEBUG_ASSERT(stb.stg_avail <= stb.stg_size,
98 "realloc_sym_storage: internal error");
99}

101/**
102 \brief Look up symbol with indicated name.
103 \return If there is already such a symbol, the pointer to the
104 existing symbol table entry; or 0 if a symbol doesn't exist.
105 \param name is a symbol name.
106 \param olength is the number of characters in the symbol name.
107 */
108SPTR
109lookupsym(const char *name, int olength)
110{
111 int length;
112 SPTR sptr; /* pointer to symbol table entry */
113 INT hashval; /* index into hashtb. */
114
115 /*
116 * Loop thru the appropriate hash link list to see if symbol is
117 * already in the table:
118 */
119
120 length = olength;
121 if (length > MAXIDLEN) {
122 length = MAXIDLEN;
123 }
124 HASH_ID(hashval, name, length);
125 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
126 if (HIDDENG(sptr))
127 continue;
128 if (strncmp(name, SYMNAME(sptr), length) != 0 ||
129 *(SYMNAME(sptr) + length) != '\0')
130 continue;
131
132 /* matching entry has been found in symbol table. return it: */
133
134 return sptr;
135 }
136 return SPTR_NULL;
137} /* lookupsym */

139/** \brief Issue diagnostic for identifer that is too long.
140
141 \param name - identifier (without terminating njull)
142 \param olength - length of identifier
143 \param max_idlen - maximum allowed length
144
145 Though this routine has only one lexical call site, it is factored
146 out to not clutter the common path in installsym_ex.
147 */
148static void
149report_too_long_identifier(const char *name, int olength, int max_idlen)
150{
151 static char *ebuf;
152 static int ebuf_sz = 0;
153 char len_buf[12];
154 if (ebuf_sz == 0) {
155 ebuf_sz = olength + 1;
156 NEW(ebuf, char, ebuf_sz);
157 } else {
158 int ii;
159 NEED(olength + 1, ebuf, char, ebuf_sz, olength + 1);
160 ii = strlen(ebuf);
161 if (ii < olength)
162 strcpy(ebuf + (ii - 2), "..."); /* there's room for at least 1 '.'*/
163 }
164 memcpy(ebuf, name, olength);
165 ebuf[olength] = '\0';
166 sprintf(len_buf, "%d", max_idlen);
167 symini_error(16, 2, gbl.lineno, ebuf, len_buf);
168}

195/**
196 \brief Enter symbol with indicated name into symbol table,
197 initialize the new entry, and return pointer to it. If there is
198 already such a symbol, just return pointer to the existing symbol
199 table entry.
200
201 \param name is the symbol name.
202 \param olength is the number of characters in the symbol name.
203 */
204SPTR
205installsym_ex(const char *name, int olength, IS_MODE mode)
206{
207 int length;
208 SPTR sptr; /* pointer to symbol table entry */
209 INT hashval; /* index into hashtb. */
210 bool toolong;
211 int nmptr;
212 static int max_idlen = MAXIDLEN;
213
214 /*
215 * Trim identifier if it is too long.
216 */
217 toolong = false;
218 length = olength;
219 if (flg.standard) {
220 max_idlen = STANDARD_MAXIDLEN;
221 }
222 if (length > max_idlen) {
223 length = max_idlen;
224 toolong = true;
225 }
226
227 nmptr = 0;
228 if (mode != IS_QUICK) {
229 /*
230 * Loop thru the appropriate hash link list to see if symbol is
231 * already in the table.
232 */
233 HASH_ID(hashval, name, length);
234 for (sptr = stb.hashtb[hashval]; sptr != 0; sptr = HASHLKG(sptr)) {
235 const char *sname;
236 int np = NMPTRG(sptr);
237 if (np + length >= stb.namavl)
238 continue;
239 sname = stb.n_base + np;
240 if (sname[0] != name[0] || sname[length] != '\0')
241 continue;
242 if (strncmp(name, sname, length) != 0)
243 continue;
244 nmptr = np;
245 if (HIDDENG(sptr))
246 continue;
247
248 /* Matching entry has been found in symbol table. Return it. */
249
250 return sptr;
251 }
252 }
253
254 /* Symbol not found. Create a new symbol table entry. */
255
256 NEWSYM(sptr);
257 if (mode != IS_QUICK) {
258 LINKSYM(sptr, hashval);
259 }
260
261 if (!nmptr)
262 nmptr = putsname(name, length);
263 NMPTRP(sptr, nmptr);
264 SYMLKP(sptr, NOSYM);
265 SCOPEP(sptr, 0);
266#ifdef LINENOP
267 LINENOP(sptr, gbl.lineno);
268#endif
269
270 if (toolong) {
271 report_too_long_identifier(name, olength, max_idlen);
272 }
273
274 return sptr;
275}

277/**
278 \brief Put a string of characters into the symbol names storage
279 area and return pointer to the string (relative to
280 stb.n_base). This routine is used to enter both symbol names and
281 character string constants.
282
283 \param name are the characters to be entered.
284 \param length is the number of characters to be entered.
285 */
286int
287putsname(const char *name, int length)
288{
289 int nptr; /* index into character storage area */
290 char *np; /* pointer into character storage area */
291 int i; /* counter */
292
293 nptr = stb.namavl;
294 stb.namavl += (length + 1);
295 while (stb.namavl > stb.n_size) {
296 /* To avoid quadratic behavior, we increase the storage area size
297 by a factor, not a constant. Use unsigned arithmetic here
298 to avoid risk of overflow. */
299 unsigned n = 2u * stb.n_size;
300 if (n > MAX_NMPTR) {
301 n = MAX_NMPTR;
302 if (stb.namavl > n)
303 symini_errfatal(7); /* names table overflow */
304 }
305 NEED(stb.namavl, stb.n_base, char, stb.n_size, n);
306 }
307 np = stb.n_base + nptr;
308 for (i = 0; i < length; i++)
309 *np++ = *name++;
310 *np = '\0';
311
312 return nptr;
313}

315/**
316 \brief Create a local copy of a name known to be stored in the 'stb.n_base'
317 area.
318
319 Used when a symbol needs to be created from a name stored in the
320 area; a purify umr error could occur if the area is realloc'd. The
321 char pointer to the copy is returned.
322 */
323char *
324local_sname(char *name)
325{
326 static char *safe_p;
327 static int safe_sz = 0;
328 int length;
329
330 length = strlen(name) + 2 + 6; /* MW: add one more character,
331 needed in semfunc2 */
332 /* Hongyon: add 6 more for
333 _cr and _nm for cref,nomixed */
334 if (safe_sz == 0) {
335 safe_sz = length + 100;
336 NEW(safe_p, char, safe_sz);
337 } else {
338 NEED(length, safe_p, char, safe_sz, length + 100);
339 }
340
341 strcpy(safe_p, name);
342
343 return safe_p;
344}

346void
347add_fp_constants(void)
348{
349 INT tmp[4];
350#if defined(LONG_DOUBLE_FLOAT128) || defined(TARGET_SUPPORTS_QUADFP)
351 INT res[4];
352#endif
353
354 tmp[0] = 0;
355 atoxf("0.0", &tmp[1], 3);
356 /***** Only the f90/hpf front-ends know aobut the sized-real *****/
357 /*
358 * NOTE (A REMINDER):
359 * 1. the order in which the following constants are created can
360 * never changed -- their sptr numbers as well what they denote
361 * is asseumed by the modfile processing.
362 * 2. additional predefined constants CANNOT be added unless
363 * we propagate how many more to symtab.c:can_map_initsym() and
364 * symtab.c:map_initsym().
365 * For now, just create them, such as stb.fltm0 and stb.dblm0,
366 * in symtab.c after symtab.c:sym_init() defines stb.firstosym
367 */
368
369 stb.flt0 = getcon(tmp, DT_REAL4);
370 atoxf("1.0", &tmp[1], 3);
371 stb.flt1 = getcon(tmp, DT_REAL4);
372 atoxf("2.0", &tmp[1], 3);
373 stb.flt2 = getcon(tmp, DT_REAL4);
374 atoxf("0.5", &tmp[1], 3);
375 stb.flthalf = getcon(tmp, DT_REAL4);
