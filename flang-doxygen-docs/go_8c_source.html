66/* Implementation of the go() macro, which passes __FILE__ and __LINE__
67 * to each call.
68 */
69int (*go_odometer(const char *file, int line))(const char *format, ...)
70{
71 if ((flg.dbg[64] & 1) || flg.dbg[65] > 0) {
72 static int odometer = 0;
73 if (++odometer < flg.dbg[65] || flg.dbg[65] <= 0) {
74 if (odometer + 1 == flg.dbg[65])
75 dbgprintf("[%d LAST!]%s(%d): ", odometer, file, line);
76 else
77 dbgprintf("[%d]%s(%d): ", odometer, file, line);
78 return yes;
79 } else {
80 dbgprintf("[%d NO]%s(%d): ", odometer, file, line);
81 return no;
82 }
83 } else {
84 return yes_silent;
85 }
86}
