32static LLUplevel *
33get_uplevel(int stblock_sptr)
34{
35 int key;
36 LLUplevel *up;
37 assert(STYPEG(stblock_sptr) == ST_BLOCK, "Uplevel key must be an ST_BLOCK",
38 stblock_sptr, ERR_Fatal);
39
40 /* Index */
41 key = PARSYMSG(stblock_sptr);
42
43 /* Locate uplevel pointer */
44 up = NULL;
45 if (key <= llmp_all_uplevels.avl)
46 up = (LLUplevel *)(&llmp_all_uplevels.base[key]);
47
48 assert(up && key, "Could not locate uplevel instance for stblock",
49 stblock_sptr, ERR_Fatal);
50
51 return up;
52}

73LLUplevel *
74llmp_create_uplevel(int stblock_sptr)
75{
76 int key;
77 LLUplevel *up;
78
79 assert(STYPEG(stblock_sptr) == ST_BLOCK, "Uplevel key must be an ST_BLOCK",
80 stblock_sptr, ERR_Fatal);
81
82 /* Avoid processing an already created uplevel */
83 if (PARSYMSG(stblock_sptr))
84 return get_uplevel(stblock_sptr);
85
86 /* Make room if necessary */
87 if (llmp_all_uplevels.avl == 0) {
88 llmp_all_uplevels.avl = 2;
89 key = 1;
90 } else {
91 key = llmp_all_uplevels.avl;
92 ++llmp_all_uplevels.avl;
93 }
94 NEED(llmp_all_uplevels.avl, llmp_all_uplevels.base, LLUplevel,
95 llmp_all_uplevels.size, llmp_all_uplevels.size + 8);
96
97 up = (LLUplevel *)(&llmp_all_uplevels.base[key]);
98 memset(up, 0, sizeof(LLUplevel));
99
100 /* Add key and map it to stblock */
101 PARSYMSP(stblock_sptr, key);
102
103 return up;
104}

127/* Uniquely add shared_sptr to up */
128int
129llmp_add_shared_var(LLUplevel *up, int shared_sptr)
130{
131 int i;
132 const int idx = up->vals_count;
133
134 /* Unique add: I really wanted to make this a hashset... */
135 for (i = 0; i < up->vals_count; ++i) {
136 if (shared_sptr == 0)
137 break;
138 if (up->vals[i] == shared_sptr)
139 return 0;
140 }
141
142 ++up->vals_count;
143 NEED(up->vals_count, up->vals, int, up->vals_size, up->vals_size + 8);
144 up->vals[idx] = shared_sptr;
145 return 1;
146}

195void
196llmp_reset_uplevel(void)
197{
198 if (llmp_all_uplevels.avl) {
199 for (int i = 1; i < llmp_all_uplevels.avl; ++i) {
200 LLUplevel *up = (LLUplevel *)(&llmp_all_uplevels.base[i]);
201 if (up->vals_count)
202 FREE(up->vals);
203 }
204 FREE(llmp_all_uplevels.base);
205 memset(&llmp_all_uplevels, 0, sizeof(llmp_all_uplevels));
206 }
207 if (llmp_all_tasks.avl) {
208 for (int i = 0; llmp_all_tasks.avl; ++i) {
209 LLTask *task = (LLTask *)(&llmp_all_tasks.base[i]);
210 if (task->privs_count) {
211 FREE(task->privs);
212 }
213 FREE(llmp_all_tasks.base);
214 memset(&llmp_all_tasks, 0, sizeof(llmp_all_tasks));
215 }
216 }
217 llmp_all_uplevels.avl = 0;
218 llmp_all_tasks.avl = 0;
219}

261LLTask *
262llmp_create_task(int scope_sptr)
263{
264 NEED(llmp_all_tasks.avl + 1, llmp_all_tasks.base, LLTask, llmp_all_tasks.size,
265 llmp_all_tasks.size + 4);
266
267 LLTask *task = (LLTask *)(&llmp_all_tasks.base[llmp_all_tasks.avl]);
268 ++llmp_all_tasks.avl;
269 memset(task, 0, sizeof(LLTask));
270 task->actual_size = llmp_task_get_base_task_size();
271 task->scope_sptr = scope_sptr;
272 return task;
273}

275/* Return the size of an empty KMPC task (no shared variables):
276 * Pointer + Pointer + int32(+pad) +
277 * kmp_cmplrdata_t(data1) + kmp_cmplrdata_t(data2)
278 * see kmp.h
279 */
280int
281llmp_task_get_base_task_size(void)
282{
283 int pad = sizeof(void *) - sizeof(int);
284#ifdef TARGET_WIN
285 return sizeof(void *) + sizeof(void *) + sizeof(int) + pad +
286 sizeof(void *) * 2;
287#else
288 return sizeof(void *) + sizeof(void *) + sizeof(int32_t) + pad +
289 sizeof(void *) * 2;
290#endif
291}

320int
321llmp_task_add_private(LLTask *task, int shared_sptr, SPTR private_sptr)
322{
323#ifndef FE90
324 int pad = 0;
325 int size;
326 int align;
327 DTYPE dtype;
328#endif
329 int offset = 0;
330 LLFirstPrivate *fp;
331 int idx = task->privs_count;
332
333 NEED(++task->privs_count, task->privs, LLFirstPrivate, task->privs_size,
334 task->privs_size + 4);
335
336 /* Create the private object */
337 fp = (LLFirstPrivate *)&(task->privs[idx]);
338 fp->private_sptr = private_sptr;
339 fp->shared_sptr = shared_sptr;
340
341/* Bump up the size of the task to contain private_sptr */
342#ifdef FE90
343 task->actual_size += size_of_var(private_sptr);
344#else
345 dtype = DTYPEG(private_sptr);
346 if (dtype) {
347 size = zsize_of(dtype);
348 align = alignment(dtype);
349 pad = ALIGN(task->actual_size, align) - task->actual_size;
350 task->actual_size += pad;
351 }
352 offset = task->actual_size;
353 task->actual_size += size_of_sym(private_sptr);
354#endif
355 return offset;
356}

358int
359llmp_task_add_loopvar(LLTask *task, int num, DTYPE dtype)
360/* put loop variables on task_alloc array after private vars */
361{
362#ifndef FE90
363 int pad = 0;
364 int size;
365 int align;
366#endif
367 int offset = 0;
368#ifdef FE90
369 /* we add it to backend only */
370#else
371 /* Bump up the size of the task to contain loop var and make sure
372 * it is integer*64 aligned.
373 */
374 size = zsize_of(dtype) * num;
375 align = alignment(dtype);
376 pad = ALIGN(task->actual_size, align) - task->actual_size;
377 task->actual_size += pad;
378 offset = task->actual_size;
379 task->actual_size += size;
380#endif
381 return offset;
382}

384void
385llmp_task_add(int scope_sptr, int shared_sptr, SPTR private_sptr)
386{
387 LLTask *task;
388 assert(scope_sptr && STYPEG(scope_sptr) == ST_BLOCK,
389 "Task key must be a scope sptr (ST_BLOCK)", scope_sptr, ERR_Fatal);
390
391 task = llmp_get_task(scope_sptr);
392 if (!task)
393 task = llmp_create_task(scope_sptr);
394 llmp_task_add_private(task, shared_sptr, private_sptr);
