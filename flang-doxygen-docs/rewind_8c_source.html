18static int
19_f90io_rewind(__INT_T *unit, __INT_T *bitv, __INT_T *iostat)
20{
21 FIO_FCB *f;
22
23 __fortio_errinit03(*unit, *bitv, iostat, "REWIND");
24 if (ILLEGAL_UNIT(*unit)) /* check for illegal unit number */
25 return __fortio_error(FIO_EUNIT);
26
27 f = __fortio_find_unit(*unit);
28
29 if (f) {
30 if (f->acc == FIO_DIRECT) /* can't rewind direct access file */
31 /* treat rewind of direct acc. file as no-op to avoid complaints */
32 return 0 /*__fortio_error(FIO_EDIRECT)*/;
33
34 /* check for outstanding async i/o */
35
36 if (f->asy_rw) { /* stop any async i/o */
37 f->asy_rw = 0;
38 if (Fio_asy_disable(f->asyptr) == -1) {
39 return (__fortio_error(__io_errno()));
40 }
41 }
42
43 /* append carriage return (maybe) */
44
45 if (f->nonadvance) {
46 f->nonadvance = FALSE;
47#if defined(_WIN64)
48 if (__io_binary_mode(f->fp))
49 __io_fputc('\r', f->fp);
50#endif
51 __io_fputc('\n', f->fp);
52 if (__io_ferror(f->fp))
53 return __io_errno();
54 }
55
56 if (__io_fseek(f->fp, 0L, SEEK_SET) != 0)
57 return __fortio_error(__io_errno());
58
59 f->nextrec = 1;
60 f->coherent = 0;
61 f->eof_flag = FALSE;
62 f->truncflag = TRUE;
63 f->skip = 0;
64 }
65
66 return 0; /* no error occurred */
67}
