66/*
67 * Configure "denormal inputs are zero" and "flush denormal results to zero"
68 * modes.
69 *
70 * TODO: Find a more portable way to configure these settings.
71 */
72
73static void
74configure_denormals(bool denorms_are_zeros, bool flush_to_zero)
75{
76 fenv_t fenv;
77 if (fegetenv(&fenv) != 0)
78 fprintf(stderr, "fegetenv() failed: %s\n", strerror(errno));
79#ifdef __x86_64__
80#ifdef _WIN64
81 unsigned int mxcsr = _mm_getcsr();
82#else
83 unsigned int mxcsr = fenv.__mxcsr;
84#endif
85 mxcsr &= ~0x0040;
86 if (denorms_are_zeros)
87 mxcsr |= 0x0040;
88 mxcsr &= ~0x8000;
89 if (flush_to_zero)
90 mxcsr |= 0x8000;
91#ifdef _WIN64
92 _mm_setcsr( mxcsr );
93#else
94 fenv.__mxcsr = mxcsr;
95#endif
96#endif
97#ifndef _WIN64
98 if (fesetenv(&fenv) != 0)
99 fprintf(stderr, "fesetenv() failed: %s\n", strerror(errno));
100#endif
101}

103/*
104 * Comparisons. These can't trap.
105 */
106
107#define COMPARE_BODY { \
108 enum fold_relation rel = FOLD_UN; \
109 fenv_t fenv; \
110 if (feholdexcept(&fenv) != 0) \
111 fprintf(stderr, "feholdexcept() failed: %s\n", strerror(errno)); \
112 configure_denormals(false, false); \
113 if (*x < *y) \
114 rel = FOLD_LT; \
115 else if (*x == *y) \
116 rel = FOLD_EQ; \
117 else if (*x > *y) \
118 rel = FOLD_GT; \
119 if (fesetenv(&fenv) != 0) \
120 fprintf(stderr, "fesetenv() failed: %s\n", strerror(errno)); \
121 return rel; \
122}

142/*
143 * Set up the floating-point environment so that exceptional conditions
144 * are recorded but don't raise traps.
145 *
146 * TODO: Consider setting the daz and flushz modes by the compiler to
147 * match the expected run-time environment (-Mdaz, -Mflushz). Those
148 * compiler options are documented as "link-time" but maybe it would
149 * be a good idea to respect them while folding.
150 */
151static void
152set_up_floating_point_environment(fenv_t *fenv)
153{
154 if (feholdexcept(fenv) != 0)
155 fprintf(stderr, "feholdexcept() failed: %s\n", strerror(errno));
156 configure_denormals(false, false);
157 errno = 0;
158}

160/*
161 * Map floating-point exceptional conditions to a folding status code.
162 */
163
164static enum fold_status
165interpret_exceptions(int exceptions, int errno_capture)
166{
167 if (exceptions & FE_INVALID)
168 return FOLD_INVALID;
169 if (exceptions & (FE_DIVBYZERO | FE_OVERFLOW))
170 return FOLD_OVERFLOW;
171 if (exceptions & FE_UNDERFLOW)
172 return FOLD_UNDERFLOW;
173 if (exceptions & FE_INEXACT)
174 return FOLD_INEXACT;
175 /* ignore any non-standard extended flags */
176 if (errno_capture == EDOM)
177 return FOLD_INVALID;
178 if (errno_capture == ERANGE)
179 return FOLD_OVERFLOW; /* can't distinguish over/underflow from errno */
180 return FOLD_OK;
181}

183/*
184 * Common exit processing: restore the processor's floating-point
185 * environment and translate any exceptions that may have arisen.
186 */
187
188static enum fold_status
189check_and_restore_floating_point_environment(const fenv_t *saved_fenv)
190{
191 int errno_capture = errno;
192 int exceptions = fetestexcept(FE_ALL_EXCEPT);
193 if (fesetenv(saved_fenv) != 0)
194 fprintf(stderr, "fesetenv() failed: %s\n", strerror(errno));
195 return interpret_exceptions(exceptions, errno_capture);
196}
