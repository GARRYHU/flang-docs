25/* For OpenMP target accelerator,
26 * the compilers outlines the same region multiple times.
27 * Therefore outliner has following states. */
28typedef enum
29{
30 outliner_not_active = 0, /* Not outlining, temp files are empty */
31 outliner_active_host_par1 = 1, /* Host outlining, building ILI from parfile1 ilm */
32 outliner_active_host_par2 = 2, /* Host outlining, building ILI from parfile2 ilm */
33 outliner_active_switchfile = 3, /* Outlining recurs */
34 outliner_reset = 4, /* Reset files, go back to main ilm file. */
35 outliner_error = 5
36} outliner_states_t;
