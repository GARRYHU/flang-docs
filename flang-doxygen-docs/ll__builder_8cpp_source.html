26LLMD_Builder
27llmd_init(LL_Module *module)
28{
29 LLMD_Builder mdb = (LLMD_Builder)calloc(1, sizeof(struct LLMD_Builder_));
30 mdb->module = module;
31 /* Start out with init_elems to save a malloc() call. */
32 mdb->elems = mdb->init_elems;
33 mdb->capacity = sizeof(mdb->init_elems) / sizeof(mdb->init_elems[0]);
34 return mdb;
35}
36
37/* Double the capacity in mdb. */
38static void
39grow(LLMD_Builder mdb)
40{
41 mdb->capacity *= 2;
42 if (mdb->elems == mdb->init_elems) {
43 /* First time we grow, don't free init_elems. */
44 mdb->elems = (LL_MDRef *)malloc(mdb->capacity * sizeof(mdb->elems[0]));
45 memcpy(mdb->elems, mdb->init_elems, sizeof(mdb->init_elems));
46 } else {
47 /* We've grown before. Just realloc. */
48 mdb->elems =
49 (LL_MDRef *)realloc(mdb->elems, mdb->capacity * sizeof(mdb->elems[0]));
50 }
51}

53void
54llmd_add_md(LLMD_Builder mdb, LL_MDRef mdnode)
55{
56 if (mdb->nelems >= mdb->capacity)
57 grow(mdb);
58 mdb->elems[mdb->nelems++] = mdnode;
59}

114void
115llmd_reverse(LLMD_Builder mdb)
116{
117 unsigned i, j;
118 LL_MDRef *e = mdb->elems;
119
120 if (mdb->nelems <= 1)
121 return;
122
123 for (i = 0, j = mdb->nelems - 1; i < j; i++, j--) {
124 LL_MDRef t = e[i];
125 e[i] = e[j];
126 e[j] = t;
127 }
128}

148LL_MDRef
149ll_finish_variable(LLMD_Builder mdb, LL_MDRef fwd)
150{
151 LL_MDRef mdref;
152 LLVMModuleRef mod = mdb->module;
153 const unsigned mdnodeTop = mod->mdnodes_count;
154
155 /* create the LL_MDRef */
156 if (mdb->is_distinct)
157 mdref =
158 ll_create_distinct_md_node(mod, mdb->mdclass, mdb->elems, mdb->nelems);
159 else
160 mdref = ll_get_md_node(mod, mdb->mdclass, mdb->elems, mdb->nelems);
161
162 /* and move it as needed */
163 if (!LL_MDREF_IS_NULL(fwd)) {
164 hash_data_t data;
165 const unsigned slot = LL_MDREF_value(fwd);
166 const unsigned mdrefSlot = LL_MDREF_value(mdref) - 1;
167 LL_MDNode *newNode = mod->mdnodes[mdrefSlot];
168 ll_set_md_node(mod, slot, newNode);
169 if (mod->mdnodes_count == mdnodeTop + 1)
170 --mod->mdnodes_count;
171 mdref = fwd;
172 data = (hash_data_t)INT2HKEY(slot);
173 hashmap_replace(mod->mdnodes_map, newNode, &data);
174 }
175
176 if (mdb->elems != mdb->init_elems)
177 free(mdb->elems);
178 free(mdb);
179
180 return mdref;
181}

183LL_MDRef
184llmd_finish(LLMD_Builder mdb)
185{
186 LL_MDRef mdref;
187
188 if (mdb->is_distinct)
189 mdref = ll_create_distinct_md_node(mdb->module, mdb->mdclass, mdb->elems,
190 mdb->nelems);
191 else
192 mdref = ll_get_md_node(mdb->module, mdb->mdclass, mdb->elems, mdb->nelems);
193
194 if (mdb->elems != mdb->init_elems)
195 free(mdb->elems);
196 free(mdb);
197
198 return mdref;
