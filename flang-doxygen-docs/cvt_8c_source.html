89 * Do integer part

93 while (fi != 0) {

95 *--p1 = (int)((fj + .03) * 10) + '0';
96 r2++;
97 }
98 while (p1 < &buf[NDIG])
99 *p++ = *p1++;
100 } else if (arg > 0) {
101 while ((fj = arg * 10) < 1) {
102 arg = fj;
103 r2--;
104 }
105 }
106 p1 = &buf[ndigits];
107 if (eflag == 0)
108 p1 += r2;
109 *decpt = r2;
110 if (p1 < &buf[0]) {
111 buf[0] = '\0';
112 return (buf);
113 }
114 while (p <= p1 && p < &buf[NDIG]) {
115 arg *= 10;
116 arg = modf(arg, &fj);
117 *p++ = (int)fj + '0';
118 }
119 if (p1 >= &buf[NDIG]) {
120 buf[NDIG - 1] = '\0';
121 return (buf);

136 }

138 }
