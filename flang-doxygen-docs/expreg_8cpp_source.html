29/** \brief Add register candidate
30 *
31 * routine to add candidates to the register candidate lists of
32 * the register module. This routine is valid for opt 1 only.
33 * This routine is called only for load and store ILI and will
34 * eliminate a candidate which depend on their types, storage
35 * class, etc.
36 */
37void
38exp_rcand(int ilix, int nmex)
39{
40 register int sym;
41
42 if (NME_TYPE(nmex) == NT_VAR /* var names only */
43 && TY_ISSCALAR(DTY(DTYPEG(sym = NME_SYM(nmex)))) /* scalars only */
44 && IS_LCL_OR_DUM(sym) /* autos, args or regs */
45 && !VOLG(sym) /* not volatile */
46 && SOCPTRG(sym) == 0 /* not equivalenced */
47 ) {
48 addrcand(ilix);
49 /* If a variable is loaded or stored in a critical section or a
50 * parallel section, set the candidate's IGNORE flag. This prevents
51 * a register from being assigned.
52 * Tprs 2077 & 2127 - can't assign registers to variables in a
53 * parallel region (could restrict this to shared variables).
54 */
55 if (NME_RAT(nmex) &&
56 (bihb.csfg || bihb.parsectfg || bihb.parfg || bihb.taskfg))
57 RCAND_IGNORE(NME_RAT(nmex)) = 1;
58 }
59}

63/** \brief Control the assignment of the registers at opt level 1
64 */
65void
66reg_assign1(void)
67{
68 int first_rat, /* first entry of the block's RAT */
69 entr; /* entry symbol of the function */
70 int null_rat;
71 int funcbih;
72 int iltt; /* local tmp for ilt */
73 int sym;
74 LST *save;
75 int tbih;
76
77 funcbih = BIHNUMG(gbl.currsub);
78 rdilts(funcbih); /* fetch the entry block */
79 entr = BIH_LABEL(funcbih); /* get the entry symbol */
80 list = 0;
81 ilt = ILT_PREV(0); /* locate the last ilt in the block */
82
83#if DEBUG
84 if (EXPDBG(8, 12))
85 fprintf(gbl.dbgfil,
86 "\n***** Register Information for Function \"%s\" *****\n",
87 getprint(entr));
88 if (EXPDBG(8, 4))
89 dmprcand();
90#endif
91
92 GET_RAT(first_rat); /* fetch a RAT item which will be
93 * the header of the table
94 */
95 /*
96 * assign the data registers from the allowed register set.
97 */
98 assign1(RATA_IR);
99 /*
100 * assign the address registers from the allowed set.
101 */
102 assign1(RATA_AR);
103 /*
104 * assign the single precision registers from the allowed set.
105 */
106 assign1(RATA_SP);
107 /*
108 * assign the double precision registers from the allowed set.
109 */
110 assign1(RATA_DP);
111 /*
112 * assign the double integer registers from the allowed set.
113 */
114 assign1(RATA_KR);
115 /*
116 * communicate to the scheduler the first global register assigned
117 * for each register class -- note that this will be the physical register
118 * number; it reflects the number of registers assigned from the physical
119 * set mapped from the generic register set. Because two or more generic
120 * register sets can map to a single register set, this information
121 * can only be computed after all of the assignments are done.
122 *
123 */
124 mr_end();
125 /*
126 * fill in the VAL field of the header entry for this RAT with
127 * the number of assignments.
128 */
129 RAT_VAL(first_rat) = ratb.stg_avail - first_rat - 1;
130
131#if DEBUG
132 if (EXPDBG(8, 8)) {
133 dmprat(first_rat);
134 fprintf(gbl.dbgfil,
135 " first_dr:%3d, first_ar:%3d, first_sp:%3d, first_dp:%3d\n",
136 aux.curr_entry->first_dr, aux.curr_entry->first_ar,
137 aux.curr_entry->first_sp, aux.curr_entry->first_dp);
138 }
139#endif
140/*
141 * link the BIH of the entry block to the RAT. this block is written
142 * out.
143 */
144 BIH_ASSN(BIH_NEXT(funcbih)) = first_rat;
145 wrilts(funcbih);
146 /*
147 * add ilis that were added to the main entry to any other entries
148 */
149 if (list) {
150 save = list;
151 for (sym = SYMLKG(gbl.entries); sym != NOSYM; sym = SYMLKG(sym)) {
152 rdilts(tbih = BIHNUMG(sym));
153 GET_RAT(null_rat);
154 RAT_VAL(null_rat) = 0;
155 BIH_ASSN(tbih) = null_rat;
156 /*
157 * if we copied params, we do this mvxx stuff in block after the
158 * entry block, so scheduler loads up dummies after copy to temp
159 * param list has been done
160 */
161 if (COPYPRMSG(sym)) {
162 wrilts(tbih);
163 tbih = exp_addbih(tbih);
164 rdilts(tbih);
165 BIH_ASSN(tbih) = null_rat;
166 }
167 iltt = ILT_PREV(0);
168 list = save;
169 while (list) {
170 iltt = addilt(iltt, list->item);
171 list = list->next;
172 }
173 BIH_ASSN(BIH_NEXT(tbih)) = first_rat;
174 wrilts(tbih);
175 }
176 freearea(LST_AREA);
177 }
178 /*
179 * dummies that have stores into and that are assigned registers , must
180 * be stored back at exit
181 */
182 storedums(expb.curbih, first_rat);
183
184 /* for the routine's exit block, assign a null rat table */
185 GET_RAT(null_rat);
186 RAT_VAL(null_rat) = 0;
187 BIH_ASSN(expb.curbih) = null_rat;
188
189 /*
190 * go through the candidate lists to clean up the back pointers.
191 * also, reset the register candidate store area.
192 */
193 endrcand();
194
195}

197/** \brief Assign registers for opt level 1.
198 *
199 * The possible set of registers to use satisfies the relation first_global <=
200 * reg <= last_global, where the bounds are found in the mach_reg structure of
201 * the reg structure for the given register type. The registers are assigned
202 * beginning at last_global and continues down to first_global. The value
203 * returned by the function is the number of registers assigned.
204 *
205 * \param rtype type of registers
206 */
207static void
208assign1(int rtype)
209{
210 int cand, /* candidate for a register */
211 rat, /* RAT for each assigned reg */
212 ilix, /* ili for the candidate */
213 areg; /* assigned register */
214
215 int candl, /* candidate list */
216 nme, /* NME for the candidate */
217 sym, /* ST for the candidate */
218 addr, /* address ili for the candidate */
219 ilitmp; /* ili temporary */
220
221 /*
222 * loop while there are candidates and registers available
223 */
224 candl = reg[rtype].rcand;
225 for (;;) {
226 /*
227 * get a candidate; if one does not exist, exit the loop
228 */
229 if ((cand = getrcand(candl)) == 0)
230 break;
231 if (RCAND_IGNORE(cand))
232 continue;
233 if (!RCAND_ISILI(cand)) {
234/*
235 * if the candidate is not type ILI (implies that just stores
236 * occurred, get the next candidate if it's not a regarg.
237 * NOTE: C only.
238 */
239 continue;
240 } else {
241 /*
242 * for the candidate, locate the load ili (ilix), the names entry
243 * (nme), and the symbol (sym)
244 */
245 ilix = RCAND_VAL(cand);
246 sym = NME_SYM(nme = ILI_OPND(ilix, 2));
247 addr = ILI_OPND(ilix, 1);
248 }
249 /*
250 * if the symbol has had its address taken, it cannot be assigned a
251 * register
252 */
253 if (ADDRTKNG(sym))
254 continue;
255 if (gbl.internal > 1 && sym == aux.curr_entry->display) {
256 /* printf("display temp %s cannot be assigned register\n",
257 SYMNAME(sym));
258 */
259 continue;
260 }
261/*
262 * if symbol is declared in an 'outer' scope, it cannot be assigned a
263 * register - happens in a language with nested procedures (c++)
264 */
265 if (GSCOPEG(sym))
266 continue;
267 if (UPLEVELG(sym))
268 continue;
269
270 /*
271 *for now, dissallow dummies if there are multiple entries
272 */
273 if (IS_DUM(sym)) {
274 if (SYMLKG(gbl.entries) != NOSYM)
275 continue;
276 } else if (SCG(sym) == SC_LOCAL && DINITG(sym) &&
277 (ADDRTKNG(sym) || RCAND_STORE(cand)) && gbl.rutype != RU_PROG)
278 /*
279 * if a local variable is data initialized, disallow it if it
280 * has been stored; the thinking is that it falls into the
281 * same category of a saved variable -- someday, may want
282 * to override this if XBIT(124,0x80) is set (also optutil.c)
283 */
284 continue;
285 /*
286 * for C regargs, use arg register saved in address field;
287 * for ftn use arg register saved in address field if it's the
288 * symbol which represents the arg's address. Otherwise, get a
289 * register from the "global" set.
290 */
291 if (IS_REGARG(sym))
292 areg = ADDRESSG(sym);
293 else {
294 /*
295 * get a register from the global set
296 */
297 if ((areg = mr_getreg(rtype)) == NO_REG)
298 break;
299 }
300
301 GET_RAT(rat) /* locate a rat entry */
302
303 RAT_REG(rat) = ad1ili(RTYPE_DF(rtype), areg);
304
305 RAT_RTYPE(rat) = rtype; /* don't forget its rtype */
306 RAT_ATYPE(rat) = RATA_NME; /* opt 1 assigns nme's only */
307 RAT_ADDR(rat) = addr; /* locate addr ili of ref */
308 RAT_CONFL(rat) = 0; /* no conflict for opt 1 */
309 RAT_STORE(rat) = RCAND_STORE(cand); /* copy up store flag */
310 RAT_VAL(rat) = nme; /* record nme assigned */
311 RAT_MSIZE(rat) = RCAND_MSIZE(cand); /* copy up mem size */
312 /*
313 * if the symbol is an argument to the function, add a move ili of
314 * the argument to its assigned register
315 */
316 /* for fortran, a REGARG sym has its address in the register, not
317 * the value of the argument. REDUC flag indicates that no extra
318 * indirection is needed.
319 */
320 if (IS_DUM(sym)) {
321 if (REGARGG(sym)) {
322 if (!REDUCG(sym)) {
323 /* we're preloading the dummy argument; we need to use
324 * the arg register that's assigned to its address as the
325 * address expression in the the load.
326 */
327 addr = ad1ili(IL_ARDF, ADDRESSG(sym));
328 if (flg.endian &&
329 (DTYPEG(sym) == DT_INT8 || DTYPEG(sym) == DT_LOG8)) {
330 if (!XBIT(124, 0x400))
331 /* 32bits of significance in 64 bits */
332 addr = ad3ili(IL_AADD, addr, ad_aconi(4), 0);
333 }
334 RAT_ADDR(rat) = addr; /* switch addr expr */
335 ilix = ad3ili(ILI_OPC(ilix), addr, ILI_OPND(ilix, 2),
336 ILI_OPND(ilix, 3));
337 ilt = addilt(ilt, ilitmp = ad2ili(MV_RTYPE(rtype), ilix, areg));
338 ADDNODE(list, ilitmp);
339 }
340 } else {
341 ilt = addilt(ilt, ilitmp = ad2ili(MV_RTYPE(rtype), ilix, areg));
342 ADDNODE(list, ilitmp);
343 }
344 } else {
345 /*
346 * FORTRAN only - if the symbol is local to the function and has
347 * been data initialized, add a move ili of the argument to its
348 * assigned register
349 */
350 if (DINITG(sym)) {
351 ilt = addilt(ilt, ilitmp = ad2ili(MV_RTYPE(rtype), ilix, areg));
352 ADDNODE(list, ilitmp);
353 }
354 }
355 }
