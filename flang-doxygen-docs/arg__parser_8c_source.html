81{
82 *parser = (arg_parser_t*) malloc(sizeof(arg_parser_t));
83 (*parser)->flags = hashmap_alloc(hash_functions_strings);

90/** Deallocate argument parser hash map */
91void
92destroy_arg_parser(arg_parser_t **parser)
93{
94 /* Deallocate entries */
95 hashmap_iterate((*parser)->flags, deallocate_arg_value, NULL);
96
97 /* Free flags data structure */
98 hashmap_free((*parser)->flags);
99
100 /* Deallocate the data structure itself */
101 free(*parser);
102 *parser = NULL;
103}

105/** Deallocate hashtable entry */
106static void
107deallocate_arg_value(hash_key_t key, hash_data_t value_ptr, void *key_context)
108{
109#pragma GCC diagnostic push
110#pragma GCC diagnostic warning "-Wcast-qual"
111 value_t *p = (value_t *)value_ptr;
112 char *k = (char *)key;
113#pragma GCC diagnostic pop
114 /* Some of the argument types require deallocation */
115 switch (p->type) {
116 case ARG_ActionMap:
117 case ARG_CombinedBoolean:
118 free(p->location);
119 break;
120 case ARG_ReverseBoolean:
121 free(k);
122 break;
123 default:
124 /* Do nothing */
125 break;
126 }
127 free(p);
128}

264/** \brief Add a generic argument, specifying argument type
265 *
266 * \param arg_parser Parser data structure
267 * \param arg_name Argument name (key to flags table)
268 * \param value_type Type of the value to store
269 * \param value_ptr Location to store value to
270 */
271static void
272add_generic_argument(arg_parser_t *parser, const char *arg_name,
273 value_type_t value_type, void *value_ptr)
274{
275 /* Mapped value */
276 value_t *value = NULL;
277 /* Old (ignored) value */
278 hash_data_t old_value = NULL;
279
280 /* Check if this argument is already registered */
281 if (hashmap_lookup(parser->flags, arg_name, &old_value)) {
282 /* Compose & throw error */
283 compose_and_throw("Argument already registered: ", arg_name);
284 }
285
286 /* Add value to flags hashmap */
287 value = (value_t*) malloc(sizeof(value_t));
288 value->type = value_type;
289 value->location = value_ptr;
290 hashmap_insert(parser->flags, arg_name, value);
291
292 /* Put the same value into hashset tracking registered values (this time it is
293 * the key) */
294 if (value_type == ARG_ActionMap) {
295 action_map_t *target = ((action_map_bundle_t *)value_ptr)->output;
296 hashset_replace(parser->values, target);
297 } else {
298 hashset_replace(parser->values, value_ptr);
299 }
300}

302/** Parse all arguments */
303void
304parse_arguments(const arg_parser_t *parser, int argc, char **argv)
305{
306 int argindex = 1; /* Argument counter */
307 char *next_string; /* Next argument */
308 int x_index, x_value; /* Index to xflags array and value to set there */
309
310 /* Reset value hits */
311 hashset_clear(parser->value_hits);
312
313 /* Make sure we can set file name */
314 if (!parser->input_file_name_ptr) {
315 interr("Input file name is not registered", 0, ERR_Fatal);
316 }
317
318 /* First grab the source file name */
319 if (*argv[1] != '-') {
320 *parser->input_file_name_ptr = argv[1];
321 argindex = 2;
322 } else if (*argv[argc - 1] != '-') {
323 *parser->input_file_name_ptr = argv[argc - 1];
324 --argc;
325 }
326
327 /* Loop through provided arguments */
328 while (argindex < argc) {
329 value_t *value = NULL;
330 char *arg = argv[argindex];
331
332 /* All switches should start with a '-' */
333 if (*arg != '-') {
334 compose_and_throw("Unrecognized argument: ", arg);
335 }
336 ++arg; /* skip over '-' */
337
338#pragma GCC diagnostic push
339#pragma GCC diagnostic warning "-Wcast-qual"
340 /* All arguments need to be in the data structure */
341 if (!hashmap_lookup(parser->flags, arg, (hash_data_t *)&value)) {
342 if (parser->fail_on_unknown_args) {
343 compose_and_throw("Unknown command line argument: ", arg);
344 } else {
345 /* Skip until next switch or end of argument list */
346 while ((argindex < argc - 1) && (*argv[++argindex] != '-'))
347 ;
348 if (argindex == argc - 1)
349 break;
350 continue;
351 }
352 }
353#pragma GCC diagnostic pop
354
355 /* Parse argument type */
356 switch (value->type) {
357 case ARG_ActionMap: {
358 const action_map_t *from = ((action_map_bundle_t *)value->location)->input;
359 action_map_t *to = ((action_map_bundle_t *)value->location)->output;
360
361 /* TODO parse lists of arguments */
362 char *phase_string = next_value(argv, &argindex);
363 char *dump_string = next_value(argv, &argindex);
364
365 copy_action(from, dump_string, to, phase_string);
366 } break;
367
368 case ARG_Boolean:
369 *((bool *)value->location) = true;
370 break;
371
372 case ARG_ReverseBoolean:
373 *((bool *)value->location) = false;
374 break;
375
376 case ARG_String:
377 next_string = next_value(argv, &argindex);
378 if (!next_string)
379 compose_and_throw("Missing value for -", arg);
380 /* Change stored value to point to passed string */
381 *((char **)value->location) = next_string;
382 break;
383
384 case ARG_StringList:
385 /* Get argument value */
386 next_string = next_value(argv, &argindex);
387 if (!next_string)
388 compose_and_throw("Missing value for -", arg);
389 /* Set argument value */
390 *((char **)value->location) = next_string;
391 /* Point to the next value on the list */
392 value->location = (void *)(((char **)value->location) + 1);
393 /* Terminate the list */
394 *((char **)value->location) = NULL;
395 break;
396
397 case ARG_InformLevel:
398 next_string = next_value(argv, &argindex);
399 if (!next_string)
400 compose_and_throw("Missing value for -", arg);
401 *((inform_level_t *)value->location) = strtoinform(next_string);
402 break;
403
404 case ARG_Integer:
405 next_string = next_value(argv, &argindex);
406 if (!next_string)
407 compose_and_throw("Missing value for -", arg);
408 *((int *)value->location) = (int)strtol(next_string, NULL, 10);
409 break;
410
411 case ARG_QFlag:
412 next_string = next_value(argv, &argindex);
413
414 if (!next_string)
415 compose_and_throw("Missing value for -", arg);
416
417 x_index = (int)strtol(next_string, NULL, 10);
418
419 next_string = next_value(argv, &argindex);
420
421 if (!next_string)
422 compose_and_throw("Missing second value for -", arg);
423
424 x_value = (int)strtol(next_string, NULL, 10);
425
426 ((int *)value->location)[x_index] |= x_value;
427 break;
428
429 case ARG_XFlag:
430 next_string = next_value(argv, &argindex);
431
432 if (!next_string)
433 compose_and_throw("Missing value for -", arg);
434 x_index = (int)strtol(next_string, NULL, 10);
435
436 next_string = next_value(argv, &argindex);
437
438 if (next_string)
439 x_value = (int)strtol(next_string, NULL, 0);
440 else
441 x_value = 1;
442
443 set_xflag_value((int *)value->location, x_index, x_value);
444 break;
445
446 case ARG_YFlag:
447 next_string = next_value(argv, &argindex);
448
449 if (!next_string)
450 compose_and_throw("Missing value for -", arg);
451 x_index = (int)strtol(next_string, NULL, 10);
452
453 next_string = next_value(argv, &argindex);
454
455 if (next_string)
456 x_value = (int)strtol(next_string, NULL, 0);
457 else
458 x_value = 1;
459
460 unset_xflag_value((int *)value->location, x_index, x_value);
461 break;
462
463 case ARG_CombinedBoolean:
464 *(((bool_string_t *)value->location)->bool_ptr) = true;
465 next_string = next_value(argv, &argindex);
466 if (next_string) {
467 *(((bool_string_t *)value->location)->string_ptr) = next_string;
468 }
469 break;
470 }
471
472 /* Remember that the value as set */
473 if (value->type == ARG_ActionMap) {
474 action_map_t *target = ((action_map_bundle_t *)value->location)->output;
475 hashset_replace(parser->value_hits, target);
476 } else {
477 hashset_replace(parser->value_hits, value->location);
478 }
479
480 ++argindex;
481 }
482}

484/** \brief Compose and throw internal compiler error
485 *
486 * Produces internal compiler error with the message composed by concatenating
487 * the two operands
488 */
489static void
490compose_and_throw(const char *first_part, const char *second_part)
491{
492 char *msg = (char*) malloc(strlen(first_part) + strlen(second_part) + 1);
493 strcpy(msg, first_part);
494 strcat(msg, second_part);
495 interr(msg, 0, ERR_Fatal);
496}

498/** \brief Point to next argument value
499 *
500 * If next argument in argument list does not start with a '-' return it and
501 * advance argument counter, otherwise return NULL and keep counter the same
502 *
503 * \return pointer to the next non-switch command line argument
504 * \param argv command line argument array
505 * \param arg_index current index into the argument array
506 */
507static char *
508next_value(char **argv, int *arg_index)
509{
510 if (*argv[*arg_index + 1] == '-')
511 return NULL;
512
513 *arg_index = *arg_index + 1;
514
515 return argv[*arg_index];
516}

518/** \brief Convert a inform level string to corresponding constant */
519static inform_level_t
520strtoinform(const char *string)
521{
522 if (!strcmp("inform", string))
523 return LV_Inform;
524 if (!strcmp("severe", string))
525 return LV_Severe;
526 if (!strcmp("warn", string))
527 return LV_Warn;
528 if (!strcmp("fatal", string))
529 return LV_Fatal;
530
531 compose_and_throw("Unrecognized inform level: ", string);
532
533 /* unreacheable */
534 return LV_Inform;
535}
