24void
25list_init(FILE *fd)
26{
27 register char **p;
28 char buf[80], buf1[40], buf2[20], buf3[20];
29 static const char *sevtxt[4] = {"inform", "warn", "severe", "fatal"};
30
31 lf = fd;
32 pgno = 1;
33
34 /* WARNING: watch for overflowing buf */
35 if (!DBGBIT(14, 3)) {
36
37 /* ... put out switch line. */
38 sprintf(
39 buf, "Switches: -%s -%s -%s -%s -%s -%s",
40 (flg.asmcode ? "asm" : "noasm"), (flg.dclchk ? "dclchk" : "nodclchk"),
41 (flg.debug ? "debug" : "nodebug"), (flg.dlines ? "dlines" : "nodlines"),
42 (flg.line ? "line" : "noline"), (flg.list ? "list" : "nolist"));
43 list_line(buf);
44
45 /* -idir lines: */
46 for (p = flg.idir; p && *p; p++)
47 list_ln(" -idir ", *p);
48
49 sprintf(buf, " -inform %s -opt %d -%s -%s -%s", sevtxt[flg.inform],
50 flg.opt, (flg.save ? "save" : "nosave"),
51 (flg.object ? "object" : "noobject"),
52 (flg.onetrip ? "onetrip" : "noonetrip"));
53 list_line(buf);
54
55 buf2[0] = buf3[0] = 0;
56 if (flg.depchk) {
57 if (flg.depwarn)
58 strcpy(buf2, "-depchk warn");
59 else
60 strcpy(buf2, "-depchk on");
61 } else
62 strcpy(buf2, "-depchk off");
63 sprintf(buf, " %s -%s %s %s", buf2,
64 (flg.standard ? "standard" : "nostandard"),
65 (flg.extend_source == 132 ? "-extend" : " "),
66 (flg.profile ? "-profile" : " "));
67 list_line(buf);
68
69 strcpy(buf1, "-show");
70 if (flg.include && flg.xref && flg.code)
71 strcat(buf1, " all");
72 else {
73 if (flg.include)
74 strcat(buf1, " include");
75 if (flg.xref)
76 strcat(buf1, " xref");
77 if (flg.code)
78 strcat(buf1, " code");
79 }
80 if (!strcmp(buf1, "-show"))
81 strcpy(buf1, " ");
82 buf2[0] = buf3[0] = 0;
83 sprintf(buf, " -%s -%s %s %s %s",
84 (flg.symbol ? "symbol" : "nosymbol"),
85 (flg.ucase ? "upcase" : "noupcase"), buf1, buf2, buf3);
86 list_line(buf);
87
88 /* ... put out filename line. */
89 list_ln("\nFilename: ", gbl.src_file);
90 }
91
92 list_line("");
93
94}

102static void
103list_ln(const char *beg, const char *txt)
104{
105 if (pgpos == 1 && !DBGBIT(14, 3)) {
106 if (!lf)
107 return; /* in case error msg written before file
108 * opened */
109 fprintf(lf, "\n\n\n%s (Version %8s) %s page %d\n\n",
110 version.lang, version.vsn, gbl.datetime, pgno);
111 pgno++;
112 pgpos = 6;
113 }
114
115 fprintf(lf, "%s%s\n", beg, txt);
116#if DEBUG
117 if (DBGBIT(0, 4))
118 fprintf(gbl.dbgfil, "%s%s\n", beg, txt);
119#endif
120 pgpos++;
121
122 if (pgpos == LPP + 4 && !DBGBIT(14, 3)) {
123 fprintf(lf, "\n\n\n");
124 pgpos = 1;
125 }
126
127}
