85{
86 /*
87 * Allocate the space necessary to hold the auxiliary information for ILM
88 * evaluation required by the expander. If necessary the space could
89 * depend on sem.ilm_size, but this is probably too much. The ilm index

98 expb.str_avail = 0;

108 expb.curbih = 0;
109 expb.isguarded = 0;
110 expb.flags.bits.noblock = 1;
111 expb.flags.bits.noheader = 1;
112 if (CHARLEN_64BIT)
113 expb.charlen_dtype = DT_INT8;
114 else
115 expb.charlen_dtype = DT_INT;
116

118 expb.flags.bits.excstat = 1;

159/*
160 * Parse an IM_FILE ilm.
161 *
162 * - ilmp is an IM_FILE ilm.
163 * - lineno_out becomes the line number, but only if the IM_FILE has a non-zero
164 * lineno operand. Otherwise, lineno_out is not touched.
165 * - findex_out becomes a valid index into the FIH table.
166 * - ftag_out becomes the ftag.
167 */
168static void
169parse_im_file(const ILM *ilmp, int *lineno_out, int *findex_out, int *ftag_out)
170{
171 /* IM_FILE lineno findex ftag */
172 int lineno = ILM_OPND(ilmp, 1);
173 int findex = ILM_OPND(ilmp, 2);
174 int ftag = ILM_OPND(ilmp, 3);
175
176 assert(ILM_OPC(ilmp) == IM_FILE, "parse_im_file: Expected IM_FILE",
177 ILM_OPC(ilmp), ERR_Fatal);
178
179 /* The bottom-up inliner will generate some IM_FILE ilms with findex
180 * operands that reference the IFIH table. These references are encoded as
181 * negative numbers. Translate them back to FIH references here. */
182 if (findex < 0) {
183 int ifindex = -findex - 1;
184 assert(ifindex < ifihb.stg_avail,
185 "parse_im_file: Invalid IFIH reference on IM_FILE", ifindex,
186 ERR_Warning);
187 findex = IFIH_FINDEX(ifindex);
188 }
189
190 assert(findex < fihb.stg_avail,
191 "parse_im_file: Invalid FIH reference on IM_FILE", findex,
192 ERR_Warning);
193
194 if (lineno_out && lineno)
195 *lineno_out = lineno;
196 if (findex_out)
197 *findex_out = findex;
198 if (ftag_out)
199 *ftag_out = ftag;
200}

204/** \brief Expand ILMs to ILIs */
205int
206expand(void)
207{
208 int ilmx; /* index of the ILM */
209 int len; /* length of the ILM */
210 ILM *ilmp; /* absolute pointer to the ILM */
211 ILM_OP opc; /* opcode of the ILM */
212 int last_label_bih = 0;
213 int last_ftag = 0;
214 int nextftag = 0, nextfindex = 0;
215 int last_cpp_branch = 0;
216
217 /*
218 * NOTE, for an ILM: ilmx is needed to access the ILM_AUX area, ilmp is
219 * needed to access the ILM area
220 */
221 exp_init();
222 /* During expand, we want to generate unique proc ili each time a
223 * proc ILM is processed. The assumption is that the scheduler will
224 * cse a proc ili if it appears multiple times in a block. E.g.,
225 * COMPLEX c(10)
226 * x = f() + f() ! two ili for calling f
227 * c(ifunc()) = ... ! 1 call to ifunc (although two uses)
228 * After expand, we share proc ili; the optimizer may create expressions
229 * which contain calls where the intent is to cse a call if it already
230 * exists in the block.
231 */
232 share_proc_ili = false;
233
234 if (!XBIT(120, 0x4000000)) {
235 set_allfiles(0);
236 } else {
237 gbl.findex = 1;
238 }
239
240 /*
241 * process all blocks for a function. For Fortran, the terminating
242 * condition is when the "end" ILM is seen (there may be multiple
243 * subprograms per compilation -- the ilm file is reused). For C,
244 * the ilm file contains the blocks for all function. The loop
245 * terminates when the "end" ILM is seen and a non-zero value is
246 * returned; if the ilm file is at end-of-file, 0 is returned.
247 */
248 do {
249 expb.nilms = rdilms();
250 nextftag = fihb.nextftag;
251 nextfindex = fihb.nextfindex;
252#if DEBUG
253 if (DBGBIT(4, 0x800))
254 dumpilms();
255#endif
256 DEBUG_ASSERT(expb.nilms, "expand:ilm end of file");
257 /*
258 * the following check could be deleted if the max ilm block size is
259 * known or if space doesn't have to be conserved during this phase
260 */
261 if (expb.nilms > expb.ilmb.stg_size) {
262 EXP_MORE(expb.ilmb, ILM_AUX, expb.nilms + 100);
263 }
264
265 /* scan through all the ilms in the current ILM block */
266
267 for (ilmx = 0; ilmx < expb.nilms; ilmx += len) {
268 int saved_curbih = expb.curbih;
269 int saved_findex = fihb.nextfindex;
270 bool followed_by_file = false;
271 bool ilmx_is_block_label = false;
272 int findex, ftag;
273
274 /* the first time an ilm is seen, it has no result */
275
276 ILM_RESULT(ilmx) = 0;
277 ILM_EXPANDED_FOR(ilmx) = 0;
278
279 ILM_RESTYPE(ilmx) = 0; /* zero out result types */
280 ILM_NME(ilmx) = 0; /* zero out name entry (?) */
281 findex = 0;
282 ftag = 0;
283
284 ilmp = (ILM *)(ilmb.ilm_base + ilmx);
285 opc = ILM_OPC(ilmp);
286
287 if (opc == IM_BR) {
288 last_cpp_branch = ILM_OPND(ilmp, 1);
289 } else if (opc == IM_LABEL) {
290 /* Scope labels don't cause block breaks. */
291 ilmx_is_block_label = !is_scope_label(ILM_OPND(ilmp, 1));
292 if (!ilmx_is_block_label) {
293 new_callee_scope = ENCLFUNCG(ILM_OPND(ilmp, 1));
294 }
295 }
296
297 DEBUG_ASSERT(opc > 0 && opc < N_ILM, "expand: bad ilm");
298 len = ilms[opc].oprs + 1; /* length is number of words for the
299 * fixed operands and the opcode */
300 if (IM_VAR(opc))
301 len += ILM_OPND(ilmp, 1); /* include the number of
302 * variable operands */
303 if (IM_TRM(opc)) {
304 eval_ilm(ilmx);
305 }
306 else if (flg.smp && len) {
307 ll_rewrite_ilms(-1, ilmx, len);
308 }
309
310 if (opc != IM_FILE) {
311 ++nextftag;
312 fihb.nextftag = nextftag;
313 } else if ((XBIT(148, 0x1) || XBIT(148, 0x1000)) && !followed_by_file) {
314 int ftag;
315 int findex;
316 parse_im_file((ILM *)&ilmb.ilm_base[ilmx], NULL, &findex, &ftag);
317 if (ftag) {
318 nextfindex = findex;
319 nextftag = ftag;
320 fihb.nextfindex = nextfindex;
321 fihb.nextftag = nextftag;
322 }
323 }
324
325 /* If a new bih is created, detect certain scenarios */
326
327 if (expb.curbih > saved_curbih) {
328
329 /* Pay special attention to the transition from inlinee to inliner.
330 * If last bih (in the inlinee) is created by an IM_LABEL followed
331 * by an IM_FILE, we need to honor the ftag info in the IM_FILE.
332 */
333
334 if ((saved_curbih != 0) && (saved_curbih == last_label_bih) &&
335 (saved_findex > fihb.nextfindex))
336 BIH_FTAG(last_label_bih) = last_ftag;
337
338 /* Flag the scenario that the new bih is created by an IM_LABEL that is
339 * followed by an IM_FILE.
340 */
341
342 if (ilmx_is_block_label && followed_by_file) {
343 last_label_bih = expb.curbih;
344 last_ftag = ftag;
345 }
346 }
347 } /* end of loop through ILM block */
348
349 new_callee_scope = 0;
350 }
351 while (opc != IM_END && opc != IM_ENDF);
352
353 if (DBGBIT(10, 2) && (bihb.stg_avail != 1)) {
354 int bih;
355 for (bih = 1; bih != 0; bih = BIH_NEXT(bih)) {
356 if (BIH_EN(bih))
357 dump_blocks(gbl.dbgfil, bih, "***** BIHs for Function \"%s\" *****", 1);
358 }
359 dmpili();
360 }
361#if DEBUG
362 verify_function_ili(VERIFY_ILI_DEEP);
363 if (DBGBIT(10, 16)) {
364 dmpnme();
365 {
366 int i, j;
367 for (i = nmeb.stg_avail - 1; i >= 2; i--) {
368 for (j = nmeb.stg_avail - 1; j >= 2; j--) {
369 if (i != j)
370 (void)conflict(i, j);
371 }
372 }
373 }
374 }
375 if (DBGBIT(8, 64)) {
376 fprintf(gbl.dbgfil, " ILM(%d)", expb.ilmb.stg_size);
377 fprintf(gbl.dbgfil, " ILI(%d)", ilib.stg_avail);
378 fprintf(gbl.dbgfil, " ILT(%d)", iltb.stg_size);
379 fprintf(gbl.dbgfil, " BIH(%d)", bihb.stg_size);
380 fprintf(gbl.dbgfil, " NME(%d)\n", nmeb.stg_avail);
381 }
382#endif
383
384 ili_lpprg_init();
385 /* for C, we don't free the ilm area until we reach end-of-file */
386 FREE(ilmb.ilm_base);
387 ilmb.ilm_base = NULL;
388 EXP_FREE(expb.ilmb);
389 freearea(STR_AREA);
390 if (flg.opt < 2) {
391 if (rcandb.stg_base) {
392 EXP_FREE(rcandb);
393 rcandb.stg_base = NULL;
394 }
395 }
396 share_proc_ili = true;
397 exp_smp_fini();
398 fihb.nextftag = fihb.currftag = 0;
399
400 if (!XBIT(120, 0x4000000)) {
401 /* Restored file indexes to where they were before expand in case
402 they got changed somewhere.
403 */
404 set_allfiles(1);
405 } else {
406 fihb.nextfindex = fihb.currfindex = 1;
407 }
408 return expb.nilms;
409}

413/*
414 * Check that operand opr of ILM ilmx has been expanded.
415 * If this will be the first use of this ILM, then set ILM_EXPANDED_FOR
416 * to ilmx.
417 */
418static void
419eval_ilm_argument1(int opr, ILM *ilmpx, int ilmx)
420{
421 int op1, ilix;
422 if ((ilix = ILI_OF(op1 = ILM_OPND(ilmpx, opr))) == 0) {
423 /* hasn't been evaluated yet */
424 eval_ilm(op1);
425 /* mark this as expanded for this ILM */
426 ILM_EXPANDED_FOR(op1) = -ilmx;
427 } else if (ILM_EXPANDED_FOR(op1) < 0 && !is_cseili_opcode(ILI_OPC(ilix))) {
428 /* This was originally added for a parent ILM, so it hasn't
429 * been used as an operand ILI yet. Take ownership of it here.
430 * When it is reused later for a parent ILM,
431 * it will get then get turned into a CSE ILI */
432 ILM_EXPANDED_FOR(op1) = -ilmx;
433 }
434} /* eval_ilm_argument1 */

436void
437eval_ilm(int ilmx)
438{
439
440 ILM *ilmpx;
441 int noprs, /* number of operands in the ILM */
442 ilix, /* ili index */
443 tmp, /* temporary */
444 op1; /* operand 1 */
445 ILM_OP opcx; /**< ILM opcode of the ILM */
446
447 int first_op = 0;
448
449 opcx = ILM_OPC(ilmpx = (ILM *)(ilmb.ilm_base + ilmx));
450
451 if (flg.smp) {
452 if (IM_TYPE(opcx) != IMTY_SMP && ll_rewrite_ilms(-1, ilmx, 0)) {
453 if (ilmx == 0 && opcx == IM_BOS) {
454 /* Set line no for EPARx */
455 gbl.lineno = ILM_OPND(ilmpx, 1);
456 }
457 return;
458 }
459 }
460
461 if (EXPDBG(8, 2))
462 fprintf(gbl.dbgfil, "---------- eval ilm %d\n", ilmx);
463
464 if (!ll_ilm_is_rewriting())
465 {
466#ifdef OMP_OFFLOAD_LLVM
467 if (flg.omptarget && gbl.ompaccel_intarget) {
468 if (opcx == IM_MP_BREDUCTION) {
469 ompaccel_notify_reduction(true);
470 exp_ompaccel_reduction(ilmpx, ilmx);
471 } else if (opcx == IM_MP_EREDUCTION) {
472 ompaccel_notify_reduction(false);
473 return;
474 }
475
476 if (ompaccel_is_reduction_region())
477 return;
478 }
479#endif
480 /*-
481 * evaluate unevaluated "fixed" arguments:
482 * For each operand which is a link to another ilm, recurse (evaluate it)
483 * if not already evaluated
484 */
485 if (opcx == IM_DCMPLX || opcx == IM_CMPLX
486#ifdef TARGET_SUPPORTS_QUADFP
487 || opcx == IM_QCMPLX
488#endif
489 ) {
490 for (tmp = 1, noprs = 1; noprs <= ilms[opcx].oprs; ++tmp, ++noprs) {
491 if (IM_OPRFLAG(opcx, noprs) == OPR_LNK) {
492 eval_ilm_argument1(noprs, ilmpx, ilmx);
493 }
494 }
495 } else {
496 for (tmp = 1, noprs = ilms[opcx].oprs; noprs > first_op; ++tmp, --noprs) {
497 if (IM_OPRFLAG(opcx, noprs) == OPR_LNK) {
498 eval_ilm_argument1(noprs, ilmpx, ilmx);
499 }
500 }
501 }
502
503 /* evaluate unevaluated "variable" arguments */
504
505 if (IM_VAR(opcx) && IM_OPRFLAG(opcx, ilms[opcx].oprs + 1) == OPR_LNK) {
506 for (noprs = ILM_OPND(ilmpx, 1); noprs > 0; --noprs, ++tmp) {
507 eval_ilm_argument1(tmp, ilmpx, ilmx);
508 }
509 }
510
511 /*-
512 * check the "fixed" arguments for any duplicated values
513 */
514 for (tmp = 1, noprs = ilms[opcx].oprs; noprs > first_op; ++tmp, --noprs) {
515 if (IM_OPRFLAG(opcx, noprs) == OPR_LNK) {
516 /* all arguments will have been evaluated by now */
517 ilix = ILI_OF(op1 = ILM_OPND(ilmpx, noprs));
518 if (ILM_EXPANDED_FOR(op1) == -ilmx) {
519 ILM_EXPANDED_FOR(op1) = ilmx;
520 } else if (ilix && ILM_EXPANDED_FOR(op1) != ilmx) {
521 if (ILM_RESTYPE(op1) != ILM_ISCMPLX &&
522 ILM_RESTYPE(op1) != ILM_ISDCMPLX
523#ifdef LONG_DOUBLE_FLOAT128
524 && ILM_RESTYPE(op1) != ILM_ISFLOAT128CMPLX
525#endif
526 )
527 /* not complex */
528 ILM_RESULT(op1) = check_ilm(op1, ilix);
529 else {
530 /* complex */
531 ILM_RRESULT(op1) = check_ilm(op1, (int)ILM_RRESULT(op1));
532 ILM_IRESULT(op1) = check_ilm(op1, (int)ILM_IRESULT(op1));
533 }
534 }
535 }
536 }
537
538 /* check the "variable" arguments for any duplicated values */
539
540 if (IM_VAR(opcx) && IM_OPRFLAG(opcx, ilms[opcx].oprs + 1) == OPR_LNK) {
541 for (noprs = ILM_OPND(ilmpx, 1); noprs > 0; --noprs, ++tmp) {
542 ilix = ILI_OF(op1 = ILM_OPND(ilmpx, tmp));
543 if (ILM_EXPANDED_FOR(op1) == -ilmx) {
544 ILM_EXPANDED_FOR(op1) = ilmx;
545 } else if (ilix && ILM_EXPANDED_FOR(op1) != ilmx) {
546 if (ILM_RESTYPE(op1) != ILM_ISCMPLX &&
547 ILM_RESTYPE(op1) != ILM_ISDCMPLX
548#ifdef LONG_DOUBLE_FLOAT128
549 && ILM_RESTYPE(op1) != ILM_ISFLOAT128CMPLX
550#endif
551 ) {
552 /* not complex */
553 ILM_RESULT(op1) = check_ilm(op1, ilix);
554 } else {
555 /* complex */
556 ILM_RRESULT(op1) = check_ilm(op1, (int)ILM_RRESULT(op1));
557 ILM_IRESULT(op1) = check_ilm(op1, (int)ILM_IRESULT(op1));
558 }
559 }
560 }
561 }
562 }
563 /*
564 * ready to evaluate the ilm. opcx is opcode of current ilm, ilmpx is
565 * pointer to current ilm, and ilmx is index to the current ilm.
566 */
567 if (EXPDBG(8, 2))
568 fprintf(gbl.dbgfil, "ilm %s, index %d, lineno %d\n", ilms[opcx].name, ilmx,
569 gbl.lineno);
570
571 if (!IM_SPEC(opcx))
572 {
573 /* expand the macro definition */
574 tmp = exp_mac(opcx, ilmpx, ilmx);
575 if (IM_I8(opcx))
576 ILM_RESTYPE(ilmx) = ILM_ISI8;
577
578 return;
579 }
580 switch (IM_TYPE(opcx)) { /* special-cased ILM */
581
582 case IMTY_REF: /* reference */
583 exp_ref(opcx, ilmpx, ilmx);
584 break;
585
586 case IMTY_LOAD: /* load */
587 exp_load(opcx, ilmpx, ilmx);
588 break;
589
590 case IMTY_STORE: /* store */
591 exp_store(opcx, ilmpx, ilmx);
592 break;
593
594 case IMTY_BRANCH: /* branch */
595 exp_bran(opcx, ilmpx, ilmx);
596 break;
597
598 case IMTY_PROC: /* procedure */
599 exp_call(opcx, ilmpx, ilmx);
600 break;
601
602 case IMTY_INTR: /* intrinsic */
603 case IMTY_ARTH: /* arithmetic */
604 case IMTY_CONS: /* constant */
605 exp_ac(opcx, ilmpx, ilmx);
606 break;
607
608 case IMTY_MISC: /* miscellaneous */
609 exp_misc(opcx, ilmpx, ilmx);
610 break;
611
612 case IMTY_FSTR: /* fortran string */
613 exp_fstring(opcx, ilmpx, ilmx);
614 break;
615
616 case IMTY_SMP: /* smp ILMs */
617 exp_smp(opcx, ilmpx, ilmx);
618 break;
619
620 default: /* error */
621 interr("eval_ilm: bad op type", IM_TYPE(opcx), ERR_Severe);
622 break;
623 } /* end of switch on ILM opc */
624
625#ifdef OMP_OFFLOAD_LLVM
626
627 if (flg.omptarget && opcx == IM_ENLAB) {
628 /* Enables creation of libomptarget related structs in the main function,
629 * but it is not recommended option. Default behaviour is to initialize and
630 * create them in the global constructor. */
631 if (XBIT(232, 0x10)) {
632 if (!ompaccel_is_tgt_registered() && !OMPACCRTG(gbl.currsub) &&
633 !gbl.outlined) {
634 ilix = ll_make_tgt_register_lib2();
635 iltb.callfg = 1;
636 chk_block(ilix);
637 ompaccel_register_tgt();
638 }
639 }
640 /* We do not initialize spmd kernel library since we do not use spmd data
641 * sharing model. It does extra work and allocates device on-chip memory.
642 * */
643 if (XBIT(232, 0x40) && gbl.ompaccel_intarget) {
644 ilix = ompaccel_nvvm_get(threadIdX);
645 ilix = ll_make_kmpc_spmd_kernel_init(ilix);
646 iltb.callfg = 1;

656/*
657 * An ESTMT ILM (or an ILI whose value is to be discarded) is processed by
658 * walking the ILI tree (located by ilix) and creating ILTs for any function
659 * calls that exist in the tree. This routine is similar to reduce_ilt
660 * (iltutil.c) except that chk_block is used to add an ILT. This is done so
661 * that the "end of block" checks are performed.
662 */
663void
664exp_estmt(int ilix)
665{
666 int noprs, i;
667
668 ILI_OP opc = ILI_OPC(ilix);
669 if (IL_TYPE(opc) == ILTY_PROC && opc >= IL_JSR) {
670 iltb.callfg = 1; /* create an ILT for the function */
671 chk_block(ilix);
672 } else if (opc == IL_DFRDP && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
673 iltb.callfg = 1;
674 chk_block(ad1ili(IL_FREEDP, ilix));
675 } else if (opc == IL_DFRSP && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
676 iltb.callfg = 1;
677 chk_block(ad1ili(IL_FREESP, ilix));
678 } else if (opc == IL_DFRCS && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
679 iltb.callfg = 1;
680 chk_block(ad1ili(IL_FREECS, ilix));
681 }
682#ifdef LONG_DOUBLE_FLOAT128
683 else if (opc == IL_FLOAT128RESULT && ILI_OPC(ILI_OPND(ilix, 1)) != IL_QJSR) {
684 iltb.callfg = 1;
685 chk_block(ad1ili(IL_FLOAT128FREE, ilix));
686 }
687#endif
688 else if (opc == IL_VA_ARG) {
689 iltb.callfg = 1;
690 chk_block(ilix);
691 } else if (IL_HAS_FENCE(opc)) {
692 chk_block(ad_free(ilix));
693 } else {
694 /* otherwise, walk all of the link operands of the ILI */
695 noprs = ilis[opc].oprs;
696 for (i = 1; i <= noprs; ++i)
697 if (IL_ISLINK(opc, i))
698 exp_estmt((int)ILI_OPND(ilix, i));
699 }
700}

704/* Expand a scope label that should be inserted as an in-stream IL_LABEL ilt
705 * instead of splitting the current block.
706 *
707 * These scope labels are generated by enter_lexical_block() and
708 * exit_lexical_block(). They are verified by scope_verify().
709 */
710static void
711exp_scope_label(int lbl)
712{
713 int ilt, ilix;
714
715 /* Each scope label can only appear in one block. The ILIBLK field for the
716 * label must point to the unique BIH containing the IL_LABEL ilt.
717 */
718 assert(ILIBLKG(lbl) == 0 || ISTASKDUPG(GBL_CURRFUNC),
719 "Duplicate appearance of scope label", lbl, ERR_Severe);
720
721 /* This IM_LABEL may have been created for a lexical scope that turned out
722 * to not contain any variables. Such a label should simply be ignored. See
723 * cancel_lexical_block(). */
724 if (!ENCLFUNCG(lbl))
725 return;
726
727 ilix = ad1ili(IL_LABEL, lbl);
728
729 /* Insert the label at the top of the current block instead of appending
730 * it. Labels are not supposed to affect code generation, but they
731 * interfere with the trailing branches in a block. We also have code which
732 * expects the last three ilts in a block to follow a certain pattern for
733 * indiction variable updates.
734 *
735 * Skip any existing labels at the beginning of the block so that multiple
736 * labels appear in source order.
737 *
738 * The first and last ilts in the current block are stored in ILT_NEXT(0)
739 * and ILT_PREV(0) respectively; BIH_ILTFIRST isn't up-to-date. See
740 * wrilts().
741 */
742 ilt = ILT_NEXT(0);
743 while (ilt && ILI_OPC(ILT_ILIP(ilt)) == IL_LABEL)
744 ilt = ILT_NEXT(ilt);
745
746 if (!ilt) {
747 /* This block is all labels. Append the new label. */
748 expb.curilt = addilt(expb.curilt, ilix);
749 } else {
750 /* Now, ilt is the first non-label ilt in the block.
751 * Insert new label before ilt.
752 * This also does the right thing when ILT_PREV(ilt) == 0.
753 */
754 addilt(ILT_PREV(ilt), ilix);
755 }
756
757 ILIBLKP(lbl, expb.curbih);
758}
759
760void
761exp_label(SPTR lbl)
762{
763 int ilix; /* ili of an ilt */
764
765 /* Handle in-stream labels by creating an IL_LABEL ilt. */
766 if (is_scope_label(lbl)) {
767 exp_scope_label(lbl);
768 /* In-stream labels newer cause a new block to be created, so we're
769 * done. */
770 return;
771 }
772
773 if (expb.flags.bits.waitlbl) {
774 /*
775 * the current ilt points to a conditional branch. saveili locates an
776 * unconditional branch. If the conditional label is lbl, then the
777 * conditional is complemented whose label is changed to locate the
778 * one specified in the unconditional. The unconditional ili is not
779 * added.
780 */
781 expb.flags.bits.waitlbl = 0;
782 ilix = ILT_ILIP(expb.curilt); /* conditional branch ili */
783
784 if (expb.curilt && (ILI_OPND(ilix, ilis[ILI_OPC(ilix)].oprs)) == lbl) {
785 ILT_ILIP(expb.curilt) = compl_br(ilix, (int)(ILI_OPND(expb.saveili, 1)));
786 RFCNTD(lbl);
787 } else {
788 if (flg.opt != 1) {
789 wr_block();
790 cr_block();
791 }
792 expb.curilt = addilt(expb.curilt, expb.saveili);
793 }
794 }
795 /*
796 * check to see if the current ilt locates an ili which is a branch to
797 * lbl -- this only happens for opt levels other than 0.
798 */
799 if (flg.opt != 0 && ILT_BR(expb.curilt)) {
800 ilix = ILT_ILIP(expb.curilt);
801 if (ILI_OPND(ilix, ilis[ILI_OPC(ilix)].oprs) == lbl &&
802 ILI_OPC(ilix) != IL_JMPA && ILI_OPC(ilix) != IL_JMPMK &&
803 ILI_OPC(ilix) != IL_JMPM) {
804 int curilt = expb.curilt;
805
806 /*
807 * delete the branch ilt -- this may create ilts which locate
808 * functions
809 */
810 if (EXPDBG(8, 32))
811 fprintf(gbl.dbgfil,
812 "---exp_label: deleting branch ili %d from block %d\n", ilix,
813 expb.curbih);
814
815 expb.curilt = ILT_PREV(curilt);
816 ILT_NEXT(expb.curilt) = 0;
817 ILT_PREV(0) = expb.curilt;
818 STG_ADD_FREELIST(iltb, curilt);
819 expb.curilt = reduce_ilt(expb.curilt, ilix);
820 RFCNTD(lbl);
821 }
822 }
823 /*-
824 * finish off by checking lbl --
825 * 1. If opt 0 is requested, the label will always begin a block
826 * if it is a user label. NOTE that this covers the case when
827 * just -debug is specified (no -opt); if debug is requested along
828 * with a higher opt, we do not allow unreferenced labels to
829 * appear in the blocks since this can drastically affect code.
830 * WARNING: coffasm needs to be follow these conventions --- see
831 * the Is_user_label macro in all versions of coffasm.c.
832 * KLUDGE: for C blocks, labels are created -- their RFCNT's must
833 * be nonzero (set by semant).
834 * 2. If the reference count is still non-zero, a new block is
835 * created labeled by lbl.
836 */
837 if (flg.opt == 0 && CCSYMG(lbl) == 0) {
838 if (BIH_LABEL(expb.curbih) != 0 ||
839 (expb.curilt != 0 && !ILT_DBGLINE(expb.curilt))) {
840 wr_block();
841 cr_block();
842 }
843 BIH_LABEL(expb.curbih) = lbl;
844 ILIBLKP(lbl, expb.curbih);
845 fihb.currftag = fihb.nextftag;
846 fihb.currfindex = fihb.nextfindex;
847 } else if (RFCNTG(lbl) != 0) {
848 if (BIH_LABEL(expb.curbih) != 0 ||
849 (expb.curilt != 0 && !ILT_DBGLINE(expb.curilt))) {
850 wr_block();
851 cr_block();
852 } else if ((XBIT(148, 0x1) || XBIT(148, 0x1000)) && (expb.curilt == 0) &&
853 (fihb.currfindex != fihb.nextfindex)) {
854 fihb.currfindex = fihb.nextfindex;
855 fihb.currftag = fihb.nextftag;
856 }
857
858 BIH_LABEL(expb.curbih) = lbl;
859 ILIBLKP(lbl, expb.curbih);
860 fihb.currftag = fihb.nextftag;
861 fihb.currfindex = fihb.nextfindex;
862 }
863
864 else if (CCSYMG(lbl) == 0 && DBGBIT(8, 4096))
865 /* defd but not refd */
866 errlabel((error_code_t)120, ERR_Informational, gbl.lineno, SYMNAME(lbl),
867 CNULL);
868}

872/*
873 * the following macro is used by the load and store code to determine if the
874 * load or store operation conflicts with the data type of the item being
875 * fetched or stored. This is done for those names entries which are
876 * constant array or indirection references.
877 * Conflicts could occur when:
878 * 1. if the operation is for a double data item and the data type is not
879 * double.
880 * 2. if the operation is for a float data item and the data type is not
881 * float.
882 * 3. if the operation is for an integral type and its size is inconsistent
883 * with the size of the data type.
884 * A conflict is resolved by creating an array (or indirection) reference
885 * which has a non-constant offset. The macro argument, "cond", specifies the
886 * whether or not there is a conflict.
887 */
888#define CHECK_NME(nme, cond) \
889 { \
890 NT_KIND i = NME_TYPE(nme); \
891 if (NME_SYM(nme) == 0 && (i == NT_ARR || i == NT_IND) && (cond)) \
892 nme = add_arrnme(i, NME_NULL, NME_NM(nme), 0, NME_SUB(nme), \
893 NME_INLARR(nme)); \
894 }

901 return n;
902 if (dtype == DT_ASSNCHAR || dtype == DT_DEFERNCHAR)
903 return n;
904 return size_of(dtype);

962 * here, we have a load of the missing optional, replace by a zero
963 */
964void
965replace_by_zero(ILM_OP opc, ILM *ilmp, int curilm)
966{
967 INT num[4];
968 int zero = 0;
969 ILM_OP newopc;
970 int i1 = ILM_OPND(ilmp, 1);
971 switch (opc) {
972 /* handle complex */
973 case IM_CLD:
974 num[0] = 0;
975 num[1] = 0;
976 zero = getcon(num, DT_CMPLX);
977 newopc = IM_CDCON;
978 break;
979 case IM_CDLD:
980 num[0] = stb.dbl0;
981 num[1] = stb.dbl0;
982 zero = getcon(num, DT_DCMPLX);
983 newopc = IM_CCON;
984 break;
985 case IM_ILD:
986 case IM_LLD:
987 case IM_LFUNC: /* LFUNC, for PRESENT calls replaced by zero */
988 zero = stb.i0;
989 newopc = IM_ICON;
990 break;
991
992 case IM_KLD:
993 case IM_KLLD:
994 zero = stb.k0;
995 newopc = IM_KCON;
996 break;
997
998 case IM_SLLD:
999 case IM_SILD:
1000 case IM_CHLD:
1001 zero = stb.i0;
1002 newopc = IM_ICON;
1003 break;
1004
1005 case IM_RLD:
1006 zero = stb.flt0;
1007 newopc = IM_RCON;
1008 break;
1009
1010 case IM_DLD:
1011 zero = stb.dbl0;
1012 newopc = IM_DCON;
1013 break;
1014
1015 case IM_PLD:
1016 zero = stb.i0;
1017 newopc = IM_ICON;
1018 break;
1019
1020 default:
1021 interr("replace_by_zero opc not cased", opc, ERR_Severe);
1022 newopc = IM_ICON;
1023 break;
1024 }
1025 /* CHANGE the ILM in place */
1026 SetILM_OPC(ilmp, newopc);
1027 ILM_OPND(ilmp, 1) = zero;
1028 /* process as a constant */
1029 eval_ilm(curilm);
1030 SetILM_OPC(ilmp, opc);
1031 ILM_OPND(ilmp, 1) = i1;
1032} /* replace_by_zero */

1091void
1092exp_load(ILM_OP opc, ILM *ilmp, int curilm)
1093{
1094 int op1;
1095 int imag; /* address of the imag. part if complex */
1096
1097 int nme; /* names entry */
1098 int addr, /* address of the load */
1099 load = 0; /* load ili generated */
1100 SPTR tmp;
1101 int siz; /* MSZ value for load */
1102 DTYPE dt;
1103 bool confl;
1104 ILM *tmpp;
1105
1106 op1 = ILM_OPND(ilmp, 1);
1107 addr = op1;
1108 nme = NME_OF(addr);
1109 if (optional_missing_ilm(ilmp)) {
1110 replace_by_zero(opc, ilmp, curilm);
1111 return;
1112 }
1113
1114 /*
1115 * if the names entry is for a variable which is an array, then a new
1116 * names entry is created which will denote the first element (offset 0)
1117 * of the array -- this catches the cases of '*(a)', where a is an array
1118 * name
1119 */
1120 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1121 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));

1129 load = ad3ili(IL_LDSCMPLX, addr, nme, MSZ_F8);
1130 goto cand_load;

1135 tmp = addnme(NT_MEM, NOSYM, nme, 4);

1137 ILM_RESTYPE(curilm) = ILM_ISCMPLX;
1138 return;
1139 }

1141 if (XBIT(70, 0x40000000)) {
1142 CHECK_NME(nme, dt_nme(nme) != DT_DCMPLX);
1143 load = ad3ili(IL_LDDCMPLX, addr, nme, MSZ_F16);
1144 goto cand_load;
1145 } else {
1146 imag = ad3ili(IL_AADD, addr, ad_aconi(size_of(DT_DBLE)), 0);
1147 tmp = addnme(NT_MEM, SPTR_NULL, nme, 0);
1148 ILM_RRESULT(curilm) = ad3ili(IL_LDDP, addr, tmp, MSZ_F8);
1149 tmp = addnme(NT_MEM, NOSYM, nme, 8);
1150 ILM_IRESULT(curilm) = ad3ili(IL_LDDP, imag, tmp, MSZ_F8);
1151 ILM_RESTYPE(curilm) = ILM_ISDCMPLX;
1152 return;
1153 }
1154#ifdef TARGET_SUPPORTS_QUADFP
1155 case IM_CQLD:
1156 if (XBIT(70, 0x40000000)) {
1157 CHECK_NME(nme, dt_nme(nme) != DT_QCMPLX);
1158 load = ad3ili(IL_LDQCMPLX, addr, nme, MSZ_F32);

1162 tmp = addnme(NT_MEM, SPTR_NULL, nme, 0);

1164 tmp = addnme(NT_MEM, NOSYM, nme, 16);
1165 ILM_IRESULT(curilm) = ad3ili(IL_LDQP, imag, tmp, MSZ_F16);
1166 ILM_RESTYPE(curilm) = ILM_ISQCMPLX;
1167 return;
1168 }
1169#endif
1170 case IM_ILD:
1171 case IM_LLD:
1172 confl = false;
1173 dt = dt_nme(nme);
1174 if (dt && DT_ISSCALAR(dt) && SCALAR_SIZE(dt, 4) != 4)
1175 confl = true;
1176 CHECK_NME(nme, confl);
1177 load = ad3ili(IL_LD, addr, nme, MSZ_WORD);
1178 cand_load:
1179 ADDRCAND(load, nme);
1180 break;
1181
1182 case IM_KLD:
1183 case IM_KLLD:
1184 confl = false;
1185 dt = dt_nme(nme);
1186 if (dt && DT_ISSCALAR(dt) && SCALAR_SIZE(dt, 8) != 8)
1187 confl = true;
1188 CHECK_NME(nme, confl);
1189 if (XBIT(124, 0x400)) {
1190 load = ad3ili(IL_LDKR, addr, nme, MSZ_I8);
1191 rcandb.kr = 1;
1192 } else {
1193 if (flg.endian)
1194 addr = ad3ili(IL_AADD, addr, ad_aconi((INT)size_of(DT_INT)), 0);
1195 load = ad3ili(IL_LD, addr, nme, MSZ_WORD);
1196 }
1197 ADDRCAND(load, nme);
1198 break;
1199
1200 case IM_SLLD:
1201 case IM_SILD:
1202 siz = MSZ_SHWORD;
1203 confl = false;
1204 dt = dt_nme(nme);
1205 if (dt && DT_ISSCALAR(dt) && size_of(dt) != 2)
1206 confl = true;
1207 CHECK_NME(nme, confl);
1208 load = ad3ili(IL_LD, addr, nme, siz);
1209 goto cand_load;
1210
1211 case IM_CHLD:
1212 siz = MSZ_SBYTE;
1213 confl = false;
1214 dt = dt_nme(nme);
1215 if (dt && DT_ISSCALAR(dt) && size_of(dt) != 1)
1216 confl = true;
1217 CHECK_NME(nme, confl);
1218 load = ad3ili(IL_LD, addr, nme, siz);
1219 goto cand_load;
1220
1221 case IM_RLD:
1222 CHECK_NME(nme, dt_nme(nme) != DT_FLOAT);
1223 load = ad3ili(IL_LDSP, addr, nme, MSZ_F4);
1224 goto cand_load;
1225
1226 case IM_DLD:
1227 CHECK_NME(nme, dt_nme(nme) != DT_DBLE);
1228 load = ad3ili(IL_LDDP, addr, nme, MSZ_F8);
1229 goto cand_load;
1230 case IM_QLD: /*m128*/
1231 CHECK_NME(nme, DTY(dt_nme(nme)) != TY_128);
1232 load = ad3ili(IL_LDQ, addr, nme, MSZ_F16);
1233 goto cand_load;
1234#ifdef TARGET_SUPPORTS_QUADFP
1235 /* transform the QFLD ilm to LDQP ili */
1236 case IM_QFLD: /* fp128 */
1237 CHECK_NME(nme, DTY(dt_nme(nme)) != TY_QUAD);
1238 load = ad3ili(IL_LDQP, addr, nme, MSZ_F16);
1239 goto cand_load;
1240#endif
1241 case IM_M256LD: /* m256 */
1242 CHECK_NME(nme, DTY(dt_nme(nme)) != TY_256);
1243 load = ad3ili(IL_LD256, addr, nme, MSZ_F32);
1244 goto cand_load;
1245#ifdef LONG_DOUBLE_FLOAT128
1246 case IM_FLOAT128LD:
1247 CHECK_NME(nme, DTY(dt_nme(nme)) != TY_FLOAT128);
1248 load = ad3ili(IL_FLOAT128LD, addr, nme, MSZ_F16);
1249 goto cand_load;
1250#endif /* LONG_DOUBLE_FLOAT128 */
1251
1252 case IM_PLD:
1253/* fortran: pointer variables are really integer variables;
1254 * later phases 'depend' on seeing references via pointers
1255 * via the 'LDA' ili.
1256 */
1257 /* if using integer*8 variables and not 64-bit precision,
1258 adjust the address of pointer */
1259 /* ???
1260 if (flg.endian && !XBIT(124,0x400))
1261 */
1262 if (flg.endian) {
1263 tmp = ILM_SymOPND(ilmp, 2);
1264 tmpp = (ILM *)(ilmb.ilm_base + ILM_OPND(ilmp, 1));
1265 if (((tmp == SPTR_NULL) && DTYPEG(ILM_OPND(tmpp, 1)) == DT_INT8) ||
1266 (SCG(tmp) == SC_BASED && DTYPEG(MIDNUMG(tmp)) == DT_INT8))
1267 addr = ad3ili(IL_AADD, addr, ad_aconi(size_of(DT_INT)), 0);
1268 }
1269 load = ad2ili(IL_LDA, addr, nme);
1270 ADDRCAND(load, nme);
1271 /*
1272 * if the 2nd operand is non-zero, then the 2nd operand is the
1273 * symbol table entry of some sort of based object. The symbol
1274 * table entry is the object in a POINTER statement
1275 *
1276 * For POINTER, a names entry of NT_IND through the pointer variable
1277 * is sufficent.
1278 *
1279 * When the PLD is to load the pointer to a character object, the
1280 * additional character information needs to be created (examine
1281 * the data type of the symbol which is the second operand.
1282 */
1283 tmp = ILM_SymOPND(ilmp, 2);
1284 if (tmp) {
1285 DTYPE dtype;
1286#if DEBUG
1287 if (!(tmp && DEVICECOPYG(tmp) && DEVCOPYG(tmp))) {
1288 assert(STYPEG(tmp) == ST_MEMBER || SCG(tmp) == SC_BASED ||
1289 SCG(tmp) == SC_EXTERN,
1290 "exp_load:PLD op#2 not based sym, member, or procedure pointer",
1291 tmp, ERR_Severe);
1292 }
1293#endif
1294 dtype = DDTG(DTYPEG(tmp));
1295 if (DTY(dtype) == TY_PTR)
1296 dtype = DTySeqTyElement(dtype);
1297 if (DTY(dtype) == TY_CHAR || DTY(dtype) == TY_NCHAR) {
1298 int mxlen, clen;
1299 mxlen = 0;
1300 if ((dtype == DT_DEFERCHAR || dtype == DT_DEFERNCHAR) && SDSCG(tmp)) {
1301 if (STYPEG(tmp) == ST_MEMBER) {
1302 int member, base;
1303 member = ILM_OPND(ilmp, 1);
1304 base = ilmb.ilm_base[member + 1];
1305 clen = exp_get_sdsc_len(tmp, ILI_OF(base), NME_OF(base));
1306 } else {
1307 clen = exp_get_sdsc_len(tmp, 0, 0);
1308 }
1309 } else
1310 if (
1311 STYPEG(tmp) != ST_MEMBER &&
1312 CLENG(tmp) > 0) {
1313 if (CHARLEN_64BIT) {
1314 int clensym, ili;
1315 clensym = CLENG(tmp);
1316 if (size_of(DTYPEG(clensym)) == 8) {
1317 ili = mk_address(CLENG(tmp));
1318 clen = ad3ili(IL_LDKR, ili, addnme(NT_VAR, CLENG(tmp), 0, 0),
1319 MSZ_I8);
1320 } else {
1321 /*
1322 * -Mlarge_arrays (large character lengths WORK-AROUND)
1323 * there are several cases where the front-end IS NOT creating
1324 * 64-bit length temps, e.g., the length temp for the adjustl
1325 * intrinisc. When we're ready to correct the support of
1326 * large character, this section of code ought to turn into
1327 * an assert.
1328 */
1329 ili = mk_address(CLENG(tmp));
1330 clen = ad3ili(IL_LD, ili, addnme(NT_VAR, CLENG(tmp), 0, 0),
1331 MSZ_WORD);
1332 clen = ad1ili(IL_IKMV, clen);
1333 }
1334 } else {
1335 int ili = mk_address(CLENG(tmp));
1336 clen =
1337 ad3ili(IL_LD, ili, addnme(NT_VAR, CLENG(tmp), 0, 0), MSZ_WORD);
1338 }
1339 }
1340 else if (DTyCharLength(dtype) == 0 && SDSCG(tmp)) {
1341 clen = exp_get_sdsc_len(tmp, 0, 0);
1342 }
1343 else if (CHARLEN_64BIT)
1344 clen = mxlen = ad_kconi(DTyCharLength(dtype));
1345 else
1346 clen = mxlen = ad_icon(DTyCharLength(dtype));
1347 ILM_CLEN(curilm) = clen;
1348 ILM_MXLEN(curilm) = mxlen;
1349 ILM_RESTYPE(curilm) = ILM_ISCHAR;
1350 } else if (STYPEG(tmp) == ST_MEMBER) {
1351 ILM_NME(curilm) = addnme(NT_IND, SPTR_NULL, nme, 0);
1352#ifdef DEVICEG
1353 } else if (DEVICEG(tmp) && DT_ISBASIC(DTYPEG(tmp))) {
1354 ILM_NME(curilm) = addnme(NT_VAR, tmp, 0, 0);
1355#ifdef TEXTUREG
1356 } else if (DEVICEG(tmp) && TEXTUREG(tmp)) {
1357 ILM_NME(curilm) = addnme(NT_VAR, tmp, 0, 0);
1358#endif
1359#endif
1360 } else if (NOCONFLICTG(tmp)) {
1361 /* the frontend has determined that this pointer-based object
1362 * cannot conflict with other references via pointers; for
1363 * example, allocatable arrays and automatic arrays.
1364 */
1365 ILM_NME(curilm) = addnme(NT_VAR, tmp, 0, 0);
1366 } else if (XBIT(125, 0x40)) {
1367 /* Cray's pointer semantics */
1368 ILM_NME(curilm) = addnme(NT_VAR, tmp, 0, 0);
1369 } else {
1370 ILM_NME(curilm) = addnme(NT_IND, SPTR_NULL, nme, 0);
1371 }
1372 } else {
1373 ILM_NME(curilm) = addnme(NT_IND, SPTR_NULL, nme, 0);
1374 }
1375 break;
1376
1377#ifdef LONG_DOUBLE_FLOAT128
1378 case IM_CFLOAT128LD:
1379 ILM_RRESULT(curilm) =
1380 ad3ili(IL_FLOAT128LD, addr, addnme(NT_MEM, 0, nme, 0), MSZ_F16);
1381 ILM_IRESULT(curilm) =
1382 ad3ili(IL_FLOAT128LD, ad3ili(IL_AADD, addr, ad_aconi(16), 0),
1383 addnme(NT_MEM, 1, nme, 16), MSZ_F16);
1384 ILM_RESTYPE(curilm) = ILM_ISFLOAT128CMPLX;
1385 return;
1386#endif /* LONG_DOUBLE_FLOAT128 */
1387
1388 default:
1389 interr("exp_load opc not cased", opc, ERR_Severe);
1390 break;
1391 }
1392
1393 ILM_RESULT(curilm) = load;
1394}

1409void
1410exp_store(ILM_OP opc, ILM *ilmp, int curilm)
1411{
1412 int nme; /* names entry */
1413 int op1, /* operand 1 of the ILM */
1414 op2; /* operand 2 of the ILM */
1415 int store = 0, /* store ili generated */
1416 addr, /* address ili where value stored */
1417 expr, /* ili of value being stored */
1418 siz, /* size of the field in the field store */
1419 ilix, /* ili indexi */
1420 ilix1; /* ili index */
1421 int tmp;
1422 DTYPE dt;
1423 bool confl;
1424
1425 int imag; /* address of the imag. part if complex */
1426
1427 op1 = ILM_OPND(ilmp, 1);
1428
1429 op2 = ILM_OPND(ilmp, 2);
1430 nme = NME_OF(op1);
1431 if (opc != IM_PSEUDOST) {
1432 if (optional_missing_ilm(ilmp)) {
1433 /* this is a store to a missing optional argument.
1434 * it must be on a path that is branched around, or it is illegal.
1435 * simply drop the expression */
1436 return;
1437 }
1438 }
1439
1440 switch (opc) {
1441 case IM_LST:
1442 case IM_IST:
1443 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1444 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1445 confl = false;
1446 dt = dt_nme(nme);
1447 if (dt && DT_ISSCALAR(dt) && SCALAR_SIZE(dt, 4) != 4)
1448 confl = true;
1449 CHECK_NME(nme, confl);
1450 ilix = ILI_OF(op2);
1451 if (IL_RES(ILI_OPC(ilix)) == ILIA_AR)
1452 ilix = ad1ili(IL_AIMV, ilix);
1453 store = ad4ili(IL_ST, ilix, ILI_OF(op1), nme, MSZ_WORD);
1454 cand_store:
1455 if (NME_TYPE(nme) == NT_VAR)
1456 ASSNP(NME_SYM(nme), 1);
1457 ADDRCAND(store, nme);
1458 SET_ASSN(nme);
1459 break;
1460
1461 case IM_KLST:
1462 case IM_KST:
1463 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1464 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1465 confl = false;
1466 dt = dt_nme(nme);
1467 if (dt && DT_ISSCALAR(dt) && SCALAR_SIZE(dt, 8) != 8)
1468 confl = true;
1469 if (XBIT(124, 0x400)) {
1470 /* problem arose with the pointer statement and the value
1471 * returned by the call to ftn_allocate being an IR
1472 * AND (as of 12/09/2010) with the result being an AR
1473 */
1474 ilix = ILI_OF(op2);
1475 if (IL_RES(ILI_OPC(ilix)) == ILIA_AR)
1476 ilix = ad1ili(IL_AKMV, ilix);
1477 else {
1478 if (IL_RES(ILI_OPC(ilix)) != ILIA_KR)
1479 ilix = ad1ili(IL_IKMV, ilix);
1480 }
1481 store = ad4ili(IL_STKR, ilix, ILI_OF(op1), nme, MSZ_I8);
1482 rcandb.kr = 1;
1483 } else {
1484 addr = ILI_OF(op1);
1485 if (flg.endian)
1486 addr = ad3ili(IL_AADD, (int)ILI_OF(op1), ad_aconi((INT)size_of(DT_INT)),
1487 0);
1488 ilix = ILI_OF(op2);
1489 if (IL_RES(ILI_OPC(ilix)) == ILIA_AR)
1490 ilix = ad1ili(IL_AIMV, ilix);
1491 else if (IL_RES(ILI_OPC(ilix)) == ILIA_KR)
1492 ilix = ad1ili(IL_KIMV, ilix);
1493 store = ad4ili(IL_ST, ilix, addr, nme, MSZ_WORD);
1494 }
1495 CHECK_NME(nme, confl);
1496 if (NME_TYPE(nme) == NT_VAR)
1497 ASSNP(NME_SYM(nme), 1);
1498 ADDRCAND(store, nme);
1499 SET_ASSN(nme);
1500 break;
1501
1502 case IM_SLST:
1503 case IM_SIST:
1504 siz = MSZ_SHWORD;
1505 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1506 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1507 confl = false;
1508 dt = dt_nme(nme);
1509 if (dt && DT_ISSCALAR(dt) && size_of(dt) != 2)
1510 confl = true;
1511 CHECK_NME(nme, confl);
1512 expr = ILI_OF(op2);
1513 store = ad4ili(IL_ST, expr, (int)ILI_OF(op1), nme, siz);
1514 goto cand_store;
1515
1516 case IM_CHST:
1517 siz = MSZ_SBYTE;
1518 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1519 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1520 confl = false;
1521 dt = dt_nme(nme);
1522 if (dt && DT_ISSCALAR(dt) && size_of(dt) != 1)
1523 confl = true;
1524 CHECK_NME(nme, confl);
1525 expr = ILI_OF(op2);
1526 store = ad4ili(IL_ST, expr, (int)ILI_OF(op1), nme, siz);
1527 goto cand_store;
1528
1529 case IM_AST:
1530 expr = ILI_OF(op2);
1531 if (IL_RES(ILI_OPC(expr)) == ILIA_AR)
1532 expr = ad1ili(IL_AIMV, expr);
1533 store = ad4ili(IL_ST, expr, (int)ILI_OF(op1), nme, MSZ_WORD);
1534 SET_ASSN(nme);
1535 break;
1536
1537 case IM_KAST:
1538 addr = ILI_OF(op1);
1539 expr = ILI_OF(op2);
1540 if (IL_RES(ILI_OPC(expr)) == ILIA_AR)
1541 expr = ad1ili(IL_AKMV, expr);
1542 store = ad4ili(IL_STKR, expr, addr, nme, MSZ_I8);
1543 SET_ASSN(nme);
1544 break;
1545
1546 case IM_PSTRG1:
1547 store = ad2ili(IL_STRG1, (int)ILI_OF(op1), op2);
1548 break;
1549
1550 case IM_PST:
1551 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1552 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1553 confl = false;
1554 dt = dt_nme(nme);
1555 if (dt && DT_ISSCALAR(dt) && SCALAR_SIZE(dt, 8) != 8)
1556 confl = true;
1557 CHECK_NME(nme, confl);
1558 expr = ILI_OF(op2);
1559 switch (ILI_OPC(expr)) {
1560 case IL_AIMV:
1561 case IL_AKMV:
1562 expr = ILI_OPND(expr, 1);
1563 break;
1564 default:
1565 break;
1566 }
1567 if (IL_RES(ILI_OPC(expr)) != ILIA_AR) {
1568 expr = ad1ili(IL_KAMV, expr);
1569 }
1570 store = ad3ili(IL_STA, expr, (int)ILI_OF(op1), nme);
1571
1572 /*
1573 * check if &var is being stored. If so, the base symbol's "address
1574 * taken" flag is set.
1575 */
1576 loc_of((int)NME_OF(op2));
1577
1578 /*
1579 * store the names result of the store -- this is just an indirection
1580 * based on the names entry of the STA
1581 */
1582 ILM_NME(curilm) = addnme(NT_IND, SPTR_NULL, nme, (INT)0);
1583 goto cand_store;
1584
1585 case IM_RST:
1586 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1587 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1588 CHECK_NME(nme, dt_nme(nme) != DT_FLOAT);
1589 store = ad4ili(IL_STSP, (int)ILI_OF(op2), (int)ILI_OF(op1), nme, MSZ_F4);
1590 goto cand_store;

1592 case IM_DST:

1594 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1595 CHECK_NME(nme, dt_nme(nme) != DT_DBLE);
1596 store = ad4ili(IL_STDP, (int)ILI_OF(op2), (int)ILI_OF(op1), nme, MSZ_F8);
1597 goto cand_store;
1598 case IM_QST: /*m128*/
1599 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)
1600 nme = add_arrnme(NT_ARR, SPTR_NULL, nme, 0, ad_icon(0), NME_INLARR(nme));
1601 CHECK_NME(nme, DTY(dt_nme(nme)) != TY_128);
1602 store = ad4ili(IL_STQ, (int)ILI_OF(op2), (int)ILI_OF(op1), nme, MSZ_F16);
1603 goto cand_store;
1604#ifdef TARGET_SUPPORTS_QUADFP
1605 /* transform the QFST ilm TO STQP ili */
1606 case IM_QFST:
1607 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)

1614 if (NME_TYPE(nme) == NT_VAR && DTY(DTYPEG(NME_SYM(nme))) == TY_ARRAY)

1629 case IM_SMOVE: /* make sure this works for both languages */
1630 SET_ASSN(NME_OF(op1));

1655 ilix1 = ILI_OPND(ilix1, 2); /* IL_GARGRET */
1656 ILI_OPND(ilix1, 1) = ILI_OF(op1);

1666 ILM_RESULT(curilm) = ILI_OF(op2);
1667 ILM_NME(curilm) = NME_OF(op2);
1668 return;

1673 ILM_RESULT(curilm) = 0;
1674 ILM_NME(curilm) = NME_UNK;
1675 return;
1676
1677 case IM_PSEUDOST:

1699 store = ad1ili(IL_FREESP, expr);
1700 break;
1701
1702 case ILIA_DP:
1703 if (ILM_RESTYPE(op2) == ILM_ISDCMPLX) {
1704 store = ad1ili(IL_FREEDP, (int)ILM_IRESULT(op2));
1705 chk_block(store);
1706 ILM_IRESULT(curilm) = store;
1707 ILM_RESTYPE(curilm) = ILM_ISDCMPLX;
1708 if (EXPDBG(8, 16))
1709 fprintf(gbl.dbgfil, "store imag: ilm %d, block %d, ili %d\n", curilm,
1710 expb.curbih, store);
1711 }
1712 store = ad1ili(IL_FREEDP, expr);
1713 break;
1714#ifdef ILIA_CS
1715 case ILIA_CS:
1716 store = ad1ili(IL_FREECS, expr);
1717 break;
1718 case ILIA_CD:
1719 store = ad1ili(IL_FREECD, expr);
1720 break;
1721#endif
1722 case ILIA_AR:
1723 store = ad1ili(IL_FREEAR, expr);
1724 ILM_NME(curilm) = NME_OF(op2);
1725 break;
1726
1727 case ILIA_KR:
1728 store = ad1ili(IL_FREEKR, expr);
1729 break;
1730
1731#ifdef LONG_DOUBLE_FLOAT128
1732 case ILIA_FLOAT128:
1733 if (ILM_RESTYPE(op2) == ILM_ISFLOAT128CMPLX) {
1734 store = ad1ili(IL_FLOAT128FREE, (int)ILM_IRESULT(op2));
1735 chk_block(store);
1736 ILM_IRESULT(curilm) = store;
1737 ILM_RESTYPE(curilm) = ILM_ISFLOAT128CMPLX;
1738 if (EXPDBG(8, 16))
1739 fprintf(gbl.dbgfil, "store imag: ilm %d, block %d, ili %d\n", curilm,
1740 expb.curbih, store);
1741 }
1742 store = ad1ili(IL_FLOAT128FREE, expr);
1743 break;
1744#endif /* LONG_DOUBLE_FLOAT128 */
1745
1746 case ILIA_LNK:
1747 dt = ili_get_vect_dtype(expr);
1748 if (dt) {
1749 store = ad2ili(IL_FREE, expr, dt);
1750 break;
1751 }

2321 pattern += ilis[newili.opc].oprs + 2;

2323 } /*** end of ilicnt loop ***/

2335 resdt = (DTYPE)-1;
2336 p = nm;
