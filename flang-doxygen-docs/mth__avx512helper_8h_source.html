16/*
17 * mth_avx512helper.h - helper macros for AVX512.
18 *
19 * Two objectives:
20 * 1) Provide AVX/AVX2 and SKYLAKE-AVX512 compatibility.
21 * There are instructions in the AVX/AVX2 extensions that do not exist
22 * with SKYLAKE-AVX512. Create macros that provide identical functionality
23 * to AVX/AVX2 with AVX512 - though using 512-bit registers.
24 *
25 * Example:
26 * Vector compare in the AVX/AVX2 extensions set a resulting
27 * vector register with a -1 (32 or 64-bit) where the results of the
28 * comparison match. AVX512 uses the K registers for the result of the
29 * compare. So extend _mm256_cmpeq_epi32(a,b) to _MM512_CMPEQ_EPI32 as:
30 * (__m512i) _mm512_maskz_set1_epi32(_mm512_cmpeq_epi32_mask(a, b), -1))
31 *
32 *
33 * 2) Provide KNC and SKYLAKE-AVX512 compatibility.
34 * Another complication is that we currently build to have a common object
35 * between KNL and AVX512F (CPUID flags AVX512F for AVX-512, KNCNI for KNC)
36 * thus AVX512 instructions.
37 *
38 * Example:
39 * The KNC extensions do not have a "floating point" boolean AND()
40 * instruciton.
41 * Extend _mm512_and_ps(a,b) to _MM512_AND_PS as:
42 * (__m512) _mm512_and_si512(a, _mm512_castps_si512(b))
43 *
44 * Macro FCN_AVX512(name) is used to create unique (entry point) names
45 * based upon the extensions "_knl" or "_512" depending whether KNL is
46 * targetted or not.
47 *
48 * Note: Not every possible AVX/AVX2 intrinsic is currently defined.
49 * They helper macros are created as needed in porting the FMA3 version
50 * of the math intrinsics source code.
51 */
52
