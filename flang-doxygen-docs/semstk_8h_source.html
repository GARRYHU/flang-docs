25/* INT members within the union portion of the stack entry need to be
26 * padded for systems, such as the decalpha, where ints are 32-bits and
27 * pointers are 64-bits. We cannot have pointer members in the stack entry
28 * overlapping two INT members.
29 */
30#define SST_INT(m) INT m

32typedef struct sst {
33 short id; /**< type of this stack entry */
34 unsigned char flag; /**< general flag */
35 unsigned f1 : 1; /**< plain expr flag - 0 => no parens */
36 unsigned f2 : 1; /**< id is an alias */
37 int ast; /**< the AST for this stack entry */
38 int mnoff; /**< derived type flag & information */
39 int sr; /**< save & restore word */
40 int lineno; /**< line number associated with this stack entry */
41 int col; /**< column number associated with this stack entry */
42
43 union { /**< value of this stack entry */
44 struct {/**< general purpose word value */
45 SST_INT(w1);
46 SST_INT(w2);
47 SST_INT(w3);
48 SST_INT(w4);
49 SST_INT(w5);
50 } wval;
51 struct { /**< constructor value */
52 SST_INT(dum1); /**< needs wval.w1 */
53 SST_INT(dum2); /**< needs wval.w2 */
54 SST_INT(dum3); /**< needs wval.w3 */
55 SST_INT(dum4); /**< needs wval.w4 */
56 ACL *acl;
57 } cnval;
58 struct { /**< equivalence item */
59 SST_INT(dum1); /**< needs wval.w1 (SYM)*/
60 SST_INT(substring);
61 SST_INT(offset);
62 SST_INT(subscript);
63 } eqvval;
64 struct {/**< item list value */
65 ITEM *beg;
66 ITEM *end;
67 SST_INT(count);
68 } ilval;
69 struct {/**< variable name list for initializers */
70 VAR *beg;
71 VAR *end;
72 SST_INT(count);
73 } vlval;
74 struct {/**< constant list for initializers */
75 ACL *beg;
76 ACL *end;
77 SST_INT(count);
78 } clval;
79 struct { /**< derived type value */
80 ITEM *dum1; /**< needs ilval.beg */
81 ITEM *dum2; /**< needs ilval.end */
82 ITEM *beg;
83 ITEM *end;
84 } dtval;
85 struct {/**< vector slice triplet notation */
86 struct sst *next;
87 struct sst *e1;
88 struct sst *e2;
89 struct sst *e3;
90 } tlval;
91 } value;
92} SST;
