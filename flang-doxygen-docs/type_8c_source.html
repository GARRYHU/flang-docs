274/** \brief Check whether two polymorphic types are conformable.
275 *
276 * This routine is similar to the conformable routines in rdst.c, but
277 * it is for two polymorphic scalar objects instead of arrays.
278 *
279 * This is needed in polymorphic allocatable assignment. If two types
280 * are conformable or the type of the left hand side expression is large
281 * enough to hold the value(s) on the right hand side, then we do not have
282 * to reallocate the left hand side if it's already allocated.
283 *
284 * \param ab is the address of the first object.
285 * \param ab is the first object's descriptor.
286 * \param bd is the second object's descriptor.
287 * \param flag can be 0, 1, 2. See flag's description for poly_asn().
288 *
289 * \return 1 if types are conformable; 0 if types are not conformable but
290 * \param ab is big enough to hold \param bd; -1 if \param ab is not
291 * conformable, not big enough, or not allocated.
292 */
293int
294ENTF90(POLY_CONFORM_TYPES, poly_conform_types)(char *ab, F90_Desc *ad,
295 F90_Desc *bd, __INT_T flag)
296{
297 /* Possible return values. Do not change the integer values */
298 enum {
299 NOT_BIG_ENOUGH = -1, /* not conformable, not big enough */
300 BIG_ENOUGH = 0, /* not conformable but big enough */
301 CONFORMABLE = 1 /* conformable */
302 };
303
304 TYPE_DESC *src_td, *dest_td;
305 int src_sz, dest_sz;
306 int src_is_array = 0, dest_is_array = 0;
307
308 if (!I8(__fort_allocated)(ab)) {
309 return NOT_BIG_ENOUGH;
310 }
311
312 get_source_and_dest_sizes(ad, bd, &dest_sz, &src_sz, &dest_is_array,
313 &src_is_array, &dest_td, &src_td, flag);
314
315 if (dest_td != 0 && src_td != 0) {
316 if (dest_td == src_td && dest_sz == src_sz) {
317 return CONFORMABLE;
318 } else if (dest_sz >= src_sz) {
319 return BIG_ENOUGH;
320 }
321 }
322 return NOT_BIG_ENOUGH;
323}

368/** \brief Compute address of an element in a polymorphic array.
369 *
370 * This routine is intended for arrays with 4 or more dimensions.
371 * For 1, 2, or 3 dimensional arrays, use the specialized routines below.
372 * Those routines do not incur the overhead of variable arguments.
373 *
374 * \param ab is the base address of the array.
375 * \param ad is the array's descriptor.
376 * \param result is the address of the pointer that will hold the result.
377 * \param variable arguments include an __INT_T* for each index of the array
378 * element that we are computing.
379 */
380void
381ENTF90(POLY_ELEMENT_ADDR, poly_element_addr)(char *ab, F90_Desc *ad,
382 char **result, ...)
383
384{
385 va_list va;
386 __INT_T sz;
387 int i, numdims;
388 __INT_T index[MAXDIMS];
389 __INT_T offset;
390 DECL_DIM_PTRS(add);
391
392 va_start(va, result);
393
394 sz = ENTF90(GET_OBJECT_SIZE, get_object_size)(ad);
395 numdims = F90_RANK_G(ad);
396
397 for (i = 0; i < numdims; ++i) {
398 SET_DIM_PTRS(add, ad, i);
399 index[i] = *va_arg(va, __INT_T *) - F90_DPTR_LBOUND_G(add);
400 }
401
402 i = numdims - 1;
403 offset = index[i];
404 for (--i; i >= 0; --i) {
405 SET_DIM_PTRS(add, ad, i);
406 offset = index[i] + (F90_DPTR_EXTENT_G(add) * offset);
407 }
408 *result = ab + sz * offset;
409 va_end(va);
410}

604/** \brief Returns a type descriptor pointer of a specified ancestor of
605 * a type descriptor.
606 *
607 * \param src_td is the type descriptor used to locate the ancestor type
608 * type descriptor.
609 * \param level specifies the heirarchical position in the inheritance graph
610 * of the desired ancestor type descriptor. To find its immediate
611 * parent, specify a level equal to src_td's level.
612 *
613 * \return a type descriptor representing the ancestor or NULL if there is no
614 * ancestor.
615 */
616static TYPE_DESC *
617get_parent_pointer(TYPE_DESC *src_td, __INT_T level)
618{
619 __INT_T offset, src_td_level;
620 TYPE_DESC *parent;
621
622 if (level <= 0 || src_td == NULL)
623 return NULL;
624
625 src_td_level = src_td->obj.level;
626 if (src_td_level < 0 || level > src_td_level)
627 return NULL;
628
629 if (src_td->parents != NULL) {
630 /* The parents field is filled in, so use it to get the desired parent */

647static void

649{
650 /* See also Cuda Fortran version in rte/cudafor/hammer/src/dev_allo.c
651 * call dev_process_final_procedures()
652 */
653 OBJECT_DESC *src = (OBJECT_DESC *)sd;
654 TYPE_DESC *src_td, *tmp_td;

663 if (tmp_td && (tmp_td->obj.tag > 0 && tmp_td->obj.tag < __NTYPES)) {
664 src_td = tmp_td;

666 src_td = 0;
667 }
668 } else {

678

706 ld->tag != 'F') ||
707 ld->offset < 0) {
708 continue;
709 }

712 cb = area + ld->offset;
713 if (cb && !fd && ld->declType) {

737 continue;
738 }

740 }
741 }
742

744 /* process parent finals */

752 }
753 } else {
754 process_final_procedures(area, (F90_Desc *)parent);
755 }
756 }
757}

1114 __INT_T flag)
1115{

1117 * Assumes destination descriptor, ad, is a full descriptor.
1118 * If flag == 0, then source descriptor, bd, is a scalar "fake" descriptor
1119 * If flag == 1, assume full descriptor for bd
1120 * If flag == 2, assume full descriptor for bd and copy bd into ad.
1121 */
1122
1123 OBJECT_DESC *src = (OBJECT_DESC *)bd;

1126 int dest_is_array, src_is_array, i;

1129 &src_is_array, &dest_td, &src_td, flag);
1130

1132 (!ad || ad->tag != __DESC || !dest_td || dest_td->obj.tag != __POLY))
1133 sz = src_sz;
1134 else if (!src_sz ||
1135 (ad && ad->tag == __DESC && dest_td && dest_td->obj.tag == __POLY &&
1136 (!src_td || src_td->obj.tag != __POLY)))
1137 sz = dest_sz;
1138 else

1140
1141 if (src_td && src_td->obj.size && dest_is_array && !src_is_array) {
1142 for (i = 0; i < dest_sz; i += src_td->obj.size) {
1143 __fort_bcopy(ab + i, bb, src_sz);
1144 }
1145 } else if (src_sz && !flag && ISSCALAR(bd) && bd->tag != __POLY &&
1146 bd->tag < __NTYPES) {
1147 for (i = 0; i < sz; i += src_sz) {
1148 __fort_bcopy(ab + i, bb, src_sz);
1149 }
1150 } else {
1151 __fort_bcopy(ab, bb, sz);
1152 }
1153
1154 if (flag && ad && bd && ad != bd && F90_TAG_G(bd) == __DESC &&
1155 (F90_TAG_G(ad) == __DESC || flag == 2)) {
1156 __fort_bcopy((char *)ad, (char *)bd,
1157 SIZE_OF_RANK_n_ARRAY_DESC(F90_RANK_G(bd)));
1158 SET_F90_DIST_DESC_PTR(ad, F90_RANK_G(ad));
1159 /* check for align-target to self */
1160 if (DIST_ALIGN_TARGET_G(ad) == ad) {
1161 DIST_ALIGN_TARGET_P(ad, ad);
1162 }
1163 } else if (flag > 0 && src_td) {
1164 ENTF90(SET_TYPE, set_type)(ad, src);
1165 dest_td = src_td;
1166 }
1167

1171 sourced_alloc_and_assign(ab, bb, src_td->obj.type);

1173 sourced_alloc_and_assign_array(ad->lsize, ab, bb, dest_td->obj.type);
1174 } else if (dest_is_array) {
1175 sourced_alloc_and_assign_array_from_scalar(ad->lsize, ab, bb,
1176 dest_td->obj.type);
1177 }
1178 }
