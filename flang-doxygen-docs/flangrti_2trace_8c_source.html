50/*
51 * this routine prints a message and terminates the process
52 * sv values
53 * 0 - no traceback
54 * 1 - traceback (other errors)
55 * 2 - traceback (perror)
56 * 3 - traceback (signal)
57 */
58

60#include <process.h> // for _getpid, _exit
61#define getpid _getpid
62#define _Exit _exit
63#endif

65void
66__abort(int sv, const char *msg)
67{
68 char cmd[128];
69 char *p;
70 int n;
71 const char * dbg_env = "F90_TERM_DEBUG";
72 const char * dbg_cmd = "gdb -p %d";
73
74 if (msg != NULL) {
75 fprintf(__io_stderr(), "Error: %s\n", msg);
76 }
77 dbg_stop_before_exit(); /* for the debugger */
78 if (sv == 0) { /* nothing more to do */
79 exit(127);
80 }
81 fflush(__io_stderr());
82 if (tracopt & T_DEBUG) {
83 p = getenv(dbg_env);
84 p = (char *)(p == NULL ? dbg_cmd : p);
85 sprintf(cmd, p, getpid());
86 system(cmd);
87 } else if (tracopt & T_TRACE) {
88 if (sv == 3) /* don't display "abort" routines */
89 n = 2;
90 else if (sv == 2)
91 n = 2;
92 else
93 n = 1;
94 __abort_trace(n);
95 }
96 if (tracopt & T_TEST) {
97 if (sv == 3) /* don't display "abort" routines */
98 n = 3;
99 else if (sv == 2)
100 n = 3;
101 else
102 n = 2;
103 __abort_trace(n);
104 }
105 if (tracopt & T_ABORT) { /* try for core file */
106 signal(SIGABRT, SIG_DFL);
107 abort();
108 }
109 _Exit(127);
110}

126 * this routine is called from the initialization

129void
130__abort_init(char *path)
131{
132 char *p;
133 struct opts *op;
134 int n;
135 int neg;
136
137#if defined(_WIN64)
138 fn = path;
139#endif
140 p = getenv("TRACE_TERM");
141 if (p != NULL) {
142 while (*p != '\0') {
143 if (strncmp(p, "no", 2) == 0) {
144 neg = 1;
145 p += 2;
146 } else {
147 neg = 0;
148 }
149 op = opts;
150 while (1) {
151 if (op->opt == NULL) {
152 fprintf(__io_stderr(), "Error: TRACE_TERM invalid value\n");
153 exit(127);
154 }
155 n = strlen(op->opt);
156 if (strncmp(p, op->opt, n) == 0) {
157 break;
158 }
159 op++;
160 }
161 if (neg == 0) {
162 tracopt |= op->val;
163 } else {
164 tracopt &= ~op->val;
165 }
166 p += n;
167 if (*p == ',') {
168 p++;
169 } else if (*p == '\0') {
170 break;
171 } else {
172 fprintf(__io_stderr(), "Error: TRACE_TERM invalid value\n");
173 exit(127);
174 }
175 }
176 }
177 if (tracopt & T_SIGNAL) {
178 __abort_sig_init();
179 }
180}
