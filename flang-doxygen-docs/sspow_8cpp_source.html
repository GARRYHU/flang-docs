156/*
157 * Before returning see if we need to set any of the processor
158 * exception flags.
159 *
160 * Domain error: a is negative, and b is a finite noninteger
161 * we need to raise the Invalid-Operation flag. This can be done by
162 * taking the square root of a negative number.
163 *
164 * Pole error: a is zero and b is negative we need to raise the
165 * divide by zero flag. This can be done by dividing by zero.
166 */
167
168 if (a_is_neg_flag && (!b_is_integer_flag)) {
169 __m128 volatile invop = _mm_sqrt_ps(a);
170 }
171
172 if (a_is_any_zero_flag && b_is_lt_zero_flag) {
173 __m128 volatile divXzero = _mm_div_ps(ONE_F_VEC,ZERO_F_VEC);
174 }
175
176
177 return _mm_cvtss_f32(res_exp);
