20void
21basenam(const char *orig_path, const char *optional_suffix, char *basename)
22{
23 const char *fn = strrchr(orig_path, '/');
24 size_t length;
25
26 if (fn == NULL)
27 fn = orig_path;
28 else
29 ++fn;
30 length = strlen(fn);
31
32 if (optional_suffix != NULL) {
33 size_t suffix_length = strlen(optional_suffix);
34 if (suffix_length >= length &&
35 strcmp(fn + length - suffix_length, optional_suffix) == 0)
36 length -= suffix_length;
37 }
38
39 memcpy(basename, fn, length);
40 basename[length] = '\0';
41}

43void
44dirnam(const char *orig_path, char *dirname)
45{
46 const char *slash = strrchr(orig_path, '/');
47 if (slash == NULL) {
48 strcpy(dirname, "./");
49 } else if (slash == orig_path) {
50 strcpy(dirname, "/");
51 } else {
52 size_t length = slash - orig_path;
53 memcpy(dirname, orig_path, length);
54 dirname[length] = '\0';
55 }
56}

58int
59fndpath(const char *target, char *path, size_t max_length, const char *dirlist)
60{
61 size_t target_length = target ? strlen(target) : 0;
62 if (target_length == 0)
63 return -1;
64
65 /* The legacy fndpath supplies a default dirlist of '.', which seems
66 * unsafe.
67 */
68 if (dirlist == NULL || !*dirlist)
69 dirlist = ".";
70
71 while (*dirlist != '\0') {
72 const char *end = strchr(dirlist, ':');
73 size_t component_length = end ? end - dirlist : strlen(dirlist);
74 while (component_length > 1 &&
75 dirlist[component_length - 1] == '/') {
76 /* ignore trailing '/', unless it's the only character */
77 --component_length;
78 }
79 if (component_length > 0 &&
80 component_length + 1 /* '/' */ + target_length + 1 <= max_length) {
81 char *p = path;
82 memcpy(p, dirlist, component_length);
83 p += component_length;
84 *p++ = '/';
85 memcpy(p, target, target_length);
86 p[target_length] = '\0';
87 if (access(path, 0) == 0)
88 return 0; /* path exists */
89 }
90 if (end == NULL)
91 break;
92 dirlist = end + 1;
93 }
94
95 return -1;
96}
