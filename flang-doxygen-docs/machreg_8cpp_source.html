105/** \brief Initialize mach_reg structs and reg array. This is done for each
106 * function (subprogram)
107 */
108void
109mr_init()
110{
111 int i;
112
113 aux.curr_entry->first_dr = reg[RATA_IR].mach_reg->first_global;
114 aux.curr_entry->first_sp = reg[RATA_SP].mach_reg->first_global;
115 aux.curr_entry->first_dp = reg[RATA_DP].mach_reg->first_global;
116 aux.curr_entry->first_ar = reg[RATA_AR].mach_reg->first_global;
117
118 for (i = 0; i < MR_UNIQ; i++) {
119 mach_reg[i].next_global = mach_reg[i].first_global;
120 mach_reg[i].nused = 0;
121 }
122
123 for (i = 0; i <= RATA_RTYPES_ACTIVE; i++) {
124 reg[i].nused = 0;
125 reg[i].rcand = 0;
126 }
127
128 /* for pic code, we need to reserve %ebx -- treat it like it
129 * has already been assigned. Since it is register #1, this
130 * is not too difficult.
131 */
132 if (XBIT(62, 8)) {
133 mach_reg[0].next_global++;
134 mach_reg[0].nused = 1;
135 reg[RATA_IR].nused = 1;
136 reg[RATA_AR].nused = 1;
137 }
138
139}

168/** \brief get a global register for a given register type (RATA_IR, etc.).
169 * NOTE that the global registers are allocated in increasing order.
170 * next_global locates the next available global register. The range
171 * of global register values is from first_global to last_global,
172 * inclusive.
173 */
174int
175mr_getreg(int rtype)
176{
177 int rg;
178
179 rg = _mr_getreg(rtype);
180
181 return rg;
182}

184static int
185_mr_getreg(int rtype)
186{
187 register MACH_REG *mr;
188
189 if (reg[rtype].nused >= reg[rtype].max)
190 return NO_REG;
191
192 mr = reg[rtype].mach_reg;
193 if (mr->next_global > mr->last_global)
194 return NO_REG;
195
196 if (BIH_SMOVE(gbl.entbih) && mr->next_global > 1)
197 return NO_REG;
198
199 /* currently, only allow more than one floating point
200 * global register if an xbit is set.
201 */
202 if ((rtype == RATA_SP || rtype == RATA_DP || rtype == RATA_CSP ||
203 rtype == RATA_CDP) &&
204 (!XBIT(4, 0x4) || ratb.mexits))
205 if (mr->next_global > mr->first_global)
206 return NO_REG;
207
208 /* floating point globals need to always start from fp2 (fp1 is
209 * by convention where the return value of fp functions is placed)
210 * and then increment for each inner loop being processed. Thus,
211 * the nused field records the largest number of fp registers
212 * assigned to any loop. This is done differently for the I386
213 * fp as opposed to the I386 integers or any other register set
214 * (due primarily to the fact that the fp registers on x86 are
215 * actually a stack).
216 */
217 if ((rtype != RATA_SP && rtype != RATA_DP && rtype != RATA_CSP &&
218 rtype != RATA_CDP) ||
219 (mr->next_global - mr->first_global + 1 > mr->nused)) {
220 reg[rtype].nused++;
221 mr->nused++;
222 }
223 return (mr->next_global++);
224}
225
226/** \brief map a register type and global register number to an index value in
227 * the range 0 .. MR_NUMGLB-1, taking into consideration that certain
228 * register types map to the same machine register set.
229 *
230 * This is used by * the optimizer to index into its register history table.
231 */
232int
233mr_gindex(int rtype, int regno)
234{
235 MACH_REG *mr = reg[rtype].mach_reg;
236 return ((regno - mr->first_global) + mr->mapbase);
237}

239/** \brief communicate to the scheduler the first global register not assigned
240 * for each register class
241 *
242 * Note that this will be the physical register
243 * number; it reflects the number of registers assigned from the physical
244 * set mapped from the generic register set. Because two or more generic
245 * register sets can map to a single register set, this information
246 * can only be computed after all of the assignments are done.
247 *
248 */
249void
250mr_end()
251{
252 aux.curr_entry->first_dr += reg[RATA_IR].mach_reg->nused;
253 aux.curr_entry->first_ar += reg[RATA_AR].mach_reg->nused;
254 aux.curr_entry->first_sp += reg[RATA_SP].mach_reg->nused;
255 aux.curr_entry->first_dp += reg[RATA_DP].mach_reg->nused;
256
257}

268/** \brief Initialize for scanning the entire machine register set used for
269 * rtype.
270 *
271 * This mechanism for retrieving registers is done when we can no longer
272 * retrieve registers from mr_getreg (we're out of rtype registers).
273 * Ensuing calls to mr_getnext will attempt to retrieve a register
274 * from the set. The assumption is that the caller (optimizer)
275 * will first call mr_reset, and then call mr_getnext one or more
276 * times.
277 */
278void
279mr_reset(int rtype)
280{
281 getnext_reg = reg[rtype].mach_reg->first_global;
282
283 /* if we are generating pic code, we must exclude %ebx as
284 * a potential register.
285 */
286 if ((rtype == RATA_IR || rtype == RATA_AR || rtype == RATA_KR) && XBIT(62, 8))
287 getnext_reg++;
288
289}
290
291/** \func Attempt to retrieve the next available register from the set used
292 * for rtype.
293 *
294 * If one is found, it may be necessary to update the
295 * mach_reg info since we're scanning the entire set. mr_getreg uses a
296 * portion of the set (as defined by the reg structure); things could
297 * get out of sync when registers of different rtypes share the same
298 * register set.
299 */
300
301int
302mr_getnext(int rtype)
303{
304 int rg;
305
306 rg = _mr_getnext(rtype);
307 return rg;
308}
309
310static
311int _mr_getnext(int rtype)
312{
313 int mreg;
314 MACH_REG *mr;
315
316 mr = reg[rtype].mach_reg;
317 if (getnext_reg > mr->last_global)
318 return NO_REG;
319 if (BIH_SMOVE(gbl.entbih) && mr->next_global > 1)
320 return NO_REG;
321
322 if ((rtype == RATA_SP || rtype == RATA_DP || rtype == RATA_CSP ||
323 rtype == RATA_CDP) &&
324 (!XBIT(4, 0x4) || ratb.mexits))
325 if (getnext_reg > mr->first_global)
326 return NO_REG;
327
328 mreg = getnext_reg;
329 getnext_reg++;
330 if (mreg >= mr->next_global) {
331 mr_restore = true;
332 mr_restore_nused = mr->nused;
333 mr_restore_next_global = mr->next_global;
334 /* same comment as in _mr_getreg */
335 if ((rtype != RATA_SP && rtype != RATA_DP && rtype != RATA_CSP &&
336 rtype != RATA_CDP) ||
337 ((mr->next_global - mr->first_global + 1) > mr->nused))
338 mr->nused++;
339 mr->next_global = getnext_reg;
340 }
341 return mreg;
342}

368 bihx = BIH_NEXT(bihx);

373/** \brief allocate and initialize a RGSET entry. */
374int
375mr_get_rgset()
376{
377 int rgset;
378
379 rgset = rgsetb.stg_avail++;
380 if (rgsetb.stg_avail > MAXRAT)
381 error((error_code_t)7, ERR_Fatal, 0, CNULL, CNULL);
382 NEED(rgsetb.stg_avail, rgsetb.stg_base, RGSET, rgsetb.stg_size,
383 rgsetb.stg_size + 100);
384 if (rgsetb.stg_base == NULL)
385 error((error_code_t)7, ERR_Fatal, 0, CNULL, CNULL);
386
387 RGSET_XR(rgset) = 0;
388
389 return rgset;
390}
