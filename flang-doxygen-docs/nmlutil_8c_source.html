15/** \brief
16 * Return the size of a name list item
17 */
18__BIGINT_T
19I8(siz_of)(NML_DESC *descp)
20{
21 int sz;
22
23 switch (descp->type) {
24 case __DERIVED:
25 sz = descp->len;
26 break;
27 case __STR:
28 sz = descp->len;
29 F90_Desc *sd = get_descriptor(descp);
30 if (sd != NULL) {
31 descp->len = sz = F90_LEN_G(sd);
32 }
33 break;
34 case __NCHAR:
35 sz = descp->len * FIO_TYPE_SIZE(descp->type);
36 break;
37 default:
38 sz = FIO_TYPE_SIZE(descp->type);
39 break;
40 }
41 return sz;
42}

46/** \brief
47 * Return the number of elements in a name list item.
48 */
49int
50nelems_of(NML_DESC *descp)
51{
52 int nelems;
53 int k;
54 __POINT_T *desc_dims;
55 __POINT_T actual_ndims;
56
57 /* count up number of constants ( == 1 unless array): */
58 if (descp->ndims == 0)
59 return 1;
60
61 /* MAX_DIM (30) is the number number we add to actual ndims in semfin.c */
62 if (descp->ndims == MAX_DIM)
63 return 1;
64
65 /* write one at a time */
66 if (descp->ndims == -1 || descp->ndims == -2) { /* for pointer/allocatable */
67 return 1;
68 }
69
70 desc_dims = (__POINT_T *)((char *)descp + sizeof(NML_DESC));
71 nelems = desc_dims[1] - desc_dims[0] + 1;
72 ACTUAL_NDIMS(actual_ndims);
73 for (k = 1; k < actual_ndims; k++)
74 nelems *= (desc_dims[2 * k + 1] - desc_dims[2 * k] + 1);
75
76 return nelems;
77}
