73/** \brief Initiate annotation of the assembly file.
74 *
75 * This is called before code is generated. Augment the current annotation
76 * chain with a sorted list of the new linenumber records (ANNO) for this
77 * module.
78 *
79 * \param ahead head of annotation list
80 * \return new head of annotation list
81 */
82static ANNO *
83annomod_initx(ANNO *ahead)
84{
85 static ANNO *static_ahead;
86 static int lastcnt = 0;
87 static INT file_pos = 0;
88
89 ANNO *aptr, *aptr1, *anptr;
90 int cnt = 0;
91 int cnt1 = 0;
92 int bihd, i, j;
93 int old_line_num;
94 int first_bih;
95 char *p;
96
97 /* *************************************************** *
98 * loop thru all of the ili blocks and setup line info
99 * *************************************************** */
100
101 /* ********
102 * step 1:
103 * open source file.
104 * count the number of non-zero blocks
105 * allocate and zero first ANNO node.
106 * chain to old ANNO chain.
107 */
108
109 if (modcnt == 0) {
110 if ((fanno = fopen(SOURCE_FILE, "rb")) == NULL) {
111 error((error_code_t)2, ERR_Warning, 0, SOURCE_FILE, CNULL);
112 return ahead;
113 }
114 } else if (fanno == NULL)
115 return ahead; /* don't process if src not open */
116
117 for (bihd = BIH_NEXT(0), cnt = 0; bihd; bihd = BIH_NEXT(bihd)) {
118 if (BIH_LINENO(bihd))
119 cnt++;
120 }
121
122 if (cnt == 0)
123 return NULL;
124
125 if (cnt > lastcnt) { /* get a new list after freeing old list */
126 if (static_ahead)
127 FREE(static_ahead);
128 NEW(static_ahead, ANNO, cnt);
129 BZERO(static_ahead, ANNO, cnt);
130 lastcnt = cnt;
131 } else /* Just zero out the current chain */
132 BZERO(static_ahead, ANNO, lastcnt);
133
134 anptr = aptr = ahead = static_ahead;
135
136 /* anptr should point to tail of list */
137
138 /* ********
139 * step 2:
140 * walk BIH list and chain non-zero lineno blocks
141 * into ANNO chain. There are 'cnt' non-0 lineno
142 * blocks.
143 */
144 aptr1 = aptr;
145 first_bih = -1;
146
147 for (bihd = BIH_NEXT(0), cnt1 = 0; bihd; bihd = BIH_NEXT(bihd)) {
148
149 if (BIH_LINENO(bihd)) {
150 if (first_bih == -1)
151 first_bih = bihd;
152
153 /* per flyspray 15590, check if this is inline block, get line number
154 from fihb.stg_base[].lineno. Inliner has kept a good record of
155 line numbers. Otherwise, an error lines too long will occurs
156 because p = NULL, a return from get_anno_line() when line number
157 in BIH_LINENO(bihd) is larger than line number in compile file.
158 Note: Include does not yet work correctly especially for c++ .
159 */
160 if (FIH_INL(BIH_FINDEX(bihd))) {
161 /* Go to main source file of this */
162 for (j = BIH_FINDEX(bihd); j != 0; j = FIH_PARENT(j)) {
163 if (FIH_PARENT(j) == 1) {
164 aptr1->lineno = FIH_LINENO(j);
165 break;
166 }
167 }
168 } else {
169 aptr1->lineno = BIH_LINENO(bihd);
170 }
171
172 aptr1->bihd = bihd;
173 aptr1->module = modcnt;
174 aptr1->tmp = cnt1;
175 aptr1->next = NULL;
176 aptr1++;
177 cnt1++;
178 }
179 }
180
181 assert(cnt == cnt1, "annomod_init: cnt != cnt1 :", cnt1, ERR_Severe);
182
183/* ********
184 * step 3:
185 * Sort the newest list into ascending lineno.
186 */
187
188#if DEBUG
189 if (DBGBIT(16, 4)) {
190 fprintf(gbl.dbgfil,
191 "annomod_init: Pre QSORT Linear dump of anno recs: cnt: %d\n", cnt);
192 dmp_all_anno(ahead, gbl.dbgfil, cnt);
193 }
194#endif
195
196 qsort((char *)aptr, cnt, sizeof(ANNO), qs_anno);
197
198#if DEBUG
199 if (DBGBIT(16, 4)) {
200 fprintf(gbl.dbgfil,
201 "annomod_init: Post QSORT Linear dump of anno recs: cnt: %d\n",
202 cnt);
203 dmp_all_anno(ahead, gbl.dbgfil, cnt);
204 }
205#endif
206 /* ********
207 * step 3a:
208 * Now go back and link the ANNO records according to index occurence.
209 */
210 {
211 int idx, i;
212
213 aptr = ahead;
214 aptr1 = NULL;
215 for (i = 0, idx = 0; i < cnt; i++) {
216
217 idx = find_idx(aptr, cnt, i, idx);
218 if (i == 0)
219 ahead = &aptr[idx];
220
221 assert(idx <= cnt && idx >= 0, "outside anno loop: i:", i, ERR_Informational);
222
223 if (aptr1 == NULL)
224 aptr1 = &aptr[idx];
225
226 aptr1->next = &aptr[idx];
227 aptr1 = &aptr[idx];
228 aptr1->next = NULL;
229 }
230 }
231
232#if DEBUG
233 if (DBGBIT(16, 4)) {
234 fprintf(gbl.dbgfil, "annomod_init3: Linked Dump of anno recs: cnt: %d\n",
235 cnt);
236 dmp_all_anno(ahead, gbl.dbgfil, 0);
237 }
238#endif
239
240 /* ********:
241 * step 4:
242 * At this point:
243 * o ahead points a NULL terminated linked list;
244 * o aptr points at the latest contiguous list of sorted
245 * (by ascending lineno) ANNO records.
246 * o The 'next' link chains the new ANNO chain in the order
247 * of their bih occurrence.
248 * o cnt is the number of linenumbers in aptr list.
249 * aptr[0..cnt-1]
250 *
251 * Now, get file locations for each line by streaming thru
252 * source file and recording file position for each line
253 * that corresponds to a BIH LINENO.
254 * When completed, restore file and curlin to the current
255 * positions.
256 */
257
258 if (lanno.curlin < 1)
259 lanno.curlin = 1;
260 file_pos = ftell(fanno);
261 old_line_num = lanno.curlin;
262
263 for (i = 0; i < cnt; i++) {
264 int l;
265
266 l = aptr[i].lineno;
267
268 /* per flyspray 15590, keep line the same is it is include file */
269 if (FIH_INC(BIH_FINDEX(aptr[i].bihd))) {
270 l = lanno.curlin;
271 }
272
273 while (lanno.curlin <= l) {
274 if (lanno.curlin < l) {
275 p = get_anno_line(fanno);
276 if (p == NULL) {
277 flg.anno = 0;
278 return NULL;
279 }
280 }
281
282 if (lanno.curlin == l) {
283 if (lanno.curpos < 0) {
284#if DEBUG
285 interr("annomod_init(): cannot ftell into source for line:",
286 lanno.curlin, ERR_Warning);
287#endif
288 return NULL;
289 }
290 aptr[i].fileloc = (long)lanno.curpos;
291 aptr[i].fd = fanno;
292 aptr[i].filename = SOURCE_FILE; /* Not alloc'd for now */
293 break;
294 }
295 }
296 }
297
298 fseek(fanno, file_pos, 0);
299 lanno.curlin = old_line_num;
300 lanno.curpos = file_pos = ftell(fanno);
301
302#if DEBUG
303 if (DBGBIT(16, 4)) {
304 fprintf(gbl.dbgfil, "annomod_init4: Dump of Linked anno recs: cnt:%d\n",
305 cnt);
306 dmp_all_anno(ahead, gbl.dbgfil, 0);
307 fprintf(gbl.dbgfil, "annomod_init4: Dump of linear anno recs: cnt: %d\n",
308 cnt);
309 dmp_all_anno(aptr, gbl.dbgfil, cnt);
310 }
311#endif
312
313 return aptr;
314} /* endroutine annomod_init */

316/** \brief Quicksort comparison routine for annotations.
317 *
318 * Compares source file line numbers that the blocks correspond to.
319 *
320 * \param p1 first block
321 * \param p2 second block
322 *
323 * Note: parameters are void* because that is required by qsort.
324 */
325static int
326qs_anno(const void *p1, const void *p2)
327{
328 if (((const ANNO *)p1)->lineno < ((const ANNO *)p2)->lineno)
329 return (-1);
330 if (((const ANNO *)p1)->lineno > ((const ANNO *)p2)->lineno)
331 return (1);
332#ifdef HOST_MSDOS
333 return 0;
334#else
335 return (-1); /* compare equality as being 'less than' */
336#endif
337} /* endroutine qs_anno */

351/** \brief Annotate assembly file
352 *
353 * Just seek to file position indicated and output from there to next file
354 * position or eof.
355 */
356extern void
357annomod_asm(int blkno)
358{
359 int l;
360 char *p;
361
362 if (BIH_LINENO(blkno) == 0 || amod == NULL)
363 return;
364
365 /* amod should pt to record for the code that begins with
366 this bih. annomod_asm should output code from this
367 node to the *(amod+1) if amod->next is not null. */
368
369 if (amod->bihd != blkno) {
370#if DEBUG
371 interr("Inconsistent anno records for blkno: ", blkno, ERR_Informational);
372#endif
373 flg.anno = 0;
374 return;
375 }
376
377again:
378 lanno.curpos = ftell(fanno);
379 if (lanno.curpos < 0) {
380 interr("annomod_asm(): cannot ftell into source for module:", modcnt, ERR_Warning);
381 amod = NULL;
382 return;
383 }
384
385 if (lanno.curpos != amod->fileloc) {
386 if (fseek(fanno, (long)amod->fileloc, 0)) {
387 interr("annomod_asm(): cannot fseek into source for lineno:",
388 amod->lineno, ERR_Warning);
389 amod = NULL;
390 return;
391 }
392
393 goto again;
394 }
395
396 lanno.curlin = amod->lineno;
397
398 if (amod->next == NULL)
399 l = lanno.curlin + 1; /* Only go to the next line if last block */
400 else {
401 if ((amod + 1)->lineno == 0)
402 l = lanno.curlin + 1;
403 else
404 l = (amod + 1)->lineno;
405 }
406
407 emit_str("\n", 0);
408 while (lanno.curlin < l) {
409 p = get_anno_line(fanno);
410 if (p == NULL)
411 break;
412 put_anno_line(p);
413 if (lanno.curlin == l)
414 break;
415 }
416
417 amod = amod->next;
418} /* endroutine annomod_asm() */

441/** \brief Get a line from the file
442 *
443 * \param fptr file pointer
444 *
445 * \return ptr to static file buffer if enough room or return NULL if problem
446 * encountered.
447 */
448static char *
449get_anno_line(FILE *fptr)
450{
451#define BLEN 80
452 char *p, *p1;
453 char lbuf[BLEN + 2];
454
455 lanno.buf[0] = '\n';
456 lanno.buf[ALEN - 1] = lanno.buf[ALEN - 2] = lanno.buf[ALEN - 3] = ' ';
457 lanno.buf[1] = '\0';
458
459 p = fgets(lanno.buf, ALEN - 2, fptr);
460 if (p && (int)strlen(lanno.buf) >= ALEN - 3 && lanno.buf[ALEN - 4] != '\n') {
461 strcat(lanno.buf, "...\n");
462 while (1) {
463 p1 = fgets(lbuf, BLEN - 2, fptr);
464 if (p1 && (int)strlen(lbuf) >= BLEN - 3 && lbuf[BLEN - 4] != '\n')
465 continue;
466 else
467 break;
468 }
469 }
470
471 if (p)
472 lanno.curlin++;
473
474 lanno.curpos = ftell(fptr);
475 return p;
476
477} /* endroutine get_anno_line */

493static int
494find_idx(ANNO *p, int cnt, /* # of records in linear list */
495 int idx, /* idx to search for in idx field */
496 int last_idx /* last idx found */
497 )
498{
499 int i;
500
501 for (i = last_idx; i < cnt; i++) {
502 if (p[i].tmp == idx)
503 return i;
504 }
505
506 /* if can't find forward, start from beginning and research */
507 for (i = 0; i < cnt; i++) {
508 if (p[i].tmp == idx)
509 return i;
510 }
511
512 return -1;
513} /* endroutine find_idx */
