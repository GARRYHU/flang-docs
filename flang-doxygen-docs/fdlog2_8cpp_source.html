20// casts int to double
21inline
22__m128d __internal_fast_int2dbl(__m128i a)
23{
24 __m128i const INT2DBL_HI = _mm_set1_epi64x(INT2DBL_HI_D);
25 __m128i const INT2DBL_LO = _mm_set1_epi64x(INT2DBL_LO_D);
26 __m128d const INT2DBL = (__m128d)_mm_set1_epi64x(INT2DBL_D);
27
28 __m128i t = _mm_xor_si128(INT2DBL_LO, a);
29 t = _mm_blend_epi32(INT2DBL_HI, t, 0x5);
30 return _mm_sub_pd((__m128d)t, INT2DBL);
31}

33// special cases for log
34__m128d __attribute__ ((noinline)) __pgm_log_d_vec128_special_cases(__m128d const a, __m128d z)
35{
36 __m128d const ZERO = _mm_set1_pd(ZERO_D);
37 __m128i const ALL_ONES_EXPONENT = _mm_set1_epi64x(ALL_ONES_EXPONENT_D);
38 __m128d const NAN_VAL = (__m128d)_mm_set1_epi64x(NAN_VAL_D);
39 __m128d const NEG_INF = (__m128d)_mm_set1_epi64x(NEG_INF_D);
40
41
42 __m128i detect_inf_nan = (__m128i)_mm_sub_pd(a, a);
43 __m128d inf_nan_mask = (__m128d)_mm_cmpeq_epi64(_mm_and_si128(detect_inf_nan, ALL_ONES_EXPONENT), ALL_ONES_EXPONENT);
44
45 // inf + inf = inf = log(inf). nan + nan = nan = log(nan).
46 __m128i inf_nan = (__m128i)_mm_add_pd(a, a);
47 z = _mm_blendv_pd(z, (__m128d)inf_nan, inf_nan_mask);
48
49 __m128d non_positive_mask = _mm_cmp_pd(a, ZERO, _CMP_LT_OQ);
50 // log(negative number) = NaN
51 z = _mm_blendv_pd(z, NAN_VAL, non_positive_mask);
52
53 // log(0) = -inf
54 __m128d zero_mask = _mm_cmp_pd(a, ZERO, _CMP_EQ_OQ);
55 z = _mm_blendv_pd(z, NEG_INF, zero_mask);
56
57 return z;
58}

104 __m128d a_mut, m, f;
105 __m128i expo, expo_plus1;
106 __m128d thresh_mask;
107
108 // isolate mantissa
109 a_mut = _mm_and_pd(a, HI_CONST_1);
110 a_mut = _mm_or_pd(a_mut, HI_CONST_2);
111
112 // magic trick to improve accuracy (divide mantissa by 2 and increase exponent by 1)
113 thresh_mask = _mm_cmp_pd(a_mut, (__m128d)HI_THRESH, _CMP_GT_OS);
114 m = (__m128d)_mm_sub_epi32((__m128i)a_mut, HALFIFIER);
115 m = _mm_blendv_pd(a_mut, m, thresh_mask);
116
117 // compute exponent
118 expo = _mm_srli_epi64((__m128i)a, D52_D);
119 expo = _mm_sub_epi64(expo, TEN_23);
120 expo_plus1 = _mm_add_epi64(expo, ONE);
121 expo = (__m128i)_mm_blendv_pd((__m128d)expo, (__m128d)expo_plus1, thresh_mask);
122
123 // computing polynomial for log(1+m)
124 m = _mm_sub_pd(m, ONE_F);
125
126 // estrin scheme for highest 16 terms, then estrin again for the next 8. Finally finish off with horner.
127 __m128d z9 = _mm_fmadd_pd(LOG_C10_VEC, m, LOG_C9_VEC);
128 __m128d z11 = _mm_fmadd_pd(LOG_C12_VEC, m, LOG_C11_VEC);
129 __m128d z13 = _mm_fmadd_pd(LOG_C14_VEC, m, LOG_C13_VEC);
130 __m128d z15 = _mm_fmadd_pd(LOG_C16_VEC, m, LOG_C15_VEC);
131 __m128d z17 = _mm_fmadd_pd(LOG_C18_VEC, m, LOG_C17_VEC);
132 __m128d z19 = _mm_fmadd_pd(LOG_C20_VEC, m, LOG_C19_VEC);
133 __m128d z21 = _mm_fmadd_pd(LOG_C22_VEC, m, LOG_C21_VEC);
134 __m128d z23 = _mm_fmadd_pd(LOG_C24_VEC, m, LOG_C23_VEC);
135
136 __m128d m2 = _mm_mul_pd(m, m);
137 z9 = _mm_fmadd_pd(z11, m2, z9);
138 z13 = _mm_fmadd_pd(z15, m2, z13);
139 z17 = _mm_fmadd_pd(z19, m2, z17);
140 z21 = _mm_fmadd_pd(z23, m2, z21);
141
142 __m128d m4 = _mm_mul_pd(m2, m2);
143 z9 = _mm_fmadd_pd(z13, m4, z9);
144 z17 = _mm_fmadd_pd(z21, m4, z17);
145
146 __m128d m8 = _mm_mul_pd(m4, m4);
147 z9 = _mm_fmadd_pd(z17, m8, z9);
148
149 // estrin for the next 8 terms
150 __m128d z8 = _mm_fmadd_pd(z9, m, LOG_C8_VEC);
151 __m128d z6 = _mm_fmadd_pd(LOG_C7_VEC, m, LOG_C6_VEC);
152 __m128d z4 = _mm_fmadd_pd(LOG_C5_VEC, m, LOG_C4_VEC);
153 __m128d z2 = _mm_fmadd_pd(LOG_C3_VEC, m, LOG_C2_VEC);
154
155 z6 = _mm_fmadd_pd(z8, m2, z6);
156 z2 = _mm_fmadd_pd(z4, m2, z2);
157 __m128d z = _mm_fmadd_pd(z6, m4, z2);
158
159 // finish computation with horner
160 z = _mm_fmadd_pd(z, m, LOG_C1_VEC);
161 z = _mm_mul_pd(z, m);
162
163 f = __internal_fast_int2dbl(expo);
164 z = _mm_fmadd_pd(f, LN2_HI, z);
165
166 // compute special cases (inf, NaN, negative, 0)
167 __m128i detect_inf_nan = (__m128i)_mm_sub_pd(a, a);
168 __m128i detect_non_positive = (__m128i)_mm_cmp_pd(a, ZERO, _CMP_LE_OQ);
169 __m128i inf_nan_mask = _mm_cmpeq_epi64(_mm_and_si128(detect_inf_nan, ALL_ONES_EXPONENT), ALL_ONES_EXPONENT);
170
171#if defined(TARGET_LINUX_POWER)
172 int specMask = _vec_any_nz((__m128i)_mm_or_si128(detect_non_positive, inf_nan_mask));
173#else
174 int specMask = _mm_movemask_pd((__m128d)_mm_or_si128(detect_non_positive, inf_nan_mask));
175#endif
176 if(__builtin_expect(specMask, 0)) {
177 return __pgm_log_d_vec128_special_cases(a, z);
178 }
179 return z;
