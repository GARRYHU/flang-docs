28{

42/** @brief returns results of executing CPUID with function cpuid_func and

48 *
49 * Returns false(0): if cpuid_func not supported
50 * true(1): CPUID successfully executed with cpuid_func+ecx and:

55int
56__pgi_cpuid_ecx(uint32_t f, uint32_t *r, uint32_t c)
57{
58 if (__pgi_cpuid_getmax(f) == 0) return 0;

60 : "=a"(r[0]), "=b"(r[1]), "=c"(r[2]), "=d"(r[3])
