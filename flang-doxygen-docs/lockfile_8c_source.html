8/*
9 * Four routines to be able to lock access to a file in the Linux
10 * approved manner. This seems to be the only method that works
11 * over NFS:
12 *
13 * __pg_make_lock_file( char* dir )
14 * Creates a uniquely-named file in the given directory.
15 * To work, this must be created on the same filesystem
16 * as the file which we are attempting to lock.
17 * If there are multiple processes running each competing for the
18 * same lock, each gets a unique file here.
19 * return -1 if it can't get a unique file in 50 tries
20 * __pg_get_lock( char* lname )
21 * The argument is the name of the lock.
22 * Each process tries to create a hard link with this name
23 * to its own uniquely-named file from __pg_make_lock_file().
24 * The one that succeeds is the new lock owner. The others
25 * fail and try again. There is a fail-over to handle the case
26 * where the process with the lock dies, which is inherently unsafe,
27 * but we haven't come up with a better solution.
28 * __pg_release_lock( char* lname )
29 * The argument is the same name for the lock.
30 * The lock is released by deleting (calling unlink) for the
31 * hard link we had just created.
32 * __pg_delete_lock_file()
33 * Clean up by deleting the uniquely named file we had created earlier.
34 */

66int
67__pg_make_lock_file(char *dir)
68{
69 pid_t pid;
70 char hostname[999];
71 int r;
72 FILE *F;
73 udir = (char *)malloc(strlen(dir) + 1);
74 strcpy(udir, dir);
75 pid = getpid();
76 r = gethostname(hostname, 998);
77 if (r != 0) {
78 fprintf(stderr, "gethostname fails\n");
79 exit(1);
80 }
81 uname = (char *)malloc(strlen(hostname) + strlen(udir) + 25);
82 r = 0;
83 do {
84 ++r;
85 /* create a unique file */
86 sprintf(uname, "%s/lock.%s.%ld%d", udir, hostname, (long)pid, r);
87 F = fopen(uname, "w");
88 } while (F == NULL && r < 51);
89 if (F == NULL) {
90 return -1;
91 }
92 fprintf(F, "%s\n", uname);
93 fclose(F);
94 uwaiting = 0;
95 return 0;
96} /* __pg_make_lock_file */

98void
99__pg_get_lock(char *lname)
100{
101 int r;
102 struct stat ss, ss1;
103 int count;
104 char *fullname;
105 if (udir == NULL || uname == NULL)
106 return;
107 fullname = (char *)malloc(strlen(lname) + strlen(udir) +
108 2); /* +1 for / +1 for null terminator */
109 strcpy(fullname, udir);
110 strcat(fullname, "/");
111 strcat(fullname, lname);
112 memset(&ss1, 0, sizeof(ss1));
113 count = 0;
114 do {
115 r = link(uname, fullname);
116 if (r != 0) {
117 /* link had some problem, see if the number of links
118 * to the new file is now two */
119 r = stat(uname, &ss);
120 if (r == 0) {
121 if (ss.st_nlink != 2) {
122 if (ss.st_nlink != 1) {
123 fprintf(stderr, "get_lock: %d links to %s\n", (int)ss.st_nlink,
124 uname);
125 exit(1);
126 }
127 /* see if we should forcefully TAKE the lock */
128 r = lstat(fullname, &ss);
129 if (count == 0) {
130 memcpy(&ss1, &ss, sizeof(ss));
131 count = 1;
132 } else if (ss1.st_dev != ss.st_dev || ss1.st_ino != ss.st_ino ||
133 ss1.st_mode != ss.st_mode || ss1.st_uid != ss.st_uid ||
134 ss1.st_gid != ss.st_gid || ss1.st_size != ss.st_size ||
135 ss1.st_mtime != ss.st_mtime) {
136 /* some else got there first */
137 memcpy(&ss1, &ss, sizeof(ss));
138 count = 1;
139 } else {
140 ++count;
141 if (count > 20) {
142 /* we've waited long enough. */
143 r = unlink(fullname);
144 /* ignore errors on the unlink */
145 }
146 }
147 sleep(1);
148 r = 1;
149 }
150 }
151 if (r)
152 ++uwaiting;
153 }
154 } while (r != 0);
155 free(fullname);
156} /* __pg_get_lock */

158void
159__pg_release_lock(char *lname)
160{
161 int r;
162 char *fullname;
163 if (udir == NULL || uname == NULL)
164 return;
165 fullname = (char *)malloc(strlen(lname) + strlen(udir) + 2);
166 strcpy(fullname, udir);
167 strcat(fullname, "/");
168 strcat(fullname, lname);
169 /* release the lock by deleting the link */
170 r = unlink(fullname);
171 if (r != 0) {
172 fprintf(stderr, "release_lock: unlink %s fails\n", lname);
173 exit(1);
174 }
175 free(fullname);
176} /* __pg_release_lock */

189 free(udir);
190 uname = NULL;
191 udir = NULL;
192} /* __pg_delete_lock_file */
