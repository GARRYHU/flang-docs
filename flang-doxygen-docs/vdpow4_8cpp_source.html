34// double-double "fma"
35inline
36__m256d_2 __internal_ddfma (__m256d_2 x, __m256d_2 y, __m256d_2 z)
37{
38 __m256d e;
39 __m256d_2 t, m;
40 t.y = _mm256_mul_pd(x.y, y.y);
41 t.x = _mm256_fmsub_pd(x.y, y.y, t.y);
42 t.x = _mm256_fmadd_pd(x.y, y.x, t.x);
43 t.x = _mm256_fmadd_pd(x.x, y.y, t.x);
44
45 m.y = _mm256_add_pd (z.y, t.y);
46 e = _mm256_sub_pd (z.y, m.y);
47 m.x = _mm256_add_pd (_mm256_add_pd(_mm256_add_pd (e, t.y), t.x), z.x);
48
49 return m;
50}

78// slowpath for exp. used to improve accuracy on larger inputs
79__m256d __attribute__ ((noinline)) __pgm_exp_d_vec256_slowpath(__m256i const i, __m256d const t, __m256d const bloga, __m256d z, __m256d prodx)
80{
81 __m256d const UPPERBOUND_1 = (__m256d)_mm256_set1_epi64x(UPPERBOUND_1_D);
82 __m256d const UPPERBOUND_2 = (__m256d)_mm256_set1_epi64x(UPPERBOUND_2_D);
83 __m256d const ZERO = _mm256_set1_pd(ZERO_D);
84 __m256d const INF = (__m256d)_mm256_set1_epi64x(INF_D);
85 __m256i const HI_ABS_MASK = _mm256_set1_epi64x(HI_ABS_MASK_D);
86 __m256i const ABS_MASK = _mm256_set1_epi64x(ABS_MASK_D);
87 __m256i const MULT_CONST = _mm256_set1_epi64x(MULT_CONST_D);
88
89 __m256i abs_bloga = _mm256_and_si256((__m256i)bloga, ABS_MASK);
90 __m256d abs_lt = (__m256d)_mm256_and_si256((__m256i)abs_bloga, HI_ABS_MASK);
91 __m256d lt_zero_mask = _mm256_cmp_pd(bloga, ZERO, _CMP_LT_OS); // compute bloga < 0.0
92 __m256d slowpath_mask = _mm256_cmp_pd(abs_lt, UPPERBOUND_1, _CMP_LT_OS);
93
94 __m256d a_plus_inf = _mm256_add_pd(bloga, INF); // check if bloga is too big
95 __m256d zero_inf_blend = _mm256_blendv_pd(a_plus_inf, ZERO, lt_zero_mask);
96
97 __m256d accurate_scale_mask = (__m256d)_mm256_cmp_pd(abs_lt, UPPERBOUND_2, 1);
98
99 // compute accurate scale
100 __m256i k = _mm256_srli_epi64(i, 1); // k = i / 2
101 __m256i i_scale_acc = _mm256_slli_epi64(k, D52_D); // shift to HI and shift 20
102 k = _mm256_sub_epi32(i, k); // k = i - k
103 __m256i i_scale_acc_2 = _mm256_slli_epi64(k, D52_D); // shift to HI and shift 20
104 __m256d multiplier = (__m256d)_mm256_add_epi64(i_scale_acc_2, MULT_CONST);
105 multiplier = _mm256_blendv_pd(ZERO, multiplier, accurate_scale_mask); // quick fix for overflows in case they're being trapped
106
107 __m256d res = (__m256d)_mm256_add_epi32(i_scale_acc, (__m256i)t);
108 res = _mm256_mul_pd(res, multiplier);
109
110 __m256d slowpath_blend = _mm256_blendv_pd(zero_inf_blend, res, accurate_scale_mask);
111 z = _mm256_blendv_pd(slowpath_blend, z, slowpath_mask);
112
113 __m256i isinf_mask = _mm256_cmpeq_epi64((__m256i)INF, (__m256i)z); // special case for inf
114 __m256d z_fixed = _mm256_fmadd_pd(z, prodx, z); // add only if not inf
115 z = _mm256_blendv_pd(z_fixed, z, (__m256d)isinf_mask);
116
117 return z;
118}

120// special cases for pow
121// implementation derived from cudart/device_functions_impl.c
122__m256d __attribute__ ((noinline)) __pgm_pow_d_vec256_special_cases(__m256d const a, __m256d const b, __m256d t)
123{
124 __m256i const HI_MASK = _mm256_set1_epi64x(HI_MASK_D);
125 __m256d const SGN_EXP_MASK = (__m256d)_mm256_set1_epi64x(SGN_EXP_MASK_D);
126 __m256i const ABS_MASK = _mm256_set1_epi64x(ABS_MASK_D);
127 __m256d const NEG_ONE = _mm256_set1_pd(NEG_ONE_D);
128 __m256d const ZERO = _mm256_set1_pd(ZERO_D);
129 __m256d const ONE = _mm256_set1_pd(ONE_D);
130 __m256d const HALF = _mm256_set1_pd(HALF_D);
131 __m256d const SGN_MASK = (__m256d)_mm256_set1_epi64x(SGN_MASK_D);
132 __m256d const INF = (__m256d)_mm256_set1_epi64x(INF_D);
133 __m256i const INF_FAKE = _mm256_set1_epi64x(INF_FAKE_D);
134 __m256d const NAN_MASK = (__m256d)_mm256_set1_epi64x(NAN_MASK_D);
135 __m256d const NEG_ONE_CONST = (__m256d)_mm256_set1_epi64x(NEG_ONE_CONST_D);
136
137 __m256i const TEN_23 = _mm256_set1_epi64x(TEN_23_D);
138 __m256i const ELEVEN = _mm256_set1_epi64x(ELEVEN_D);
139
140 __m256i a_sign, b_sign, bIsOddInteger;
141 __m256d abs_a, abs_b;
142 __m256i shiftb;
143
144 a_sign = (__m256i)_mm256_and_pd(a, SGN_MASK);
145 b_sign = (__m256i)_mm256_and_pd(b, SGN_MASK);
146 abs_a = _mm256_and_pd(a, (__m256d)ABS_MASK);
147 abs_b = _mm256_and_pd(b, (__m256d)ABS_MASK);
148
149 // determining if b is an odd integer, since there are special cases for it
150 shiftb = (__m256i)_mm256_and_pd (b, SGN_EXP_MASK);
151 shiftb = _mm256_srli_epi64(shiftb, D52_D);
152 shiftb = _mm256_sub_epi64(shiftb, TEN_23);
153 shiftb = _mm256_add_epi64(shiftb, ELEVEN);
154
155 __m256i b_is_half = _mm256_cmpeq_epi64((__m256i)abs_b, (__m256i)HALF);
156 bIsOddInteger = _mm256_sllv_epi64((__m256i)b, shiftb);
157 bIsOddInteger = _mm256_cmpeq_epi64((__m256i)SGN_MASK, bIsOddInteger);
158 // fix for b = +/-0.5 being incorrectly identified as an odd integer
159 bIsOddInteger = _mm256_andnot_si256(b_is_half, bIsOddInteger);
160
161 // corner cases where a <= 0
162 // if ((ahi < 0) && bIsOddInteger)
163 __m256d ahi_lt_0 = (__m256d)_mm256_cmpeq_epi64(a_sign, (__m256i)ZERO);
164 __m256d ahilt0_and_boddint_mask = _mm256_andnot_pd(ahi_lt_0, (__m256d)bIsOddInteger);
165
166 t = _mm256_blendv_pd(t, neg(t), ahilt0_and_boddint_mask);
167
168 // else if ((ahi < 0) && (b != trunc(b)))
169 __m256d b_ne_trunc = _mm256_cmp_pd(b, _mm256_round_pd(b, _MM_FROUND_TO_ZERO), _CMP_NEQ_UQ);
170 __m256d nan_mask = _mm256_andnot_pd(ahi_lt_0, b_ne_trunc);
171 t = _mm256_blendv_pd(t, NAN_MASK, nan_mask);
172
173 // if (a == 0.0)
174 __m256d a_is_0_mask = (__m256d)_mm256_cmpeq_epi64((__m256i)abs_a, (__m256i)ZERO);
175 __m256d thi_when_ais0 = ZERO;
176
177 // if (bIsOddInteger && a == 0.0)
178 thi_when_ais0 = _mm256_blendv_pd(thi_when_ais0, (__m256d)a, (__m256d)bIsOddInteger);
179
180 // if (bhi < 0 && a == 0.0)
181 __m256d bhi_lt_0 = (__m256d)_mm256_cmpeq_epi64(b_sign, (__m256i)ZERO); //this mask is inverted
182 __m256d thi_or_INF = _mm256_or_pd(thi_when_ais0, INF);
183 thi_when_ais0 = _mm256_blendv_pd(thi_or_INF, thi_when_ais0, bhi_lt_0);
184 __m256d t_when_ais0 = _mm256_and_pd(thi_when_ais0, (__m256d)HI_MASK);
185
186 t = _mm256_blendv_pd(t, t_when_ais0, a_is_0_mask);
187
188 // else if (a is INF)
189 __m256d a_inf_mask = (__m256d)_mm256_cmpeq_epi64((__m256i)abs_a, (__m256i)INF);
190 // use bhi_lt_0 backwards to evaluate bhi >= 0
191 __m256d thi_when_aisinf = _mm256_blendv_pd(thi_when_aisinf, INF, bhi_lt_0);
192
193 // now evaluate ((ahi < 0) && bIsOddInteger)
194 __m256d thi_xor_sgn = _mm256_xor_pd(thi_when_aisinf, SGN_MASK);
195 thi_when_aisinf = _mm256_blendv_pd(thi_when_aisinf, thi_xor_sgn, ahilt0_and_boddint_mask);
196
197 t = _mm256_blendv_pd(t, thi_when_aisinf, a_inf_mask);
198
199 // else if (abs(b) is INF)
200 __m256d b_inf_mask = (__m256d)_mm256_cmpeq_epi64((__m256i)abs_b, (__m256i)INF);
201 __m256d thi_when_bisinf = ZERO;
202 __m256d absa_gt_one = _mm256_cmp_pd(abs_a, ONE, _CMP_GT_OS); // evaluating (abs(a) > 1)
203 thi_when_bisinf = _mm256_blendv_pd(thi_when_bisinf, INF, absa_gt_one);
204
205 __m256d thi_xor_inf = _mm256_xor_pd(thi_when_bisinf, INF);
206 thi_when_bisinf = _mm256_blendv_pd(thi_xor_inf, thi_when_bisinf, bhi_lt_0); // bhi < 0
207
208 __m256d a_is_negone = (__m256d)_mm256_cmpeq_epi64((__m256i)a, (__m256i)NEG_ONE);
209 thi_when_bisinf = _mm256_blendv_pd(thi_when_bisinf, NEG_ONE_CONST, a_is_negone); //a == -1
210
211 t = _mm256_blendv_pd(t, thi_when_bisinf, b_inf_mask);
212
213 // if(a is NAN || B is NAN) <=> !(a is a number && b is a number)
214 __m256i a_nan_mask = _mm256_cmpeq_epi64(_mm256_max_epu32((__m256i)abs_a, INF_FAKE), INF_FAKE);
215 __m256i b_nan_mask = _mm256_cmpeq_epi64(_mm256_max_epu32((__m256i)abs_b, INF_FAKE), INF_FAKE);
216 __m256i aorb_nan_mask = _mm256_and_si256(a_nan_mask, b_nan_mask); //this mask is inverted
217 t = _mm256_blendv_pd(_mm256_add_pd(a,b), t, (__m256d)aorb_nan_mask);
218
219 // if a == 1 or b == 0, answer = 1
220 __m256d a_is_one_mask = (__m256d)_mm256_cmpeq_epi64((__m256i)a, (__m256i)ONE);
221 __m256d b_is_zero_mask = (__m256d)_mm256_cmpeq_epi64((__m256i)abs_b, (__m256i)ZERO);
222 __m256d ans_equals_one_mask = _mm256_or_pd(a_is_one_mask, b_is_zero_mask);
223 t = _mm256_blendv_pd(t, ONE, ans_equals_one_mask);
224 // ******************************************************************************************
225
226
227 __m256i a_is_neg = (__m256i)_mm256_cmp_pd(a, ZERO, _CMP_LT_OS);
228 int a_is_neg_flag = _mm256_movemask_epi8((__m256i)a_is_neg);
229
230 __m256i b_is_integer = (__m256i)_mm256_cmp_pd(b, _mm256_floor_pd(b), _CMP_EQ_OQ);
231 int b_is_integer_flag = _mm256_movemask_epi8((__m256i)b_is_integer);
232
233 __m256i b_is_lt_zero = (__m256i)_mm256_cmp_pd(b, ZERO, _CMP_LT_OS);
234 int b_is_lt_zero_flag = _mm256_movemask_epi8((__m256i)b_is_lt_zero);
235
236 __m256d const MINUS_ZERO = _mm256_set1_pd((double)-0.0);
237 __m256i a_is_pos_zero = _mm256_cmpeq_epi64( (__m256i)a, (__m256i)ZERO);
238 __m256i a_is_neg_zero = _mm256_cmpeq_epi64( (__m256i)a, (__m256i)MINUS_ZERO);
239 __m256i a_is_any_zero = _mm256_or_si256(a_is_pos_zero, a_is_neg_zero);
240 int a_is_any_zero_flag = _mm256_movemask_epi8((__m256i)a_is_any_zero);
241
242/*
243 * Before returning see if we need to set any of the processor
244 * exception flags.
245 *
246 * Domain error: a is negative, and b is a finite noninteger
247 * we need to raise the Invalid-Operation flag. This can be done by
248 * taking the square root of a negative number.
249 *
250 * Pole error: a is zero and b is negative we need to raise the
251 * divide by zero flag. This can be done by dividing by zero.
252 */
253
254 if (a_is_neg_flag && (!b_is_integer_flag)) {
255 __m256d volatile invop = _mm256_sqrt_pd(a);
256 }
257
258 if (a_is_any_zero_flag && b_is_lt_zero_flag) {
259 __m256d volatile divXzero = _mm256_div_pd(ONE,ZERO);
260 }
261
262
263
264
265
266
267
268
269
270
271
272
273 return t;
274}

276__m256d __fvd_pow_fma3_256(__m256d const a, __m256d const b)
277{
278 //////////////////////////////////////////////////// log constants
279 __m256d const LOG_POLY_6 = _mm256_set1_pd(LOG_POLY_6_D);
280 __m256d const LOG_POLY_5 = _mm256_set1_pd(LOG_POLY_5_D);
281 __m256d const LOG_POLY_4 = _mm256_set1_pd(LOG_POLY_4_D);
282 __m256d const LOG_POLY_3 = _mm256_set1_pd(LOG_POLY_3_D);
283 __m256d const LOG_POLY_2 = _mm256_set1_pd(LOG_POLY_2_D);
284 __m256d const LOG_POLY_1 = _mm256_set1_pd(LOG_POLY_1_D);
285 __m256d const LOG_POLY_0 = _mm256_set1_pd(LOG_POLY_0_D);
286
287 __m256d const CC_CONST_Y = _mm256_set1_pd(CC_CONST_Y_D);
288 __m256d const CC_CONST_X = _mm256_set1_pd(CC_CONST_X_D);
289
290 __m256i const EXPO_MASK = _mm256_set1_epi64x(EXPO_MASK_D);
291 __m256d const HI_CONST_1 = (__m256d)_mm256_set1_epi64x(HI_CONST_1_D);
292 __m256d const HI_CONST_2 = (__m256d)_mm256_set1_epi64x(HI_CONST_2_D);
293 __m256i const HALFIFIER = _mm256_set1_epi64x(HALFIFIER_D);
294 __m256i const HI_THRESH = _mm256_set1_epi64x(HI_THRESH_D); //~sqrt(2)
295 __m256d const ONE_F = _mm256_set1_pd(ONE_F_D);
296 __m256d const TWO = _mm256_set1_pd(TWO_D);
297 __m256d const ZERO = _mm256_set1_pd(ZERO_D);
298
299 __m256d const LN2_HI = _mm256_set1_pd(LN2_HI_D);
300 __m256d const LN2_LO = _mm256_set1_pd(LN2_LO_D);
301 //////////////////////////////////////////////////////////////////
302
303 //////////////////////////////////////////////////// exp constants
304 __m256d const L2E = _mm256_set1_pd(L2E_D);
305 __m256d const NEG_LN2_HI = _mm256_set1_pd(NEG_LN2_HI_D);
306 __m256d const NEG_LN2_LO = _mm256_set1_pd(NEG_LN2_LO_D);
307
308 __m256d const EXP_POLY_B = _mm256_set1_pd(EXP_POLY_B_D);
309 __m256d const EXP_POLY_A = _mm256_set1_pd(EXP_POLY_A_D);
310 __m256d const EXP_POLY_9 = _mm256_set1_pd(EXP_POLY_9_D);
311 __m256d const EXP_POLY_8 = _mm256_set1_pd(EXP_POLY_8_D);
312 __m256d const EXP_POLY_7 = _mm256_set1_pd(EXP_POLY_7_D);
313 __m256d const EXP_POLY_6 = _mm256_set1_pd(EXP_POLY_6_D);
314 __m256d const EXP_POLY_5 = _mm256_set1_pd(EXP_POLY_5_D);
315 __m256d const EXP_POLY_4 = _mm256_set1_pd(EXP_POLY_4_D);
316 __m256d const EXP_POLY_3 = _mm256_set1_pd(EXP_POLY_3_D);
317 __m256d const EXP_POLY_2 = _mm256_set1_pd(EXP_POLY_2_D);
318 __m256d const EXP_POLY_1 = _mm256_set1_pd(EXP_POLY_1_D);
319 __m256d const EXP_POLY_0 = _mm256_set1_pd(EXP_POLY_0_D);
320
321 __m256d const DBL2INT_CVT = _mm256_set1_pd(DBL2INT_CVT_D);
322
323 __m256d const UPPERBOUND_1 = (__m256d)_mm256_set1_epi64x(UPPERBOUND_1_D);
324 __m256i const HI_ABS_MASK = _mm256_set1_epi64x(HI_ABS_MASK_D);
325
326 //////////////////////////////////////////////////// pow constants
327 __m256i const HI_MASK = _mm256_set1_epi64x(HI_MASK_D);
328 __m256d const SGN_EXP_MASK = (__m256d)_mm256_set1_epi64x(SGN_EXP_MASK_D);
329 __m256i const ABS_MASK = _mm256_set1_epi64x(ABS_MASK_D);
330 __m256i const ONE = _mm256_set1_epi64x(ONE_D);
331
332 __m256i const TEN_23 = _mm256_set1_epi64x(TEN_23_D);
333 __m256i const ALL_ONES_EXPONENT = _mm256_set1_epi64x(ALL_ONES_EXPONENT_D);
334
335
336 __m256d abs_a, a_mut;
337 __m256d_2 loga;
338 __m256d t_hi, t_lo, tmp, e, bloga, prodx;
339
340 __m256d_2 qq, cc, uu, tt;
341 __m256d f, g, u, v, q, ulo, m;
342 __m256i ihi, expo, expo_plus1;
343 __m256d thresh_mask;
344 __m256d b_is_one_mask;
345
346 /*
347 * Check whether all exponents in vector b are 1.0, and if so take
348 * a quick exit.
349 * Note: b_is_one_mask is needed later in case some elements in b are 1.0.
350 */
351
352 b_is_one_mask = _mm256_cmp_pd(b, ONE_F, _CMP_EQ_OQ);
353 if (_mm256_movemask_pd(b_is_one_mask) == 0xf) {
354 return a;
355 }
356
357 // *****************************************************************************************
358 // computing log(abs(a))
359 abs_a = _mm256_and_pd(a, (__m256d)ABS_MASK);
360 a_mut = _mm256_and_pd(abs_a, HI_CONST_1);
361 a_mut = _mm256_or_pd(a_mut, HI_CONST_2);
362 m = (__m256d)_mm256_sub_epi32((__m256i)a_mut, HALFIFIER); // divide by 2
363
364 ihi = _mm256_and_si256((__m256i)a_mut, HI_MASK);
365 thresh_mask = _mm256_cmp_pd((__m256d)ihi, (__m256d)HI_THRESH,_CMP_GT_OS);
366 m = _mm256_blendv_pd(a_mut, m, thresh_mask);
367
368 expo = _mm256_srli_epi64((__m256i)abs_a, D52_D);
369 expo = _mm256_sub_epi64(expo, TEN_23);
370 expo_plus1 = _mm256_add_epi64(expo, ONE); // add one to exponent instead
371 expo = (__m256i)_mm256_blendv_pd((__m256d)expo, (__m256d)expo_plus1, thresh_mask);
372
373 // begin computing log(m)
374 f = _mm256_sub_pd(m, ONE_F);
375 g = _mm256_add_pd(m, ONE_F);
376 g = _mm256_div_pd(ONE_F, g);
377
378 u = _mm256_mul_pd(_mm256_mul_pd(TWO, f), g);
379
380 // u = 2.0 * (m - 1.0) / (m + 1.0)
381 v = _mm256_mul_pd(u, u);
382
383 // polynomial is used to approximate atanh(v)
384 // an estrin evaluation scheme is used.
385 __m256d c0 = _mm256_fmadd_pd(LOG_POLY_1, v, LOG_POLY_0);
386 __m256d c2 = _mm256_fmadd_pd(LOG_POLY_3, v, LOG_POLY_2);
387 __m256d c4 = _mm256_fmadd_pd(LOG_POLY_5, v, LOG_POLY_4);
388 __m256d v2 = _mm256_mul_pd(v, v);
389 __m256d v4 = _mm256_mul_pd(v2, v2);
390
391 c0 = _mm256_fmadd_pd(c2, v2, c0);
392 c4 = _mm256_fmadd_pd(LOG_POLY_6, v2, c4);
393 q = _mm256_fmadd_pd(c4, v4, c0);
394 q = _mm256_mul_pd(q, v);
395
396 tmp = _mm256_mul_pd(TWO, _mm256_sub_pd(f, u));
397 tmp = _mm256_fmadd_pd(neg(u), f, tmp);
398 ulo = _mm256_mul_pd(g, tmp);
399
400 // double-double computation begins
401 qq.y = q;
402 qq.x = ZERO;
403 uu.y = u;
404 uu.x = ulo;
405 cc.y = CC_CONST_Y;
406 cc.x = CC_CONST_X;
407
408 qq = __internal_ddadd_yisdouble(cc, qq);
409
410 // computing log(m) in double-double format
411 cc.y = _mm256_mul_pd(uu.y, uu.y);
412 cc.x = _mm256_fmsub_pd(uu.y, uu.y, cc.y);
413 cc.x = _mm256_fmadd_pd(uu.y,
414 (__m256d)_mm256_add_epi32((__m256i)uu.x, HALFIFIER),
415 cc.x); // u ** 2
416
417 tt.y = _mm256_mul_pd(cc.y, uu.y);
418 tt.x = _mm256_fmsub_pd(cc.y, uu.y, tt.y);
419 tt.x = _mm256_fmadd_pd(cc.y, uu.x, tt.x);
420 tt.x = _mm256_fmadd_pd(cc.x, uu.y, tt.x); // u ** 3 un-normalized
421
422 uu = __internal_ddfma(qq, tt, uu);
423
424 // computing log a = log(m) + log(2)*expo
425 f = __internal_fast_int2dbl (expo);
426 q = _mm256_fmadd_pd(f, LN2_HI, uu.y);
427 tmp = _mm256_fmadd_pd(neg(f), LN2_HI, q);
428 tmp = _mm256_sub_pd(tmp, uu.y);
429 loga.y = q;
430 loga.x = _mm256_sub_pd(uu.x, tmp);
431 loga.x = _mm256_fmadd_pd(f, LN2_LO, loga.x);
432
433 // finish log(a)
434 // ******************************************************************************************
435
436 // compute b * log(a)
437 t_hi = _mm256_mul_pd(loga.y, b);
438 t_lo = _mm256_fmsub_pd(loga.y, b, t_hi);
439 t_lo = _mm256_fmadd_pd(loga.x, b, t_lo);
440 bloga = e = _mm256_add_pd(t_hi, t_lo);
441 prodx = _mm256_add_pd(_mm256_sub_pd(t_hi, e), t_lo);
442
443 // ***************************************************************************************
444 // computing exp(b * log(a))
445 // calculating exponent; stored in the LO of each 64-bit block
446 __m256i i = (__m256i) _mm256_fmadd_pd(bloga, L2E, DBL2INT_CVT);
447
448 // calculate mantissa
449 __m256d t = _mm256_sub_pd ((__m256d)i, DBL2INT_CVT);
450 __m256d z = _mm256_fmadd_pd (t, NEG_LN2_HI, bloga);
451 z = _mm256_fmadd_pd (t, NEG_LN2_LO, z);
452
453 // use polynomial to calculate exp
454 // mixed estrin/horner scheme: estrin on the higher 8 coefficients, horner on the lowest 4.
455 // provided speedup without loss of precision compared to full horner
456 __m256d t4 = _mm256_fmadd_pd(EXP_POLY_5, z, EXP_POLY_4);
457 __m256d t6 = _mm256_fmadd_pd(EXP_POLY_7, z, EXP_POLY_6);
458 __m256d t8 = _mm256_fmadd_pd(EXP_POLY_9, z, EXP_POLY_8);
459 __m256d t10 = _mm256_fmadd_pd(EXP_POLY_B, z, EXP_POLY_A);
460
461 __m256d z2 = _mm256_mul_pd(z, z);
462 __m256d z4 = _mm256_mul_pd(z2, z2);
463
464 t4 = _mm256_fmadd_pd(t6, z2, t4);
465 t8 = _mm256_fmadd_pd(t10, z2, t8);
466 t4 = _mm256_fmadd_pd(t8, z4, t4);
467
468 t = _mm256_fmadd_pd(t4, z, EXP_POLY_3);
469 t = _mm256_fmadd_pd(t, z, EXP_POLY_2);
470 t = _mm256_fmadd_pd(t, z, EXP_POLY_1);
471 t = _mm256_fmadd_pd(t, z, EXP_POLY_0);
472
473 // fast scale
474 __m256i i_scale = _mm256_slli_epi64(i,D52_D);
475 __m256d ztemp = z = (__m256d)_mm256_add_epi32(i_scale, (__m256i)t);
476
477 // slowpath detection for exp
478 __m256d abs_bloga = (__m256d)_mm256_and_si256((__m256i)bloga, HI_ABS_MASK);
479 int exp_slowmask = _mm256_movemask_pd(_mm256_cmp_pd(abs_bloga, UPPERBOUND_1, _CMP_GE_OS));
480
481 z = _mm256_fmadd_pd(z, prodx, z);
482
483 if (__builtin_expect(exp_slowmask, 0)) {
484 z = __pgm_exp_d_vec256_slowpath(i, t, bloga, ztemp, prodx);
485 }
486 // finished exp(b * log (a))
487 // ************************************************************************************************
488
489 // Now special case if some elements of exponent (b) are 1.0.
490 z = _mm256_blendv_pd(z, a, b_is_one_mask);
491
492 // compute if we have special cases (inf, nan, etc). see man pow for full list of special cases
493 __m256i detect_inf_nan = (__m256i)_mm256_add_pd(a, b); // check for inf/nan
494 __m256i overridemask = _mm256_cmpeq_epi64( (__m256i)a, (__m256i)ONE_F); // if a == 1
495 __m256i overridemask2 = _mm256_cmpeq_epi64(_mm256_and_si256(detect_inf_nan, ALL_ONES_EXPONENT), ALL_ONES_EXPONENT);
496 overridemask = _mm256_or_si256(overridemask, (__m256i)_mm256_cmp_pd(a, ZERO, _CMP_LE_OQ)); // if a < 0
497
498 int specMask = _mm256_movemask_pd((__m256d)_mm256_or_si256(overridemask, overridemask2));
499 if(__builtin_expect(specMask, 0)) {
500 return __pgm_pow_d_vec256_special_cases(a, b, z);
501 }
502 return z;
503}
