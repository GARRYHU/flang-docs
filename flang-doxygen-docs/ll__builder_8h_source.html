19/*
20 * LLMD_Builder -- Builder for metadata nodes.
21 *
22 * AN LLMD_Builder can be used to construct a single LLVM metadata node by
23 * adding one element at a time. A set of llmd_add_* functions are provided for
24 * adding different types of elements to the node.
25 *
26 * The LLMD_Builder struct is opaque. Allocate one with llmd_init() and destroy
27 * it with llmd_finish().
28 */
29typedef struct LLMD_Builder_ *LLMD_Builder;
30
31#define LL_MD_IS_NULL(MD) (!(MD))
