49/** \brief A routine to find the loops in a flow graph
50 */
51void
52findloop(int hlopt_bv)
53{
54 int edge, lp, head, tail, max_level, i;
55 LP *p;
56 int exit;
57 int pt;
58 bool precedes;
59 PSI_P q;
60#ifndef FE90
61 int save_rgset0, *save_rtemps0;
62#endif
63 bool any_malformed;
64
65 if (OPTDBG(9, 8))
66 fprintf(gbl.dbgfil, "\n---------- findloop trace for function \"%s\"\n",
67 getprint(BIH_LABEL(gbl.entbih)));
68
69#if DEBUG
70 if (XBIT(6, 0x100000))
71 lp_topsort = 1;
72#endif
73
74 opt.nloops = 0;
75 opt.lpb.stg_avail = 1;
76/* Optimizer sets two fields in opt.lpb.stg_avail[0] before findloop is called.
77 * Save and restore those */
78#ifndef FE90
79 save_rgset0 = LP_RGSET(0);
80 save_rtemps0 = LP_RTEMPS(0);
81#endif
82 STG_CLEAR(opt.lpb, 0);
83#ifndef FE90
84 LP_RGSET(0) = save_rgset0;
85 LP_RTEMPS(0) = save_rtemps0;
86 LP_HEAD(0) = BIH_TO_FG(gbl.entbih);
87 LP_TAIL(0) = BIH_TO_FG(opt.exitbih);
88#else
89 LP_HEAD(0) = BIH_TO_FG(gbl.entbih);
90 LP_TAIL(0) = opt.exitfg;
91#endif
92 LP_CNCALL(0) = 1;
93 LP_TAIL_AEXE(0) = 1;
94
95 if (NUM_RTE == 0)
96 goto build_region0;
97 /*
98 * go through the retreating edges and find the back edges using the
99 * dominance requirement. Note that a non-back edge is denoted by
100 * by a 0 successor field.
101 */
102 for (edge = 0; edge < NUM_RTE; edge++) {
103 if (!is_dominator((int)EDGE_SUCC(edge), (int)EDGE_PRED(edge))) {
104 if (OPTDBG(9, 8))
105 fprintf(gbl.dbgfil, "edge %d (%d, %d) is not back edge\n", edge,
106 EDGE_PRED(edge), EDGE_SUCC(edge));
107 EDGE_SUCC(edge) = 0;
108 }
109 }
110 /*
111 * Now, those back edges which have identical heads are linked together
112 * (i.e., the natural loops of the edges with the same head are combined
113 * into a single loop). Look ahead in the table of edges. If an edge
114 * with the same "head" is found, link the current edge to it. Also,
115 * the "tail" with the maximum dfn is computed which will become
116 * the tail of the loop; this ensures that if "head==tail", we have a
117 * one block loop. The list will be used to find the other edges
118 * (actually the extra tails) so that the build of the loop is done
119 * correctly.
120 *
121 * NOTE that a back edge which is added to the list will have its
122 * successor field set to zero.
123 */
124 for (edge = 0; edge < NUM_RTE; edge++) {
125 head = EDGE_SUCC(edge);
126 if (head == 0)
127 continue; /* not a back edge or already processed */
128 tail = EDGE_PRED(edge); /* tail with maximum dfn */
129 if (OPTDBG(9, 8))
130 fprintf(gbl.dbgfil, "back edge %d, (%d %d)\n", edge, tail, head);
131 for (i = edge + 1; i < NUM_RTE; i++) {
132 /*
133 * as edges with the same head are found, ensure that the
134 * edges' tails lexically follow their respective heads.
135 * NOTE that this only checks the multiple edges which follow
136 * this edge. Single edges and the first edge of the multiple
137 * case are checked in the final scan of the edges.
138 */
139 precedes = false;
140 if (head == EDGE_SUCC(i)) {
141 if (OPTDBG(9, 8))
142 fprintf(gbl.dbgfil, " add tail %d to edge\n", EDGE_PRED(i));
143 if (EDGE_PRED(i) < EDGE_SUCC(i)) {
144 if (OPTDBG(9, 8))
145 fprintf(gbl.dbgfil,
146 " mult.edge %d (%d %d), tail precedes head\n", i,
147 EDGE_PRED(i), EDGE_SUCC(i));
148 precedes = true;
149 }
150 EDGE_NEXT(i) = EDGE_NEXT(edge);
151 EDGE_NEXT(edge) = i;
152 if (FG_DFN(tail) < FG_DFN(EDGE_PRED(i))) {
153 if (OPTDBG(9, 8))
154 fprintf(gbl.dbgfil, " tail exchange: old %d, new %d\n", tail,
155 EDGE_PRED(i));
156 /* exchange predecessors (tails) of the two edges */
157 tail = EDGE_PRED(i); /* new maximum */
158 EDGE_PRED(i) = EDGE_PRED(edge);
159 EDGE_PRED(edge) = tail;
160 }
161 EDGE_SUCC(i) = 0;
162 }
163 if (precedes) {
164 if (OPTDBG(9, 8))
165 fprintf(gbl.dbgfil,
166 " multiple edge %d (%d %d), tail precedes head\n", edge,
167 tail, EDGE_SUCC(edge));
168 EDGE_SUCC(edge) = 0;
169 }
170 }
171 }
172 /*
173 * Go through the back edges and create an entry in the loop table;
174 * don't allow those whose tail lexically precedes the head.
175 */
176 any_malformed = false;
177 max_level = 0;
178 for (edge = 0; edge < NUM_RTE; edge++) {
179 head = EDGE_SUCC(edge);
180 if (head == 0)
181 continue;
182 tail = EDGE_PRED(edge);
183 if (head <= tail) {
184 opt.nloops = STG_NEXT(opt.lpb);
185 p = opt.lpb.stg_base + opt.nloops;
186 BZERO(p, LP, 1);
187 p->head = head;
188 p->tail = tail;
189 p->level = 1;
190 p->exits = PSI_P_NULL;
191 p->flags.bits.innermost = 1; /* loop is innermost unless
192 * proven otherwise */
193 p->flags.bits.cncall = 1; /* loop is cncall unless
194 * proven otherwise */
195 p->edge = edge;
196
197#ifndef FE90
198 /*
199 * allocate an area for the loop's register used information and
200 * the rtemps information - zero them out.
201 */
202 p->rused = (RUSED *)getitem(RUSED_AREA, MR_NUMGLB * sizeof(RUSED));
203 BZERO(p->rused, RUSED, MR_NUMGLB);
204 p->rtemps = (int *)getitem(RUSED_AREA, RTEMPS * sizeof(int));
205 BZERO(p->rtemps, int, RTEMPS);
206#endif
207#if DEBUG
208 if (OPTDBG(9, 8)) {
209 assert(tail == head || !is_dominator(tail, head), "bad dominator rel",
210 head, ERR_Informational);
211 fprintf(gbl.dbgfil, "---LOOPS--- %d is (%d, %d)\n", opt.nloops, tail,
212 head);
213 }
214#endif
215 } else {
216 if (OPTDBG(9, 8))
217 fprintf(gbl.dbgfil, "edge %d (%d %d), tail precedes.0 head\n", edge,
218 tail, head);
219 any_malformed = true;
220 }
221 /*next_edge: ;*/
222 }
223
224 /*
225 * go through the loops and determine their level and parent values
226 */
227 for (lp = 1; lp <= opt.nloops; lp++) {
228 p = opt.lpb.stg_base + lp;
229 p->exits = PSI_P_NULL;
230 /*
231 * if this a 1 block loop, there is no need to determine if any loops
232 * are contained in it
233 */
234 if ((p->head) == (p->tail)) {
235 for (q = FG_SUCC(p->tail); q != PSI_P_NULL; q = PSI_NEXT(q))
236 if ((exit = PSI_NODE(q)) != p->tail) {
237 add_lpexit(lp, exit);
238 PSI_EXIT(q) = 1;
239 }
240 continue;
241 }
242 /*
243 * otherwise, collect all the blocks which are in the loop -- this
244 * "region" will contain all of the blocks in loops which are
245 * contained in the loop
246 */
247 build_loop(lp);
248 if (OPTDBG(9, 8))
249 fprintf(gbl.dbgfil, "for natural loop %5d\n", lp);
250
251 /* scan natural loop to find if there are multiple exits and
252 * detect any extended range loops.
253 */
254
255 for (i = LP_FG(lp); i != 0; i = FG_NEXT(i)) {
256 if (OPTDBG(9, 8))
257 fprintf(gbl.dbgfil, " - fg %5d bih %5d\n", i, FG_TO_BIH(i));
258 for (q = FG_SUCC(i); q != PSI_P_NULL; q = PSI_NEXT(q)) {
259 if (FG_LOOP(exit = PSI_NODE(q)) != lp) {
260 add_lpexit(lp, exit);
261 PSI_EXIT(q) = 1;
262 }
263 }
264 if (i < LP_HEAD(lp) || i > LP_TAIL(lp)) {
265 if (OPTDBG(9, 8))
266 fprintf(gbl.dbgfil, " - fg %d is not enclosed within loop\n", i);
267 /*
268 fprintf(stderr,
269 "loop %d, edge (%d %d) does not enclose node %d\n",
270 lp, LP_TAIL(lp), LP_HEAD(lp), i);
271 */
272 LP_XTNDRNG(lp) = 1;
273 }
274 }
275 /*
276 * scan through the other loops to determine if they are contained in
277 * the loop lp. This is determined by checking the loop field for
278 * the flow graph nodes of the head and tail of their back edges
279 */
280 for (i = 1; i <= opt.nloops; i++) {
281 if (i == lp)
282 continue;
283
284 /*
285 * if both the head and the tail of the back edge are in the
286 * loop, then this loop is contained in lp
287 */
288 if (FG_LOOP(LP_HEAD(i)) == 0)
289 continue;
290 if (FG_LOOP(LP_TAIL(i)) == 0)
291 continue;
292
293 /*
294 * increment the level of the contained loop. Also, set the
295 * maximum level if necessary
296 */
297 if (max_level < (++LP_LEVEL(i)))
298 max_level = LP_LEVEL(i);
299
300 /*
301 * lp is the parent of loop i if this is the first time or the
302 * level of lp is greater than the level of the current parent of
303 * loop i
304 */
305 if (LP_PARENT(i) == 0 || p->level > LP_LEVEL(LP_PARENT(i))) {
306 LP_PARENT(i) = lp;
307 LP_INNERMOST(lp) = 0;
308 if (OPTDBG(9, 8))
309 fprintf(gbl.dbgfil, " new parent of %d is %d\n", i, lp);
310 }
311 }
312
313 /* cleanup the region built by build_loop */
314 /* mark nodes as unvisited and clear FG_LOOP field */
315 unvisit(naturalloop, 1);
316 LP_FG(lp) = 0;
317 }
318
319 /* sort the loops in innermost order */
320
321 if (opt.nloops)
322 top_sort();
323
324 /*
325 * go through the loops according to their sorted order to build the
326 * regions for each loop. At this time, a loop's region will exclude any
327 * blocks which belong to any enclosed loops
328 */
329 for (i = 1; i <= opt.nloops; ++i) {
330 lp = LP_LOOP(i);
331 /*
332 * add this loop to its parent's sibling list. must do this for
333 * all of the natural loops which were discovered including those
334 * which are ignored or merged with other loops (i.e., their
335 * FG_LOOP fields are zero).
336 */
337 pt = LP_PARENT(lp);
338 LP_SIBLING(lp) = LP_CHILD(pt);
339 LP_CHILD(pt) = lp;
340 /* update the count of the loop; its count may be nonzero since
341 * all of the loops contained by this loop have been counted.
342 */
343 LP_COUNT(lp)++;
344 /* propagate the loop's count to its parent
345 */
346 LP_COUNT(pt) += LP_COUNT(lp);
347
348 /*
349 * only build the region if the head and tail nodes are not members
350 * of another loop
351 */
352 if (FG_LOOP(LP_HEAD(lp)) || FG_LOOP(LP_TAIL(lp)))
353 continue;
354 build_loop(lp);
355
356#if DEBUG
357 if (OPTDBG(9, 8)) {
358 int z;
359 fprintf(gbl.dbgfil, "regions for loop %5d\n", lp);
360 for (z = LP_FG(lp); z != 0; z = FG_NEXT(z))
361 fprintf(gbl.dbgfil, " - fg %5d bih %5d\n", z, FG_TO_BIH(z));
362 }
363#endif
364 /* mark nodes as unvisited , but don't clear FG_LOOP field */
365 unvisit(naturalloop, 0);
366
367 /*
368 * if lp is some type of while loop, convert it to an "if - repeat"
369 * loop
370 */
371
372 if (hlopt_bv & HLOPT_ENDTEST)
373 convert_loop(lp);
374
375 LP_TAIL_AEXE(lp) = is_tail_aexe(lp);
376 }
377 if (any_malformed) {
378 for (edge = 0; edge < NUM_RTE; edge++) {
379 head = EDGE_SUCC(edge);
380 if (head == 0)
381 continue;
382 tail = EDGE_PRED(edge);
383 if (head > tail) {
384 if (OPTDBG(9, 8))
385 fprintf(gbl.dbgfil, "edge %d (%d %d), tail precedes.1 head\n", edge,
386 tail, head);
387 malformed(head, tail);
388 }
389 }
390 }
391
392/* build region 0 */
393
394build_region0:
395 p = opt.lpb.stg_base;
396 p->fg = 0;
397 p->flags.bits.callfg = BIH_EX(gbl.entbih);
398 for (i = opt.dfn; i; i--) {
399 head = VTX_NODE(i);
400 if (FG_LOOP(head) == 0) {
401 FG_NEXT(head) = p->fg;
402 p->fg = head;
403 }
404 }
405#ifndef FE90
406 LP_RUSED(0) = (RUSED *)getitem(RUSED_AREA, MR_NUMGLB * sizeof(RUSED));
407 BZERO(LP_RUSED(0), RUSED, MR_NUMGLB);
408 /*
409 * NOTE: LP_RTEMPS(0) has already been allocated by function_init
410 * in optimize.c
411 */
412
413 /*
414 * Now that loops have been found, we can compute post_dominance
415 * and build the control dependence data structures. Necessary
416 * to find loops first since we identify infinite loops to
417 * insert pseudo edges to the exit node. We need a path from exit
418 * to every node in order to compute the post_dominator tree.
419 */
420 build_apt(0); /* 1 = dominance frontiers, 0 = control dependence */
421#endif
422}

439/** \brief Topological sort
440 *
441 * This routine is used to sort the loops found in the flowgraph such that a
442 * loop X is processed for optimizations before any of the loops containing X.
443 * The algorithm used is the topological sort as defined in Knuth (p. 262), The
444 * Art of Computer Programming, Volume 1/Fundamental Algorithms.
445 */
446static void
447top_sort(void)
448{
449
450#define TOP(i) topb[i].top
451#define COUNT(i) topb[i].count
452#define QLINK(i) topb[i].count
453#define SUCC(p) p->succ
454#define NEXT(p) p->next
455
456 typedef struct topi_tag {
457 short succ;
458 struct topi_tag *next;
459 } TOPI;
460 typedef struct {
461 short count;
462 TOPI *top;
463 } TOPB;
464 TOPB *topb;
465 TOPI *p;
466 int k, r;
467 int lp;
468 int n;
469
470 /*
471 * allocate storage for the top array and intitalize the count and top
472 * fields. There is a need for a top entry for each loop and loop "0",
473 * and one for linking together the entries whose counts are zero. Entry
474 * 0 is reserved for the "level-0" loop which represents the outermost
475 * region of a function and contains all loops. Entries 1 through nloops
476 * are used for the respective loops. Entry n is only used for linking
477 * up those loops which have zero counts (i.e, the innermost loops).
478 */
479 n = opt.nloops + 1;
480 NEW(topb, TOPB, n + 1);
481 for (k = 0; k < n; k++) {
482 TOP(k) = NULL;
483 COUNT(k) = 0;
484 }
485
486 /*
487 * go through the loops and define the partial ordering, lp < k, where k
488 * is the parent of lp (i.e., lp is contained in k). Note that at least
489 * one relation of the form i < 0 exists.
490 */
491 for (lp = 1; lp <= opt.nloops; lp++) {
492 k = LP_PARENT(lp); /* relation lp < k */
493 if (OPTDBG(9, 8))
494 fprintf(gbl.dbgfil, " relation %d < %d\n", lp, k);
495 COUNT(k)++; /* # predecessors of k */
496 p = (TOPI *)getitem(TOPI_AREA, sizeof(TOPI));
497 SUCC(p) = k; /* k is a successor of lp */
498 NEXT(p) = TOP(lp);
499 TOP(lp) = p;
500 }
501
502 /*
503 * Initialize the queue used to hold loops in topological order by
504 * scanning for loops which have no predecessors (loop 0 has at least one
505 * predecessor. The list produced is always non- empty given that there
506 * are loops
507 */
508
509 r = n;
510 for (k = 1; k < n; k++)
511 if (COUNT(k) == 0) {
512 r = QLINK(r) = k;
513 if (OPTDBG(9, 8))
514 fprintf(gbl.dbgfil, " innermost loop %d\n", k);
515 }
516 assert(r != n, "top_sort: wrong qlink", r, ERR_Severe);
517
518 /*
519 * Go through the relations and create the order - this continues until
520 * loop zero is seen (it is always the last one seen)
521 */
522
523 k = 0;
524 for (lp = QLINK(n); lp > 0; lp = QLINK(lp)) {
525 LP_LOOP(++k) = lp;
526 if (OPTDBG(9, 8))
527 fprintf(gbl.dbgfil, " next loop: %d\n", lp);
528 n--;
529 for (p = TOP(lp); p != NULL; p = NEXT(p))
530 if (--COUNT(SUCC(p)) == 0)
531 r = QLINK(r) = SUCC(p);
532 }
533 assert(n == 1, "wrong top_sort", n, ERR_Severe);
534
535 /* free up the top array and the area used for the successors */
536
537 FREE(topb);
538 freearea(TOPI_AREA);
539}

543/** \brief Determine if a node dominates another
544 *
545 * \param v first node
546 * \param w second node
547 * \return true if 'v' dominates 'w'
548 *
549 * Walk up dominator tree from 'w', stop at 'v' (return true) or when we reach
550 * a node above 'v' in the spanning tree.
551 */
552bool
553is_dominator(int v, int w)
554{
555 int vv, vw;
556 vv = FG_DFN(v);
557 if (v == 0)
558 return true;
559 while (w != 0) {
560 if (v == w)
561 return true;
562 /* this test is so complicated because either v or w
563 * may have been added after the depth-first tree was built
564 * and nodes were numbered; if they are both numbered, then
565 * we can stop when we reach a 'w' above 'v' in the spanning tree. */
566 if (vv > 0 && (vw = FG_DFN(w)) > 0 && vw < vv)
567 return false;
568 w = FG_DOM(w);
569 }
570 return false;
571} /* is_dominator */
572
573#if defined(FG_PDOM)
574/*
575 * return true if 'v' postdominates 'w'
576 * walk up postdominator tree from 'w'
577 */
578bool
579is_post_dominator(int v, int w)
580{
581 int vv;
582 for (vv = w; vv > 0; vv = FG_PDOM(vv)) {
583 if (vv == v)
584 return true;
585 }
586 return false;
587} /* is_post_dominator */
588#endif

592/*
593 * Determine if the tail of a loop always executed:
594 */
595bool
596is_tail_aexe(int lp)
597{
598 int fg;
599 int cnt;
600 PSI_P p;
601 if (LP_MEXITS(lp))
602 return false;
603 fg = LP_TAIL(lp);
604 if (LP_HEAD(lp) == fg)
605 return true;
606 if (!BIH_FT(FG_TO_BIH(fg)))
607 /*
608 * the multi-block loop exits from another point in the loop;
609 * therefore, the tail is not always executed.
610 */
611 return false;
612 cnt = 0;
613 for (p = FG_PRED(LP_HEAD(lp)); p != PSI_P_NULL; p = PSI_NEXT(p)) {
614 if (FG_LOOP(PSI_NODE(p)) == lp)
615 cnt++;
616 if (cnt > 1)
617 return false;
618 }
619 return true;
620}

622/** \brief Build the natural loop and loop region given the head and tail flow
623 * graph nodes.
624 *
625 * The natural loop is represented by a linked list headed by the variable
626 * naturalloop. As nodes are visited, they are added to this list.
627 * The natural loop is built by first adding the loop head to the list
628 * and then visiting the tail of the loop and recursively visiting its
629 * predecessors. The natural loop contains the nodes of any nested loops.
630 * Note that if there are other retreating edges whose head values are
631 * identical to the head of the loop, these extra edges are taken into
632 * consideration so that the loops defined by retreating edges with identical
633 * head values are combined into a single loop.
634 *
635 * The loop region is defined to be the set of nodes in the natural loop
636 * which do not belong to any nested loops. The loop region is represented
637 * as a linked list headed by the field LP_FG of the current loop. The nodes
638 * which belong to the current loop region are linked together using the
639 * FG_NEXT fields. The field FG_LOOP records the loop to which it
640 * belongs. Also, the loop's LP_CALLFG and LP_NOBLA fields are set if
641 * the respective criteria are met.
642 *
643 * NOTE that build_loop is used for two purposes:
644 * 1. to aid in determining the level and parent values of the loops
645 * so that the loops can be sorted,
646 * 2. to build the regions of the loops after the loops have been sorted.
647 * For the first case, the natural loop list and loop region list are
648 * identical.
649 */
650static void
651build_loop(int loop)
652{
653 int head;
654 int edge;
655
656 /* begin building the loop by adding the loop head to the region */
657
658 LP_FG(loop) = head = LP_HEAD(loop);
659 LP_CALLFG(loop) = BIH_EX(FG_TO_BIH(head));
660#ifdef BIH_NOBLA
661 LP_NOBLA(loop) = BIH_NOBLA(FG_TO_BIH(head));
662#endif
663 LP_JMP_TBL(loop) = FG_JMP_TBL(head);
664#ifdef FE90
665 LP_PARLOOP(loop) = FG_PAR(head);
666#else
667 LP_PARLOOP(loop) = BIH_PARLOOP(FG_TO_BIH(head));
668#endif
669#ifdef LP_PARALN
670 LP_PARALN(loop) = BIH_PARALN(FG_TO_BIH(head));
671#endif
672
673 /* init linked list of regions assigned to this loop */
674 FG_LOOP(head) = loop;
675 FG_NEXT(head) = 0;
676
677 /* init linked list of regions in this naturalloop */
678 naturalloop = head;
679 FG_NATNXT(head) = 0;
680 FG_VISITED(head) = 1;
681
682 current_lp = loop;
683 current_lp_tail = head;
684 /*
685 * repeat for the tail of the loop and all of its predecessors;
686 * also, include the predecessors of any back edges whose
687 * head values are identical.
688 */
689 add_to_region((int)LP_TAIL(loop));
690 edge = EDGE_NEXT(LP_EDGE(loop));
691 while (edge >= 0) {
692 add_to_region((int)EDGE_PRED(edge));
693 edge = EDGE_NEXT(edge);
694 }
695}

697/** \brief Add a node to the loop region
698 */
699static void
700add_to_region(int v)
701{
702 PSI_P p;
703
704 if (FG_VISITED(v) == 1)
705 return;
706
707 /* add v to nodes in naturalloop */
708 FG_VISITED(v) = 1;
709 FG_NATNXT(v) = naturalloop;
710 naturalloop = v;
711
712 if (lp_topsort) {
713 /* first recurse on non-fall-through edges.
714 * then do the fall-through edges.
715 * this tends to keep the fall-through edges adjacent. */
716 for (p = FG_PRED(v); p != PSI_P_NULL; p = PSI_NEXT(p)) {
717 if (!PSI_FT(p))
718 add_to_region((int)PSI_NODE(p));
719 }
720 for (p = FG_PRED(v); p != PSI_P_NULL; p = PSI_NEXT(p)) {
721 if (PSI_FT(p))
722 add_to_region((int)PSI_NODE(p));
723 }
724 }
725
726 if (FG_LOOP(v) == 0) {
727 /* if no enclosed loop has grabbed this node, */
728 /* add v to current_lp's assigned regions. */
729
730 /* add to linked list of assigned regions */
731 if (lp_topsort) {
732 FG_NEXT(v) = 0;
733 FG_NEXT(current_lp_tail) = v;
734 current_lp_tail = v;
735 } else {
736 FG_NEXT(v) = LP_FG(current_lp);
737 LP_FG(current_lp) = v;
738 }
739
740 FG_LOOP(v) = current_lp;
741 if (BIH_EX(FG_TO_BIH(v)))
742 LP_CALLFG(current_lp) = 1;
743#ifdef BIH_NOBLA
744 if (BIH_NOBLA(FG_TO_BIH(v)))
745 LP_NOBLA(current_lp) = 1;
746#endif
747 if (FG_JMP_TBL(v))
748 LP_JMP_TBL(current_lp) = 1;
749 }
750 /* recurse for the predecessors of v */
751 if (!lp_topsort) {
752 for (p = FG_PRED(v); p != PSI_P_NULL; p = PSI_NEXT(p)) {
753 add_to_region((int)PSI_NODE(p));
754 }
755 }
756
757}

861#ifndef FE90
862/** \brief A routine to find loops whose control flow follows the pattern of a
863 * "while" loop.
864 *
865 * A "while _cond_" loop is changed to "if _cond_ do ... while !_cond_". A
866 * "for" loop is a form of a "while" loop.
867 */
868static void
869convert_loop(int loop)
870{
871 int head, tail;
872 int headbih, tailbih, iltx, newhead, exit;
873 SPTR label, tmpsptr;
874 int tmp, bihx, lastheadbih, fgx;
875 int oldheadbih;
876 int i, br_ilt;
877 PSI_P p, q;
878 int new_tree;
879
880 if (XBIT(6, 0x80000000))
881 return;
882 head = LP_HEAD(loop);
883 headbih = FG_TO_BIH(head);
884 /*
885 * if expander has marked a Doloop as zerotrip or if a previous call
886 * to findloop on this function (i.e., by the vectorizer) has converted
887 * a loop, pass along this info and return.
888 */
889 if (BIH_ZTRP(headbih) == 1) {
890 LP_ZEROTRIP(loop) = 1;
891 return;
892 }
893
894 /* check if head contains <= ILT_THRESH ILTs;
895 * do nothing if the loop only has one block */
896 if (head == (tail = LP_TAIL(loop)))
897 return;
898
899 /* label of head does not exist or it is used more than once */
900
901 if ((label = BIH_LABEL(headbih)) == 0 || RFCNTG(label) != 1) {
902 if (OPTDBG(9, 8))
903 fprintf(gbl.dbgfil, "---convert_loop(%d): bih%d wrong label %d\n", loop,
904 headbih, label);
905 return;
906 }
907/* check if head contains <= ILT_THRESH ILTs */
908/* note that the 'head' may be more than one block in some cases */
909
910#define ILT_THRESH 4
911
912 i = 0;
913 for (bihx = headbih; 1; bihx = BIH_NEXT(bihx)) {
914 lastheadbih = bihx;
915 fgx = BIH_TO_FG(bihx);
916 if (FG_LOOP(fgx) != loop)
917 return;
918 iltx = BIH_ILTFIRST(bihx);
919 if (ILT_DBGLINE(iltx))
920 iltx = ILT_NEXT(iltx);
921 for (tmp = iltx; tmp; tmp = ILT_NEXT(tmp))
922 ++i;
923
924 if (i > ILT_THRESH) {
925 if (OPTDBG(9, 8))
926 fprintf(gbl.dbgfil, "---convert_loop(%d): bih%d > %d ILT\n", loop, bihx,
927 ILT_THRESH);
928 return;
929 }
930 br_ilt = BIH_ILTLAST(bihx);
931 /* if there are two successors of bihx, this is the branch.
932 * if there is one successor of bihx and it is a fall-through,
933 * go around this loop again. */
934 p = FG_SUCC(fgx);
935 if (PSI_NEXT(p) == PSI_P_NULL) {
936 /* only one successor */
937 int target;
938 target = PSI_NODE(p);
939 if (FG_LOOP(target) != loop || FG_TO_BIH(target) != BIH_NEXT(bihx) ||
940 (br_ilt && IL_TYPE(ILI_OPC(ILT_ILIP(br_ilt))) == ILTY_BRANCH)) {
941 if (OPTDBG(9, 8))
942 fprintf(gbl.dbgfil, "---convert_loop(%d): bih%d single target with "
943 "jump or not fall through\n",
944 loop, bihx);
945 return;
946 }
947 } else {
948 /* Branch at br_ilt needs to be a conditional to avoid crashing
949 * below when its condition code is reversed.
950 */
951 if (br_ilt <= 0 ||
952 !(is_integer_comparison_opcode(ILI_OPC(ILT_ILIP(br_ilt))) ||
953 is_floating_comparison_opcode(ILI_OPC(ILT_ILIP(br_ilt))))) {
954 if (OPTDBG(9, 8))
955 fprintf(gbl.dbgfil,
956 "---convert_loop(%d): bih%d missing conditional jump\n", loop,
957 bihx);
958 return;
959 }
960 /*
961 * there can only be two successors of head -- the first one is the one
962 * which the head falls through to and the second is the one which the
963 * head exits to.
964 */
965 newhead = PSI_NODE(p);
966 if (FG_LOOP(newhead) != loop) {
967 if (OPTDBG(9, 8))
968 fprintf(gbl.dbgfil,
969 "---convert_loop(%d): first succ, %d, not in loop\n", loop,
970 newhead);
971 return;
972 }
973 p = PSI_NEXT(p);
974 exit = PSI_NODE(p);
975 if (FG_LOOP(exit) == loop) {
976 if (OPTDBG(9, 8))
977 fprintf(
978 gbl.dbgfil,
979 "---convert_loop(%d) succ node, %d, of head, %d, is in loop\n",
980 loop, exit, fgx);
981 return;
982 }
983 if (PSI_NEXT(p) != PSI_P_NULL) {
984 if (OPTDBG(9, 8))
985 fprintf(gbl.dbgfil, "---convert loop(%d): >2 succ(h) of head ,%d\n",
986 loop, fgx);
987 return;
988 }
989 break; /* found the loop exit branch */
990 }
991 }
992
993 /*
994 * the next physical block of the tail must be the block which the head
995 * exits to
996 */
997 tailbih = FG_TO_BIH(tail);
998 if (BIH_NEXT(tailbih) != FG_TO_BIH(exit)) {
999 return;
1000 }
1001 /* the number of successors of tail is 1 */
1002
1003 if (PSI_NEXT(FG_SUCC(tail)) != PSI_P_NULL) {
1004 if (OPTDBG(9, 8))
1005 fprintf(gbl.dbgfil,
1006 "---convert_loop(%d) more than one success of tail %d\n", loop,
1007 tail);
1008 return;
1009 }
1010 assert(ILI_OPC(ILT_ILIP(BIH_ILTLAST(tailbih))) == IL_JMP,
1011 "convert_loop: tail not IL_JMP", loop, ERR_Severe);
1012
1013 if (BIH_PAR(FG_TO_BIH(newhead)) != BIH_PAR(tailbih)) {
1014 if (OPTDBG(9, 8))
1015 fprintf(gbl.dbgfil,
1016 "---convert_loop(%d): BIH_PAR of %d and %d do not match\n", loop,
1017 FG_TO_BIH(newhead), tailbih);
1018 return;
1019 }
1020 if (BIH_TASK(FG_TO_BIH(newhead)) != BIH_TASK(tailbih)) {
1021 if (OPTDBG(9, 8))
1022 fprintf(gbl.dbgfil,
1023 "---convert_loop(%d): BIH_TASK of %d and %d do not match\n", loop,
1024 FG_TO_BIH(newhead), tailbih);
1025 return;
1026 }
1027
1028 /***** loop can be converted *****/
1029
1030 if (OPTDBG(9, 8))
1031 fprintf(gbl.dbgfil, "---convert_loop(%d), newhead:%d, tail:%d, exit:%d\n",
1032 loop, newhead, tail, exit);
1033
1034 BIH_LABEL(headbih) = SPTR_NULL; /* old head no longer has a label */
1035
1036 /*
1037 * check if the new head (the block following old head) already has a
1038 * label. If so, use it.
1039 */
1040 oldheadbih = headbih;
1041 headbih = FG_TO_BIH(newhead);
1042 tmpsptr = BIH_LABEL(headbih);
1043 if (tmpsptr != SPTR_NULL) {
1044 RFCNTD(label);
1045 RFCNTI(tmpsptr);
1046 label = tmpsptr;
1047 } else {
1048 BIH_LABEL(headbih) = label;
1049 ILIBLKP(label, headbih);
1050 }
1051
1052 /*
1053 * get the tail block and add the ilts from the old head block (except
1054 * for the last ilt) to the tail after the ilt which precedes its branch
1055 */
1056 rdilts(tailbih); /* sets bihb.callfg */
1057
1058 tmp = ILT_PREV(BIH_ILTLAST(tailbih)); /* where to begin adding ilts */
1059
1060 /*
1061 * go through and add the ilts from head blocks which occur before the branch
1062 * to
1063 * just before the last ilt in the tail. iltx is the first ilt which has
1064 * to be added.
1065 * NOTE: can't simply reuse the ili for the ilt; problem
1066 * occurs if a jsr appears in the tree -- by definition,
1067 * jsr's aren't shared unless an explicit cse situation
1068 * occurs. To ensure jsr's are rewritten, use the rewrite
1069 * ili mechanism using the IL_NULL as the 'old' & 'new'
1070 * arguments.
1071 */
1072 for (bihx = oldheadbih; 1; bihx = BIH_NEXT(bihx)) {
1073 iltx = BIH_ILTFIRST(bihx);
1074 if (ILT_DBGLINE(iltx))
1075 iltx = ILT_NEXT(iltx);
1076 for (; iltx && iltx != br_ilt; iltx = ILT_NEXT(iltx)) {
1077 int ilix = ILT_ILIP(iltx);
1078
1079 /* we don't want IL_LABEL that marks beggining of a scope to be moved to
1080 * last block of loop
1081 */
1082 if (ilix && ILI_OPC(ilix) == IL_LABEL && BEGINSCOPEG(ILI_OPND(ilix, 1)))
1083 continue;
1084 new_tree = rewr_ili(ilix, 1, 1);
1085 tmp = addilt(tmp, new_tree); /* updates bihb.callfg */
1086 if (OPTDBG(9, 8))
1087 fprintf(gbl.dbgfil, "---convert_loop(%d), oldtree:%d, newtree:%d\n",
1088 loop, ILT_ILIP(iltx), new_tree);
1089 }
1090 BIH_QJSR(tailbih) |= BIH_QJSR(bihx);
1091 bihb.callfg |= BIH_EX(bihx);
1092 if (bihx == lastheadbih)
1093 break;
1094 }
1095
1096 /*
1097 * Since the last ilt is one which branches back to the old head, change
1098 * that branch to the complement of the one which is in the old head. The
1099 * label referenced is the one which labels the new head
1100 */
1101 tmp = ILT_NEXT(tmp);
1102 assert(tmp == BIH_ILTLAST(tailbih), "convert_loop: wrong last ilt", tmp, ERR_Severe);
1103 new_tree = rewr_ili((int)ILT_ILIP(br_ilt), 1, 1);
1104 ILT_ILIP(tmp) = compl_br((int)new_tree, label);
1105 if (OPTDBG(9, 8))
1106 fprintf(gbl.dbgfil, "---convert_loop(%d), oldbrtree:%d, newbrtree:%d\n",
1107 loop, ILT_ILIP(br_ilt), ILT_ILIP(tmp));
1108 ILT_EX(tmp) = ILT_EX(br_ilt);
1109 BIH_EX(tailbih) = bihb.callfg | ILT_EX(tmp);
1110 wrilts(tailbih);
1111 rewr_cln_ili();
1112
1113 /*
1114 * remove old head nodes from the current loop and define the new head of the
1115 * loop
1116 */
1117
1118 for (bihx = oldheadbih; 1; bihx = BIH_NEXT(bihx)) {
1119 fgx = BIH_TO_FG(bihx);
1120 rm_node_from_loop(fgx);
1121 FG_LOOP(fgx) = 0;
1122 FG_NEXT(fgx) = 0;
1123 if (bihx == lastheadbih)
1124 break;
1125 }
1126
1127 LP_HEAD(loop) = newhead; /* already in the region */
1128 FG_NEXT(head) = 0; /* note: head is the last block in the */
1129 FG_NEXT(tail) = 0; /* region and tail is the next to last */
1130
1131 /* replace head in the successor list of tail with exit */
1132
1133 for (p = FG_SUCC(tail); p != PSI_P_NULL; p = PSI_NEXT(p)) {
1134 if (PSI_NODE(p) == head) {
1135 PSI_NODE(p) = exit;
1136 PSI_FT(p) = 1; /* tail now falls thru */
1137 break;
1138 }
1139 }
1140 assert(p != PSI_P_NULL, "convert_loop: head not succ of tail", tail, ERR_Severe);
1141
1142 /*
1143 * remove tail from the predecessor list of head and add tail to the
1144 * predecessor list of exit
1145 */
1146 q = PSI_P_NULL;
1147 for (p = FG_PRED(head); p != PSI_P_NULL; p = PSI_NEXT(p)) {
1148 if (PSI_NODE(p) == tail) { /* item is for the node tail */
1149 if (q == PSI_P_NULL)
1150 FG_PRED(head) = PSI_NEXT(p);
1151 else
1152 PSI_NEXT(q) = PSI_NEXT(p);
1153 PSI_NEXT(p) = FG_PRED(exit); /* add the item to pred(exit) */
1154 FG_PRED(exit) = p;
1155 break;
1156 }
1157 q = p;
1158 }
1159 assert(p != PSI_P_NULL, "convert_loop: tail not pred of head", head, ERR_Severe);
1160 BIH_FT(tailbih) = 1;
1161
1162 /*
1163 * add newhead to the successor list of tail and add tail to the
1164 * precessor list of newhead
1165 */
1166 (void)add_succ(tail, newhead);
1167 (void)add_pred(newhead, tail);
1168
1169 /*
1170 * mark this loop as one which has been converted -- the flag is zerotrip
1171 * meaning that there exists a test prior to the loop which may prevent
1172 * the loop from being executed. This flag may be used by induction.
1173 */
1174 LP_ZEROTRIP(loop) = 1;
1175 BIH_ZTRP(headbih) = 1;
1176 /*
1177 * the line number of the new loop head should be the line number of the
1178 * the old loop head instead of the new head's current line number.
1179 * necessary to maintain consistency with loop-scoped pragmas/directives.
1180 */
1181 BIH_LINENO(headbih) = BIH_LINENO(FG_TO_BIH(head));
1182
1183#if DEBUG
1184 if (DBGBIT(10, 2)) {
1185 fprintf(gbl.dbgfil, "after convert loop, bihs: head %d, tail %d\n", headbih,
1186 tailbih);
1187 dmpilt(headbih);
1188 if (headbih != tailbih)
1189 dmpilt(tailbih);
1190 }
1191#endif
1192
1193}

1241/*
1242 * reinsert each loop into its parent loop child list
1243 * so the order of the loop children matches the order in
1244 * the BIH list. Similarly, reinsert each FG node into its
1245 * loop FG list so the FG list matches the order in the BIH list.
1246 */
1247void
1248sortloops()
1249{
1250 int lpx, fgx, bihx, px;
1251 /* clear the LP_CHILD links */
1252 for (lpx = 0; lpx <= opt.nloops; ++lpx) {
1253 LP_CHILD(lpx) = 0;
1254 LP_FG(lpx) = 0;
1255 LP_SIBLING(lpx) = 0;
1256 }
1257 /* clear FG_NEXT link */
1258 for (fgx = 1; fgx < opt.num_nodes; ++fgx) {
1259 FG_NEXT(fgx) = 0;
1260 }
1261 for (bihx = gbl.entbih; bihx; bihx = BIH_NEXT(bihx)) {
1262 fgx = BIH_TO_FG(bihx);
1263 lpx = FG_LOOP(fgx);
1264 FG_NEXT(fgx) = LP_FG(lpx);
1265 LP_FG(lpx) = fgx; /* insert in reverse order, reverse later */
1266 if (lpx && fgx == LP_HEAD(lpx)) {
1267 px = LP_PARENT(lpx);
1268 LP_SIBLING(lpx) = LP_CHILD(px);
1269 LP_CHILD(px) = lpx;
1270 }
1271 }
1272 for (lpx = 0; lpx <= opt.nloops; ++lpx) {
1273 /* reverse LP_CHILD links, FG_NEXT */
1274 int childx, nextchildx, newchildx;
1275 int nextfgx, newfgx;
1276 childx = LP_CHILD(lpx);
1277 newchildx = 0;
1278 for (; childx; childx = nextchildx) {
1279 nextchildx = LP_SIBLING(childx);
1280 LP_SIBLING(childx) = newchildx;
1281 newchildx = childx;
1282 }
1283 LP_CHILD(lpx) = newchildx;
1284 fgx = LP_FG(lpx);
1285 newfgx = 0;
1286 for (; fgx; fgx = nextfgx) {
1287 nextfgx = FG_NEXT(fgx);
1288 FG_NEXT(fgx) = newfgx;
1289 newfgx = fgx;
1290 }
1291 LP_FG(lpx) = newfgx;
1292 }
1293} /* sortloops */
