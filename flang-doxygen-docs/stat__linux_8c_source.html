48void __fort_gettb(struct tb *t)
49{
50#if !defined(_WIN64)
51 struct timeval tv0;
52 struct timezone tz0;
53 struct rusage rs0, rc0;
54
55 /* Use an approximation here to avoid using inexact arithmetic */
56 unsigned long long tapprox;
57 union ieee v;
58
59 gettimeofday(&tv0, &tz0);
60 getrusage(RUSAGE_SELF, &rs0);
61 getrusage(RUSAGE_CHILDREN, &rc0);
62 v.i[0] = 0;
63 v.i[1] = 0x3d700000; /* 2 ** -40 */
64 tapprox = (unsigned long long)tv0.tv_usec; /* around 2**30 */
65 tapprox = tapprox * 1099512UL; /* mpy by a 21 bit value */
66 tapprox &= 0xfffffffffffc0000ULL; /* Lop off enough to be exact */
67 t->r = (double)tv0.tv_sec + v.d * (double)tapprox;
68 /* printf("BDL %d %d %22.15le\n",tv0.tv_sec,tv0.tv_usec,t->r); */
69
70 tapprox = rs0.ru_utime.tv_usec;
71 tapprox = tapprox * 1099512UL;
72 tapprox &= 0xfffffffffffc0000ULL;
73 t->u = (double)rs0.ru_utime.tv_sec + v.d * (double)tapprox;
74
75 tapprox = rs0.ru_stime.tv_usec;
76 tapprox = tapprox * 1099512UL;
77 tapprox &= 0xfffffffffffc0000ULL;
78 t->s = (double)rs0.ru_stime.tv_sec + v.d * (double)tapprox;
79
80 tapprox = rc0.ru_utime.tv_usec;
81 tapprox = tapprox * 1099512UL;
82 tapprox &= 0xfffffffffffc0000ULL;
83 t->u += (double)rc0.ru_utime.tv_sec + v.d * (double)tapprox;
84
85 tapprox = rc0.ru_stime.tv_usec;
86 tapprox = tapprox * 1099512UL;
87 tapprox &= 0xfffffffffffc0000ULL;
88 t->s += (double)rc0.ru_stime.tv_sec + v.d * (double)tapprox;
89 t->maxrss = rs0.ru_maxrss;
90 t->minflt = rs0.ru_minflt;
91 t->majflt = rs0.ru_majflt;
92 t->nsignals = rs0.ru_nsignals;
93 t->nvcsw = rs0.ru_nvcsw;
94 t->nivcsw = rs0.ru_nivcsw;
95 t->sbrk = (double)((long)sbrk(0));
96 t->gsbrk = (GET_DIST_HEAPZ == 0 ? 0.0 : (double)((long)__fort_sbrk(0)));
97 nodename(t->host);
98#endif
99}

103double
104__fort_second()
105{
106#if !defined(_WIN64)
107 struct timeval v;
108 struct timezone t;
109 double d;
110 int s;
111
112 s = gettimeofday(&v, &t);
113 if (s == -1) {
114 __fort_abortp("gettimeofday");
115 }
116 d = (double)v.tv_sec + (double)v.tv_usec / 1000000;
117 if (first == 0.0) {
118 first = d;
119 }
120 return (d - first);
121#endif
122}
