52/** \brief Layout descriptor
53 *
54 * A layout descriptor will have a list of the following integers representing
55 * information on components in the object. For now, we only specify a
56 * layout for pointer/allocatable/finalizable components. The layout
57 * descriptor is used with sourced allocation (to allocate/clone nested
58 * components in an object) and with final subroutines.
59 *
60 * There are five tags (used in the tag field) to indicate the type of
61 * component we're dealing with:
62 *
63 * - 'T' => an "allocated" pointer (has allocatable attribute)
64 * - 'D' => a regular pointer to a derived type (non-recursive)
65 * - 'R' => a "recursive" pointer to a derived type
66 * - 'P' => a general purpose pointer. Can be a pointer to an array, a
67 * scalar, etc.
68 * - 'F' => a non-pointer/non-allocatable finalizable component
69 * - 'S' => procedure pointer
70 */
71struct layout_desc {
72 __INT_T tag; /**< the tag -- one of T,D,R,P,F,S -- see
73 description above */
74 __INT_T type; /**< runtime type of component, base type if
75 pointer (not yet used) */
76 __INT_T offset; /**< byte offset of component */
77 __INT_T length; /**< length in bytes of component. 0 if unknown */
78 __INT_T desc_offset; /**< byte offset of component's descriptor. -1 if
79 N/A */
80 __INT_T padding; /**< reserved */
81 POINT(TYPE_DESC, declType); /**< ptr to declared type or 0 if N/A */
82};
