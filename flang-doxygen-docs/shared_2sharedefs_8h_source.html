11/*
12 * STG_DECLARE(name, datatype, indextype) - declare structure
13 * STG_ALLOC(name, size) - allocate
14 * STG_CLEAR(name) - clear all fields up to stg_avail
15 * STG_DELETE(name) - deallocate
16 * STG_RESET(name) - reset stg_avail
17 * i = STG_NEXT(name) - return next available index (no free list)
18 * i = STG_NEXT_SIZE(name, size) - return next available index, allocate size
19 * i = STG_NEXT_FREELIST(name) - return index from free list
20 * STG_NEED(name) - test avail vs size, realloc if needed
21 * STG_RETURN(name) - return lastest added element
22 * STG_ADD_FREELIST(name, i) - add to free list
23 * STG_ALLOC_SIDECAR(basename, name, datatype)
24 * allocate name the same size as basename
25 * register name on the sidecar list of basename
26 * STG_DELETE_SIDECAR(basename, name)
27 * remove name from the sidecar list of basename
28 * deallocate name
29 */
