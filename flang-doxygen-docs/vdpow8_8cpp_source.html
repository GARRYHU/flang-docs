36// double-double "fma"
37inline
38__m512d_2 __internal_ddfma (__m512d_2 x, __m512d_2 y, __m512d_2 z)
39{
40 __m512d e;
41 __m512d_2 t, m;
42 t.y = _mm512_mul_pd(x.y, y.y);
43 t.x = _mm512_fmsub_pd(x.y, y.y, t.y);
44 t.x = _mm512_fmadd_pd(x.y, y.x, t.x);
45 t.x = _mm512_fmadd_pd(x.x, y.y, t.x);
46
47 m.y = _mm512_add_pd (z.y, t.y);
48 e = _mm512_sub_pd (z.y, m.y);
49 m.x = _mm512_add_pd (_mm512_add_pd(_mm512_add_pd (e, t.y), t.x), z.x);
50
51 return m;
52}

80// slowpath for exp. used to improve accuracy on larger inputs
81static __m512d __attribute__ ((noinline)) __pgm_exp_d_vec512_slowpath(__m512i const i, __m512d const t, __m512d const bloga, __m512d z, __m512d prodx)
82{
83 __m512d const UPPERBOUND_1 = (__m512d)_mm512_set1_epi64(UPPERBOUND_1_D);
84 __m512d const UPPERBOUND_2 = (__m512d)_mm512_set1_epi64(UPPERBOUND_2_D);
85 __m512d const ZERO = _mm512_set1_pd(ZERO_D);
86 __m512d const INF = (__m512d)_mm512_set1_epi64(INF_D);
87 __m512i const HI_ABS_MASK = _mm512_set1_epi64(HI_ABS_MASK_D);
88 __m512i const ABS_MASK = _mm512_set1_epi64(ABS_MASK_D);
89 __m512i const MULT_CONST = _mm512_set1_epi64(MULT_CONST_D);
90
91 __m512i abs_bloga = _mm512_and_si512((__m512i)bloga, ABS_MASK);
92 __m512d abs_lt = (__m512d)_mm512_and_si512((__m512i)abs_bloga, HI_ABS_MASK);
93 __m512d lt_zero_mask = _MM512_CMP_PD(bloga, ZERO, _CMP_LT_OS); // compute bloga < 0.0
94 __m512d slowpath_mask = _MM512_CMP_PD(abs_lt, UPPERBOUND_1, _CMP_LT_OS);
95
96 __m512d a_plus_inf = _mm512_add_pd(bloga, INF); // check if bloga is too big
97 __m512d zero_inf_blend = _MM512_BLENDV_PD(a_plus_inf, ZERO, lt_zero_mask);
98
99 __m512d accurate_scale_mask = (__m512d)_MM512_CMP_PD(abs_lt, UPPERBOUND_2, 1);
100
101 // compute accurate scale
102 __m512i k = _mm512_srli_epi64(i, 1); // k = i / 2
103 __m512i i_scale_acc = _mm512_slli_epi64(k, D52_D); // shift to HI and shift 20
104 k = _mm512_sub_epi32(i, k); // k = i - k
105 __m512i i_scale_acc_2 = _mm512_slli_epi64(k, D52_D); // shift to HI and shift 20
106 __m512d multiplier = (__m512d)_mm512_add_epi64(i_scale_acc_2, MULT_CONST);
107 multiplier = _MM512_BLENDV_PD(ZERO, multiplier, accurate_scale_mask); // quick fix for overflows in case they're being trapped
108
109 __m512d res = (__m512d)_mm512_add_epi32(i_scale_acc, (__m512i)t);
110 res = _mm512_mul_pd(res, multiplier);
111
112 __m512d slowpath_blend = _MM512_BLENDV_PD(zero_inf_blend, res, accurate_scale_mask);
113 z = _MM512_BLENDV_PD(slowpath_blend, z, slowpath_mask);
114
115 __m512i isinf_mask = _MM512_CMPEQ_EPI64((__m512i)INF, (__m512i)z); // special case for inf
116 __m512d z_fixed = _mm512_fmadd_pd(z, prodx, z); // add only if not inf
117 z = _MM512_BLENDV_PD(z_fixed, z, (__m512d)isinf_mask);
118
119 return z;
120}

122// special cases for pow
123// implementation derived from cudart/device_functions_impl.c
124static __m512d __attribute__ ((noinline)) __pgm_pow_d_vec512_special_cases(__m512d const a, __m512d const b, __m512d t)
125{
126 __m512i const HI_MASK = _mm512_set1_epi64(HI_MASK_D);
127 __m512d const SGN_EXP_MASK = (__m512d)_mm512_set1_epi64(SGN_EXP_MASK_D);
128 __m512i const ABS_MASK = _mm512_set1_epi64(ABS_MASK_D);
129 __m512d const NEG_ONE = _mm512_set1_pd(NEG_ONE_D);
130 __m512d const ZERO = _mm512_set1_pd(ZERO_D);
131 __m512d const ONE = _mm512_set1_pd(ONE_D);
132 __m512d const HALF = _mm512_set1_pd(HALF_D);
133 __m512d const SGN_MASK = (__m512d)_mm512_set1_epi64(SGN_MASK_D);
134 __m512d const INF = (__m512d)_mm512_set1_epi64(INF_D);
135 __m512i const INF_FAKE = _mm512_set1_epi64(INF_FAKE_D);
136 __m512d const NAN_MASK = (__m512d)_mm512_set1_epi64(NAN_MASK_D);
137 __m512d const NEG_ONE_CONST = (__m512d)_mm512_set1_epi64(NEG_ONE_CONST_D);
138
139 __m512i const TEN_23 = _mm512_set1_epi64(TEN_23_D);
140 __m512i const ELEVEN = _mm512_set1_epi64(ELEVEN_D);
141
142 __m512i a_sign, b_sign, bIsOddInteger;
143 __m512d abs_a, abs_b;
144 __m512i shiftb;
145
146 a_sign = (__m512i)_MM512_AND_PD(a, SGN_MASK);
147 b_sign = (__m512i)_MM512_AND_PD(b, SGN_MASK);
148 abs_a = _MM512_AND_PD(a, (__m512d)ABS_MASK);
149 abs_b = _MM512_AND_PD(b, (__m512d)ABS_MASK);
150
151 // determining if b is an odd integer, since there are special cases for it
152 shiftb = (__m512i)_MM512_AND_PD (b, SGN_EXP_MASK);
153 shiftb = _mm512_srli_epi64(shiftb, D52_D);
154 shiftb = _mm512_sub_epi64(shiftb, TEN_23);
155 shiftb = _mm512_add_epi64(shiftb, ELEVEN);
156
157 __m512i b_is_half = _MM512_CMPEQ_EPI64((__m512i)abs_b, (__m512i)HALF);
158 bIsOddInteger = _mm512_sllv_epi64((__m512i)b, shiftb);
159 bIsOddInteger = _MM512_CMPEQ_EPI64((__m512i)SGN_MASK, bIsOddInteger);
160 // fix for b = +/-0.5 being incorrectly identified as an odd integer
161 bIsOddInteger = _mm512_andnot_si512(b_is_half, bIsOddInteger);
162
163 // corner cases where a <= 0
164 // if ((ahi < 0) && bIsOddInteger)
165 __m512d ahi_lt_0 = (__m512d)_MM512_CMPEQ_EPI64(a_sign, (__m512i)ZERO);
166 __m512d ahilt0_and_boddint_mask = _MM512_ANDNOT_PD(ahi_lt_0, (__m512d)bIsOddInteger);
167
168 t = _MM512_BLENDV_PD(t, neg(t), ahilt0_and_boddint_mask);
169
170 // else if ((ahi < 0) && (b != trunc(b)))
171 __m512d b_ne_trunc = _MM512_CMP_PD(b, _MM512_ROUND_PD(b, _MM_FROUND_TO_ZERO), _CMP_NEQ_UQ);
172 __m512d nan_mask = _MM512_ANDNOT_PD(ahi_lt_0, b_ne_trunc);
173 t = _MM512_BLENDV_PD(t, NAN_MASK, nan_mask);
174
175 // if (a == 0.0)
176 __m512d a_is_0_mask = (__m512d)_MM512_CMPEQ_EPI64((__m512i)abs_a, (__m512i)ZERO);
177 __m512d thi_when_ais0 = ZERO;
178
179 // if (bIsOddInteger && a == 0.0)
180 thi_when_ais0 = _MM512_BLENDV_PD(thi_when_ais0, (__m512d)a, (__m512d)bIsOddInteger);
181
182 // if (bhi < 0 && a == 0.0)
183 __m512d bhi_lt_0 = (__m512d)_MM512_CMPEQ_EPI64(b_sign, (__m512i)ZERO); //this mask is inverted
184 __m512d thi_or_INF = _MM512_OR_PD(thi_when_ais0, INF);
185 thi_when_ais0 = _MM512_BLENDV_PD(thi_or_INF, thi_when_ais0, bhi_lt_0);
186 __m512d t_when_ais0 = _MM512_AND_PD(thi_when_ais0, (__m512d)HI_MASK);
187
188 t = _MM512_BLENDV_PD(t, t_when_ais0, a_is_0_mask);
189
190 // else if (a is INF)
191 __m512d a_inf_mask = (__m512d)_MM512_CMPEQ_EPI64((__m512i)abs_a, (__m512i)INF);
192 // use bhi_lt_0 backwards to evaluate bhi >= 0
193 __m512d thi_when_aisinf = _MM512_BLENDV_PD(thi_when_aisinf, INF, bhi_lt_0);
194
195 // now evaluate ((ahi < 0) && bIsOddInteger)
196 __m512d thi_xor_sgn = _MM512_XOR_PD(thi_when_aisinf, SGN_MASK);
197 thi_when_aisinf = _MM512_BLENDV_PD(thi_when_aisinf, thi_xor_sgn, ahilt0_and_boddint_mask);
198
199 t = _MM512_BLENDV_PD(t, thi_when_aisinf, a_inf_mask);
200
201 // else if (abs(b) is INF)
202 __m512d b_inf_mask = (__m512d)_MM512_CMPEQ_EPI64((__m512i)abs_b, (__m512i)INF);
203 __m512d thi_when_bisinf = ZERO;
204 __m512d absa_gt_one = _MM512_CMP_PD(abs_a, ONE, _CMP_GT_OS); // evaluating (abs(a) > 1)
205 thi_when_bisinf = _MM512_BLENDV_PD(thi_when_bisinf, INF, absa_gt_one);
206
207 __m512d thi_xor_inf = _MM512_XOR_PD(thi_when_bisinf, INF);
208 thi_when_bisinf = _MM512_BLENDV_PD(thi_xor_inf, thi_when_bisinf, bhi_lt_0); // bhi < 0
209
210 __m512d a_is_negone = (__m512d)_MM512_CMPEQ_EPI64((__m512i)a, (__m512i)NEG_ONE);
211 thi_when_bisinf = _MM512_BLENDV_PD(thi_when_bisinf, NEG_ONE_CONST, a_is_negone); //a == -1
212
213 t = _MM512_BLENDV_PD(t, thi_when_bisinf, b_inf_mask);
214
215 // if(a is NAN || B is NAN) <=> !(a is a number && b is a number)
216 __m512i a_nan_mask = _MM512_CMPEQ_EPI64(_mm512_max_epu32((__m512i)abs_a, INF_FAKE), INF_FAKE);
217 __m512i b_nan_mask = _MM512_CMPEQ_EPI64(_mm512_max_epu32((__m512i)abs_b, INF_FAKE), INF_FAKE);
218 __m512i aorb_nan_mask = _mm512_and_si512(a_nan_mask, b_nan_mask); //this mask is inverted
219 t = _MM512_BLENDV_PD(_mm512_add_pd(a,b), t, (__m512d)aorb_nan_mask);
220
221 // if a == 1 or b == 0, answer = 1
222 __m512d a_is_one_mask = (__m512d)_MM512_CMPEQ_EPI64((__m512i)a, (__m512i)ONE);
223 __m512d b_is_zero_mask = (__m512d)_MM512_CMPEQ_EPI64((__m512i)abs_b, (__m512i)ZERO);
224 __m512d ans_equals_one_mask = _MM512_OR_PD(a_is_one_mask, b_is_zero_mask);
225 t = _MM512_BLENDV_PD(t, ONE, ans_equals_one_mask);
226 // ******************************************************************************************
227
228
229 __m512i a_is_neg = (__m512i)_MM512_CMP_PD(a, ZERO, _CMP_LT_OS);
230 int a_is_neg_flag = _MM512_MOVEMASK_PD(_mm512_castsi512_pd(a_is_neg));
231
232 __m512i b_is_integer = (__m512i)_MM512_CMP_PD(b, _mm512_floor_pd(b), _CMP_EQ_OQ);
233 int b_is_integer_flag = _MM512_MOVEMASK_PD(_mm512_castsi512_pd(b_is_integer));
234
235 __m512i b_is_lt_zero = (__m512i)_MM512_CMP_PD(b, ZERO, _CMP_LT_OS);
236 int b_is_lt_zero_flag = _MM512_MOVEMASK_PD(_mm512_castsi512_pd(b_is_lt_zero));
237
238 __m512d const MINUS_ZERO = _mm512_set1_pd((double)-0.0);
239 __m512i a_is_pos_zero = (__m512i) _MM512_CMPEQ_PD( a, ZERO);
240 __m512i a_is_neg_zero = (__m512i) _MM512_CMPEQ_PD( a, MINUS_ZERO);
241 __m512i a_is_any_zero = _mm512_or_si512(a_is_pos_zero, a_is_neg_zero);
242 int a_is_any_zero_flag = _MM512_MOVEMASK_PD(_mm512_castsi512_pd(a_is_any_zero));
243
244/*
245 * Before returning see if we need to set any of the processor
246 * exception flags.
247 *
248 * Domain error: a is negative, and b is a finite noninteger
249 * we need to raise the Invalid-Operation flag. This can be done by
250 * taking the square root of a negative number.
251 *
252 * Pole error: a is zero and b is negative we need to raise the
253 * divide by zero flag. This can be done by dividing by zero.
254 */
255
256 if (a_is_neg_flag && (!b_is_integer_flag)) {
257 __m512d volatile invop = _mm512_sqrt_pd(a);
258 }
259
260 if (a_is_any_zero_flag && b_is_lt_zero_flag) {
261 __m512d volatile divXzero = _mm512_div_pd(ONE,ZERO);
262 }
263
264
265
266
267
268
269
270
271
272
273
274
275 return t;
276}

281 __m512d const LOG_POLY_6 = _mm512_set1_pd(LOG_POLY_6_D);

284 __m512d const LOG_POLY_3 = _mm512_set1_pd(LOG_POLY_3_D);
285 __m512d const LOG_POLY_2 = _mm512_set1_pd(LOG_POLY_2_D);
286 __m512d const LOG_POLY_1 = _mm512_set1_pd(LOG_POLY_1_D);
287 __m512d const LOG_POLY_0 = _mm512_set1_pd(LOG_POLY_0_D);
288
289 __m512d const CC_CONST_Y = _mm512_set1_pd(CC_CONST_Y_D);

291
292 __m512i const EXPO_MASK = _mm512_set1_epi64(EXPO_MASK_D);

305 //////////////////////////////////////////////////// exp constants
306 __m512d const L2E = _mm512_set1_pd(L2E_D);
307 __m512d const NEG_LN2_HI = _mm512_set1_pd(NEG_LN2_HI_D);

324
325 __m512d const UPPERBOUND_1 = (__m512d)_mm512_set1_epi64(UPPERBOUND_1_D);
326 __m512i const HI_ABS_MASK = _mm512_set1_epi64(HI_ABS_MASK_D);

329 __m512i const HI_MASK = _mm512_set1_epi64(HI_MASK_D);
330 __m512d const SGN_EXP_MASK = (__m512d)_mm512_set1_epi64(SGN_EXP_MASK_D);
331 __m512i const ABS_MASK = _mm512_set1_epi64(ABS_MASK_D);
332 __m512i const ONE = _mm512_set1_epi64(ONE_D);
333
334 __m512i const TEN_23 = _mm512_set1_epi64(TEN_23_D);
335 __m512i const ALL_ONES_EXPONENT = _mm512_set1_epi64(ALL_ONES_EXPONENT_D);

338 __m512d abs_a, a_mut;
339 __m512d_2 loga;
340 __m512d t_hi, t_lo, tmp, e, bloga, prodx;
341
342 __m512d_2 qq, cc, uu, tt;
343 __m512d f, g, u, v, q, ulo, m;
344 __m512i ihi, expo, expo_plus1;
345 __m512d thresh_mask;
346 __m512d b_is_one_mask;
347
348 /*
349 * Check whether all exponents in vector b are 1.0, and if so take
350 * a quick exit.
351 * Note: b_is_one_mask is needed later in case some elements in b are 1.0.
352 */
353
354 b_is_one_mask = _MM512_CMP_PD(b, ONE_F, _CMP_EQ_OQ);
355//Correct - but slower
356// if (_MM512_MOVEMASK_PD(b_is_one_mask) == 0xff) {
357 if (_mm512_cmp_pd_mask(b, ONE_F, _CMP_EQ_OQ) == 0xff) {
358 return a;
359 }
360
361 // *****************************************************************************************
362 // computing log(abs(a))
363 abs_a = _MM512_AND_PD(a, (__m512d)ABS_MASK);
364 a_mut = _MM512_AND_PD(abs_a, HI_CONST_1);
365 a_mut = _MM512_OR_PD(a_mut, HI_CONST_2);
366 m = (__m512d)_mm512_sub_epi32((__m512i)a_mut, HALFIFIER); // divide by 2
367
368 ihi = _mm512_and_si512((__m512i)a_mut, HI_MASK);
369 thresh_mask = _MM512_CMP_PD((__m512d)ihi, (__m512d)HI_THRESH,_CMP_GT_OS);
370 m = _MM512_BLENDV_PD(a_mut, m, thresh_mask);
371
372 expo = _mm512_srli_epi64((__m512i)abs_a, D52_D);
373 expo = _mm512_sub_epi64(expo, TEN_23);
374 expo_plus1 = _mm512_add_epi64(expo, ONE); // add one to exponent instead
375 expo = (__m512i)_MM512_BLENDV_PD((__m512d)expo, (__m512d)expo_plus1, thresh_mask);
376
377 // begin computing log(m)
378 f = _mm512_sub_pd(m, ONE_F);
379 g = _mm512_add_pd(m, ONE_F);
380 g = _mm512_div_pd(ONE_F, g);
381
382 u = _mm512_mul_pd(_mm512_mul_pd(TWO, f), g);
383
384 // u = 2.0 * (m - 1.0) / (m + 1.0)
385 v = _mm512_mul_pd(u, u);
386
387 // polynomial is used to approximate atanh(v)
388 // an estrin evaluation scheme is used.
389 __m512d c0 = _mm512_fmadd_pd(LOG_POLY_1, v, LOG_POLY_0);
390 __m512d c2 = _mm512_fmadd_pd(LOG_POLY_3, v, LOG_POLY_2);
391 __m512d c4 = _mm512_fmadd_pd(LOG_POLY_5, v, LOG_POLY_4);
392 __m512d v2 = _mm512_mul_pd(v, v);
393 __m512d v4 = _mm512_mul_pd(v2, v2);
394
395 c0 = _mm512_fmadd_pd(c2, v2, c0);
396 c4 = _mm512_fmadd_pd(LOG_POLY_6, v2, c4);
397 q = _mm512_fmadd_pd(c4, v4, c0);
398 q = _mm512_mul_pd(q, v);
399
400 tmp = _mm512_mul_pd(TWO, _mm512_sub_pd(f, u));
401 tmp = _mm512_fmadd_pd(neg(u), f, tmp);
402 ulo = _mm512_mul_pd(g, tmp);
403
404 // double-double computation begins
405 qq.y = q;
406 qq.x = ZERO;
407 uu.y = u;
408 uu.x = ulo;
409 cc.y = CC_CONST_Y;
410 cc.x = CC_CONST_X;
411
412 qq = __internal_ddadd_yisdouble(cc, qq);
413
414 // computing log(m) in double-double format
415 cc.y = _mm512_mul_pd(uu.y, uu.y);
416 cc.x = _mm512_fmsub_pd(uu.y, uu.y, cc.y);
417 cc.x = _mm512_fmadd_pd(uu.y,
418 (__m512d)_mm512_add_epi32((__m512i)uu.x, HALFIFIER),
419 cc.x); // u ** 2
420
421 tt.y = _mm512_mul_pd(cc.y, uu.y);
422 tt.x = _mm512_fmsub_pd(cc.y, uu.y, tt.y);
423 tt.x = _mm512_fmadd_pd(cc.y, uu.x, tt.x);
424 tt.x = _mm512_fmadd_pd(cc.x, uu.y, tt.x); // u ** 3 un-normalized
425
426 uu = __internal_ddfma(qq, tt, uu);
427
428 // computing log a = log(m) + log(2)*expo
429 f = __internal_fast_int2dbl (expo);
430 q = _mm512_fmadd_pd(f, LN2_HI, uu.y);
431 tmp = _mm512_fmadd_pd(neg(f), LN2_HI, q);
432 tmp = _mm512_sub_pd(tmp, uu.y);
433 loga.y = q;
434 loga.x = _mm512_sub_pd(uu.x, tmp);
435 loga.x = _mm512_fmadd_pd(f, LN2_LO, loga.x);
436
437 // finish log(a)
438 // ******************************************************************************************
439
440 // compute b * log(a)
441 t_hi = _mm512_mul_pd(loga.y, b);
442 t_lo = _mm512_fmsub_pd(loga.y, b, t_hi);
443 t_lo = _mm512_fmadd_pd(loga.x, b, t_lo);
444 bloga = e = _mm512_add_pd(t_hi, t_lo);
445 prodx = _mm512_add_pd(_mm512_sub_pd(t_hi, e), t_lo);
446
447 // ***************************************************************************************
448 // computing exp(b * log(a))
449 // calculating exponent; stored in the LO of each 64-bit block
450 __m512i i = (__m512i) _mm512_fmadd_pd(bloga, L2E, DBL2INT_CVT);
451
452 // calculate mantissa
453 __m512d t = _mm512_sub_pd ((__m512d)i, DBL2INT_CVT);
454 __m512d z = _mm512_fmadd_pd (t, NEG_LN2_HI, bloga);
455 z = _mm512_fmadd_pd (t, NEG_LN2_LO, z);
456
457 // use polynomial to calculate exp
458 // mixed estrin/horner scheme: estrin on the higher 8 coefficients, horner on the lowest 4.
459 // provided speedup without loss of precision compared to full horner
460 __m512d t4 = _mm512_fmadd_pd(EXP_POLY_5, z, EXP_POLY_4);
461 __m512d t6 = _mm512_fmadd_pd(EXP_POLY_7, z, EXP_POLY_6);
462 __m512d t8 = _mm512_fmadd_pd(EXP_POLY_9, z, EXP_POLY_8);
463 __m512d t10 = _mm512_fmadd_pd(EXP_POLY_B, z, EXP_POLY_A);
464
465 __m512d z2 = _mm512_mul_pd(z, z);
466 __m512d z4 = _mm512_mul_pd(z2, z2);
467
468 t4 = _mm512_fmadd_pd(t6, z2, t4);
469 t8 = _mm512_fmadd_pd(t10, z2, t8);
470 t4 = _mm512_fmadd_pd(t8, z4, t4);
471
472 t = _mm512_fmadd_pd(t4, z, EXP_POLY_3);
473 t = _mm512_fmadd_pd(t, z, EXP_POLY_2);
474 t = _mm512_fmadd_pd(t, z, EXP_POLY_1);
475 t = _mm512_fmadd_pd(t, z, EXP_POLY_0);
476
477 // fast scale
478 __m512i i_scale = _mm512_slli_epi64(i,D52_D);
479 __m512d ztemp = z = (__m512d)_mm512_add_epi32(i_scale, (__m512i)t);
480
481 // slowpath detection for exp
482 __m512d abs_bloga = (__m512d)_mm512_and_si512((__m512i)bloga, HI_ABS_MASK);
483 int exp_slowmask = _MM512_MOVEMASK_PD(_MM512_CMP_PD(abs_bloga, UPPERBOUND_1, _CMP_GE_OS));
484
485 z = _mm512_fmadd_pd(z, prodx, z);
486
487 if (__builtin_expect(exp_slowmask, 0)) {
488 z = __pgm_exp_d_vec512_slowpath(i, t, bloga, ztemp, prodx);
489 }
490 // finished exp(b * log (a))
491 // ************************************************************************************************
492
493 // Now special case if some elements of exponent (b) are 1.0.
494 z = _MM512_BLENDV_PD(z, a, b_is_one_mask);
495
496 // compute if we have special cases (inf, nan, etc). see man pow for full list of special cases
497 __m512i detect_inf_nan = (__m512i)_mm512_add_pd(a, b); // check for inf/nan
498 __m512i overridemask = _MM512_CMPEQ_EPI64( (__m512i)a, (__m512i)ONE_F); // if a == 1
499 __m512i overridemask2 = _MM512_CMPEQ_EPI64(_mm512_and_si512(detect_inf_nan, ALL_ONES_EXPONENT), ALL_ONES_EXPONENT);
500 overridemask = _mm512_or_si512(overridemask, (__m512i)_MM512_CMP_PD(a, ZERO, _CMP_LE_OQ)); // if a < 0
501
502 int specMask = _MM512_MOVEMASK_PD((__m512d)_mm512_or_si512(overridemask, overridemask2));
503 if(__builtin_expect(specMask, 0)) {
504 return __pgm_pow_d_vec512_special_cases(a, b, z);
505 }
506 return z;
507}
