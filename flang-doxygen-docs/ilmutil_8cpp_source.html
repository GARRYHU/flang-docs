188/**
189 * Add 'n' ILM words to current ILM buffer, including the opc.
190 */
191int
192adNilm(int n, int opc, ...)
193{
194 ILM_T *p;
195 int i, opr;
196 va_list vargs;
197
198 assert(n > 5, "adNilm should only be used for ILMs with >5 arguments", opc,
199 ERR_Fatal);
200
201#if DEBUG
202 if (DBGBIT(4, 0x4))
203 fprintf(gbl.dbgfil, "%5d %s %d operands\n", ilmb.ilmavl, ILMNAME(opc), n);
204#endif
205 NEED(ilmb.ilmavl + n, ilmb.ilm_base, ILM_T, ilmb.ilm_size,
206 ilmb.ilmavl + 1000);
207 p = ilmb.ilm_base + ilmb.ilmavl;
208 ilmb.ilmavl += n;
209
210 va_start(vargs, opc);
211
212 *p++ = opc;
213 for (i = 1; i < n; ++i, ++p) {
214 opr = va_arg(vargs, int);
215 *p = opr;
216 }
217 --p;
218 va_end(vargs);
219 return ilmb.ilmavl - n;
220}

255{
256 DEBUG_ASSERT(0 < ilmp[0] && ilmp[0] < N_ILM,
257 "ilm_return_slot_index: bad ILM");

261 switch (ilmp[0]) {

423/*
424 * write one block of ILM's to ILM file.
425 */
426void
427wrilms(int linenum)
428{
429 void dmpilms();
430 ILM_T *p;
431 int nw;
432
433 /* if nocode, then just return */
434 if (ilmb.ilmavl == BOS_SIZE)
435 return;
436
437 p = ilmb.ilm_base;
438 *p++ = IM_BOS;
439 if (linenum == -1 || linenum == 0)
440 *p++ = gbl.lineno;
441 else
442 *p++ = linenum;
443
444 *p++ = gbl.findex;
445
446 *p = ilmb.ilmavl;
447
448 if (sem.wrilms) {
449 nw = fwrite((char *)ilmb.ilm_base, sizeof(ILM_T), ilmb.ilmavl, gbl.ilmfil);
450 if (nw != ilmb.ilmavl)
451 error(F_0010_File_write_error_occurred_OP1, ERR_Fatal, 0, "(IL file)", CNULL);
452 }
453
454 if (DBGBIT(4, 1))
455 dmpilms();
456
457 ilmb.ilmavl = BOS_SIZE;
458 gilmb_mode = 0; /* next rdilms comes from file */
459}

463/*
464 * if the ILM area is not empty, allocate a block in the indicated working
465 * storage area and copy the current ILM's into it. The first word of the
466 * block will contain the number of ILM words. The ILM area is reset to the
467 * empty state.
468 */
469ILM_T *
470save_ilms(int area)
471{
472 ILM_T *p;
473 int count;
474 ILM_T *q;
475
476 if (ilmb.ilmavl == BOS_SIZE)
477 return NULL;
478 count = ilmb.ilmavl - BOS_SIZE;
479 q = p = (ILM_T *)getitem(area, sizeof(ILM_T) * (count + 1));
480 *p++ = count;
481 BCOPY(p, ilmb.ilm_base + BOS_SIZE, ILM_T, count);
482#if DEBUG
483 if (DBGBIT(4, 8)) {
484 ILMA(0) = IM_BOS;
485 ILMA(1) = gbl.lineno;
486 ILMA(2) = gbl.findex;
487 ILMA(BOS_SIZE - 1) = ilmb.ilmavl;
488 dmpilms();
489 }
490#endif
491 ilmb.ilmavl = BOS_SIZE;
492 return q;
493}

552/*
553 * Copy block of ILM's, previously saved by save_ilms, directly into the ILM
554 * buffer and relocate links.
555 */
556void
557reloc_ilms(ILM_T *p)
558{
559 int count;
560 int need;
561 int ilmptr;
562 int rlc;
563
564 if (p == NULL)
565 return;
566 ilmptr = ilmb.ilmavl;
567 rlc = ilmb.ilmavl - BOS_SIZE;
568 count = *p++;
569 assert(count > 0, "reloc_ilms: non-positive count", 0, ERR_Fatal);
570 need = count;
571 if (need < 1000)
572 need = 1000;
573 NEED(ilmb.ilmavl + count, ilmb.ilm_base, ILM_T, ilmb.ilm_size,
574 ilmb.ilmavl + need);
575 BCOPY(ilmb.ilm_base + ilmb.ilmavl, p, ILM_T, count);
576 ilmb.ilmavl += count;
577
578 /* scan current ILM buffer, and relocate links */
579 do { /* loop once for each opcode */
580 int opc, len, noprs, varpart, opnd;
581
582 opc = ILMA(ilmptr);
583 len = noprs = ilms[opc].oprs; /* number of "fixed" operands */
584 if (IM_VAR(opc)) {
585 varpart = ILMA(ilmptr + 1);
586 len += varpart;
587 /*
588 * for an ILM with a variable number operands, we only want to
589 * examine the operands only if they are links. In any case,
590 * we want to begin at the second operand.
591 */
592 if (IM_OPRFLAG(opc, noprs + 1) != OPR_LNK)
593 varpart = 0;
594 noprs--;
595 opnd = 2;
596 } else {
597 /*
598 * the ILM does not have any variable ILM links -- the
599 * analysis begins with the first operand.
600 */
601 varpart = 0;
602 opnd = 1;
603 }
604 for (;; opnd++) {
605 if (noprs == 0) {
606 if ((varpart--) == 0)
607 break;
608 } else {
609 noprs--;
610 if (IM_OPRFLAG(opc, opnd) != OPR_LNK)
611 continue;
612 }
613#if DEBUG
614 assert(ILMA(ilmptr + opnd) >= BOS_SIZE && ILMA(ilmptr + opnd) < ilmptr,
615 "reloc_ilms: bad lnk", ilmptr, ERR_Severe);
616#endif
617 ILMA(ilmptr + opnd) += rlc;
618 }
619 ilmptr += (len + 1);
620 } while (ilmptr < ilmb.ilmavl);
621}

625/*
626 * Convert ILM 'tree' pointed to by ilmptr into a series of one or more
627 * conditional branches which have the net effect of branching to truelb
628 * if the condition has the truth value of flag:
629 */
630void
631mkbranch(int ilmptr, int truelb, int flag)
632{
633 int opc, falselb;
634
635 opc = ILMA(ilmptr);
636 if (opc == IM_LAND) {
637 if (!flag) {
638 /* if (!ilm1 || !ilm2) goto truelb */
639 mkbranch(ILMA(ilmptr + 1), truelb, false);
640 mkbranch(ILMA(ilmptr + 2), truelb, false);
641 /* erase the AND ilm: */
642 if (ilmptr + 3 == ilmb.ilmavl)
643 ilmb.ilmavl = ilmptr;
644 else {
645 ILMA(ilmptr++) = IM_NOP;
646 ILMA(ilmptr++) = IM_NOP;
647 ILMA(ilmptr) = IM_NOP;
648 }
649 } else {
650 /* if (ilm1 && ilm2) goto truelb -->
651 *
652 * if (!ilm1) goto falselb
653 * if (ilm2) goto truelb
654 * falselb:
655 */
656 falselb = getlab();
657 mkbranch(ILMA(ilmptr + 1), falselb, false);
658 mkbranch(ILMA(ilmptr + 2), truelb, true);
659 ILMA(ilmptr++) = IM_LABEL;
660 ILMA(ilmptr++) = falselb;
661 ILMA(ilmptr) = IM_NOP;
662 }
663 } else if (opc == IM_LAND8) {
664 /*
665 * tpr3035, the relational expressions are always logical*4. In the
666 * presence of a logical*8 expression, the relational is converted
667 * with a IM_ITOI8. Unfortunately, this means that the ILM order of
668 * the operands to LAND8 to LOR8 may change, and the assumptions to
669 * effect short-circuiting are no longer valid; e.g.,
670 * (1) a
671 * (2) b
672 * (3) LOR (1) (2) [operand 1 precedes operand 2]
673 * becomes
674 * (1) a
675 * (2) b
676 * (3) ITOI8 a
677 * (4) LOR8 (3) (2) [operand 1 no longer precedes operand 2]
678 *
679 * Need to detect this situation and to correct the order.
680 */
681 if (ILMA(ilmptr + 1) > ILMA(ilmptr + 2)) {
682 int i1, i2;
683 i1 = ILMA(ilmptr + 1);
684 i2 = ILMA(ilmptr + 2);
685 ILMA(ilmptr + 1) = i2;
686 ILMA(ilmptr + 2) = i1;
687 mkbranch(ilmptr, truelb, flag);
688 return;
689 }
690 if (!flag) {
691 /* if (!ilm1 || !ilm2) goto truelb */
692 mkbranch(ILMA(ilmptr + 1), truelb, false);
693 mkbranch(ILMA(ilmptr + 2), truelb, false);
694 /* erase the AND ilm: */
695 if (ilmptr + 3 == ilmb.ilmavl)
696 ilmb.ilmavl = ilmptr;
697 else {
698 ILMA(ilmptr++) = IM_NOP;
699 ILMA(ilmptr++) = IM_NOP;
700 ILMA(ilmptr) = IM_NOP;
701 }
702 } else {
703 /* if (ilm1 && ilm2) goto truelb -->
704 *
705 * if (!ilm1) goto falselb
706 * if (ilm2) goto truelb
707 * falselb:
708 */
709 falselb = getlab();
710 mkbranch(ILMA(ilmptr + 1), falselb, false);
711 mkbranch(ILMA(ilmptr + 2), truelb, true);
712 ILMA(ilmptr++) = IM_LABEL;
713 ILMA(ilmptr++) = falselb;
714 ILMA(ilmptr) = IM_NOP;
715 }
716 } else if (opc == IM_LOR) {
717 if (flag) {
718 /* if (ilm1 || ilm2) goto truelb */
719 mkbranch(ILMA(ilmptr + 1), truelb, true);
720 mkbranch(ILMA(ilmptr + 2), truelb, true);
721
722 /* erase the OR ilm: */
723 if (ilmptr + 3 == ilmb.ilmavl)
724 ilmb.ilmavl = ilmptr;
725 else {
726 ILMA(ilmptr++) = IM_NOP;
727 ILMA(ilmptr++) = IM_NOP;
728 ILMA(ilmptr) = IM_NOP;
729 }
730 } else {
731 /* if (!ilm1 && !ilm2) goto truelb -->
732 *
733 * if (ilm1) goto falselb
734 * if (!ilm2) goto truelb
735 * falselb:
736 */
737 falselb = getlab();
738 mkbranch(ILMA(ilmptr + 1), falselb, true);
739 mkbranch(ILMA(ilmptr + 2), truelb, false);
740 ILMA(ilmptr++) = IM_LABEL;
741 ILMA(ilmptr++) = falselb;
742 ILMA(ilmptr) = IM_NOP;
743 }
744 } else if (opc == IM_LOR8) {
745 /*
746 * tpr3035, the relational expressions are always logical*4. In the
747 * presence of a logical*8 expression, the relational is converted
748 * with a IM_ITOI8. Unfortunately, this means that the ILM order of
749 * the operands to LAND8 to LOR8 may change, and the assumptions to
750 * effect short-circuiting are no longer valid; e.g.,
751 * (1) a
752 * (2) b
753 * (3) LOR (1) (2) [operand 1 precedes operand 2]
754 * becomes
755 * (1) a
756 * (2) b
757 * (3) ITOI8 a
758 * (4) LOR8 (3) (2) [operand 1 no longer precedes operand 2]
759 *
760 * Need to detect this situation and to correct the order.
761 */
762 if (ILMA(ilmptr + 1) > ILMA(ilmptr + 2)) {
763 int i1, i2;
764 i1 = ILMA(ilmptr + 1);
765 i2 = ILMA(ilmptr + 2);
766 ILMA(ilmptr + 1) = i2;
767 ILMA(ilmptr + 2) = i1;
768 mkbranch(ilmptr, truelb, flag);
769 return;
770 }
771 if (flag) {
772 /* if (ilm1 || ilm2) goto truelb */
773 mkbranch(ILMA(ilmptr + 1), truelb, true);
774 mkbranch(ILMA(ilmptr + 2), truelb, true);
775
776 /* erase the OR ilm: */
777 if (ilmptr + 3 == ilmb.ilmavl)
778 ilmb.ilmavl = ilmptr;
779 else {
780 ILMA(ilmptr++) = IM_NOP;
781 ILMA(ilmptr++) = IM_NOP;
782 ILMA(ilmptr) = IM_NOP;
783 }
784 } else {
785 /* if (!ilm1 && !ilm2) goto truelb -->
786 *
787 * if (ilm1) goto falselb
788 * if (!ilm2) goto truelb
789 * falselb:
790 */
791 falselb = getlab();
792 mkbranch(ILMA(ilmptr + 1), falselb, true);
793 mkbranch(ILMA(ilmptr + 2), truelb, false);
794 ILMA(ilmptr++) = IM_LABEL;
795 ILMA(ilmptr++) = falselb;
796 ILMA(ilmptr) = IM_NOP;
797 }
798 } else {
799 if (IS_COMPARE(opc)) {
800 /* follow opcode with BRT or BRF*/
801 ILMA(ilmptr + 2) = flag ? IM_BRT : IM_BRF;
802 ILMA(ilmptr + 3) = ilmptr;
803 ILMA(ilmptr + 4) = truelb;
804 } else if (opc == IM_LNOP || opc == IM_LNOP8) {
805 ILMA(ilmptr) = flag ? IM_BRT : IM_BRF;
806 ILMA(ilmptr + 2) = truelb;
807 } else if (opc == IM_LNOT || opc == IM_LNOT8) {
808 ILMA(ilmptr) = flag ? IM_BRF : IM_BRT;
809 ILMA(ilmptr + 2) = truelb;
810 } else
811 (void)ad3ilm(flag ? IM_BRT : IM_BRF, ilmptr, truelb);
812
813 RFCNTI(truelb); /* increment label reference count */
814 }
815}
