68/**
69 \brief write two lines of info for symbol elem to listing.
70*/
71static void
72putentry(int elem)
73{
74 char buf[200];
75 const char *ptr, *sc_p;
76 int stype;
77 DTYPE dtype;
78 char hyp;
79 static const char *scs[] = {"n/a", "local", "static", "dummy", "common",
80 "extern", "based", "private", "eight"};
81
82 stype = STYPEG(elem);
83
84 ptr = SYMNAME(elem);
85 if (stype == ST_LABEL)
86 ptr += 2; /* delete ".L" from front of label name */
87
88 dtype = DT_NONE;
89 if (stype != ST_LABEL && stype != ST_CMBLK) {
90 dtype = DTYPEG(elem);
91 }
92
93 sprintf(buf, "%-16.16s ", ptr);
94 getdtype(dtype, &buf[17]);
95 strcat(buf, " ");
96 strcat(buf, stb.stypes[stype]);
97 list_line(buf);
98
99 if ((int)strlen(ptr) > 16) {
100 ptr += 16;
101 hyp = '-';
102 } else {
103 ptr = "";
104 hyp = ' ';
105 }
106
107 switch (stype) {
108 case ST_LABEL:
109 sprintf(buf, "%c%-15.15s addr: %08" ISZ_PF "x", hyp, ptr, ADDRESSG(elem));
110 break;
111 case ST_PARAM:
112/* Changed to display ALL PARAMETER data types LDE */
113 sprintf(buf, "%c%-15.15s value: %s", hyp, ptr, parmprint(elem));
114 break;
115 case ST_ENTRY:
116 sprintf(buf, "%c%-15.15s addr: %08" ISZ_PF "x", hyp, ptr, ADDRESSG(elem));
117 break;
118 case ST_CMBLK:
119 sprintf(buf, "%c%-15.15s size: %" ISZ_PF "d bytes", hyp, ptr,
120 SIZEG(elem));
121 break;
122
123 case ST_VAR:
124 case ST_IDENT:
125 case ST_ARRAY:
126 case ST_STRUCT:
127 case ST_UNION:
128 if (SCG(elem) == SC_CMBLK)
129 sc_p = SYMNAME(MIDNUMG(elem));
130 else
131 sc_p = scs[SCG(elem)];
132 sprintf(buf, "%c%-15.15s sc: %s addr: %08" ISZ_PF "x", hyp, ptr, sc_p,
133 ADDRESSG(elem));
134 break;
135
136 case ST_PROC:
137 case ST_STFUNC:
138 case ST_GENERIC:
139 case ST_INTRIN:
140 default:
141 if (hyp == '-')
142 sprintf(buf, "%c%-15.15s", hyp, ptr);
143 else
144 hyp = 0;
145 }
146
147 if (hyp != 0)
148 list_line(buf);
149
150}

154static void
155printxref(SPTR liststart)
156{
157 SPTR sptr;
158 int refptr;
159 int i;
160 char buf[200];
161
162 /* print header */
163 list_line("Cross Reference Listing:");
164 list_line("");
165
166 /* for each ref'd element of symbol table with a name */
167 for (sptr = liststart; sptr != SPTR_NULL; sptr = HASHLKG(sptr)) {
168 assert((NMPTRG(sptr) != 0), "printxref bad record name", sptr, ERR_Severe);
169 assert((XREFLKG(sptr) != 0), "printxref bad record link", sptr, ERR_Severe);
170
171 /* print stuff about symbol table entry */
172 putentry(sptr);
173
174 refptr = XREFLKG(sptr);
175
176#define ENDOFLINE 80
177 /* put references info */
178 i = 999; /* i is char position in line being printed */
179 do {
180 if (i > (ENDOFLINE - 5)) {
181 if (i != 999)
182 list_line(buf);
183 sprintf(buf, " ");
184 i = 16;
185 }
186 sprintf(&buf[i], "%5d%c", (baseptr + refptr)->lineno,
187 (baseptr + refptr)->use);
188 refptr = (baseptr + refptr)->link;
189 i += 6;
190 } while (refptr != 0);
191 list_line(buf);
192 }
193
194 /* print trailer */
195 list_line("---------------------------------");
196 list_page();
197}

201static SPTR
202insertsrt(int q, int r)
203{
204 int i, last;
205 SPTR next;
206 char *ptr;
207
208 HASHLKP(0, SPTR_NULL);
209
210 for (i = q; i <= r; i++) {
211 /* skip over those records without ref lists */
212 if (XREFLKG(i) == 0)
213 continue;
214 ptr = SYMNAME(i);
215 last = 0;
216 next = HASHLKG(0);
217 while ((next != 0) && (strcmp(ptr, SYMNAME(next)) >= 0)) {
218 last = next;
219 next = HASHLKG(next);
220 }
221 HASHLKP(i, next);
222 HASHLKP(last, (SPTR)i);
223 }
224 return HASHLKG(0);
225}

229static SPTR
230merge(int q, int r)
231{
232 int i, j, k;
233
234 k = 0;
235 i = q;
236 j = r;
237 while ((i != 0) && (j != 0)) {
238 if (strcmp(SYMNAME(i), SYMNAME(j)) <= 0) {
239 HASHLKP(k, (SPTR)i);
240 k = i;
241 i = HASHLKG(i);
242 } else {
243 HASHLKP(k, (SPTR)j);
244 k = j;
245 j = HASHLKG(j);
246 }
247 }
248 HASHLKP(k, (i == 0) ? (SPTR)j : (SPTR)i);
249 return HASHLKG(0);
250}

254static SPTR
255mergesrt(int low, int high)
256{
257 int mid;
258 SPTR p;
259 int q, r;
260#define SORTLIM 16
261
262 if ((high - low) < SORTLIM) {
263 p = insertsrt(low, high);
264 } else {
265 mid = (low + high) / 2;
266 q = mergesrt(low, mid);
267 r = mergesrt(mid + 1, high);
268 p = merge(q, r);
269 }
270 return p;
271}

275void
276xref(void)
277{
278 struct memitem *iptr;
279 SPTR sptr;
280 int index;
281 int i;
282 SPTR liststart;
283 int nsyms;
284
285 if (!xrefcnt)
286 return;
287
288 /* rewind file */
289 if (fseek(fd, 0L, 0) == EOF)
290 interr("xref's fseek failed", 0, ERR_Fatal);
291
292 /* allocate space for ref lists */
293 NEW(baseptr, struct memitem, xrefcnt + 1);
294 if (baseptr == NULL)
295 interr("xref couldn't alloc enough space", 0, ERR_Fatal);
296
297 /* fill symbol table's hash field with NULLs (sort links) */
298 /* fill w7 field in symbol table with NULLs (ref list head) */
299 nsyms = stb.stg_avail;
300 for (i = 1; i <= nsyms; i++) {
301 HASHLKP(i, SPTR_NULL);
302 XREFLKP(i, 0);
303 }
304
305 /* read ref info into memory - 0 pos is wasted */
306 if (fread((char *)(baseptr + 1), sizeof(struct memitem), xrefcnt, fd) == 0)
307 interr("xref fread failed", xrefcnt, ERR_Fatal);
308
309 /* process in reverse order so lists are in correct order */
310 iptr = baseptr + xrefcnt;
311 index = xrefcnt;
312 while (iptr > baseptr) {
313 if (iptr->type == 0) {
314 sptr = iptr->link;
315 iptr->link = XREFLKG(sptr);
316 XREFLKP(sptr, index);
317 }
318 index--;
319 iptr--;
320 }
321
322 /* sort symbol table on SYMNAMES */
323 liststart = mergesrt(1, nsyms);
324
325 printxref(liststart);
326 if (!XBIT(0, 0x200)) {
327 FREE(baseptr);
328 fclose(fd);
329 }
330}

349/* data collection per each parallel region - in sorted order by lineno */
350typedef struct {
351 int lineno;
352 int shared;
353 int Private;
354 int prev;
355 int next;
356} RGN;
357static RGN *rgn_base;
358static int rgn_avl;
359static int rgn_size;
360static int rgn_cur;
361
362static int
363find_rgn(int lineno)
364{
365 int ll;
366 int i, here;
367
368 ll = rgn_base[rgn_cur].lineno;
369 if (lineno == ll)
370 return rgn_cur;
371 if (lineno > ll)
372 i = rgn_cur;
373 else
374 i = 0;
375 here = i;
376 for (here = i; (i = rgn_base[i].next); here = i) {
377 if (lineno < rgn_base[i].lineno) {
378 break;
379 }
380 if (lineno == rgn_base[i].lineno) {
381 rgn_cur = i;
382 return i;
383 }
384 }
385 /* Insert after 'here' */
386 i = rgn_avl++;
387 ll = rgn_base[here].next;
388 NEED(rgn_avl + 1, rgn_base, RGN, rgn_size, rgn_size + 16);
389 rgn_base[i].lineno = lineno;
390 rgn_base[i].shared = rgn_base[i].Private = 0;
391 rgn_base[i].prev = here;
392 rgn_base[i].next = ll;
393 rgn_base[here].next = i;
394 rgn_base[ll].prev = i;
395
396 rgn_cur = i;
397 return i;
398}

400static void
401add_to_list(int *list, int index)
402{
403 int s;
404 int prev;
405
406 prev = *list;
407 for (s = prev; s; prev = s, s = baseptr[s].use) {
408 if (baseptr[s].link == baseptr[index].link)
409 return;
410 }
411 if (prev) {
412 baseptr[index].use = baseptr[prev].use;
413 baseptr[prev].use = index;
414 } else {
415 *list = index;
416 baseptr[index].use = 0;
417 }
418}

420static void
421par_body(const char *hdr, int *list)
422{
423 char *nm;
424 char buf[200];
425 int s;
426 int cnt;
427#undef WIDTH
428#define WIDTH 8
429
430 list_line(hdr);
431 cnt = 0;
432 for (s = *list; s; s = baseptr[s].use) {
433 nm = SYMNAME(baseptr[s].link);
434 if (strlen(nm) + cnt > 80) {
435 list_line(buf);
436 cnt = 0;
437 }
438 sprintf(buf + cnt, " %-*s", WIDTH, nm);
439 cnt += 4;
440 if (strlen(nm) <= WIDTH)
441 cnt += WIDTH;
442 else
443 cnt += strlen(nm);
444 }
445 if (cnt)
446 list_line(buf);
447 list_line("");
448}

467void
468par_xref(void)
469{
470 struct memitem *iptr;
471 SPTR sptr;
472 int index;
473 int i;
474
475 if (!xrefcnt)
476 return;
477
478 /* rewind file */
479 if (fseek(fd, 0L, 0) == EOF)
480 interr("par_xref's fseek failed", 0, ERR_Fatal);
481
482 if (!flg.xref) {
483 /* allocate space for ref lists */
484 NEW(baseptr, struct memitem, xrefcnt + 1);
485 if (baseptr == NULL)
486 interr("par_xref couldn't alloc enough space", 0, ERR_Fatal);
487 }
488
489 /* read ref info into memory - 0 pos is wasted */
490 if (fread((char *)(baseptr + 1), sizeof(struct memitem), xrefcnt, fd) == 0)
491 interr("par_xref fread failed", xrefcnt, ERR_Fatal);
492
493 rgn_size = 16;
494 rgn_avl = 1;
495 NEW(rgn_base, RGN, rgn_size);
496 rgn_base[0].lineno = -1;
497 rgn_base[0].prev = rgn_base[0].next = 0;
498 rgn_base[0].shared = rgn_base[0].Private = 0;
499 rgn_cur = 0;
500
501 /* process in forward order - in most cases, line numbers increase */
502 iptr = baseptr + 1;
503 for (index = 1; index <= xrefcnt; index++, iptr++) {
504 if (iptr->type == 1) {
505 i = find_rgn(iptr->lineno);
506 sptr = iptr->link;
507 if (iptr->use == SC_PRIVATE)
508 add_to_list(&rgn_base[i].Private, index);
509 else
510 add_to_list(&rgn_base[i].shared, index);
511 }
512 }
513
514 print_par_xref();
515
516 FREE(baseptr);
517 FREE(rgn_base);
518 fclose(fd);
519}
