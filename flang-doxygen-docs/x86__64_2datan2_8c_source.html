617 This avoids any possible later denormalised arithmetic. */

621 scaleUpDouble1024(uy, &uy);
622 PUT_BITS_DP64(ux, x);
623 PUT_BITS_DP64(uy, y);

625 yexp = (int)((uy & EXPBITS_DP64) >> EXPSHIFTBITS_DP64);
626 diffexp = yexp - xexp;

630 is insignificant compared to piby2 */

633 else

669 return +-pi/4 or +- 3pi/4 according to
670 signs. */

674 else
675 return val_with_flags(three_piby4, AMD_F_INEXACT);
676 } else {

678 return val_with_flags(-piby4, AMD_F_INEXACT);
679 else

681 }

684 /* General case: take absolute values of arguments */
685
686 u = x;
687 v = y;
688 if (xneg)
689 u = -x;
690 if (yneg)
691 v = -y;

693 /* Swap u and v if necessary to obtain 0 < v < u. Compute v/u. */
694
695 swap_vu = (u < v);
696 if (swap_vu) {
697 uu = u;
698 u = v;
699 v = uu;
700 }
701 vbyu = v / u;
702
703 if (vbyu > 0.0625) { /* General values of v/u. Use a look-up
704 table and series expansion. */
705
706 index = (int)(256 * vbyu + 0.5);
707 q1 = atan_jby256_lead[index - 16];
708 q2 = atan_jby256_tail[index - 16];
709 c = index * 1. / 256;
710 GET_BITS_DP64(u, ui);
711 m = (int)((ui & EXPBITS_DP64) >> EXPSHIFTBITS_DP64) - EXPBIAS_DP64;
712 u = scaleDouble_2(u, -m);
713 v = scaleDouble_2(v, -m);
714 GET_BITS_DP64(u, ui);
715 PUT_BITS_DP64(0xfffffffff8000000 & ui, u1); /* 26 leading bits of u */
716 u2 = u - u1;
717
718 r = ((v - c * u1) - c * u2) / (u + c * v);
719
720 /* Polynomial approximation to atan(r) */
721
722 s = r * r;
723 q2 = q2 + r -
724 r * (s * (0.33333333333224095522 - s * (0.19999918038989143496)));
725 } else if (vbyu < 1.e-8) { /* v/u is small enough that atan(v/u) = v/u */
726 q1 = 0.0;
727 q2 = vbyu;
728 } else /* vbyu <= 0.0625 */
729 {
730 /* Small values of v/u. Use a series expansion
731 computed carefully to minimise cancellation */
732
733 GET_BITS_DP64(u, ui);
734 PUT_BITS_DP64(0xffffffff00000000 & ui, u1);
735 GET_BITS_DP64(vbyu, ui);
736 PUT_BITS_DP64(0xffffffff00000000 & ui, vu1);
737 u2 = u - u1;
738 vu2 = vbyu - vu1;
739
740 q1 = 0.0;
741 s = vbyu * vbyu;
742 q2 = vbyu +
743 ((((v - u1 * vu1) - u2 * vu1) - u * vu2) / u -
744 (vbyu * s * (0.33333333333333170500 -
745 s * (0.19999999999393223405 -
746 s * (0.14285713561807169030 -
747 s * (0.11110736283514525407 -
748 s * (0.90029810285449784439E-01)))))));
749 }
750
751 /* Tidy-up according to which quadrant the arguments lie in */
752
753 if (swap_vu) {
754 q1 = piby2_head - q1;
755 q2 = piby2_tail - q2;
756 }
757 if (xneg) {
758 q1 = pi_head - q1;
759 q2 = pi_tail - q2;
760 }
761 q1 = q1 + q2;
762
763 if (yneg)
764 q1 = -q1;
765
766 return q1;
