99void
100print_back_trace_line(char *bt_str, void const *const addr)
101{
102 char addr2line_cmd[512], cmd_out[1024];
103 char *ptr_qmark, *ptr_colon, *ptr_lparen, *ptr_rparen, *ptr_plus;
104 static char buffer[4096];
105 FILE *fp;
106 static char exec_name[PATH_MAX]; // Includes terminating NULL
107 static char *pexec_name = exec_name;
108
109 if (NULL == saved_argv) {
110 saved_argv = __io_get_argv();
111 }
112
113 if (NULL == saved_argv) {
114 /*
115 * Most likely a C program where __io_set_argv() was not called on startup.
116 * Let's see if we can get executable's name from /proc/<PID>/cmdline.
117 *
118 * Here we use buffer "exec_name" to capture the executable's filename.
119 * But there are issues with pathname lengths. See:
120 * https://eklitzke.org/path-max-is-tricky
121 */
122
123 FILE *f;
124
125 snprintf(exec_name, sizeof exec_name, "/proc/%" PRIu64 "/cmdline",
126 (uint64_t)getpid());
127 f = fopen(exec_name, "r");
128 if (NULL == f) {
129 /*
130 * Not sure how this could happen and if it would be better to
131 * abort/exit(EXIT_FAILURE) instead of returning.
132 * exit(EXIT_FAILURE);
133 */
134 return;
135 }
136 fread(exec_name, sizeof exec_name, 1, f);
137 exec_name[PATH_MAX - 1] = '\0'; // In case filename+pathname > PATH_MAX
138 fclose(f);
139 saved_argv = &pexec_name;
140 }
141
142 sprintf(addr2line_cmd, "addr2line -e %s %p", saved_argv[0], addr);
143
144 fp = popen(addr2line_cmd, "r");
145 if (fp != NULL) {
146 if (fgets(cmd_out, sizeof(cmd_out) - 1, fp) != NULL) {
147 if (cmd_out[strlen(cmd_out) - 1] == '\n')
148 cmd_out[strlen(cmd_out) - 1] = '\0';
149 ptr_qmark = strchr(cmd_out, '?');
150 ptr_colon = strchr(cmd_out, ':');
151 ptr_lparen = strchr(bt_str, '(');
152 ptr_rparen = strchr(bt_str, ')');
153 if (ptr_lparen != NULL && ptr_lparen != NULL && ptr_lparen < ptr_rparen)
154 ptr_plus = strchr(ptr_lparen, '+');
155 if (ptr_qmark != NULL) {
156 fprintf(__io_stderr(), " %s\n", bt_str);
157 } else if (ptr_colon != NULL && ptr_plus != NULL &&
158 ptr_plus < ptr_rparen) {
159 /* replace offset by line number */
160 strncpy(buffer, bt_str, ptr_plus - bt_str);
161 sprintf(buffer + (ptr_plus - bt_str), "%s%s", ptr_colon, ptr_rparen);
162 fprintf(__io_stderr(), " %s\n", buffer);
163 }
164 } else {
165 fprintf(__io_stderr(), " %s\n", bt_str);
166 }
167 fclose(fp);
168 } else {
169 fprintf(__io_stderr(), " %s\n", bt_str);
170 }
171}

173void
174__abort_trace(int skip)
175{
176 void *array[MAXTRACE];
177 size_t size;
178 char **strings;
179 size_t i;
180
181 if (regs != (FLANGRTI_GREGSET_T *)0) {
182 dumpregs(regs);
183 }
184
185 size = backtrace(array, MAXTRACE);
186 if (skip + 1 >= size) {
187 fprintf(__io_stderr(), " --- traceback not available\n");
188 return;
189 }
190 strings = backtrace_symbols(array, size);
191 if (size < 100) {
192 for (i = skip + 1; i < size; i++)
193 print_back_trace_line(strings[i], array[i]);
194 } else {
195 for (i = skip + 1; i < 40; i++)
196 print_back_trace_line(strings[i], array[i]);
197 fprintf(__io_stderr(), " --- skipping traceback entries\n");
198 for (i = size - 40; i < size; i++)
199 print_back_trace_line(strings[i], array[i]);
200 }
201 free(strings);
202}

204/*
205 * this routine is a signal handler, it prints a message and terminates the
206 * process
207 */

209static void
210__abort_sig_hand(int sig, siginfo_t *in, FLANGRTI_UCONTEXT_T *u)
211{
212 const char *p;
213 char b[128];
214 int n, m;
215 struct sigaction new;
216 struct sigaction old;
217
218 new.sa_sigaction = (void (*)(int, siginfo_t *, void *))SIG_DFL;
219 sigemptyset(&new.sa_mask);
220 new.sa_flags = SA_SIGINFO;
221 n = 0;
222 while (sigs[n].sig != 0) {
223 sigaction(sigs[n].sig, &new, &old);
224 n++;
225 }
226
227 regs = getRegs(u);
228
229 n = 0;
230 while ((sigs[n].sig != 0) && (sigs[n].sig != sig)) {
231 n++;
232 }
233 if (sigs[n].sig == 0) {
234 sprintf(b, "signal %d", sig);
235 p = b;
236 } else {
237 p = sigs[n].str;
238 m = 0;
239 if (sigs[n].cod != CODNULL) {
240 while ((sigs[n].cod[m].code != 0) &&
241 (sigs[n].cod[m].code != in->si_code)) {
242 m++;
243 }
244 if (sigs[n].cod[m].code != 0) {
245 sprintf(b, "%s, %s", p, sigs[n].cod[m].str);
246 p = b;
247 }
248 }
249 }
250 __abort(3, p);
251}
