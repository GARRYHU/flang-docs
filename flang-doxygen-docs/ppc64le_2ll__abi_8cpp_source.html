34/* ELF v2 ABI has the concept of a homogeneous aggregrate. It is an aggregate
35 * type where all the fundamental types are the same after flattening all
36 * structs and arrays.
37 */
38struct pwr_homogeneous_aggr {
39 LL_Module *module;
40 LL_Type *base_type;
41 ISZ_T base_bytes;
42};
43
44/* Return 1 if dtype is inconsistent with the homogeneous aggregate
45 * information pointed to by context. */
46static int
47update_homogeneous(void *context, DTYPE dtype, unsigned address,
48 int member_sptr)
49{
50 struct pwr_homogeneous_aggr *ha = (struct pwr_homogeneous_aggr *)context;
51 ISZ_T size;
52 LL_Type *llt;
53
54 dtype = DT_BASETYPE(dtype);
55
56 if (DTY(dtype) == TY_ARRAY)
57 dtype = (DTYPE)DTY(dtype + 1); // ???
58
59 switch (dtype) {
60 case DT_CMPLX:
61 dtype = DT_FLOAT;
62 break;
63 case DT_DCMPLX:
64 dtype = DT_DBLE;
65 break;
66 default:
67 break;
68 }
69
70 size = zsize_of(dtype);
71 llt = ll_convert_dtype(ha->module, dtype);
72
73 /* Check if this type is allowed as the base type of a homogeneous
74 * aggregate. */
75 switch (llt->data_type) {
76 case LL_FLOAT:
77 case LL_DOUBLE:
78 /* OK. */
79 break;
80 case LL_VECTOR:
81 /* Allow naturally sized vectors only. */
82 if (size != 16)
83 return 1;
84 break;
85 default:
86 /* Nope. */
87 return 1;
88 }
89
90 /* Record base type on the first call. */
91 if (!ha->base_type) {
92 if (address != 0)
93 return 1;
94 ha->base_type = llt;
95 ha->base_bytes = size;
96 return 0;
97 }
98
99 /* Check if dtype is consistent with the existing base type. */
100 if (llt != ha->base_type)
101 return 1;
102
103 /* Reject unaligned entries. */
104 if (!size || address % size != 0)
105 return 1;
106
107 return 0;
108}

110/* Check if dtype is a vector register candidate. */
111static LL_Type *
112check_vector_registers(LL_Module *module, DTYPE dtype)
113{
114 struct pwr_homogeneous_aggr aggr = {module, NULL, 0};
115 ISZ_T size = zsize_of(dtype);
116
117 /* Check if dtype is a homogeneous aggregate, or a single value which can
118 * be passed in vector registers. */
119 if (visit_flattened_dtype(update_homogeneous, &aggr, dtype, 0, 0))
120 return NULL;
121 if (!aggr.base_type)
122 return NULL;
123
124 /* A non-aggregated scalar will simply be copied to base_type. */
125 switch (aggr.base_type->data_type) {
126 case LL_FLOAT:
127 case LL_DOUBLE:
128 break;
129 case LL_VECTOR:
130 if (aggr.base_bytes != 8 && aggr.base_bytes != 16)
131 return NULL;
132 break;
133 default:
134 return NULL;
135 }
136
137 /* We have a scalar or a homogeneous aggregate of the right type. The ABI
138 * supports one to eight elements of the base type. */
139 if (size > 8 * aggr.base_bytes)
140 return NULL;
141
142 if (size == aggr.base_bytes)
143 return aggr.base_type;
144
145 /* If we supported IBM extended precision or _Decimal128 data types, we
146 * should impose a 4 element limit, but all other types allow up to eight
147 * elements to be passed in vector registers.
148 */
149
150 return ll_get_array_type(aggr.base_type, size / aggr.base_bytes, 0);
151}

153/* Classify an integer type for return or arg. */
154static enum LL_ABI_ArgKind
155classify_int(DTYPE dtype)
156{
157 /* Integer types smaller than a register must be sign/zero extended. */
158 if (zsize_of(dtype) < 8)
159 return DT_ISUNSIGNED(dtype) ? LL_ARG_ZEROEXT : LL_ARG_SIGNEXT;
160
161 return LL_ARG_DIRECT;
162}

164/* Classify common to args and return values. */
165static bool
166classify_common(LL_ABI_Info *abi, LL_ABI_ArgInfo *arg, DTYPE dtype)
167{
168 LL_Type *haggr;
169 if (DT_ISINT(dtype)) {
170 arg->kind = classify_int(dtype);
171 return true;
172 }
173
174 /* Basic types can be returned in registers directly. Complex types also
175 * get handled correctly. */
176 if (dtype == DT_VOIDNONE || DT_ISSCALAR(dtype)) {
177 arg->kind = LL_ARG_DIRECT;
178 return true;
179 }
180
181 /* Check for vector register arguments, including homogeneous aggregrates. */
182 if ((haggr = check_vector_registers(abi->module, dtype))) {
183 arg->kind = LL_ARG_DIRECT;
184 arg->type = haggr;
185 return true;
186 }
187
188 return false;
189}

191void
192ll_abi_classify_return_dtype(LL_ABI_Info *abi, DTYPE dtype)
193{
194 enum LL_BaseDataType bdt = LL_NOTYPE;
195 ISZ_T size;
196
197 dtype = DT_BASETYPE(dtype);
198
199 if (classify_common(abi, &abi->arg[0], dtype))
200 return;
201
202 /* Small structs can be returned in up to two GPRs. */
203 size = zsize_of(dtype);
204 if (size <= 16) {
205 abi->arg[0].kind = LL_ARG_COERCE;
206 abi->arg[0].type = ll_coercion_type(abi->module, dtype, size, 8);
207 return;
208 }
209
210 /* Large types must be returned in memory via an sret pointer argument. */
211 abi->arg[0].kind = LL_ARG_INDIRECT;
212}

214void
215ll_abi_classify_arg_dtype(LL_ABI_Info *abi, LL_ABI_ArgInfo *arg, DTYPE dtype)
216{
217 dtype = DT_BASETYPE(dtype);
218
219 if (classify_common(abi, arg, dtype))
220 return;
221
222 /* Fortran, by default, is pass by reference.
223 * dtype information is not enough to determine if this argument is direct
224 * or not. All classification occurs in process_ll_abi_func_ftn_mod and
225 * only when there is a known prototype.
226 */
227 if (arg->ftn_pass_by_val &&
228 (DT_ISCMPLX(dtype) || (DTY(dtype) == TY_STRUCT))) {
229 arg->kind = LL_ARG_COERCE;
230 arg->type = ll_coercion_type(abi->module, dtype, zsize_of(dtype), 8);
231 return;
232 }
233
234 /* ???: 64 == 8 x eightwords, which is the most we can pass in GPRs. */
235 if ((DTY(dtype) == TY_STRUCT) && (zsize_of(dtype) > 64)) {
236 LL_Type *llt;
237 arg->kind = LL_ARG_BYVAL;
238 llt = ll_get_struct_type(abi->module, dtype, 0);
239 if (!llt) {
240 llt = ll_convert_dtype(abi->module, dtype);
241 assert(llt, "expected LL_Type*", dtype, ERR_Fatal);
242 }
243 arg->type = ll_get_pointer_type(llt);
244 return;
245 }
246
247 /* All other arguments are coerced. LLVM will figure out which parts go in
248 * registers. */
249 arg->kind = LL_ARG_COERCE;
250 arg->type = ll_coercion_type(abi->module, dtype, zsize_of(dtype), 8);
251}
